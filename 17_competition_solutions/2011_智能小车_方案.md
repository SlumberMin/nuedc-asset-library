# 2011年：智能小车 - Orange Pi 5解决方案

## 题目概述

设计两辆智能小车，能够在跑道上交替领跑。前车引导后车跟随，后车在特定位置超越前车成为新的领跑车，实现双车交替领跑。需要视觉循迹、无线通信、速度控制等功能。

### 1. 视觉循迹算法

```python
#!/usr/bin/env python3
"""
视觉循迹算法 - 跑道检测
"""

import cv2
import numpy as np

class LineTracker:
    """循迹线检测器"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
        
        # 跑道参数
        self.track_color_lower = np.array([0, 0, 0])      # 黑色跑道线
        self.track_color_upper = np.array([180, 255, 50])
        
        # ROI区域 (只处理图像下半部分)
        self.roi_top = 120  # 从中间开始
        self.roi_bottom = 240
        
        # PID参数
        self.kp = 0.5
        self.ki = 0.01
        self.kd = 0.1
        
        # PID状态
        self.last_error = 0
        self.error_sum = 0
    
    def detect_lane(self, frame):
        """
        检测跑道线
        返回: 偏移量 (-1.0 到 1.0)
        """
        # 提取ROI
        roi = frame[self.roi_top:self.roi_bottom, :]
        
        # 转换到HSV
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        
        # 创建掩码
        mask = cv2.inRange(hsv, self.track_color_lower, self.track_color_upper)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None, mask
        
        # 找到最大轮廓
        largest = max(contours, key=cv2.contourArea)
        
        if cv2.contourArea(largest) < 100:
            return None, mask
        
        # 计算中心位置
        M = cv2.moments(largest)
        if M["m00"] > 0:
            cx = int(M["m10"] / M["m00"])
            
            # 计算偏移量 (-1.0 到 1.0)
            frame_center = roi.shape[1] // 2
            offset = (cx - frame_center) / frame_center
            
            return offset, mask
        
        return None, mask
    
    def detect_crossroad(self, mask):
        """
        检测十字路口
        """
        # 统计白色像素
        white_pixels = np.sum(mask > 0)
        total_pixels = mask.shape[0] * mask.shape[1]
        
        # 如果白色像素比例过高，可能是十字路口
        if white_pixels / total_pixels > 0.3:
            return True
        
        return False
    
    def detect_curve(self, offset_history):
        """
        检测弯道
        """
        if len(offset_history) < 5:
            return False, 0
        
        # 检查偏移量变化趋势
        recent = offset_history[-5:]
        trend = recent[-1] - recent[0]
        
        if abs(trend) > 0.3:
            return True, trend
        
        return False, 0
    
    def pid_control(self, offset, dt):
        """
        PID控制
        返回: (left_speed, right_speed)
        """
        if offset is None:
            return 0, 0
        
        # 比例项
        p = self.kp * offset
        
        # 积分项
        self.error_sum += offset * dt
        self.error_sum = max(-100, min(100, self.error_sum))
        i = self.ki * self.error_sum
        
        # 微分项
        d = self.kd * (offset - self.last_error) / dt if dt > 0 else 0
        self.last_error = offset
        
        # 计算速度
        base_speed = 50
        correction = p + i + d
        
        left_speed = base_speed + correction * 50
        right_speed = base_speed - correction * 50
        
        # 限幅
        left_speed = max(-100, min(100, left_speed))
        right_speed = max(-100, min(100, right_speed))
        
        return left_speed, right_speed
    
    def process_frame(self, frame):
        """
        处理一帧图像
        返回: (left_speed, right_speed, info)
        """
        offset, mask = self.detect_lane(frame)
        
        # 计算控制量
        dt = 0.033  # 30fps
        left_speed, right_speed = self.pid_control(offset, dt)
        
        info = {
            'offset': offset,
            'mask': mask,
            'is_crossroad': self.detect_crossroad(mask) if mask is not None else False
        }
        
        return left_speed, right_speed, info
    
    def release(self):
        """释放资源"""
        self.cap.release()
```

```python
#!/usr/bin/env python3
"""
电机控制 - PWM + GPIO
"""

import gpiod
import time

class MotorController:
    """电机控制器"""
    
    def __init__(self):
        # GPIO配置
        self.chip = gpiod.Chip('1')
        
        # 左电机
        self.left_pwm = self.chip.get_line(24)    # GPIO1_A0
        self.left_in1 = self.chip.get_line(25)    # GPIO1_A1
        self.left_in2 = self.chip.get_line(26)    # GPIO1_A2
        
        # 右电机
        self.right_pwm = self.chip.get_line(32)   # GPIO1_B0
        self.right_in1 = self.chip.get_line(33)   # GPIO1_B1
        self.right_in2 = self.chip.get_line(34)   # GPIO1_B2
        
        # 配置GPIO
        config = gpiod.LineSettings(direction=gpiod.Direction.OUTPUT, output_value=0)
        
        self.left_pwm.request(consumer="motor", config=config)
        self.left_in1.request(consumer="motor", config=config)
        self.left_in2.request(consumer="motor", config=config)
        self.right_pwm.request(consumer="motor", config=config)
        self.right_in1.request(consumer="motor", config=config)
        self.right_in2.request(consumer="motor", config=config)
        
        # 编码器
        self.left_encoder_a = self.chip.get_line(48)   # GPIO3_A0
        self.left_encoder_b = self.chip.get_line(49)   # GPIO3_A1
        self.right_encoder_a = self.chip.get_line(56)  # GPIO3_B0
        self.right_encoder_b = self.chip.get_line(57)  # GPIO3_B1
        
        encoder_config = gpiod.LineSettings(direction=gpiod.Direction.INPUT)
        self.left_encoder_a.request(consumer="encoder", config=encoder_config)
        self.left_encoder_b.request(consumer="encoder", config=encoder_config)
        self.right_encoder_a.request(consumer="encoder", config=encoder_config)
        self.right_encoder_b.request(consumer="encoder", config=encoder_config)
        
        # 编码器计数
        self.left_count = 0
        self.right_count = 0
        
        # 速度
        self.left_speed = 0
        self.right_speed = 0
    
    def set_left_motor(self, speed):
        """
        设置左电机速度
        speed: -100 到 100
        """
        if speed > 0:
            self.left_in1.set_value(1)
            self.left_in2.set_value(0)
        elif speed < 0:
            self.left_in1.set_value(0)
            self.left_in2.set_value(1)
        else:
            self.left_in1.set_value(0)
            self.left_in2.set_value(0)
        
        # PWM控制 (软件PWM)
        duty = abs(speed) / 100.0
        self.left_pwm.set_value(1 if duty > 0.5 else 0)  # 简化实现
    
    def set_right_motor(self, speed):
        """
        设置右电机速度
        speed: -100 到 100
        """
        if speed > 0:
            self.right_in1.set_value(1)
            self.right_in2.set_value(0)
        elif speed < 0:
            self.right_in1.set_value(0)
            self.right_in2.set_value(1)
        else:
            self.right_in1.set_value(0)
            self.right_in2.set_value(0)
        
        duty = abs(speed) / 100.0
        self.right_pwm.set_value(1 if duty > 0.5 else 0)
    
    def set_motors(self, left_speed, right_speed):
        """设置两个电机速度"""
        self.set_left_motor(left_speed)
        self.set_right_motor(right_speed)
        self.left_speed = left_speed
        self.right_speed = right_speed
    
    def stop(self):
        """停止所有电机"""
        self.set_motors(0, 0)
    
    def read_encoder(self):
        """读取编码器"""
        left_a = self.left_encoder_a.get_value()
        right_a = self.right_encoder_a.get_value()
        
        # 简化的编码器读取
        return left_a, right_a
    
    def get_speed(self):
        """获取当前速度"""
        return self.left_speed, self.right_speed
```

```python
#!/usr/bin/env python3
"""
无线通信 - ESP-NOW
"""

import serial
import json
import time
import threading

class WirelessComm:
    """无线通信管理"""
    
    def __init__(self, port='/dev/ttyS3', baudrate=115200):
        self.serial = serial.Serial(port, baudrate, timeout=0.1)
        
        # 车辆ID
        self.car_id = 0  # 0 或 1
        
        # 对方车辆信息
        self.peer_position = None
        self.peer_speed = None
        self.peer_state = None
        
        # 通信线程
        self.is_running = False
        self.rx_thread = None
        
        # 回调函数
        self.on_message_received = None
    
    def start(self):
        """启动通信"""
        self.is_running = True
        self.rx_thread = threading.Thread(target=self._rx_loop)
        self.rx_thread.daemon = True
        self.rx_thread.start()
    
    def stop(self):
        """停止通信"""
        self.is_running = False
        if self.rx_thread:
            self.rx_thread.join()
        self.serial.close()
    
    def _rx_loop(self):
        """接收循环"""
        while self.is_running:
            try:
                line = self.serial.readline().decode('utf-8').strip()
                if line:
                    self._parse_message(line)
            except:
                pass
            time.sleep(0.01)
    
    def _parse_message(self, message):
        """解析消息"""
        try:
            data = json.loads(message)
            
            if data.get('id') != self.car_id:
                self.peer_position = data.get('position')
                self.peer_speed = data.get('speed')
                self.peer_state = data.get('state')
                
                if self.on_message_received:
                    self.on_message_received(data)
        except:
            pass
    
    def send_status(self, position, speed, state):
        """
        发送状态
        position: (x, y) 或 偏移量
        speed: 速度
        state: 状态 ('lead', 'follow', 'overtake')
        """
        data = {
            'id': self.car_id,
            'position': position,
            'speed': speed,
            'state': state,
            'timestamp': time.time()
        }
        
        message = json.dumps(data) + '\n'
        self.serial.write(message.encode('utf-8'))
    
    def send_overtake_request(self):
        """发送超车请求"""
        data = {
            'id': self.car_id,
            'command': 'overtake_request',
            'timestamp': time.time()
        }
        
        message = json.dumps(data) + '\n'
        self.serial.write(message.encode('utf-8'))
    
    def send_overtake_ack(self):
        """发送超车确认"""
        data = {
            'id': self.car_id,
            'command': 'overtake_ack',
            'timestamp': time.time()
        }
        
        message = json.dumps(data) + '\n'
        self.serial.write(message.encode('utf-8'))


class ESP32Bridge:
    """ESP32通信桥接"""
    
    def __init__(self, port='/dev/ttyS3'):
        self.serial = serial.Serial(port, 115200, timeout=0.1)
    
    def configure_espnow(self, peer_mac):
        """
        配置ESP-NOW
        peer_mac: 对方MAC地址
        """
        # 发送配置命令
        cmd = f'CONFIG:{peer_mac}\n'
        self.serial.write(cmd.encode())
        
        # 等待响应
        response = self.serial.readline().decode().strip()
        return response == 'OK'
    
    def send_data(self, data):
        """发送数据"""
        message = f'DATA:{json.dumps(data)}\n'
        self.serial.write(message.encode())
    
    def receive_data(self):
        """接收数据"""
        line = self.serial.readline().decode().strip()
        if line.startswith('DATA:'):
            return json.loads(line[5:])
        return None
```

### 4. 状态机与决策

```python
#!/usr/bin/env python3
"""
状态机 - 双车交替领跑决策
"""

import time

class CarState:
    """车辆状态"""
    IDLE = 'idle'
    LEADING = 'leading'
    FOLLOWING = 'following'
    OVERTAKING = 'overtaking'
    BEING_OVERTAKEN = 'being_overtaken'
    WAITING = 'waiting'

class DecisionMaker:
    """决策控制器"""
    
    def __init__(self, car_id, wireless_comm):
        self.car_id = car_id
        self.wireless = wireless_comm
        
        # 当前状态
        self.state = CarState.IDLE
        
        # 超车相关
        self.overtake_distance = 0.5  # 超车触发距离
        self.overtake_cooldown = 10   # 超车冷却时间
        self.last_overtake_time = 0
        
        # 圈数计数
        self.lap_count = 0
        self.lap_start_time = 0
        
        # 距离估算
        self.distance_to_peer = float('inf')
    
    def update(self, vision_info, peer_info):
        """
        更新决策
        vision_info: 视觉信息
        peer_info: 对方车辆信息
        """
        # 更新距离估算
        if peer_info and peer_info.get('position') is not None:
            self._estimate_distance(peer_info)
        
        # 状态机
        if self.state == CarState.IDLE:
            return self._handle_idle()
        
        elif self.state == CarState.LEADING:
            return self._handle_leading(vision_info, peer_info)
        
        elif self.state == CarState.FOLLOWING:
            return self._handle_following(vision_info, peer_info)
        
        elif self.state == CarState.OVERTAKING:
            return self._handle_overtaking(vision_info, peer_info)
        
        elif self.state == CarState.BEING_OVERTAKEN:
            return self._handle_being_overtaken(vision_info, peer_info)
        
        elif self.state == CarState.WAITING:
            return self._handle_waiting(vision_info, peer_info)
        
        return None
    
    def _handle_idle(self):
        """空闲状态"""
        # 根据ID决定初始角色
        if self.car_id == 0:
            self.state = CarState.LEADING
            return {'action': 'lead', 'speed': 50}
        else:
            self.state = CarState.FOLLOWING
            return {'action': 'follow', 'speed': 40}
    
    def _handle_leading(self, vision_info, peer_info):
        """领跑状态"""
        current_time = time.time()
        
        # 检查是否需要让后车超车
        if self._should_allow_overtake(current_time, peer_info):
            self.state = CarState.BEING_OVERTAKEN
            self.wireless.send_overtake_ack()
            return {'action': 'slow_down', 'speed': 30}
        
        # 正常领跑
        return {'action': 'lead', 'speed': 50}
    
    def _handle_following(self, vision_info, peer_info):
        """跟随状态"""
        current_time = time.time()
        
        # 检查是否可以超车
        if self._can_overtake(current_time, peer_info):
            self.state = CarState.OVERTAKING
            self.wireless.send_overtake_request()
            return {'action': 'overtake', 'speed': 60}
        
        # 跟随前车
        follow_speed = self._calculate_follow_speed(peer_info)
        return {'action': 'follow', 'speed': follow_speed}
    
    def _handle_overtaking(self, vision_info, peer_info):
        """超车状态"""
        # 检查是否完成超车
        if self._is_overtake_complete(vision_info, peer_info):
            self.state = CarState.LEADING
            self.last_overtake_time = time.time()
            self.lap_count += 1
            return {'action': 'lead', 'speed': 50}
        
        # 继续超车
        return {'action': 'overtake', 'speed': 60}
    
    def _handle_being_overtaken(self, vision_info, peer_info):
        """被超车状态"""
        # 检查超车是否完成
        if self._is_overtaken_complete(peer_info):
            self.state = CarState.FOLLOWING
            return {'action': 'follow', 'speed': 40}
        
        # 减速让行
        return {'action': 'slow_down', 'speed': 30}
    
    def _handle_waiting(self, vision_info, peer_info):
        """等待状态"""
        return {'action': 'wait', 'speed': 0}
    
    def _should_allow_overtake(self, current_time, peer_info):
        """判断是否应该让后车超车"""
        # 距离足够近
        if self.distance_to_peer < self.overtake_distance:
            # 冷却时间已过
            if current_time - self.last_overtake_time > self.overtake_cooldown:
                return True
        return False
    
    def _can_overtake(self, current_time, peer_info):
        """判断是否可以超车"""
        # 距离足够近
        if self.distance_to_peer < self.overtake_distance:
            # 冷却时间已过
            if current_time - self.last_overtake_time > self.overtake_cooldown:
                return True
        return False
    
    def _is_overtake_complete(self, vision_info, peer_info):
        """判断超车是否完成"""
        # 如果已经领先足够距离
        if self.distance_to_peer > self.overtake_distance * 2:
            return True
        return False
    
    def _is_overtaken_complete(self, peer_info):
        """判断被超车是否完成"""
        if self.distance_to_peer > self.overtake_distance * 2:
            return True
        return False
    
    def _calculate_follow_speed(self, peer_info):
        """计算跟随速度"""
        if peer_info and peer_info.get('speed') is not None:
            # 保持与前车相同速度
            return peer_info['speed'] * 0.9  # 略慢于前车
        return 40
    
    def _estimate_distance(self, peer_info):
        """估算与对方车辆的距离"""
        # 简化实现：基于位置差
        if peer_info.get('position') is not None:
```

```python
class SmartCar:
    """智能小车系统"""
    
    def __init__(self, car_id=0):

        self.tracker = LineTracker(camera_id=0)
        self.motor = MotorController()

        self.is_running = True
        self.start_time = time.time()
        
        print("系统已启动")
    
    def stop(self):
        """停止系统"""
        self.is_running = False
        self.motor.stop()
        self.wireless.stop()
        print("系统已停止")
    
    def run(self):
        """主循环"""
        self.start()
        
        print("按 'q' 退出")
        print("按 's' 停止电机")
        print("按 'r' 重启")
        
        try:
            while self.is_running:
                # 获取图像
                ret, frame = self.tracker.cap.read()
                if not ret:
                    continue
                
                # 视觉处理
                left_speed, right_speed, vision_info = self.tracker.process_frame(frame)
                
                # 获取对方车辆信息
                peer_info = {
                    'position': self.wireless.peer_position,
                    'speed': self.wireless.peer_speed,
                    'state': self.wireless.peer_state
                }
                
                # 决策
                decision = self.decision.update(vision_info, peer_info)
                
                # 执行决策
                if decision:
                    if decision['action'] == 'lead':
                        self.motor.set_motors(left_speed, right_speed)
                    elif decision['action'] == 'follow':
                        self.motor.set_motors(
                            left_speed * decision['speed'] / 50,
                            right_speed * decision['speed'] / 50
                        )
                    elif decision['action'] == 'overtake':
                        # 超车时稍微偏移
                        self.motor.set_motors(
                            decision['speed'] + 10,
                            decision['speed'] - 10
                        )
                    elif decision['action'] == 'slow_down':
                        self.motor.set_motors(
                            left_speed * 0.5,
                            right_speed * 0.5
                        )
                    elif decision['action'] == 'wait':
                        self.motor.stop()
                
                # 发送状态
                self.wireless.send_status(
                    position=vision_info.get('offset', 0),
                    speed=(left_speed + right_speed) / 2,
                    state=self.decision.state
                )
                
                # 显示画面
                if vision_info.get('mask') is not None:
                    display = vision_info['mask']
                else:
                    display = frame
                
                # 添加状态信息
                cv2.putText(display, f"Car {self.car_id}", (10, 20),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
                cv2.putText(display, f"State: {self.decision.state}", (10, 40),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
                cv2.putText(display, f"Laps: {self.decision.lap_count}", (10, 60),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
                
                cv2.imshow(f'Car {self.car_id}', display)
                
                # 键盘控制
                key = cv2.waitKey(1) & 0xFF
                if key == ord('q'):
                    break
                elif key == ord('s'):
                    self.motor.stop()
                elif key == ord('r'):
                    self.decision.state = CarState.IDLE
                
                self.frame_count += 1
                
                # 控制频率
                time.sleep(0.033)  # ~30fps
        
        except KeyboardInterrupt:
            print("\n用户中断")
        
        finally:
            self.stop()
            cv2.destroyAllWindows()
            
            # 打印统计
            elapsed = time.time() - self.start_time
            print(f"\n统计信息:")
            print(f"运行时间: {elapsed:.1f} 秒")
            print(f"处理帧数: {self.frame_count}")
            print(f"平均FPS: {self.frame_count / elapsed:.1f}")
```

```python
class OvertakeProtocol:
    """超车协议"""
    
    def __init__(self, wireless_comm):
        self.wireless = wireless_comm
        self.state = 'idle'
        self.timeout = 5.0  # 超时时间
        self.request_time = 0
    
    def request_overtake(self):
        """请求超车"""
        self.state = 'requesting'
        self.request_time = time.time()
        self.wireless.send_overtake_request()
    
    def handle_request(self):
        """处理超车请求"""
        self.state = 'acknowledged'
        self.wireless.send_overtake_ack()
    
    def check_timeout(self):
        """检查超时"""
        if self.state == 'requesting':
            if time.time() - self.request_time > self.timeout:
                self.state = 'idle'
                return True
        return False
    
    def complete(self):
        """完成超车"""
        self.state = 'idle'
```

```python
class SpeedSynchronizer:
    """速度同步器"""
    
    def __init__(self, wireless_comm):
        self.wireless = wireless_comm
        self.target_speed = 50
        self.sync_interval = 0.5  # 同步间隔
        self.last_sync_time = 0
    
    def sync_speed(self, current_speed):
        """同步速度"""
        current_time = time.time()
        
        if current_time - self.last_sync_time >= self.sync_interval:
            # 发送速度信息
            self.wireless.send_status(
                position=0,
                speed=current_speed,
                state='sync'
            )
            self.last_sync_time = current_time
        
        # 获取对方速度
        peer_speed = self.wireless.peer_speed
        
        if peer_speed is not None:
            # 平滑调整速度
            self.target_speed = (self.target_speed + peer_speed) / 2
        
        return self.target_speed
```
