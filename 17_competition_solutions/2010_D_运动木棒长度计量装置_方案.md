# 2010年D题：运动木棒长度计量装置 - Orange Pi 5解决方案

## 题目概述

设计一个长度计量装置，能够自动测量运动中的木棒长度。木棒在传送带上运动，需要通过视觉或其他传感器实时检测木棒的端点位置，计算其长度并显示。

---

## 方案选择

### 方案A：单摄像头视觉测量（推荐）

**优点：**
- 成本低
- 精度可达毫米级
- 可测量任意形状

**缺点：**
- 需要标定
- 受光照影响

### 方案B：双摄像头立体视觉

**优点：**
- 可测量三维尺寸
- 更高精度

**缺点：**
- 成本高
- 标定复杂

### 方案C：激光测距

**优点：**
- 精度高
- 不受光照影响

**缺点：**
- 只能测量单点
- 需要扫描机构

### 1. 摄像头标定

```python
#!/usr/bin/env python3
"""
摄像头标定 - 消除畸变，建立像素到物理尺寸的映射
"""

import cv2
import numpy as np

class CameraCalibrator:
    """摄像头标定器"""
    
    def __init__(self):
        # 标定参数
        self.camera_matrix = None
        self.dist_coeffs = None
        self.homography = None
        
        # 棋盘格参数
        self.chessboard_size = (9, 6)  # 内角点数
        self.square_size = 25.0  # 每格25mm
    
    def calibrate_intrinsics(self, images):
        """
        标定内参（焦距、畸变）
        images: 棋盘格图像列表
        """
        # 准备棋盘格点
        obj_points = []  # 3D点
        img_points = []  # 2D点
        
        # 生成棋盘格坐标
        objp = np.zeros((self.chessboard_size[0] * self.chessboard_size[1], 3), np.float32)
        objp[:, :2] = np.mgrid[0:self.chessboard_size[0], 
                                0:self.chessboard_size[1]].T.reshape(-1, 2)
        objp *= self.square_size
        
        for img in images:
            gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
            
            # 查找棋盘格角点
            ret, corners = cv2.findChessboardCorners(gray, self.chessboard_size, None)
            
            if ret:
                # 亚像素精化
                criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001)
                corners = cv2.cornerSubPix(gray, corners, (11, 11), (-1, -1), criteria)
                
                obj_points.append(objp)
                img_points.append(corners)
        
        if len(obj_points) < 3:
            print("标定图像不足")
            return False
        
        # 标定
        ret, self.camera_matrix, self.dist_coeffs, rvecs, tvecs = cv2.calibrateCamera(
            obj_points, img_points, gray.shape[::-1], None, None
        )
        
        if ret:
            print("标定成功")
            print(f"焦距: fx={self.camera_matrix[0,0]:.1f}, fy={self.camera_matrix[1,1]:.1f}")
            print(f"光心: cx={self.camera_matrix[0,2]:.1f}, cy={self.camera_matrix[1,2]:.1f}")
            return True
        
        return False
    
    def calibrate_homography(self, image_points, world_points):
        """
        标定透视变换（图像坐标到世界坐标）
        image_points: 图像中的4个点
        world_points: 对应的世界坐标（mm）
        """
        self.homography, _ = cv2.findHomography(
            np.array(image_points, dtype=np.float32),
            np.array(world_points, dtype=np.float32)
        )
        
        return self.homography is not None
    
    def undistort(self, image):
        """去畸变"""
        if self.camera_matrix is None:
            return image
        
        return cv2.undistort(image, self.camera_matrix, self.dist_coeffs)
    
    def pixel_to_world(self, pixel_point):
        """
        像素坐标转世界坐标（mm）
        """
        if self.homography is None:
            return pixel_point
        
        pt = np.array([pixel_point[0], pixel_point[1], 1])
        world_pt = self.homography @ pt
        world_pt /= world_pt[2]
        
        return (world_pt[0], world_pt[1])
    
    def world_to_pixel(self, world_point):
        """
        世界坐标转像素坐标
        """
        if self.homography is None:
            return world_point
        
        inv_homography = np.linalg.inv(self.homography)
        pt = np.array([world_point[0], world_point[1], 1])
        pixel_pt = inv_homography @ pt
        pixel_pt /= pixel_pt[2]
        
        return (int(pixel_pt[0]), int(pixel_pt[1]))
    
    def save_calibration(self, filename):
        """保存标定结果"""
        np.savez(filename,
                 camera_matrix=self.camera_matrix,
                 dist_coeffs=self.dist_coeffs,
                 homography=self.homography)
    
    def load_calibration(self, filename):
        """加载标定结果"""
        data = np.load(filename)
        self.camera_matrix = data['camera_matrix']
        self.dist_coeffs = data['dist_coeffs']
        self.homography = data['homography']
```

### 2. 木棒检测

```python
#!/usr/bin/env python3
"""
木棒检测 - 边缘检测 + 轮廓分析
"""

import cv2
import numpy as np

class StickDetector:
    """木棒检测器"""
    
    def __init__(self, calibrator):
        self.calibrator = calibrator
        
        # 检测参数
        self.min_area = 1000  # 最小面积（像素）
        self.max_area = 100000  # 最大面积
        self.min_length = 50  # 最小长度（像素）
        self.aspect_ratio_threshold = 3.0  # 长宽比阈值
        
        # 颜色范围（木棒颜色）
        self.color_lower = np.array([10, 30, 50])   # HSV下限
        self.color_upper = np.array([30, 200, 200])  # HSV上限
        
        # 背景模型
        self.background = None
        self.bg_subtractor = cv2.createBackgroundSubtractorMOG2(
            history=500, varThreshold=50, detectShadows=False
        )
    
    def set_background(self, frame):
        """设置背景"""
        self.background = frame.copy()
    
    def detect_stick(self, frame):
        """
        检测木棒
        返回: [(x1, y1, x2, y2, length_mm), ...]
        """
        # 去畸变
        undistorted = self.calibrator.undistort(frame)
        
        # 预处理
        processed = self._preprocess(undistorted)
        
        # 查找轮廓
        contours = self._find_contours(processed)
        
        # 筛选木棒
        sticks = []
        for contour in contours:
            stick = self._analyze_contour(contour)
            if stick is not None:
                sticks.append(stick)
        
        return sticks
    
    def _preprocess(self, frame):
        """图像预处理"""
        # 转换到HSV
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 颜色分割
        mask_color = cv2.inRange(hsv, self.color_lower, self.color_upper)
        
        # 背景差分
        if self.background is not None:
            diff = cv2.absdiff(frame, self.background)
            gray_diff = cv2.cvtColor(diff, cv2.COLOR_BGR2GRAY)
            _, mask_motion = cv2.threshold(gray_diff, 30, 255, cv2.THRESH_BINARY)
        else:
            mask_motion = np.ones_like(mask_color) * 255
        
        # 合并掩码
        mask = cv2.bitwise_and(mask_color, mask_motion)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        return mask
    
    def _find_contours(self, mask):
        """查找轮廓"""
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        # 按面积筛选
        filtered = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if self.min_area <= area <= self.max_area:
                filtered.append(contour)
        
        return filtered
    
    def _analyze_contour(self, contour):
        """分析轮廓，判断是否为木棒"""
        # 计算最小外接矩形
        rect = cv2.minAreaRect(contour)
        center, size, angle = rect
        
        width, height = size
        
        # 确保宽度 < 高度
        if width > height:
            width, height = height, width
            angle += 90
        
        # 检查长宽比
        if height < self.min_length:
            return None
        
        aspect_ratio = height / width if width > 0 else 0
        if aspect_ratio < self.aspect_ratio_threshold:
            return None
        
        # 计算端点
        # 将轮廓点转换到最小外接矩形坐标系
        box = cv2.boxPoints(rect)
        box = np.int0(box)
        
        # 找到长轴方向的两个端点
        endpoints = self._find_endpoints(contour, angle)
        
        if endpoints is None:
            return None
        
        # 转换为世界坐标
        world_endpoints = [
            self.calibrator.pixel_to_world(endpoints[0]),
            self.calibrator.pixel_to_world(endpoints[1])
        ]
        
        # 计算实际长度（mm）
        dx = world_endpoints[1][0] - world_endpoints[0][0]
        dy = world_endpoints[1][1] - world_endpoints[0][1]
        length_mm = np.sqrt(dx**2 + dy**2)
        
        return {
            'endpoints_pixel': endpoints,
            'endpoints_world': world_endpoints,
            'length_mm': length_mm,
            'angle': angle,
            'center': center,
            'box': box
        }
    
    def _find_endpoints(self, contour, angle):
        """找到木棒的两个端点"""
        # 计算轮廓的凸包
        hull = cv2.convexHull(contour)
        
        # 找到最远的两个点
        max_dist = 0
        point1 = None
        point2 = None
        
        for i in range(len(hull)):
            for j in range(i+1, len(hull)):
                p1 = tuple(hull[i][0])
                p2 = tuple(hull[j][0])
                
                dist = np.sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)
                
                if dist > max_dist:
                    max_dist = dist
                    point1 = p1
                    point2 = p2
        
        if point1 is not None and point2 is not None:
            return (point1, point2)
        
        return None
    
    def draw_detection(self, frame, sticks):
        """绘制检测结果"""
        display = frame.copy()
        
        for stick in sticks:
            # 绘制轮廓
            cv2.drawContours(display, [stick['box']], 0, (0, 255, 0), 2)
            
            # 绘制端点
            ep = stick['endpoints_pixel']
            cv2.circle(display, ep[0], 5, (0, 0, 255), -1)
            cv2.circle(display, ep[1], 5, (0, 0, 255), -1)
            
            # 绘制长度
            center = (int(stick['center'][0]), int(stick['center'][1]))
            cv2.putText(display, f"{stick['length_mm']:.1f}mm", 
                       (center[0]-30, center[1]-20),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 2)
        
        return display
```

### 3. 长度计算

```python
#!/usr/bin/env python3
"""
长度计算 - 多帧融合 + 滤波
"""

import numpy as np
from collections import deque

class LengthCalculator:
    """长度计算器"""
    
    def __init__(self, calibrator):
        self.calibrator = calibrator
        
        # 测量历史
        self.measurement_history = deque(maxlen=10)
        
        # 滤波参数
        self.filter_window = 5
        
        # 统计信息
        self.total_measured = 0
        self.length_sum = 0
    
    def add_measurement(self, length_mm):
        """添加测量值"""
        self.measurement_history.append(length_mm)
        self.total_measured += 1
        self.length_sum += length_mm
    
    def get_filtered_length(self):
        """获取滤波后的长度"""
        if len(self.measurement_history) == 0:
            return None
        
        # 中值滤波
        lengths = list(self.measurement_history)
        filtered = np.median(lengths)
        
        return filtered
    
    def get_average_length(self):
        """获取平均长度"""
        if self.total_measured == 0:
            return None
        
        return self.length_sum / self.total_measured
    
    def get_statistics(self):
        """获取统计信息"""
        if len(self.measurement_history) == 0:
            return None
        
        lengths = list(self.measurement_history)
        
        return {
            'mean': np.mean(lengths),
            'std': np.std(lengths),
            'min': np.min(lengths),
            'max': np.max(lengths),
            'median': np.median(lengths),
            'count': self.total_measured
        }
    
    def validate_measurement(self, length_mm):
        """验证测量值是否合理"""
        # 检查是否在合理范围内
        if length_mm < 50 or length_mm > 2000:
            return False
        
        # 检查与历史值的偏差
        if len(self.measurement_history) > 0:
            avg = np.mean(list(self.measurement_history))
            if abs(length_mm - avg) > avg * 0.2:  # 偏差超过20%
                return False
        
        return True
    
    def calculate_from_sticks(self, sticks):
        """
        从检测结果计算长度
        """
        if not sticks:
            return None
        
        # 取最长的木棒
        longest = max(sticks, key=lambda s: s['length_mm'])
        
        length_mm = longest['length_mm']
        
        # 验证
        if self.validate_measurement(length_mm):
            self.add_measurement(length_mm)
            return self.get_filtered_length()
        
        return None
    
    def reset(self):
        """重置"""
        self.measurement_history.clear()
        self.total_measured = 0
        self.length_sum = 0
```

### 4. 传送带同步

```python
#!/usr/bin/env python3
"""
传送带同步 - 编码器 + 触发信号
"""

import gpiod
import time
import threading

class ConveyorSync:
    """传送带同步控制器"""
    
    def __init__(self):
        # GPIO配置
        self.chip = gpiod.Chip('1')
        
        # 光电传感器
        self.sensor1 = self.chip.get_line(24)  # GPIO1_A0 - 入口
        self.sensor2 = self.chip.get_line(25)  # GPIO1_A1 - 出口
        
        sensor_config = gpiod.LineSettings(direction=gpiod.Direction.INPUT)
        self.sensor1.request(consumer="conveyor", config=sensor_config)
        self.sensor2.request(consumer="conveyor", config=sensor_config)
        
        # 编码器
        self.encoder_a = self.chip.get_line(48)  # GPIO3_A0
        self.encoder_b = self.chip.get_line(49)  # GPIO3_A1
        
        encoder_config = gpiod.LineSettings(direction=gpiod.Direction.INPUT)
        self.encoder_a.request(consumer="encoder", config=encoder_config)
        self.encoder_b.request(consumer="encoder", config=encoder_config)
        
        # 状态
        self.is_running = False
        self.conveyor_speed = 0  # mm/s
        self.encoder_count = 0
        self.last_encoder_time = 0
        
        # 木棒队列
        self.stick_queue = []
        
        # 回调函数
        self.on_stick_detected = None
        self.on_stick_passed = None
        
        # 线程
        self.monitor_thread = None
    
    def start(self):
        """启动监控"""
        self.is_running = True
        self.monitor_thread = threading.Thread(target=self._monitor_loop)
        self.monitor_thread.daemon = True
        self.monitor_thread.start()
    
    def stop(self):
        """停止监控"""
        self.is_running = False
        if self.monitor_thread:
            self.monitor_thread.join()
    
    def _monitor_loop(self):
        """监控循环"""
        last_sensor1 = 0
        last_sensor2 = 0
        
        while self.is_running:
            # 读取传感器
            sensor1_val = self.sensor1.get_value()
            sensor2_val = self.sensor2.get_value()
            
            # 检测木棒进入
            if sensor1_val == 1 and last_sensor1 == 0:
                self._on_stick_enter()
            
            # 检测木棒离开
            if sensor2_val == 1 and last_sensor2 == 0:
                self._on_stick_exit()
            
            last_sensor1 = sensor1_val
            last_sensor2 = sensor2_val
            
            # 更新编码器
            self._update_encoder()
            
            time.sleep(0.001)  # 1kHz采样
    
    def _on_stick_enter(self):
        """木棒进入"""
        stick_info = {
            'enter_time': time.time(),
            'enter_position': self.encoder_count,
            'exit_time': None,
            'exit_position': None
        }
        
        self.stick_queue.append(stick_info)
        
        if self.on_stick_detected:
            self.on_stick_detected(stick_info)
    
    def _on_stick_exit(self):
        """木棒离开"""
        if self.stick_queue:
            stick_info = self.stick_queue[-1]
            stick_info['exit_time'] = time.time()
            stick_info['exit_position'] = self.encoder_count
            
            if self.on_stick_passed:
                self.on_stick_passed(stick_info)
    
    def _update_encoder(self):
        """更新编码器计数"""
        a_val = self.encoder_a.get_value()
        current_time = time.time()
        
        # 简化的编码器读取
        if a_val == 1 and current_time - self.last_encoder_time > 0.001:
            self.encoder_count += 1
            self.last_encoder_time = current_time
            
            # 计算速度
            # 假设编码器每转100脉冲，轮子周长200mm
            pulses_per_rev = 100
            wheel_circumference = 200  # mm
            
            # 速度 = 脉冲频率 * 周长 / 每转脉冲数
            self.conveyor_speed = wheel_circumference / pulses_per_rev * 1000  # mm/s
    
    def get_speed(self):
        """获取传送带速度"""
        return self.conveyor_speed
    
    def get_position(self):
        """获取当前位置"""
        # 假设编码器每转100脉冲，轮子周长200mm
        pulses_per_rev = 100
        wheel_circumference = 200  # mm
        
        return self.encoder_count * wheel_circumference / pulses_per_rev
    
    def get_stick_info(self):
        """获取当前木棒信息"""
        if self.stick_queue:
            return self.stick_queue[-1]
        return None
```

        self.calculator = LengthCalculator(self.calibrator)
        self.conveyor = ConveyorSync()
        
        # 摄像头
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1920)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 1080)
        
        # 运行状态
        self.is_running = False
        self.is_measuring = False
        
        # 测量结果
        self.current_length = None
        self.measurement_history = []
    
    def initialize(self):
        """初始化系统"""
        print("初始化测量系统...")
        
        # 加载标定
        try:
            self.calibrator.load_calibration('calibration.npz')
            print("标定文件已加载")
        except:
            print("需要进行标定")
            self._do_calibration()
        
        # 设置背景
        print("请移除传送带上的物体，按Enter拍摄背景...")
        input()
        ret, frame = self.cap.read()
        if ret:
            self.detector.set_background(frame)
            print("背景已设置")
        
        # 启动传送带监控
        self.conveyor.start()
        self.conveyor.on_stick_detected = self._on_stick_detected
        self.conveyor.on_stick_passed = self._on_stick_passed
        
        print("初始化完成")
    
    def _do_calibration(self):
        """进行标定"""
        print("开始标定...")
        print("请拍摄棋盘格图像 (至少3张)")
        
        images = []
        for i in range(5):
            print(f"拍摄第 {i+1} 张，按Enter...")
            input()
            ret, frame = self.cap.read()
            if ret:
                images.append(frame)
                cv2.imshow('Calibration', frame)
                cv2.waitKey(1000)
        
        cv2.destroyAllWindows()
        
        if len(images) >= 3:
            self.calibrator.calibrate_intrinsics(images)
            self.calibrator.save_calibration('calibration.npz')
            print("标定完成")
        else:
            print("标定图像不足")
    
    def _on_stick_detected(self, stick_info):
        """木棒进入检测区域"""
        self.is_measuring = True
        print("检测到木棒，开始测量...")
    
    def _on_stick_passed(self, stick_info):
        """木棒离开检测区域"""
        self.is_measuring = False
        
        if self.current_length is not None:
            self.measurement_history.append({
                'time': time.time(),
                'length': self.current_length,
                'speed': self.conveyor.get_speed()
            })
            print(f"测量完成: {self.current_length:.1f} mm")
    
    def run(self):
        """主循环"""
        self.initialize()
        
        print("\n开始测量 (按 'q' 退出)")
        print("按 'r' 重置统计")
        print("按 's' 保存结果")
        
        self.is_running = True
        
        try:
            while self.is_running:
                # 获取图像
                ret, frame = self.cap.read()
                if not ret:
                    continue
                
                # 检测木棒
                sticks = self.detector.detect_stick(frame)
                
                # 计算长度
                if sticks:
                    length = self.calculator.calculate_from_sticks(sticks)
                    if length is not None:
                        self.current_length = length
                
                # 绘制结果
                display = self.detector.draw_detection(frame, sticks)
                
                # 添加信息
                self._draw_info(display)
                
                # 显示
                cv2.imshow('Stick Measurement', display)
                
                # 键盘控制
                key = cv2.waitKey(1) & 0xFF
                if key == ord('q'):
                    break
                elif key == ord('r'):
                    self.calculator.reset()
                    self.measurement_history.clear()
                    print("统计数据已重置")
                elif key == ord('s'):
                    self._save_results()
        
        except KeyboardInterrupt:
            print("\n用户中断")
        
        finally:
            self.stop()
    
    def stop(self):
        """停止系统"""
        self.is_running = False
        self.conveyor.stop()
        self.cap.release()
        cv2.destroyAllWindows()
    
    def _draw_info(self, frame):
        """绘制信息"""
        # 当前长度
        if self.current_length is not None:
            cv2.putText(frame, f"Length: {self.current_length:.1f} mm", 
                       (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
        
        # 统计信息
        stats = self.calculator.get_statistics()
        if stats:
            cv2.putText(frame, f"Avg: {stats['mean']:.1f} mm", 
                       (10, 70), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 1)
            cv2.putText(frame, f"Std: {stats['std']:.1f} mm", 
                       (10, 100), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 1)
            cv2.putText(frame, f"Count: {stats['count']}", 
                       (10, 130), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 1)
        
        # 传送带速度
        speed = self.conveyor.get_speed()
        cv2.putText(frame, f"Speed: {speed:.1f} mm/s", 
                   (10, 170), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 1)
        
        # 状态
        status = "Measuring" if self.is_measuring else "Idle"
        cv2.putText(frame, f"Status: {status}", 
                   (10, 200), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 255), 1)
    
    def _save_results(self):
        """保存结果"""
        filename = f"measurement_{time.strftime('%Y%m%d_%H%M%S')}.csv"
        
        with open(filename, 'w') as f:
            f.write("时间,长度(mm),速度(mm/s)\n")
            for item in self.measurement_history:
                f.write(f"{item['time']},{item['length']:.1f},{item['speed']:.1f}\n")
        
        print(f"结果已保存到 {filename}")

def main():
    system = StickMeasurementSystem()
    system.run()

if __name__ == '__main__':
    main()

```python
def subpixel_edge_detection(image, contour):
    """亚像素边缘检测"""
    # 转换为灰度
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    # 亚像素精化
    criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001)
    
    # 精化轮廓点
    contour_float = contour.astype(np.float32)
    refined = cv2.cornerSubPix(gray, contour_float, (5, 5), (-1, -1), criteria)
    
    return refined
```

### 2. 多帧融合

```python
def multi_frame_fusion(measurements, weights=None):
    """
    多帧融合
    measurements: [(length, confidence), ...]
    """
    if weights is None:
        weights = [1.0] * len(measurements)
    
    total_weight = sum(weights)
    weighted_sum = sum(m * w for m, w in zip(measurements, weights))
    
    return weighted_sum / total_weight
```

### 3. 温度补偿

```python
def temperature_compensation(length_mm, temperature):
    """
    温度补偿
    考虑热胀冷缩
    """
    # 木材线膨胀系数 (大约)
    alpha = 5e-6  # 每摄氏度
    
    # 参考温度
    T_ref = 20.0
    
    # 补偿
    compensated = length_mm * (1 + alpha * (temperature - T_ref))
    
    return compensated
```

        self.result = None
        self.running = False

        self.running = True
        self.thread = threading.Thread(target=self._process_loop)
