## 一、题目分析

### 1.1 核心要求
- 设计声音导引系统，通过声音信号引导移动声源到达指定位置
- 系统包含1个可移动声源和3个声音接收器（A、B、C）
- 声源发出特定频率声音，接收器检测信号强度
- 根据信号强度差异计算声源位置并导引其移动

### 1.2 技术难点
- 声音信号的精确采集与处理
- 基于声音强度的定位算法
- 实时位置计算与运动控制
- 导引精度与响应速度的平衡

```python
#!/usr/bin/env python3
"""
声音采集模块 - Orange Pi 5
使用I2S接口采集3路麦克风信号
"""

import numpy as np
import pyaudio
import struct
from collections import deque

class SoundCapture:
    def __init__(self, sample_rate=44100, chunk_size=1024, channels=3):
        self.sample_rate = sample_rate
        self.chunk_size = chunk_size
        self.channels = channels
        self.audio = pyaudio.PyAudio()
        self.stream = None
        self.buffer = deque(maxlen=10)  # 10帧缓冲
        
    def start(self):
        """启动音频采集"""
        self.stream = self.audio.open(
            format=pyaudio.paInt32,
            channels=self.channels,
            rate=self.sample_rate,
            input=True,
            frames_per_buffer=self.chunk_size,
            input_device_index=self._find_i2s_device()
        )
        print(f"音频采集已启动: {self.sample_rate}Hz, {self.channels}通道")
        
    def _find_i2s_device(self):
        """查找I2S音频设备"""
        for i in range(self.audio.get_device_count()):
            dev_info = self.audio.get_device_info_by_index(i)
            if 'I2S' in dev_info['name'] or 'i2s' in dev_info['name']:
                return i
        return 0  # 默认设备
        
    def read_chunk(self):
        """读取一帧音频数据"""
        if self.stream is None:
            return None
            
        try:
            data = self.stream.read(self.chunk_size, exception_on_overflow=False)
            # 解析为32位整数
            samples = struct.unpack(f'{self.chunk_size * self.channels}i', data)
            
            # 分离各通道
            ch0 = np.array(samples[0::3], dtype=np.float64) / (2**31)
            ch1 = np.array(samples[1::3], dtype=np.float64) / (2**31)
            ch2 = np.array(samples[2::3], dtype=np.float64) / (2**31)
            
            return {'mic_a': ch0, 'mic_b': ch1, 'mic_c': ch2}
        except Exception as e:
            print(f"音频读取错误: {e}")
            return None
            
    def stop(self):
        """停止音频采集"""
        if self.stream:
            self.stream.stop_stream()
            self.stream.close()
        self.audio.terminate()
```

### 4.3 信号处理模块

```python
#!/usr/bin/env python3
"""
信号处理模块 - 声音信号的滤波与特征提取
"""

import numpy as np
from scipy import signal
from scipy.fft import fft, fftfreq

class SignalProcessor:
    def __init__(self, sample_rate=44100, target_freq=2000):
        self.sample_rate = sample_rate
        self.target_freq = target_freq
        self.nyquist = sample_rate / 2
        
        # 设计带通滤波器（以目标频率为中心）
        low = (target_freq - 200) / self.nyquist
        high = (target_freq + 200) / self.nyquist
        self.b, self.a = signal.butter(4, [low, high], btype='band')
        
    def filter_signal(self, data):
        """带通滤波"""
        return signal.filtfilt(self.b, self.a, data)
        
    def calculate_rms(self, data):
        """计算信号RMS值"""
        filtered = self.filter_signal(data)
        return np.sqrt(np.mean(filtered**2))
        
    def calculate_amplitude(self, data):
        """计算信号幅度（峰值检测）"""
        filtered = self.filter_signal(data)
        peaks, _ = signal.find_peaks(np.abs(filtered), height=0)
        if len(peaks) > 0:
            return np.mean(np.abs(filtered[peaks]))
        return 0
        
    def cross_correlate(self, sig1, sig2):
        """计算两路信号的互相关，用于TDOA"""
        # 希尔伯特变换提取包络
        analytic1 = signal.hilbert(sig1)
        analytic2 = signal.hilbert(sig2)
        
        env1 = np.abs(analytic1)
        env2 = np.abs(analytic2)
        
        # 互相关
        correlation = signal.correlate(env1, env2, mode='full')
        lags = signal.correlation_lags(len(env1), len(env2), mode='full')
        
        # 找到峰值对应的延迟
        peak_idx = np.argmax(np.abs(correlation))
        delay_samples = lags[peak_idx]
        delay_time = delay_samples / self.sample_rate
        
        return delay_time, correlation[peak_idx]
        
    def extract_features(self, data):
        """提取声音特征"""
        filtered = self.filter_signal(data)
        
        features = {
            'rms': self.calculate_rms(data),
            'amplitude': self.calculate_amplitude(data),
            'zero_crossings': self._count_zero_crossings(filtered),
            'spectral_centroid': self._spectral_centroid(filtered)
        }
        
        return features
        
    def _count_zero_crossings(self, data):
        """计算过零率"""
        return np.sum(np.diff(np.sign(data)) != 0)
        
    def _spectral_centroid(self, data):
        """计算频谱质心"""
        spectrum = np.abs(fft(data))
        freqs = fftfreq(len(data), 1/self.sample_rate)
        positive_mask = freqs > 0
        return np.sum(freqs[positive_mask] * spectrum[positive_mask]) / np.sum(spectrum[positive_mask])
```

### 4.4 声音定位算法

```python
#!/usr/bin/env python3
"""
声音定位算法模块
基于TDOA（到达时间差）和RSSI（接收信号强度）的混合定位
"""

import numpy as np
from scipy.optimize import minimize

class SoundLocalizer:
    def __init__(self, mic_positions, method='tdoa'):
        """
        mic_positions: 麦克风位置坐标 [(x1,y1), (x2,y2), (x3,y3)]
        method: 'tdoa' 或 'rssi'
        """
        self.mic_positions = np.array(mic_positions)
        self.method = method
        self.sound_speed = 343  # 声速 m/s
        
    def localize_tdoa(self, delays):
        """
        基于TDOA的三边定位
        delays: 到达时间差 [τ_AB, τ_AC]
        """
        # 转换为距离差
        d_AB = delays[0] * self.sound_speed
        d_AC = delays[1] * self.sound_speed
        
        # 双曲线定位
        x1, y1 = self.mic_positions[0]
        x2, y2 = self.mic_positions[1]
        x3, y3 = self.mic_positions[2]
        
        # 建立方程组求解
        def equations(pos):
            x, y = pos
            d1 = np.sqrt((x-x1)**2 + (y-y1)**2)
            d2 = np.sqrt((x-x2)**2 + (y-y2)**2)
            d3 = np.sqrt((x-x3)**2 + (y-y3)**2)
            
            eq1 = (d2 - d1) - d_AB
            eq2 = (d3 - d1) - d_AC
            
            return eq1**2 + eq2**2
            
        # 初始猜测（三角形中心）
        x0 = np.mean(self.mic_positions[:, 0])
        y0 = np.mean(self.mic_positions[:, 1])
        
        result = minimize(equations, [x0, y0], method='Nelder-Mead')
        
        if result.success:
            return result.x
        else:
            return None
            
    def localize_rssi(self, amplitudes):
        """
        基于RSSI的三边定位
        amplitudes: 各麦克风接收的信号幅度 [A1, A2, A3]
        """
        # 假设声源功率恒定，距离与幅度成反比
        # A = k / d^2
        # d = sqrt(k / A)
        
        # 归一化幅度
        A_norm = amplitudes / np.max(amplitudes)
        
        # 估计相对距离
        distances = 1.0 / np.sqrt(A_norm + 1e-6)
        distances = distances / np.max(distances)  # 归一化
        
        # 加权质心法
        weights = 1.0 / (distances**2 + 1e-6)
        weights = weights / np.sum(weights)
        
        position = np.zeros(2)
        for i in range(3):
            position += weights[i] * self.mic_positions[i]
            
        return position
        
    def localize_hybrid(self, delays, amplitudes, alpha=0.7):
        """
        混合定位算法
        alpha: TDOA权重 (0-1)
        """
        pos_tdoa = self.localize_tdoa(delays)
        pos_rssi = self.localize_rssi(amplitudes)
        
        if pos_tdoa is None:
            return pos_rssi
            
        return alpha * pos_tdoa + (1 - alpha) * pos_rssi
        
    def calculate_angle(self, source_pos):
        """计算声源相对于原点的角度"""
        dx = source_pos[0] - self.mic_positions[0][0]
        dy = source_pos[1] - self.mic_positions[0][1]
        angle = np.arctan2(dy, dx)
        return np.degrees(angle)
```

### 4.5 运动控制模块

```python
#!/usr/bin/env python3
"""
运动控制模块 - 差速驱动小车的PID控制
"""

import time
import numpy as np

class MotionController:
    def __init__(self, wheel_base=0.15, wheel_radius=0.0325):
        self.wheel_base = wheel_base  # 两轮间距
        self.wheel_radius = wheel_radius  # 轮子半径
        
        # PID参数
        self.pid_linear = {'kp': 1.0, 'ki': 0.1, 'kd': 0.05}
        self.pid_angular = {'kp': 2.0, 'ki': 0.2, 'kd': 0.1}
        
        # 状态变量
        self.position = np.array([0.0, 0.0])  # x, y
        self.theta = 0.0  # 朝向角
        self.left_speed = 0
        self.right_speed = 0
        
        # PID误差累积
        self.linear_error_sum = 0
        self.linear_error_prev = 0
        self.angular_error_sum = 0
        self.angular_error_prev = 0
        
    def update_odometry(self, left_encoder, right_encoder, dt):
        """根据编码器更新里程计"""
        # 计算轮子线速度
        v_left = left_encoder * self.wheel_radius / dt
        v_right = right_encoder * self.wheel_radius / dt
        
        # 计算车体线速度和角速度
        v = (v_left + v_right) / 2
        omega = (v_right - v_left) / self.wheel_base
        
        # 更新位姿
        self.theta += omega * dt
        self.theta = self.theta % (2 * np.pi)
        
        self.position[0] += v * np.cos(self.theta) * dt
        self.position[1] += v * np.sin(self.theta) * dt
        
    def navigate_to_target(self, target_pos):
        """
        导航到目标位置
        返回: (左轮速度, 右轮速度)
        """
        # 计算误差
        dx = target_pos[0] - self.position[0]
        dy = target_pos[1] - self.position[1]
        
        distance = np.sqrt(dx**2 + dy**2)
        target_angle = np.arctan2(dy, dx)
        
        # 角度误差（归一化到-pi到pi）
        angle_error = target_angle - self.theta
        while angle_error > np.pi:
            angle_error -= 2 * np.pi
        while angle_error < -np.pi:
            angle_error += 2 * np.pi
            
        # PID控制
        linear_output = self._pid_control(
            distance, self.pid_linear,
            'linear'
        )
        
        angular_output = self._pid_control(
            angle_error, self.pid_angular,
            'angular'
        )
        
        # 限制速度
        max_speed = 0.5  # m/s
        linear_output = np.clip(linear_output, -max_speed, max_speed)
        angular_output = np.clip(angular_output, -2.0, 2.0)
        
        # 转换为左右轮速度
        v_left = linear_output - angular_output * self.wheel_base / 2
        v_right = linear_output + angular_output * self.wheel_base / 2
        
        return v_left, v_right
        
    def _pid_control(self, error, pid_params, control_type):
        """PID控制器"""
        kp = pid_params['kp']
        ki = pid_params['ki']
        kd = pid_params['kd']
        
        if control_type == 'linear':
            self.linear_error_sum += error
            derivative = error - self.linear_error_prev
            self.linear_error_prev = error
            output = kp * error + ki * self.linear_error_sum + kd * derivative
        else:
            self.angular_error_sum += error
            derivative = error - self.angular_error_prev
            self.angular_error_prev = error
            output = kp * error + ki * self.angular_error_sum + kd * derivative
            
        return output
        
    def stop(self):
        """停止运动"""
        self.left_speed = 0
        self.right_speed = 0
        self.linear_error_sum = 0
        self.linear_error_prev = 0
        self.angular_error_sum = 0
        self.angular_error_prev = 0
        
    def get_status(self):
        """获取当前状态"""
        return {
            'position': self.position.tolist(),
            'theta': np.degrees(self.theta),
            'left_speed': self.left_speed,
            'right_speed': self.right_speed
        }
```

```python
#!/usr/bin/env python3
"""
声音导引系统主程序 - Orange Pi 5
"""

import time
import threading
import numpy as np
from sound_capture import SoundCapture
from signal_processor import SignalProcessor
from sound_localizer import SoundLocalizer
from motion_controller import MotionController

class SoundGuidanceSystem:
    def __init__(self):
        # 麦克风位置（等边三角形布局，边长0.5m）
        self.mic_positions = [
            (0.0, 0.0),      # 麦克风A
            (0.5, 0.0),      # 麦克风B
            (0.25, 0.433)    # 麦克风C
        ]
        
        # 目标频率
        self.target_freq = 2000  # Hz
        
        # 初始化各模块
        self.sound_capture = SoundCapture(sample_rate=44100, chunk_size=2048)
        self.signal_processor = SignalProcessor(sample_rate=44100, target_freq=self.target_freq)
        self.localizer = SoundLocalizer(self.mic_positions, method='hybrid')
        self.motion_controller = MotionController()
        
        # 目标位置
        self.target_position = np.array([0.25, 0.15])  # 默认目标
        
        # 运行标志
        self.running = False
        
        # 统计信息
        self.stats = {
            'position_history': [],
            'update_count': 0,
            'start_time': 0
        }
        
    def start(self):
        """启动系统"""
        print("=" * 50)
        print("声音导引系统启动")
        print("=" * 50)
        
        self.sound_capture.start()
        self.running = True
        self.stats['start_time'] = time.time()
        
        # 启动处理线程
        self.process_thread = threading.Thread(target=self._process_loop)
        self.process_thread.start()
        
        print(f"目标位置: {self.target_position}")
        print("等待声源信号...")
        
    def _process_loop(self):
        """主处理循环"""
        while self.running:
            # 读取音频数据
            audio_data = self.sound_capture.read_chunk()
            if audio_data is None:
                time.sleep(0.01)
                continue
                
            # 提取各通道数据
            mic_a = audio_data['mic_a']
            mic_b = audio_data['mic_b']
            mic_c = audio_data['mic_c']
            
            # 计算信号幅度
            amp_a = self.signal_processor.calculate_amplitude(mic_a)
            amp_b = self.signal_processor.calculate_amplitude(mic_b)
            amp_c = self.signal_processor.calculate_amplitude(mic_c)
            
            # 计算互相关延迟（TDOA）
            delay_ab, _ = self.signal_processor.cross_correlate(mic_a, mic_b)
            delay_ac, _ = self.signal_processor.cross_correlate(mic_a, mic_c)
            
            # 定位声源
            position = self.localizer.localize_hybrid(
                delays=[delay_ab, delay_ac],
                amplitudes=[amp_a, amp_b, amp_c],
                alpha=0.7
            )
            
            if position is not None:
                # 计算到目标的距离和角度
                distance = np.linalg.norm(position - self.target_position)
                angle = self.localizer.calculate_angle(position)
```

```python
#!/usr/bin/env python3
"""
视觉辅助定位模块 - 使用摄像头辅助校准声源位置
"""

import cv2
import numpy as np

class VisionAssist:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        # 声源标记颜色范围（假设声源有LED标记）
        self.color_lower = np.array([0, 100, 100])  # HSV下界
        self.color_upper = np.array([10, 255, 255])  # HSV上界
        
    def detect_source(self):
        """检测声源位置"""
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 转换到HSV颜色空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 颜色过滤
        mask = cv2.inRange(hsv, self.color_lower, self.color_upper)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.erode(mask, kernel, iterations=2)
        mask = cv2.dilate(mask, kernel, iterations=2)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找到最大轮廓
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if area > 100:  # 最小面积阈值
                M = cv2.moments(largest)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    return (cx, cy, area)
                    
        return None
        
    def release(self):
        """释放资源"""
        self.cap.release()
```

```python
    axes[0].plot(t, audio_data['mic_b'], label='Mic B')

    axes[0].set_ylabel('幅度')
    axes[0].legend()
    axes[0].set_title('时域波形')

    for i, (key, data) in enumerate(audio_data.items()):
```
