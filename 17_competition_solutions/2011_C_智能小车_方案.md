# 2011年电赛C题 - 智能小车 Orange Pi 5解决方案

## 一、题目分析

两辆智能小车在跑道上交替领跑, 实现循迹行驶、超车、领跑切换等功能。

### 核心要求
1. 视觉循迹 (黑线/白线跟踪)
2. 双车通信协调
3. 交替领跑与超车
4. 速度控制与安全距离

---

```
│ └─────────────────┘ │    │ └─────────────────┘ │

└─────────────────────┘    └─────────────────────┘
         ↕  WiFi通信 (UDP)
```

### 领跑策略
1. A车在前领跑, B车跟随保持安全距离
2. 到达切换点, B车加速超车到A车前方
3. B车成为领跑车, A车跟随
4. 循环交替

### 4.1 视觉循迹

```python
import cv2
import numpy as np

class LineTracker:
    """视觉循迹模块"""

    def __init__(self, roi_y=300, roi_h=180):
        self.roi_y = roi_y    # 感兴趣区域起始y
        self.roi_h = roi_h    # ROI高度

        # PID参数
        self.Kp = 0.5
        self.Ki = 0.01
        self.Kd = 0.3
        self.integral = 0
        self.prev_error = 0

    def detect_line(self, frame):
        """
        检测赛道线位置
        返回: (offset, angle, line_pts)
        offset: 偏移量 (-1.0~1.0), 负=偏左, 正=偏右
        angle: 线角度 (用于弯道预判)
        """
        h, w = frame.shape[:2]

        # ROI裁剪
        roi = frame[self.roi_y:self.roi_y+self.roi_h, :]

        # 转灰度+二值化
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        # 自适应阈值
        binary = cv2.adaptiveThreshold(
            gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 11, 2
        )

        # 分段扫描 (3段, 从近到远)
        offsets = []
        for i in range(3):
            y_start = i * (self.roi_h // 3)
            y_end = y_start + (self.roi_h // 3)
            segment = binary[y_start:y_end, :]

            # 扫描每行找黑线位置
            line_positions = []
            for row in segment:
                indices = np.where(row > 0)[0]
                if len(indices) > 0:
                    line_positions.append(np.mean(indices))

            if line_positions:
                center_x = np.mean(line_positions)
                offset = (center_x - w/2) / (w/2)  # 归一化
                offsets.append(offset)

        if not offsets:
            return 0, 0, None

        # 主偏移量 (近处权重更大)
        main_offset = offsets[0] * 0.5 + offsets[1] * 0.3 + \
                      offsets[2] * 0.2 if len(offsets) >= 3 else offsets[0]

        # 弯道角度估算
        angle = 0
        if len(offsets) >= 2:
            angle = offsets[-1] - offsets[0]  # 近远差值

        return main_offset, angle, offsets

    def compute_steer(self, offset, angle):
        """
        循迹PID转向计算
        返回: 差速转向值 (-1.0~1.0)
        """
        # PID
        error = offset
        self.integral += error
        self.integral = max(-10, min(10, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error

        steer = (self.Kp * error +
                 self.Ki * self.integral +
                 self.Kd * derivative)

        # 弯道预补偿
        steer += 0.2 * angle

        return max(-1.0, min(1.0, steer))
```

### 4.2 双车通信与协调

```python
import socket
import json
import threading
import time

class CarCommunicator:
    """双车WiFi UDP通信"""

    def __init__(self, car_id, remote_ip, local_port=5000, remote_port=5001):
        """
        car_id: 'A' or 'B'
        """
        self.car_id = car_id
        self.remote_ip = remote_ip

        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock.bind(('0.0.0.0', local_port))
        self.sock.settimeout(0.01)

        self.remote_port = remote_port

        # 接收到的对方状态
        self.peer_state = {
            'position': 0,      # 前方距离
            'speed': 0,
            'mode': 'FOLLOW',
            'line_offset': 0,
        }

        # 接收线程
        self.running = True
        self.rx_thread = threading.Thread(target=self._receive_loop)
        self.rx_thread.daemon = True
        self.rx_thread.start()

    def send_state(self, state):
        """发送本车状态"""
        data = json.dumps({
            'id': self.car_id,
            'position': state.get('position', 0),
            'speed': state.get('speed', 0),
            'mode': state.get('mode', 'FOLLOW'),
            'line_offset': state.get('line_offset', 0),
            'timestamp': time.time(),
        }).encode()
        self.sock.sendto(data, (self.remote_ip, self.remote_port))

    def _receive_loop(self):
        """接收线程"""
        while self.running:
            try:
                data, addr = self.sock.recvfrom(1024)
                msg = json.loads(data.decode())
                if msg.get('id') != self.car_id:
                    self.peer_state = msg
            except socket.timeout:
                continue
            except Exception as e:
                pass

    def get_peer_state(self):
        return self.peer_state.copy()

    def close(self):
        self.running = False
        self.sock.close()
```

### 4.3 主程序

```python
import cv2
import time
import serial
import struct
from line_tracker import LineTracker
from communicator import CarCommunicator

class SmartCar:
    """智能小车主程序"""

    def __init__(self, car_id='A', peer_ip='192.168.1.100'):
        self.car_id = car_id

        self.cap = cv2.VideoCapture(0)
        self.tracker = LineTracker()

        port = 5000 if car_id == 'A' else 5001
        remote_port = 5001 if car_id == 'A' else 5000
        self.comm = CarCommunicator(car_id, peer_ip, port, remote_port)

        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)

        # 状态
        self.mode = 'STOP'    # STOP / LEAD / FOLLOW / OVERTAKE
        self.speed = 0.0      # 基础速度
        self.steer = 0.0      # 转向
        self.distance_front = 100.0  # 前方距离(cm)
        self.distance_rear = 100.0   # 后方距离(cm)

        # 配置
        self.base_speed = 0.5         # 基础速度
        self.safe_distance = 30.0     # 安全距离(cm)
        self.overtake_speed = 0.8     # 超车速度
        self.lap_count = 0            # 圈数
        self.overtake_zone = False    # 是否在超车区

    def send_motor_cmd(self, left_speed, right_speed):
        """发送电机指令"""
        buf = bytearray(10)
        buf[0] = 0xAA
        struct.pack_into('<f', buf, 1, left_speed)
        struct.pack_into('<f', buf, 5, right_speed)
        buf[9] = 0x55
        self.ser.write(buf)

    def read_sensors(self):
        """读取传感器"""
        if self.ser.in_waiting >= 12:
            data = self.ser.read(12)
            if data[0] == 0xBB:
                df, dr = struct.unpack('<ff', data[1:9])
                self.distance_front = df
                self.distance_rear = dr

    def detect_overtake_zone(self, frame):
        """检测超车区标记"""
        # 检测特定标记 (如红色区域)
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        lower = np.array([0, 150, 150])
        upper = np.array([10, 255, 255])
        mask = cv2.inRange(hsv, lower, upper)

        # 检查ROI区域是否有红色标记
        roi = mask[200:400, 250:390]
        red_ratio = np.count_nonzero(roi) / roi.size

        return red_ratio > 0.1

    def run(self):
        print(f"=== 智能小车 {self.car_id} 启动 ===")
        print("指令: s=开始, p=停止, 1=领跑, 2=跟随")

        import numpy as np
        last_time = time.time()

        while True:
            ret, frame = self.cap.read()
            if not ret:
                continue

            dt = time.time() - last_time
            last_time = time.time()

            # 读取传感器
            self.read_sensors()

            # 循迹检测
            offset, angle, _ = self.tracker.detect_line(frame)
            self.steer = self.tracker.compute_steer(offset, angle)

            # 超车区检测
            self.overtake_zone = self.detect_overtake_zone(frame)

            # 获取对方状态
            peer = self.comm.get_peer_state()

            # 状态机
            if self.mode == 'LEAD':
                # 领跑模式
                self.speed = self.base_speed

                # 检查是否需要切换
                if self.overtake_zone and peer.get('mode') == 'FOLLOW':
                    # 发送超车信号
                    self.comm.send_state({
                        'position': self.distance_front,
                        'speed': self.speed,
                        'mode': 'OVERTAKE_SIGNAL',
                        'line_offset': offset
                    })
                    self.mode = 'SLOW_DOWN'  # 减速让后车超车
                    print("发出超车信号, 减速...")

            elif self.mode == 'FOLLOW':
                # 跟随模式
                # 根据前车距离调节速度
                if self.distance_front > self.safe_distance * 2:
                    self.speed = self.base_speed * 1.2  # 加速追赶
                elif self.distance_front < self.safe_distance:
                    self.speed = self.base_speed * 0.5  # 减速保持距离
                else:
                    self.speed = self.base_speed

            elif self.mode == 'OVERTAKE':
                # 超车模式: 加速+偏移轨迹
                self.speed = self.overtake_speed
                self.steer += 0.3  # 向外偏移超车

                if self.distance_rear > self.safe_distance * 3:
                    # 已超车完成
                    self.mode = 'LEAD'
                    print("超车完成, 成为领跑车!")

            elif self.mode == 'SLOW_DOWN':
                # 减速模式
                self.speed = self.base_speed * 0.3
                if peer.get('mode') == 'OVERTAKE':
                    # 等后车开始超车
                    pass

            elif self.mode == 'STOP':
                self.speed = 0
                self.steer = 0

            # 计算差速
            left_speed = self.speed + self.steer
            right_speed = self.speed - self.steer

            # 发送电机指令
            self.send_motor_cmd(left_speed, right_speed)

            # 发送状态
            self.comm.send_state({
                'position': self.distance_front,
                'speed': self.speed,
                'mode': self.mode,
                'line_offset': offset
            })

            # 可视化
            self._draw(frame, offset, angle)

            key = cv2.waitKey(1) & 0xFF
            if key == ord('q'):
                break
            elif key == ord('s'):
                self.mode = 'LEAD' if self.car_id == 'A' else 'FOLLOW'
                print(f"模式: {self.mode}")
            elif key == ord('p'):
                self.mode = 'STOP'
            elif key == ord('1'):
                self.mode = 'LEAD'
            elif key == ord('2'):
                self.mode = 'FOLLOW'
            elif key == ord('o'):
                self.mode = 'OVERTAKE'

        self.send_motor_cmd(0, 0)
        self.comm.close()
        self.cap.release()
        cv2.destroyAllWindows()

    def _draw(self, frame, offset, angle):
        """绘制信息"""
        cv2.putText(frame, f"Car {self.car_id}", (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
        cv2.putText(frame, f"Mode: {self.mode}", (10, 50),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)
        cv2.putText(frame, f"Speed: {self.speed:.2f}", (10, 75),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 255), 2)
        cv2.putText(frame, f"Steer: {self.steer:.2f}", (10, 100),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 200, 0), 2)
        cv2.putText(frame, f"Front: {self.distance_front:.0f}cm",
                    (10, 125), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (200, 200, 200), 2)

        # 循迹偏移指示
        h, w = frame.shape[:2]
        cx = int(w/2 + offset * w/4)
        cv2.circle(frame, (cx, h-30), 10, (0, 0, 255), -1)
        cv2.line(frame, (w//2, h-40), (w//2, h-20), (255, 255, 255), 2)

        cv2.imshow(f'Car {self.car_id}', frame)

if __name__ == '__main__':
    import sys
    car_id = sys.argv[1] if len(sys.argv) > 1 else 'A'
    peer_ip = sys.argv[2] if len(sys.argv) > 2 else '192.168.1.100'
    car = SmartCar(car_id, peer_ip)
    car.run()
```

## 五、调试要点

### 5.1 循迹PID
| 参数 | 推荐值 | 说明 |
|------|--------|------|
| Kp | 0.3~0.8 | 直道响应 |
| Ki | 0~0.02 | 消除偏移 |
| Kd | 0.1~0.5 | 弯道阻尼 |

### 5.2 双车协调
1. **WiFi通信延迟**: < 50ms, 需要稳定的热点
2. **超车时机**: 必须在直道超车区进行
3. **安全距离**: 根据车速动态调整
4. **编号区分**: A车先领跑, B车先跟随
