# 2011年电赛B题 - 基于自由摆的平板控制系统 Orange Pi 5解决方案

## 一、题目分析

自由摆下端连接平板, 通过控制平板的倾斜角度, 使平板上放置的物体在摆运动过程中保持稳定不滑落。

### 核心要求
1. 实时检测摆的角度和角速度
2. 控制平板始终保持水平 (或补偿角度)
3. 硬币/物体在摆动过程中不滑落
4. 可实现平板角度主动控制

---

## 四、软件方案

### 4.1 IMU数据融合与摆角计算

```c
// imu_fusion.c - STM32上运行
#include "stm32f1xx_hal.h"
#include "mpu6050.h"
#include <math.h>

typedef struct {
    float angle_x;  // 摆的X轴角度 (rad)
    float angle_y;  // 摆的Y轴角度 (rad)
    float omega_x;  // X轴角速度 (rad/s)
    float omega_y;  // Y轴角速度 (rad/s)
    float accel_x;  // X轴加速度
    float accel_y;  // Y轴加速度
    float accel_z;  // Z轴加速度
} IMUData;

IMUData imu = {0};

// 互补滤波器
#define ALPHA 0.96f

void IMU_Update(float dt) {
    float gx, gy, gz, ax, ay, az;
    MPU6050_Read(&gx, &gy, &gz, &ax, &ay, &az);

    // 加速度计角度
    float acc_angle_x = atan2f(ay, sqrtf(ax*ax + az*az));
    float acc_angle_y = atan2f(-ax, sqrtf(ay*ay + az*az));

    // 互补滤波
    imu.angle_x = ALPHA * (imu.angle_x + gx * dt) +
                  (1 - ALPHA) * acc_angle_x;
    imu.angle_y = ALPHA * (imu.angle_y + gy * dt) +
                  (1 - ALPHA) * acc_angle_y;

    // 角速度 (低通滤波)
    imu.omega_x = 0.9f * imu.omega_x + 0.1f * gx;
    imu.omega_y = 0.9f * imu.omega_y + 0.1f * gy;

    imu.accel_x = ax;
    imu.accel_y = ay;
    imu.accel_z = az;
}

// 平板补偿控制
void Plate_Compensate(float dt) {
    // 核心: 平板需要补偿摆的角度, 保持水平
    // 舵机输出 = -(摆角) + 微分补偿

    // X轴补偿
    float comp_x = -imu.angle_x;
    // 加入角速度前馈阻尼
    comp_x += 0.3f * imu.omega_x;

    // Y轴补偿
    float comp_y = -imu.angle_y;
    comp_y += 0.3f * imu.omega_y;

    // 限幅
    #define MAX_TILT 0.5f  // ~30°
    comp_x = fmaxf(-MAX_TILT, fminf(MAX_TILT, comp_x));
    comp_y = fmaxf(-MAX_TILT, fminf(MAX_TILT, comp_y));

    // 转换为舵机PWM
    Servo_SetAngle_X(comp_x);
    Servo_SetAngle_Y(comp_y);
}
```

### 4.2 视觉辅助检测

```python
import cv2
import numpy as np
import math

class PendulumVisualDetector:
    """摆和平板视觉检测器"""

    def __init__(self):
        # 用于检测平板上的物体位置
        self.lower_color = np.array([0, 100, 100])
        self.upper_color = np.array([10, 255, 255])

    def detect_plate_angle(self, frame):
        """
        检测平板实际倾斜角度
        通过平板上的标记点计算
        """
        # 检测平板边缘线
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        edges = cv2.Canny(gray, 50, 150)

        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50,
                                minLineLength=50, maxLineGap=5)

        if lines is not None:
            # 分析水平线角度
            angles = []
            for line in lines:
                x1, y1, x2, y2 = line[0]
                angle = math.degrees(math.atan2(y2-y1, x2-x1))
                if abs(angle) < 30:  # 近似水平的线
                    angles.append(angle)

            if angles:
                return np.median(angles)
        return None

    def detect_object_position(self, frame):
        """
        检测平板上物体(硬币)的位置
        返回: (x, y) 像素坐标, 用于判断是否滑动
        """
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.lower_color, self.upper_color)

        contours, _ = cv2.findContours(
            mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if contours:
            largest = max(contours, key=cv2.contourArea)
            if cv2.contourArea(largest) > 50:
                M = cv2.moments(largest)
                if M['m00'] > 0:
                    cx = int(M['m10'] / M['m00'])
                    cy = int(M['m01'] / M['m00'])
                    return (cx, cy)
        return None

    def detect_pendulum_rod(self, frame):
        """
        检测摆杆角度 (视觉辅助)
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        edges = cv2.Canny(gray, 30, 100)

        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 80,
                                minLineLength=100, maxLineGap=10)
        if lines is not None:
            # 选择最长的线段作为摆杆
            longest = max(lines, key=lambda l:
                         math.hypot(l[0][2]-l[0][0], l[0][3]-l[0][1]))
            x1, y1, x2, y2 = longest[0]
            angle = math.degrees(math.atan2(x2-x1, -(y2-y1)))
            return angle, (x1, y1, x2, y2)

        return None, None
```

### 4.3 主程序

```python
import cv2
import time
import serial
import struct
import math
from visual_detector import PendulumVisualDetector

class PendulumPlateControlSystem:
    def __init__(self):
        self.cap = cv2.VideoCapture(0)
        self.detector = PendulumVisualDetector()
        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)

        # IMU数据 (从STM32读取)
        self.pendulum_angle_x = 0.0
        self.pendulum_angle_y = 0.0
        self.pendulum_omega_x = 0.0
        self.pendulum_omega_y = 0.0

        # 控制模式
        self.mode = 'COMPENSATE'  # COMPENSATE / MANUAL / TRACK
        self.manual_angle_x = 0.0
        self.manual_angle_y = 0.0

        # 物体位置记录 (检测滑动)
        self.object_positions = []

    def read_imu_data(self):
        """从STM32读取IMU数据"""
        if self.ser.in_waiting >= 20:
            data = self.ser.read(20)
            if data[0] == 0xBB and data[19] == 0x55:
                self.pendulum_angle_x, self.pendulum_angle_y, \
                self.pendulum_omega_x, self.pendulum_omega_y = \
                    struct.unpack('<ffff', data[1:17])

    def send_plate_cmd(self, angle_x, angle_y):
        """发送平板角度指令"""
        buf = bytearray(10)
        buf[0] = 0xAA
        struct.pack_into('<f', buf, 1, angle_x)
        struct.pack_into('<f', buf, 5, angle_y)
        buf[9] = 0x55
        self.ser.write(buf)

    def run(self):
        print("=== 自由摆平板控制系统 ===")
        print("指令: c=补偿模式, m=手动模式, q=退出")

        last_time = time.time()

        while True:
            ret, frame = self.cap.read()
            if not ret:
                continue

            dt = time.time() - last_time
            last_time = time.time()

            # 读取IMU
            self.read_imu_data()

            # 视觉检测
            plate_angle = self.detector.detect_plate_angle(frame)
            object_pos = self.detector.detect_object_position(frame)

            if object_pos:
                self.object_positions.append(
                    (time.time(), object_pos[0], object_pos[1]))
                if len(self.object_positions) > 100:
                    self.object_positions.pop(0)

            # 控制
            if self.mode == 'COMPENSATE':
                # 补偿模式: 平板反向补偿摆角
                # 带角速度前馈的PD控制
                Kp = 1.0
                Kd = 0.3

                cmd_x = -Kp * self.pendulum_angle_x + \
                         Kd * self.pendulum_omega_x
                cmd_y = -Kp * self.pendulum_angle_y + \
                         Kd * self.pendulum_omega_y

                self.send_plate_cmd(cmd_x, cmd_y)

            elif self.mode == 'MANUAL':
                self.send_plate_cmd(
                    self.manual_angle_x, self.manual_angle_y)

            # 可视化
            self._draw(frame, plate_angle, object_pos)

            key = cv2.waitKey(1) & 0xFF
            if key == ord('q'):
                break
            elif key == ord('c'):
                self.mode = 'COMPENSATE'
            elif key == ord('m'):
                self.mode = 'MANUAL'

        self.send_plate_cmd(0, 0)
        self.cap.release()
        cv2.destroyAllWindows()

    def _draw(self, frame, plate_angle, object_pos):
        """绘制信息"""
        cv2.putText(frame, f"Mode: {self.mode}", (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        cv2.putText(frame,
            f"Pendulum: {math.degrees(self.pendulum_angle_x):.1f} deg",
            (10, 50), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)

        if object_pos:
            cv2.circle(frame, object_pos, 10, (0, 0, 255), 3)

        # 物体轨迹
        if len(self.object_positions) > 1:
            for i in range(1, len(self.object_positions)):
                p1 = self.object_positions[i-1]
                p2 = self.object_positions[i]
                cv2.line(frame, (int(p1[1]), int(p1[2])),
                        (int(p2[1]), int(p2[2])), (0, 255, 255), 1)

        cv2.imshow('Pendulum Plate Control', frame)

if __name__ == '__main__':
    system = PendulumPlateControlSystem()
    system.run()
```

## 五、调试要点

### 5.1 关键参数
| 参数 | 推荐值 | 说明 |
|------|--------|------|
| 互补滤波α | 0.96~0.98 | 越大越信任陀螺仪 |
| 补偿Kp | 0.8~1.2 | 过大会振荡 |
| 补偿Kd | 0.2~0.5 | 角速度前馈阻尼 |
| 舵机响应 | <20ms | 需要高速舵机 |

### 5.2 注意事项
1. **IMU安装位置**: 尽量靠近摆的转轴
2. **舵机扭矩**: 需要足够扭矩驱动平板+物体
3. **摆长影响**: 摆越长, 周期越长, 控制越容易
4. **重心设计**: 平板+舵机重心尽量在摆轴线上
