# 2012年E题 — 激光枪自动射击装置

## 目标检测与追踪

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class TargetDetector:
    """目标靶检测器（NPU加速）"""
    
    def __init__(self, model_path='target_detector.rknn'):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime()
        
        # 卡尔曼滤波器（目标跟踪）
        self.kalman = cv2.KalmanFilter(4, 2)
        self.kalman.measurementMatrix = np.array([[1,0,0,0],[0,1,0,0]], np.float32)
        self.kalman.transitionMatrix = np.array([[1,0,1,0],[0,1,0,1],[0,0,1,0],[0,0,0,1]], np.float32)
        self.kalman.processNoiseCov = np.eye(4, dtype=np.float32) * 0.03
    
    def detect_target(self, frame):
        """检测目标靶心"""
        img = cv2.resize(frame, (640, 640))
        
        # NPU推理
        outputs = self.rknn.inference(inputs=[img])
        
        # 方法1: 基于颜色的靶心检测（辅助）
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 红色靶心检测
        mask1 = cv2.inRange(hsv, (0, 100, 100), (10, 255, 255))
        mask2 = cv2.inRange(hsv, (160, 100, 100), (180, 255, 255))
        mask = mask1 | mask2
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 取最大轮廓
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 500:
                M = cv2.moments(c)
                cx = int(M['m10'] / M['m00'])
                cy = int(M['m01'] / M['m00'])
                
                # 卡尔曼滤波平滑
                self.kalman.predict()
                measurement = np.array([[np.float32(cx)], [np.float32(cy)]])
                self.kalman.correct(measurement)
                
                filtered = self.kalman.statePost
                return (int(filtered[0]), int(filtered[1])), True
        
        return None, False
    
    def get_target_offset(self, target_pos, frame_center):
        """计算目标相对于画面中心的偏移"""
        if target_pos is None:
            return None
        dx = target_pos[0] - frame_center[0]
        dy = target_pos[1] - frame_center[1]
        return (dx, dy)
```

## 云台PID控制

```python
import smbus
import time

class GimbalController:
    """双轴舵机云台控制"""
    
    def __init__(self):
        self.bus = smbus.SMBus(1)
        self.pca_addr = 0x40
        self._init_pca9685()
        
        # 当前角度
        self.yaw = 90    # 水平
        self.pitch = 90  # 俯仰
        
        # PID参数
        self.pid_yaw = PIDController(kp=0.3, ki=0.01, kd=0.1)
        self.pid_pitch = PIDController(kp=0.3, ki=0.01, kd=0.1)
        
        # 像素到角度的映射系数（需标定）
        self.px_to_deg_x = 0.1  # 度/像素
        self.px_to_deg_y = 0.1
    
    def _init_pca9685(self):
        self.bus.write_byte_data(self.pca_addr, 0xFE, 0x1E)
        self.bus.write_byte_data(self.pca_addr, 0x00, 0x20)
    
    def set_angle(self, channel, angle):
        """设置舵机角度"""
        angle = max(0, min(180, angle))
        pulse = int(500 + angle / 180 * 2000)
        tick = int(pulse * 4096 / 20000)
        ch = channel * 4
        self.bus.write_word_data(self.pca_addr, 0x06 + ch, 0)
        self.bus.write_word_data(self.pca_addr, 0x08 + ch, tick)
    
    def track_target(self, offset):
        """PID跟踪目标"""
        if offset is None:
            return False
        
        dx, dy = offset
        
        # PID计算
        adj_yaw = self.pid_yaw.update(dx)
        adj_pitch = self.pid_pitch.update(dy)
        
        # 更新角度
        self.yaw = max(0, min(180, self.yaw - adj_yaw))
        self.pitch = max(0, min(180, self.pitch - adj_pitch))
        
        self.set_angle(0, self.yaw)
        self.set_angle(1, self.pitch)
        
        # 判断是否瞄准（偏移小于阈值）
        return abs(dx) < 10 and abs(dy) < 10


class PIDController:
    """PID控制器"""
    
    def __init__(self, kp, ki, kd):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.prev_error = 0
        self.integral = 0
    
    def update(self, error):
        self.integral += error
        self.integral = max(-1000, min(1000, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
    
    def reset(self):
        self.prev_error = 0
        self.integral = 0
```

## 激光控制

```python
import OPi.GPIO as GPIO

class LaserController:
    """激光模组控制"""
    
    LASER_PIN = 11  # GPIO引脚
    
    def __init__(self):
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.LASER_PIN, GPIO.OUT)
        self.is_on = False
    
    def fire(self, duration=0.5):
        """发射激光"""
        GPIO.output(self.LASER_PIN, GPIO.HIGH)
        self.is_on = True
        time.sleep(duration)
        GPIO.output(self.LASER_PIN, GPIO.LOW)
        self.is_on = False
    
    def continuous_mode(self, enable):
        """持续照射模式"""
        GPIO.output(self.LASER_PIN, GPIO.HIGH if enable else GPIO.LOW)
        self.is_on = enable
```

## 主控程序

```python
import cv2
import time

class LaserShootingSystem:
    """激光射击系统主控"""
    
    def __init__(self):
        self.camera = cv2.VideoCapture(0)
        self.camera.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        self.detector = TargetDetector()
        self.gimbal = GimbalController()
        self.laser = LaserController()
        
        self.frame_center = (320, 240)
        self.is_aimed = False
        self.aim_count = 0
    
    def run(self):
        """主循环"""
        print("激光射击系统启动...")
        
        while True:
            ret, frame = self.camera.read()
            if not ret:
                continue
            
            # 检测目标
            target_pos, detected = self.detector.detect_target(frame)
            
            if detected:
                # 计算偏移
                offset = self.detector.get_target_offset(target_pos, self.frame_center)
                
                # 云台跟踪
                self.is_aimed = self.gimbal.track_target(offset)
                
                # 连续3帧瞄准才发射（防误触发）
                if self.is_aimed:
                    self.aim_count += 1
                    if self.aim_count >= 3:
                        print("瞄准目标，发射激光！")
                        self.laser.fire(duration=0.5)
                        self.aim_count = 0
                else:
                    self.aim_count = 0
                
                # 画框
                cv2.circle(frame, target_pos, 5, (0, 0, 255), -1)
                cv2.putText(frame, f"Aimed: {self.is_aimed}", (10, 30),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
            
            cv2.imshow('Laser Shooting', frame)
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        self.cleanup()
    
    def cleanup(self):
        self.laser.continuous_mode(False)
        self.camera.release()
        cv2.destroyAllWindows()

if __name__ == '__main__':
    system = LaserShootingSystem()
    system.run()
```

## 评分要点与优化策略

| 评分项 | 策略 |
|--------|------|
| 瞄准精度 | 高帧率检测+PID调参+卡尔曼滤波平滑 |
| 响应速度 | NPU加速推理，≥30fps检测 |
| 射击准确率 | 多帧确认+预测补偿 |
| 多目标切换 | 目标优先级排序，最近/最大目标优先 |

## 调试技巧

1. **PID调参**：先P后D最后I，从小到大逐步调整
2. **标定**：精确标定像素-角度映射，消除镜头畸变
3. **延迟补偿**：检测到瞄准有延迟，可加入速度前馈
4. **安全措施**：激光功率限制，加入急停按钮
