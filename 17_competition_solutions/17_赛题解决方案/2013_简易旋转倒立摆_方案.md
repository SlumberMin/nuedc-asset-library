# 2013年电赛 - 简易旋转倒立摆系统 - Orange Pi 5解决方案

## 题目要求

设计制作一个简易旋转倒立摆控制系统，实现：
1. 摆杆从自然下垂状态摆起至倒立状态
2. 倒立状态下保持稳定（抗扰动）
3. 旋转臂带动摆杆旋转一定角度后恢复倒立

## 系统架构

```
┌──────────────┐    CSI/USB     ┌──────────────┐    UART     ┌──────────────┐
│  摄像头俯视   │──────────────→│  Orange Pi 5 │────────────→│  STM32主控   │
│  检测摆杆角度  │              │  RK3588S     │             │  电机驱动    │
└──────────────┘              │  视觉+LQR    │             └──────┬───────┘
                               └──────────────┘                    │
                                                              ┌────┴────┐
                                                              │ 直流电机 │
                                                              │ 旋转臂   │
                                                              └─────────┘
```

## 核心设计

### 1. 视觉检测摆角

**方案：OpenCV单目视觉俯视检测**

摄像头从侧面拍摄摆杆，通过颜色标记点检测摆杆角度：

```python
import cv2
import numpy as np
import time
import struct
import serial

class RotaryInvertedPendulumVision:
    """旋转倒立摆视觉检测"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 60)
        
        # 旋转轴心点（需标定）
        self.pivot_point = (320, 240)
        
        # 摆杆末端颜色标记（红色）
        self.hsv_lower = np.array([0, 120, 100])
        self.hsv_upper = np.array([10, 255, 255])
        
        # 摆角历史（用于角速度估计）
        self.angle_history = []
        self.time_history = []
        self.max_history = 10
        
    def detect_pendulum_angle(self):
        """检测摆杆角度（相对竖直方向）"""
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 检测摆杆末端红色标记
        mask1 = cv2.inRange(hsv, self.hsv_lower, self.hsv_upper)
        mask2 = cv2.inRange(hsv, np.array([170, 120, 100]), np.array([180, 255, 255]))
        mask = cv2.bitwise_or(mask1, mask2)
        
        # 形态学处理
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找最大轮廓
            largest = max(contours, key=cv2.contourArea)
            M = cv2.moments(largest)
            if M["m00"] > 100:
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                
                # 计算摆角（相对于竖直向下方向）
                dx = cx - self.pivot_point[0]
                dy = cy - self.pivot_point[1]
                angle_rad = np.arctan2(dx, dy)  # 摆杆偏离竖直方向的角度
                angle_deg = np.degrees(angle_rad)
                
                # 记录历史
                now = time.time()
                self.angle_history.append(angle_rad)
                self.time_history.append(now)
                if len(self.angle_history) > self.max_history:
                    self.angle_history.pop(0)
                    self.time_history.pop(0)
                
                # 可视化
                cv2.circle(frame, (cx, cy), 8, (0, 255, 0), -1)
                cv2.circle(frame, self.pivot_point, 5, (255, 0, 0), -1)
                cv2.line(frame, self.pivot_point, (cx, cy), (0, 255, 0), 2)
                cv2.putText(frame, f"Angle: {angle_deg:.1f} deg", (10, 30),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)
                
                return angle_rad
        
        return None
    
    def estimate_angular_velocity(self):
        """估计角速度（数值微分）"""
        if len(self.angle_history) < 2:
            return 0.0
        
        n = len(self.angle_history)
        dt = self.time_history[-1] - self.time_history[-2]
        if dt < 1e-6:
            return 0.0
        
        # 二阶差分
        if n >= 3:
            d_angle = (3*self.angle_history[-1] - 4*self.angle_history[-2] + self.angle_history[-3])
            d_time = self.time_history[-1] - self.time_history[-3]
            return d_angle / d_time if d_time > 1e-6 else 0.0
        else:
            return (self.angle_history[-1] - self.angle_history[-2]) / dt
    
    def estimate_arm_angle(self):
        """估计旋转臂角度（通过旋转臂末端标记，需额外标记点）"""
        # 简化：可从STM32编码器获取
        return 0.0
    
    def release(self):
        self.cap.release()
```

### 2. LQR控制器设计

```python
import numpy as np
from scipy import linalg

class LQRRotaryInvertedPendulum:
    """
    LQR控制器 - 旋转倒立摆
    
    状态向量: x = [theta_arm, theta_pend, omega_arm, omega_pend]^T
    - theta_arm: 旋转臂角度
    - theta_pend: 摆杆角度（竖直向上为0）
    - omega_arm: 旋转臂角速度
    - omega_pend: 摆杆角速度
    """
    
    def __init__(self):
        # 系统参数（需根据实际系统辨识）
        self.M_arm = 0.15       # 旋转臂质量(kg)
        self.L_arm = 0.20       # 旋转臂长度(m)
        self.m_pend = 0.05      # 摆杆质量(kg)
        self.L_pend = 0.30      # 摆杆长度(m)
        self.g = 9.81
        self.J_arm = self.M_arm * self.L_arm**2 / 3
        self.J_pend = self.m_pend * self.L_pend**2 / 3
        
        # 线性化（在倒立平衡点附近）
        # A矩阵（线性化后的状态矩阵）
        self.A = np.array([
            [0, 0, 1, 0],
            [0, 0, 0, 1],
            [0, -self.m_pend*self.g*self.L_pend/self.J_arm, 0, 0],
            [0, (self.M_arm+self.m_pend)*self.g*self.L_pend/(self.J_pend), 0, 0]
        ])
        
        # B矩阵
        self.B = np.array([
            [0],
            [0],
            [1/self.J_arm],
            [0]
        ])
        
        # LQR权重矩阵
        self.Q = np.diag([100, 1000, 10, 50])  # 摆杆角度权重最大
        self.R = np.array([[1]])                  # 控制量权重
        
        # 计算LQR增益
        self.K = self._compute_lqr_gain()
        
        # 控制参数
        self.u_max = 5.0  # 最大控制量
        self.arm_angle_target = 0.0  # 目标臂角度
        
    def _compute_lqr_gain(self):
        """求解代数Riccati方程"""
        try:
            P = linalg.solve_continuous_are(self.A, self.B, self.Q, self.R)
            K = np.linalg.inv(self.R) @ self.B.T @ P
            return K
        except Exception as e:
            print(f"LQR求解失败: {e}")
            return np.array([[50, 200, 10, 30]])
    
    def compute_control(self, state):
        """
        计算LQR控制输出
        state: [theta_arm, theta_pend, omega_arm, omega_pend]
        """
        # 摆杆角度修正（竖直向上为0）
        state[1] = self._normalize_angle(state[1])
        
        # 目标状态
        x_ref = np.array([self.arm_angle_target, 0, 0, 0])
        
        # 状态误差
        x_err = state - x_ref
        
        # LQR控制律 u = -K * x_err
        u = -self.K @ x_err
        
        # 限幅
        u = np.clip(u, -self.u_max, self.u_max)
        
        return float(u[0])
    
    def _normalize_angle(self, angle):
        """角度归一化到[-pi, pi]"""
        while angle > np.pi:
            angle -= 2 * np.pi
        while angle < -np.pi:
            angle += 2 * np.pi
        return angle


class EnergySwingUp:
    """
    能量摆起控制器
    将摆杆从自然下垂摆起到倒立位置
    """
    
    def __init__(self, pendulum_length=0.30, pendulum_mass=0.05):
        self.L = pendulum_length
        self.m = pendulum_mass
        self.g = 9.81
        
        # 目标能量（倒立位置能量）
        self.E_target = self.m * self.g * self.L
        
        # 能量增益
        self.k_energy = 2.0
        
        # 摆起状态判断阈值
        self.angle_threshold = np.radians(15)  # 15度以内切换到LQR
    
    def compute_energy(self, theta, omega):
        """计算摆杆能量（以最低点为参考）"""
        # 动能
        KE = 0.5 * self.m * (self.L * omega)**2
        # 势能（以最低点为参考）
        PE = self.m * self.g * self.L * (1 - np.cos(theta))
        return KE + PE
    
    def compute_control(self, theta, omega):
        """
        能量摆起控制律
        theta: 摆杆角度（下垂为0，倒立为pi）
        omega: 摆杆角速度
        """
        E = self.compute_energy(theta, omega)
        E_error = E - self.E_target
        
        # 能量泵浦控制律
        # u = k * E_error * sign(omega * cos(theta))
        u = self.k_energy * E_error * np.sign(omega * np.cos(theta))
        
        return u
    
    def should_switch_to_lqr(self, theta):
        """判断是否切换到LQR控制"""
        # 摆杆接近倒立位置
        theta_from_vertical = abs(np.pi - abs(theta))
        return theta_from_vertical < self.angle_threshold


class RotaryPendulumController:
    """旋转倒立摆主控制器（状态机）"""
    
    STATE_SWING_UP = 0
    STATE_BALANCING = 1
    STATE_FALLING = 2
    
    def __init__(self):
        self.lqr = LQRRotaryInvertedPendulum()
        self.swing_up = EnergySwingUp()
        self.state = self.STATE_SWING_UP
        
        # UART通信
        self.ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
        
        # 旋转臂角度（从编码器读取）
        self.arm_angle = 0.0
        self.arm_velocity = 0.0
        
    def update(self, pend_angle, pend_velocity, arm_angle=0, arm_velocity=0):
        """
        主控制循环
        返回电机控制量
        """
        self.arm_angle = arm_angle
        self.arm_velocity = arm_velocity
        
        if self.state == self.STATE_SWING_UP:
            u = self.swing_up.compute_control(pend_angle, pend_velocity)
            
            if self.swing_up.should_switch_to_lqr(pend_angle):
                self.state = self.STATE_BALANCING
                print("切换到LQR平衡控制")
                
        elif self.state == self.STATE_BALANCING:
            state = np.array([arm_angle, pend_angle, arm_velocity, pend_velocity])
            u = self.lqr.compute_control(state)
            
            # 检测是否倒下
            if abs(np.pi - abs(pend_angle)) > np.radians(45):
                self.state = self.STATE_FALLING
                u = 0
                print("摆杆倒下，停止控制")
        else:
            u = 0
            # 等待复位
            if abs(pend_angle) < np.radians(30):
                self.state = self.STATE_SWING_UP
                print("复位，重新摆起")
        
        return u
    
    def send_to_stm32(self, pwm_value):
        """发送控制量到STM32"""
        pwm_int = int(np.clip(pwm_value * 100, -500, 500))
        data = struct.pack('<hh', pwm_int, 0)
        self.ser.write(b'\xAA' + data + b'\x55')


def main():
    vision = RotaryInvertedPendulumVision(camera_id=0)
    controller = RotaryPendulumController()
    
    print("旋转倒立摆控制系统启动")
    print("按下 'q' 退出")
    
    try:
        while True:
            # 视觉检测
            pend_angle = vision.detect_pendulum_angle()
            pend_velocity = vision.estimate_angular_velocity()
            
            if pend_angle is not None:
                # 从STM32读取编码器数据（旋转臂角度）
                # arm_angle, arm_velocity = read_encoder(controller.ser)
                arm_angle = 0.0
                arm_velocity = 0.0
                
                # 控制计算
                u = controller.update(pend_angle, pend_velocity, arm_angle, arm_velocity)
                
                # 发送到STM32
                controller.send_to_stm32(u)
                
                print(f"状态: {controller.state}, 摆角: {np.degrees(pend_angle):.1f}°, "
                      f"控制量: {u:.3f}")
            
            cv2.waitKey(1)
            
    except KeyboardInterrupt:
        pass
    finally:
        controller.send_to_stm32(0)
        vision.release()
        controller.ser.close()
        print("系统已停止")
```

// PWM输出控制电机
void Motor_SetPWM(int16_t pwm) {
    if (pwm > 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_SET);   // 正转
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, pwm);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_SET);   // 反转
        __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, -pwm);
    }
}

## 调试要点

1. **摄像头标定**：准确标定旋转轴心点位置
2. **LQR参数调优**：Q矩阵权重影响响应速度和稳定性
3. **摆起增益调节**：k_energy过大会导致摆过头
4. **切换阈值**：角度阈值决定何时从摆起切换到LQR
5. **采样率**：视觉检测需≥30fps保证控制实时性
