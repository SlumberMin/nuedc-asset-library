### 2.1 еүҚи§ҶзӣёжңәеҫӘиҝ№пјҲжӣҝд»ЈзәўеӨ–дј ж„ҹеҷЁйҳөеҲ—пјү
```python
import cv2
import numpy as np

class LineTracker:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 90)
        
        # ROIеҢәеҹҹпјҲеҸӘеӨ„зҗҶеӣҫеғҸдёӢеҚҠйғЁеҲҶпјү
        self.roi_top_ratio = 0.4  # д»Һ40%й«ҳеәҰејҖе§ӢеҸ–
        
        # й»‘зәҝжЈҖжөӢйҳҲеҖј
        self.line_threshold = 60  # зҒ°еәҰйҳҲеҖј
        
        # з«ҷзӮ№ж Үи®°йўңиүІпјҲеҒҮи®ҫз«ҷзӮ№жңүеҪ©иүІж Үи®°пјү
        self.station_colors = {
            'A': {'lower': np.array([0, 100, 100]), 'upper': np.array([10, 255, 255])},
            'B': {'lower': np.array([100, 100, 100]), 'upper': np.array([130, 255, 255])},
            'C': {'lower': np.array([35, 100, 100]), 'upper': np.array([85, 255, 255])},
            'D': {'lower': np.array([140, 100, 100]), 'upper': np.array([170, 255, 255])},
        }
    
    def detect_line_offset(self, frame):
        """
        жЈҖжөӢй»‘зәҝзӣёеҜ№дәҺз”»йқўдёӯеҝғзҡ„еҒҸз§»йҮҸ
        иҝ”еӣһ: offset (еғҸзҙ ), -дёәе·ҰеҒҸ, +дёәеҸіеҒҸ, None=дёўзәҝ
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        h, w = gray.shape
        
        # еҸ–ROIеҢәеҹҹпјҲйқ иҝ‘иҪҰеӨҙзҡ„еҢәеҹҹпјҢжқғйҮҚжӣҙеӨ§пјү
        roi_y = int(h * self.roi_top_ratio)
        roi = gray[roi_y:, :]
        
        # еӨҡиЎҢжү«жҸҸпјҢеҠ жқғе№іеқҮ
        offsets = []
        weights = []
        num_scans = 5
        
        for i in range(num_scans):
            y = int(roi.shape[0] * (0.3 + 0.15 * i))  # д»Һиҝ‘еҲ°иҝң
            row = roi[y, :]
            
            # дәҢеҖјеҢ–
            _, binary = cv2.threshold(row, self.line_threshold, 255, cv2.THRESH_BINARY_INV)
            
            # жүҫй»‘зәҝдҪҚзҪ®
            indices = np.where(binary > 0)[0]
            if len(indices) < 3:
                continue
            
            # еҸ–дёӯй—ҙеҢәеҹҹдҪңдёәзәҝдҪҚзҪ®
            line_x = np.mean(indices)
            offset = line_x - w / 2
            weight = 1.0 / (1 + i * 0.5)  # иҝ‘еӨ„жқғйҮҚжӣҙеӨ§
            
            offsets.append(offset)
            weights.append(weight)
        
        if not offsets:
            return None
        
        # еҠ жқғе№іеқҮеҒҸз§»
        weighted_offset = np.average(offsets, weights=weights)
        return weighted_offset
    
    def detect_station(self, frame):
        """жЈҖжөӢжҳҜеҗҰеҲ°иҫҫз«ҷзӮ№"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        h, w = frame.shape[:2]
        
        # жЈҖжҹҘз”»йқўдёӢж–№1/3еҢәеҹҹжҳҜеҗҰжңүз«ҷзӮ№ж Үи®°
        roi = hsv[int(h*0.6):, :]
        
        for station_id, color_range in self.station_colors.items():
            mask = cv2.inRange(roi, color_range['lower'], color_range['upper'])
            pixel_count = cv2.countNonZero(mask)
            
            if pixel_count > 500:  # йҳҲеҖј
                # жЈҖжҹҘж Үи®°еӨ§е°ҸеҲӨж–ӯи·қзҰ»
                contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
                if contours:
                    c = max(contours, key=cv2.contourArea)
                    area = cv2.contourArea(c)
                    if area > 2000:  # еӨҹеӨ§иҜҙжҳҺи¶іеӨҹиҝ‘
                        return station_id
        
        return None
    
    def detect_intersection(self, frame):
        """
        жЈҖжөӢеҚҒеӯ—и·ҜеҸЈ/еҲҶеІ”еҸЈ
        иҝ”еӣһ: 'cross', 'left_turn', 'right_turn', None
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        h, w = gray.shape
        
        roi = gray[int(h*0.4):int(h*0.7), :]
        _, binary = cv2.threshold(roi, self.line_threshold, 255, cv2.THRESH_BINARY_INV)
        
        # жЈҖжҹҘе·ҰеҸідёӨдҫ§жҳҜеҗҰжңүй»‘зәҝ
        left_region = binary[:, :int(w*0.3)]
        right_region = binary[:, int(w*0.7):]
        center_region = binary[:, int(w*0.3):int(w*0.7)]
        
        left_count = cv2.countNonZero(left_region)
        right_count = cv2.countNonZero(right_region)
        center_count = cv2.countNonZero(center_region)
        
        threshold = roi.shape[0] * 20
        
        has_left = left_count > threshold
        has_right = right_count > threshold
        has_center = center_count > threshold
        
        if has_left and has_right and has_center:
            return 'cross'
        elif has_left and has_center:
            return 'left_turn'
        elif has_right and has_center:
            return 'right_turn'
        
        return None
```

### 3.1 и§Ҷи§үPIDеҫӘиҝ№
```python
class LineFollowPID:
    def __init__(self):
        self.Kp = 0.5
        self.Ki = 0.001
        self.Kd = 0.3
        
        self.integral = 0
        self.prev_error = 0
        self.integral_limit = 500
        
        # еҹәзЎҖйҖҹеәҰ
        self.base_speed = 60  # PWMеҚ з©әжҜ”%
        
        # ејҜйҒ“еҮҸйҖҹ
        self.turn_speed = 35
    
    def compute(self, offset, dt):
        """
        offset: еғҸзҙ еҒҸз§»йҮҸ
        иҝ”еӣһ: (left_speed, right_speed)
        """
        error = offset
        
        # PID
        self.integral += error * dt
        self.integral = np.clip(self.integral, -self.integral_limit, self.integral_limit)
        derivative = (error - self.prev_error) / dt if dt > 0 else 0
        self.prev_error = error
        
        correction = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        
        # ејҜйҒ“жЈҖжөӢпјҲеҒҸз§»еӨ§еҲҷеҮҸйҖҹпјү
        abs_error = abs(error)
        if abs_error > 100:
            base = self.turn_speed
        else:
            base = self.base_speed
        
        # е·®йҖҹжҺ§еҲ¶
        left_speed = base + correction
        right_speed = base - correction
        
        # йҷҗе№…
        left_speed = np.clip(left_speed, 0, 100)
        right_speed = np.clip(right_speed, 0, 100)
        
        return left_speed, right_speed
```

class PathPlanner:
    def __init__(self):
        # з«ҷзӮ№еәҸеҲ—
        self.route = ['A', 'B', 'C', 'D', 'A']
        self.current_idx = 0
        self.state = 'following'  # 'following', 'approaching', 'stopped', 'turning'
        
        # и·ҜеҸЈиҪ¬ејҜжҢҮд»ӨпјҲж №жҚ®ең°еӣҫпјү
        self.intersection_rules = {
            'A': {'to_B': 'straight', 'to_D': 'left'},
            'B': {'to_C': 'right', 'to_A': 'straight'},
            'C': {'to_D': 'straight', 'to_B': 'left'},
            'D': {'to_A': 'right', 'to_C': 'straight'},
        }
    
    def get_current_target(self):
        return self.route[self.current_idx]
    
    def get_next_target(self):
        if self.current_idx + 1 < len(self.route):
            return self.route[self.current_idx + 1]
        return None
    
    def on_station_reached(self, station_id):
        """еҲ°иҫҫз«ҷзӮ№"""
        if station_id == self.get_current_target():
            self.state = 'stopped'
            # еҒңйқ 3з§’
            return 'stop'
        return None
    
    def on_departure(self):
        """зҰ»з«ҷ"""
        self.current_idx = (self.current_idx + 1) % len(self.route)
        self.state = 'following'
    
    def get_intersection_action(self, intersection_type):
        """еҲ°иҫҫи·ҜеҸЈж—¶зҡ„еҶізӯ–"""
        current = self.get_current_target()
        next_target = self.get_next_target()
        
        if next_target is None:
            return 'stop'
        
        rule = self.intersection_rules.get(current, {}).get(f'to_{next_target}', 'straight')
        
        if intersection_type == 'cross':
            return rule
        elif intersection_type == 'left_turn' and rule in ['left', 'straight']:
            return rule
        elif intersection_type == 'right_turn' and rule in ['right', 'straight']:
            return rule
        
        return 'straight'
```

class SpeedController:
    def __init__(self, kp=1.0, ki=0.5):
        self.kp = kp
        self.ki = ki
        self.integral_left = 0
        self.integral_right = 0
    
    def compute(self, target_left, target_right, actual_left, actual_right, dt):
        err_l = target_left - actual_left
        err_r = target_right - actual_right
        
        self.integral_left += err_l * dt
        self.integral_right += err_r * dt
        
        pwm_l = self.kp * err_l + self.ki * self.integral_left
        pwm_r = self.kp * err_r + self.ki * self.integral_right
        
        return np.clip(pwm_l, 0, 100), np.clip(pwm_r, 0, 100)
```
