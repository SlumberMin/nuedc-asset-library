```python
class PathType(Enum):
    STRAIGHT = 0    # зӣҙзәҝ
    LEFT_TURN = 1   # е·ҰејҜ
    RIGHT_TURN = 2  # еҸіејҜ
    CROSS = 3       # еҚҒеӯ—и·ҜеҸЈ
    T_LEFT = 4      # TеһӢе·ҰиҪ¬
    T_RIGHT = 5     # TеһӢеҸіиҪ¬
    FORK = 6        # YеһӢеҲҶеҸү
    DEAD_END = 7    # ж–ӯеӨҙи·Ҝ
    GAP = 8         # ж–ӯзәҝ

class LineDetector:
    def __init__(self):
        self.line_lower = (0, 0, 0)    # й»‘иүІHSVдёӢйҷҗ
        self.line_upper = (180, 255, 60)  # й»‘иүІHSVдёҠйҷҗ
        
    def detect_line(self, frame):
        """жЈҖжөӢеј•еҜјзәҝдҪҚзҪ®е’Ңзұ»еһӢ"""
        h, w = frame.shape[:2]
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # й»‘иүІзәҝи·ҜжҺ©з Ғ
        mask = cv2.inRange(hsv, self.line_lower, self.line_upper)
        
        # еҪўжҖҒеӯҰеӨ„зҗҶ
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # еӨҡиЎҢжү«жҸҸ (7иЎҢ)
        scan_positions = [0.9, 0.8, 0.7, 0.6, 0.5, 0.4, 0.3]
        line_points = []
        
        for ratio in scan_positions:
            row = int(h * ratio)
            scan_line = mask[row, :]
            indices = np.where(scan_line > 0)[0]
            
            if len(indices) > 0:
                cx = int(np.mean(indices))
                width = indices[-1] - indices[0]
                line_points.append({
                    'y': row, 'x': cx,
                    'width': width,
                    'normalized_x': (cx - w/2) / (w/2)
                })
            else:
                line_points.append(None)
        
        # и·Ҝеҫ„зұ»еһӢеҲҶзұ»
        path_type = self._classify_path(mask, line_points, w)
        
        # еҒҸз§»йҮҸ (з”ЁжңҖиҝ‘жңүж•ҲиЎҢ)
        offset = 0
        for pt in line_points:
            if pt is not None:
                offset = pt['normalized_x']
                break
        
        return {'offset': offset, 'path_type': path_type}
    
    def _classify_path(self, mask, line_points, width):
        """еҲҶзұ»и·Ҝеҫ„зұ»еһӢ"""
        valid_points = [p for p in line_points if p is not None]
        
        if len(valid_points) < 2:
            return PathType.GAP
        
        # зәҝе®ҪзӘҒеўһ вҶ’ еҚҒеӯ—и·ҜеҸЈ
        widths = [p['width'] for p in valid_points]
        if widths[0] > np.mean(widths) * 3 and widths[0] > width * 0.5:
            return PathType.CROSS
        
        # еҒҸз§»и¶ӢеҠҝ вҶ’ ејҜйҒ“
        offsets = [p['normalized_x'] for p in valid_points]
        if len(offsets) >= 3:
            trend = offsets[-1] - offsets[0]
            if trend < -0.3:
                return PathType.LEFT_TURN
            elif trend > 0.3:
                return PathType.RIGHT_TURN
        
        return PathType.STRAIGHT
```

## дёүгҖҒе·ЎзәҝPIDжҺ§еҲ¶

```python
class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
        self.integral_limit = 100
    
    def compute(self, error):
        self.integral = max(-self.integral_limit, 
                           min(self.integral_limit, 
                               self.integral + error))
        derivative = error - self.prev_error
        self.prev_error = error
        return (self.kp * error + 
                self.ki * self.integral + 
                self.kd * derivative)

class DifferentialDrive:
    def __init__(self):
        self.pid = PIDController(kp=40, ki=2, kd=15)
        self.base_speed = 50
        
    def follow_line(self, line_offset, path_type):
        """е·ЎзәҝжҺ§еҲ¶"""
        # ж №жҚ®и·Ҝеҫ„зұ»еһӢи°ғж•ҙйҖҹеәҰ
        if path_type == PathType.STRAIGHT:
            speed = self.base_speed
        elif path_type in [PathType.LEFT_TURN, PathType.RIGHT_TURN]:
            speed = self.base_speed * 0.6
        elif path_type == PathType.CROSS:
            speed = self.base_speed * 0.5
        elif path_type == PathType.GAP:
            speed = self.base_speed * 0.4
        else:
            speed = self.base_speed * 0.5
        
        # PIDиҪ¬еҗ‘дҝ®жӯЈ
        correction = self.pid.compute(line_offset)
        
        # е·®йҖҹиҪ¬еҗ‘
        left_speed = speed + correction
        right_speed = speed - correction
        
        self.set_speed(left_speed, right_speed)
```

```c
void HandleTurn(PathType_t type, float offset) {
    static float last_offset = 0;
    static uint16_t gap_count = 0;
    
    switch(type) {
        case PATH_STRAIGHT:
            gap_count = 0;
            // жӯЈеёёPIDе·Ўзәҝ
            correction = PID_Calc(&g_track_pid, 0, offset);
            speed = BASE_SPEED;
            break;
            
        case PATH_LEFT_TURN:
        case PATH_RIGHT_TURN:
            gap_count = 0;
            // еҮҸйҖҹ + еўһеӨ§PIDеўһзӣҠ
            speed = BASE_SPEED * 0.6;
            correction = PID_Calc(&g_track_pid, 0, offset) * 1.5;
            break;
            
        case PATH_CROSS:
            gap_count = 0;
            // дҝқжҢҒзӣҙиЎҢ
            speed = BASE_SPEED * 0.5;
            correction = 0;
            break;
            
        case PATH_GAP:
            gap_count++;
            if (gap_count < MAX_GAP_FRAMES) {
                // жғҜжҖ§зӣҙиЎҢ
                speed = BASE_SPEED * 0.4;
                correction = last_offset * KpжғҜжҖ§;
            } else {
                // й•ҝж—¶й—ҙж–ӯзәҝпјҢеҒңиҪҰ
                speed = 0;
                correction = 0;
            }
            break;
    }
    
    last_offset = offset;
    Motor_SetSpeed(speed + correction, speed - correction);
}
```

## дә”гҖҒдё»зЁӢеәҸ

```python
def main():
    camera = cv2.VideoCapture(0)
    camera.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
    camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
    camera.set(cv2.CAP_PROP_FPS, 30)
    
    line_det = LineDetector()
    drive = DifferentialDrive()
    
    gap_count = 0
    max_gap_frames = 15
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        result = line_det.detect_line(frame)
        offset = result['offset']
        path_type = result['path_type']
        
        # ж–ӯзәҝеӨ„зҗҶ
        if path_type == PathType.GAP:
            gap_count += 1
            if gap_count <= max_gap_frames:
                drive.follow_line(0, PathType.STRAIGHT)
            else:
                drive.set_speed(0, 0)
            continue
        else:
            gap_count = 0
        
        # е·ЎзәҝжҺ§еҲ¶
        drive.follow_line(offset, path_type)
        
        time.sleep(0.02)  # 50Hz
```
