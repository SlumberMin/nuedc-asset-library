# 2013年G题：手写绘图板 - Orange Pi 5解决方案

## 题目概述

设计一个手写绘图板系统，能够检测笔尖在平板上的位置，并将手写轨迹实时显示在屏幕上。支持基本的绘图功能，如画线、擦除、颜色选择等。

---

## 方案选择

### 方案A：电阻式触摸屏（推荐）

**优点：**
- 成本低，易于获取
- 接口简单（4线/5线模拟接口）
- 可用任何物体触摸（手指、笔）

**缺点：**
- 精度相对较低
- 不支持压感

### 方案B：电磁式手写板

**优点：**
- 精度高（可达0.1mm）
- 支持压感
- 支持悬停检测

**缺点：**
- 需要专用电磁笔
- 成本较高

### 方案C：视觉检测方案

**优点：**
- 无需特殊触摸硬件
- 可检测任意笔迹

**缺点：**
- 精度受摄像头分辨率限制
- 需要标定

### 1. 触摸屏驱动

```python
#!/usr/bin/env python3
"""
触摸屏驱动 - XPT2046
"""

import spidev
import gpiod
import time
import threading
from collections import deque

class XPT2046Driver:
    """XPT2046触摸屏控制器驱动"""
    
    def __init__(self, spi_bus=1, spi_device=0, cs_pin=24, irq_pin=25):
        # 初始化SPI
        self.spi = spidev.SpiDev()
        self.spi.open(spi_bus, spi_device)
        self.spi.max_speed_hz = 1000000  # 1MHz
        self.spi.mode = 0
        
        # GPIO配置
        self.chip = gpiod.Chip('1')
        self.cs_line = self.chip.get_line(cs_pin)
        self.irq_line = self.chip.get_line(irq_pin)
        
        cs_config = gpiod.LineSettings(direction=gpiod.Direction.OUTPUT, output_value=1)
        irq_config = gpiod.LineSettings(direction=gpiod.Direction.INPUT)
        
        self.cs_line.request(consumer="xpt2046", config=cs_config)
        self.irq_line.request(consumer="xpt2046", config=irq_config)
        
        # 校准参数
        self.x_min = 200
        self.x_max = 3900
        self.y_min = 200
        self.y_max = 3900
        self.screen_width = 800
        self.screen_height = 480
        
        # 触摸状态
        self.is_touched = False
        self.x = 0
        self.y = 0
        self.pressure = 0
        
        # 回调函数
        self.touch_callback = None
        self.release_callback = None
        
        # 采样线程
        self.running = False
        self.sample_thread = None
    
    def _read_channel(self, channel):
        """读取ADC通道"""
        # XPT2046命令字节
        cmd = 0x80 | (channel << 4) | 0x04  # 启动位 + 通道 + 12位模式
        
        self.cs_line.set_value(0)
        result = self.spi.xfer2([cmd, 0x00, 0x00])
        self.cs_line.set_value(1)
        
        # 组合12位结果
        value = ((result[1] << 8) | result[2]) >> 3
        return value
    
    def read_touch(self):
        """读取触摸坐标"""
        # 检查IRQ (低电平表示触摸)
        if self.irq_line.get_value() == 1:
            return None
        
        # 读取X, Y坐标 (多次采样取平均)
        samples = 5
        x_sum = 0
        y_sum = 0
        
        for _ in range(samples):
            x_sum += self._read_channel(1)  # 通道1 = X
            y_sum += self._read_channel(5)  # 通道5 = Y
        
        x_raw = x_sum // samples
        y_raw = y_sum // samples
        
        # 转换为屏幕坐标
        x_screen = int((x_raw - self.x_min) * self.screen_width / (self.x_max - self.x_min))
        y_screen = int((y_raw - self.y_min) * self.screen_height / (self.y_max - self.y_min))
        
        # 限幅
        x_screen = max(0, min(self.screen_width - 1, x_screen))
        y_screen = max(0, min(self.screen_height - 1, y_screen))
        
        return (x_screen, y_screen)
    
    def calibrate(self, points):
        """
        校准触摸屏
        points: [(screen_x, screen_y, raw_x, raw_y), ...]
        """
        if len(points) < 3:
            return False
        
        # 最小二乘法计算校准矩阵
        # 简化版本：使用3点校准
        src = np.array([[p[2], p[3]] for p in points], dtype=np.float32)
        dst = np.array([[p[0], p[1]] for p in points], dtype=np.float32)
        
        # 计算仿射变换矩阵
        self.transform_matrix = cv2.getAffineTransform(src[:3], dst[:3])
        
        return True
    
    def _sample_loop(self):
        """采样循环"""
        while self.running:
            touch = self.read_touch()
            
            if touch is not None:
                if not self.is_touched:
                    self.is_touched = True
                    if self.touch_callback:
                        self.touch_callback(touch[0], touch[1])
                
                self.x = touch[0]
                self.y = touch[1]
            else:
                if self.is_touched:
                    self.is_touched = False
                    if self.release_callback:
                        self.release_callback()
            
            time.sleep(0.005)  # 200Hz采样率
    
    def start(self):
        """启动采样"""
        self.running = True
        self.sample_thread = threading.Thread(target=self._sample_loop)
        self.sample_thread.daemon = True
        self.sample_thread.start()
    
    def stop(self):
        """停止采样"""
        self.running = False
        if self.sample_thread:
            self.sample_thread.join()
```

### 2. 绘图引擎

```python
#!/usr/bin/env python3
"""
绘图引擎 - 处理绘图逻辑
"""

import numpy as np
from collections import deque

class DrawingEngine:
    """绘图引擎"""
    
    def __init__(self, width=800, height=480):
        self.width = width
        self.height = height
        
        # 绘图层
        self.canvas = np.zeros((height, width, 3), dtype=np.uint8)
        self.canvas.fill(255)  # 白色背景
        
        # 当前状态
        self.current_color = (0, 0, 0)  # 黑色
        self.current_thickness = 2
        self.current_tool = 'pen'  # pen, eraser, line, rect, circle
        
        # 绘图历史 (用于撤销)
        self.history = deque(maxlen=50)
        self.save_state()
        
        # 当前绘制的点
        self.current_stroke = []
        
        # 工具设置
        self.colors = {
            'black': (0, 0, 0),
            'red': (0, 0, 255),
            'green': (0, 255, 0),
            'blue': (255, 0, 0),
            'yellow': (0, 255, 255),
        }
        
        self.thicknesses = [1, 2, 3, 5, 8]
    
    def save_state(self):
        """保存当前画布状态"""
        self.history.append(self.canvas.copy())
    
    def undo(self):
        """撤销"""
        if self.history:
            self.canvas = self.history.pop()
    
    def clear(self):
        """清空画布"""
        self.save_state()
        self.canvas.fill(255)
    
    def set_color(self, color_name):
        """设置颜色"""
        if color_name in self.colors:
            self.current_color = self.colors[color_name]
    
    def set_thickness(self, thickness):
        """设置线宽"""
        self.current_thickness = thickness
    
    def set_tool(self, tool):
        """设置工具"""
        self.current_tool = tool
    
    def start_stroke(self, x, y):
        """开始新的笔画"""
        self.save_state()
        self.current_stroke = [(x, y)]
    
    def continue_stroke(self, x, y):
        """继续笔画"""
        if not self.current_stroke:
            return
        
        self.current_stroke.append((x, y))
        
        if self.current_tool == 'pen':
            # 画线
            if len(self.current_stroke) >= 2:
                pt1 = self.current_stroke[-2]
                pt2 = self.current_stroke[-1]
                cv2.line(self.canvas, pt1, pt2, self.current_color, self.current_thickness)
        
        elif self.current_tool == 'eraser':
            # 擦除
            if len(self.current_stroke) >= 2:
                pt1 = self.current_stroke[-2]
                pt2 = self.current_stroke[-1]
                cv2.line(self.canvas, pt1, pt2, (255, 255, 255), self.current_thickness * 3)
    
    def end_stroke(self):
        """结束笔画"""
        if not self.current_stroke:
            return
        
        if self.current_tool == 'line' and len(self.current_stroke) >= 2:
            pt1 = self.current_stroke[0]
            pt2 = self.current_stroke[-1]
            cv2.line(self.canvas, pt1, pt2, self.current_color, self.current_thickness)
        
        elif self.current_tool == 'rect' and len(self.current_stroke) >= 2:
            pt1 = self.current_stroke[0]
            pt2 = self.current_stroke[-1]
            cv2.rectangle(self.canvas, pt1, pt2, self.current_color, self.current_thickness)
        
        elif self.current_tool == 'circle' and len(self.current_stroke) >= 2:
            pt1 = self.current_stroke[0]
            pt2 = self.current_stroke[-1]
            center = ((pt1[0] + pt2[0]) // 2, (pt1[1] + pt2[1]) // 2)
            radius = int(((pt2[0] - pt1[0])**2 + (pt2[1] - pt1[1])**2)**0.5 / 2)
            cv2.circle(self.canvas, center, radius, self.current_color, self.current_thickness)
        
        self.current_stroke = []
    
    def get_preview(self, x, y):
        """获取预览图 (带当前绘制的临时内容)"""
        preview = self.canvas.copy()
        
        if self.current_stroke:
            if self.current_tool in ['line', 'rect', 'circle']:
                pt1 = self.current_stroke[0]
                pt2 = (x, y)
                
                if self.current_tool == 'line':
                    cv2.line(preview, pt1, pt2, self.current_color, self.current_thickness)
                elif self.current_tool == 'rect':
                    cv2.rectangle(preview, pt1, pt2, self.current_color, self.current_thickness)
                elif self.current_tool == 'circle':
                    center = ((pt1[0] + pt2[0]) // 2, (pt1[1] + pt2[1]) // 2)
                    radius = int(((pt2[0] - pt1[0])**2 + (pt2[1] - pt1[1])**2)**0.5 / 2)
                    cv2.circle(preview, center, radius, self.current_color, self.current_thickness)
        
        return preview
    
    def add_text(self, x, y, text, font_scale=1.0):
        """添加文字"""
        self.save_state()
        cv2.putText(self.canvas, text, (x, y), cv2.FONT_HERSHEY_SIMPLEX, 
                   font_scale, self.current_color, self.current_thickness)
    
    def save_image(self, filename):
        """保存图片"""
        cv2.imwrite(filename, self.canvas)
    
    def load_image(self, filename):
        """加载图片"""
        self.save_state()
        img = cv2.imread(filename)
        if img is not None:
            img = cv2.resize(img, (self.width, self.height))
            self.canvas = img
```

### 3. 用户界面

```python
#!/usr/bin/env python3
"""
用户界面 - 工具栏和绘图区
"""

import cv2
import numpy as np

class DrawingUI:
    """绘图界面"""
    
    def __init__(self, width=800, height=480):
        self.width = width
        self.height = height
        
        # 工具栏区域
        self.toolbar_height = 60
        self.drawing_area_y = self.toolbar_height
        
        # 工具栏布局
        self.buttons = []
        self._create_toolbar()
        
        # 当前选中的按钮
        self.selected_button = None
    
    def _create_toolbar(self):
        """创建工具栏按钮"""
        button_width = 60
        button_height = 40
        padding = 10
        x = padding
        y = 10
        
        # 工具按钮
        tools = [
            ('Pen', 'pen'), ('Eraser', 'eraser'), ('Line', 'line'),
            ('Rect', 'rect'), ('Circle', 'circle'), ('Text', 'text')
        ]
        
        for name, tool in tools:
            self.buttons.append({
                'rect': (x, y, x + button_width, y + button_height),
                'name': name,
                'tool': tool,
                'type': 'tool'
            })
            x += button_width + padding
        
        # 分隔线
        x += 10
        
        # 颜色按钮
        colors = [
            ('Black', (0, 0, 0)), ('Red', (0, 0, 255)),
            ('Green', (0, 255, 0)), ('Blue', (255, 0, 0)),
            ('Yellow', (0, 255, 255))
        ]
        
        for name, color in colors:
            self.buttons.append({
                'rect': (x, y, x + button_width, y + button_height),
                'name': name,
                'color': color,
                'type': 'color'
            })
            x += button_width + padding
        
        # 分隔线
        x += 10
        
        # 线宽按钮
        thicknesses = [1, 2, 3, 5, 8]
        for thickness in thicknesses:
            self.buttons.append({
                'rect': (x, y, x + button_width, y + button_height),
                'name': str(thickness),
                'thickness': thickness,
                'type': 'thickness'
            })
            x += button_width + padding
        
        # 功能按钮
        x += 10
        functions = [('Clear', 'clear'), ('Undo', 'undo'), ('Save', 'save')]
        for name, func in functions:
            self.buttons.append({
                'rect': (x, y, x + button_width, y + button_height),
                'name': name,
                'function': func,
                'type': 'function'
            })
            x += button_width + padding
    
    def draw_toolbar(self, frame):
        """绘制工具栏"""
        # 工具栏背景
        cv2.rectangle(frame, (0, 0), (self.width, self.toolbar_height), 
                     (200, 200, 200), -1)
        
        # 绘制按钮
        for button in self.buttons:
            x1, y1, x2, y2 = button['rect']
            
            # 按钮背景
            if button == self.selected_button:
                color = (150, 150, 255)  # 选中状态
            else:
                color = (220, 220, 220)
            
            cv2.rectangle(frame, (x1, y1), (x2, y2), color, -1)
            cv2.rectangle(frame, (x1, y1), (x2, y2), (100, 100, 100), 1)
            
            # 颜色按钮特殊处理
            if button['type'] == 'color':
                color = button['color']
                cv2.rectangle(frame, (x1+5, y1+5), (x2-5, y2-5), color, -1)
            else:
                # 文字
                text_size = cv2.getTextSize(button['name'], cv2.FONT_HERSHEY_SIMPLEX, 0.4, 1)[0]
                text_x = x1 + (x2 - x1 - text_size[0]) // 2
                text_y = y1 + (y2 - y1 + text_size[1]) // 2
                cv2.putText(frame, button['name'], (text_x, text_y),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.4, (0, 0, 0), 1)
        
        return frame
    
    def check_button_click(self, x, y):
        """检查按钮点击"""
        for button in self.buttons:
            x1, y1, x2, y2 = button['rect']
            if x1 <= x <= x2 and y1 <= y <= y2:
                self.selected_button = button
                return button
        return None
    
    def draw_drawing_area(self, frame, canvas):
        """绘制绘图区"""
        # 将画布复制到帧
        frame[self.drawing_area_y:, :] = canvas[self.drawing_area_y:, :]
        return frame
    
    def show_status(self, frame, tool, color, thickness):
        """显示状态信息"""
        status = f"Tool: {tool} | Color: {color} | Width: {thickness}"
        cv2.putText(frame, status, (10, self.height - 10),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, (100, 100, 100), 1)
        return frame
```

### 4. 主程序

```python
#!/usr/bin/env python3
"""
手写绘图板主程序
"""

import cv2
import numpy as np
from touchscreen_driver import XPT2046Driver
from drawing_engine import DrawingEngine
from drawing_ui import DrawingUI

class HandwritingTablet:
    """手写绘图板系统"""
    
    def __init__(self):
        # 初始化组件
        self.touchscreen = XPT2046Driver()
        self.engine = DrawingEngine(800, 480)
        self.ui = DrawingUI(800, 480)
        
        # 连接触摸事件
        self.touchscreen.touch_callback = self.on_touch
        self.touchscreen.release_callback = self.on_release
        
        # 绘图状态
        self.is_drawing = False
        self.last_point = None
        
        # 显示窗口
        self.window_name = 'Handwriting Tablet'
    
    def on_touch(self, x, y):
        """触摸事件处理"""
        # 检查是否点击了工具栏
        if y < self.ui.toolbar_height:
            button = self.ui.check_button_click(x, y)
            if button:
                self._handle_button(button)
            return
        
        # 绘图区域
        self.is_drawing = True
        self.engine.start_stroke(x, y)
        self.last_point = (x, y)
    
    def on_release(self):
        """释放事件处理"""
        if self.is_drawing:
            self.engine.end_stroke()
            self.is_drawing = False
            self.last_point = None
    
    def _handle_button(self, button):
        """处理按钮点击"""
        if button['type'] == 'tool':
            self.engine.set_tool(button['tool'])
        
        elif button['type'] == 'color':
            # 将BGR转换为名称
            color_map = {
                (0, 0, 0): 'black',
                (0, 0, 255): 'red',
                (0, 255, 0): 'green',
                (255, 0, 0): 'blue',
                (0, 255, 255): 'yellow'
            }
            color_name = color_map.get(button['color'], 'black')
            self.engine.set_color(color_name)
        
        elif button['type'] == 'thickness':
            self.engine.set_thickness(button['thickness'])
        
        elif button['type'] == 'function':
            if button['function'] == 'clear':
                self.engine.clear()
            elif button['function'] == 'undo':
                self.engine.undo()
            elif button['function'] == 'save':
                self.engine.save_image('drawing.png')
                print("图片已保存为 drawing.png")
    
    def run(self):
        """主循环"""
        print("手写绘图板启动...")
        print("按 'q' 退出")
        
        self.touchscreen.start()
        
        try:
            while True:
                # 创建显示帧
                frame = np.zeros((480, 800, 3), dtype=np.uint8)
                
                # 绘制工具栏
                frame = self.ui.draw_toolbar(frame)
                
                # 绘制画布
                canvas = self.engine.canvas.copy()
                
                # 如果正在绘制，显示预览
                if self.is_drawing and self.touchscreen.is_touched:
                    canvas = self.engine.get_preview(self.touchscreen.x, self.touchscreen.y)
                
                # 合并画布到帧
                frame[self.ui.drawing_area_y:, :] = canvas[self.ui.drawing_area_y:, :]
                
                # 显示状态
                frame = self.ui.show_status(frame, 
                                           self.engine.current_tool,
                                           self.engine.current_color,
                                           self.engine.current_thickness)
                
                # 显示
                cv2.imshow(self.window_name, frame)
                
                # 键盘控制
                key = cv2.waitKey(1) & 0xFF
                if key == ord('q'):
                    break
                elif key == ord('c'):
                    self.engine.clear()
                elif key == ord('z'):
                    self.engine.undo()
                elif key == ord('s'):
                    self.engine.save_image('drawing.png')
                elif key == ord('1'):
                    self.engine.set_tool('pen')
```

```python
#!/usr/bin/env python3
"""
视觉检测方案 - 使用摄像头检测笔尖位置
"""

import cv2
import numpy as np

class VisionPenDetector:
    """视觉笔尖检测器"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        # 笔尖检测参数
        # 使用颜色检测 (假设笔尖为深色)
        self.pen_color_lower = np.array([0, 0, 0])
        self.pen_color_upper = np.array([180, 255, 50])
        
        # 或使用ArUco标记
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.aruco_detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        
        # 标定参数
        self.homography = None
        
    def calibrate(self, image_points, board_points):
        """
        标定摄像头到画板的映射
        image_points: 图像中的点
        board_points: 画板上的对应点
        """
        self.homography, _ = cv2.findHomography(
            np.array(image_points), 
            np.array(board_points)
        )
    
    def detect_pen_color(self, frame):
        """使用颜色检测笔尖"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.pen_color_lower, self.pen_color_upper)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找到最大的轮廓
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if area > 50:  # 最小面积
                M = cv2.moments(largest)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    return (cx, cy)
        
        return None
    
    def detect_pen_aruco(self, frame):
        """使用ArUco标记检测笔尖"""
        corners, ids, rejected = self.aruco_detector.detectMarkers(frame)
        
        if ids is not None and len(ids) > 0:
            # 使用第一个标记的中心
            corner = corners[0][0]
            cx = int(np.mean(corner[:, 0]))
            cy = int(np.mean(corner[:, 1]))
            return (cx, cy)
        
        return None
    
    def image_to_board(self, image_point):
        """将图像坐标转换为画板坐标"""
        if self.homography is None:
            return image_point
        
        pt = np.array([image_point[0], image_point[1], 1])
        board_pt = self.homography @ pt
        board_pt /= board_pt[2]
        
        return (int(board_pt[0]), int(board_pt[1]))
    
    def detect(self, frame):
        """检测笔尖位置"""
        # 尝试颜色检测
        point = self.detect_pen_color(frame)
        
        # 尝试ArUco检测
        if point is None:
            point = self.detect_pen_aruco(frame)
        
        if point is not None:
            # 转换到画板坐标
            board_point = self.image_to_board(point)
            return board_point
        
        return None
    
    def release(self):
        """释放资源"""
        self.cap.release()
```

```python
#!/usr/bin/env python3
"""
手写字符识别
"""

import numpy as np
from PIL import Image

class HandwritingRecognizer:
    """手写识别器"""
    
    def __init__(self):
        # 加载预训练模型 (如果有)
        self.model = None
        
    def preprocess_stroke(self, stroke_points):
        """预处理笔画数据"""
        if len(stroke_points) < 2:
            return None
        
        # 归一化坐标
        points = np.array(stroke_points)
        min_x, min_y = points.min(axis=0)
        max_x, max_y = points.max(axis=0)
        
        # 避免除零
        width = max(max_x - min_x, 1)
        height = max(max_y - min_y, 1)
        
        normalized = (points - [min_x, min_y]) / [width, height]
        
        # 缩放到28x28 (MNIST尺寸)
        scaled = (normalized * 27).astype(int)
        
        # 创建图像
        img = np.zeros((28, 28), dtype=np.uint8)
        for i in range(len(scaled) - 1):
            cv2.line(img, tuple(scaled[i]), tuple(scaled[i+1]), 255, 2)
        
        return img
    
    def recognize(self, stroke_points):
        """识别手写字符"""
        img = self.preprocess_stroke(stroke_points)
        
        if img is None:
            return None
        
        # 如果有模型，进行预测
        if self.model is not None:
            # 预处理为模型输入格式
            input_data = img.reshape(1, 28, 28, 1).astype(np.float32) / 255.0
            prediction = self.model.predict(input_data)
            return prediction
        
        return None
```

```python
class DoubleBuffer:
    """双缓冲显示"""
    
    def __init__(self):
        self.front_buffer = np.zeros((480, 800, 3), dtype=np.uint8)
        self.back_buffer = np.zeros((480, 800, 3), dtype=np.uint8)
        self.lock = threading.Lock()
    
    def update_back(self, canvas):
        """更新后缓冲"""
        with self.lock:
            np.copyto(self.back_buffer, canvas)
    
    def swap(self):
        """交换前后缓冲"""
        with self.lock:
            self.front_buffer, self.back_buffer = self.back_buffer, self.front_buffer
    
    def get_front(self):
        """获取前缓冲"""
        return self.front_buffer.copy()
```
