# 2021年电赛 - 送药小车 方案

> 基于 Orange Pi 5 + OpenCV视觉 + 自动寻径 + 双车协同

## 一、赛题分析

设计送药小车系统，能够从药房出发，沿预设路径行驶至指定病房送药，再自动返回。要求自主导航、识别路径标志、双车协同避让。

### 3.2 接口设计

```
Orange Pi 5:
  UART2 (Pin8/10)  → STM32 通信 (115200)
  MIPI CSI          → 摄像头
  USB               → 备用摄像头
  GPIO21 (Pin29)    → 蜂鸣器(到达提示)

STM32F103:
  TIM1 CH1-4        → 4路电机PWM
  TIM2/3            → 编码器输入
  USART1            → Orange Pi通信
  USART2            → ESP32双车通信
  ADC               → 电池电压检测
```

### 4.1 视觉路径识别

```python
#!/usr/bin/env python3
"""送药小车视觉导航"""
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class VisionNavigator:
    def __init__(self):
        # 路径颜色HSV阈值 (黑色引导线)
        self.line_lower = np.array([0, 0, 0])
        self.line_upper = np.array([180, 255, 80])
        # 标志牌颜色阈值
        self.sign_colors = {
            "pharmacy":  {"lower": np.array([0,100,100]), "upper": np.array([10,255,255])},  # 红色-药房
            "ward_1":    {"lower": np.array([100,100,100]), "upper": np.array([130,255,255])}, # 蓝色-病房1
            "ward_2":    {"lower": np.array([35,100,100]), "upper": np.array([85,255,255])},  # 绿色-病房2
            "crossroad": {"lower": np.array([15,100,100]), "upper": np.array([30,255,255])},  # 黄色-路口
        }
        # NPU模型加载(用于数字/标志识别)
        self.rknn = RKNNLite()
        self.rknn.load_rknn("sign_detector.rknn")
        self.rknn.init_runtime()

    def detect_line(self, frame):
        """检测引导线，返回偏移量和角度"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.line_lower, self.line_upper)

        # 透视变换(俯视角)
        h, w = frame.shape[:2]
        src = np.float32([[0,h], [w,h], [w*0.6,h*0.4], [w*0.4,h*0.4]])
        dst = np.float32([[0,h], [w,h], [w,0], [0,0]])
        M = cv2.getPerspectiveTransform(src, dst)
        warped = cv2.warpPerspective(mask, M, (w, h))

        # 扫描底部2/3区域
        roi = warped[int(h*0.3):, :]
        offset = 0
        angle = 0

        # 多行扫描取平均
        rows_to_scan = [int(roi.shape[0] * r) for r in [0.2, 0.4, 0.6, 0.8]]
        offsets = []
        for r in rows_to_scan:
            row = roi[r, :]
            white = np.where(row > 0)[0]
            if len(white) > 10:
                center = int(np.mean(white))
                offsets.append(center - w // 2)

        if offsets:
            offset = int(np.mean(offsets))
            # 用前后行差估算角度
            if len(offsets) >= 2:
                angle = np.degrees(np.arctan2(offsets[-1] - offsets[0], h * 0.6))

        return offset, angle

    def detect_sign(self, frame):
        """识别标志牌"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

        # 基于颜色的快速检测
        for name, thresh in self.sign_colors.items():
            mask = cv2.inRange(hsv, thresh["lower"], thresh["upper"])
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
            for cnt in contours:
                area = cv2.contourArea(cnt)
                if 500 < area < 15000:
                    x, y, w, h = cv2.boundingRect(cnt)
                    aspect = w / h
                    if 0.7 < aspect < 1.4:  # 近似正方形
                        return {"type": name, "bbox": (x,y,w,h), "area": area}

        # NPU精确识别(数字、箭头等)
        roi = frame[50:350, 150:490]  # 预设标志区域
        input_data = cv2.resize(roi, (224, 224))
        input_data = np.expand_dims(input_data, axis=0).astype(np.uint8)
        outputs = self.rknn.inference(inputs=[input_data])
        # 解析输出...
        return None

    def process_frame(self, frame):
        """单帧处理"""
        offset, angle = self.detect_line(frame)
        sign = self.detect_sign(frame)
        return {
            "offset": offset,       # 偏移量(像素), 负=偏左
            "angle": angle,         # 偏航角(度)
            "sign": sign,           # 检测到的标志
            "status": "tracking" if abs(offset) < 100 else "searching"
        }
```

### 4.2 路径规划与状态机

```python
#!/usr/bin/env python3
"""送药小车状态机与路径规划"""
import time
from enum import Enum

class State(Enum):
    IDLE = "idle"
    NAVIGATING = "navigating"
    INTERSECTION = "intersection"
    DELIVERING = "delivering"
    RETURNING = "returning"
    AVOIDING = "avoiding"
    DONE = "done"

class MapGraph:
    """地图拓扑图: 节点=路口/目标, 边=路径"""
    def __init__(self):
        # 节点: (x, y, type)
        self.nodes = {
            "start":    (0, 0, "start"),
            "cross_1":  (2, 0, "crossroad"),
            "pharmacy": (0, 2, "pharmacy"),
            "cross_2":  (2, 2, "crossroad"),
            "ward_1":   (4, 1, "ward"),
            "ward_2":   (4, 3, "ward"),
        }
        # 边: (node_a, node_b, 距离, 方向)
        self.edges = [
            ("start", "cross_1", 2, "right"),
            ("cross_1", "cross_2", 2, "up"),
            ("cross_1", "ward_1", 2, "right"),
            ("cross_2", "ward_2", 2, "right"),
            ("cross_2", "pharmacy", 2, "left"),
        ]

    def shortest_path(self, start, goal):
        """Dijkstra最短路径"""
        import heapq
        dist = {n: float('inf') for n in self.nodes}
        dist[start] = 0
        prev = {}
        pq = [(0, start)]
        while pq:
            d, u = heapq.heappop(pq)
            if u == goal:
                break
            for a, b, w, _ in self.edges:
                v = b if a == u else (a if b == u else None)
                if v and d + w < dist[v]:
                    dist[v] = d + w
                    prev[v] = u
                    heapq.heappush(pq, (dist[v], v))
        # 回溯路径
        path = [goal]
        while path[-1] != start:
            path.append(prev[path[-1]])
        return list(reversed(path))

class NavigationController:
    def __init__(self):
        self.state = State.IDLE
        self.map = MapGraph()
        self.current_path = []
        self.path_idx = 0
        self.current_order = None  # 当前送药任务
        self.tasks = []            # 任务队列

    def set_order(self, ward, medicine):
        """接收送药任务"""
        self.current_order = {"ward": ward, "medicine": medicine}
        start = "pharmacy"  # 从药房出发
        goal = ward         # 目标病房
        self.current_path = self.map.shortest_path(start, goal)
        self.path_idx = 0
        self.state = State.NAVIGATING
        print(f"任务: {medicine} → {ward}, 路径: {self.current_path}")

    def update(self, vision_result, ultrasonic_dist):
        """根据传感器数据更新控制"""
        cmd = {"left_speed": 0, "right_speed": 0, "action": None}

        if self.state == State.NAVIGATING:
            offset = vision_result["offset"]
            angle = vision_result["angle"]

            # PID循迹
            Kp, Ki, Kd = 0.5, 0.01, 0.2
            # 简单P控制
            correction = Kp * offset
            base_speed = 150  # PWM
            cmd["left_speed"] = int(base_speed + correction)
            cmd["right_speed"] = int(base_speed - correction)

            # 检测到路口标志
            if vision_result.get("sign") and vision_result["sign"]["type"] == "crossroad":
                self.state = State.INTERSECTION
                cmd["action"] = "slow_down"

            # 检测到目标
            if vision_result.get("sign") and vision_result["sign"]["type"] == self.current_order["ward"]:
                self.state = State.DELIVERING
                cmd = {"left_speed": 0, "right_speed": 0, "action": "stop"}

            # 超声波避障
            if ultrasonic_dist < 15:
                self.state = State.AVOIDING
                cmd = {"left_speed": -100, "right_speed": -100, "action": "backup"}

        elif self.state == State.INTERSECTION:
            # 路口决策: 根据路径规划转向
            next_node = self.current_path[self.path_idx + 1]
            # 计算转向方向...
            cmd["action"] = "turn_" + self._get_turn_direction()
            self.path_idx += 1
            if self.path_idx >= len(self.current_path) - 1:
                self.state = State.NAVIGATING

        elif self.state == State.DELIVERING:
            cmd["action"] = "deliver"
            time.sleep(3)  # 等待取药
            # 开始返程
            self.current_path = self.map.shortest_path(
                self.current_order["ward"], "start"
            )
            self.path_idx = 0
            self.state = State.RETURNING

        elif self.state == State.RETURNING:
            # 类似NAVIGATING，但方向相反
            offset = vision_result["offset"]
            correction = 0.5 * offset
            cmd["left_speed"] = int(150 - correction)
            cmd["right_speed"] = int(150 + correction)  # 反向

        elif self.state == State.AVOIDING:
            # 后退→转向→重新寻线
            time.sleep(1)
            cmd = {"left_speed": 100, "right_speed": -100, "action": "spin"}
            time.sleep(0.5)
            self.state = State.NAVIGATING

        return cmd

    def _get_turn_direction(self):
        # 根据当前节点和下一节点的位置关系计算
        return "right"  # 简化

class DualCarCoordinator:
    """双车协同通信"""
    def __init__(self, car_id):
        self.car_id = car_id
        self.peer_status = None

    def broadcast_status(self, position, path):
        """广播自身状态"""
        import socket
        msg = {
            "car_id": self.car_id,
            "position": position,
            "path": path,
            "timestamp": time.time()
        }
        # UDP广播
        sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        sock.setsockopt(socket.SOL_SOCKET, socket.SO_BROADCAST, 1)
        sock.sendto(json.dumps(msg).encode(), ("255.255.255.255", 9999))

    def check_conflict(self, my_path, peer_path):
        """检测路径冲突，决定避让策略"""
        # 如果路径交叉点相同且时间接近，低优先级车避让
        my_priority = self.car_id  # 简单: ID小优先
        return "yield" if my_priority > self.peer_status["car_id"] else "go"
```

### 4.3 STM32通信协议

```python
"""Orange Pi ↔ STM32 UART协议"""
import serial
import struct
import threading

class STM32Comm:
    def __init__(self, port="/dev/ttyS2", baud=115200):
        self.ser = serial.Serial(port, baud, timeout=0.1)
        self.lock = threading.Lock()
        self.encoder_data = {"left": 0, "right": 0}
        self.ultrasonic = [0, 0, 0]

    def send_motor(self, left_pwm, right_pwm):
        """发送电机控制 [-1000, 1000]"""
        left_pwm = max(-1000, min(1000, int(left_pwm)))
        right_pwm = max(-1000, min(1000, int(right_pwm)))
        cmd = struct.pack('<BhhB', 0xAA, left_pwm, right_pwm, 0x55)
        with self.lock:
            self.ser.write(cmd)

    def send_servo(self, angle):
        """发送舵机角度 [0, 180]"""
        cmd = struct.pack('<BBB', 0xAB, int(angle), 0x55)
        with self.lock:
            self.ser.write(cmd)

    def read_feedback(self):
        """读取STM32反馈"""
        while True:
            if self.ser.read(1) == b'\xAA':
                data = self.ser.read(11)
                if len(data) == 11 and data[-1] == 0x55:
                    enc_l, enc_r = struct.unpack('<hh', data[0:4])
                    us1, us2, us3 = struct.unpack('<HHH', data[4:10])
                    self.encoder_data = {"left": enc_l, "right": enc_r}
                    self.ultrasonic = [us1, us2, us3]

    def start(self):
        t = threading.Thread(target=self.read_feedback, daemon=True)
        t.start()
```

### 4.4 主程序

```python
#!/usr/bin/env python3
"""送药小车主程序"""
import cv2
import time

def main():
    vision = VisionNavigator()
    nav = NavigationController()
    stm = STM32Comm(port="/dev/ttyS2")
    coord = DualCarCoordinator(car_id=1)
    stm.start()

    # 接收任务
    nav.set_order("ward_1", "降压药")

    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    while True:
        ret, frame = cap.read()
        if not ret:
            continue

        # 视觉处理
        result = vision.process_frame(frame)

        # 导航控制
        us_dist = min(stm.ultrasonic) * 0.017  # cm
        cmd = nav.update(result, us_dist)

        # 执行
        if cmd["action"] == "deliver":
            stm.send_motor(0, 0)
            # 蜂鸣/LED提示
            print("送药完成，等待取药...")
        elif cmd["action"] == "stop":
            stm.send_motor(0, 0)
        else:
            stm.send_motor(cmd["left_speed"], cmd["right_speed"])

        # 双车间状态广播
        coord.broadcast_status(nav.path_idx, nav.current_path)

        # 调试显示
        cv2.putText(frame, f"State: {nav.state.value}", (10, 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 1, (0,255,0), 2)
        cv2.putText(frame, f"Offset: {result['offset']}", (10, 60),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0,255,0), 2)
        cv2.imshow("Debug", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

        if nav.state == State.DONE:
            print("任务完成!")
            break

    stm.send_motor(0, 0)
    cap.release()

if __name__ == "__main__":
    main()
```

## 五、调试技巧

1. **循线调参**: 先调P值使车沿直线走，再加D消除震荡
2. **路口识别**: 用不同颜色标记不同方向
3. **双车调试**: 先单独调通，再联网测试
4. **路径录制**: 先手动遥控跑一遍，录制路径数据

## 六、评分要点

| 项目 | 方案对应 |
|------|---------|
| 路径识别准确率 | 双重检测(颜色+NPU) |
| 导航精度 | PID循迹 + 编码器 |
| 送药成功率 | 状态机 + 重试机制 |
| 双车协同 | UDP广播 + 冲突检测 |
