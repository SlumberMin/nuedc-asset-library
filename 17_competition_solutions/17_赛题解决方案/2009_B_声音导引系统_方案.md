# 2009年电赛B题 - 声音导引系统 Orange Pi 5解决方案

## 一、题目分析

利用声音信号导引小车运动到指定位置, 实现声源定位和运动控制。

### 核心要求
1. 声音信号发射与接收
2. 声源定位 (TOA/TDOA)
3. 小车向声源位置运动
4. 定位精度 ≤ 5cm

---

### 4.1 声音TDOA定位

```python
import numpy as np
import time
from scipy.signal import correlate
from collections import deque

class SoundLocalizer:
    """基于TDOA的声音定位系统"""

    def __init__(self, mic_positions=None, sample_rate=860):
        """
        mic_positions: 麦克风位置 [(x,y), ...] 单位m
        sample_rate: ADS1115采样率
        """
        # 默认三角阵列 (边长12cm)
        if mic_positions is None:
            d = 0.12  # 12cm
            self.mic_positions = np.array([
                [-d/2, 0],      # M1 左
                [d/2, 0],       # M2 右
                [0, d*np.sqrt(3)/2]  # M3 后
            ])
        else:
            self.mic_positions = np.array(mic_positions)

        self.sample_rate = sample_rate
        self.speed_of_sound = 343.0  # m/s

        # 采样缓冲
        self.buffer_size = 200  # 采样点数
        self.buffers = [deque(maxlen=self.buffer_size) for _ in range(3)]

        # 定位结果滤波
        self.pos_history = deque(maxlen=10)

    def add_samples(self, ch0, ch1, ch2):
        """添加ADC采样数据"""
        self.buffers[0].append(ch0)
        self.buffers[1].append(ch1)
        self.buffers[2].append(ch2)

    def compute_tdoa(self, signal1, signal2):
        """
        计算两个信号的到达时间差
        使用互相关方法
        """
        # 去直流
        s1 = signal1 - np.mean(signal1)
        s2 = signal2 - np.mean(signal2)

        # 带通滤波 (保留目标频率)
        s1 = self._bandpass(s1, 38000, 42000)
        s2 = self._bandpass(s2, 38000, 42000)

        # 互相关
        correlation = correlate(s1, s2, mode='full')
        lags = np.arange(-len(s1)+1, len(s1))

        # 找峰值
        peak_idx = np.argmax(np.abs(correlation))
        lag = lags[peak_idx]

        # 时间差
        tdoa = lag / self.sample_rate

        return tdoa

    def _bandpass(self, signal, low_freq, high_freq):
        """简单带通滤波"""
        # 对于40kHz超声波, 用简单的滑动平均近似
        # 实际应使用scipy.signal.butter
        from scipy.signal import butter, filtfort
        nyq = self.sample_rate / 2
        if nyq <= high_freq:
            return signal  # 采样率不够, 直接返回
        b, a = butter(4, [low_freq/nyq, high_freq/nyq], btype='band')
        return filtfilt(b, a, signal)

    def localize(self):
        """
        TDOA定位
        返回: (x, y) 声源位置, 或 None
        """
        if len(self.buffers[0]) < self.buffer_size:
            return None

        # 转为numpy数组
        signals = [np.array(list(buf)) for buf in self.buffers]

        # 计算TDOA (以M1为参考)
        tdoa_12 = self.compute_tdoa(signals[0], signals[1])  # M1-M2
        tdoa_13 = self.compute_tdoa(signals[0], signals[2])  # M1-M3

        # TDOA转距离差
        d12 = tdoa_12 * self.speed_of_sound
        d13 = tdoa_13 * self.speed_of_sound

        # 解双曲线方程
        position = self._solve_hyperbola(d12, d13)

        if position is not None:
            # 位置滤波 (中值)
            self.pos_history.append(position)
            if len(self.pos_history) >= 3:
                positions = np.array(list(self.pos_history))
                return np.median(positions, axis=0)

        return position

    def _solve_hyperbola(self, d12, d13):
        """
        解双曲线定位方程
        d12 = dist(P,M1) - dist(P,M2)  (M1-M2距离差)
        d13 = dist(P,M1) - dist(P,M3)  (M1-M3距离差)
        """
        m1 = self.mic_positions[0]
        m2 = self.mic_positions[1]
        m3 = self.mic_positions[2]

        # 最小二乘法求解
        # 设声源位置为(x,y), 已知M1位置(x1,y1)
        # (x-x1)^2 + (y-y1)^2 - (x-x2)^2 - (y-y2)^2 = d12^2 + 2*d12*r1
        # 简化: 线性化处理

        # 使用迭代法 (牛顿法)
        # 初始猜测: 阵列中心
        x, y = 0.0, 0.5

        for _ in range(20):
            r1 = np.sqrt((x-m1[0])**2 + (y-m1[1])**2)
            r2 = np.sqrt((x-m2[0])**2 + (y-m2[1])**2)
            r3 = np.sqrt((x-m3[0])**2 + (y-m3[1])**2)

            # 残差
            f1 = (r1 - r2) - d12
            f2 = (r1 - r3) - d13

            # 雅可比矩阵
            J = np.array([
                [(x-m1[0])/r1 - (x-m2[0])/r2,
                 (y-m1[1])/r1 - (y-m2[1])/r2],
                [(x-m1[0])/r1 - (x-m3[0])/r3,
                 (y-m1[1])/r1 - (y-m3[1])/r3]
            ])

            F = np.array([f1, f2])

            # 更新
            try:
                delta = np.linalg.solve(J, -F)
                x += delta[0]
                y += delta[1]

                if np.linalg.norm(delta) < 0.001:
                    break
            except np.linalg.LinAlgError:
                return None

        # 检查合理性
        if -2 < x < 2 and -2 < y < 2:
            return np.array([x, y])
        return None
```

### 4.2 视觉辅助定位

```python
import cv2
import numpy as np
from cv2 import aruco

class VisualAssist:
    """视觉辅助定位"""

    def __init__(self):
        self.aruco_dict = aruco.getPredefinedDictionary(aruco.DICT_4X4_50)
        self.camera_matrix = np.load('camera_matrix.npy')
        self.dist_coeffs = np.load('dist_coeffs.npy')

        # 地标位置 (已知)
        self.landmarks = {
            0: (0, 0),
            1: (2.0, 0),
            2: (0, 2.0),
            3: (2.0, 2.0),
        }
        self.marker_size = 0.1  # 10cm

    def detect_landmarks(self, frame):
        """检测ArUco地标"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, _ = aruco.detectMarkers(gray, self.aruco_dict)
        return corners, ids

    def estimate_position(self, frame):
        """
        通过地标估计小车位置
        返回: (x, y) 或 None
        """
        corners, ids = self.detect_landmarks(frame)
        if ids is None:
            return None

        positions = []
        for i, mid in enumerate(ids.flatten()):
            if mid not in self.landmarks:
                continue

            obj_pts = np.array([
                [-self.marker_size/2, -self.marker_size/2, 0],
                [ self.marker_size/2, -self.marker_size/2, 0],
                [ self.marker_size/2,  self.marker_size/2, 0],
                [-self.marker_size/2,  self.marker_size/2, 0]
            ], dtype=np.float32)

            ret, rvec, tvec = cv2.solvePnP(
                obj_pts, corners[i][0],
                self.camera_matrix, self.dist_coeffs
            )

            if ret:
                R, _ = cv2.Rodrigues(rvec)
                cam_pos = (-R.T @ tvec).flatten()

                lm = self.landmarks[mid]
                world_x = lm[0] + cam_pos[0]
                world_y = lm[1] + cam_pos[1]
                positions.append((world_x, world_y))

        if positions:
            return np.mean(positions, axis=0)
        return None
```

### 4.3 融合定位与运动控制

```python
import numpy as np

class FusionLocalizer:
    """声音+视觉融合定位 (卡尔曼滤波)"""

    def __init__(self):
        # 状态: [x, y, vx, vy]
        self.x = np.zeros(4)

        # 状态转移
        self.F = np.eye(4)
        self.F[0, 2] = 0.1  # dt
        self.F[1, 3] = 0.1

        # 过程噪声
        self.Q = np.eye(4) * 0.01

        # 状态协方差
        self.P = np.eye(4) * 1.0

        # 声音观测矩阵
        self.H_sound = np.array([[1, 0, 0, 0],
                                  [0, 1, 0, 0]])
        self.R_sound = np.eye(2) * 0.1  # 声音定位噪声较大

        # 视觉观测矩阵
        self.H_visual = np.array([[1, 0, 0, 0],
                                   [0, 1, 0, 0]])
        self.R_visual = np.eye(2) * 0.01  # 视觉定位噪声较小

    def predict(self, dt=0.1):
        """预测步骤"""
        self.F[0, 2] = dt
        self.F[1, 3] = dt
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q

    def update_sound(self, z):
        """声音观测更新"""
        z = np.array(z)
        y = z - self.H_sound @ self.x
        S = self.H_sound @ self.P @ self.H_sound.T + self.R_sound
        K = self.P @ self.H_sound.T @ np.linalg.inv(S)
        self.x = self.x + K @ y
        self.P = (np.eye(4) - K @ self.H_sound) @ self.P

    def update_visual(self, z):
        """视觉观测更新"""
        z = np.array(z)
        y = z - self.H_visual @ self.x
        S = self.H_visual @ self.P @ self.H_visual.T + self.R_visual
        K = self.P @ self.H_visual.T @ np.linalg.inv(S)
        self.x = self.x + K @ y
        self.P = (np.eye(4) - K @ self.H_visual) @ self.P

    def get_position(self):
        return self.x[:2].copy()

    def get_velocity(self):
        return self.x[2:].copy()


class MotionController:
    """运动控制器"""

    def __init__(self):
        self.Kp_linear = 0.5
        self.Kp_angular = 2.0
        self.target_pos = np.array([0.0, 0.0])

    def compute(self, current_pos, current_heading):
        """
        计算左右轮速度
        current_pos: (x, y)
        current_heading: 航向角 (rad)
        返回: (v_left, v_right)
        """
        dx = self.target_pos[0] - current_pos[0]
        dy = self.target_pos[1] - current_pos[1]

        distance = np.sqrt(dx**2 + dy**2)
        target_angle = np.arctan2(dy, dx)

        # 角度误差
        angle_error = target_angle - current_heading
        # 归一化到 [-π, π]
        angle_error = (angle_error + np.pi) % (2*np.pi) - np.pi

        # 距离足够近时停车
        if distance < 0.05:
            return 0, 0

        # 控制律
        linear_speed = self.Kp_linear * min(distance, 1.0)
        angular_speed = self.Kp_angular * angle_error

        # 差速
        v_left = linear_speed - angular_speed * 0.1  # 轮距/2
        v_right = linear_speed + angular_speed * 0.1

        return v_left, v_right
```

### 4.4 主程序

```python
import cv2
import time
import serial
import struct
import numpy as np
from sound_localizer import SoundLocalizer
from visual_assist import VisualAssist
from fusion import FusionLocalizer, MotionController
import ads1115  # ADS1115 ADC驱动

class SoundGuidedCar:
    """声音导引小车"""

    def __init__(self):
        self.adc = ads1115.ADS1115(address=0x48)
        self.adc.set_mode('continuous')
        self.adc.set_data_rate(860)  # 860 SPS
        self.adc.set_programmable_gain(0.256)  # ±256mV

        self.sound_loc = SoundLocalizer()
        self.visual_loc = VisualAssist()
        self.fusion = FusionLocalizer()
        self.controller = MotionController()

        self.cap = cv2.VideoCapture(0)
        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)

        self.heading = 0.0  # 航向角 (从IMU获取)
        self.mode = 'STOP'

    def send_motor_cmd(self, v_left, v_right):
        buf = bytearray(10)
        buf[0] = 0xAA
        struct.pack_into('<f', buf, 1, v_left)
        struct.pack_into('<f', buf, 5, v_right)
        buf[9] = 0x55
        self.ser.write(buf)

    def read_adc(self):
        """读取3通道麦克风ADC"""
        ch0 = self.adc.read(0, differential=False)
        ch1 = self.adc.read(1, differential=False)
        ch2 = self.adc.read(2, differential=False)
        return ch0, ch1, ch2

    def read_imu_heading(self):
        """从STM32读取IMU航向角"""
        if self.ser.in_waiting >= 8:
            data = self.ser.read(8)
            if data[0] == 0xBB:
                heading = struct.unpack('<f', data[1:5])[0]
                return heading
        return self.heading

    def run(self):
        print("=== 声音导引系统 ===")
        print("指令: g=开始导航, s=停止, t=设目标位置")

        last_time = time.time()
        adc_count = 0

        while True:
            ret, frame = self.cap.read()
            current_time = time.time()
            dt = current_time - last_time
            last_time = current_time

            # 1. 声音采集 (高频)
            ch0, ch1, ch2 = self.read_adc()
            self.sound_loc.add_samples(ch0, ch1, ch2)
            adc_count += 1

            # 2. 融合定位 (每10帧)
            self.fusion.predict(dt)

            # 声音定位 (每20个采样点)
            if adc_count >= 20:
                sound_pos = self.sound_loc.localize()
                if sound_pos is not None:
                    self.fusion.update_sound(sound_pos)
                adc_count = 0

            # 视觉定位 (每帧)
            if ret:
                visual_pos = self.visual_loc.estimate_position(frame)
                if visual_pos is not None:
                    self.fusion.update_visual(visual_pos)

            # 3. 运动控制
            if self.mode == 'NAVIGATE':
                pos = self.fusion.get_position()
                v_left, v_right = self.controller.compute(
                    pos, self.heading)
                self.send_motor_cmd(v_left, v_right)

                # 到达目标?
                dist = np.linalg.norm(
                    pos - self.controller.target_pos)
                if dist < 0.05:
                    self.send_motor_cmd(0, 0)
                    self.mode = 'ARRIVED'
                    print(f"到达目标! 位置: {pos}")

            # 4. 更新航向
            self.heading = self.read_imu_heading()

            # 5. 可视化
            if ret:
                self._draw(frame)

            key = cv2.waitKey(1) & 0xFF
            if key == ord('q'):
                break
            elif key == ord('g'):
                self.mode = 'NAVIGATE'
                print("开始导航...")
            elif key == ord('s'):
                self.mode = 'STOP'
                self.send_motor_cmd(0, 0)
            elif key == ord('t'):
                # 设置目标 (示例: 声源位置)
                self.controller.target_pos = np.array([1.0, 1.5])
                print(f"目标: {self.controller.target_pos}")

        self.send_motor_cmd(0, 0)
        self.cap.release()
        cv2.destroyAllWindows()

    def _draw(self, frame):
        pos = self.fusion.get_position()

        cv2.putText(frame, f"Mode: {self.mode}", (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        cv2.putText(frame,
            f"Pos: ({pos[0]:.2f}, {pos[1]:.2f})", (10, 50),
            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)
        cv2.putText(frame,
            f"Tgt: ({self.controller.target_pos[0]:.2f}, "
            f"{self.controller.target_pos[1]:.2f})", (10, 75),
            cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 255), 2)

        # 地图显示
        map_img = np.zeros((200, 200, 3), dtype=np.uint8)
        # 小车位置
        px = int(pos[0] * 80 + 100)
        py = int(-pos[1] * 80 + 180)
        cv2.circle(map_img, (px, py), 5, (0, 255, 0), -1)
        # 目标位置
        tx = int(self.controller.target_pos[0] * 80 + 100)
        ty = int(-self.controller.target_pos[1] * 80 + 180)
        cv2.circle(map_img, (tx, ty), 5, (0, 0, 255), -1)
        # 地标
        for mid, (lx, ly) in self.visual_loc.landmarks.items():
            lx_px = int(lx * 80 + 100)
            ly_px = int(-ly * 80 + 180)
            cv2.rectangle(map_img,
                (lx_px-3, ly_px-3), (lx_px+3, ly_px+3),
                (255, 255, 0), -1)

        frame[10:210, 430:630] = map_img
        cv2.rectangle(frame, (430, 10), (630, 210), (255, 255, 255), 1)

        cv2.imshow('Sound Guided Car', frame)

if __name__ == '__main__':
    car = SoundGuidedCar()
    car.run()
```

## 五、调试要点

### 5.1 声音定位精度
| 因素 | 措施 |
|------|------|
| 采样率 | ADS1115设860SPS, 时间分辨率~1.2ms |
| 麦克风间距 | 12~15cm, 太小精度差 |
| 频率选择 | 40kHz超声波抗干扰强 |
| 多径反射 | 空旷场地测试, 加吸音材料 |

### 5.2 融合定位
- 声音定位: 大范围粗定位 (~10cm精度)
- 视觉定位: 近距离精定位 (~1cm精度)
- 融合策略: 远距离用声音, 近距离用视觉

### 5.3 注意事项
