# 2013 全国大学生电子设计竞赛 - 手写绘图板

## 题目概述
设计一个手写绘图板，能在平板上用笔书写或绘制图形，实时显示在屏幕上，要求具有较高精度和流畅度。

## 软件架构

### 方案A：电阻式触摸屏 + ADS1115

```python
import board
import busio
import adafruit_ads1x15.ads1115 as ADS
from adafruit_ads1x15.analog_in import AnalogIn

class ResistiveTouch:
    """4线电阻触摸屏驱动"""
    def __init__(self, width=1024, height=768):
        i2c = busio.I2C(board.SCL, board.SDA)
        self.ads = ADS.ADS1115(i2c)
        self.width = width
        self.height = height

    def read_x(self):
        """读取X坐标（切换引脚模式）"""
        # 需要GPIO切换触摸屏引脚方向
        chan = AnalogIn(self.ads, ADS.P0, ADS.P1)
        return chan.value

    def read_y(self):
        """读取Y坐标"""
        chan = AnalogIn(self.ads, ADS.P2, ADS.P3)
        return chan.value

    def get_position(self):
        x = self.read_x()
        y = self.read_y()
        if x > 100 and y > 100:  # 有触摸
            return (int(x * self.width / 65535), int(y * self.height / 65535))
        return None
```

### 方案C：视觉检测手写

```python
import cv2
import numpy as np

class VisionPenTracker:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.bg_subtractor = cv2.createBackgroundSubtractorMOG2()
        self.prev_point = None

    def detect_pen_tip(self, frame):
        """检测笔尖位置（深色笔在白板上）"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 检测深色笔尖
        lower = np.array([0, 0, 0])
        upper = np.array([180, 255, 80])
        mask = cv2.inRange(hsv, lower, upper)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((3,3), np.uint8))
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 50:
                M = cv2.moments(c)
                if M["m00"] > 0:
                    return (int(M["m10"]/M["m00"]), int(M["m01"]/M["m00"]))
        return None
```

### 绘图引擎（PyQt5 / OpenCV窗口）

```python
import cv2
import numpy as np
from collections import deque

class DrawingCanvas:
    def __init__(self, width=1280, height=720):
        self.canvas = np.ones((height, width, 3), dtype=np.uint8) * 255
        self.points = deque(maxlen=5000)
        self.stroke_width = 2
        self.color = (0, 0, 0)

    def add_point(self, point, pen_down=True):
        if pen_down and point:
            self.points.append(point)
            if len(self.points) > 1:
                cv2.line(self.canvas, self.points[-2], self.points[-1],
                        self.color, self.stroke_width)

    def smooth_stroke(self, points, window=5):
        """平滑笔迹（移动平均）"""
        if len(points) < window:
            return points
        smoothed = []
        for i in range(len(points)):
            start = max(0, i - window//2)
            end = min(len(points), i + window//2 + 1)
            avg_x = int(np.mean([p[0] for p in points[start:end]]))
            avg_y = int(np.mean([p[1] for p in points[start:end]]))
            smoothed.append((avg_x, avg_y))
        return smoothed

    def save(self, filename):
        cv2.imwrite(filename, self.canvas)

    def clear(self):
        self.canvas[:] = 255
        self.points.clear()
```

### 主程序

```python
import cv2
import time

def main():
    # 选择输入方式
    # touch = ResistiveTouch()
    pen_tracker = VisionPenTracker(camera_id=0)
    canvas = DrawingCanvas()
    display = cv2.namedWindow("Drawing Board", cv2.WINDOW_NORMAL)

    while True:
        ret, frame = pen_tracker.cap.read()
        if not ret:
            break

        point = pen_tracker.detect_pen_tip(frame)
        canvas.add_point(point)

        # 显示
        cv2.imshow("Drawing Board", canvas.canvas)

        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break
        elif key == ord('s'):
            canvas.save("drawing.png")
        elif key == ord('c'):
            canvas.clear()

    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

## 关键技术要点

### 坐标标定
- **四点标定法**：触摸屏四角对应屏幕坐标
- **线性插值**：`screen_x = (touch_x - offset_x) * scale_x`
- **畸变校正**：非线性区域用多项式拟合

### 笔迹平滑
- 移动平均滤波消除抖动
- 贝塞尔曲线插值使笔迹流畅
- 压力感应（电阻屏）控制线条粗细

### 图形识别（加分项）
- 基本图形识别：直线、圆、矩形
- 使用Douglas-Peucker算法简化轨迹
- 模板匹配识别手写字符
