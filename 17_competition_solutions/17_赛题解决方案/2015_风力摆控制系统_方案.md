# 2015年电赛 - 风力摆控制系统 - Orange Pi 5解决方案

## 题目要求

设计制作风力摆控制系统，实现：
1. 通过4个风扇控制摆杆运动
2. 摆杆画圆、画直线等轨迹
3. 快速响应，小超调
4. 精确轨迹跟踪

## 系统架构

```

                                                              │ 十字布局 │
                                                              └─────────┘
```

## 核心设计

### 1. 视觉检测摆锤位置

```python
import cv2
import numpy as np
import time
import struct
import serial
import math

class WindPendulumVision:
    """风力摆视觉检测"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 120)  # 高帧率
        
        # 悬挂点（标定）
        self.pivot_point = (320, 50)
        
        # 摆锤颜色标记（绿色）
        self.hsv_lower = np.array([35, 100, 100])
        self.hsv_upper = np.array([85, 255, 255])
        
        # 坐标系转换参数（像素→实际距离mm）
        self.pixels_per_mm = 5.0  # 需标定
        
        # 位置历史
        self.pos_history = []
        self.time_history = []
        self.max_history = 20
        
    def detect_pendulum_position(self):
        """检测摆锤在平面上的(x, y)位置"""
        ret, frame = self.cap.read()
        if not ret:
            return None, None
            
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.hsv_lower, self.hsv_upper)
        
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            M = cv2.moments(largest)
            if M["m00"] > 50:
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                
                # 转换为实际坐标（mm）
                x_mm = (cx - self.pivot_point[0]) / self.pixels_per_mm
                y_mm = (cy - self.pivot_point[1]) / self.pixels_per_mm
                
                # 记录历史
                now = time.time()
                self.pos_history.append((x_mm, y_mm))
                self.time_history.append(now)
                if len(self.pos_history) > self.max_history:
                    self.pos_history.pop(0)
                    self.time_history.pop(0)
                
                # 可视化
                cv2.circle(frame, (cx, cy), 8, (0, 255, 0), -1)
                cv2.putText(frame, f"({x_mm:.0f}, {y_mm:.0f})mm", (10, 30),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
                
                return x_mm, y_mm
        
        return None, None
    
    def get_velocity(self):
        """估计摆锤速度"""
        if len(self.pos_history) < 2:
            return 0.0, 0.0
        
        dt = self.time_history[-1] - self.time_history[-2]
        if dt < 1e-6:
            return 0.0, 0.0
        
        vx = (self.pos_history[-1][0] - self.pos_history[-2][0]) / dt
        vy = (self.pos_history[-1][1] - self.pos_history[-2][1]) / dt
        return vx, vy
    
    def release(self):
        self.cap.release()
```

### 2. 四风机协调控制

```python
class FourFanController:
    """
    四风扇控制器
    风扇布局（俯视）:
         风扇0(前)
           ↑
    风扇3(左) ← → 风扇1(右)
           ↓
         风扇2(后)
    """
    
    def __init__(self, ser):
        self.ser = ser
        
        # PID参数（X和Y方向独立）
        self.pid_x = PIDController(Kp=2.0, Ki=0.1, Kd=1.5)
        self.pid_y = PIDController(Kp=2.0, Ki=0.1, Kd=1.5)
        
        # 风扇最大PWM
        self.pwm_max = 999
        
    def compute_fan_pwm(self, target_x, target_y, current_x, current_y, vx=0, vy=0):
        """
        根据目标位置和当前位置计算4路风扇PWM
        
        target: 目标位置(mm)
        current: 当前位置(mm)
        """
        # PID计算X和Y方向的推力分量
        fx = self.pid_x.compute(target_x, current_x)
        fy = self.pid_y.compute(target_y, current_y)
        
        # 添加速度前馈（阻尼）
        fx -= vx * 0.001
        fy -= vy * 0.001
        
        # 分配到4个风扇
        # 风扇0(+Y方向推), 风扇1(+X方向推), 风扇2(-Y方向推), 风扇3(-X方向推)
        fan = [0] * 4
        fan[0] = max(0, fy)     # 前风扇：Y正方向
        fan[2] = max(0, -fy)    # 后风扇：Y负方向
        fan[1] = max(0, fx)     # 右风扇：X正方向
        fan[3] = max(0, -fx)    # 左风扇：X负方向
        
        # 归一化并转换为PWM
        pwm = [int(np.clip(f / 100 * self.pwm_max, 0, self.pwm_max)) for f in fan]
        
        return pwm
    
    def send_fan_pwm(self, pwm):
        """发送PWM到STM32"""
        data = struct.pack('<HHHH', pwm[0], pwm[1], pwm[2], pwm[3])
        self.ser.write(b'\xAA' + data + b'\x55')
    
    def stop_all(self):
        """停止所有风扇"""
        self.send_fan_pwm([0, 0, 0, 0])


class PIDController:
    """PID控制器"""
    
    def __init__(self, Kp, Ki, Kd, output_max=100, output_min=-100):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.output_max = output_max
        self.output_min = output_min
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None
        
    def compute(self, target, current):
        error = target - current
        now = time.time()
        
        if self.prev_time is None:
            self.prev_time = now
            self.prev_error = error
            return 0
        
        dt = now - self.prev_time
        if dt < 1e-6:
            dt = 1e-6
            
        # P
        p_term = self.Kp * error
        
        # I（带抗积分饱和）
        self.integral += error * dt
        self.integral = np.clip(self.integral, -50, 50)
        i_term = self.Ki * self.integral
        
        # D
        d_term = self.Kd * (error - self.prev_error) / dt
        
        output = p_term + i_term + d_term
        output = np.clip(output, self.output_min, self.output_max)
        
        self.prev_error = error
        self.prev_time = now
        
        return output
    
    def reset(self):
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None


class TrajectoryGenerator:
    """轨迹生成器"""
    
    @staticmethod
    def circle(t, radius=150, period=4.0):
        """圆形轨迹"""
        omega = 2 * np.pi / period
        x = radius * np.cos(omega * t)
        y = radius * np.sin(omega * t)
        return x, y
    
    @staticmethod
    def line(t, length=300, angle=0, period=4.0):
        """直线轨迹"""
        t_mod = t % period
        if t_mod < period / 2:
            ratio = t_mod / (period / 2)
        else:
            ratio = 1 - (t_mod - period / 2) / (period / 2)
        
        x = length * (ratio - 0.5) * np.cos(np.radians(angle))
        y = length * (ratio - 0.5) * np.sin(np.radians(angle))
        return x, y
    
    @staticmethod
    def figure_eight(t, radius=100, period=6.0):
        """8字形轨迹"""
        omega = 2 * np.pi / period
        x = radius * np.sin(omega * t)
        y = radius * np.sin(2 * omega * t) / 2
        return x, y


def main():
    vision = WindPendulumVision(camera_id=0)
    ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
    controller = FourFanController(ser)
    trajectory = TrajectoryGenerator()
    
    mode = 'circle'  # circle/line/figure_eight/point
    target_x, target_y = 0, 0
    
    print("风力摆控制系统启动")
    print("模式: circle-画圆, line-画直线, figure8-8字, point-定点")
    
    t_start = time.time()
    
    try:
        while True:
            t = time.time() - t_start
            
            # 视觉检测
            x, y = vision.detect_pendulum_position()
            vx, vy = vision.get_velocity()
            
            if x is not None:
                # 生成目标轨迹
                if mode == 'circle':
                    target_x, target_y = trajectory.circle(t, radius=150, period=5)
                elif mode == 'line':
                    target_x, target_y = trajectory.line(t, length=300, angle=45)
                elif mode == 'figure_eight':
                    target_x, target_y = trajectory.figure_eight(t)
                elif mode == 'point':
                    pass  # 使用固定目标
                
                # 计算风扇PWM
                pwm = controller.compute_fan_pwm(target_x, target_y, x, y, vx, vy)
                
                # 发送
                controller.send_fan_pwm(pwm)
                
                print(f"目标:({target_x:.0f},{target_y:.0f}) 实际:({x:.0f},{y:.0f}) PWM:{pwm}")
            
            time.sleep(0.01)  # 100Hz控制周期
            
    except KeyboardInterrupt:
        pass
    finally:
        controller.stop_all()
        vision.release()
        ser.close()
        print("系统已停止")
```

### 3. STM32端风扇驱动

```c
// 4路风扇PWM驱动
// 接收Orange Pi 5的PWM值，驱动4个风扇

void Fan_SetPWM(uint16_t fan0, uint16_t fan1, uint16_t fan2, uint16_t fan3) {
    __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, fan0);  // 前
    __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_2, fan1);  // 右
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, fan2);  // 后
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, fan3);  // 左
}
```

## 接线说明

| Orange Pi 5引脚 | 连接目标 | 说明 |
|-----------------|---------|------|
| CSI接口 | 摄像头模块 | 俯视拍摄 |
| UART3_TX | STM32 RX | 控制指令 |
| 5V | STM32 VCC | 供电 |
| GND | 公共地 | - |

## 调试要点

1. **坐标系标定**：准确标定像素到mm的转换系数
2. **PID调参**：先调P使系统响应，再加D减超调，最后加I消除稳态误差
3. **风扇响应延迟**：需在控制中补偿
4. **轨迹平滑**：避免目标突变导致摆杆剧烈摆动
