# 2016 全国大学生电子设计竞赛 - 自动循迹小车

## 题目概述
设计一辆自动循迹小车，能在白色地面的黑色引导线上行驶，完成循迹、避障、转弯等任务，要求速度快且不脱离轨迹。

## 软件架构

### 1. 视觉循迹（核心）

```python
import cv2
import numpy as np

class LineTracker:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
        self.cap.set(cv2.CAP_PROP_FPS, 30)

    def detect_line(self, frame):
        """检测引导线，返回偏移量(像素)和角度"""
        # 裁剪下半部分（近处视野）
        h, w = frame.shape[:2]
        roi = frame[int(h*0.5):, :]

        # 灰度 + 二值化
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        _, binary = cv2.threshold(gray, 60, 255, cv2.THRESH_BINARY_INV)

        # 形态学处理
        kernel = np.ones((5,5), np.uint8)
        binary = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)

        # 检测黑线轮廓
        contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 200:
                # 计算线的中心
                M = cv2.moments(c)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    # 用拟合直线计算角度
                    vx, vy, x0, y0 = cv2.fitLine(c, cv2.DIST_L2, 0, 0.01, 0.01)
                    angle = np.arctan2(vy[0], vx[0]) * 180 / np.pi
                    offset = cx - w // 2  # 正=偏右，负=偏左
                    return offset, angle, binary
        return 0, 0, binary

    def detect_intersection(self, binary):
        """检测交叉路口"""
        h, w = binary.shape
        # 在特定行检测黑线宽度
        row = binary[int(h*0.3), :]
        black_pixels = np.sum(row == 255)
        if black_pixels > w * 0.4:
            return "intersection"
        return "normal"

    def detect_finish(self, binary):
        """检测终点线（横向粗线）"""
        h, w = binary.shape
        top_row = binary[0, :]
        if np.sum(top_row == 255) > w * 0.6:
            return True
        return False
```

### 2. PID转向控制器

```python
import time

class SteerController:
    def __init__(self):
        self.Kp = 0.8
        self.Ki = 0.02
        self.Kd = 0.5
        self.integral = 0
        self.prev_error = 0
        self.max_integral = 500

    def compute(self, offset, angle):
        """根据偏移和角度计算转向量"""
        # 偏移权重 + 角度权重
        error = offset * 0.7 + angle * 0.3

        self.integral += error
        self.integral = max(-self.max_integral, min(self.max_integral, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error

        steer = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        return steer

    def reset(self):
        self.integral = 0
        self.prev_error = 0
```

### 3. 电机驱动控制

```python
import serial

class MotorDriver:
    def __init__(self, uart_port='/dev/ttyS3'):
        self.ser = serial.Serial(uart_port, 115200)

    def set_motors(self, left_speed, right_speed):
        """设置左右电机速度 -255 ~ +255"""
        left_speed = max(-255, min(255, int(left_speed)))
        right_speed = max(-255, min(255, int(right_speed)))
        # 协议: [0xAA, 左高, 左低, 右高, 右低, 0x55]
        cmd = bytes([
            0xAA,
            (left_speed >> 8) & 0xFF, left_speed & 0xFF,
            (right_speed >> 8) & 0xFF, right_speed & 0xFF,
            0x55
        ])
        self.ser.write(cmd)

    def stop(self):
        self.set_motors(0, 0)

    def forward(self, speed=150):
        self.set_motors(speed, speed)
```

### 4. 主控制循环

```python
import cv2
import time

class AutoCar:
    def __init__(self):
        self.tracker = LineTracker(camera_id=0)
        self.steer = SteerController()
        self.motor = MotorDriver()
        self.base_speed = 120  # 基础速度
        self.running = True

    def run(self):
        while self.running:
            ret, frame = self.tracker.cap.read()
            if not ret:
                continue

            offset, angle, binary = self.tracker.detect_line(frame)

            # 检测特殊区域
            if self.tracker.detect_intersection(binary):
                self.handle_intersection()
            elif self.tracker.detect_finish(binary):
                self.stop()
                break
            else:
                # 正常循迹
                steer_value = self.steer.compute(offset, angle)
                left_speed = self.base_speed + steer_value
                right_speed = self.base_speed - steer_value
                self.motor.set_motors(left_speed, right_speed)

            # 调试显示
            cv2.imshow("Binary", binary)
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break

            time.sleep(0.01)  # ~100Hz

        self.motor.stop()
        cv2.destroyAllWindows()

    def handle_intersection(self):
        """处理路口（根据题目要求选择方向）"""
        self.motor.set_motors(80, 80)  # 减速通过
        time.sleep(0.3)

    def stop(self):
        self.motor.stop()
        self.running = False

if __name__ == "__main__":
    car = AutoCar()
    car.run()
```

## 关键技术要点

### 视觉循迹优化
- **ROI裁剪**：只处理画面下半部分，减少计算
- **分辨率降低**：320×240足够，提高帧率
- **前瞻控制**：看远处的线提前转向，减少超调

### PID调参方法
1. 先设Ki=Kd=0，逐渐增大Kp直到系统振荡
2. Kp取振荡值的60%
3. 逐渐增大Kd消除超调
4. 最后加入Ki消除稳态误差

### 弯道处理
- 急弯时降低基础速度
- 根据曲率动态调节速度：`speed = base_speed / (1 + curvature * factor)`
- 前瞻距离根据速度动态调整
