# 2019年电赛B题 - 巡线机器人 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：机器人沿地面引导线自主行驶，识别并处理十字路口、弯道、分支、断线等复杂路径，完成全程巡线行驶。

**关键指标**：
- 巡线精度（偏离量）
- 复杂路径处理能力（十字/分支/断线）
- 行驶速度
- 全程完成率

### 4.1 视觉巡线模块

```python
# line_detector.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite
from enum import Enum

class PathType(Enum):
    STRAIGHT = 0
    LEFT_TURN = 1
    RIGHT_TURN = 2
    CROSS = 3       # 十字路口
    T_LEFT = 4      # T型左转
    T_RIGHT = 5     # T型右转
    FORK = 6        # Y型分叉
    DEAD_END = 7    # 断头路
    GAP = 8         # 断线

class LineDetector:
    """视觉巡线检测器"""
    def __init__(self, model_path='line_seg.rknn'):
        # RKNN分割模型（可选）
        self.use_rknn = model_path is not None
        if self.use_rknn:
            self.rknn = RKNNLite()
            self.rknn.load_rknn(model_path)
            self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0)
        
        # 色彩参数（黑色引导线）
        self.line_lower = (0, 0, 0)
        self.line_upper = (180, 255, 60)
        
    def detect_line(self, frame):
        """检测引导线位置和类型"""
        h, w = frame.shape[:2]
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 黑色线路掩码
        mask = cv2.inRange(hsv, self.line_lower, self.line_upper)
        
        # 形态学处理
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # 多行扫描
        scan_positions = [0.9, 0.8, 0.7, 0.6, 0.5, 0.4, 0.3]
        line_points = []
        
        for ratio in scan_positions:
            row = int(h * ratio)
            scan_line = mask[row, :]
            
            # 找黑色区域
            indices = np.where(scan_line > 0)[0]
            if len(indices) > 0:
                cx = int(np.mean(indices))
                width = indices[-1] - indices[0]
                line_points.append({
                    'y': row,
                    'x': cx,
                    'width': width,
                    'normalized_x': (cx - w/2) / (w/2)
                })
            else:
                line_points.append(None)
        
        # 判断路径类型
        path_type = self._classify_path(mask, line_points, w)
        
        # 计算偏移量（用最近的有效扫描行）
        offset = 0
        for pt in line_points:
            if pt is not None:
                offset = pt['normalized_x']
                break
        
        return {
            'offset': offset,           # -1 ~ 1 归一化偏移
            'line_points': line_points,
            'path_type': path_type,
            'mask': mask
        }
    
    def _classify_path(self, mask, line_points, width):
        """分类路径类型"""
        valid_points = [p for p in line_points if p is not None]
        
        if len(valid_points) < 2:
            return PathType.GAP
        
        # 检查线路宽度（十字路口处宽度突增）
        widths = [p['width'] for p in valid_points]
        avg_width = np.mean(widths)
        near_width = widths[0] if widths else 0
        
        if near_width > avg_width * 3 and near_width > width * 0.5:
            return PathType.CROSS
        
        # 检查偏移趋势（弯道判断）
        offsets = [p['normalized_x'] for p in valid_points]
        
        if len(offsets) >= 3:
            trend = offsets[-1] - offsets[0]
            if trend < -0.3:
                return PathType.LEFT_TURN
            elif trend > 0.3:
                return PathType.RIGHT_TURN
        
        # T型路口检测（底部有线但上部分叉）
        h_seg = mask.shape[0]
        top_half = mask[:h_seg//2, :]
        bottom_half = mask[h_seg//2:, :]
        
        top_contours, _ = cv2.findContours(top_half, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        bottom_contours, _ = cv2.findContours(bottom_half, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if len(top_contours) >= 2 and len(bottom_contours) >= 1:
            # 分析分叉方向
            centers = []
            for c in top_contours:
                M = cv2.moments(c)
                if M['m00'] > 0:
                    centers.append(M['m10']/M['m00'])
            if len(centers) >= 2:
                if max(centers) - min(centers) > width * 0.3:
                    return PathType.FORK
        
        return PathType.STRAIGHT
```

### 4.2 运动控制器

```python
# motor_controller.py
import time

class DifferentialDrive:
    """差速驱动底盘控制"""
    def __init__(self, left_pin_fwd=17, left_pin_rev=27, 
                 right_pin_fwd=22, right_pin_rev=23):
        import OPi.GPIO as GPIO
        import RPi.GPIO
        
        GPIO.setmode(GPIO.BOARD)
        for pin in [left_pin_fwd, left_pin_rev, right_pin_fwd, right_pin_rev]:
            GPIO.setup(pin, GPIO.OUT)
        
        # PWM初始化
        self.pwm_lf = GPIO.PWM(left_pin_fwd, 1000)
        self.pwm_lr = GPIO.PWM(left_pin_rev, 1000)
        self.pwm_rf = GPIO.PWM(right_pin_fwd, 1000)
        self.pwm_rr = GPIO.PWM(right_pin_rev, 1000)
        
        self.pwm_lf.start(0)
        self.pwm_lr.start(0)
        self.pwm_rf.start(0)
        self.pwm_rr.start(0)
        
        # PID参数
        self.pid = PIDController(kp=40, ki=2, kd=15)
        self.base_speed = 50  # 基础速度 %
        
    def set_speed(self, left_pct, right_pct):
        """设置左右电机速度 (-100 ~ 100%)"""
        left_pct = max(-100, min(100, left_pct))
        right_pct = max(-100, min(100, right_pct))
        
        if left_pct >= 0:
            self.pwm_lf.ChangeDutyCycle(left_pct)
            self.pwm_lr.ChangeDutyCycle(0)
        else:
            self.pwm_lf.ChangeDutyCycle(0)
            self.pwm_lr.ChangeDutyCycle(-left_pct)
        
        if right_pct >= 0:
            self.pwm_rf.ChangeDutyCycle(right_pct)
            self.pwm_rr.ChangeDutyCycle(0)
        else:
            self.pwm_rf.ChangeDutyCycle(0)
            self.pwm_rr.ChangeDutyCycle(-right_pct)
    
    def follow_line(self, line_offset, path_type):
        """巡线控制"""
        if path_type == PathType.DEAD_END:
            self.set_speed(0, 0)
            return
        
        # PID计算转向修正
        correction = self.pid.compute(line_offset)
        
        # 根据路径类型调整速度
        if path_type == PathType.STRAIGHT:
            speed = self.base_speed
        elif path_type in [PathType.LEFT_TURN, PathType.RIGHT_TURN]:
            speed = self.base_speed * 0.6  # 弯道减速
        elif path_type == PathType.CROSS:
            speed = self.base_speed * 0.5
        elif path_type == PathType.GAP:
            speed = self.base_speed * 0.4  # 断线慢行
        else:
            speed = self.base_speed * 0.5
        
        # 差速转向
        left_speed = speed + correction
        right_speed = speed - correction
        
        self.set_speed(left_speed, right_speed)


class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
        self.integral_limit = 100
    
    def compute(self, error):
        self.integral = max(-self.integral_limit, 
                           min(self.integral_limit, self.integral + error))
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
```

### 4.3 编码器里程计

```python
# encoder_odometry.py
import time
import math

class EncoderOdometry:
    """编码器里程计"""
    def __init__(self, wheel_radius=0.032, wheel_base=0.16, ppr=600):
        self.R = wheel_radius
        self.L = wheel_base
        self.ppr = ppr
        
        self.left_count = 0
        self.right_count = 0
        
        self.x = 0.0
        self.y = 0.0
        self.theta = 0.0
        
    def update(self, left_ticks, right_ticks):
        """更新里程计"""
        # 计算轮子行程
        left_dist = (left_ticks / self.ppr) * 2 * math.pi * self.R
        right_dist = (right_ticks / self.ppr) * 2 * math.pi * self.R
        
        # 差速模型
        dist = (left_dist + right_dist) / 2
        dtheta = (right_dist - left_dist) / self.L
        
        self.theta += dtheta
        self.x += dist * math.cos(self.theta)
        self.y += dist * math.sin(self.theta)
        
        return self.x, self.y, self.theta
    
    def get_position(self):
        return self.x, self.y, self.theta
```

### 4.4 任务主流程

```python
# main.py
import cv2
import time

def main():
    camera = cv2.VideoCapture(0)
    camera.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
    camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
    camera.set(cv2.CAP_PROP_FPS, 30)
    
    line_det = LineDetector(model_path=None)  # 纯色彩方案
    drive = DifferentialDrive()
    odom = EncoderOdometry()
    
    frame_count = 0
    gap_count = 0
    max_gap_frames = 15  # 最大断线容忍帧数
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        # 1. 巡线检测
        result = line_det.detect_line(frame)
        offset = result['offset']
        path_type = result['path_type']
        
        # 2. 断线处理
        if path_type == PathType.GAP:
            gap_count += 1
            if gap_count <= max_gap_frames:
                # 惯性直行
                drive.follow_line(0, PathType.STRAIGHT)
            else:
                # 停车
                drive.set_speed(0, 0)
                print("长时间断线，停车等待")
            continue
        else:
            gap_count = 0
        
        # 3. 特殊路径处理
        if path_type == PathType.CROSS:
            print("十字路口，保持直行")
        elif path_type == PathType.LEFT_TURN:
            print("左弯道")
        elif path_type == PathType.RIGHT_TURN:
            print("右弯道")
        elif path_type == PathType.FORK:
            print("分叉路口，默认走右")
        
        # 4. 执行巡线
        drive.follow_line(offset, path_type)
        
        # 5. 里程计更新
        x, y, theta = odom.get_position()
        
        frame_count += 1
        if frame_count % 100 == 0:
            print(f"Frame {frame_count}: offset={offset:.2f}, type={path_type.name}, pos=({x:.2f},{y:.2f})")
        
        time.sleep(0.02)  # 50Hz

if __name__ == '__main__':
    main()
```

## 五、关键技术要点

### 5.1 复杂路径识别
| 路径类型 | 识别特征 | 处理策略 |
|----------|----------|----------|
| 直线 | 偏移稳定 | PID巡线 |
| 弯道 | 偏移趋势增大 | 减速+加大PID |
| 十字路口 | 线宽突增 | 保持直行 |
| T型分叉 | 上部多轮廓 | 默认右转 |
| Y型分叉 | 轮廓分叉 | 选择主道 |
| 断线 | 无有效扫描 | 惯性直行 |

### 5.2 提速策略
- **前瞻控制**：用远处扫描行预测弯道，提前减速
- **变速控制**：直道满速，弯道降速
- **动态PID**：根据曲率调整PID参数
