## 二、视觉方案（Orange Pi 5）

### 2.1 靶标检测与定位

```python
import cv2
import numpy as np
from picamera2 import Picamera2
import math

class TargetDetector:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (1280, 720), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # 靶标参数
        self.target_real_size = 60  # 靶标实际直径 cm
        self.focal_length = 800    # 相机焦距（像素）
        self.camera_height = 120   # 相机安装高度 cm
        self.camera_angle = 15     # 相机俯仰角 度
        
    def detect_target(self):
        """检测靶标并计算位置"""
        frame = self.picam2.capture_array()
        hsv = cv2.cvtColor(frame, cv2.COLOR_RGB2HSV)
        
        # 检测红色靶标
        lower_red1 = np.array([0, 100, 100])
        upper_red1 = np.array([10, 255, 255])
        lower_red2 = np.array([160, 100, 100])
        upper_red2 = np.array([180, 255, 255])
        mask = cv2.inRange(hsv, lower_red1, upper_red1) | \
               cv2.inRange(hsv, lower_red2, upper_red2)
        
        # 形态学处理
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找最大轮廓
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if area > 500:
                # 拟合圆
                (cx, cy), radius = cv2.minEnclosingCircle(largest)
                cx, cy, radius = int(cx), int(cy), int(radius)
                
                # 计算距离
                distance = (self.target_real_size * self.focal_length) / (2 * radius)
                
                # 计算水平偏移角度
                pixel_offset = cx - 640  # 图像中心偏移
                angle_offset = math.degrees(math.atan2(pixel_offset, self.focal_length))
                
                # 计算垂直偏移角度
                pixel_offset_y = cy - 360
                angle_elevation = math.degrees(math.atan2(pixel_offset_y, self.focal_length))
                
                return {
                    'detected': True,
                    'center': (cx, cy),
                    'radius': radius,
                    'distance': distance,
                    'angle_h': angle_offset,      # 水平偏移角
                    'angle_v': angle_elevation,   # 垂直偏移角
                    'area': area
                }
        
        return {'detected': False}
        
    def calculate_trajectory(self, distance, target_height=150):
        """计算弹道参数"""
        g = 9.81  # 重力加速度
        
        # 简化弹道模型：抛物线
        # 需要根据电磁炮特性调整
        launch_angle = math.degrees(math.atan(
            (target_height - self.camera_height + 
             g * distance**2 / (2 * 50**2)) / distance  # 假设初速度50m/s
        ))
        
        # 充电时间（控制发射能量）
        charge_time = distance * 0.02  # 简化计算
        
        return {
            'launch_angle': max(0, min(90, launch_angle)),
            'charge_time': charge_time
        }
```

### 2.2 弹道修正算法

```python
class BallisticComputer:
    def __init__(self):
        self.g = 9.81
        self.v0 = 50.0  # 初速度 m/s（需要实测校准）
        self.shot_history = []
        
    def compute_launch_params(self, distance_m, height_diff_m):
        """计算发射参数"""
        # 解析弹道方程
        # y = x*tan(θ) - g*x²/(2*v0²*cos²(θ))
        
        # 简化：使用迭代法求解
        best_angle = 45.0
        best_error = float('inf')
        
        for angle in range(10, 80):
            theta = math.radians(angle)
            t = distance_m / (self.v0 * math.cos(theta))
            y = self.v0 * math.sin(theta) * t - 0.5 * self.g * t**2
            error = abs(y - height_diff_m)
            
            if error < best_error:
                best_error = error
                best_angle = angle
                
        return best_angle
        
    def adaptive_correction(self, hit_pos, target_pos):
        """基于命中反馈的自适应修正"""
        self.shot_history.append({
            'target': target_pos,
            'hit': hit_pos,
            'error': (hit_pos[0] - target_pos[0], hit_pos[1] - target_pos[1])
        })
        
        if len(self.shot_history) >= 2:
            # 计算平均偏差
            avg_dx = np.mean([s['error'][0] for s in self.shot_history[-3:]])
            avg_dy = np.mean([s['error'][1] for s in self.shot_history[-3:]])
            
            return -avg_dx * 0.5, -avg_dy * 0.5
            
        return 0, 0
```

## 三、控制方案

### 3.1 STM32控制程序

```c
// 电磁炮控制
typedef struct {
    float horizontal_angle;  // 水平角
    float vertical_angle;    // 俯仰角
    uint16_t charge_time_ms; // 充电时间
    uint8_t fire;            // 发射指令
} CannonCommand_t;

// 步进电机控制
void Stepper_MoveAngle(StepperMotor_t *motor, float angle) {
    int32_t steps = (int32_t)(angle / 360.0 * STEPS_PER_REV * GEAR_RATIO);
    motor->target_steps = steps;
    motor->direction = (steps > 0) ? 1 : 0;
}

// 电磁线圈控制
void Cannon_Fire(uint16_t charge_time_ms) {
    // 1. 电容充电
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_SET);  // 充电开关
    HAL_Delay(charge_time_ms);
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_0, GPIO_PIN_RESET);
    
    // 2. 放电发射
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_1, GPIO_PIN_SET);  // 放电开关
    HAL_Delay(10);  // 放电时间
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_1, GPIO_PIN_RESET);
    
    // 3. 安全间隔
    HAL_Delay(500);
}

// 主循环
void Cannon_MainLoop(void) {
    CannonCommand_t cmd;
    
    while (1) {
        // 接收视觉系统指令
        if (UART_ReceiveCommand(&cmd)) {
            // 水平旋转
            Stepper_MoveAngle(&stepper_h, cmd.horizontal_angle);
            // 俯仰调整
            Stepper_MoveAngle(&stepper_v, cmd.vertical_angle);
            
            // 等待到位
            while (!Stepper_IsReady(&stepper_h) || !Stepper_IsReady(&stepper_v)) {
                HAL_Delay(10);
            }
            
            // 发射
            if (cmd.fire) {
                Cannon_Fire(cmd.charge_time_ms);
            }
        }
    }
}
```

### 3.2 角度控制PID

```python
class AngleController:
    def __init__(self):
        self.ser = serial.Serial('/dev/ttyS5', 115200)
        self.current_h_angle = 0
        self.current_v_angle = 0
        
    def aim_at(self, angle_h, angle_v):
        """瞄准目标角度"""
        # 计算需要转动的角度
        delta_h = angle_h - self.current_h_angle
        delta_v = angle_v - self.current_v_angle
        
        # 发送控制指令
        cmd = bytes([
            0xAA,
            int(delta_h * 10) & 0xFF,  # 水平角 x10
            int(delta_v * 10) & 0xFF,  # 俯仰角 x10
            0x00,  # 不发射
            0x55
        ])
        self.ser.write(cmd)
        
        self.current_h_angle = angle_h
        self.current_v_angle = angle_v
        
    def fire(self, charge_time_ms):
        """发射"""
        cmd = bytes([
            0xAA,
            0x00,
            0x00,
            charge_time_ms & 0xFF,
            0x55
        ])
        self.ser.write(cmd)
```

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2019年H题 - 模拟电磁曲射炮 主程序
"""
import cv2
import numpy as np
import serial
import time
import math

class ElectromagneticCannon:
    def __init__(self):
        self.detector = TargetDetector()
        self.ballistic = BallisticComputer()
        self.controller = AngleController()
        
        self.state = 'AIM'  # AIM, FIRE, RELOAD
        self.shot_count = 0
        self.max_shots = 5
        
    def run(self):
        print("电磁曲射炮系统启动...")
        
        while self.shot_count < self.max_shots:
            if self.state == 'AIM':
                # 检测靶标
                target = self.detector.detect_target()
                
                if target['detected']:
                    distance = target['distance'] / 100  # cm -> m
                    height_diff = 0  # 假设靶标与炮同高
                    
                    # 计算发射参数
                    launch_angle = self.ballistic.compute_launch_params(
                        distance, height_diff
                    )
                    
                    # 瞄准
                    self.controller.aim_at(
                        target['angle_h'],
                        launch_angle
                    )
                    
                    print(f"瞄准: 距离={distance:.1f}m, "
                          f"水平角={target['angle_h']:.1f}°, "
                          f"俯仰角={launch_angle:.1f}°")
                    
                    # 等待稳定
                    time.sleep(0.5)
                    
                    # 计算充电时间
                    charge_time = int(distance * 20)  # ms
                    
                    self.state = 'FIRE'
                    
            elif self.state == 'FIRE':
                print(f"发射! 充电时间={charge_time}ms")
                self.controller.fire(charge_time)
                self.shot_count += 1
                
                # 等待并检测命中
                time.sleep(1.0)
                
                # 自适应修正
                hit_target = self.detector.detect_target()
                if hit_target['detected']:
                    dx, dy = self.ballistic.adaptive_correction(
                        hit_target['center'],
                        target['center']
                    )
                    print(f"弹道修正: dx={dx:.1f}, dy={dy:.1f}")
                
                self.state = 'AIM'
                time.sleep(0.5)
                
        print(f"射击完成，共发射 {self.shot_count} 发")

if __name__ == '__main__':
    cannon = ElectromagneticCannon()
    cannon.run()
```

## 七、调试要点

1. **弹道校准**：不同距离射击，建立距离-角度-充电时间对应表
2. **电磁线圈**：优化线圈匝数和电容容量，提高初速度
3. **视觉标定**：相机内参精确标定，提高距离测量精度
4. **安全防护**：高压电路隔离，电容放电安全
5. **稳定时间**：步进电机到位后等待振动衰减再发射
