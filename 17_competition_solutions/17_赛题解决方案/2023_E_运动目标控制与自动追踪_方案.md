```python
import cv2
import numpy as np
from picamera2 import Picamera2
import threading

class DualCameraSystem:
    def __init__(self):
        # 前方相机
        self.cam_front = Picamera2(0)
        config_front = self.cam_front.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.cam_front.configure(config_front)
        
        # 侧方相机
        self.cam_side = Picamera2(1)
        config_side = self.cam_side.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.cam_side.configure(config_side)
        
        self.cam_front.start()
        self.cam_side.start()
        
        # 相机内参（标定得到）
        self.K_front = np.array([[650, 0, 320], [0, 650, 240], [0, 0, 1]], dtype=np.float32)
        self.K_side = np.array([[650, 0, 320], [0, 650, 240], [0, 0, 1]], dtype=np.float32)
        
        # 相机外参（相对位姿）
        self.R = np.eye(3)  # 旋转矩阵
        self.T = np.array([[30], [0], [0]], dtype=np.float32)  # 平移向量 cm
        
    def capture_sync(self):
        """同步采集双相机图像"""
        frame_front = self.cam_front.capture_array()
        frame_side = self.cam_side.capture_array()
        return frame_front, frame_side
```

### 2.2 红绿激光光斑检测

```python
class LaserDetector:
    def __init__(self):
        # 红色激光HSV范围
        self.red_lower1 = np.array([0, 150, 150])
        self.red_upper1 = np.array([10, 255, 255])
        self.red_lower2 = np.array([160, 150, 150])
        self.red_upper2 = np.array([180, 255, 255])
        
        # 绿色激光HSV范围
        self.green_lower = np.array([35, 150, 150])
        self.green_upper = np.array([85, 255, 255])
        
    def detect_laser_spot(self, frame, color='red'):
        """检测激光光斑"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_RGB2HSV)
        
        if color == 'red':
            mask = cv2.inRange(hsv, self.red_lower1, self.red_upper1) | \
                   cv2.inRange(hsv, self.red_lower2, self.red_upper2)
        elif color == 'green':
            mask = cv2.inRange(hsv, self.green_lower, self.green_upper)
        
        # 形态学处理
        kernel = np.ones((3, 3), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                       cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找最亮的光斑
            best_cnt = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(best_cnt)
            
            if 10 < area < 5000:  # 光斑大小范围
                M = cv2.moments(best_cnt)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    return {
                        'detected': True,
                        'center': (cx, cy),
                        'area': area,
                        'color': color
                    }
        
        return {'detected': False}
        
    def detect_both_lasers(self, frame):
        """同时检测红绿激光"""
        red = self.detect_laser_spot(frame, 'red')
        green = self.detect_laser_spot(frame, 'green')
        return red, green
```

### 2.3 运动目标检测与追踪

```python
class TargetTracker:
    def __init__(self):
        # 目标检测参数
        self.target_color = 'white'  # 目标板颜色
        self.target_size = 10  # cm
        
        # 卡尔曼滤波器
        self.kf = cv2.KalmanFilter(4, 2)  # 状态: [x, y, vx, vy], 观测: [x, y]
        self.kf.measurementMatrix = np.array([[1,0,0,0], [0,1,0,0]], np.float32)
        self.kf.transitionMatrix = np.array([[1,0,1,0], [0,1,0,1], 
                                              [0,0,1,0], [0,0,0,1]], np.float32)
        self.kf.processNoiseCov = np.eye(4, dtype=np.float32) * 0.03
        self.kf.measurementNoiseCov = np.eye(2, dtype=np.float32) * 0.1
        
        self.initialized = False
        
    def detect_target(self, frame):
        """检测运动目标"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_RGB2HSV)
        
        # 检测白色/浅色目标
        lower = np.array([0, 0, 200])
        upper = np.array([180, 30, 255])
        mask = cv2.inRange(hsv, lower, upper)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                       cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if 500 < area < 50000:
                x, y, w, h = cv2.boundingRect(largest)
                cx = x + w // 2
                cy = y + h // 2
                return True, (cx, cy), area
        
        return False, (0, 0), 0
        
    def update(self, measurement):
        """卡尔曼滤波更新"""
        if not self.initialized:
            self.kf.statePre = np.array([[measurement[0]], [measurement[1]], 
                                          [0], [0]], np.float32)
            self.initialized = True
            
        # 预测
        prediction = self.kf.predict()
        
        # 更新
        self.kf.correct(np.array([[measurement[0]], [measurement[1]]], np.float32))
        
        # 返回预测位置
        state = self.kf.statePost
        return int(state[0]), int(state[1])
        
    def predict_next(self):
        """预测下一帧位置"""
        state = self.kf.statePost
        vx, vy = state[2], state[3]
        x, y = state[0], state[1]
        return int(x + vx), int(y + vy)
```

### 2.4 双相机三维定位

```python
class Stereo3DLocator:
    def __init__(self, camera_system):
        self.cam_sys = camera_system
        
    def triangulate(self, point_front, point_side):
        """三角测量计算3D坐标"""
        # 归一化坐标
        p1 = np.array([point_front[0], point_front[1], 1.0])
        p2 = np.array([point_side[0], point_side[1], 1.0])
        
        # 投影矩阵
        P1 = self.cam_sys.K_front @ np.hstack([np.eye(3), np.zeros((3, 1))])
        P2 = self.cam_sys.K_side @ np.hstack([self.cam_sys.R, self.cam_sys.T])
        
        # DLT三角测量
        A = np.array([
            p1[0] * P1[2] - P1[0],
            p1[1] * P1[2] - P1[1],
            p2[0] * P2[2] - P2[0],
            p2[1] * P2[2] - P2[1]
        ])
        
        _, _, Vt = np.linalg.svd(A)
        X = Vt[-1]
        X = X / X[3]
        
        return X[:3]  # [x, y, z] in cm
```

## 三、控制方案

### 3.1 云台PID控制

```python
class GimbalController:
    def __init__(self):
        self.ser = serial.Serial('/dev/ttyS5', 115200)
        
        # X轴PID（水平）
        self.x_pid = PIDController(kp=0.3, ki=0.005, kd=0.15)
        # Y轴PID（垂直）
        self.y_pid = PIDController(kp=0.3, ki=0.005, kd=0.15)
        
        # 云台当前角度
        self.angle_x = 90  # 初始位置
        self.angle_y = 90
        
        # 图像中心
        self.center_x = 320
        self.center_y = 240
        
    def track_target(self, target_x, target_y):
        """追踪目标"""
        # 计算像素误差
        error_x = target_x - self.center_x
        error_y = target_y - self.center_y
        
        # PID计算角度调整
        delta_x = self.x_pid.compute(0, error_x)
        delta_y = self.y_pid.compute(0, error_y)
        
        # 更新角度
        self.angle_x = np.clip(self.angle_x + delta_x, 0, 180)
        self.angle_y = np.clip(self.angle_y + delta_y, 0, 180)
        
        # 发送命令
        self.send_gimbal_command(int(self.angle_x), int(self.angle_y))
        
        # 返回误差
        return error_x, error_y
        
    def aim_laser(self, target_x, target_y, laser_color='red'):
        """瞄准并发射激光"""
        # 先瞄准
        error_x, error_y = self.track_target(target_x, target_y)
        
        # 判断是否足够接近中心
        if abs(error_x) < 10 and abs(error_y) < 10:
            self.send_laser_command(laser_color, True)
            return True
        else:
            self.send_laser_command(laser_color, False)
            return False
            
    def send_gimbal_command(self, angle_x, angle_y):
        """发送云台控制命令"""
        cmd = bytes([
            0xAA,
            0x01,  # 云台控制
            angle_x,
            angle_y,
            0x55
        ])
        self.ser.write(cmd)
        
    def send_laser_command(self, color, on):
        """发送激光控制命令"""
        cmd = bytes([
            0xAA,
            0x02,  # 激光控制
            0x01 if color == 'red' else 0x02,
            0x01 if on else 0x00,
            0x55
        ])
        self.ser.write(cmd)
```

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2023年E题 - 运动目标控制与自动追踪 主程序
"""
import cv2
import numpy as np
import serial
import time
import threading

class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
        
    def compute(self, target, current):
        error = target - current
        self.integral += error
        self.integral = np.clip(self.integral, -100, 100)
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative

class MotionTargetTracker:
    def __init__(self):
        self.dual_cam = DualCameraSystem()
        self.laser_detector = LaserDetector()
        self.target_tracker = TargetTracker()
        self.gimbal = GimbalController()
        self.stereo = Stereo3DLocator(self.dual_cam)
        
        self.mode = 'TRACK_RED'  # TRACK_RED, TRACK_GREEN, DUAL_TRACK
        self.running = True
        
    def track_red_laser(self):
        """红激光追踪目标"""
        frame_front, frame_side = self.dual_cam.capture_sync()
        
        # 检测目标
        detected, target_pos, area = self.target_tracker.detect_target(frame_front)
        
        if detected:
            # 卡尔曼滤波平滑
            smoothed_pos = self.target_tracker.update(target_pos)
            
            # 云台追踪并发射红激光
            hit = self.gimbal.aim_laser(smoothed_pos[0], smoothed_pos[1], 'red')
            
            if hit:
                print(f"✓ 红激光命中目标 ({smoothed_pos[0]}, {smoothed_pos[1]})")
            else:
                print(f"→ 追踪中 偏移:({smoothed_pos[0]-320}, {smoothed_pos[1]-240})")
                
            return smoothed_pos
        else:
            # 使用预测位置
            predicted = self.target_tracker.predict_next()
            self.gimbal.track_target(predicted[0], predicted[1])
            print("目标丢失，使用预测位置...")
            return None
            
    def track_green_laser(self):
        """绿激光追踪目标"""
        frame_front, frame_side = self.dual_cam.capture_sync()
        
        # 检测目标
        detected, target_pos, area = self.target_tracker.detect_target(frame_front)
        
        if detected:
            smoothed_pos = self.target_tracker.update(target_pos)
            hit = self.gimbal.aim_laser(smoothed_pos[0], smoothed_pos[1], 'green')
            
            if hit:
                print(f"✓ 绿激光命中目标 ({smoothed_pos[0]}, {smoothed_pos[1]})")
            
            return smoothed_pos
        return None
        
    def dual_laser_tracking(self):
        """双激光同时追踪"""
        frame_front, frame_side = self.dual_cam.capture_sync()
        
        # 检测红绿激光位置（在目标板上）
        red_laser = self.laser_detector.detect_laser_spot(frame_front, 'red')
        green_laser = self.laser_detector.detect_laser_spot(frame_front, 'green')
        
        # 检测目标
        detected, target_pos, area = self.target_tracker.detect_target(frame_front)
        
        if detected:
            smoothed_pos = self.target_tracker.update(target_pos)
            
            # 红激光追踪目标
            self.gimbal.aim_laser(smoothed_pos[0], smoothed_pos[1], 'red')
            
            # 绿激光追踪红激光的偏移位置
            if red_laser['detected']:
                offset_x = red_laser['center'][0] - smoothed_pos[0]
                offset_y = red_laser['center'][1] - smoothed_pos[1]
                green_target_x = smoothed_pos[0] - offset_x
                green_target_y = smoothed_pos[1] - offset_y
                self.gimbal.aim_laser(green_target_x, green_target_y, 'green')
            
            # 3D定位
            if red_laser['detected'] and green_laser['detected']:
                pos_3d = self.stereo.triangulate(
                    red_laser['center'], green_laser['center']
                )
                print(f"3D位置: ({pos_3d[0]:.1f}, {pos_3d[1]:.1f}, {pos_3d[2]:.1f})cm")
                
    def run(self):
        print("运动目标追踪系统启动...")
        print(f"模式: {self.mode}")
        
        while self.running:
            if self.mode == 'TRACK_RED':
                self.track_red_laser()
            elif self.mode == 'TRACK_GREEN':
                self.track_green_laser()
            elif self.mode == 'DUAL_TRACK':
                self.dual_laser_tracking()
                
            time.sleep(0.02)  # 50Hz
            
    def set_mode(self, mode):
        """切换模式"""
        self.mode = mode
        # 关闭所有激光
        self.gimbal.send_laser_command('red', False)
        self.gimbal.send_laser_command('green', False)
        print(f"切换到模式: {mode}")

if __name__ == '__main__':
    tracker = MotionTargetTracker()
    
    # 演示：依次执行各功能
    print("=== 基本要求1: 红激光追踪 ===")
    tracker.set_mode('TRACK_RED')
    tracker.run()
```

## 七、调试要点

1. **相机标定**：双相机内外参精确标定，提高3D精度
2. **激光对准**：激光发射方向与相机光轴平行
3. **PID整定**：追踪PID需快速响应但不振荡
4. **颜色过滤**：不同光照下调整HSV阈值
5. **预测参数**：调整卡尔曼过程噪声，平衡响应速度和平滑度
6. **帧率优化**：适当降低分辨率提高处理速度
