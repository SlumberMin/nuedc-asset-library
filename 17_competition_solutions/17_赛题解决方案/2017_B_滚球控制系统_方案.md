### 2.2 ArUco标记追踪
```python
import cv2
import numpy as np
from cv2 import aruco

class BallTracker:
    def __init__(self):
        # ArUco字典（球上贴标记，或直接用颜色追踪）
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict)
        
        # 像素→物理坐标映射矩阵（标定后填入）
        # homography矩阵：像素坐标 → 平台坐标(mm)
        self.H = None  # 3×3单应性矩阵
        
        # HSV颜色追踪（备选方案，追踪红色小球）
        self.lower_hsv = np.array([0, 100, 100])
        self.upper_hsv = np.array([10, 255, 255])
    
    def calibrate(self, pixel_pts, world_pts):
        """
        标定：4个角点的像素坐标 → 实际平台坐标
        pixel_pts: [[x1,y1],[x2,y2],[x3,y3],[x4,y4]] 像素
        world_pts: [[X1,Y1],[X2,Y2],[X3,Y3],[X4,Y4]] mm
        """
        self.H, _ = cv2.findHomography(np.float32(pixel_pts), np.float32(world_pts))
    
    def detect_ball(self, frame):
        """返回小球在平台上的物理坐标(mm)"""
        # 方法1：颜色追踪
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.lower_hsv, self.upper_hsv)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((5,5)))
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if not contours:
            return None
        
        c = max(contours, key=cv2.contourArea)
        ((px, py), radius) = cv2.minEnclosingCircle(c)
        
        if radius < 5 or radius > 50:  # 滤除噪声
            return None
        
        # 像素坐标 → 物理坐标
        pixel = np.array([[[px, py]]], dtype=np.float32)
        world = cv2.perspectiveTransform(pixel, self.H)
        x_mm, y_mm = world[0][0]
        
        return (x_mm, y_mm, radius)

    def get_ball_velocity(self, pos_curr, pos_prev, dt):
        """计算小球速度(mm/s)用于前馈"""
        vx = (pos_curr[0] - pos_prev[0]) / dt
        vy = (pos_curr[1] - pos_prev[1]) / dt
        return (vx, vy)
```

### 3.2 PID控制器
```python
import time

class DualAxisPID:
    def __init__(self):
        # X轴PID参数
        self.pid_x = PID(Kp=0.8, Ki=0.01, Kd=0.15, setpoint=0)
        # Y轴PID参数
        self.pid_y = PID(Kp=0.8, Ki=0.01, Kd=0.15, setpoint=0)
        
        # 前馈系数（根据小球速度补偿）
        self.ff_gain = 0.1
        
        # 输出限幅（舵机角度范围 ±15°）
        self.output_limit = 15.0
    
    def compute(self, target, current, velocity, dt):
        """
        target: (x_mm, y_mm) 目标位置
        current: (x_mm, y_mm) 当前位置
        velocity: (vx, vy) mm/s
        dt: 时间间隔
        返回: (servo_angle_x, servo_angle_y)
        """
        # PID计算
        err_x = target[0] - current[0]
        err_y = target[1] - current[1]
        
        out_x = self.pid_x.compute(err_x, dt)
        out_y = self.pid_y.compute(err_y, dt)
        
        # 前馈补偿（基于速度预测）
        ff_x = self.ff_gain * velocity[0]
        ff_y = self.ff_gain * velocity[1]
        
        # 合成输出
        angle_x = np.clip(out_x + ff_x, -self.output_limit, self.output_limit)
        angle_y = np.clip(out_y + ff_y, -self.output_limit, self.output_limit)
        
        return angle_x, angle_y


class PID:
    def __init__(self, Kp, Ki, Kd, setpoint=0):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.setpoint = setpoint
        self.integral = 0
        self.prev_error = 0
        self.integral_limit = 100
    
    def compute(self, error, dt):
        self.integral += error * dt
        self.integral = np.clip(self.integral, -self.integral_limit, self.integral_limit)
        derivative = (error - self.prev_error) / dt if dt > 0 else 0
        self.prev_error = error
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative
```

### 3.3 轨迹跟踪
```python
class TrajectoryFollower:
    def __init__(self):
        self.waypoints = []  # [(x1,y1,t1), (x2,y2,t2), ...]
        self.current_idx = 0
    
    def set_circle_trajectory(self, center, radius, num_points=36):
        """圆形轨迹"""
        self.waypoints = []
        for i in range(num_points):
            angle = 2 * np.pi * i / num_points
            x = center[0] + radius * np.cos(angle)
            y = center[1] + radius * np.sin(angle)
            self.waypoints.append((x, y))
    
    def set_line_trajectory(self, start, end, num_points=20):
        """直线轨迹"""
        self.waypoints = []
        for i in range(num_points):
            t = i / (num_points - 1)
            x = start[0] + t * (end[0] - start[0])
            y = start[1] + t * (end[1] - start[1])
            self.waypoints.append((x, y))
    
    def get_current_target(self, ball_pos, reach_threshold=5):
        """返回当前目标点，到达后切换下一个"""
        target = self.waypoints[self.current_idx]
        dist = np.sqrt((ball_pos[0]-target[0])**2 + (ball_pos[1]-target[1])**2)
        if dist < reach_threshold:
            self.current_idx = (self.current_idx + 1) % len(self.waypoints)
        return self.waypoints[self.current_idx]
```

## 七、调试要点
1. **先标定**：确保视觉坐标系准确
2. **分轴调试**：先单独调X轴PID，再调Y轴
3. **前馈作用**：小球高速运动时前馈明显改善跟踪性能
4. **积分抗饱和**：设置积分限幅防止超调
5. **低通滤波**：视觉数据加一阶低通滤除抖动
