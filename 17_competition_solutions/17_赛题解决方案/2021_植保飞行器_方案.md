# 2021 植保飞行器 - Orange Pi 5 解决方案

## 视觉识别系统

### 作物区域检测
```python
import cv2
import numpy as np

class CropDetector:
    """下视相机识别作物区域标记"""
    def __init__(self):
        # 相机标定(下视安装)
        self.camera_mtx = np.load('cam_mtx.npy')
        self.altitude = 2.0  # 当前飞行高度(m)

    def detect_crop_rows(self, frame):
        """检测作物行标记(彩色胶带/LED标记)"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

        # 绿色作物行
        green_mask = cv2.inRange(hsv, (35,60,60), (85,255,255))

        # 形态学处理
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (15,5))
        green_mask = cv2.morphologyEx(green_mask, cv2.MORPH_CLOSE, kernel)
        green_mask = cv2.morphologyEx(green_mask, cv2.MORPH_OPEN, kernel)

        # 提取作物行中心线
        contours, _ = cv2.findContours(green_mask, cv2.RETR_EXTERNAL,
                                        cv2.CHAIN_APPROX_SIMPLE)

        rows = []
        for c in contours:
            area = cv2.contourArea(c)
            if area < 1000:
                continue
            rect = cv2.minAreaRect(c)
            center, (w, h), angle = rect
            # 长宽比大的才是行
            if max(w,h) / (min(w,h)+1) > 3:
                rows.append({
                    'center': center,
                    'width': min(w, h),
                    'length': max(w, h),
                    'angle': angle
                })

        return rows

    def detect_spray_target(self, frame):
        """检测喷洒目标区域(红色标记=需要喷洒)"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        red_mask = cv2.inRange(hsv, (0,100,100), (10,255,255))
        red_mask |= cv2.inRange(hsv, (160,100,100), (180,255,255))

        contours, _ = cv2.findContours(red_mask, cv2.RETR_EXTERNAL,
                                        cv2.CHAIN_APPROX_SIMPLE)
        targets = []
        for c in contours:
            if cv2.contourArea(c) > 500:
                M = cv2.moments(c)
                if M['m00'] > 0:
                    cx = int(M['m10']/M['m00'])
                    cy = int(M['m01']/M['m00'])
                    targets.append((cx, cy, cv2.contourArea(c)))

        return targets

    def pixel_to_world(self, px, py, frame_shape):
        """像素坐标转地面世界坐标"""
        h, w = frame_shape[:2]
        # 下视相机简化映射
        fx = self.camera_mtx[0, 0]
        fy = self.camera_mtx[1, 1]
        cx = self.camera_mtx[0, 2]
        cy = self.camera_mtx[1, 2]

        # 地面坐标(相对相机中心正下方)
        x_world = (px - cx) * self.altitude / fx
        y_world = (py - cy) * self.altitude / fy
        return x_world, y_world
```

## 飞行控制

```python
import serial
import struct

class SprayFlightController:
    """植保飞行控制"""
    def __init__(self, port='/dev/ttyS2', baud=115200):
        self.ser = serial.Serial(port, baud, timeout=0.1)
        self.current_pos = [0, 0, 2.0]  # x, y, z
        self.target_alt = 2.0
        self.spray_active = False

    def send_command(self, vx, vy, vz, yaw_rate):
        """发送速度指令"""
        msg = struct.pack('<ffff', vx, vy, vz, yaw_rate)
        self.ser.write(b'\xAA' + msg + bytes([sum(msg) & 0xFF]))

    def goto_position(self, target_x, target_y, target_z=2.0):
        """飞向目标位置"""
        dx = target_x - self.current_pos[0]
        dy = target_y - self.current_pos[1]
        dz = target_z - self.current_pos[2]

        dist = np.sqrt(dx**2 + dy**2)
        speed = min(0.5, dist * 0.8)  # 接近时减速

        if dist > 0.1:
            vx = dx / dist * speed
            vy = dy / dist * speed
        else:
            vx, vy = 0, 0

        vz = np.clip(dz * 0.5, -0.3, 0.3)
        self.send_command(vx, vy, vz, 0)
        return dist < 0.1  # 到达

    def hover(self):
        """悬停"""
        self.send_command(0, 0, 0, 0)
```

## 喷洒控制

```python
import gpiod

class Sprayer:
    """喷洒系统控制"""
    def __init__(self):
        self.chip = gpiod.Chip('0')
        self.pump_pin = self.chip.get_line(16)   # PA16: 水泵继电器
        self.valve_pin = self.chip.get_line(17)  # PA17: 电磁阀
        self.pump_pin.request(consumer='spray', type=gpiod.LINE_REQ_DIR_OUT)
        self.valve_pin.request(consumer='spray', type=gpiod.LINE_REQ_DIR_OUT)

        self.flow_rate = 0.5  # ml/s
        self.spray_width = 1.0  # 喷幅(m)
        self.total_sprayed = 0  # ml

    def start_spray(self):
        self.pump_pin.set_value(1)
        self.valve_pin.set_value(1)

    def stop_spray(self):
        self.pump_pin.set_value(0)
        self.valve_pin.set_value(0)

    def spray_area(self, area_m2, dose_ml_per_m2=10):
        """计算区域喷洒量"""
        return area_m2 * dose_ml_per_m2
```

## 任务规划

```python
class SprayMissionPlanner:
    """植保任务规划 - 往返航线"""
    def __init__(self, field_length=5.0, field_width=3.0, swath=1.0):
        self.field_l = field_length
        self.field_w = field_width
        self.swath = swath

    def generate_waypoints(self, origin=(0,0)):
        """生成往复式喷洒航线"""
        waypoints = []
        num_strips = int(np.ceil(self.field_w / self.swath))

        for i in range(num_strips):
            y = origin[1] + i * self.swath + self.swath/2
            if i % 2 == 0:
                # 去程
                waypoints.append((origin[0], y))
                waypoints.append((origin[0] + self.field_l, y))
            else:
                # 回程
                waypoints.append((origin[0] + self.field_l, y))
                waypoints.append((origin[0], y))

        return waypoints

    def plan_spray_zones(self, targets, altitude=2.0):
        """根据视觉识别的目标生成喷洒区域"""
        zones = []
        for (tx, ty, area) in targets:
            zones.append({
                'center': (tx, ty),
                'radius': np.sqrt(area / np.pi) * 0.01,  # 像素→米
                'dose': 'FULL'
            })
        return zones
```

## Orange Pi 5 接口分配

| 接口 | 功能 | 说明 |
|------|------|------|
| MIPI CSI-0 | 下视摄像头 | OV5647识别作物 |
| MIPI CSI-1 | 前视摄像头 | 避障(可选) |
| UART2 | 飞控通信 | MAVLink |
| GPIO PA16 | 水泵继电器 | 喷洒开关 |
| GPIO PA17 | 电磁阀 | 喷头开关 |
| I2C3 | 气压计 | 高度辅助 |
| SPI0 | 流量传感器 | 喷洒计量 |

## 关键指标

| 指标 | 目标值 |
|------|--------|
| 定位精度 | ±10cm |
| 飞行高度 | 1.5~3m |
| 航线精度 | ±20cm |
| 喷洒覆盖 | >95% |
| 续航时间 | >10min |
| 避障响应 | <0.5s |

## 调试流程

1. **地面标定**：相机-喷头相对位置标定
2. **悬停测试**：定点悬停稳定性验证
3. **航线飞行**：空飞验证航线精度
4. **喷洒测试**：地面标记覆盖率验证
5. **完整任务**：端到端全流程测试
