## 一、题目分析

### 1.1 核心要求
- 设计帆板控制系统，通过风扇控制帆板角度
- 帆板能在0-90°范围内任意角度保持稳定
- 系统能快速响应角度变化并调整风扇风力
- 具有良好的稳定性和抗干扰能力

### 1.2 技术难点
- 帆板角度的精确检测
- 风力与角度的非线性关系
- 系统响应速度与稳定性平衡
- 外界干扰（如自然风）的抑制

- **安装**: 固定在支架上，帆板可自由旋转

### 4.2 MPU6050驱动模块

```python
#!/usr/bin/env python3
"""
MPU6050驱动模块 - 读取加速度和陀螺仪数据
用于帆板角度检测
"""

import smbus2
import time
import numpy as np

class MPU6050:
    # MPU6050寄存器地址
    PWR_MGMT_1 = 0x6B
    SMPLRT_DIV = 0x19
    CONFIG = 0x1A
    GYRO_CONFIG = 0x1B
    ACCEL_CONFIG = 0x1C
    ACCEL_XOUT_H = 0x3B
    ACCEL_YOUT_H = 0x3D
    ACCEL_ZOUT_H = 0x3F
    GYRO_XOUT_H = 0x43
    GYRO_YOUT_H = 0x45
    GYRO_ZOUT_H = 0x47
    
    def __init__(self, bus_id=1, address=0x68):
        self.bus = smbus2.SMBus(bus_id)
        self.address = address
        
        # 初始化MPU6050
        self.bus.write_byte_data(self.address, self.PWR_MGMT_1, 0x00)  # 唤醒
        time.sleep(0.1)
        
        # 设置采样率 1kHz
        self.bus.write_byte_data(self.address, self.SMPLRT_DIV, 0x07)
        
        # 设置低通滤波 44Hz
        self.bus.write_byte_data(self.address, self.CONFIG, 0x03)
        
        # 设置陀螺仪量程 ±250°/s
        self.bus.write_byte_data(self.address, self.GYRO_CONFIG, 0x00)
        
        # 设置加速度计量程 ±2g
        self.bus.write_byte_data(self.address, self.ACCEL_CONFIG, 0x00)
        
        # 校准偏移量
        self.accel_offset = [0, 0, 0]
        self.gyro_offset = [0, 0, 0]
        self.calibrate()
        
    def read_raw_data(self, addr):
        """读取原始数据"""
        high = self.bus.read_byte_data(self.address, addr)
        low = self.bus.read_byte_data(self.address, addr + 1)
        
        value = (high << 8) + low
        if value > 32768:
            value = value - 65536
        return value
        
    def read_accel(self):
        """读取加速度值（单位：g）"""
        x = self.read_raw_data(self.ACCEL_XOUT_H) / 16384.0
        y = self.read_raw_data(self.ACCEL_YOUT_H) / 16384.0
        z = self.read_raw_data(self.ACCEL_ZOUT_H) / 16384.0
        
        # 应用校准偏移
        x -= self.accel_offset[0]
        y -= self.accel_offset[1]
        z -= self.accel_offset[2]
        
        return np.array([x, y, z])
        
    def read_gyro(self):
        """读取陀螺仪值（单位：°/s）"""
        x = self.read_raw_data(self.GYRO_XOUT_H) / 131.0
        y = self.read_raw_data(self.GYRO_YOUT_H) / 131.0
        z = self.read_raw_data(self.GYRO_ZOUT_H) / 131.0
        
        # 应用校准偏移
        x -= self.gyro_offset[0]
        y -= self.gyro_offset[1]
        z -= self.gyro_offset[2]
        
        return np.array([x, y, z])
        
    def calibrate(self, samples=1000):
        """校准传感器"""
        print("校准MPU6050，请保持帆板水平静止...")
        
        accel_sum = np.zeros(3)
        gyro_sum = np.zeros(3)
        
        for _ in range(samples):
            accel_sum += self.read_raw_data_vec(self.ACCEL_XOUT_H)
            gyro_sum += self.read_raw_data_vec(self.GYRO_XOUT_H)
            time.sleep(0.001)
            
        # 计算平均偏移
        accel_avg = accel_sum / samples / 16384.0
        gyro_avg = gyro_sum / samples / 131.0
        
        # 加速度Z轴应该为1g
        self.accel_offset = [accel_avg[0], accel_avg[1], accel_avg[2] - 1.0]
        self.gyro_offset = gyro_avg
        
        print(f"校准完成: accel_offset={self.accel_offset}, gyro_offset={self.gyro_offset}")
        
    def read_raw_data_vec(self, base_addr):
        """读取三轴原始数据向量"""
        data = []
        for i in range(3):
            high = self.bus.read_byte_data(self.address, base_addr + i * 2)
            low = self.bus.read_byte_data(self.address, base_addr + i * 2 + 1)
            value = (high << 8) + low
            if value > 32768:
                value = value - 65536
            data.append(value)
        return np.array(data)
        
    def get_angle(self):
        """
        计算帆板倾斜角（使用互补滤波）
        返回: (roll, pitch) 角度（度）
        """
        accel = self.read_accel()
        gyro = self.read_gyro()
        
        # 从加速度计计算角度
        accel_roll = np.arctan2(accel[1], accel[2]) * 180 / np.pi
        accel_pitch = np.arctan2(-accel[0], np.sqrt(accel[1]**2 + accel[2]**2)) * 180 / np.pi
        
        # 互补滤波融合
        alpha = 0.98
        dt = 0.01  # 10ms采样周期
        
        if not hasattr(self, 'roll'):
            self.roll = accel_roll
            self.pitch = accel_pitch
        else:
            self.roll = alpha * (self.roll + gyro[0] * dt) + (1 - alpha) * accel_roll
            self.pitch = alpha * (self.pitch + gyro[1] * dt) + (1 - alpha) * accel_pitch
            
        return self.roll, self.pitch
        
    def get_sail_angle(self):
        """
        获取帆板角度（相对于水平面）
        假设帆板绕X轴旋转
        返回: 角度（度），正值表示向风扇方向倾斜
        """
        _, pitch = self.get_angle()
        return pitch
```

### 4.3 卡尔曼滤波器

```python
#!/usr/bin/env python3
"""
卡尔曼滤波器 - 用于角度估计和滤波
"""

import numpy as np

class KalmanFilter:
    def __init__(self, dt=0.01, process_noise=0.01, measurement_noise=0.1):
        """
        dt: 采样时间
        process_noise: 过程噪声
        measurement_noise: 测量噪声
        """
        self.dt = dt
        
        # 状态向量 [角度, 角速度]
        self.x = np.array([[0], [0]])
        
        # 状态转移矩阵
        self.F = np.array([[1, dt],
                           [0, 1]])
        
        # 测量矩阵
        self.H = np.array([[1, 0]])
        
        # 过程噪声协方差
        self.Q = np.array([[dt**4/4, dt**3/2],
                           [dt**3/2, dt**2]]) * process_noise
        
        # 测量噪声协方差
        self.R = np.array([[measurement_noise]])
        
        # 误差协方差矩阵
        self.P = np.eye(2)
        
    def predict(self):
        """预测步骤"""
        # 状态预测
        self.x = self.F @ self.x
        
        # 协方差预测
        self.P = self.F @ self.P @ self.F.T + self.Q
        
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def update(self, measurement):
        """更新步骤"""
        # 卡尔曼增益
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        # 状态更新
        y = measurement - self.H @ self.x  # 测量残差
        self.x = self.x + K @ y
        
        # 协方差更新
        I = np.eye(2)
        self.P = (I - K @ self.H) @ self.P
        
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def get_state(self):
        """获取当前状态"""
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def reset(self):
        """重置滤波器"""
        self.x = np.array([[0], [0]])
        self.P = np.eye(2)
```

```python
#!/usr/bin/env python3
"""
风扇控制模块 - PWM调速控制
"""

import RPi.GPIO as GPIO

class FanController:
    def __init__(self, pwm_pin, enable_pin=None):
        """
        pwm_pin: PWM控制引脚
        enable_pin: 使能引脚（可选）
        """
        self.pwm_pin = pwm_pin
        self.enable_pin = enable_pin
        
        # GPIO设置
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(pwm_pin, GPIO.OUT)
        
        if enable_pin:
            GPIO.setup(enable_pin, GPIO.OUT)
            GPIO.output(enable_pin, GPIO.HIGH)  # 使能风扇
        
        # PWM初始化（25kHz，适合风扇调速）
        self.pwm = GPIO.PWM(pwm_pin, 25000)
        self.pwm.start(0)
        
        self.current_duty = 0
        
    def set_speed(self, duty_cycle):
        """
        设置风扇速度
        duty_cycle: 0-100
        """
        duty_cycle = max(0, min(100, duty_cycle))
        self.pwm.ChangeDutyCycle(duty_cycle)
        self.current_duty = duty_cycle
        
    def set_speed_normalized(self, speed):
        """
        设置风扇速度（归一化）
        speed: 0.0-1.0
        """
        duty = speed * 100
        self.set_speed(duty)
        
    def stop(self):
        """停止风扇"""
        self.set_speed(0)
        
    def enable(self):
        """使能风扇"""
        if self.enable_pin:
            GPIO.output(self.enable_pin, GPIO.HIGH)
            
    def disable(self):
        """禁用风扇"""
        if self.enable_pin:
            GPIO.output(self.enable_pin, GPIO.LOW)
        self.stop()
        
    def get_speed(self):
        """获取当前速度"""
        return self.current_duty
        
    def cleanup(self):
        """清理GPIO"""
        self.stop()
        self.pwm.stop()
        if self.enable_pin:
            GPIO.cleanup(self.enable_pin)
```

### 4.5 帆板控制算法

```python
#!/usr/bin/env python3
"""
帆板控制算法 - PID控制 + 前馈补偿
"""

import numpy as np

class SailController:
    def __init__(self, pid_params=None):
        """
        pid_params: PID参数字典 {'kp': ..., 'ki': ..., 'kd': ...}
        """
        if pid_params is None:
            self.pid_params = {
                'kp': 3.0,
                'ki': 0.2,
                'kd': 0.8
            }
        else:
            self.pid_params = pid_params
            
        # PID状态变量
        self.error_sum = 0
        self.error_prev = 0
        
        # 目标角度
        self.target_angle = 0
        
        # 输出限制
        self.output_min = 0    # 最小风速
        self.output_max = 100  # 最大风速
        
        # 前馈参数
        self.feedforward_gain = 0.5
        
        # 风力-角度映射表（用于前馈）
        self.wind_angle_map = {
            0: 0,
            15: 20,
            30: 40,
            45: 55,
            60: 70,
            75: 85,
            90: 100
        }
        
    def compute(self, current_angle, current_angular_velocity=None):
        """
        计算控制输出
        current_angle: 当前帆板角度（度）
        current_angular_velocity: 当前角速度（度/秒），可选
        
        返回: 风扇速度 (0-100)
        """
        # 计算误差
        error = self.target_angle - current_angle
        
        # PID计算
        # 比例项
        p_term = self.pid_params['kp'] * error
        
        # 积分项（带抗饱和）
        self.error_sum += error
        self.error_sum = np.clip(self.error_sum, -100, 100)  # 抗饱和
        i_term = self.pid_params['ki'] * self.error_sum
        
        # 微分项
        if current_angular_velocity is not None:
            # 使用角速度作为微分项（更准确）
            d_term = -self.pid_params['kd'] * current_angular_velocity
        else:
            # 使用误差差分
            d_term = self.pid_params['kd'] * (error - self.error_prev)
            
        self.error_prev = error
        
        # PID输出
        pid_output = p_term + i_term + d_term
        
        # 前馈补偿
        feedforward = self._feedforward(self.target_angle)
        
        # 总输出
        output = pid_output + feedforward
        
        # 限制输出范围
        output = np.clip(output, self.output_min, self.output_max)
        
        return output
        
    def _feedforward(self, target_angle):
        """
        前馈补偿
        根据目标角度预测所需风速
        """
        # 线性插值
        angles = sorted(self.wind_angle_map.keys())
        
        if target_angle <= angles[0]:
            return self.wind_angle_map[angles[0]] * self.feedforward_gain
        if target_angle >= angles[-1]:
            return self.wind_angle_map[angles[-1]] * self.feedforward_gain
            
        # 找到目标角度所在的区间
        for i in range(len(angles) - 1):
            if angles[i] <= target_angle <= angles[i+1]:
                # 线性插值
                t = (target_angle - angles[i]) / (angles[i+1] - angles[i])
                wind_speed = self.wind_angle_map[angles[i]] + t * (
                    self.wind_angle_map[angles[i+1]] - self.wind_angle_map[angles[i]]
                )
                return wind_speed * self.feedforward_gain
                
        return 0
        
    def reset(self):
        """重置控制器状态"""
        self.error_sum = 0
        self.error_prev = 0
        
    def set_target(self, angle):
        """设置目标角度"""
        self.target_angle = angle
        
    def tune_pid(self, kp=None, ki=None, kd=None):
        """调整PID参数"""
        if kp is not None:
            self.pid_params['kp'] = kp
        if ki is not None:
            self.pid_params['ki'] = ki
        if kd is not None:
            self.pid_params['kd'] = kd
            
    def calibrate_feedforward(self, angle_wind_pairs):
        """
        校准前馈映射表
        angle_wind_pairs: [(角度, 风速), ...]
        """
        self.wind_angle_map = {angle: wind for angle, wind in angle_wind_pairs}
```

```python
        self.stats['start_time'] = time.time()
        
        # 启动控制线程
        self.control_thread = threading.Thread(target=self._control_loop)
        self.control_thread.start()
        
        print("系统就绪，请设置目标角度...")
        
    def _control_loop(self):
        """主控制循环（100Hz）"""
        dt = 0.01  # 10ms
        
        while self.running:
            current_time = time.time()
            
            # 读取IMU数据
            accel = self.imu.read_accel()
            gyro = self.imu.read_gyro()
            
            # 使用加速度计计算角度
            accel_angle = np.arctan2(-accel[0], np.sqrt(accel[1]**2 + accel[2]**2)) * 180 / np.pi
            
            # 使用陀螺仪获取角速度
            angular_velocity = gyro[1]  # 绕Y轴旋转
            
            # 卡尔曼滤波融合
            self.kalman.predict()
            self.current_angle, self.current_angular_velocity = self.kalman.update(accel_angle)
            
            # 计算控制输出
            self.fan_speed = self.controller.compute(
                self.current_angle,
                self.current_angular_velocity
            )
            
            # 设置风扇速度
            self.fan.set_speed(self.fan_speed)
            
            # 更新统计
            self.stats['control_count'] += 1
            self.stats['max_angle'] = max(self.stats['max_angle'], self.current_angle)
            self.stats['min_angle'] = min(self.stats['min_angle'], self.current_angle)
            
            # 记录数据
            elapsed = current_time - self.stats['start_time']
            self.data_log['time'].append(elapsed)
            self.data_log['angle'].append(self.current_angle)
            self.data_log['angular_velocity'].append(self.current_angular_velocity)
            self.data_log['fan_speed'].append(self.fan_speed)
            
            # 等待下一个周期
            elapsed_loop = time.time() - current_time
            if elapsed_loop < dt:
                time.sleep(dt - elapsed_loop)
                
    def set_target_angle(self, angle):
        """设置目标角度"""
        angle = max(0, min(90, angle))
        self.controller.set_target(angle)
        print(f"目标角度设置为: {angle}°")
        
    def get_status(self):
        """获取系统状态"""
        return {
            'angle': self.current_angle,
            'angular_velocity': self.current_angular_velocity,
            'fan_speed': self.fan_speed,
            'target_angle': self.controller.target_angle,
            'max_angle': self.stats['max_angle'],
            'min_angle': self.stats['min_angle'],
            'control_count': self.stats['control_count'],
            'elapsed': time.time() - self.stats['start_time']
        }
        
    def save_log(self, filename='sail_log.csv'):
        """保存数据日志"""
        import csv
        with open(filename, 'w', newline='') as f:
            writer = csv.writer(f)
            writer.writerow(['Time', 'Angle', 'AngularVelocity', 'FanSpeed'])
            for i in range(len(self.data_log['time'])):
                writer.writerow([
                    self.data_log['time'][i],
                    self.data_log['angle'][i],
                    self.data_log['angular_velocity'][i],
                    self.data_log['fan_speed'][i]
                ])
        print(f"数据已保存到 {filename}")
        
    def plot_data(self):
        """绘制数据曲线"""
        import matplotlib.pyplot as plt
        
        fig, axes = plt.subplots(3, 1, figsize=(10, 8))
        
        axes[0].plot(self.data_log['time'], self.data_log['angle'])
        axes[0].axhline(y=self.controller.target_angle, color='r', linestyle='--', label='目标')
        axes[0].set_xlabel('时间 (s)')

        plt.tight_layout()
        plt.savefig('sail_data.png', dpi=150)
        plt.show()
        
    def stop(self):
        """停止系统"""
        self.running = False
        
        # 等待线程结束
        if hasattr(self, 'control_thread'):
            self.control_thread.join()
            
        # 停止风扇
        self.fan.cleanup()
        
        print("系统已停止")
        
    def auto_tune(self):
        """自动PID参数整定"""
        print("开始自动PID参数整定...")
        print("请保持帆板静止，按Enter开始...")
        input()
        
        # 测试不同参数
        test_params = [
            {'kp': 1.0, 'ki': 0.05, 'kd': 0.3},
            {'kp': 2.0, 'ki': 0.1, 'kd': 0.5},
            {'kp': 3.0, 'ki': 0.15, 'kd': 0.7},
            {'kp': 4.0, 'ki': 0.2, 'kd': 0.9},
        ]
        
        best_params = None
        best_score = float('inf')
        
        for params in test_params:
            print(f"\n测试参数: {params}")
            self.controller.tune_pid(**params)
            
            # 设置测试角度
            self.set_target_angle(30)
            
            # 运行5秒测试
            self.data_log = {
                'time': [],
                'angle': [],
                'angular_velocity': [],
                'fan_speed': []
            }
            
            time.sleep(5)
            
            # 计算评分（角度误差平方和越小越好）
            if len(self.data_log['angle']) > 0:
                errors = [abs(a - 30) for a in self.data_log['angle']]
                score = np.mean(errors)
                print(f"评分: {score:.2f}")
                
                if score < best_score:
                    best_score = score
                    best_params = params
                    
        if best_params:
            print(f"\n最佳参数: {best_params}")
            self.controller.tune_pid(**best_params)
            
        # 恢复目标角度
        self.set_target_angle(0)
        
        return best_params
```

```python
def calibrate_wind_angle_map(system):
    """
    校准风力-角度映射
```

### 5.2 常见问题及解决

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| 帆板不响应 | 风扇接线错误 | 检查PWM接线和电源 |
| 帆板振荡 | PID参数不合适 | 减小kp，增大kd |
| 帆板偏移 | IMU未校准 | 重新校准IMU |
| 响应延迟 | 控制频率太低 | 提高控制频率到100Hz以上 |
| 角度不准 | IMU安装位置 | 确保IMU安装在旋转轴上 |

### 5.3 性能优化

1. **提高控制频率**: 使用硬件PWM，减少软件延迟
2. **优化滤波**: 调整卡尔曼滤波器参数
3. **前馈补偿**: 根据目标角度预测所需风速
4. **自适应控制**: 根据帆板状态自动调整PID参数
5. **扰动抑制**: 添加扰动观测器
