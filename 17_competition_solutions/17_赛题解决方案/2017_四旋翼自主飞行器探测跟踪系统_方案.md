# 2017年全国大学生电子设计竞赛 - 四旋翼自主飞行器探测跟踪系统

## 题目分析
设计一个四旋翼自主飞行器，能够在指定区域内自主探测目标（如地面标志物），并完成对移动目标的跟踪飞行。

#### 1. MAVSDK通信接口 (mavsdk_interface.py)
```python
import asyncio
from mavsdk import System
from mavsdk.offboard import PositionNedYaw, VelocityNedYaw

class MavlinkInterface:
    def __init__(self, config):
        self.drone = System()
        self.connection = config.get('connection', 'serial:///dev/ttyACM0:57600')
        self.connected = False

    async def connect(self):
        """连接飞控"""
        await self.drone.connect(system_address=self.connection)
        async for state in self.drone.core.connection_state():
            if state.is_connected:
                print("飞控已连接")
                self.connected = True
                break

        # 等待GPS锁定(室外)或准备好室内模式
        async for health in self.drone.telemetry.health():
            if health.is_gyrometer_calibration_ok and health.is_accelerometer_calibration_ok:
                print("传感器校准完成")
                break

    async def arm_and_takeoff(self, altitude=3.0):
        """解锁并起飞"""
        await self.drone.action.arm()
        await self.drone.action.set_takeoff_altitude(altitude)
        await self.drone.action.takeoff()

        # 等待到达目标高度
        async for position in self.drone.telemetry.position():
            if abs(position.relative_altitude_m - altitude) < 0.3:
                print(f"到达目标高度: {position.relative_altitude_m:.1f}m")
                break

    async def start_offboard(self):
        """进入Offboard模式"""
        await self.drone.offboard.set_position_ned(
            PositionNedYaw(0, 0, 0, 0))
        await self.drone.offboard.start()
        print("Offboard模式已启动")

    async def send_position_command(self, north, east, down, yaw=0):
        """发送位置指令(NED坐标系)"""
        await self.drone.offboard.set_position_ned(
            PositionNedYaw(north, east, down, yaw))

    async def send_velocity_command(self, vn, ve, vd, yaw=0):
        """发送速度指令"""
        await self.drone.offboard.set_velocity_ned(
            VelocityNedYaw(vn, ve, vd, yaw))

    async def get_position(self):
        """获取当前位置"""
        async for pos in self.drone.telemetry.position():
            return pos

    async def land(self):
        """降落"""
        await self.drone.action.land()

    async def return_to_launch(self):
        """返航"""
        await self.drone.action.return_to_launch()
```

#### 2. 视觉目标跟踪 (visual_tracker.py)
```python
import cv2
import numpy as np
from collections import deque

class VisualTracker:
    """视觉目标跟踪器"""

    def __init__(self, config):
        # 目标特征(颜色/形状)
        self.target_type = config.get('target_type', 'aruco')
        self.marker_size = config.get('marker_size', 0.3)  # m

        # ArUco检测器
        if self.target_type == 'aruco':
            aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
            self.detector = cv2.aruco.ArucoDetector(aruco_dict)

        # 跟踪器(KCF/CSRT)
        self.tracker = None
        self.tracking = False
        self.track_buffer = deque(maxlen=30)

        # 相机参数
        self.camera_matrix = np.array(config['camera_matrix'])
        self.dist_coeffs = np.array(config['dist_coeffs'])

    def detect_target(self, frame):
        """检测目标位置"""
        if self.target_type == 'aruco':
            return self._detect_aruco(frame)
        elif self.target_type == 'color':
            return self._detect_color(frame)
        return None

    def _detect_aruco(self, frame):
        """ArUco码检测"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, _ = self.detector.detectMarkers(gray)

        if ids is not None and len(ids) > 0:
            # 取第一个检测到的标记
            rvecs, tvecs, _ = cv2.aruco.estimatePoseSingleMarkers(
                corners, self.marker_size, self.camera_matrix, self.dist_coeffs)

            # 目标在相机坐标系中的位置
            target_pos = tvecs[0][0]  # [x, y, z] in meters

            # 图像中心偏移
            cx = np.mean(corners[0][0][:, 0])
            cy = np.mean(corners[0][0][:, 1])
            h, w = frame.shape[:2]
            offset_x = (cx - w / 2) / w  # 归一化偏移
            offset_y = (cy - h / 2) / h

            return {
                'found': True,
                'position': target_pos,
                'image_offset': (offset_x, offset_y),
                'corners': corners[0][0],
                'id': ids[0][0]
            }
        return {'found': False}

    def _detect_color(self, frame):
        """颜色目标检测"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 检测红色目标
        mask1 = cv2.inRange(hsv, np.array([0, 100, 100]), np.array([10, 255, 255]))
        mask2 = cv2.inRange(hsv, np.array([160, 100, 100]), np.array([180, 255, 255]))
        mask = mask1 | mask2

        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 500:
                M = cv2.moments(c)
                cx = int(M['m10'] / M['m00'])
                cy = int(M['m01'] / M['m00'])
                h, w = frame.shape[:2]
                return {
                    'found': True,
                    'image_offset': ((cx - w/2) / w, (cy - h/2) / h),
                    'contour': c
                }
        return {'found': False}

    def track_target(self, frame, bbox):
        """KCF/CSRT跟踪"""
        if not self.tracking:
            self.tracker = cv2.TrackerCSRT_create()
            self.tracker.init(frame, bbox)
            self.tracking = True
            return True

        success, box = self.tracker.update(frame)
        if success:
            x, y, w, h = [int(v) for v in box]
            cx, cy = x + w // 2, y + h // 2
            img_h, img_w = frame.shape[:2]
            self.track_buffer.append((cx, cy))
            return {
                'found': True,
                'bbox': (x, y, w, h),
                'center': (cx, cy),
                'image_offset': ((cx - img_w/2) / img_w, (cy - img_h/2) / img_h)
            }
        else:
            self.tracking = False
            return {'found': False}
```

#### 3. 任务管理器 (mission_manager.py)
```python
import asyncio
from enum import Enum, auto

class MissionState(Enum):
    IDLE = auto()
    TAKEOFF = auto()
    SEARCH = auto()
    DETECTED = auto()
    TRACKING = auto()
    LOST = auto()
    LANDING = auto()
    RETURN_HOME = auto()

class MissionManager:
    """任务状态机"""

    def __init__(self, mavlink, tracker, planner, config):
        self.mavlink = mavlink
        self.tracker = tracker
        self.planner = planner
        self.config = config

        self.state = MissionState.IDLE
        self.home_position = (0, 0, 0)
        self.search_altitude = config.get('search_altitude', 5.0)
        self.track_altitude = config.get('track_altitude', 3.0)
        self.lost_timeout = config.get('lost_timeout', 5.0)  # 秒
        self.lost_timer = 0

    async def run(self, frame):
        """主循环 - 根据状态执行对应逻辑"""
        if self.state == MissionState.IDLE:
            return

        elif self.state == MissionState.TAKEOFF:
            await self.mavlink.arm_and_takeoff(self.search_altitude)
            self.state = MissionState.SEARCH
            print("起飞完成, 进入搜索模式")

        elif self.state == MissionState.SEARCH:
            # 按预设路径搜索
            target = self.tracker.detect_target(frame)
            if target['found']:
                self.state = MissionState.DETECTED
                print("目标已发现!")
            else:
                # 沿搜索路径飞行
                waypoint = self.planner.get_next_search_waypoint()
                await self.mavlink.send_position_command(
                    waypoint[0], waypoint[1], -self.search_altitude)

        elif self.state == MissionState.DETECTED:
            # 目标刚发现, 切换到跟踪
            target = self.tracker.detect_target(frame)
            if target['found']:
                # 初始化跟踪器
                corners = target.get('corners')
                if corners is not None:
                    x_min = int(corners[:, 0].min())
                    y_min = int(corners[:, 1].min())
                    x_max = int(corners[:, 0].max())
                    y_max = int(corners[:, 1].max())
                    bbox = (x_min, y_min, x_max - x_min, y_max - y_min)
                    self.tracker.track_target(frame, bbox)
                self.state = MissionState.TRACKING
                self.lost_timer = 0

        elif self.state == MissionState.TRACKING:
            target = self.tracker.detect_target(frame)

            if target['found']:
                self.lost_timer = 0
                offset_x, offset_y = target['image_offset']

                # 计算跟踪速度指令
                max_speed = self.config.get('max_track_speed', 2.0)  # m/s
                vn = -offset_y * max_speed  # 前后
                ve = offset_x * max_speed   # 左右
                vd = 0  # 保持高度

                # 根据距离调整高度
                if 'position' in target:
                    dist = np.linalg.norm(target['position'][:2])
                    if dist < 1.0:  # 太近, 提升
                        vd = 0.5
                    elif dist > 3.0:  # 太远, 降低
                        vd = -0.3

                await self.mavlink.send_velocity_command(vn, ve, vd)

            else:
                self.lost_timer += 0.1  # 假设10Hz循环
                if self.lost_timer > self.lost_timeout:
                    self.state = MissionState.LOST
                    print("目标丢失, 进入搜索")

        elif self.state == MissionState.LOST:
            # 回到最后已知位置搜索
            target = self.tracker.detect_target(frame)
            if target['found']:
                self.state = MissionState.TRACKING
                self.lost_timer = 0
            else:
                # 执行搜索模式
                waypoint = self.planner.get_search_near_lost()
                await self.mavlink.send_position_command(
                    waypoint[0], waypoint[1], -self.search_altitude)

        elif self.state == MissionState.LANDING:
            await self.mavlink.land()

        elif self.state == MissionState.RETURN_HOME:
            await self.mavlink.return_to_launch()
```

#### 4. 主程序 (main.py)
```python
import asyncio
import cv2
import yaml
import numpy as np
from camera import Camera
from mavsdk_interface import MavlinkInterface
from visual_tracker import VisualTracker
from path_planner import PathPlanner
from mission_manager import MissionManager, MissionState

async def main():
    with open('config.yaml', 'r') as f:
        config = yaml.safe_load(f)

    # 初始化模块
    camera = Camera(config['camera'])
    mavlink = MavlinkInterface(config['mavlink'])
    tracker = VisualTracker(config['tracker'])
    planner = PathPlanner(config['planner'])
    mission = MissionManager(mavlink, tracker, planner, config['mission'])

    # 连接飞控
    await mavlink.connect()

    # 启动Offboard模式
    await mavlink.start_offboard()

    mission.state = MissionState.TAKEOFF

    print("=== 四旋翼探测跟踪系统启动 ===")

    try:
        while mission.state != MissionState.IDLE:
            frame = camera.capture()
            if frame is None:
                continue

            await mission.run(frame)

            # 调试显示
            if config.get('debug', False):
                target = tracker.detect_target(frame)
                if target['found']:
                    cv2.putText(frame, "TARGET FOUND", (10, 30),
                               cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
                cv2.putText(frame, f"State: {mission.state.name}", (10, 70),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
                cv2.imshow('Tracker', frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    mission.state = MissionState.LANDING
                    break

            await asyncio.sleep(0.1)  # 10Hz控制循环

    finally:
        camera.release()
        cv2.destroyAllWindows()

if __name__ == '__main__':
    asyncio.run(main())
```

## 关键技术要点

### 1. 视觉定位方案选择
| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| ArUco码 | 精度高, 计算快 | 需布置标记 | 固定场地 |
| 颜色检测 | 简单快速 | 受光照影响大 | 颜色鲜明目标 |
| 深度学习 | 通用性强 | 计算量大 | 复杂目标 |
| 光流+IMU | 不需外部标记 | 累积误差 | 辅助定位 |

### 2. 通信架构
```
Orange Pi 5 ←─USB/UART─→ 飞控(CubeOrange)
     │                          │
     │  MAVLink(串口)           │  PWM → ESC → 电机
     │  115200 baud             │
     ↓                          ↓
  视觉处理结果              飞行姿态控制
```

### 3. Offboard模式注意事项
- 必须以≥2Hz持续发送指令, 否则飞控自动退出Offboard
- 先设置指令再启动Offboard模式
- 安全起见设置围栏(Geofence)
