# 2017 全国大学生电子设计竞赛 - 管道内钢珠运动测量装置

## 题目概述
设计一个测量装置，能检测管道内钢珠的运动状态（位置、速度、加速度），管道可倾斜不同角度，要求实时测量并显示钢珠运动参数。

## 软件架构

### 1. 视觉检测钢珠位置（方案A）

```python
import cv2
import numpy as np
from collections import deque
import time

class SteelBallTracker:
    def __init__(self, camera_id=0, fps=120):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, fps)
        self.actual_fps = self.cap.get(cv2.CAP_PROP_FPS)
        # 标定：像素到毫米的映射
        self.mm_per_pixel = 0.5  # 需要标定
        # 运动历史
        self.positions = deque(maxlen=100)
        self.timestamps = deque(maxlen=100)
        # 背景模型
        self.bg_model = None

    def calibrate(self, known_length_mm, pixel_length):
        """标定比例尺"""
        self.mm_per_pixel = known_length_mm / pixel_length

    def detect_ball(self, frame):
        """检测钢珠位置"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        # 高斯模糊
        blurred = cv2.GaussianBlur(gray, (9, 9), 2)
        # 霍夫圆检测
        circles = cv2.HoughCircles(blurred, cv2.HOUGH_GRADIENT,
                                   dp=1, minDist=30,
                                   param1=100, param2=40,
                                   minRadius=5, maxRadius=30)
        if circles is not None:
            circles = np.uint16(np.around(circles))
            x, y, r = circles[0][0]
            return (int(x), int(y), int(r))
        return None

    def detect_ball_diff(self, frame):
        """差分法检测运动中的钢珠"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        if self.bg_model is None:
            self.bg_model = gray.astype(float)
            return None
        # 更新背景
        cv2.accumulateWeighted(gray, self.bg_model, 0.01)
        bg = self.bg_model.astype(np.uint8)
        diff = cv2.absdiff(gray, bg)
        _, thresh = cv2.threshold(diff, 30, 255, cv2.THRESH_BINARY)
        contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            M = cv2.moments(c)
            if M["m00"] > 0:
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                return (cx, cy)
        return None

    def update(self, frame):
        """更新位置和时间记录"""
        ball = self.detect_ball(frame)
        if ball:
            now = time.time()
            pos_mm = (ball[0] * self.mm_per_pixel, ball[1] * self.mm_per_pixel)
            self.positions.append(pos_mm)
            self.timestamps.append(now)
            return ball
        return None
```

### 2. 运动参数计算

```python
class MotionAnalyzer:
    def __init__(self, tracker):
        self.tracker = tracker

    def get_position(self):
        """当前位置(mm)"""
        if self.tracker.positions:
            return self.tracker.positions[-1]
        return None

    def get_velocity(self):
        """瞬时速度(mm/s)"""
        pos = self.tracker.positions
        ts = self.tracker.timestamps
        if len(pos) < 2:
            return None
        dt = ts[-1] - ts[-2]
        if dt <= 0:
            return None
        dx = pos[-1][0] - pos[-2][0]
        dy = pos[-1][1] - pos[-2][1]
        velocity = np.sqrt(dx**2 + dy**2) / dt
        return velocity

    def get_average_velocity(self, window=5):
        """平均速度(mm/s)"""
        pos = self.tracker.positions
        ts = self.tracker.timestamps
        if len(pos) < window:
            return None
        dt = ts[-1] - ts[-window]
        if dt <= 0:
            return None
        dx = pos[-1][0] - pos[-window][0]
        dy = pos[-1][1] - pos[-window][1]
        return np.sqrt(dx**2 + dy**2) / dt

    def get_acceleration(self):
        """加速度(mm/s²)"""
        pos = self.tracker.positions
        ts = self.tracker.timestamps
        if len(pos) < 3:
            return None
        # 三点差分计算加速度
        v1 = self._velocity_between(0, 1)
        v2 = self._velocity_between(1, 2)
        if v1 is None or v2 is None:
            return None
        dt = ts[-1] - ts[-2]
        if dt <= 0:
            return None
        return (v2 - v1) / dt

    def _velocity_between(self, i, j):
        pos = self.tracker.positions
        ts = self.tracker.timestamps
        n = len(pos)
        dt = ts[n-1-j] - ts[n-1-i]
        if dt <= 0:
            return None
        dx = pos[n-1-j][0] - pos[n-1-i][0]
        dy = pos[n-1-j][1] - pos[n-1-i][1]
        return np.sqrt(dx**2 + dy**2) / dt

    def get_theoretical_acceleration(self, angle_deg, g=9800):
        """理论加速度 a = g·sin(θ) (mm/s²)"""
        return g * np.sin(np.radians(angle_deg))
```

### 3. 光电传感器方案（方案B）

```python
import serial
import time

class PhotoelectricSensor:
    """通过STM32读取光电传感器阵列"""
    def __init__(self, uart_port='/dev/ttyS3', num_sensors=6, spacing_mm=50):
        self.ser = serial.Serial(uart_port, 115200, timeout=0.1)
        self.num_sensors = num_sensors
        self.spacing_mm = spacing_mm
        self.trigger_times = [None] * num_sensors

    def read_triggers(self):
        """读取传感器触发时间戳"""
        self.ser.write(b'\xAA\x01\x55')
        resp = self.ser.read(2 + self.num_sensors * 4)
        if len(resp) >= 2 + self.num_sensors * 4:
            if resp[0] == 0xAA:
                times = []
                for i in range(self.num_sensors):
                    t = int.from_bytes(resp[2+i*4:6+i*4], 'big')
                    times.append(t if t > 0 else None)
                self.trigger_times = times
                return times
        return None

    def calc_velocity(self):
        """通过传感器间距和时间差计算速度"""
        triggers = self.trigger_times
        velocities = []
        for i in range(len(triggers)-1):
            if triggers[i] is not None and triggers[i+1] is not None:
                dt = (triggers[i+1] - triggers[i]) / 1000000.0  # us -> s
                if dt > 0:
                    v = self.spacing_mm / dt
                    velocities.append(v)
        return velocities
```

### 4. 主程序

```python
import cv2
import time
import numpy as np

def main():
    tracker = SteelBallTracker(camera_id=0)
    analyzer = MotionAnalyzer(tracker)

    # 标定
    tracker.calibrate(known_length_mm=100, pixel_length=200)

    print("管道内钢珠运动测量装置")
    print("按键: q-退出, c-清零, s-截图")

    while True:
        ret, frame = tracker.cap.read()
        if not ret:
            break

        ball = tracker.update(frame)

        # 显示检测结果
        if ball:
            cv2.circle(frame, (ball[0], ball[1]), ball[2], (0, 255, 0), 2)

        # 计算并显示参数
        pos = analyzer.get_position()
        vel = analyzer.get_velocity()
        acc = analyzer.get_acceleration()

        y_offset = 30
        if pos:
            cv2.putText(frame, f"Pos: ({pos[0]:.1f}, {pos[1]:.1f}) mm",
                       (10, y_offset), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,255,0), 2)
            y_offset += 30
        if vel:
            cv2.putText(frame, f"Velocity: {vel:.1f} mm/s",
                       (10, y_offset), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,255,0), 2)
            y_offset += 30
        if acc:
            cv2.putText(frame, f"Accel: {acc:.1f} mm/s2",
                       (10, y_offset), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,255,0), 2)

        cv2.imshow("Ball Tracker", frame)

        key = cv2.waitKey(1) & 0xFF
        if key == ord('q'):
            break
        elif key == ord('c'):
            tracker.positions.clear()
            tracker.timestamps.clear()
        elif key == ord('s'):
            cv2.imwrite("measurement.png", frame)

    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

## 关键技术要点

### 高速检测
- 使用120fps摄像头减少运动模糊
- 降低分辨率到320×240提高帧率
- ROI只处理管道区域

### 滤波去噪
- 位置数据：卡尔曼滤波
- 速度计算：滑动窗口平均
- 加速度计算：二次差分+低通滤波

### 理论验证
- 自由落体：a = g·sin(θ)
- 有摩擦时：a = g·sin(θ) - μ·g·cos(θ)
- 可反推摩擦系数μ
