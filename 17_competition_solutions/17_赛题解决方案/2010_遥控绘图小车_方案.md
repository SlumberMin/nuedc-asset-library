## 一、题目分析

### 1.1 核心要求
- 设计遥控绘图小车，能在指定区域内按轨迹自动行驶
- 小车能绘制指定图形（圆形、三角形、矩形等）
- 支持遥控和自动两种模式
- 绘图轨迹精度高，重复性好

### 1.2 技术难点
- 精确的路径规划与轨迹跟踪
- 高精度的位置控制与定位
- 图形绘制的平滑性与准确性
- 遥控与自动模式的无缝切换

### 4.2 图形路径生成模块

```python
#!/usr/bin/env python3
"""
图形路径生成模块
支持圆形、三角形、矩形、五角星等图形
"""

import numpy as np
from dataclasses import dataclass
from typing import List, Tuple
from enum import Enum

class ShapeType(Enum):
    CIRCLE = "circle"
    TRIANGLE = "triangle"
    RECTANGLE = "rectangle"
    PENTAGRAM = "pentagram"
    CUSTOM = "custom"

@dataclass
class PathPoint:
    x: float
    y: float
    pen_down: bool = True  # True=落笔，True=抬笔

class PathGenerator:
    def __init__(self, center=(0.0, 0.0), scale=1.0):
        self.center = np.array(center)
        self.scale = scale
        
    def generate_circle(self, radius, num_points=100):
        """生成圆形路径"""
        points = []
        for i in range(num_points + 1):
            angle = 2 * np.pi * i / num_points
            x = self.center[0] + radius * np.cos(angle)
            y = self.center[1] + radius * np.sin(angle)
            points.append(PathPoint(x * self.scale, y * self.scale, pen_down=True))
        return points
        
    def generate_triangle(self, side_length, num_points_per_side=30):
        """生成等边三角形路径"""
        points = []
        # 三个顶点
        vertices = []
        for i in range(3):
            angle = 2 * np.pi * i / 3 - np.pi / 2
            x = self.center[0] + side_length * np.cos(angle)
            y = self.center[1] + side_length * np.sin(angle)
            vertices.append((x, y))
            
        # 在边上插值
        for i in range(3):
            start = vertices[i]
            end = vertices[(i + 1) % 3]
            for j in range(num_points_per_side):
                t = j / num_points_per_side
                x = start[0] + t * (end[0] - start[0])
                y = start[1] + t * (end[1] - start[1])
                points.append(PathPoint(x * self.scale, y * self.scale, pen_down=True))
                
        # 闭合路径
        points.append(PathPoint(vertices[0][0] * self.scale, vertices[0][1] * self.scale, pen_down=True))
        return points
        
    def generate_rectangle(self, width, height, num_points_per_side=30):
        """生成矩形路径"""
        points = []
        # 四个顶点
        vertices = [
            (self.center[0] - width/2, self.center[1] - height/2),
            (self.center[0] + width/2, self.center[1] - height/2),
            (self.center[0] + width/2, self.center[1] + height/2),
            (self.center[0] - width/2, self.center[1] + height/2)
        ]
        
        # 在边上插值
        for i in range(4):
            start = vertices[i]
            end = vertices[(i + 1) % 4]
            for j in range(num_points_per_side):
                t = j / num_points_per_side
                x = start[0] + t * (end[0] - start[0])
                y = start[1] + t * (end[1] - start[1])
                points.append(PathPoint(x * self.scale, y * self.scale, pen_down=True))
                
        # 闭合路径
        points.append(PathPoint(vertices[0][0] * self.scale, vertices[0][1] * self.scale, pen_down=True))
        return points
        
    def generate_pentagram(self, outer_radius, inner_radius_ratio=0.382, num_points=100):
        """生成五角星路径"""
        points = []
        inner_radius = outer_radius * inner_radius_ratio
        
        for i in range(num_points + 1):
            angle = 2 * np.pi * i / num_points - np.pi / 2
            
            # 五角星：每72度在内外半径之间切换
            segment_angle = (angle + np.pi / 2) % (2 * np.pi / 5)
            if segment_angle < np.pi / 5:
                r = outer_radius
            else:
                r = inner_radius
                
            x = self.center[0] + r * np.cos(angle)
            y = self.center[1] + r * np.sin(angle)
            points.append(PathPoint(x * self.scale, y * self.scale, pen_down=True))
            
        return points
        
    def generate_custom_path(self, waypoints, smooth=True):
        """生成自定义路径"""
        if smooth:
            # 使用贝塞尔曲线平滑
            points = self._bezier_smooth(waypoints, num_points=200)
        else:
            # 直接连接
            points = [PathPoint(wp[0] * self.scale, wp[1] * self.scale, True) for wp in waypoints]
            
        return points
        
    def _bezier_smooth(self, waypoints, num_points=200):
        """贝塞尔曲线平滑"""
        points = []
        n = len(waypoints) - 1
        
        for t in np.linspace(0, 1, num_points):
            x, y = 0, 0
            for i, (wx, wy) in enumerate(waypoints):
                # 伯恩斯坦基函数
                bernstein = self._bernstein_basis(n, i, t)
                x += wx * bernstein
                y += wy * bernstein
            points.append(PathPoint(x * self.scale, y * self.scale, pen_down=True))
            
        return points
        
    def _bernstein_basis(self, n, i, t):
        """伯恩斯坦基函数"""
        from math import comb
        return comb(n, i) * (t ** i) * ((1 - t) ** (n - i))
```

```python
#!/usr/bin/env python3
"""
视觉定位模块 - 使用ArUco标记进行精确定位
"""

import cv2
import numpy as np

class VisionLocalizer:
    def __init__(self, camera_id=0, marker_size=0.05):
        """
        camera_id: 摄像头ID
        marker_size: ArUco标记的实际尺寸（米）
        """
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)
        
        self.marker_size = marker_size
        
        # ArUco字典和参数
        self.aruco_dict = cv2.aruco.Dictionary_get(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters_create()
        
        # 相机标定参数（需要预先标定）
        self.camera_matrix = None
        self.dist_coeffs = None
        
        # 坐标变换矩阵（图像坐标到世界坐标）
        self.homography = None
        
    def calibrate_camera(self, images):
        """相机标定"""
        # 准备标定板角点
        obj_points = []  # 3D点
        img_points = []  # 2D点
        
        # 标定板参数
        board_size = (9, 6)  # 棋盘格内角点数
        square_size = 0.025  # 格子大小（米）
        
        # 准备3D点
        objp = np.zeros((board_size[0] * board_size[1], 3), np.float32)
        objp[:, :2] = np.mgrid[0:board_size[0], 0:board_size[1]].T.reshape(-1, 2) * square_size
        
        for img in images:
            gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
            ret, corners = cv2.findChessboardCorners(gray, board_size, None)
            
            if ret:
                obj_points.append(objp)
                img_points.append(corners)
                
        if obj_points:
            ret, self.camera_matrix, self.dist_coeffs, _, _ = cv2.calibrateCamera(
                obj_points, img_points, gray.shape[::-1], None, None
            )
            return ret
        return False
        
    def calibrate_homography(self, image_points, world_points):
        """
        标定图像到世界坐标的变换
        image_points: 图像中的4个点 [(x1,y1), ...]
        world_points: 对应的世界坐标 [(X1,Y1), ...]
        """
        image_points = np.array(image_points, dtype=np.float32)
        world_points = np.array(world_points, dtype=np.float32)
        self.homography = cv2.getPerspectiveTransform(image_points, world_points)
        
    def detect_marker(self, marker_id=0):
        """
        检测ArUco标记并返回位置
        返回: (x, y, angle) 或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 检测标记
        corners, ids, rejected = cv2.aruco.detectMarkers(
            frame, self.aruco_dict, parameters=self.aruco_params
        )
        
        if ids is not None and marker_id in ids:
            # 找到目标标记
            idx = np.where(ids == marker_id)[0][0]
            marker_corners = corners[idx][0]
            
            # 计算中心点
            center = np.mean(marker_corners, axis=0)
            
            # 计算朝向角
            top_center = np.mean(marker_corners[:2], axis=0)
            angle = np.arctan2(
                top_center[1] - center[1],
                top_center[0] - center[0]
            )
            
            # 如果有标定参数，进行姿态估计
            if self.camera_matrix is not None:
                rvec, tvec, _ = cv2.aruco.estimatePoseSingleMarkers(
                    corners[idx], self.marker_size,
                    self.camera_matrix, self.dist_coeffs
                )
                # tvec包含世界坐标系下的位置
                x, y = tvec[0][0][0], tvec[0][0][1]
            else:
                # 使用透视变换
                if self.homography is not None:
                    pt = np.array([[[center[0], center[1]]]], dtype=np.float32)
                    world_pt = cv2.perspectiveTransform(pt, self.homography)
                    x, y = world_pt[0][0]
                else:
                    x, y = center[0], center[1]
                    
            return (x, y, np.degrees(angle))
            
        return None
        
    def get_frame_with_marker(self, marker_id=0):
        """获取带有标记检测结果的图像帧"""
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        corners, ids, rejected = cv2.aruco.detectMarkers(
            frame, self.aruco_dict, parameters=self.aruco_params
        )
        
        if ids is not None:
            cv2.aruco.drawDetectedMarkers(frame, corners, ids)
            
            if marker_id in ids:
                idx = np.where(ids == marker_id)[0][0]
                center = np.mean(corners[idx][0], axis=0).astype(int)
                cv2.circle(frame, tuple(center), 5, (0, 255, 0), -1)
                
        return frame
        
    def release(self):
        """释放资源"""
        self.cap.release()
```

### 4.4 轨迹跟踪控制模块

```python
#!/usr/bin/env python3
"""
轨迹跟踪控制模块
Pure Pursuit算法实现
"""

import numpy as np
from typing import List, Tuple
from dataclasses import dataclass

@dataclass
class ControlCommand:
    linear_velocity: float  # 线速度 m/s
    angular_velocity: float  # 角速度 rad/s
    pen_state: bool  # True=落笔，False=抬笔

class TrajectoryTracker:
    def __init__(self, lookahead_distance=0.3, max_linear_vel=0.3, max_angular_vel=1.5):
        """
        lookahead_distance: 前视距离
        max_linear_vel: 最大线速度
        max_angular_vel: 最大角速度
        """
        self.lookahead_distance = lookahead_distance
        self.max_linear_vel = max_linear_vel
        self.max_angular_vel = max_angular_vel
        
        # PID参数
        self.pid_linear = {'kp': 1.5, 'ki': 0.1, 'kd': 0.05}
        self.pid_angular = {'kp': 3.0, 'ki': 0.2, 'kd': 0.1}
        
        # 状态变量
        self.linear_error_sum = 0
        self.linear_error_prev = 0
        self.angular_error_sum = 0
        self.angular_error_prev = 0
        
    def compute_control(self, current_pos, current_theta, path_points, current_index):
        """
        计算控制命令
        current_pos: 当前位置 (x, y)
        current_theta: 当前朝向角 (rad)
        path_points: 路径点列表 [(x, y), ...]
        current_index: 当前路径点索引
        
        返回: ControlCommand
        """
        current_pos = np.array(current_pos)
        
        # 查找前视点
        lookahead_point, new_index = self._find_lookahead_point(
            current_pos, path_points, current_index
        )
        
        if lookahead_point is None:
            # 已到达终点
            return ControlCommand(0, 0, False), new_index
            
        # 计算到前视点的距离和角度
        dx = lookahead_point[0] - current_pos[0]
        dy = lookahead_point[1] - current_pos[1]
        distance = np.sqrt(dx**2 + dy**2)
        target_angle = np.arctan2(dy, dx)
        
        # 角度误差
        angle_error = target_angle - current_theta
        angle_error = np.arctan2(np.sin(angle_error), np.cos(angle_error))  # 归一化
        
        # PID控制
        linear_vel = self._pid_linear(distance)
        angular_vel = self._pid_angular(angle_error)
        
        # 限制速度
        linear_vel = np.clip(linear_vel, 0, self.max_linear_vel)
        angular_vel = np.clip(angular_vel, -self.max_angular_vel, self.max_angular_vel)
        
        # 接近目标时减速
        if distance < 0.1:
            linear_vel *= 0.5
            
        return ControlCommand(linear_vel, angular_vel, True), new_index
        
    def _find_lookahead_point(self, current_pos, path_points, current_index):
        """查找前视点"""
        if current_index >= len(path_points) - 1:
            return None, len(path_points) - 1
            
        # 从当前点开始向前搜索
        for i in range(current_index, len(path_points)):
            point = np.array(path_points[i])
            distance = np.linalg.norm(point - current_pos)
            
            if distance >= self.lookahead_distance:
                return point, i
                
        # 返回终点
        return np.array(path_points[-1]), len(path_points) - 1
        
    def _pid_linear(self, error):
        """线速度PID"""
        self.linear_error_sum += error
        derivative = error - self.linear_error_prev
        self.linear_error_prev = error
        
        output = (self.pid_linear['kp'] * error +
                  self.pid_linear['ki'] * self.linear_error_sum +
                  self.pid_linear['kd'] * derivative)
        return output
        
    def _pid_angular(self, error):
        """角速度PID"""
        self.angular_error_sum += error
        derivative = error - self.angular_error_prev
        self.angular_error_prev = error
        
        output = (self.pid_angular['kp'] * error +
                  self.pid_angular['ki'] * self.angular_error_sum +
                  self.pid_angular['kd'] * derivative)
        return output
        
    def reset(self):
        """重置PID状态"""
        self.linear_error_sum = 0
        self.linear_error_prev = 0
        self.angular_error_sum = 0
        self.angular_error_prev = 0
```

#!/usr/bin/env python3
"""
电机驱动模块 - L298N双H桥驱动
"""

import RPi.GPIO as GPIO
import time

class MotorDriver:
    def __init__(self, left_pins, right_pins):
        """
        left_pins: (ENA, IN1, IN2) 左电机引脚
        right_pins: (ENB, IN3, IN4) 右电机引脚
        """
        self.left_pins = left_pins
        self.right_pins = right_pins
        
        # GPIO设置
        GPIO.setmode(GPIO.BCM)
        GPIO.setwarnings(False)
        
        # 设置引脚
        for pin in left_pins + right_pins:
            GPIO.setup(pin, GPIO.OUT)
            
        # PWM初始化
        self.left_pwm = GPIO.PWM(left_pins[0], 1000)
        self.right_pwm = GPIO.PWM(right_pins[0], 1000)
        self.left_pwm.start(0)
        self.right_pwm.start(0)
        
    def set_left_motor(self, speed):
        """
        设置左电机速度
        speed: -100 到 100
        """
        speed = max(-100, min(100, speed))
        
        if speed >= 0:
            GPIO.output(self.left_pins[1], GPIO.HIGH)
            GPIO.output(self.left_pins[2], GPIO.LOW)
        else:
            GPIO.output(self.left_pins[1], GPIO.LOW)
            GPIO.output(self.left_pins[2], GPIO.HIGH)
            
        self.left_pwm.ChangeDutyCycle(abs(speed))
        
    def set_right_motor(self, speed):
        """
        设置右电机速度
        speed: -100 到 100
        """
        speed = max(-100, min(100, speed))
        
        if speed >= 0:
            GPIO.output(self.right_pins[1], GPIO.HIGH)
            GPIO.output(self.right_pins[2], GPIO.LOW)
        else:
            GPIO.output(self.right_pins[1], GPIO.LOW)
            GPIO.output(self.right_pins[2], GPIO.HIGH)
            
        self.right_pwm.ChangeDutyCycle(abs(speed))
        
    def set_motors(self, left_speed, right_speed):
        """设置两个电机的速度"""
        self.set_left_motor(left_speed)
        self.set_right_motor(right_speed)
        
    def stop(self):
        """停止所有电机"""
        self.set_motors(0, 0)
        
    def cleanup(self):
        """清理GPIO"""
        self.stop()
        self.left_pwm.stop()
        self.right_pwm.stop()

import RPi.GPIO as GPIO

class PenController:
    def __init__(self, servo_pin, up_angle=90, down_angle=0):
        """
        servo_pin: 舵机PWM引脚
        up_angle: 抬笔角度
        down_angle: 落笔角度
        """
        self.servo_pin = servo_pin
        self.up_angle = up_angle
        self.down_angle = down_angle
        
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(servo_pin, GPIO.OUT)
        
        self.pwm = GPIO.PWM(servo_pin, 50)  # 50Hz
        self.pwm.start(0)
        
        self.pen_down = False
        self._set_angle(up_angle)  # 初始抬笔
        
    def _set_angle(self, angle):
        """设置舵机角度"""
        duty = 2.5 + angle / 18.0  # 角度转换为占空比
        self.pwm.ChangeDutyCycle(duty)
        time.sleep(0.3)  # 等待舵机到位
        
    def pen_up(self):
        """抬笔"""
        if self.pen_down:
            self._set_angle(self.up_angle)
            self.pen_down = False
            print("抬笔")
            
    def pen_down(self):
        """落笔"""
        if not self.pen_down:
            self._set_angle(self.down_angle)
            self.pen_down = True
            print("落笔")
            
    def set_pen_state(self, down):
        """设置笔状态"""
        if down:
            self.pen_down()
        else:
            self.pen_up()
            
    def cleanup(self):
        """清理"""
        self.pen_up()
        self.pwm.stop()

#!/usr/bin/env python3
"""
遥控绘图小车主程序 - Orange Pi 5
"""

import time
import threading
import numpy as np
from path_generator import PathGenerator, ShapeType
from vision_localizer import VisionLocalizer
from trajectory_tracker import TrajectoryTracker
from motor_driver import MotorDriver
from pen_controller import PenController

class DrawingCar:
    def __init__(self):
        # 硬件初始化
        self.motor = MotorDriver(
            left_pins=(18, 23, 24),   # ENA, IN1, IN2
            right_pins=(19, 27, 17)   # ENB, IN3, IN4
        )
        self.pen = PenController(servo_pin=12)
        self.vision = VisionLocalizer(camera_id=0)
        
        # 控制器初始化
        self.path_generator = PathGenerator(center=(0.5, 0.5), scale=1.0)
        self.tracker = TrajectoryTracker(lookahead_distance=0.2)
        
        # 状态变量
        self.current_pos = np.array([0.0, 0.0])
        self.current_theta = 0.0
        self.current_path = []
        self.current_index = 0
        self.is_drawing = False
        self.running = False
        
        # 控制参数
        self.control_rate = 20  # Hz
        self.wheel_base = 0.18  # 轮距
        self.wheel_radius = 0.0325  # 轮半径
        
    def start(self):
        """启动系统"""
        print("=" * 50)
        print("遥控绘图小车启动")
        print("=" * 50)
        
        self.running = True
        
        # 启动控制线程
        self.control_thread = threading.Thread(target=self._control_loop)
        self.control_thread.start()
        
        # 启动视觉定位线程
        self.vision_thread = threading.Thread(target=self._vision_loop)
        self.vision_thread.start()
        
        print("系统就绪，等待命令...")
        
    def _control_loop(self):
        """主控制循环"""
        while self.running:
            if self.is_drawing and self.current_path:
                # 获取当前位置
                pos = self.vision.detect_marker(marker_id=0)
                if pos:
                    self.current_pos = np.array([pos[0], pos[1]])
                    self.current_theta = np.radians(pos[2])
                    
                # 计算控制命令
                command, new_index = self.tracker.compute_control(
                    self.current_pos, self.current_theta,
                    self.current_path, self.current_index
                )
                
                self.current_index = new_index
                
                # 设置笔状态
                self.pen.set_pen_state(command.pen_state)
                
                # 计算左右轮速度
                v_left = command.linear_velocity - command.angular_velocity * self.wheel_base / 2
                v_right = command.linear_velocity + command.angular_velocity * self.wheel_base / 2
                
                # 转换为PWM值
                pwm_left = int(v_left / self.wheel_radius * 100 / 10)
                pwm_right = int(v_right / self.wheel_radius * 100 / 10)
                
                # 设置电机
                self.motor.set_motors(pwm_left, pwm_right)
                
                # 检查是否完成
                if self.current_index >= len(self.current_path) - 1:
                    print("绘图完成！")
                    self.stop_drawing()
                    
            else:
                self.motor.stop()
                
            time.sleep(1.0 / self.control_rate)
            
    def _vision_loop(self):
        """视觉定位循环"""
        while self.running:
            pos = self.vision.detect_marker(marker_id=0)
            if pos:
                self.current_pos = np.array([pos[0], pos[1]])
                self.current_theta = np.radians(pos[2])
            time.sleep(0.05)  # 20Hz
            
    def draw_shape(self, shape_type, **kwargs):
        """绘制指定图形"""
        print(f"开始绘制: {shape_type.value}")
        
        # 生成路径
        if shape_type == ShapeType.CIRCLE:
            self.current_path = self.path_generator.generate_circle(
                radius=kwargs.get('radius', 0.3)
            )
        elif shape_type == ShapeType.TRIANGLE:
            self.current_path = self.path_generator.generate_triangle(
                side_length=kwargs.get('side_length', 0.5)
            )
        elif shape_type == ShapeType.RECTANGLE:
            self.current_path = self.path_generator.generate_rectangle(
                width=kwargs.get('width', 0.5),
                height=kwargs.get('height', 0.3)
            )
        elif shape_type == ShapeType.PENTAGRAM:
            self.current_path = self.path_generator.generate_pentagram(
                outer_radius=kwargs.get('outer_radius', 0.3)
            )
        else:
            print(f"未知图形类型: {shape_type}")
            return
            
        self.current_index = 0
        self.is_drawing = True
        print(f"路径点数: {len(self.current_path)}")
        
    def stop_drawing(self):
        """停止绘图"""
        self.is_drawing = False
        self.motor.stop()
        self.pen.pen_up()
        print("已停止绘图")
        
    def remote_control(self, linear_vel, angular_vel):
        """遥控模式"""
        if not self.is_drawing:
            v_left = linear_vel - angular_vel * self.wheel_base / 2
            v_right = linear_vel + angular_vel * self.wheel_base / 2
            
            pwm_left = int(v_left * 100)
            pwm_right = int(v_right * 100)
            
            self.motor.set_motors(pwm_left, pwm_right)
            
    def stop(self):
        """停止系统"""
        self.running = False
        self.stop_drawing()
        self.motor.cleanup()
        self.pen.cleanup()
        self.vision.release()
        print("系统已停止")

# 主程序入口
if __name__ == '__main__':

    try:
