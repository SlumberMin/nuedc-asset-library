### 2.1 数字识别模块

```python
import cv2
import numpy as np
from picamera2 import Picamera2

class NumberRecognizer:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # 使用模板匹配或CNN进行数字识别
        self.templates = self.load_templates()
        
    def load_templates(self):
        """加载数字模板"""
        templates = {}
        for i in range(1, 10):
            img = cv2.imread(f'templates/num_{i}.png', cv2.IMREAD_GRAYSCALE)
            if img is not None:
                templates[i] = cv2.resize(img, (40, 60))
        return templates
        
    def recognize_number(self, roi):
        """识别ROI区域中的数字"""
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY) if len(roi.shape) == 3 else roi
        gray = cv2.resize(gray, (40, 60))
        
        best_score = 0
        best_num = 0
        
        for num, template in self.templates.items():
            result = cv2.matchTemplate(gray, template, cv2.TM_CCOEFF_NORMED)
            score = result.max()
            if score > best_score:
                best_score = score
                best_num = num
                
        return best_num if best_score > 0.6 else 0
        
    def detect_room_signs(self):
        """检测房间号标识"""
        frame = self.picam2.capture_array()
        hsv = cv2.cvtColor(frame, cv2.COLOR_RGB2HSV)
        
        # 检测白色底板上的黑色数字
        lower_white = np.array([0, 0, 200])
        upper_white = np.array([180, 30, 255])
        mask = cv2.inRange(hsv, lower_white, upper_white)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        results = []
        for cnt in contours:
            area = cv2.contourArea(cnt)
            if 1000 < area < 20000:
                x, y, w, h = cv2.boundingRect(cnt)
                aspect = w / h
                if 0.5 < aspect < 2.0:
                    roi = frame[y:y+h, x:x+w]
                    number = self.recognize_number(roi)
                    if number > 0:
                        cx = x + w // 2
                        results.append({
                            'number': number,
                            'center': (cx, y + h // 2),
                            'bbox': (x, y, w, h),
                            'area': area
                        })
        
        return results
```

### 2.2 地图与路径规划

```python
import heapq

class DeliveryMap:
    def __init__(self, grid_size=10):
        self.grid_size = grid_size
        self.grid = np.zeros((grid_size, grid_size), dtype=np.int8)
        # 0=通路, 1=障碍, 2=房间, 3=药房
        self.rooms = {}  # 房间号 -> 坐标
        self.drug_room = None  # 药房坐标
        
    def build_map(self, detected_signs, car_pos):
        """根据检测到的标识构建地图"""
        for sign in detected_signs:
            # 根据相对位置计算房间在地图中的位置
            grid_x, grid_y = self.pixel_to_grid(sign['center'], car_pos)
            self.rooms[sign['number']] = (grid_x, grid_y)
            self.grid[grid_y, grid_x] = 2
            
    def pixel_to_grid(self, pixel_pos, car_pos):
        """像素坐标转栅格坐标"""
        dx = pixel_pos[0] - 320
        dy = pixel_pos[1] - 240
        grid_x = car_pos[0] + int(dx / 100)
        grid_y = car_pos[1] + int(dy / 100)
        return grid_x, grid_y
        
    def find_path(self, start, goal):
        """A*寻路算法"""
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == goal:
                path = [current]
                while current in came_from:
                    current = came_from[current]
                    path.append(current)
                return path[::-1]
            
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                neighbor = (current[0]+dx, current[1]+dy)
                if (0 <= neighbor[0] < self.grid_size and 
                    0 <= neighbor[1] < self.grid_size and
                    self.grid[neighbor[1], neighbor[0]] != 1):
                    
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score = tentative_g + heuristic(neighbor, goal)
                        heapq.heappush(open_set, (f_score, neighbor))
        
        return None  # 无路径
```

### 2.3 循线导航

```python
class LineFollower:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
    def detect_line(self):
        """检测地面黑线"""
        frame = self.picam2.capture_array()
        # 取底部1/3区域
        roi = frame[320:480, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        _, binary = cv2.threshold(gray, 60, 255, cv2.THRESH_BINARY_INV)
        
        # 扫描找到黑线位置
        mid_row = binary[60, :]
        line_positions = np.where(mid_row > 0)[0]
        
        if len(line_positions) > 0:
            line_center = int(np.mean(line_positions))
            offset = line_center - 320  # 偏移量
            return True, offset
        
        return False, 0
        
    def detect_intersection(self):
        """检测路口（十字/T字）"""
        frame = self.picam2.capture_array()
        roi = frame[300:480, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        _, binary = cv2.threshold(gray, 60, 255, cv2.THRESH_BINARY_INV)
        
        # 分析水平和垂直线条
        h_lines = np.sum(binary[80, :]) // 255
        v_left = np.sum(binary[:, 160]) // 255
        v_right = np.sum(binary[:, 480]) // 255
        
        has_left = v_left > 30
        has_right = v_right > 30
        has_forward = h_lines > 200
        
        if has_left and has_right and has_forward:
            return 'CROSS'  # 十字路口
        elif has_left or has_right:
            return 'T'      # T字路口
        else:
            return 'NONE'
```

    uint8_t target_room;    // 目标房间号
    uint8_t action;         // 0=取药, 1=送药
} Task_t;

// PID结构体
typedef struct {
    float kp, ki, kd;
    float integral, prev_error;
} PID_t;

int16_t PID_Compute(PID_t *pid, float error) {
    pid->integral += error;
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    return (int16_t)(pid->kp * error + pid->ki * pid->integral + pid->kd * derivative);
}

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2021年F题 - 智能送药小车 主程序
"""
import cv2
import numpy as np
import serial
import time
from collections import deque

class SmartDeliveryCar:
    def __init__(self):
        # 模块初始化
        self.number_rec = NumberRecognizer()
        self.line_follower = LineFollower()
        self.delivery_map = DeliveryMap(grid_size=10)
        
        # 串口
        self.ser = serial.Serial('/dev/ttyS5', 115200, timeout=0.01)
        
        # 状态
        self.state = 'EXPLORE'  # EXPLORE, DELIVER, RETURN
        self.current_pos = (0, 0)
        self.current_heading = 0  # 0=北, 1=东, 2=南, 3=西
        self.task_queue = deque()
        self.explored_rooms = {}
        
    def send_command(self, cmd_type, param1=0, param2=0):
        cmd = bytes([0xAA, cmd_type, param1 & 0xFF, param2 & 0xFF, 0x55])
        self.ser.write(cmd)
        
    def read_feedback(self):
        if self.ser.in_waiting >= 4:
            data = self.ser.read(4)
            if data[0] == 0xAA and data[3] == 0x55:
                return {'status': data[1], 'value': data[2]}
        return None
        
    def follow_line(self):
        """循线行驶"""
        detected, offset = self.line_follower.detect_line()
        if detected:
            kp = 0.5
            steering = int(kp * offset)
            left = np.clip(100 + steering, 0, 200)
            right = np.clip(100 - steering, 0, 200)
            self.send_command(0x01, left, right)
            return True
        else:
            self.send_command(0x00, 0, 0)  # 停止
            return False
            
    def handle_intersection(self, direction):
        """处理路口转弯"""
        if direction == 'left':
            self.send_command(0x02)  # 左转命令
            self.current_heading = (self.current_heading - 1) % 4
        elif direction == 'right':
            self.send_command(0x03)  # 右转命令
            self.current_heading = (self.current_heading + 1) % 4
        elif direction == 'forward':
            pass  # 继续直行
        time.sleep(1.5)  # 等待转弯完成
        
    def update_position(self):
        """更新位置"""
        direction_delta = [(0, 1), (1, 0), (0, -1), (-1, 0)]
        dx, dy = direction_delta[self.current_heading]
        self.current_pos = (self.current_pos[0] + dx, self.current_pos[1] + dy)
        
    def explore_phase(self):
        """探索阶段：遍历地图识别房间"""
        signs = self.number_rec.detect_room_signs()
        for sign in signs:
            if sign['number'] not in self.explored_rooms:
                self.explored_rooms[sign['number']] = self.current_pos
                print(f"发现房间 {sign['number']} 在位置 {self.current_pos}")
                
        # 循线行驶
        if self.follow_line():
            intersection = self.line_follower.detect_intersection()
            if intersection != 'NONE':
                self.update_position()
                
    def deliver_phase(self):
        """送药阶段：规划路径送药到指定房间"""
        if not self.task_queue:
            return
            
        target_room = self.task_queue[0]
        if target_room in self.explored_rooms:
            target_pos = self.explored_rooms[target_room]
            path = self.delivery_map.find_path(self.current_pos, target_pos)
            
            if path and len(path) > 1:
                next_pos = path[1]
                self.navigate_to(next_pos)
                
                if self.current_pos == target_pos:
                    print(f"到达房间 {target_room}，执行送药")
                    self.send_command(0x10)  # 送药动作
                    time.sleep(2)
                    self.task_queue.popleft()
                    
    def navigate_to(self, target_pos):
        """导航到目标位置"""
        dx = target_pos[0] - self.current_pos[0]
        dy = target_pos[1] - self.current_pos[1]
        
        # 计算需要的朝向
        if dx > 0: target_heading = 1  # 东
        elif dx < 0: target_heading = 3  # 西
        elif dy > 0: target_heading = 0  # 北
        else: target_heading = 2  # 南
        
        # 转向
        turn_diff = (target_heading - self.current_heading) % 4
        if turn_diff == 1:
            self.handle_intersection('right')
        elif turn_diff == 3:
            self.handle_intersection('left')
        elif turn_diff == 2:
            self.handle_intersection('left')
            self.handle_intersection('left')
            
    def run(self):
        print("智能送药小车启动...")
        while True:
            if self.state == 'EXPLORE':
                self.explore_phase()
                if len(self.explored_rooms) >= 4:  # 找到所有房间
                    self.state = 'DELIVER'
                    print("探索完成，切换到送药模式")
                    
            elif self.state == 'DELIVER':
                self.deliver_phase()
                if not self.task_queue:
                    self.state = 'EXPLORE'
                    
            feedback = self.read_feedback()
            if feedback:
                print(f"反馈: {feedback}")
                
            time.sleep(0.03)

if __name__ == '__main__':
    car = SmartDeliveryCar()
    # 添加送药任务
    car.task_queue.append(3)  # 送到3号房间
    car.task_queue.append(5)  # 送到5号房间
    car.run()
```
