# 2015年 多旋翼自主飞行器 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个多旋翼自主飞行器，能够自主完成起飞、悬停、航线飞行、降落等任务。需通过视觉系统实现定位，不依赖GPS。

### 1.2 关键指标
- 飞行器类型：四旋翼/六旋翼
- 起飞重量：<2kg
- 飞行时间：>10分钟
- 定位精度：±10cm（水平），±5cm（垂直）
- 任务：定点悬停、航线飞行、目标降落

### 1.3 技术难点
1. **视觉定位**：无GPS环境下实现精确定位
2. **姿态控制**：保持飞行稳定
3. **避障**：检测并避开障碍物
4. **自主决策**：根据任务自主规划航线

## 四、视觉定位算法

### 4.1 光流定位（下视）
```python
import cv2
import numpy as np

class OpticalFlowTracker:
    def __init__(self, camera_height=1.0):
        self.camera_height = camera_height  # 相机高度(m)
        
        # 光流参数
        self.feature_params = dict(
            maxCorners=100,
            qualityLevel=0.3,
            minDistance=7,
            blockSize=7
        )
        
        self.lk_params = dict(
            winSize=(15, 15),
            maxLevel=2,
            criteria=(cv2.TERM_CRITERIA_EPS | cv2.TERM_CRITERIA_COUNT, 10, 0.03)
        )
        
        self.prev_gray = None
        self.prev_points = None
        self.position = np.array([0.0, 0.0])  # 累积位置
        self.scale = None  # 像素到米的比例
    
    def calibrate_scale(self, frame, known_distance_m=1.0):
        """标定像素到米的比例"""
        # 在地面上放置已知距离的标记
        # 检测标记并计算比例
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 检测角点
        points = cv2.goodFeaturesToTrack(gray, **self.feature_params)
        
        if points is not None and len(points) >= 2:
            # 假设前两个点是标定点
            p1, p2 = points[0][0], points[1][0]
            pixel_distance = np.linalg.norm(p2 - p1)
            
            self.scale = known_distance_m / pixel_distance
            return True
        
        return False
    
    def update(self, frame):
        """更新光流计算"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        if self.prev_gray is None:
            self.prev_gray = gray
            self.prev_points = cv2.goodFeaturesToTrack(gray, **self.feature_params)
            return self.position
        
        if self.prev_points is None or len(self.prev_points) < 10:
            self.prev_points = cv2.goodFeaturesToTrack(self.prev_gray, **self.feature_params)
        
        if self.prev_points is not None:
            # 计算光流
            next_points, status, _ = cv2.calcOpticalFlowPyrLK(
                self.prev_gray, gray, self.prev_points, None, **self.lk_params)
            
            # 筛选好的跟踪点
            good_old = self.prev_points[status == 1]
            good_new = next_points[status == 1]
            
            if len(good_old) > 5:
                # 计算平均位移
                displacements = good_new - good_old
                avg_displacement = np.mean(displacements, axis=0)
                
                # 转换为米
                if self.scale:
                    movement_m = avg_displacement * self.scale
                else:
                    # 使用默认比例（需要标定）
                    movement_m = avg_displacement * 0.001
                
                # 累积位置
                self.position += movement_m
        
        # 更新状态
        self.prev_gray = gray
        self.prev_points = cv2.goodFeaturesToTrack(gray, **self.feature_params)
        
        return self.position
    
    def get_velocity(self):
        """获取当前速度估计"""
        if hasattr(self, 'prev_position'):
            velocity = (self.position - self.prev_position) * 30  # 假设30fps
            self.prev_position = self.position.copy()
            return velocity
        
        self.prev_position = self.position.copy()
        return np.array([0.0, 0.0])
```

### 4.2 双目视觉避障
```python
class StereoObstacleDetector:
    def __init__(self, baseline=0.06, focal_length=600):
        self.baseline = baseline  # 双目基线(m)
        self.focal_length = focal_length  # 焦距(pixels)
        
        # 立体匹配参数
        self.stereo = cv2.StereoBM_create(numDisparities=64, blockSize=15)
        self.stereo.setMinDisparity(0)
        self.stereo.setNumDisparities(64)
        self.stereo.setBlockSize(15)
        self.stereo.setPreFilterCap(31)
        self.stereo.setUniquenessRatio(10)
        self.stereo.setSpeckleWindowSize(100)
        self.stereo.setSpeckleRange(32)
    
    def compute_depth(self, left_frame, right_frame):
        """计算深度图"""
        # 转换为灰度
        left_gray = cv2.cvtColor(left_frame, cv2.COLOR_BGR2GRAY)
        right_gray = cv2.cvtColor(right_frame, cv2.COLOR_BGR2GRAY)
        
        # 计算视差图
        disparity = self.stereo.compute(left_gray, right_gray)
        
        # 转换为深度图
        # depth = baseline * focal_length / disparity
        depth = np.zeros_like(disparity, dtype=np.float32)
        valid_mask = disparity > 0
        depth[valid_mask] = (self.baseline * self.focal_length) / disparity[valid_mask]
        
        return depth
    
    def detect_obstacles(self, depth_map, threshold=2.0):
        """
        检测障碍物
        threshold: 障碍物距离阈值(m)
        """
        h, w = depth_map.shape
        
        # 划分区域
        regions = {
            'left': depth_map[:, :w//3],
            'center': depth_map[:, w//3:2*w//3],
            'right': depth_map[:, 2*w//3:]
        }
        
        obstacles = {}
        for region_name, region_depth in regions.items():
            # 近距离像素
            near_pixels = region_depth[(region_depth > 0) & (region_depth < threshold)]
            
            if len(near_pixels) > 100:  # 足够多的近距离点
                obstacles[region_name] = {
                    'distance': np.mean(near_pixels),
                    'density': len(near_pixels) / region_depth.size
                }
        
        return obstacles
    
    def get_avoidance_command(self, obstacles):
        """根据障碍物生成避障命令"""
        if not obstacles:
            return 'forward', 0  # 无障碍，前进
        
        # 选择最安全的方向
        if 'left' not in obstacles and 'center' not in obstacles:
            return 'turn_left', 30
        elif 'right' not in obstacles and 'center' not in obstacles:
            return 'turn_right', 30
        elif 'center' in obstacles:
            # 中间有障碍，选择左右较宽的一侧
            left_dist = obstacles.get('left', {}).get('distance', 0)
            right_dist = obstacles.get('right', {}).get('distance', 0)
            
            if left_dist > right_dist:
                return 'turn_left', 45
            else:
                return 'turn_right', 45
        else:
            return 'hover', 0  # 悬停等待
```

### 4.3 SLAM地图构建
```python
class VisualSLAM:
    def __init__(self):
        # ORB特征提取器
        self.orb = cv2.ORB_create(nfeatures=1000)
        
        # 特征匹配器
        self.bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=True)
        
        # 地图点
        self.map_points = []
        self.keyframes = []
        
        # 当前位姿
        self.current_pose = np.eye(4)
    
    def process_frame(self, frame):
        """处理新帧"""
        # 提取特征
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        keypoints, descriptors = self.orb.detectAndCompute(gray, None)
        
        if len(self.keyframes) == 0:
            # 第一帧
            self.keyframes.append({
                'keypoints': keypoints,
                'descriptors': descriptors,
                'pose': self.current_pose.copy()
            })
            return self.current_pose
        
        # 与上一关键帧匹配
        last_kf = self.keyframes[-1]
        matches = self.bf.match(last_kf['descriptors'], descriptors)
        matches = sorted(matches, key=lambda x: x.distance)
        
        if len(matches) > 20:
            # 估计运动
            src_pts = np.float32([last_kf['keypoints'][m.queryIdx].pt for m in matches])
            dst_pts = np.float32([keypoints[m.trainIdx].pt for m in matches])
            
            # 计算本质矩阵
            E, mask = cv2.findEssentialMat(src_pts, dst_pts, focal=600, 
                                            pp=(320, 240), method=cv2.RANSAC)
            
            if E is not None:
                # 恢复位姿
                _, R, t, mask = cv2.recoverPose(E, src_pts, dst_pts, 
                                                  focal=600, pp=(320, 240))
                
                # 更新位姿
                transform = np.eye(4)
                transform[:3, :3] = R
                transform[:3, 3] = t.flatten()
                
                self.current_pose = self.current_pose @ transform
                
                # 是否需要新的关键帧
                if self._need_new_keyframe(matches):
                    self.keyframes.append({
                        'keypoints': keypoints,
                        'descriptors': descriptors,
                        'pose': self.current_pose.copy()
                    })
        
        return self.current_pose
    
    def _need_new_keyframe(self, matches):
        """判断是否需要新的关键帧"""
        # 匹配点数少于阈值
        if len(matches) < 50:
            return True
        
        # 距离上一关键帧足够远
        last_pose = self.keyframes[-1]['pose']
        translation = np.linalg.norm(self.current_pose[:3, 3] - last_pose[:3, 3])
        
        if translation > 0.5:  # 移动超过0.5m
            return True
        
        return False
    
    def get_position(self):
        """获取当前位置"""
        return self.current_pose[:3, 3]
    
    def get_heading(self):
        """获取当前航向角"""
        R = self.current_pose[:3, :3]
        heading = np.arctan2(R[1, 0], R[0, 0])
        return np.degrees(heading)
```

```python
import numpy as np

class AttitudeController:
    def __init__(self):
        # PID参数
        self.roll_pid = PIDController(Kp=1.5, Ki=0.1, Kd=0.3)
        self.pitch_pid = PIDController(Kp=1.5, Ki=0.1, Kd=0.3)
        self.yaw_pid = PIDController(Kp=2.0, Ki=0.05, Kd=0.5)
        self.alt_pid = PIDController(Kp=2.0, Ki=0.3, Kd=0.8)
    
    def compute(self, target, current):
        """
        计算控制输出
        target: 目标状态 [roll, pitch, yaw, throttle]
        current: 当前状态 [roll, pitch, yaw, altitude]
        """
        # 各轴PID控制
        roll_output = self.roll_pid.update(target[0] - current[0])
        pitch_output = self.pitch_pid.update(target[1] - current[1])
        yaw_output = self.yaw_pid.update(target[2] - current[2])
        alt_output = self.alt_pid.update(target[3] - current[3])
        
        return roll_output, pitch_output, yaw_output, alt_output

class PIDController:
    def __init__(self, Kp, Ki, Kd):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.prev_error = 0
        self.integral = 0
    
    def update(self, error):
        self.integral += error
        self.integral = max(-100, min(100, self.integral))  # 积分限幅
        derivative = error - self.prev_error
        self.prev_error = error
        
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative
```

### 5.2 电机混控
```python
class MotorMixer:
    def __init__(self, motor_count=4):
        self.motor_count = motor_count
        
        # X型四旋翼混控矩阵
        # [roll, pitch, yaw, throttle]
        if motor_count == 4:
            self.mix_matrix = np.array([
                [ 1, -1, -1, 1],  # 前左 (M1)
                [ 1,  1,  1, 1],  # 前右 (M2)
                [-1,  1, -1, 1],  # 后右 (M3)
                [-1, -1,  1, 1],  # 后左 (M4)
            ])
    
    def mix(self, roll, pitch, yaw, throttle):
        """混控计算"""
        control = np.array([roll, pitch, yaw, throttle])
        
        # 矩阵乘法
        motor_outputs = self.mix_matrix @ control
        
        # 归一化到0~1000
        motor_outputs = np.clip(motor_outputs, 0, 1000)
        
        return motor_outputs.astype(int)
```

### 5.3 位置控制器
```python
class PositionController:
    def __init__(self):
        self.x_pid = PIDController(Kp=0.8, Ki=0.05, Kd=0.3)
        self.y_pid = PIDController(Kp=0.8, Ki=0.05, Kd=0.3)
        self.z_pid = PIDController(Kp=1.5, Ki=0.1, Kd=0.5)
        
        self.max_tilt = 15  # 最大倾斜角(度)
    
    def compute(self, target_pos, current_pos, target_heading, current_heading):
        """
        计算姿态指令
        target_pos: 目标位置 [x, y, z]
        current_pos: 当前位置 [x, y, z]
        """
        # 位置误差
        error_x = target_pos[0] - current_pos[0]
        error_y = target_pos[1] - current_pos[1]
        error_z = target_pos[2] - current_pos[2]
        
        # PID控制
        pitch_cmd = self.x_pid.update(error_x)
        roll_cmd = self.y_pid.update(error_y)
        throttle_cmd = self.z_pid.update(error_z)
        
        # 航向控制
        heading_error = target_heading - current_heading
        # 归一化到[-180, 180]
        heading_error = (heading_error + 180) % 360 - 180
        yaw_cmd = heading_error * 0.5
        
        # 限幅
        roll_cmd = max(-self.max_tilt, min(self.max_tilt, roll_cmd))
        pitch_cmd = max(-self.max_tilt, min(self.max_tilt, pitch_cmd))
        
        return roll_cmd, pitch_cmd, yaw_cmd, throttle_cmd
```

```python
        if target is None:
            return True
        
        distance = np.linalg.norm(target['position'] - current_pos)
        
        if distance < threshold:
            # 到达航点
            if target['hover_time'] > 0:
                # 需要悬停

            self.current_waypoint += 1
            return True
        
        return False
    
    def generate_mission(self, task_type='square'):
        """生成示例任务"""
        if task_type == 'square':
            # 方形航线
            size = 2.0
            altitude = 1.5
            
            self.add_waypoint(0, 0, altitude, hover_time=3)  # 起飞悬停
            self.add_waypoint(size, 0, altitude)
            self.add_waypoint(size, size, altitude)
            self.add_waypoint(0, size, altitude)
            self.add_waypoint(0, 0, altitude, hover_time=3)  # 返回悬停
        
        elif task_type == 'figure8':

                y = radius * np.sin(2*angle) / 2
```

```python
            self.send_motor_command(throttle, throttle, throttle, throttle)
            time.sleep(0.1)
        
        self.flying = True
        print("已到达目标高度")
    
    def land(self):
        """自动降落"""
        print("降落中...")
        
        # 缓慢降低油门
        throttle = 400
        while self.current_position[2] > 0.1:
            throttle = max(throttle - 5, 200)
            self.send_motor_command(throttle, throttle, throttle, throttle)
            time.sleep(0.1)
        
        # 关闭电机
        self.send_motor_command(0, 0, 0, 0)
        self.flying = False
        self.armed = False
        print("已降落")
    
    def run_mission(self):
        """执行任务"""
        # 生成任务
        self.mission.generate_mission('square')
        
        # 起飞
        self.takeoff(1.5)
        
        # 执行任务
        while not self.mission.mission_complete:
            # 更新位置
            self.update_position()
            
            # 获取目标
            target = self.mission.get_current_target()
            
            if target is None:
                break
            
            # 检查避障
            obstacles = self.check_obstacles()
            
            if obstacles:
                # 执行避障
                self.avoid_obstacles(obstacles)
            else:
                # 飞向目标
                self.fly_to_target(target)
            
            # 检查是否到达航点
            if self.mission.check_waypoint_reached(self.current_position):
                print(f"到达航点 {self.mission.current_waypoint}")
            
            time.sleep(0.033)  # 30Hz
        
        # 降落
        self.land()
    
    def update_position(self):
        """更新位置估计"""
        # 光流定位
        down_frame = self.down_camera.capture()
        flow_position = self.optflow.update(down_frame)
        
        # SLAM定位
        front_frame = self.stereo_camera.capture_left()
        slam_pose = self.slam.process_frame(front_frame)
        slam_position = self.slam.get_position()
        
        # 传感器融合（简化版，实际应使用EKF）
        # 这里简单取平均
        self.current_position = (flow_position + slam_position) / 2
        self.current_position[2] = self.get_altitude()  # 使用超声波/气压计高度
    
    def fly_to_target(self, target):
        """飞向目标"""
        # 位置控制
        roll, pitch, yaw, throttle = self.pos_ctrl.compute(
            target['position'],
            self.current_position,
            target['heading'],
            self.current_attitude[2]
        )
        
        # 混控
        motors = self.mixer.mix(roll, pitch, yaw, throttle)
        
        # 发送电机指令
        self.send_motor_command(*motors)
    
    def check_obstacles(self):
        """检查障碍物"""
        left_frame, right_frame = self.stereo_camera.capture()
        depth = self.stereo.compute_depth(left_frame, right_frame)
        obstacles = self.stereo.detect_obstacles(depth, threshold=1.5)
        
        return obstacles
    
    def avoid_obstacles(self, obstacles):
        """避障动作"""
        action, angle = self.stereo.get_avoidance_command(obstacles)
        
        if action == 'hover':
            pass  # 悬停
        elif action == 'turn_left':
            # 左转
            self.send_motor_command(300, 500, 300, 500)
        elif action == 'turn_right':
            # 右转
            self.send_motor_command(500, 300, 500, 300)
    
    def send_motor_command(self, m1, m2, m3, m4):
        """发送电机指令"""
        cmd = f"M,{m1},{m2},{m3},{m4}\n"
        self.flight_controller.write(cmd.encode())
    
    def send_command(self, cmd):
        """发送通用命令"""
        self.flight_controller.write(f"{cmd}\n".encode())

if __name__ == '__main__':
    flight = AutonomousFlight()
```

## 八、安全机制

### 8.1 安全保护
```python
class SafetyMonitor:
    def __init__(self, flight_controller):
        self.fc = flight_controller
        
        # 安全限制
        self.max_altitude = 10.0  # 最大高度(m)
        self.max_tilt = 30.0  # 最大倾斜角(度)
        self.max_speed = 5.0  # 最大速度(m/s)
        self.battery_warning = 3.5  # 电池警告电压(V/cell)
        self.battery_critical = 3.3  # 电池危险电压
        
        # 失联处理
        self.last_heartbeat = time.time()
        self.heartbeat_timeout = 5.0  # 秒
    
    def check_safety(self, state):
        """检查安全状态"""
        warnings = []
        
        # 高度检查
        if state['altitude'] > self.max_altitude:
            warnings.append('ALTITUDE_EXCEEDED')
        
        # 姿态检查
        if abs(state['roll']) > self.max_tilt or abs(state['pitch']) > self.max_tilt:
            warnings.append('ATTITUDE_EXCEEDED')
        
        # 电池检查
        if state['battery_voltage'] < self.battery_critical:
            warnings.append('BATTERY_CRITICAL')
        elif state['battery_voltage'] < self.battery_warning:
            warnings.append('BATTERY_LOW')
        
        # 失联检查
        if time.time() - self.last_heartbeat > self.heartbeat_timeout:
            warnings.append('CONNECTION_LOST')
        
        return warnings
    
    def handle_warnings(self, warnings):
        """处理警告"""
        if 'BATTERY_CRITICAL' in warnings:
            print("电池危险！立即降落！")
            return 'LAND_NOW'
        
        if 'CONNECTION_LOST' in warnings:
            print("失去连接！执行返航！")
            return 'RETURN_HOME'
        
        if 'ALTITUDE_EXCEEDED' in warnings:
            print("超出高度限制！")
            return 'DESCEND'
        
        if 'ATTITUDE_EXCEEDED' in warnings:
            print("姿态异常！")
            return 'STABILIZE'
        
        return 'OK'
```
