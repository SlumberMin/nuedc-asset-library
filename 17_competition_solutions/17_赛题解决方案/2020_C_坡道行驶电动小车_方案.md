## 二、视觉方案（Orange Pi 5）

### 2.1 坡度视觉检测

```python
import cv2
import numpy as np
from picamera2 import Picamera2
import math

class SlopeVisualDetector:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        self.prev_line_angle = 0
        
    def detect_slope_from_lines(self):
        """通过地面线条透视变化检测坡度"""
        frame = self.picam2.capture_array()
        
        # 取上半部分图像
        roi = frame[0:320, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        
        # 边缘检测
        edges = cv2.Canny(gray, 50, 150)
        
        # 霍夫变换检测直线
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50, 
                                minLineLength=30, maxLineGap=10)
        
        if lines is not None:
            # 计算线条角度分布
            angles = []
            for line in lines:
                x1, y1, x2, y2 = line[0]
                angle = math.degrees(math.atan2(y2-y1, x2-x1))
                if abs(angle) < 60:  # 过滤接近水平的线
                    angles.append(angle)
            
            if angles:
                avg_angle = np.mean(angles)
                # 角度变化表示坡度
                slope_angle = abs(avg_angle) * 0.5  # 简化映射
                return min(slope_angle, 45)
        
        return 0
        
    def detect_line_on_slope(self):
        """坡道上的循线检测"""
        frame = self.picam2.capture_array()
        
        # 使用更低的ROI（适应坡道视角变化）
        roi = frame[350:480, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        
        # 自适应阈值（坡道光照可能不均匀）
        binary = cv2.adaptiveThreshold(
            gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 15, 3
        )
        
        # 扫描中线
        mid_row = binary[50, :]
        line_pos = np.where(mid_row > 0)[0]
        
        if len(line_pos) > 10:
            center = int(np.mean(line_pos))
            offset = center - 320
            return True, offset
        
        return False, 0
```

### 2.2 坡道特征融合

```python
class SlopeAnalyzer:
    def __init__(self):
        self.visual_detector = SlopeVisualDetector()
        self.slope_history = []
        
    def analyze_slope(self, gyro_pitch, accel_z):
        """融合多种传感器分析坡度"""
        # 1. 陀螺仪坡度
        gyro_slope = abs(gyro_pitch)
        
        # 2. 加速度计坡度
        accel_slope = math.degrees(math.acos(np.clip(accel_z / 9.81, -1, 1)))
        
        # 3. 视觉坡度
        visual_slope = self.visual_detector.detect_slope_from_lines()
        
        # 加权融合
        fused_slope = (
            0.5 * gyro_slope +
            0.3 * accel_slope +
            0.2 * visual_slope
        )
        
        # 平滑滤波
        self.slope_history.append(fused_slope)
        if len(self.slope_history) > 5:
            self.slope_history.pop(0)
        
        return np.mean(self.slope_history)
```

### 3.1 自适应速度控制

```c
// 坡度自适应PID
typedef struct {
    float base_speed;
    float slope_compensation;
    float max_speed;
    float min_speed;
} SpeedController_t;

typedef struct {
    float kp, ki, kd;
    float integral, prev_error;
} PID_t;

// 根据坡度调整速度
float Speed_GetTarget(SpeedController_t *ctrl, float slope_angle) {
    // 上坡：增加速度补偿
    if (slope_angle > 0) {
        ctrl->slope_compensation = slope_angle * 2.0;  // 每度增加2个速度单位
    } else {
        // 下坡：减速
        ctrl->slope_compensation = slope_angle * 1.0;
    }
    
    float target = ctrl->base_speed + ctrl->slope_compensation;
    return CLAMP(target, ctrl->min_speed, ctrl->max_speed);
}

// 陀螺仪数据获取
typedef struct {
    float pitch;  // 俯仰角
    float roll;   // 横滚角
    float yaw;    // 偏航角
    float accel_x, accel_y, accel_z;
} IMU_Data_t;

void MPU6050_GetSlope(IMU_Data_t *imu) {
    // 互补滤波
    float accel_pitch = atan2(imu->accel_y, sqrt(imu->accel_x*imu->accel_x + imu->accel_z*imu->accel_z));
    float gyro_pitch_rate = imu->pitch;  // 陀螺仪角速度
    
    static float filtered_pitch = 0;
    filtered_pitch = 0.98 * (filtered_pitch + gyro_pitch_rate * 0.01) + 0.02 * accel_pitch;
    
    imu->pitch = filtered_pitch * 180 / M_PI;
}
```

```python
class SlopeController:
    def __init__(self):
        self.ser = serial.Serial('/dev/ttyS5', 115200)
        self.base_speed = 100
        self.current_slope = 0
        
    def compute_speed(self, slope_angle, line_offset):
        """计算目标速度"""
        # 基础速度
        speed = self.base_speed
        
        # 坡度补偿
        if slope_angle > 5:  # 上坡
            speed += slope_angle * 1.5
        elif slope_angle < -5:  # 下坡
            speed += slope_angle * 0.8  # 减速
            
        # 偏移补偿（偏移大时减速）
        offset_penalty = abs(line_offset) / 100 * 30
        speed -= offset_penalty
        
        return int(np.clip(speed, 50, 200))
        
    def compute_steering(self, line_offset, slope_angle):
        """计算转向量"""
        # 基础PID转向
        kp = 0.5
        kd = 0.2
        
        steering = kp * line_offset
        
        # 坡道横滚补偿（如果有）
        # steering += slope_compensation
        
        return int(np.clip(steering, -100, 100))
```

```python
#!/usr/bin/env python3
"""
2020年C题 - 坡道行驶电动小车 主程序
"""
import cv2
import numpy as np
import serial
import time
import math

class SlopeDrivingCar:
    def __init__(self):
        self.slope_analyzer = SlopeAnalyzer()
        self.line_follower = VisualLineFollower()
        self.speed_ctrl = SlopeController()
        
        self.ser = serial.Serial('/dev/ttyS5', 115200, timeout=0.01)
        
        # PID参数
        self.line_kp = 0.6
        self.line_ki = 0.005
        self.line_kd = 0.3
        self.line_integral = 0
        self.line_prev_error = 0
        
        self.slope_angle = 0
        self.on_slope = False
        
    def send_command(self, left, right):
        """发送电机命令"""
        left = int(np.clip(left, -255, 255))
        right = int(np.clip(right, -255, 255))
        
        cmd = bytes([
            0xAA,
            0x01 if left >= 0 else 0x00,
            abs(left),
            0x01 if right >= 0 else 0x00,
            abs(right),
            0x55
        ])
        self.ser.write(cmd)
        
    def read_imu(self):
        """读取IMU数据"""
        if self.ser.in_waiting >= 8:
            data = self.ser.read(8)
            if data[0] == 0xBB:
                # 解析坡度数据
                slope_raw = (data[1] << 8) | data[2]
                if slope_raw > 32767:
                    slope_raw -= 65536
                self.slope_angle = slope_raw / 100.0
                return True
        return False
        
    def line_pid(self, error):
        """循线PID"""
        self.line_integral += error
        self.line_integral = np.clip(self.line_integral, -100, 100)
        derivative = error - self.line_prev_error
        self.line_prev_error = error
        
        return self.line_kp * error + self.line_ki * self.line_integral + self.line_kd * derivative
        
    def run(self):
        print("坡道行驶小车启动...")
        
        while True:
            # 读取IMU
            self.read_imu()
            
            # 检测是否在坡道上
            self.on_slope = abs(self.slope_angle) > 3
            
            # 循线检测
            detected, line_offset = self.line_follower.detect_line_on_slope()
            
            if detected:
                # 计算转向
                steering = self.line_pid(line_offset)
                
                # 计算速度（自适应坡度）
                forward = self.speed_ctrl.compute_speed(self.slope_angle, line_offset)
                
                # 计算左右轮速
                left = int(forward + steering)
                right = int(forward - steering)
                
                self.send_command(left, right)
                
                status = "上坡" if self.slope_angle > 3 else ("下坡" if self.slope_angle < -3 else "平地")
                print(f"{status} 坡度:{self.slope_angle:.1f}° "
                      f"偏移:{line_offset:.0f} 速度:{forward}")
            else:
                # 丢失线，慢速搜索
                self.send_command(40, 40)
                self.line_integral = 0
                print("丢失目标...")
                
            time.sleep(0.02)

if __name__ == '__main__':
    car = SlopeDrivingCar()
    try:
        car.run()
    except KeyboardInterrupt:
        car.send_command(0, 0)
        print("停止")
```

## 七、调试要点

1. **坡度补偿系数**：从小值开始，逐步增加，避免振荡
2. **PID参数**：坡道上和平地可能需要不同参数
3. **电机扭矩**：确保电机功率足够爬坡
4. **重心位置**：电池位置调整，降低重心
5. **下坡控制**：适当制动，防止失控加速
