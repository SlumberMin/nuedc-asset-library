# 2010年C题 — 遥控绘图小车

## 题目概述
设计一个遥控绘图小车，能够接收遥控指令在地面上绘制图形。

## 差速驱动运动学

```python
import math
import time

class DifferentialDrive:
    """差速驱动运动学模型"""
    
    def __init__(self, wheel_base=0.15, wheel_radius=0.032):
        """
        wheel_base: 左右轮距(m)
        wheel_radius: 轮子半径(m)
        """
        self.L = wheel_base
        self.R = wheel_radius
        
        # 位置状态
        self.x = 0.0
        self.y = 0.0
        self.theta = 0.0
        
        # 编码器计数
        self.left_ticks = 0
        self.right_ticks = 0
        self.ticks_per_rev = 390  # 编码器每转脉冲数
    
    def update_odometry(self, left_ticks, right_ticks):
        """根据编码器更新里程计"""
        d_left = 2 * math.pi * self.R * left_ticks / self.ticks_per_rev
        d_right = 2 * math.pi * self.R * right_ticks / self.ticks_per_rev
        
        d_center = (d_left + d_right) / 2
        d_theta = (d_right - d_left) / self.L
        
        self.x += d_center * math.cos(self.theta + d_theta / 2)
        self.y += d_center * math.sin(self.theta + d_theta / 2)
        self.theta += d_theta
        
        # 规范化角度
        self.theta = math.atan2(math.sin(self.theta), math.cos(self.theta))
    
    def inverse_kinematics(self, v, omega):
        """逆运动学: 速度->轮速"""
        v_left = v - omega * self.L / 2
        v_right = v + omega * self.L / 2
        return v_left, v_right
```

## 路径规划与轨迹生成

```python
import numpy as np

class PathPlanner:
    """路径规划器"""
    
    def __init__(self, drive: DifferentialDrive):
        self.drive = drive
        self.path = []
    
    def generate_line(self, start, end, num_points=100):
        """生成直线轨迹"""
        self.path = []
        for i in range(num_points + 1):
            t = i / num_points
            x = start[0] + t * (end[0] - start[0])
            y = start[1] + t * (end[1] - start[1])
            self.path.append((x, y))
        return self.path
    
    def generate_circle(self, center, radius, num_points=100):
        """生成圆形轨迹"""
        self.path = []
        for i in range(num_points + 1):
            angle = 2 * math.pi * i / num_points
            x = center[0] + radius * math.cos(angle)
            y = center[1] + radius * math.sin(angle)
            self.path.append((x, y))
        return self.path
    
    def generate_rectangle(self, origin, width, height):
        """生成矩形轨迹"""
        self.path = []
        self.path.extend(self.generate_line(origin, (origin[0]+width, origin[1]), 50))
        self.path.extend(self.generate_line((origin[0]+width, origin[1]), 
                                            (origin[0]+width, origin[1]+height), 50))
        self.path.extend(self.generate_line((origin[0]+width, origin[1]+height),
                                            (origin[0], origin[1]+height), 50))
        self.path.extend(self.generate_line((origin[0], origin[1]+height),
                                            origin, 50))
        return self.path
    
    def generate_triangle(self, p1, p2, p3):
        """生成三角形轨迹"""
        self.path = []
        self.path.extend(self.generate_line(p1, p2, 50))
        self.path.extend(self.generate_line(p2, p3, 50))
        self.path.extend(self.generate_line(p3, p1, 50))
        return self.path
    
    def generate_figure8(self, center, radius, num_points=200):
        """生成8字形轨迹"""
        self.path = []
        for i in range(num_points + 1):
            t = 2 * math.pi * i / num_points
            x = center[0] + radius * math.sin(t)
            y = center[1] + radius * math.sin(2*t) / 2
            self.path.append((x, y))
        return self.path
    
    def parse_gcode(self, gcode_str):
        """解析简单G-code轨迹"""
        self.path = []
        for line in gcode_str.strip().split('\n'):
            if line.startswith('G1'):
                parts = line.split()
                x = float([p[1:] for p in parts if p.startswith('X')][0])
                y = float([p[1:] for p in parts if p.startswith('Y')][0])
                self.path.append((x, y))
        return self.path
```

import OPi.GPIO as GPIO

class MotorController:
    """双路电机驱动"""
    
    def __init__(self):
        # TB6612FNG引脚
        self.pins = {
            'AIN1': 11, 'AIN2': 13, 'PWMA': 15,  # 左电机
            'BIN1': 16, 'BIN2': 18, 'PWMB': 22,  # 右电机
            'STBY': 24,
        }
        
        GPIO.setmode(GPIO.BOARD)
        for pin in self.pins.values():
            GPIO.setup(pin, GPIO.OUT)
        
        # PWM初始化
        self.pwm_left = GPIO.PWM(self.pins['PWMA'], 1000)
        self.pwm_right = GPIO.PWM(self.pins['PWMB'], 1000)
        self.pwm_left.start(0)
        self.pwm_right.start(0)
        
        GPIO.output(self.pins['STBY'], GPIO.HIGH)
    
    def set_motors(self, left_speed, right_speed):
        """设置左右电机速度 (-100 ~ 100)"""
        # 左电机
        if left_speed >= 0:
            GPIO.output(self.pins['AIN1'], GPIO.HIGH)
            GPIO.output(self.pins['AIN2'], GPIO.LOW)
        else:
            GPIO.output(self.pins['AIN1'], GPIO.LOW)
            GPIO.output(self.pins['AIN2'], GPIO.HIGH)
        self.pwm_left.ChangeDutyCycle(abs(left_speed))
        
        # 右电机
        if right_speed >= 0:
            GPIO.output(self.pins['BIN1'], GPIO.HIGH)
            GPIO.output(self.pins['BIN2'], GPIO.LOW)
        else:
            GPIO.output(self.pins['BIN1'], GPIO.LOW)
            GPIO.output(self.pins['BIN2'], GPIO.HIGH)
        self.pwm_right.ChangeDutyCycle(abs(right_speed))
    
    def stop(self):
        self.set_motors(0, 0)

## 绘图控制器

```python
class DrawingController:
    """绘图控制器"""
    
    def __init__(self):
        self.drive = DifferentialDrive()
        self.motors = MotorController()
        self.pen = PenController()
        self.planner = PathPlanner(self.drive)
        
        # 位置控制PID
        self.pid_linear = PIDController(kp=2.0, ki=0.1, kd=0.5)
        self.pid_angular = PIDController(kp=3.0, ki=0.05, kd=0.8)
        
        self.look_ahead = 0.1  # 前瞻距离(m)
    
    def follow_path(self, path, speed=30):
        """沿路径行驶"""
        self.pen.pen_down()
        
        for i, target in enumerate(path):
            while True:
                # 计算到目标点的距离和角度
                dx = target[0] - self.drive.x
                dy = target[1] - self.drive.y
                distance = math.sqrt(dx**2 + dy**2)
                target_angle = math.atan2(dy, dx)
                
                # 角度误差
                angle_error = target_angle - self.drive.theta
                angle_error = math.atan2(math.sin(angle_error), math.cos(angle_error))
                
                if distance < 0.02:  # 到达目标点(2cm)
                    break
                
                # PID控制
                linear_speed = self.pid_linear.update(distance)
                angular_speed = self.pid_angular.update(angle_error)
                
                # 限幅
                linear_speed = max(-speed, min(speed, linear_speed))
                angular_speed = max(-50, min(50, angular_speed))
                
                # 逆运动学
                v_left, v_right = self.drive.inverse_kinematics(linear_speed, angular_speed)
                
                # 驱动电机
                self.motors.set_motors(int(v_left), int(v_right))
                
                # 更新里程计（实际应从中断读取编码器）
                time.sleep(0.02)
        
        self.pen.pen_up()
        self.motors.stop()
    
    def draw_shape(self, shape_type, params):
        """绘制指定图形"""
        if shape_type == 'line':
            path = self.planner.generate_line(params['start'], params['end'])
        elif shape_type == 'circle':
            path = self.planner.generate_circle(params['center'], params['radius'])
        elif shape_type == 'rectangle':
            path = self.planner.generate_rectangle(params['origin'], 
                                                     params['width'], params['height'])
        elif shape_type == 'triangle':
            path = self.planner.generate_triangle(params['p1'], params['p2'], params['p3'])
        elif shape_type == 'figure8':
            path = self.planner.generate_figure8(params['center'], params['radius'])
        else:
            print(f"未知图形: {shape_type}")
            return
        
        self.follow_path(path)
        print(f"绘制{shape_type}完成")


class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.prev_error = 0
        self.integral = 0
    
    def update(self, error):
        self.integral += error
        self.integral = max(-500, min(500, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
```

## 遥控界面

```python
import socket
import json

class RemoteController:
    """WiFi遥控接口"""
    
    def __init__(self, port=8080):
        self.server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server.bind(('0.0.0.0', port))
        self.server.listen(1)
    
    def handle_command(self, cmd):
        """处理遥控指令"""
        if cmd['type'] == 'move':
            # 直接速度控制
            return cmd['left'], cmd['right']
        elif cmd['type'] == 'draw':
            # 绘图指令
            return cmd['shape'], cmd['params']
        elif cmd['type'] == 'pen':
            return cmd['action']  # 'up' or 'down'
```
