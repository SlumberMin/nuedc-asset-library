# 2018 全国大学生电子设计竞赛 - 无线充电电动小车

## 题目概述
设计一辆无线充电电动小车，能在指定跑道上循迹行驶，经过无线充电区域时自动充电，要求续航能力强、充电效率高、循迹精度好。

### 1. 视觉循迹（复用2016方案优化版）

```python
import cv2
import numpy as np

class LineTracker:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
        self.cap.set(cv2.CAP_PROP_FPS, 30)

    def detect_line(self, frame):
        h, w = frame.shape[:2]
        roi = frame[int(h*0.4):, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        _, binary = cv2.threshold(gray, 60, 255, cv2.THRESH_BINARY_INV)
        kernel = np.ones((5,5), np.uint8)
        binary = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)

        contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 200:
                M = cv2.moments(c)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    vx, vy, x0, y0 = cv2.fitLine(c, cv2.DIST_L2, 0, 0.01, 0.01)
                    angle = np.arctan2(vy[0], vx[0]) * 180 / np.pi
                    offset = cx - w // 2
                    return offset, angle
        return 0, 0

    def detect_charging_zone(self, frame):
        """检测充电区域标识"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 检测充电区特殊标记（如绿色标记）
        lower = np.array([35, 100, 100])
        upper = np.array([85, 255, 255])
        mask = cv2.inRange(hsv, lower, upper)
        if np.sum(mask) > 5000:
            return True
        return False
```

### 2. 充电管理系统

```python
import serial
import time

class ChargeManager:
    def __init__(self, uart_port='/dev/ttyS3'):
        self.ser = serial.Serial(uart_port, 115200)
        self.battery_voltage = 12.6  # 满电电压
        self.min_voltage = 9.0       # 低电量阈值
        self.charge_threshold = 10.5 # 需要充电的电压
        self.is_charging = False
        self.charge_start_time = 0

    def read_battery_voltage(self):
        """读取电池电压"""
        self.ser.write(b'\xAA\x02\x55')
        resp = self.ser.read(6)
        if len(resp) == 6 and resp[0] == 0xAA:
            voltage_raw = (resp[2] << 8) | resp[3]
            self.battery_voltage = voltage_raw / 1000.0
            return self.battery_voltage
        return None

    def get_battery_percent(self):
        """电池电量百分比"""
        v = self.battery_voltage
        if v >= 12.6: return 100
        if v >= 11.7: return 75
        if v >= 11.1: return 50
        if v >= 10.5: return 25
        if v >= 9.0:  return 10
        return 0

    def need_charge(self):
        """是否需要充电"""
        return self.battery_voltage < self.charge_threshold

    def is_low_battery(self):
        """是否低电量（需立即充电）"""
        return self.battery_voltage < self.min_voltage

    def get_charge_time(self):
        """建议充电时间（秒）"""
        # 根据电量差估算
        target_voltage = 12.0
        diff = target_voltage - self.battery_voltage
        # 约1V需要60秒（2A充电，2200mAh电池）
        return max(0, int(diff * 60))

    def start_charging(self):
        """通知进入充电模式"""
        self.ser.write(b'\xAA\x03\x01\x55')
        self.is_charging = True
        self.charge_start_time = time.time()

    def stop_charging(self):
        """停止充电"""
        self.ser.write(b'\xAA\x03\x00\x55')
        self.is_charging = False
```

### 3. 能量感知循迹控制器

```python
class EnergyAwareController:
    def __init__(self):
        self.Kp = 0.8
        self.Ki = 0.02
        self.Kd = 0.5
        self.integral = 0
        self.prev_error = 0

    def compute(self, offset, angle):
        error = offset * 0.7 + angle * 0.3
        self.integral += error
        self.integral = max(-500, min(500, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative

    def reset(self):
        self.integral = 0
        self.prev_error = 0
```

### 4. 电机驱动

```python
class MotorDriver:
    def __init__(self, uart_port='/dev/ttyS3'):
        self.ser = serial.Serial(uart_port, 115200)

    def set_motors(self, left, right):
        left = max(-255, min(255, int(left)))
        right = max(-255, min(255, int(right)))
        cmd = bytes([0xAA, 0x01,
                     (left >> 8) & 0xFF, left & 0xFF,
                     (right >> 8) & 0xFF, right & 0xFF, 0x55])
        self.ser.write(cmd)

    def stop(self):
        self.set_motors(0, 0)
```

### 5. 主控制程序

```python
import cv2
import time

class WirelessChargeCar:
    def __init__(self):
        self.tracker = LineTracker(camera_id=0)
        self.charge = ChargeManager()
        self.controller = EnergyAwareController()
        self.motor = MotorDriver()

        self.base_speed = 130        # 正常速度
        self.slow_speed = 80         # 慢速（低电量）
        self.charge_speed = 0        # 充电时停止
        self.min_charge_time = 10    # 最短充电时间(秒)

    def run(self):
        print("无线充电电动小车启动")
        lap_count = 0

        while True:
            ret, frame = self.tracker.cap.read()
            if not ret:
                continue

            # 读取电池状态
            self.charge.read_battery_voltage()
            battery = self.charge.get_battery_percent()

            # 检测充电区
            in_charge_zone = self.tracker.detect_charging_zone(frame)

            # 充电决策
            if in_charge_zone and self.charge.need_charge():
                self.handle_charging(battery)
                continue

            # 低电量减速
            speed = self.slow_speed if self.charge.is_low_battery() else self.base_speed

            # 正常循迹
            offset, angle = self.tracker.detect_line(frame)
            steer = self.controller.compute(offset, angle)
            left = speed + steer
            right = speed - steer
            self.motor.set_motors(left, right)

            # 状态显示
            print(f"电量:{battery}% | 偏移:{offset:.0f} | 速度:{speed}")

            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
            time.sleep(0.02)

        self.motor.stop()
        cv2.destroyAllWindows()

    def handle_charging(self, battery):
        """充电处理"""
        print(f"进入充电区，电量:{battery}%")
        self.motor.stop()
        self.charge.start_charging()

        # 充电等待
        charge_time = self.charge.get_charge_time()
        charge_time = max(self.min_charge_time, min(charge_time, 60))
        start = time.time()

        while time.time() - start < charge_time:
            v = self.charge.read_battery_voltage()
            print(f"充电中... 电压:{v:.2f}V")
            time.sleep(1)

        self.charge.stop_charging()
        print("充电完成，继续行驶")

    def optimize_speed_for_energy(self):
        """根据电量优化速度策略"""
        pct = self.charge.get_battery_percent()
        if pct > 70:
            return self.base_speed
        elif pct > 40:
            return int(self.base_speed * 0.85)
        elif pct > 20:
            return self.slow_speed
        else:
            return int(self.slow_speed * 0.7)

if __name__ == "__main__":
    car = WirelessChargeCar()
    car.run()
```

```c
uint16_t read_battery_adc(void) {
    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, 10);
    return HAL_ADC_GetValue(&hadc1);
}

float get_battery_voltage(void) {
    uint16_t adc = read_battery_adc();
    // 分压比 10k:2.2k, 参考电压3.3V, 12位ADC
    return (adc * 3.3f / 4096.0f) * (10.0f + 2.2f) / 2.2f;
}
```

## 关键技术要点

### 无线充电效率优化
- **线圈对齐**：小车行驶时居中，确保接收线圈与发射线圈对齐
- **充电距离**：保持5-10mm间距（最佳效率区间）
- **充电电流**：2A快充，约10-15分钟充入30%电量

### 能量管理策略
- **动态调速**：电量高时快速行驶，电量低时降速节能
- **智能充电**：电量低于阈值时才进入充电区
- **最小停靠**：充电区停留时间最优化

### 充电区识别
- 颜色标记识别（绿色/特定图案）
- RFID标签（可选方案）
- 距离估计判断是否在充电区
