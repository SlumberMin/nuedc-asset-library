# 2014年 四旋翼飞行器 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个四旋翼飞行器系统，实现自主悬停、航线飞行、定点降落等功能。需通过视觉系统实现精确定位，完成规定的飞行任务。

### 1.2 关键指标
- 飞行器类型：四旋翼
- 起飞重量：<1.5kg
- 定位精度：±15cm（水平），±10cm（垂直）
- 任务：定点悬停、绕杆飞行、定点降落
- 飞行时间：>8分钟

### 1.3 技术难点
1. **视觉定位**：实时准确的位置估计
2. **姿态稳定**：抗风扰、抗振动
3. **杆塔绕行**：需要精确的路径规划
4. **降落精度**：准确识别降落点

## 四、视觉定位算法

### 4.1 地面标记检测
```python
import cv2
import numpy as np

class GroundMarkerDetector:
    def __init__(self, marker_size_m=0.15):
        self.marker_size = marker_size_m  # 标记实际尺寸(m)
        
        # ArUco标记检测器
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        
        # 相机参数（需要标定）
        self.camera_matrix = None
        self.dist_coeffs = None
    
    def calibrate_camera(self, images, board_size=(7, 6), square_size=0.025):
        """相机标定"""
        obj_points = []
        img_points = []
        
        objp = np.zeros((board_size[0] * board_size[1], 3), np.float32)
        objp[:, :2] = np.mgrid[0:board_size[0], 0:board_size[1]].T.reshape(-1, 2) * square_size
        
        for img in images:
            gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
            ret, corners = cv2.findChessboardCorners(gray, board_size, None)
            
            if ret:
                obj_points.append(objp)
                corners2 = cv2.cornerSubPix(gray, corners, (11, 11), (-1, -1),
                    (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001))
                img_points.append(corners2)
        
        ret, self.camera_matrix, self.dist_coeffs, _, _ = cv2.calibrateCamera(
            obj_points, img_points, gray.shape[::-1], None, None)
        
        return ret
    
    def detect_markers(self, frame):
        """检测ArUco标记"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 检测标记
        corners, ids, rejected = self.detector.detectMarkers(gray)
        
        markers = []
        
        if ids is not None:
            for i, marker_id in enumerate(ids.flatten()):
                # 计算标记中心
                center = np.mean(corners[i][0], axis=0)
                
                # 如果有相机参数，估计位姿
                if self.camera_matrix is not None:
                    rvec, tvec, _ = cv2.aruco.estimatePoseSingleMarkers(
                        corners[i], self.marker_size, self.camera_matrix, self.dist_coeffs)
                    
                    markers.append({
                        'id': marker_id,
                        'center': center,
                        'corners': corners[i][0],
                        'rvec': rvec[0],
                        'tvec': tvec[0],
                        'distance': np.linalg.norm(tvec[0])
                    })
                else:
                    markers.append({
                        'id': marker_id,
                        'center': center,
                        'corners': corners[i][0]
                    })
        
        return markers
    
    def estimate_position(self, markers, marker_map):
        """
        根据检测到的标记估计相机（飞行器）位置
        marker_map: 标记在世界坐标系中的位置
        """
        if not markers or self.camera_matrix is None:
            return None
        
        positions = []
        
        for marker in markers:
            if marker['id'] in marker_map:
                # 标记的世界坐标
                world_pos = marker_map[marker['id']]
                
                # 相机相对于标记的位置（相机坐标系）
                tvec = marker['tvec'].flatten()
                
                # 转换到世界坐标系
                # 简化处理：假设相机垂直向下
                camera_world = world_pos + tvec
                positions.append(camera_world)
        
        if positions:
            # 多标记融合（取平均）
            return np.mean(positions, axis=0)
        
        return None
```

class ColorBlobDetector:
    def __init__(self):
        # 定义标记颜色
        self.markers = {
            'red': {
                'lower': np.array([0, 100, 100]),
                'upper': np.array([10, 255, 255]),
                'world_pos': np.array([0, 0, 0])
            },
            'blue': {
                'lower': np.array([100, 100, 100]),
                'upper': np.array([130, 255, 255]),
                'world_pos': np.array([1, 0, 0])
            },
            'green': {
                'lower': np.array([35, 100, 100]),
                'upper': np.array([85, 255, 255]),
                'world_pos': np.array([0, 1, 0])
            }
        }
    
    def detect_color_markers(self, frame):
        """检测色块标记"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        detected = []
        
        for name, config in self.markers.items():
            mask = cv2.inRange(hsv, config['lower'], config['upper'])
            
            # 形态学操作
            kernel = np.ones((5, 5), np.uint8)
            mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
            mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
            
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                            cv2.CHAIN_APPROX_SIMPLE)
            
            for contour in contours:
                area = cv2.contourArea(contour)
                if area > 500:
                    M = cv2.moments(contour)
                    if M['m00'] > 0:
                        cx = int(M['m10'] / M['m00'])
                        cy = int(M['m01'] / M['m00'])
                        
                        detected.append({
                            'name': name,
                            'center': (cx, cy),
                            'area': area,
                            'world_pos': config['world_pos']
                        })
        
        return detected

### 4.3 杆塔检测（绕杆任务）
```python
class PoleDetector:
    def __init__(self):
        self.pole_color_lower = np.array([0, 0, 100])  # 红色杆
        self.pole_color_upper = np.array([10, 255, 255])
    
    def detect_poles(self, frame):
        """检测杆塔"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 检测红色杆
        mask = cv2.inRange(hsv, self.pole_color_lower, self.pole_color_upper)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        
        poles = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if area > 1000:
                # 计算边界框
                x, y, w, h = cv2.boundingRect(contour)
                
                # 杆塔通常是细长的
                aspect_ratio = h / w
                if aspect_ratio > 3:
                    center_x = x + w // 2
                    center_y = y + h // 2
                    
                    # 估算距离（根据大小）
                    distance = self.estimate_distance(h)
                    
                    poles.append({
                        'center': (center_x, center_y),
                        'bbox': (x, y, w, h),
                        'distance': distance,
                        'angle': self.calculate_angle(center_x, frame.shape[1])
                    })
        
        return poles
    
    def estimate_distance(self, pixel_height, real_height=0.5):
        """估算到杆的距离"""
        # 使用相似三角形
        focal_length = 600  # 需要标定
        distance = (real_height * focal_length) / pixel_height
        return distance
    
    def calculate_angle(self, pixel_x, frame_width, fov=62):
        """计算杆相对于视野中心的角度"""
        center_x = frame_width / 2
        offset = pixel_x - center_x
        angle = (offset / center_x) * (fov / 2)
        return angle
```

### 5.1 状态估计
```python
import numpy as np

class StateEstimator:
    def __init__(self):
        # 互补滤波参数
        self.alpha = 0.98  # 陀螺仪权重
        
        # 状态
        self.attitude = np.array([0.0, 0.0, 0.0])  # roll, pitch, yaw
        self.position = np.array([0.0, 0.0, 0.0])   # x, y, z
        self.velocity = np.array([0.0, 0.0, 0.0])
        
        # 时间戳
        self.last_time = None
    
    def update_imu(self, gyro, accel, dt):
        """IMU数据更新"""
        # 陀螺仪积分
        self.attitude += gyro * dt
        
        # 加速度计修正roll和pitch
        accel_roll = np.arctan2(accel[1], accel[2])
        accel_pitch = np.arctan2(-accel[0], np.sqrt(accel[1]**2 + accel[2]**2))
        
        # 互补滤波
        self.attitude[0] = self.alpha * self.attitude[0] + (1 - self.alpha) * accel_roll
        self.attitude[1] = self.alpha * self.attitude[1] + (1 - self.alpha) * accel_pitch
        self.attitude[2] += gyro[2] * dt  # yaw仅用陀螺仪
    
    def update_visual(self, visual_position):
        """视觉数据更新"""
        # 视觉位置修正
        # 使用低通滤波融合
        beta = 0.3
        self.position = beta * visual_position + (1 - beta) * self.position
        
        # 从位置变化估算速度
        if self.last_time is not None:
            dt = time.time() - self.last_time
            if dt > 0:
                visual_velocity = (visual_position - self.position) / dt
                self.velocity = beta * visual_velocity + (1 - beta) * self.velocity
        
        self.last_time = time.time()
    
    def update_altitude(self, altitude):
        """高度更新（超声波/气压计）"""
        gamma = 0.5
        self.position[2] = gamma * altitude + (1 - gamma) * self.position[2]
    
    def get_state(self):
        """获取完整状态"""
        return {
            'attitude': self.attitude.copy(),
            'position': self.position.copy(),
            'velocity': self.velocity.copy()
        }
```

### 5.2 飞行控制器
```python
class FlightController:
    def __init__(self):
        # 位置PID
        self.pos_x_pid = PIDController(Kp=1.0, Ki=0.1, Kd=0.5)
        self.pos_y_pid = PIDController(Kp=1.0, Ki=0.1, Kd=0.5)
        self.pos_z_pid = PIDController(Kp=1.5, Ki=0.2, Kd=0.8)
        
        # 姿态PID
        self.roll_pid = PIDController(Kp=2.0, Ki=0.1, Kd=0.4)
        self.pitch_pid = PIDController(Kp=2.0, Ki=0.1, Kd=0.4)
        self.yaw_pid = PIDController(Kp=2.5, Ki=0.05, Kd=0.6)
        
        # 限制
        self.max_tilt = 20  # 度
        self.max_yaw_rate = 90  # 度/秒
    
    def position_control(self, target_pos, current_pos, target_yaw, current_yaw):
        """位置控制 -> 姿态指令"""
        # 位置误差
        error_x = target_pos[0] - current_pos[0]
        error_y = target_pos[1] - current_pos[1]
        error_z = target_pos[2] - current_pos[2]
        
        # 生成姿态指令
        # 前倾控制X，横滚控制Y
        pitch_cmd = self.pos_x_pid.update(error_x)
        roll_cmd = self.pos_y_pid.update(error_y)
        throttle_cmd = self.pos_z_pid.update(error_z)
        
        # 航向控制
        yaw_error = target_yaw - current_yaw
        yaw_error = (yaw_error + 180) % 360 - 180
        yaw_cmd = self.yaw_pid.update(yaw_error)
        
        # 限幅
        pitch_cmd = max(-self.max_tilt, min(self.max_tilt, pitch_cmd))
        roll_cmd = max(-self.max_tilt, min(self.max_tilt, roll_cmd))
        yaw_cmd = max(-self.max_yaw_rate, min(self.max_yaw_rate, yaw_cmd))
        
        return roll_cmd, pitch_cmd, yaw_cmd, throttle_cmd
    
    def attitude_control(self, target_attitude, current_attitude, target_throttle):
        """姿态控制 -> 电机指令"""
        roll_error = target_attitude[0] - current_attitude[0]
        pitch_error = target_attitude[1] - current_attitude[1]
        yaw_error = target_attitude[2] - current_attitude[2]
        
        roll_out = self.roll_pid.update(roll_error)
        pitch_out = self.pitch_pid.update(pitch_error)
        yaw_out = self.yaw_pid.update(yaw_error)
        
        return roll_out, pitch_out, yaw_out, target_throttle

class PIDController:
    def __init__(self, Kp, Ki, Kd):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.prev_error = 0
        self.integral = 0
    
    def update(self, error):
        self.integral += error
        self.integral = max(-100, min(100, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error
        
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative
```

class MissionExecutor:
    def __init__(self, flight_ctrl, state_estimator):
        self.fc = flight_ctrl
        self.state = state_estimator
        
        # 任务状态
        self.mission_state = 'IDLE'
        self.current_target = None
    
    def execute_hover(self, position, duration):
        """定点悬停任务"""
        self.mission_state = 'HOVERING'
        start_time = time.time()
        
        while time.time() - start_time < duration:
            current_pos = self.state.position
            roll, pitch, yaw, throttle = self.fc.position_control(
                position, current_pos, 0, self.state.attitude[2])
            
            # 发送指令到底层飞控
            self.send_attitude_command(roll, pitch, yaw, throttle)
            time.sleep(0.02)  # 50Hz
        
        self.mission_state = 'IDLE'
    
    def execute_pole_circumnavigation(self, pole_position, radius=1.0, direction='right'):
        """绕杆飞行任务"""
        self.mission_state = 'CIRCLING'
        
        # 计算圆形路径
        angles = np.linspace(0, 2 * np.pi, 36)  # 36个航点
        
        for angle in angles:
            # 计算目标位置
            if direction == 'right':
                target_x = pole_position[0] + radius * np.cos(angle)
                target_y = pole_position[1] + radius * np.sin(angle)
            else:
                target_x = pole_position[0] + radius * np.cos(-angle)
                target_y = pole_position[1] + radius * np.sin(-angle)
            
            target_pos = np.array([target_x, target_y, pole_position[2]])
            
            # 飞向目标
            while True:
                current_pos = self.state.position
                distance = np.linalg.norm(target_pos - current_pos)
                
                if distance < 0.2:  # 到达
                    break
                
                roll, pitch, yaw, throttle = self.fc.position_control(
                    target_pos, current_pos, angle, self.state.attitude[2])
                
                self.send_attitude_command(roll, pitch, yaw, throttle)
                time.sleep(0.02)
        
        self.mission_state = 'IDLE'
    
    def execute_landing(self, landing_marker):
        """定点降落任务"""
        self.mission_state = 'LANDING'
        
        # 降低高度

            target_pos = np.array([landing_marker[0], landing_marker[1], 
                                   self.state.position[2] - 0.1])
            
            roll, pitch, yaw, throttle = self.fc.position_control(
                target_pos, self.state.position, 0, self.state.attitude[2])
            
            self.send_attitude_command(roll, pitch, yaw, throttle)
            time.sleep(0.05)

    def takeoff(self, altitude):
        """自动起飞"""
        # 解锁电机
        self.send_command('ARM')
        time.sleep(1)
        
        # 增加油门
        throttle = 400
        while self.state_estimator.position[2] < altitude:
            # 更新状态
            self.update_state()
            
            # 增加油门
            throttle = min(throttle + 5, 600)
            self.send_attitude_command(0, 0, 0, throttle)
            time.sleep(0.02)
    
    def fly_to_waypoint(self, target_pos):
        """飞向指定航点"""
        while True:
            # 更新状态
            self.update_state()
            
            # 计算距离
            current_pos = self.state_estimator.position
            distance = np.linalg.norm(target_pos - current_pos)
            
            if distance < 0.2:  # 到达
                break
            
            # 位置控制
            roll, pitch, yaw, throttle = self.flight_ctrl.position_control(
                target_pos, current_pos, 0, self.state_estimator.attitude[2])
            
            # 发送指令
            self.send_attitude_command(roll, pitch, yaw, throttle)
            time.sleep(0.02)
    
    def update_state(self):
        """更新飞行状态"""
        # 获取视觉数据
        down_frame = self.down_camera.capture()
        markers = self.marker_detector.detect_markers(down_frame)
        
        # 估计位置
        visual_pos = self.marker_detector.estimate_position(markers, self.marker_map)
        
        if visual_pos is not None:
            self.state_estimator.update_visual(visual_pos)
        
        # 读取IMU数据（从STM32）
        imu_data = self.read_imu_data()
        if imu_data:
            self.state_estimator.update_imu(
                imu_data['gyro'], imu_data['accel'], imu_data['dt'])
        
        # 读取高度数据
        altitude = self.read_altitude()
        if altitude is not None:
            self.state_estimator.update_altitude(altitude)

### 7.2 PID调参建议
```
调参顺序：
1. 姿态内环（roll, pitch）
   - 先P后D，最后I
   - P: 从小到大，直到震荡
   - D: 抑制震荡
   - I: 消除稳态误差

2. 航向环（yaw）
   - 类似roll/pitch
   - 注意磁力计干扰

3. 位置外环（x, y, z）
   - 先z后xy
   - P: 控制响应速度
   - D: 控制超调
   - I: 消除位置偏移
```
