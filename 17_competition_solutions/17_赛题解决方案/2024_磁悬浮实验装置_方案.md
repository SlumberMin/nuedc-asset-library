## 一、题目分析

### 核心要求
设计一个磁悬浮实验装置：
- 电磁铁控制悬浮物（小球/圆盘）稳定悬浮
- 悬浮高度可设定（10-50mm范围内）
- 悬浮物稳定不掉落
- 实时显示悬浮高度
- 支持手动/自动模式切换

### 技术挑战
| 挑战 | 难度 | OPi5 解决方案 |
|------|------|---------------|
| 悬浮稳定性 | ★★★★★ | 高速PID控制 (1kHz+) |
| 高度检测 | ★★★★ | 视觉检测 + 激光测距 |
| 电磁控制 | ★★★★ | PWM精确调制 |
| 抗干扰 | ★★★ | 前馈补偿 |

## 四、核心算法

### 4.1 视觉高度检测

```python
import cv2
import numpy as np

class SuspensionHeightDetector:
    """视觉检测悬浮物高度"""
    
    def __init__(self, camera_matrix, dist_coeffs, 
                 reference_height_mm=0, reference_pixel_size=1.0):
        """
        camera_matrix: 相机内参
        reference_height_mm: 参考高度 (地面)
        reference_pixel_size: 参考高度时的像素/mm比例
        """
        self.K = camera_matrix
        self.D = dist_coeffs
        self.ref_height = reference_height_mm
        self.ref_pixel_size = reference_pixel_size
        
        # 悬浮物特征 (根据实际目标调整)
        self.object_diameter_mm = 25  # 悬浮物直径(mm)
        
        # 状态
        self.current_height = 0
        self.object_detected = False
        self.object_center = None
        
    def detect_suspended_object(self, frame):
        """
        检测悬浮物
        方法1: 圆形检测 (钢珠)
        方法2: 颜色检测 (带颜色标记的悬浮物)
        方法3: 轮廓检测
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        blurred = cv2.GaussianBlur(gray, (9, 9), 2)
        
        # 方法1: 霍夫圆检测
        circles = cv2.HoughCircles(
            blurred, cv2.HOUGH_GRADIENT, dp=1, minDist=50,
            param1=100, param2=30,
            minRadius=10, maxRadius=100
        )
        
        if circles is not None:
            circles = np.uint16(np.around(circles))
            # 取最大的圆
            best_circle = max(circles[0], key=lambda c: c[2])
            cx, cy, radius = best_circle
            
            self.object_center = (cx, cy)
            self.object_radius = radius
            self.object_detected = True
            
            return (cx, cy), radius
        
        # 方法2: 颜色检测 (如果悬浮物有颜色标记)
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 假设悬浮物为红色标记
        lower_red = np.array([0, 100, 100])
        upper_red = np.array([10, 255, 255])
        mask = cv2.inRange(hsv, lower_red, upper_red)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            largest = max(contours, key=cv2.contourArea)
            ((cx, cy), radius) = cv2.minEnclosingCircle(largest)
            
            if radius > 10:
                self.object_center = (int(cx), int(cy))
                self.object_radius = radius
                self.object_detected = True
                return (int(cx), int(cy)), radius
        
        self.object_detected = False
        return None, None
    
    def estimate_height_from_size(self, pixel_radius):
        """
        根据物体像素大小估计高度
        原理: 物体离相机越近，成像越大
        """
        if pixel_radius <= 0:
            return 0
        
        # 相似三角形法
        # H_real / H_pixel = distance / focal_length
        # 这里用标定好的比例关系
        
        # 简化: 像素半径越大 → 物体越近 → 悬浮越低
        # 需要标定: 记录不同高度下的像素大小
        
        # 假设已标定: height = f(pixel_size)
        # 这使用多项式拟合
        height = self._pixel_to_height(pixel_radius)
        
        return height
    
    def _pixel_to_height(self, pixel_radius):
        """
        像素大小到高度的映射 (标定后)
        使用预设的标定曲线
        """
        # 标定数据示例 (需要实际标定):
        # pixel_radius → height_mm
        # 80 → 10mm (近，成像大)
        # 40 → 30mm (中等)
        # 20 → 50mm (远，成像小)
        
        # 多项式拟合: h = a/r^2 + b/r + c
        # 或直接查表插值
        
        calibration_data = {
            80: 10, 70: 15, 60: 20, 50: 25,
            40: 30, 35: 35, 30: 40, 25: 45, 20: 50
        }
        
        # 线性插值
        radii = sorted(calibration_data.keys(), reverse=True)
        heights = [calibration_data[r] for r in radii]
        
        if pixel_radius >= radii[0]:
            return heights[0]
        if pixel_radius <= radii[-1]:
            return heights[-1]
        
        for i in range(len(radii) - 1):
            if radii[i] >= pixel_radius >= radii[i+1]:
                # 线性插值
                t = (pixel_radius - radii[i+1]) / (radii[i] - radii[i+1])
                height = heights[i+1] + t * (heights[i] - heights[i+1])
                return height
        
        return 30  # 默认值
    
    def detect_height_laser(self, distance_mm):
        """
        激光测距检测高度 (更精确)
        VL53L1X 输出的是到物体的距离
        """
        # 需要知道传感器到基座的距离
        sensor_to_base_mm = 100  # 根据实际安装调整
        height = sensor_to_base_mm - distance_mm
        
        return max(0, height)
    
    def get_height(self, frame, laser_distance=None):
        """
        综合获取高度
        优先使用激光测距，视觉作为辅助验证
        """
        # 视觉检测
        center, radius = self.detect_suspended_object(frame)
        visual_height = None
        if center:
            visual_height = self.estimate_height_from_size(radius)
        
        # 激光测距
        laser_height = None
        if laser_distance is not None:
            laser_height = self.detect_height_laser(laser_distance)
        
        # 融合
        if laser_height is not None and visual_height is not None:
            # 两者都有，加权平均
            self.current_height = 0.7 * laser_height + 0.3 * visual_height
        elif laser_height is not None:
            self.current_height = laser_height
        elif visual_height is not None:
            self.current_height = visual_height
        else:
            self.object_detected = False
        
        return self.current_height
```

### 4.2 高速PID控制器 (STM32端)

```c
// STM32F407 高速PID控制
// 采样率: 1kHz

#include "stm32f4xx.h"
#include <math.h>

// PID参数
typedef struct {
    float Kp;
    float Ki;
    float Kd;
    float integral;
    float prev_error;
    float output_min;
    float output_max;
    float integral_max;
} PID_t;

volatile PID_t height_pid = {
    .Kp = 5.0,
    .Ki = 0.1,
    .Kd = 2.0,
    .integral = 0,
    .prev_error = 0,
    .output_min = -100,
    .output_max = 100,
    .integral_max = 50
};

volatile float target_height = 30.0;  // 目标高度 (mm)
volatile float current_height = 0.0;  // 当前高度 (mm)
volatile float pid_output = 0.0;      // PID输出 (-100 到 100)
volatile int pwm_duty = 0;            // PWM占空比

// 激光测距值 (来自OPi5或直接读取)
volatile float laser_distance = 0.0;

void TIM3_1kHz_Init(void) {
    // 1kHz定时器中断
    RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
    TIM3->PSC = 84 - 1;    // 84MHz/84 = 1MHz
    TIM3->ARR = 1000 - 1;  // 1MHz/1000 = 1kHz
    TIM3->DIER |= TIM_DIER_UIE;
    TIM3->CR1 |= TIM_CR1_CEN;
    NVIC_EnableIRQ(TIM3_IRQn);
}

void TIM2_PWM_Init(void) {
    // PWM输出 (20kHz, 适合电磁铁)
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
    
    TIM2->PSC = 84 - 1;    // 84MHz/84 = 1MHz
    TIM2->ARR = 50 - 1;    // 1MHz/50 = 20kHz
    TIM2->CCR1 = 0;
    TIM2->CCMR1 = 0x60;    // PWM mode 1
    TIM2->CCER = 0x01;     // Enable CH1
    TIM2->CR1 |= TIM_CR1_CEN;
}

float PID_Compute(PID_t *pid, float setpoint, float measured, float dt) {
    float error = setpoint - measured;
    
    // 积分项
    pid->integral += error * dt;
    if (pid->integral > pid->integral_max) pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max) pid->integral = -pid->integral_max;
    
    // 微分项
    float derivative = (error - pid->prev_error) / dt;
    pid->prev_error = error;
    
    // PID输出
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;
    
    // 输出限幅
    if (output > pid->output_max) output = pid->output_max;
    if (output < pid->output_min) output = pid->output_min;
    
    return output;
}

void TIM3_IRQHandler(void) {
    if (TIM3->SR & TIM_SR_UIF) {
        TIM3->SR &= ~TIM_SR_UIF;
        
        // 1. 读取高度 (从OPi5串口或直接读传感器)
        // current_height 已在串口中断中更新
        
        // 2. 计算PID
        pid_output = PID_Compute(&height_pid, target_height, current_height, 0.001);
        
        // 3. 转换为PWM
        // pid_output: -100 到 100
        // PWM: 0 到 ARR
        // 注意: 悬浮需要保持一定基础电流
        int base_pwm = 25;  // 基础占空比 (维持悬浮)
        pwm_duty = base_pwm + (int)(pid_output * 0.25);
        
        if (pwm_duty < 0) pwm_duty = 0;
        if (pwm_duty > 49) pwm_duty = 49;
        
        // 4. 输出PWM
        TIM2->CCR1 = pwm_duty;
    }
}

// UART接收 (来自OPi5的高度数据)
void USART1_IRQHandler(void) {
    if (USART1->SR & USART_SR_RXNE) {
        char c = USART1->DR;
        // 解析高度数据协议: $H<height>\n
        static char buf[32];
        static int idx = 0;
        
        if (c == '$') {
            idx = 0;
        } else if (c == '\n') {
            buf[idx] = '\0';
            if (buf[0] == 'H') {
                current_height = atof(buf + 1);
            } else if (buf[0] == 'T') {
                target_height = atof(buf + 1);
            }
        } else if (idx < 31) {
            buf[idx++] = c;
        }
    }
}
```

### 4.3 OPi5 控制界面

```python
#!/usr/bin/env python3
"""
磁悬浮实验装置 - OPi5 控制界面
"""

import cv2
import numpy as np
import serial
import time
import threading
from luma.core.interface.serial import i2c
from luma.oled.device import ssd1306
from PIL import Image, ImageDraw, ImageFont

from height_detector import SuspensionHeightDetector

class MaglevController:
    def __init__(self):
        # 初始化
        self.height_detector = SuspensionHeightDetector(
            camera_matrix=np.eye(3),  # 需标定
            dist_coeffs=np.zeros(5)
        )
        
        # 摄像头
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
        self.cap.set(cv2.CAP_PROP_FPS, 30)
        
        # 激光测距
        import smbus
        self.i2c_bus = smbus.SMBus(1)
        self.VL53L1X_ADDR = 0x29
        
        # 串口连接STM32
        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)
        
        # OLED显示
        serial_interface = i2c(port=1, address=0x3C)
        self.oled = ssd1306(serial_interface, width=128, height=64)
        
        # 状态
        self.target_height = 30.0  # mm
        self.current_height = 0.0
        self.mode = 'auto'  # auto/manual
        self.running = False
        
        # 按键
        import OPi.GPIO as GPIO
        GPIO.setmode(GPIO.BOARD)
        for pin in [11, 12, 13, 14]:
            GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)
        
        # 历史数据
        self.height_history = []
        self.max_history = 100
    
    def read_vl53l1x(self):
        """读取VL53L1X激光测距"""
        try:
            # 简化的读取 (实际需要初始化和配置)
            data = self.i2c_bus.read_i2c_block_data(self.VL53L1X_ADDR, 0x00, 2)
            distance = (data[0] << 8) | data[1]
            return distance  # mm
        except:
            return None
    
    def send_to_stm32(self, command):
        """发送指令到STM32"""
        self.ser.write(f'{command}\n'.encode())
    
    def update_oled(self):
        """更新OLED显示"""
        img = Image.new('1', (128, 64))
        draw = ImageDraw.Draw(img)
        
        # 使用默认字体
        font = ImageFont.load_default()
        
        # 显示内容
        draw.text((0, 0), f"Target: {self.target_height:.1f}mm", fill=1, font=font)
        draw.text((0, 16), f"Current: {self.current_height:.1f}mm", fill=1, font=font)
        draw.text((0, 32), f"Mode: {self.mode}", fill=1, font=font)
        
        # 画高度柱状图
        bar_x = 100
        bar_h = 50
        bar_w = 20
        draw.rectangle([bar_x, 10, bar_x+bar_w, 10+bar_h], outline=1)
        
        # 填充当前高度
        fill_h = int(bar_h * self.current_height / 60)
        draw.rectangle([bar_x+1, 10+bar_h-fill_h, bar_x+bar_w-1, 10+bar_h-1], fill=1)
        
        # 目标线
        target_y = 10 + bar_h - int(bar_h * self.target_height / 60)
        draw.line([bar_x, target_y, bar_x+bar_w, target_y], fill=1)
        
        self.oled.display(img)
    
    def check_buttons(self):
        """检查按键"""
        import OPi.GPIO as GPIO
        
        if not GPIO.input(11):  # KEY1: 高度+
            self.target_height = min(50, self.target_height + 5)
            self.send_to_stm32(f'T{self.target_height:.1f}')
            time.sleep(0.2)
        
        if not GPIO.input(12):  # KEY2: 高度-
            self.target_height = max(10, self.target_height - 5)
            self.send_to_stm32(f'T{self.target_height:.1f}')
            time.sleep(0.2)
        
        if not GPIO.input(13):  # KEY3: 模式切换
            self.mode = 'manual' if self.mode == 'auto' else 'auto'
            time.sleep(0.2)
        
        if not GPIO.input(14):  # KEY4: 复位
            self.target_height = 30.0
            self.send_to_stm32(f'T{self.target_height:.1f}')
            time.sleep(0.2)
    
    def main_loop(self):
        """主循环"""
        self.running = True
        
        while self.running:
            # 1. 读取摄像头
            ret, frame = self.cap.read()
            if ret:
                # 2. 视觉检测高度
                visual_height = self.height_detector.get_height(frame)
                
                # 3. 激光测距 (辅助)
                laser_dist = self.read_vl53l1x()
                
                # 4. 融合高度
                if laser_dist and laser_dist < 100:
                    self.current_height = 0.7 * (100 - laser_dist) + 0.3 * visual_height
                else:
                    self.current_height = visual_height
                
                # 5. 发送到STM32
                self.send_to_stm32(f'H{self.current_height:.1f}')
                
                # 6. 记录历史
                self.height_history.append(self.current_height)
                if len(self.height_history) > self.max_history:
                    self.height_history.pop(0)
                
                # 7. 显示
                cv2.imshow('Camera', frame)
            
            # 8. 检查按键
            self.check_buttons()
            
            # 9. 更新OLED
            self.update_oled()
            
            # 10. 串口接收 (非阻塞)
            if self.ser.in_waiting:
                data = self.ser.readline().decode().strip()
                if data:
                    print(f"STM32: {data}")
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        self.cleanup()
    
    def cleanup(self):
        self.running = False
        self.cap.release()
        cv2.destroyAllWindows()
        self.ser.close()
        import OPi.GPIO as GPIO
        GPIO.cleanup()

def main():
    controller = MaglevController()
    controller.main_loop()

if __name__ == '__main__':
    main()
```

## 五、PID参数调试

### 5.1 调试步骤

```
1. 先设置基础电流 (让悬浮物有上升趋势)
2. 增加P直到系统开始振荡
3. 减小P到振荡停止的60%
4. 增加I消除稳态误差 (小量)
5. 增加D抑制超调
6. 微调至稳定
```

### 5.2 参数参考

| 参数 | 典型范围 | 说明 |
|------|---------|------|
| Kp | 2.0-10.0 | 主要响应参数 |
| Ki | 0.01-0.5 | 消除稳态误差 |
| Kd | 0.5-5.0 | 抑制振荡 |
| 基础PWM | 20-40% | 维持悬浮的最小电流 |

### 5.3 常见问题

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| 悬浮物掉落 | 基础电流不足 | 增加基础PWM |
| 持续振荡 | P过大/D过小 | 降低P，增加D |
| 响应慢 | P过小 | 增加P |
| 有稳态误差 | 积分不足 | 适当增加I |
