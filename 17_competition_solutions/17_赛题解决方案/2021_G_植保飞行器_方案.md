# 2021年电赛G题 - 植保飞行器 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：四旋翼植保飞行器自主巡航农田，识别作物（或标记），按需喷洒药液，完成自主起降、航线飞行、目标识别、定量喷洒。

**关键指标**：
- 巡航覆盖率
- 作物识别准确率
- 喷洒定位精度
- 药液用量控制

### 4.1 作物识别模块

```python
# crop_detector.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class CropDetector:
    """基于RKNN的作物识别"""
    def __init__(self, model_path='crop_detect.rknn'):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0_1_2)
        self.classes = ['healthy', 'diseased', 'weed', 'bare_soil', 'pest_damage']
    
    def detect_crop_status(self, frame):
        """检测作物状态"""
        img = cv2.resize(frame, (640, 640))
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        outputs = self.rknn.inference(inputs=[img])
        
        detections = self._postprocess(outputs)
        
        result = {
            'diseased': [],
            'weed': [],
            'pest_damage': [],
            'coverage': 0.0
        }
        
        for det in detections:
            cls_id, conf, x1, y1, x2, y2 = det
            cls_name = self.classes[int(cls_id)]
            if cls_name in result:
                result[cls_name].append({
                    'bbox': (x1, y1, x2, y2),
                    'confidence': conf,
                    'center': ((x1+x2)/2, (y1+y2)/2)
                })
        
        # 计算覆盖率
        total_area = frame.shape[0] * frame.shape[1]
        crop_area = sum(
            (d['bbox'][2]-d['bbox'][0])*(d['bbox'][3]-d['bbox'][1]) 
            for d in detections if self.classes[int(d[0])] in ['healthy', 'diseased']
        )
        result['coverage'] = crop_area / total_area
        
        return result
    
    def classify_color(self, frame, bbox):
        """基于颜色的辅助分类"""
        x1, y1, x2, y2 = [int(v) for v in bbox]
        roi = frame[y1:y2, x1:x2]
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        
        # 绿色健康作物
        green_mask = cv2.inRange(hsv, (35, 40, 40), (85, 255, 255))
        # 黄色病害
        yellow_mask = cv2.inRange(hsv, (20, 40, 40), (35, 255, 255))
        # 棕色枯萎
        brown_mask = cv2.inRange(hsv, (10, 40, 20), (20, 255, 200))
        
        green_ratio = np.sum(green_mask > 0) / green_mask.size
        yellow_ratio = np.sum(yellow_mask > 0) / yellow_mask.size
        
        if green_ratio > 0.3:
            return 'healthy'
        elif yellow_ratio > 0.2:
            return 'diseased'
        return 'unknown'
```

### 4.2 喷洒控制系统

```python
# spray_controller.py
import time
import serial
from enum import Enum

class SprayMode(Enum):
    OFF = 0
    LOW = 1       # 低流量 - 轻度病害
    MEDIUM = 2    # 中流量 - 中度病害
    HIGH = 3      # 高流量 - 重度病害/杂草

class SprayController:
    """智能喷洒控制器"""
    def __init__(self, valve_pin=17, pump_pin=27, flow_sensor_uart='/dev/ttyS3'):
        self.valve_pin = valve_pin
        self.pump_pin = pump_pin
        self.flow_serial = serial.Serial(flow_sensor_uart, 9600)
        
        # 喷洒参数
        self.flow_rates = {
            SprayMode.LOW: 100,     # ml/min
            SprayMode.MEDIUM: 200,
            SprayMode.HIGH: 400
        }
        self.total_sprayed = 0.0    # 总喷洒量 ml
        self.tank_capacity = 1000   # 药箱容量 ml
        
        # GPIO初始化
        self._init_gpio()
    
    def _init_gpio(self):
        """初始化GPIO控制"""
        import OPi.GPIO as GPIO
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.valve_pin, GPIO.OUT)
        GPIO.setup(self.pump_pin, GPIO.OUT)
        GPIO.output(self.valve_pin, GPIO.LOW)
        GPIO.output(self.pump_pin, GPIO.LOW)
    
    def spray_by_detection(self, detection_result):
        """根据识别结果决定喷洒策略"""
        diseased = len(detection_result['diseased'])
        weed = len(detection_result['weed'])
        pest = len(detection_result['pest_damage'])
        
        if weed > 3 or diseased > 5:
            return SprayMode.HIGH
        elif diseased > 2 or pest > 2:
            return SprayMode.MEDIUM
        elif diseased > 0 or pest > 0:
            return SprayMode.LOW
        return SprayMode.OFF
    
    def start_spray(self, mode):
        """启动喷洒"""
        if mode == SprayMode.OFF:
            self.stop_spray()
            return
        
        import OPi.GPIO as GPIO
        GPIO.output(self.pump_pin, GPIO.HIGH)
        GPIO.output(self.valve_pin, GPIO.HIGH)
        
        # PWM控制流量（可选）
        self.current_mode = mode
    
    def stop_spray(self):
        """停止喷洒"""
        import OPi.GPIO as GPIO
        GPIO.output(self.valve_pin, GPIO.LOW)
        GPIO.output(self.pump_pin, GPIO.LOW)
    
    def get_flow_rate(self):
        """读取实时流量"""
        data = self.flow_serial.readline()
        if data:
            return float(data.decode().strip())
        return 0.0
    
    def get_remaining(self):
        """剩余药量"""
        return self.tank_capacity - self.total_sprayed
```

### 4.3 巡航航线规划

```python
# route_planner.py
import numpy as np

class FieldRoutePlanner:
    """农田航线规划器"""
    def __init__(self, field_width, field_length, swath_width=3.0):
        self.field_w = field_width    # 田宽 m
        self.field_l = field_length   # 田长 m
        self.swath = swath_width      # 喷幅 m
    
    def generate_lawnmower(self, altitude=5.0):
        """生成弓字形覆盖航线"""
        waypoints = []
        num_strips = int(np.ceil(self.field_w / self.swath))
        
        for i in range(num_strips):
            x = i * self.swath + self.swath / 2
            if i % 2 == 0:
                waypoints.append((x, 0, altitude))
                waypoints.append((x, self.field_l, altitude))
            else:
                waypoints.append((x, self.field_l, altitude))
                waypoints.append((x, 0, altitude))
        
        return waypoints
    
    def generate_adaptive(self, hotspots, altitude=5.0):
        """自适应航线（重点区域加密）"""
        # 先生成基础覆盖航线
        base = self.generate_lawnmower(altitude)
        
        # 对热点区域增加低空航线
        extra = []
        for hs in hotspots:
            cx, cy = hs['center']
            # 围绕热点生成十字形加密航线
            extra.append((cx - 1, cy, altitude - 1))
            extra.append((cx + 1, cy, altitude - 1))
            extra.append((cx, cy - 1, altitude - 1))
            extra.append((cx, cy + 1, altitude - 1))
        
        return base + extra
    
    def estimate_time(self, waypoints, speed=3.0):
        """估算飞行时间"""
        total_dist = 0
        for i in range(1, len(waypoints)):
            dx = waypoints[i][0] - waypoints[i-1][0]
            dy = waypoints[i][1] - waypoints[i-1][1]
            total_dist += np.sqrt(dx**2 + dy**2)
        return total_dist / speed
```

### 4.4 任务主流程

```python
# main.py
import cv2
import time

def main():
    camera = cv2.VideoCapture(0)
    detector = CropDetector('crop_detect.rknn')
    spray = SprayController()
    flight = MAVLinkController('/dev/ttyS5')
    planner = FieldRoutePlanner(field_width=20, field_length=30)
    
    # 生成航线
    waypoints = planner.generate_lawnmower(altitude=5.0)
    estimated_time = planner.estimate_time(waypoints)
    print(f"预计飞行时间: {estimated_time:.1f}s")
    
    state = 'TAKEOFF'
    wp_idx = 0
    hotspots = []
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        # 实时检测作物状态
        crop_status = detector.detect_crop_status(frame)
        
        # 记录病害热点
        if crop_status['diseased'] or crop_status['weed']:
            gps_pos = flight.get_position()
            hotspots.append({
                'position': gps_pos,
                'status': crop_status,
                'time': time.time()
            })
        
        # 决策喷洒
        spray_mode = spray.spray_by_detection(crop_status)
        
        if state == 'TAKEOFF':
            flight.takeoff(5.0)
            time.sleep(8)
            state = 'CRUISE'
            
        elif state == 'CRUISE':
            if wp_idx < len(waypoints):
                wp = waypoints[wp_idx]
                flight.goto_waypoint(wp[0], wp[1], wp[2])
                
                # 到达航点附近后执行喷洒
                if flight.reached_waypoint(wp, tolerance=1.0):
                    if spray_mode != SprayMode.OFF:
                        spray.start_spray(spray_mode)
                    else:
                        spray.stop_spray()
                    wp_idx += 1
            else:
                spray.stop_spray()
                state = 'RETURN'
                
        elif state == 'RETURN':
            flight.goto_waypoint(0, 0, 5.0)
            time.sleep(10)
            state = 'LAND'
            
        elif state == 'LAND':
            flight.land()
            break
        
        time.sleep(0.1)
    
    # 输出统计
    print(f"总喷洒量: {spray.total_sprayed:.1f}ml")
    print(f"发现病害热点: {len(hotspots)}处")

if __name__ == '__main__':
    main()
```
