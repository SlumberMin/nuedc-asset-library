# 2018年全国大学生电子设计竞赛 - 灭火飞行器

## 题目分析
设计一个灭火飞行器，能够在指定区域内自主搜索火源（蜡烛/火焰），飞到火源附近并完成灭火操作。


#### 1. 火焰检测模块 (flame_detector.py)
```python
import cv2
import numpy as np
from dataclasses import dataclass

@dataclass
class FlameInfo:
    found: bool
    x: float = 0          # 图像中心x
    y: float = 0          # 图像中心y
    area: float = 0       # 火焰面积
    distance_est: float = 0  # 估计距离(m)
    intensity: float = 0  # 火焰强度(0-1)

class FlameDetector:
    """多模态火焰检测"""

    def __init__(self, config):
        self.min_area = config.get('min_area', 200)
        self.ir_camera = None

        # 热成像初始化(可选)
        if config.get('use_thermal', False):
            from thermal_detector import ThermalDetector
            self.ir_camera = ThermalDetector(config.get('thermal', {}))

    def detect_flame_rgb(self, frame) -> FlameInfo:
        """RGB图像火焰检测"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

        # 火焰颜色范围(亮黄-橙-红)
        # 方法1: HSV范围检测
        lower1 = np.array([0, 120, 200])    # 红色低
        upper1 = np.array([15, 255, 255])   # 红色高
        lower2 = np.array([15, 100, 200])   # 橙色低
        upper2 = np.array([35, 255, 255])   # 橙色高

        mask1 = cv2.inRange(hsv, lower1, upper1)
        mask2 = cv2.inRange(hsv, lower2, upper2)
        mask = mask1 | mask2

        # 形态学处理
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)

        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if contours:
            # 找最大火焰区域
            c = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(c)

            if area > self.min_area:
                M = cv2.moments(c)
                if M['m00'] > 0:
                    cx = int(M['m10'] / M['m00'])
                    cy = int(M['m01'] / M['m00'])

                    # 估计距离(火焰面积反比)
                    h, w = frame.shape[:2]
                    area_ratio = area / (w * h)
                    # 经验公式: 距离 = k / sqrt(area_ratio)
                    distance_est = 2.0 / max(np.sqrt(area_ratio), 0.01)
                    distance_est = min(distance_est, 10.0)  # 限幅

                    return FlameInfo(
                        found=True,
                        x=cx, y=cy,
                        area=area,
                        distance_est=distance_est,
                        intensity=min(area_ratio * 10, 1.0)
                    )

        return FlameInfo(found=False)

    def detect_flame_thermal(self) -> FlameInfo:
        """热成像火焰检测"""
        if self.ir_camera is None:
            return FlameInfo(found=False)

        thermal_frame = self.ir_camera.get_frame()
        if thermal_frame is None:
            return FlameInfo(found=False)

        # 热成像中火焰温度通常>100°C
        threshold = 80  # °C
        hot_mask = (thermal_frame > threshold).astype(np.uint8) * 255

        contours, _ = cv2.findContours(hot_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 50:
                M = cv2.moments(c)
                if M['m00'] > 0:
                    cx = int(M['m10'] / M['m00'])
                    cy = int(M['m01'] / M['m00'])
                    return FlameInfo(found=True, x=cx, y=cy,
                                   area=cv2.contourArea(c),
                                   distance_est=0,  # 需标定
                                   intensity=float(thermal_frame[cy, cx]) / 255)

        return FlameInfo(found=False)

    def detect(self, frame) -> FlameInfo:
        """融合RGB+热成像检测"""
        # RGB检测
        rgb_result = self.detect_flame_rgb(frame)

        # 热成像检测
        thermal_result = self.detect_flame_thermal()

        if rgb_result.found and thermal_result.found:
            # 两者都检测到, 取RGB结果(更精确)
            return rgb_result
        elif rgb_result.found:
            return rgb_result
        elif thermal_result.found:
            return thermal_result

        return FlameInfo(found=False)
```

#### 2. 灭火控制模块 (fire_extinguisher.py)
```python
import time
from enum import Enum
import RPi.GPIO as GPIO  # OPi兼容

class ExtinguishState(Enum):
    IDLE = 0
    APPROACHING = 1
    POSITIONING = 2
    EXTINGUISHING = 3
    CONFIRMED = 4

class FireExtinguisher:
    """灭火装置控制"""

    def __init__(self, config):
        self.fan_pin = config.get('fan_pin', 17)       # 风扇GPIO
        self.fan2_pin = config.get('fan2_pin', 27)     # 第二风扇
        self.servo_pitch_pin = config.get('servo_pin', 22)  # 俯仰舵机

        # 灭火参数
        self.extinguish_distance = config.get('extinguish_distance', 1.5)  # m
        self.extinguish_duration = config.get('extinguish_duration', 5.0)  # s
        self.max_attempts = config.get('max_attempts', 3)

        # GPIO初始化
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(self.fan_pin, GPIO.OUT)
        GPIO.setup(self.fan2_pin, GPIO.OUT)
        self.fan_pwm = GPIO.PWM(self.fan_pin, 1000)  # 1kHz PWM
        self.fan2_pwm = GPIO.PWM(self.fan2_pin, 1000)

        self.state = ExtinguishState.IDLE
        self.attempt_count = 0

    def start_extinguish(self, power=100):
        """启动灭火"""
        self.fan_pwm.start(power)
        self.fan2_pwm.start(power)
        self.state = ExtinguishState.EXTINGUISHING
        self.start_time = time.time()
        print(f"灭火开始, 功率: {power}%")

    def stop_extinguish(self):
        """停止灭火"""
        self.fan_pwm.stop()
        self.fan2_pwm.stop()
        self.state = ExtinguishState.IDLE
        print("灭火停止")

    def update(self, flame_info):
        """灭火状态更新"""
        if self.state == ExtinguishState.APPROACHING:
            if flame_info.found and flame_info.distance_est < self.extinguish_distance:
                self.state = ExtinguishState.POSITIONING

        elif self.state == ExtinguishState.POSITIONING:
            # 精确对准火焰
            if flame_info.found:
                # 判断是否对准(图像中心附近)
                h, w = 480, 640  # 假设分辨率
                cx, cy = flame_info.x, flame_info.y
                if abs(cx - w/2) < 50 and abs(cy - h/2) < 50:
                    self.start_extinguish()
                    self.attempt_count += 1

        elif self.state == ExtinguishState.EXTINGUISHING:
            elapsed = time.time() - self.start_time
            if elapsed > self.extinguish_duration:
                self.stop_extinguish()
                # 检查是否灭火成功
                if not flame_info.found:
                    self.state = ExtinguishState.CONFIRMED
                    print("灭火成功!")
                elif self.attempt_count < self.max_attempts:
                    self.state = ExtinguishState.APPROACHING
                    print(f"第{self.attempt_count}次尝试未成功, 再次接近")
                else:
                    print("多次尝试未成功")

    def cleanup(self):
        self.stop_extinguish()
        GPIO.cleanup()
```

#### 3. 搜索模式 (search_pattern.py)
```python
import numpy as np

class SearchPattern:
    """区域搜索路径规划"""

    def __init__(self, config):
        self.area_width = config.get('area_width', 10.0)   # m
        self.area_height = config.get('area_height', 10.0)  # m
        self.search_altitude = config.get('search_altitude', 4.0)  # m
        self.step_size = config.get('step_size', 2.0)  # m

        self.waypoints = self._generate_lawnmower()
        self.current_idx = 0

    def _generate_lawnmower(self):
        """生成弓字形搜索路径"""
        waypoints = []
        half_w = self.area_width / 2
        y = -self.area_height / 2

        while y < self.area_height / 2:
            waypoints.append((-half_w, y, -self.search_altitude))
            waypoints.append((half_w, y, -self.search_altitude))
            y += self.step_size
            if y < self.area_height / 2:
                waypoints.append((half_w, y, -self.search_altitude))
                waypoints.append((-half_w, y, -self.search_altitude))
                y += self.step_size

        return waypoints

    def get_next_waypoint(self):
        """获取下一个搜索航点"""
        if self.current_idx < len(self.waypoints):
            wp = self.waypoints[self.current_idx]
            self.current_idx += 1
            return wp
        return None

    def reset(self):
        self.current_idx = 0
```

#### 4. 主程序 (main.py)
```python
import asyncio
import cv2
import yaml
from camera import Camera
from mavlink_interface import MavlinkInterface
from flame_detector import FlameDetector
from fire_extinguisher import FireExtinguisher, ExtinguishState
from search_pattern import SearchPattern

async def main():
    with open('config.yaml', 'r') as f:
        config = yaml.safe_load(f)

    camera = Camera(config['camera'])
    mavlink = MavlinkInterface(config['mavlink'])
    detector = FlameDetector(config['detector'])
    extinguisher = FireExtinguisher(config['extinguisher'])
    search = SearchPattern(config['search'])

    await mavlink.connect()
    await mavlink.arm_and_takeoff(config['mission']['altitude'])
    await mavlink.start_offboard()

    print("=== 灭火飞行器启动 ===")

    state = 'SEARCH'  # SEARCH, APPROACH, EXTINGUISH, DONE

    try:
        while state != 'DONE':
            frame = camera.capture()
            if frame is None:
                continue

            flame = detector.detect(frame)

            if state == 'SEARCH':
                if flame.found:
                    state = 'APPROACH'
                    print(f"发现火焰! 距离估计: {flame.distance_est:.1f}m")
                else:
                    wp = search.get_next_waypoint()
                    if wp:
                        await mavlink.send_position_command(wp[0], wp[1], wp[2])
                    else:
                        print("搜索完成, 未发现火源")
                        state = 'DONE'

            elif state == 'APPROACH':
                if flame.found:
                    # 向火焰移动
                    max_speed = 1.5  # m/s
                    h, w = frame.shape[:2]
                    offset_x = (flame.x - w/2) / w
                    offset_y = (flame.y - h/2) / h

                    vn = -offset_y * max_speed
                    ve = offset_x * max_speed

                    # 距离够近时降低
                    if flame.distance_est < 2.0:
                        vd = -0.5  # 缓慢下降
                    else:
                        vd = 0

                    await mavlink.send_velocity_command(vn, ve, vd)

                    if flame.distance_est < 1.5:
                        state = 'EXTINGUISH'
                        extinguisher.state = ExtinguishState.POSITIONING
                        print("接近火源, 准备灭火!")
                else:
                    # 火焰丢失, 回到搜索
                    state = 'SEARCH'

            elif state == 'EXTINGUISH':
                extinguisher.update(flame)
                if extinguisher.state == ExtinguishState.CONFIRMED:
                    print("灭火成功!")
                    state = 'DONE'
                elif extinguisher.state == ExtinguishState.IDLE:
                    # 灭火失败, 重新接近
                    state = 'APPROACH'

            # 调试显示
            if config.get('debug', False):
                if flame.found:
                    cv2.circle(frame, (int(flame.x), int(flame.y)), 20, (0, 0, 255), 3)
                    cv2.putText(frame, f"Dist: {flame.distance_est:.1f}m",
                               (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
                cv2.putText(frame, f"State: {state}", (10, 60),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255, 255, 0), 2)
                cv2.imshow('Fire Fighter', frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break

            await asyncio.sleep(0.1)

    finally:
        extinguisher.cleanup()
        camera.release()
        cv2.destroyAllWindows()
        await mavlink.land()

if __name__ == '__main__':
    asyncio.run(main())
```

## 火焰检测优化

### 红外滤镜方案
在普通摄像头前加装红外通过滤镜(850nm), 可以:
- 过滤可见光干扰
- 增强火焰红外辐射
- 降低环境光影响

### 多光谱融合
```
RGB相机 ──→ 颜色检测 ──┐
                        ├──→ 融合判断 → 火源定位
热成像   ──→ 温度检测 ──┘
```

## 安全注意事项

1. **高度限制**: 设置最大飞行高度, 防止失控飞高
2. **电池监控**: 低电量自动返航
3. **遥控器接管**: 随时可切换手动控制
4. **灭火剂安全**: CO2灭火注意通风, 风扇灭火远离可燃物
5. **飞行区域**: 室内注意天花板/墙壁避障
