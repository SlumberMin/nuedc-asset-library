# 2014 全国大学生电子设计竞赛 - 风洞控制系统

## 题目概述
设计一个风洞控制系统，通过控制风扇转速产生不同风速的气流，并能控制风向（偏转板角度），实现对小球/旗帜等物体的悬浮、定位控制。

## Orange Pi 5 系统架构

### 核心方案：无刷电机调速 + 舵机角度控制 + 视觉/PID闭环

## 软件架构

### 1. 风速传感器读取

```python
import serial
import struct

class WindSpeedSensor:
    """通过STM32读取差压传感器/热线风速计"""
    def __init__(self, uart_port='/dev/ttyS3'):
        self.ser = serial.Serial(uart_port, 115200, timeout=0.05)

    def read_speed(self):
        """读取风速(m/s)"""
        self.ser.write(b'\x01\x03\x00\x00\x00\x01\x84\x0A')  # Modbus读取
        resp = self.ser.read(7)
        if len(resp) == 7:
            value = struct.unpack('>H', resp[3:5])[0]
            return value / 100.0  # 转换为m/s
        return None

    def read_angle(self):
        """读取偏转板角度(度)"""
        self.ser.write(b'\x02\x03\x00\x00\x00\x01\xxx\xxx')
        resp = self.ser.read(7)
        if len(resp) == 7:
            return struct.unpack('>H', resp[3:5])[0] / 10.0
        return None
```

### 2. 小球悬浮高度视觉检测

```python
import cv2
import numpy as np

class BallHeightDetector:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        # 标定参数（像素到实际高度的映射）
        self.pixels_per_cm = 10.0  # 需要标定

    def get_ball_height(self):
        ret, frame = self.cap.read()
        if not ret:
            return None
        # 检测小球（亮色球在深色风道中）
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        _, thresh = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            (x, y), radius = cv2.minEnclosingCircle(c)
            if radius > 5:
                # 小球底部Y坐标转高度
                height_cm = (frame.shape[0] - y) / self.pixels_per_cm
                return height_cm
        return None
```

### 3. PID控制器

```python
import time

class PIDController:
    def __init__(self, Kp, Ki, Kd, output_min, output_max):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.output_min = output_min
        self.output_max = output_max
        self.integral = 0
        self.prev_error = 0
        self.prev_time = time.time()

    def compute(self, setpoint, measured):
        now = time.time()
        dt = now - self.prev_time
        if dt <= 0:
            dt = 0.001
        self.prev_time = now

        error = setpoint - measured
        self.integral += error * dt
        self.integral = max(-100, min(100, self.integral))  # 积分限幅
        derivative = (error - self.prev_error) / dt
        self.prev_error = error

        output = self.Kp*error + self.Ki*self.integral + self.Kd*derivative
        return max(self.output_min, min(self.output_max, output))
```

### 4. 风洞主控制器

```python
class WindTunnelController:
    def __init__(self):
        self.sensor = WindSpeedSensor()
        self.detector = BallHeightDetector()
        # 风速PID（控制风扇转速）
        self.speed_pid = PIDController(Kp=5.0, Ki=0.5, Kd=1.0, output_min=0, output_max=100)
        # 角度PID（控制偏转板）
        self.angle_pid = PIDController(Kp=3.0, Ki=0.3, Kd=0.8, output_min=-45, output_max=45)

    def set_fan_speed(self, percent):
        """设置风扇转速 0-100%"""
        pwm_value = int(percent * 255 / 100)
        # 通过UART发送给STM32
        cmd = bytes([0xAA, 0x01, pwm_value, 0x55])
        self.sensor.ser.write(cmd)

    def set_deflector_angle(self, angle_deg):
        """设置偏转板角度 -45° ~ +45°"""
        pwm = int((angle_deg + 45) * 2000 / 90 + 500)  # 500-2500us
        cmd = bytes([0xAA, 0x02, (pwm >> 8) & 0xFF, pwm & 0xFF, 0x55])
        self.sensor.ser.write(cmd)

    def suspend_ball(self, target_height_cm):
        """悬浮小球到指定高度"""
        while True:
            height = self.detector.get_ball_height()
            if height is None:
                continue
            wind_speed = self.sensor.read_speed()
            if wind_speed is None:
                continue

            # 高度PID -> 风速设定
            speed_setpoint = self.speed_pid.compute(target_height_cm, height)
            # 风速PID -> 风扇PWM
            fan_pwm = self.speed_pid.compute(speed_setpoint, wind_speed)
            self.set_fan_speed(fan_pwm)

            print(f"高度: {height:.1f}cm, 风速: {wind_speed:.1f}m/s, 风扇: {fan_pwm:.0f}%")
            time.sleep(0.02)

    def run(self):
        """主控制循环"""
        target_height = 30.0  # cm
        self.suspend_ball(target_height)

if __name__ == "__main__":
    controller = WindTunnelController()
    controller.run()
```

### 5. STM32 PWM输出（关键代码）

```c
// 风扇ESC控制：50Hz PWM, 1000-2000us脉宽
void set_fan_esc(uint8_t percent) {
    uint16_t pulse = 1000 + (percent * 10);  // 1000-2000us
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, pulse);
}

// 舵机控制：50Hz PWM, 500-2500us脉宽
void set_servo_angle(int16_t angle_deg) {
    uint16_t pulse = 1500 + (angle_deg * 11);  // 约11us/度
    pulse = (pulse < 500) ? 500 : (pulse > 2500) ? 2500 : pulse;
    __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, pulse);
}
```

## 关键技术要点

### 串级PID控制
```
目标高度 ──→ [高度PID] ──→ 风速设定 ──→ [风速PID] ──→ 风扇PWM
                  ↑                           ↑
                  │                           │
              视觉高度反馈                 风速传感器
```

### 风速-高度关系建模
- 风洞中风速与高度近似平方关系：`v² ∝ h`
- 需要实验标定不同风扇转速下的风速分布
- 建立查找表或拟合公式用于前馈控制

### 稳定性优化
- 加入前馈项，预估维持高度所需的风速
- 使用卡尔曼滤波融合视觉和风速传感器数据
- 软启动：风扇转速缓慢上升，避免气流突变
