# 2020年电赛D题 - 绕障飞行器 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：四旋翼飞行器在室内环境中自主避障飞行，穿过指定通道、绕过障碍物，到达目标区域，全程无人遥控。

**关键指标**：
- 障碍物检测与规避
- 通道穿越能力
- 路径规划效率
- 飞行稳定性

### 4.1 深度感知模块

```python
# depth_perception.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class DepthPerception:
    """双目视觉+深度估计"""
    def __init__(self, model_path='midas_v2.rknn'):
        # 双目标定参数
        self.stereo = cv2.StereoBM_create(numDisparities=64, blockSize=15)
        
        # 或使用MiDaS深度估计模型
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0_1_2)
        
        # 摄像头参数
        self.focal_length = 3.6  # mm
        self.baseline = 60       # mm 双目基线
        
    def get_depth_stereo(self, frame_left, frame_right):
        """双目立体匹配计算深度"""
        gray_l = cv2.cvtColor(frame_left, cv2.COLOR_BGR2GRAY)
        gray_r = cv2.cvtColor(frame_right, cv2.COLOR_BGR2GRAY)
        
        disparity = self.stereo.compute(gray_l, gray_r).astype(np.float32) / 16.0
        
        # 转换为深度图
        depth = np.zeros_like(disparity)
        valid = disparity > 0
        depth[valid] = (self.focal_length * self.baseline) / disparity[valid]
        
        return depth
    
    def get_depth_mono(self, frame):
        """单目深度估计（MiDaS）"""
        img = cv2.resize(frame, (256, 256))
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        outputs = self.rknn.inference(inputs=[img])
        depth = outputs[0].reshape(256, 256)
        
        # 上采样到原图大小
        depth = cv2.resize(depth, (frame.shape[1], frame.shape[0]))
        return depth
    
    def detect_obstacles(self, depth_map, min_depth=200, max_depth=5000):
        """从深度图检测障碍物"""
        # 有效深度范围过滤
        mask = (depth_map > min_depth) & (depth_map < max_depth)
        
        # 分区域分析
        h, w = depth_map.shape
        regions = {
            'left': depth_map[:, :w//3],
            'center': depth_map[:, w//3:2*w//3],
            'right': depth_map[:, 2*w//3:],
            'upper': depth_map[:h//2, :],
            'lower': depth_map[h//2:, :],
            'center_near': depth_map[h//3:2*h//3, w//4:3*w//4]
        }
        
        obstacles = {}
        for name, region in regions.items():
            valid = region[(region > min_depth) & (region < max_depth)]
            if len(valid) > 0:
                obstacles[name] = {
                    'min_dist': float(np.min(valid)),
                    'mean_dist': float(np.mean(valid)),
                    'std_dist': float(np.std(valid))
                }
            else:
                obstacles[name] = {
                    'min_dist': max_depth,
                    'mean_dist': max_depth,
                    'std_dist': 0
                }
        
        return obstacles
    
    def find_passage(self, depth_map, min_width=800, max_depth=3000):
        """寻找可通过的通道"""
        h, w = depth_map.shape
        
        # 在中段高度扫描
        scan_row = depth_map[int(h*0.3):int(h*0.7), :]
        
        # 找深度大于阈值的连续区域（通道）
        depth_ok = np.mean(scan_row, axis=0) > max_depth
        
        # 找最长连续区域
        passages = []
        start = None
        for i in range(len(depth_ok)):
            if depth_ok[i] and start is None:
                start = i
            elif not depth_ok[i] and start is not None:
                width = i - start
                if width >= min_width // 10:  # 像素宽度
                    center = (start + i) // 2
                    passages.append({
                        'center_px': center,
                        'width_px': width,
                        'center_normalized': (center - w/2) / (w/2)
                    })
                start = None
        
        if passages:
            return max(passages, key=lambda p: p['width_px'])
        return None
```

### 4.2 避障策略

```python
# obstacle_avoidance.py
import numpy as np
from enum import Enum

class AvoidanceAction(Enum):
    GO_STRAIGHT = 0
    TURN_LEFT = 1
    TURN_RIGHT = 2
    GO_UP = 3
    GO_DOWN = 4
    SLOW_DOWN = 5
    STOP = 6

class ObstacleAvoidance:
    """避障决策引擎"""
    def __init__(self):
        self.safe_distance = 1500   # mm
        self.warning_distance = 800 # mm
        self.emergency_distance = 300 # mm
        
    def decide(self, obstacles, passage=None):
        """根据障碍信息决定动作"""
        center = obstacles.get('center_near', {})
        left = obstacles.get('left', {})
        right = obstacles.get('right', {})
        upper = obstacles.get('upper', {})
        lower = obstacles.get('lower', {})
        
        center_dist = center.get('min_dist', 9999)
        left_dist = left.get('min_dist', 9999)
        right_dist = right.get('min_dist', 9999)
        
        # 紧急避障
        if center_dist < self.emergency_distance:
            if left_dist > right_dist:
                return AvoidanceAction.TURN_LEFT, 1.0
            else:
                return AvoidanceAction.TURN_RIGHT, 1.0
        
        # 前方有障碍
        if center_dist < self.safe_distance:
            # 有通道则穿过去
            if passage is not None:
                offset = passage['center_normalized']
                if abs(offset) < 0.3:
                    return AvoidanceAction.GO_STRAIGHT, 0.5
                elif offset < 0:
                    return AvoidanceAction.TURN_LEFT, 0.5
                else:
                    return AvoidanceAction.TURN_RIGHT, 0.5
            
            # 无通道，选择较宽侧绕行
            if left_dist > right_dist + 200:
                return AvoidanceAction.TURN_LEFT, 0.6
            elif right_dist > left_dist + 200:
                return AvoidanceAction.TURN_RIGHT, 0.6
            else:
                # 上方是否有空间
                if upper.get('min_dist', 9999) > self.safe_distance * 2:
                    return AvoidanceAction.GO_UP, 0.4
                else:
                    return AvoidanceAction.TURN_LEFT, 0.5
        
        # 通路畅通
        return AvoidanceAction.GO_STRAIGHT, 1.0
    
    def compute_velocity(self, action, intensity):
        """将动作转换为速度指令"""
        base_speed = 0.5 * intensity
        
        velocities = {
            AvoidanceAction.GO_STRAIGHT: (base_speed, 0, 0),
            AvoidanceAction.TURN_LEFT: (base_speed * 0.5, -base_speed, 0),
            AvoidanceAction.TURN_RIGHT: (base_speed * 0.5, base_speed, 0),
            AvoidanceAction.GO_UP: (0, 0, -base_speed),
            AvoidanceAction.GO_DOWN: (0, 0, base_speed),
            AvoidanceAction.SLOW_DOWN: (base_speed * 0.3, 0, 0),
            AvoidanceAction.STOP: (0, 0, 0),
        }
        
        vx, vy, vz = velocities.get(action, (0, 0, 0))
        return vx, vy, vz
```

# path_planner.py
import numpy as np

class VFHPlanner:
    """VFH（Vector Field Histogram）路径规划"""
    def __init__(self, grid_size=100, cell_size=50):
        self.grid_size = grid_size
        self.cell_size = cell_size  # mm
        self.grid = np.zeros((grid_size, grid_size), dtype=np.float32)
        
    def update_obstacles(self, obstacles, current_pos):
        """更新障碍物栅格"""
        # 栅格衰减
        self.grid *= 0.95
        
        # 根据深度图更新障碍
        for region, info in obstacles.items():
            if info['min_dist'] < 3000:
                # 计算障碍在栅格中的位置
                self._mark_obstacle(current_pos, region, info['min_dist'])
    
    def plan(self, target_direction, current_heading):
        """VFH算法规划方向"""
        # 构建极坐标直方图
        histogram = np.zeros(72)  # 5度分辨率
        
        for angle_idx in range(72):
            angle = angle_idx * 5
            # 累加该方向上的障碍物密度
            histogram[angle_idx] = self._sector_density(angle)
        
        # 找候选方向（低障碍密度）
        threshold = 0.3
        candidates = []
        for i in range(72):
            if histogram[i] < threshold:
                candidates.append(i * 5)
        
        if not candidates:
            return current_heading  # 无安全方向，保持当前
        
        # 选择最接近目标方向的候选
        best = min(candidates, key=lambda a: abs(a - target_direction))
        return best
    
    def _sector_density(self, angle):
        """计算扇区障碍密度"""
        # 简化实现

### 4.4 任务主流程

```python
# main.py
import cv2
import time

def main():
    # 初始化
    cam_left = cv2.VideoCapture(0)
    cam_right = cv2.VideoCapture(2)
    depth = DepthPerception('midas_v2.rknn')
    avoidance = ObstacleAvoidance()
    planner = VFHPlanner()
    flight = MAVLinkController('/dev/ttyS5')
    
    # 任务参数
    target_pos = (10.0, 5.0, 3.0)  # 目标位置m
    waypoints = [(5.0, 0, 3.0), (10.0, 5.0, 3.0)]
    
    state = 'TAKEOFF'
    wp_idx = 0
    
    while True:
        ret_l, frame_l = cam_left.read()
        ret_r, frame_r = cam_right.read()
        if not ret_l:
            continue
        
        # 1. 深度感知
        depth_map = depth.get_depth_stereo(frame_l, frame_r)
        obstacles = depth.detect_obstacles(depth_map)
        passage = depth.find_passage(depth_map)
        
        # 2. 避障决策
        action, intensity = avoidance.decide(obstacles, passage)
        vx, vy, vz = avoidance.compute_velocity(action, intensity)
        
        # 3. 状态机
        if state == 'TAKEOFF':
            flight.takeoff(3.0)
            time.sleep(8)
            state = 'NAVIGATE'
            
        elif state == 'NAVIGATE':
            if action == AvoidanceAction.STOP:
                flight.set_velocity_body(0, 0, 0)
            else:
                # 避障优先，叠加航点引导
                wp = waypoints[wp_idx]
                wp_vx, wp_vy, wp_vz = flight.compute_guidance(wp)
                
                # 融合避障和导航速度
                final_vx = vx * 0.7 + wp_vx * 0.3
                final_vy = vy * 0.7 + wp_vy * 0.3
                final_vz = vz * 0.7 + wp_vz * 0.3
                
                flight.set_velocity_body(final_vx, final_vy, final_vz)
            
            # 到达航点？
            if flight.reached_waypoint(waypoints[wp_idx]):
                wp_idx += 1
                if wp_idx >= len(waypoints):
                    state = 'LAND'
                    
        elif state == 'LAND':
            flight.land()
            break
        
        time.sleep(0.05)  # 20Hz

if __name__ == '__main__':
    main()
```
