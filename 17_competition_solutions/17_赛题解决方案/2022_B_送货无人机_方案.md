# 2022年电赛B题 - 送货无人机 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：四旋翼无人机自主识别地面目标（二维码/色块），按指令将货物投递至指定位置，完成起降、巡航、识别、投放全流程。

**关键指标**：
- 起飞与降落精度
- 目标识别准确率
- 货物投递命中率
- 全程自主飞行稳定性

    """基于RKNN的送货目标检测"""
    def __init__(self, model_path='delivery_target.rknn'):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0_1_2)
        self.classes = ['qr_code', 'red_block', 'blue_block', 'green_block', 'landing_pad']
    
    def detect(self, frame):
        """检测目标并返回位置"""
        img = cv2.resize(frame, (640, 640))
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        outputs = self.rknn.inference(inputs=[img])
        
        detections = self._postprocess(outputs)
        results = []
        for det in detections:
            cls_id, conf, x1, y1, x2, y2 = det
            cx, cy = (x1 + x2) / 2, (y1 + y2) / 2
            results.append({
                'class': self.classes[int(cls_id)],
                'confidence': conf,
                'center': (cx, cy),
                'bbox': (x1, y1, x2, y2),
                'area': (x2 - x1) * (y2 - y1)
            })
        return results

import serial
import struct
import time

class MAVLinkController:
    """MAVLink协议飞控通信"""
    def __init__(self, port='/dev/ttyS5', baud=115200):
        self.ser = serial.Serial(port, baud, timeout=0.1)
    
    def takeoff(self, altitude=3.0):
        """起飞到指定高度"""
        self._send_command_long(
            command=22,  # MAV_CMD_NAV_TAKEOFF
            param7=altitude
        )
    
    def goto_waypoint(self, lat, lon, alt):
        """飞往航点"""
        self._send_mission_item(
            frame=6,  # GLOBAL_RELATIVE_ALT
            command=16,  # MAV_CMD_NAV_WAYPOINT
            lat=lat, lon=lon, alt=alt
        )
    
    def set_position_ned(self, north, east, down):
        """NED坐标系位置控制（相对）"""
        msg = self._build_set_position_target(
            type_mask=0b0000_1111_1111_1000,
            x=north, y=east, z=down
        )
        self.ser.write(msg)
    
    def set_velocity_body(self, vx, vy, vz, yaw_rate=0):
        """机体坐标系速度控制"""
        msg = self._build_set_position_target(

### 4.4 视觉降落与投递控制

```python
# delivery_controller.py
import cv2
import numpy as np
import time

class DeliveryController:
    """送货投递控制器"""
    def __init__(self, detector, flight_ctrl):
        self.detector = detector
        self.flight = flight_ctrl
        self.target_id = None
        self.state = 'IDLE'
        
        # 控制参数
        self.search_alt = 5.0      # 搜索高度
        self.approach_alt = 3.0    # 接近高度
        self.drop_alt = 1.5        # 投放高度
        self.pos_tolerance = 30    # 像素容差
        
    def search_target(self, frame):
        """搜索投递目标"""
        detections = self.detector.detect(frame)
        for det in detections:
            if det['class'] in ['red_block', 'blue_block', 'green_block']:
                if det['confidence'] > 0.7:
                    self.target_id = det['class']
                    self.state = 'APPROACH'
                    return det
        return None
    
    def approach_target(self, frame):
        """视觉引导接近目标"""
        detections = self.detector.detect(frame)
        target = self._find_target(detections)
        
        if target is None:
            return False, "目标丢失"
        
        cx, cy = target['center']
        h, w = frame.shape[:2]
        
        # 计算图像中心偏差
        err_x = cx - w / 2
        err_y = cy - h / 2
        
        # PID控制无人机位置
        vx = -self.pid_y.update(err_y) * 0.3  # 前进
        vy = self.pid_x.update(err_x) * 0.3   # 横移
        
        # 根据目标面积判断高度（面积越大越近）
        area = target['area']
        if area < 5000:  # 太远，下降
            vz = -0.2
        elif area > 15000:  # 太近，上升
            vz = 0.2
        else:
            vz = 0
        
        self.flight.set_velocity_body(vx, vy, vz)
        
        # 判断是否到达投放点
        if abs(err_x) < self.pos_tolerance and abs(err_y) < self.pos_tolerance:
            altitude = self.flight.get_altitude()
            if altitude < self.drop_alt + 0.3:
                return True, "到达投放点"
        
        return False, "接近中"
    
    def drop_cargo(self):
        """投放货物"""
        self.flight.release_cargo(servo_id=5, pwm=2000)
        time.sleep(0.5)
        self.flight.set_velocity_body(0, 0, 0.3)  # 投后上升
        self.state = 'RETURN'
    
    def precision_landing(self, frame):
        """精确降落（利用地面标记）"""
        detections = self.detector.detect(frame)
        pad = None
        for det in detections:
            if det['class'] == 'landing_pad' and det['confidence'] > 0.8:
                pad = det
                break
        
        if pad is None:
            # 无标记，直接GPS降落
            return False
        
        cx, cy = pad['center']
        h, w = frame.shape[:2]
        err_x = cx - w / 2
        err_y = cy - h / 2
        
        vx = -self.pid_y.update(err_y) * 0.15
        vy = self.pid_x.update(err_x) * 0.15
        vz = -0.15  # 缓慢下降
        
        self.flight.set_velocity_body(vx, vy, vz)
        
        if pad['area'] > 50000:  # 足够近
            return True
        return False
```

## 五、关键技术要点

### 5.1 目标识别策略
- **二维码识别**：OpenCV QRCodeDetector解码投递指令
- **色块识别**：HSV色彩空间分割 + 形态学过滤
- **降落标记**：ArUco marker精确定位

### 5.2 投递精度保障
- 高空搜索 → 低空确认 → 缓慢接近 → 精确投放
- 面积反馈高度控制（目标面积∝距离倒数）
- 双PID位置环（图像误差→速度指令）
