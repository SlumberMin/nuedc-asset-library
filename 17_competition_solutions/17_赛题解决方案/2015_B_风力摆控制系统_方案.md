## 二、视觉方案

### 2.1 俯视相机追踪激光点
```python
import cv2
import numpy as np

class LaserTracker:
    def __init__(self, camera_id=0, resolution=(640, 480)):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, resolution[0])
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, resolution[1])
        self.cap.set(cv2.CAP_PROP_FPS, 120)
        
        # 激光点HSV范围（红色激光）
        self.lower_hsv1 = np.array([0, 150, 200])
        self.upper_hsv1 = np.array([10, 255, 255])
        self.lower_hsv2 = np.array([170, 150, 200])
        self.upper_hsv2 = np.array([180, 255, 255])
        
        # 坐标映射矩阵
        self.H = None
    
    def calibrate(self, pixel_pts, real_pts):
        """标定像素→实际坐标(mm)"""
        self.H, _ = cv2.findHomography(
            np.float32(pixel_pts), np.float32(real_pts)
        )
    
    def detect_laser(self, frame):
        """检测激光点位置，返回实际坐标(mm)"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 红色激光（跨0°的H范围）
        mask1 = cv2.inRange(hsv, self.lower_hsv1, self.upper_hsv1)
        mask2 = cv2.inRange(hsv, self.lower_hsv2, self.upper_hsv2)
        mask = cv2.bitwise_or(mask1, mask2)
        
        # 形态学去噪
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((3,3)))
        mask = cv2.morphologyEx(mask, cv2.MORPH_DILATE, np.ones((3,3)))
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None
        
        # 取最大亮斑
        c = max(contours, key=cv2.contourArea)
        M = cv2.moments(c)
        if M["m00"] == 0:
            return None
        
        px = M["m10"] / M["m00"]
        py = M["m01"] / M["m00"]
        
        # 像素→物理坐标
        pixel = np.array([[[px, py]]], dtype=np.float32)
        real = cv2.perspectiveTransform(pixel, self.H)
        x_mm, y_mm = real[0][0]
        
        return (x_mm, y_mm)
    
    def get_trajectory_params(self, points_buffer, fps):
        """
        从轨迹点计算摆动参数
        points_buffer: 最近N帧的(x,y)坐标
        返回: (radius, angle, angular_velocity)
        """
        if len(points_buffer) < 10:
            return None
        
        pts = np.array(points_buffer)
        
        # 最小二乘拟合圆
        center, radius = self.fit_circle(pts)
        
        # 计算角速度
        angles = np.arctan2(pts[:,1]-center[1], pts[:,0]-center[0])
        angle_diff = np.diff(np.unwrap(angles))
        angular_vel = np.mean(angle_diff) * fps  # rad/s
        
        current_angle = angles[-1]
        
        return {
            'center': center,
            'radius': radius,
            'angle': current_angle,
            'angular_velocity': angular_vel
        }
    
    def fit_circle(self, points):
        """最小二乘拟合圆，返回(圆心, 半径)"""
        x = points[:, 0]
        y = points[:, 1]
        
        A = np.column_stack([2*x, 2*y, np.ones(len(x))])
        b = x**2 + y**2
        
        result = np.linalg.lstsq(A, b, rcond=None)[0]
        cx, cy = result[0], result[1]
        radius = np.sqrt(result[2] + cx**2 + cy**2)
        
        return (cx, cy), radius
```

### 3.2 坐标分解到四风机
```python
class FanController:
    def __init__(self):
        # 风机安装角度（相对于X轴正方向，逆时针）
        self.fan_angles = [90, 0, 270, 180]  # 上、右、下、左
        self.fan_pwm = [0, 0, 0, 0]
        
        # PID控制器
        self.pid_x = PID(Kp=2.0, Ki=0.05, Kd=0.8)
        self.pid_y = PID(Kp=2.0, Ki=0.05, Kd=0.8)
        
        self.max_pwm = 100  # 最大占空比%
    
    def compute_fan_outputs(self, target, current, dt):
        """
        target: (x_mm, y_mm)
        current: (x_mm, y_mm)
        返回: [fan1_pwm, fan2_pwm, fan3_pwm, fan4_pwm]
        """
        err_x = target[0] - current[0]
        err_y = target[1] - current[1]
        
        # PID输出 → XY方向力
        fx = self.pid_x.compute(err_x, dt)
        fy = self.pid_y.compute(err_y, dt)
        
        # 分解到四风机
        # 风机1(+Y)、风机2(+X)、风机3(-Y)、风机4(-X)
        fan_output = [
            max(0, fy),              # 风机1: Y正方向
            max(0, fx),              # 风机2: X正方向
            max(0, -fy),             # 风机3: Y负方向
            max(0, -fx),             # 风机4: X负方向
        ]
        
        # 限幅
        fan_output = [np.clip(p, 0, self.max_pwm) for p in fan_output]
        
        return fan_output
    
    def compute_circle_fan(self, angle, radius, center, current_angular_vel, target_angular_vel):
        """
        画圆模式：计算四风机输出
        angle: 当前角度(rad)
        radius: 当前半径
        """
        # 切向力（维持角速度）
        tangent_force = self.pid_speed.compute(
            target_angular_vel - current_angular_vel, dt=0.01
        )
        
        # 径向力（维持半径）
        radial_force = self.pid_radius.compute(
            target_radius - radius, dt=0.01
        )
        
        # 合力分解到XY
        fx = tangent_force * np.cos(angle + np.pi/2) + radial_force * np.cos(angle)
        fy = tangent_force * np.sin(angle + np.pi/2) + radial_force * np.sin(angle)
        
        return self._xy_to_fans(fx, fy)
```

### 3.3 画圆算法
```python
class CircleDrawer:
    def __init__(self, fan_ctrl, laser_tracker):
        self.fan_ctrl = fan_ctrl
        self.tracker = laser_tracker
        self.target_radius = 150  # mm
        self.target_omega = 1.0   # rad/s
    
    def update(self, current_pos, dt):
        """每帧调用"""
        # 计算当前极坐标
        dx = current_pos[0] - self.center[0]
        dy = current_pos[1] - self.center[1]
        r = np.sqrt(dx**2 + dy**2)
        theta = np.arctan2(dy, dx)
        
        # 目标位置（匀速圆周运动）
        target_theta = theta + self.target_omega * dt
        target_x = self.center[0] + self.target_radius * np.cos(target_theta)
        target_y = self.center[1] + self.target_radius * np.sin(target_theta)
        
        return self.fan_ctrl.compute_fan_outputs(
            (target_x, target_y), current_pos, dt
        )
```

### 3.4 画直线算法
```python
class LineDrawer:
    def __init__(self, fan_ctrl, start, end):
        self.fan_ctrl = fan_ctrl
        self.start = np.array(start)
        self.end = np.array(end)
        self.direction = self.end - self.start
        self.direction = self.direction / np.linalg.norm(self.direction)
        self.normal = np.array([-self.direction[1], self.direction[0]])
        self.line_progress = 0
        self.oscillation_freq = 0.5  # Hz
        self.amplitude = 150  # mm
    
    def get_target(self, t):
        """直线往复运动目标点"""
        # 沿直线方向的往复
        progress = self.amplitude * np.sin(2 * np.pi * self.oscillation_freq * t)
        midpoint = (self.start + self.end) / 2
        target = midpoint + self.direction * progress
        return target
```
