# 2012年D题：声音定位系统 - Orange Pi 5解决方案

## 题目概述

设计一个声音定位系统，通过多个麦克风阵列接收声音信号，计算声源的方位角和距离，并在显示器上显示声源位置。系统需要能够实时追踪移动声源。

```python
#!/usr/bin/env python3
"""
音频采集模块 - Orange Pi 5 I2S
"""

import numpy as np
import threading
import queue
import time

class I2SAudioCapture:
    """I2S音频采集"""
    
    def __init__(self, sample_rate=16000, channels=4, chunk_size=1024):
        self.sample_rate = sample_rate
        self.channels = channels
        self.chunk_size = chunk_size
        
        # 音频缓冲区
        self.buffer_size = sample_rate * 2  # 2秒缓冲
        self.buffers = [np.zeros(self.buffer_size, dtype=np.int32) for _ in range(channels)]
        self.write_pos = 0
        
        # 采集状态
        self.is_running = False
        self.capture_thread = None
        
        # 音频队列
        self.audio_queue = queue.Queue(maxsize=100)
        
    def start(self):
        """开始采集"""
        self.is_running = True
        self.capture_thread = threading.Thread(target=self._capture_loop)
        self.capture_thread.daemon = True
        self.capture_thread.start()
    
    def stop(self):
        """停止采集"""
        self.is_running = False
        if self.capture_thread:
            self.capture_thread.join()
    
    def _capture_loop(self):
        """采集循环"""
        # 实际实现需要使用ALSA或PyAudio
        # 这里使用模拟数据
        
        while self.is_running:
            # 生成模拟音频数据
            chunk = self._generate_test_audio()
            
            # 存入缓冲区
            for ch in range(self.channels):
                end_pos = self.write_pos + self.chunk_size
                if end_pos <= self.buffer_size:
                    self.buffers[ch][self.write_pos:end_pos] = chunk[ch]
                else:
                    # 循环缓冲
                    first_part = self.buffer_size - self.write_pos
                    second_part = self.chunk_size - first_part
                    self.buffers[ch][self.write_pos:] = chunk[ch][:first_part]
                    self.buffers[ch][:second_part] = chunk[ch][first_part:]
            
            self.write_pos = (self.write_pos + self.chunk_size) % self.buffer_size
            
            # 放入队列
            self.audio_queue.put(chunk)
            
            # 控制采集速度
            time.sleep(self.chunk_size / self.sample_rate)
    
    def _generate_test_audio(self):
        """生成测试音频 (用于调试)"""
        t = np.arange(self.chunk_size) / self.sample_rate
        
        # 模拟声源位置 (角度45度，距离1米)
        angle = np.radians(45)
        distance = 1.0
        
        # 声速
        c = 343.0
        
        # 计算各麦克风的延迟
        delays = []
        for mic_pos in MICROPHONE_POSITIONS:
            # 计算声源到麦克风的距离
            dx = distance * np.cos(angle) - mic_pos[0]
            dy = distance * np.sin(angle) - mic_pos[1]
            dist = np.sqrt(dx**2 + dy**2)
            delay = dist / c
            delays.append(delay)
        
        # 生成音频信号
        frequency = 1000  # 1kHz测试信号
        chunk = []
        for ch in range(self.channels):
            phase = 2 * np.pi * frequency * delays[ch]
            signal = np.sin(2 * np.pi * frequency * t + phase) * 32767
            signal += np.random.randn(self.chunk_size) * 1000  # 添加噪声
            chunk.append(signal.astype(np.int32))
        
        return chunk
    
    def get_latest_chunk(self):
        """获取最新的音频块"""
        try:
            return self.audio_queue.get(timeout=0.1)
        except queue.Empty:
            return None
    
    def get_buffer_data(self, channel, num_samples):
        """获取缓冲区数据"""
        end_pos = self.write_pos
        start_pos = (end_pos - num_samples) % self.buffer_size
        
        if start_pos < end_pos:
            return self.buffers[channel][start_pos:end_pos]
        else:
            return np.concatenate([
                self.buffers[channel][start_pos:],
                self.buffers[channel][:end_pos]
            ])


class USBAudioCapture:
    """USB音频采集"""
    
    def __init__(self, sample_rate=16000, channels=4, chunk_size=1024):
        self.sample_rate = sample_rate
        self.channels = channels
        self.chunk_size = chunk_size
        
        # PyAudio初始化
        import pyaudio
        self.pa = pyaudio.PyAudio()
        
        # 查找USB音频设备
        self.device_index = self._find_usb_audio_device()
        
        # 打开音频流
        self.stream = self.pa.open(
            format=pyaudio.paInt32,
            channels=channels,
            rate=sample_rate,
            input=True,
            input_device_index=self.device_index,
            frames_per_buffer=chunk_size
        )
        
        self.is_running = False
    
    def _find_usb_audio_device(self):
        """查找USB音频设备"""
        for i in range(self.pa.get_device_count()):
            info = self.pa.get_device_info_by_index(i)
            if 'USB' in info['name'] and info['maxInputChannels'] >= self.channels:
                return i
        return None
    
    def start(self):
        """开始采集"""
        self.is_running = True
        self.stream.start_stream()
    
    def stop(self):
        """停止采集"""
        self.is_running = False
        self.stream.stop_stream()
        self.stream.close()
        self.pa.terminate()
    
    def read_chunk(self):
        """读取一个音频块"""
        data = self.stream.read(self.chunk_size, exception_on_overflow=False)
        # 转换为numpy数组
        audio_data = np.frombuffer(data, dtype=np.int32)
        # 分离各通道
        channels = [audio_data[i::self.channels] for i in range(self.channels)]
        return channels
```

### 2. TDOA定位算法

```python
#!/usr/bin/env python3
"""
TDOA定位算法
"""

import numpy as np
from scipy.optimize import minimize

class TDOALocalizer:
    """TDOA定位器"""
    
    def __init__(self, mic_positions, sample_rate=16000):
        """
        mic_positions: 麦克风位置列表 [(x1,y1), (x2,y2), ...]
        """
        self.mic_positions = np.array(mic_positions)
        self.sample_rate = sample_rate
        self.num_mics = len(mic_positions)
        
        # 声速 (m/s)
        self.speed_of_sound = 343.0
        
        # 互相关参数
        self.max_delay_samples = int(0.001 * sample_rate)  # 最大延迟1ms
    
    def gcc_phat(self, sig1, sig2):
        """
        广义互相关-相位变换 (GCC-PHAT)
        返回: 延迟（采样点数）
        """
        n = len(sig1) + len(sig2) - 1
        
        # FFT
        SIG1 = np.fft.rfft(sig1, n)
        SIG2 = np.fft.rfft(sig2, n)
        
        # 互功率谱
        R = SIG1 * np.conj(SIG2)
        
        # PHAT加权
        magnitude = np.abs(R)
        magnitude[magnitude < 1e-10] = 1e-10
        R_phat = R / magnitude
        
        # IFFT得到互相关函数
        cc = np.fft.irfft(R_phat, n)
        
        # 找到最大值
        max_shift = self.max_delay_samples
        cc = np.concatenate([cc[-max_shift:], cc[:max_shift+1]])
        
        # 找到峰值
        peak_idx = np.argmax(np.abs(cc))
        delay = peak_idx - max_shift
        
        return delay
    
    def calculate_tdoa(self, audio_data):
        """
        计算各麦克风对的TDOA
        audio_data: [mic1_data, mic2_data, mic3_data, mic4_data]
        返回: TDOA矩阵
        """
        tdoa = np.zeros((self.num_mics, self.num_mics))
        
        for i in range(self.num_mics):
            for j in range(i+1, self.num_mics):
                delay_samples = self.gcc_phat(audio_data[i], audio_data[j])
                delay_time = delay_samples / self.sample_rate
                tdoa[i][j] = delay_time
                tdoa[j][i] = -delay_time
        
        return tdoa
    
    def localize_tdoa(self, tdoa):
        """
        使用TDOA定位声源
        返回: (x, y) 坐标
        """
        # 选择参考麦克风 (MIC 0)
        ref_idx = 0
        
        # 构建方程组
        # 使用最小二乘法求解
        A = []
        b = []
        
        for i in range(1, self.num_mics):
            # 距离差 = TDOA * 声速
            distance_diff = tdoa[ref_idx][i] * self.speed_of_sound
            
            # 线性化方程
            xi = self.mic_positions[i][0]
            yi = self.mic_positions[i][1]
            x0 = self.mic_positions[ref_idx][0]
            y0 = self.mic_positions[ref_idx][1]
            
            A.append([2*(xi-x0), 2*(yi-y0)])
            b.append(distance_diff**2 - (xi**2 + yi**2) + (x0**2 + y0**2))
        
        A = np.array(A)
        b = np.array(b)
        
        # 最小二乘求解
        try:
            result = np.linalg.lstsq(A, b, rcond=None)
            position = result[0]
            return position
        except:
            return None
    
    def localize_sphere(self, audio_data):
        """
        球面插值法定位
        更鲁棒的定位方法
        """
        # 计算TDOA
        tdoa = self.calculate_tdoa(audio_data)
        
        # 初始猜测
        x0 = [0.0, 0.0]
        
        # 目标函数：最小化TDOA误差
        def objective(params):
            x, y = params
            error = 0
            
            for i in range(self.num_mics):
                for j in range(i+1, self.num_mics):
                    # 计算理论TDOA
                    di = np.sqrt((x - self.mic_positions[i][0])**2 + 
                                (y - self.mic_positions[i][1])**2)
                    dj = np.sqrt((x - self.mic_positions[j][0])**2 + 
                                (y - self.mic_positions[j][1])**2)
                    
                    theoretical_tdoa = (di - dj) / self.speed_of_sound
                    
                    # 误差
                    error += (theoretical_tdoa - tdoa[i][j])**2
            
            return error
        
        # 优化
        result = minimize(objective, x0, method='Nelder-Mead')
        
        if result.success:
            return result.x
        return None
    
    def estimate_distance(self, audio_data, position):
        """
        估算声源距离
        基于信号能量衰减
        """
        if position is None:
            return None
        
        # 计算到各麦克风的距离
        distances = []
        for mic_pos in self.mic_positions:
            d = np.sqrt((position[0] - mic_pos[0])**2 + 
                       (position[1] - mic_pos[1])**2)
            distances.append(d)
        
        # 平均距离
        avg_distance = np.mean(distances)
        
        # 使用信号能量修正距离估计
        energies = [np.mean(data**2) for data in audio_data]
        avg_energy = np.mean(energies)
        
        # 能量与距离的平方成反比
        # 参考能量（1米处）
        ref_energy = 1e6  # 需要校准
        estimated_distance = np.sqrt(ref_energy / avg_energy)
        
        # 结合几何距离和能量估计
        final_distance = 0.5 * avg_distance + 0.5 * estimated_distance
        
        return final_distance
    
    def calculate_angle(self, position):
        """
        计算声源方位角
        返回: 角度（度）
        """
        if position is None:
            return None
        
        x, y = position
        angle = np.degrees(np.arctan2(y, x))
        
        # 转换到0-360度
        if angle < 0:
            angle += 360
        
        return angle
```

### 3. 声源追踪滤波

```python
#!/usr/bin/env python3
"""
声源追踪 - 卡尔曼滤波
"""

import numpy as np

class KalmanTracker:
    """卡尔曼滤波追踪器"""
    
    def __init__(self, dt=0.05):
        self.dt = dt  # 采样间隔
        
        # 状态向量: [x, y, vx, vy]
        self.state = np.zeros(4)
        
        # 状态协方差
        self.P = np.eye(4) * 100
        
        # 状态转移矩阵
        self.F = np.array([
            [1, 0, dt, 0],
            [0, 1, 0, dt],
            [0, 0, 1, 0],
            [0, 0, 0, 1]
        ])
        
        # 观测矩阵
        self.H = np.array([
            [1, 0, 0, 0],
            [0, 1, 0, 0]
        ])
        
        # 过程噪声
        self.Q = np.array([
            [dt**4/4, 0, dt**3/2, 0],
            [0, dt**4/4, 0, dt**3/2],
            [dt**3/2, 0, dt**2, 0],
            [0, dt**3/2, 0, dt**2]
        ]) * 0.1
        
        # 观测噪声
        self.R = np.eye(2) * 0.5
        
        # 是否已初始化
        self.initialized = False
        
        # 历史轨迹
        self.history = []
    
    def initialize(self, x, y):
        """初始化追踪器"""
        self.state = np.array([x, y, 0, 0])
        self.initialized = True
        self.history = [(x, y)]
    
    def predict(self):
        """预测步骤"""
        # 状态预测
        self.state = self.F @ self.state
        
        # 协方差预测
        self.P = self.F @ self.P @ self.F.T + self.Q
        
        return self.state[:2]
    
    def update(self, measurement):
        """更新步骤"""
        if not self.initialized:
            self.initialize(measurement[0], measurement[1])
            return self.state[:2]
        
        # 预测
        self.predict()
        
        # 计算卡尔曼增益
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        # 更新状态
        y = measurement - self.H @ self.state
        self.state = self.state + K @ y
        
        # 更新协方差
        I = np.eye(4)
        self.P = (I - K @ self.H) @ self.P
        
        # 保存历史
        self.history.append(tuple(self.state[:2]))
        
        return self.state[:2]
    
    def get_position(self):
        """获取当前位置"""
        return self.state[:2]
    
    def get_velocity(self):
        """获取当前速度"""
        return self.state[2:]
    
    def get_speed(self):
        """获取速度大小"""
        vel = self.get_velocity()
        return np.sqrt(vel[0]**2 + vel[1]**2)
    
    def get_direction(self):
        """获取运动方向（角度）"""
        vel = self.get_velocity()
        return np.degrees(np.arctan2(vel[1], vel[0]))
    
    def is_moving(self):
        """判断是否在移动"""
        return self.get_speed() > 0.1  # 阈值


class ParticleTracker:
    """粒子滤波追踪器"""
    
    def __init__(self, num_particles=100):
        self.num_particles = num_particles
        
        # 粒子状态 [x, y, vx, vy]
        self.particles = np.zeros((num_particles, 4))
        self.weights = np.ones(num_particles) / num_particles
        
        # 初始化粒子
        self._initialize_particles()
    
    def _initialize_particles(self):
        """初始化粒子"""
        # 在工作空间内随机分布
        self.particles[:, 0] = np.random.uniform(-5, 5, self.num_particles)  # x
        self.particles[:, 1] = np.random.uniform(-5, 5, self.num_particles)  # y
        self.particles[:, 2] = np.random.randn(self.num_particles) * 0.1     # vx
        self.particles[:, 3] = np.random.randn(self.num_particles) * 0.1     # vy
    
    def predict(self, dt=0.05):
        """预测步骤"""
        # 状态转移
        self.particles[:, 0] += self.particles[:, 2] * dt
        self.particles[:, 1] += self.particles[:, 3] * dt
        
        # 添加过程噪声
        self.particles[:, 0] += np.random.randn(self.num_particles) * 0.1
        self.particles[:, 1] += np.random.randn(self.num_particles) * 0.1
        self.particles[:, 2] += np.random.randn(self.num_particles) * 0.05
        self.particles[:, 3] += np.random.randn(self.num_particles) * 0.05
    
    def update(self, measurement):
        """更新步骤"""
        # 计算每个粒子的似然度
        for i in range(self.num_particles):
            dx = self.particles[i, 0] - measurement[0]
            dy = self.particles[i, 1] - measurement[1]
            distance = np.sqrt(dx**2 + dy**2)
            
            # 高斯似然
            self.weights[i] = np.exp(-distance**2 / 0.5)
        
        # 归一化权重
        self.weights /= np.sum(self.weights)
        
        # 重采样
        self._resample()
    
    def _resample(self):
        """重采样"""
        indices = np.random.choice(
            self.num_particles,
            size=self.num_particles,
            p=self.weights
        )
        
        self.particles = self.particles[indices]
        self.weights = np.ones(self.num_particles) / self.num_particles
    
    def get_estimate(self):
        """获取估计位置"""
        # 加权平均
        x = np.average(self.particles[:, 0], weights=self.weights)
        y = np.average(self.particles[:, 1], weights=self.weights)
        return np.array([x, y])
    
    def get_uncertainty(self):
        """获取不确定性"""
        x_var = np.average((self.particles[:, 0] - self.get_estimate()[0])**2, 
                          weights=self.weights)
        y_var = np.average((self.particles[:, 1] - self.get_estimate()[1])**2, 
                          weights=self.weights)
        return np.array([x_var, y_var])
```

```python
        self.width = width
        self.height = height
        
        # 显示范围
        self.x_range = (-5, 5)  # 米
        self.y_range = (-5, 5)
        
        # 颜色定义
        self.colors = {
            'background': (30, 30, 30),
            'grid': (60, 60, 60),
            'mic': (0, 255, 0),
            'sound': (0, 0, 255),
            'trail': (0, 100, 255),
            'text': (255, 255, 255)
        }
        
        # 轨迹历史
        self.trail_history = []
        self.max_trail_length = 100
    
    def world_to_pixel(self, x, y):
        """世界坐标转像素坐标"""
        px = int((x - self.x_range[0]) / (self.x_range[1] - self.x_range[0]) * self.width)
        py = int((y - self.y_range[0]) / (self.y_range[1] - self.y_range[0]) * self.height)
        return (px, py)
    
    def draw_background(self, frame):
        """绘制背景和网格"""
        # 背景
        frame[:] = self.colors['background']
        
        # 网格线
        for x in np.arange(self.x_range[0], self.x_range[1] + 1, 1):
            px, _ = self.world_to_pixel(x, 0)
            cv2.line(frame, (px, 0), (px, self.height), self.colors['grid'], 1)
        
        for y in np.arange(self.y_range[0], self.y_range[1] + 1, 1):
            _, py = self.world_to_pixel(0, y)
            cv2.line(frame, (0, py), (self.width, py), self.colors['grid'], 1)
        
        # 坐标轴
        center_x, center_y = self.world_to_pixel(0, 0)
        cv2.line(frame, (center_x, 0), (center_x, self.height), (100, 100, 100), 2)
        cv2.line(frame, (0, center_y), (self.width, center_y), (100, 100, 100), 2)
        
        return frame
    
    def draw_microphones(self, frame, mic_positions):
        """绘制麦克风位置"""
        for i, (x, y) in enumerate(mic_positions):
            px, py = self.world_to_pixel(x, y)
            
            # 绘制麦克风标记
            cv2.circle(frame, (px, py), 8, self.colors['mic'], -1)
            cv2.putText(frame, f'MIC{i+1}', (px+10, py), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['mic'], 1)
        
        return frame
    
    def draw_sound_source(self, frame, position, confidence=None):
        """绘制声源位置"""
        if position is None:
            return frame
        
        x, y = position
        px, py = self.world_to_pixel(x, y)
        
        # 添加到轨迹
        self.trail_history.append((px, py))
        if len(self.trail_history) > self.max_trail_length:
            self.trail_history.pop(0)
        
        # 绘制轨迹
        if len(self.trail_history) > 1:
            for i in range(1, len(self.trail_history)):
                # 轨迹颜色渐变
                alpha = i / len(self.trail_history)
                color = (
                    int(self.colors['trail'][0] * alpha),
                    int(self.colors['trail'][1] * alpha),
                    int(self.colors['trail'][2] * alpha)
                )
                cv2.line(frame, self.trail_history[i-1], self.trail_history[i], 
                        color, 2)
        
        # 绘制当前位置
        cv2.circle(frame, (px, py), 15, self.colors['sound'], -1)
        cv2.circle(frame, (px, py), 20, self.colors['sound'], 2)
        
        # 显示坐标
        cv2.putText(frame, f'({x:.2f}, {y:.2f})', (px+25, py-10),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['text'], 1)
        
        # 如果有置信度，显示
        if confidence is not None:
            cv2.putText(frame, f'Conf: {confidence:.2f}', (px+25, py+10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
        
        return frame
    
    def draw_info_panel(self, frame, angle, distance, speed=None):
        """绘制信息面板"""
        # 背景
        panel_x = 10
        panel_y = 10
        panel_width = 200
        panel_height = 120
        cv2.rectangle(frame, (panel_x, panel_y), 
                     (panel_x + panel_width, panel_y + panel_height),
                     (50, 50, 50), -1)
        cv2.rectangle(frame, (panel_x, panel_y), 
                     (panel_x + panel_width, panel_y + panel_height),
                     (100, 100, 100), 2)
        
        # 信息文字
        y_offset = panel_y + 25
        cv2.putText(frame, 'Sound Source Info', (panel_x + 10, y_offset),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['text'], 1)
        
        y_offset += 25
        cv2.putText(frame, f'Angle: {angle:.1f} deg', (panel_x + 10, y_offset),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
        
        y_offset += 20
        cv2.putText(frame, f'Distance: {distance:.2f} m', (panel_x + 10, y_offset),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
        
        if speed is not None:
            y_offset += 20
            cv2.putText(frame, f'Speed: {speed:.2f} m/s', (panel_x + 10, y_offset),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
        
        return frame
    
    def draw_angle_indicator(self, frame, angle):
        """绘制角度指示器"""
        center_x = self.width - 80
        center_y = 80
        radius = 50
        
        # 绘制圆形
        cv2.circle(frame, (center_x, center_y), radius, (100, 100, 100), 2)
        
        # 绘制角度线
        angle_rad = np.radians(angle)
        end_x = int(center_x + radius * np.cos(angle_rad))
        end_y = int(center_y + radius * np.sin(angle_rad))
        cv2.line(frame, (center_x, center_y), (end_x, end_y), 
                self.colors['sound'], 3)
        
        # 绘制角度值
        cv2.putText(frame, f'{angle:.0f}°', (center_x - 15, center_y + radius + 20),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['text'], 1)
        
        return frame
    
    def draw_direction_arrow(self, frame, from_pos, to_pos):
        """绘制方向箭头"""
        if from_pos is None or to_pos is None:
            return frame
        
        px1, py1 = self.world_to_pixel(from_pos[0], from_pos[1])
        px2, py2 = self.world_to_pixel(to_pos[0], to_pos[1])
        
        # 绘制箭头
        cv2.arrowedLine(frame, (px1, py1), (px2, py2), 
                       self.colors['sound'], 2, tipLength=0.1)
        
        return frame
    
    def clear_trail(self):
        """清除轨迹"""
        self.trail_history.clear()
```

```python
#!/usr/bin/env python3
"""
声音定位系统主程序
"""

import cv2
import numpy as np
import time
from audio_capture import I2SAudioCapture
from tdoa_localizer import TDOALocalizer
from kalman_tracker import KalmanTracker
from sound_visualizer import SoundVisualizer

class SoundLocalizationSystem:
    """声音定位系统"""
    
    def __init__(self):
        # 麦克风位置
        self.mic_positions = [
            (0.075, 0.075),    # MIC 1
            (0.075, -0.075),   # MIC 2
            (-0.075, -0.075),  # MIC 3
            (-0.075, 0.075),   # MIC 4
        ]
        
        # 初始化组件
        self.audio_capture = I2SAudioCapture(sample_rate=16000, channels=4)
        self.localizer = TDOALocalizer(self.mic_positions)
        self.tracker = KalmanTracker()
        self.visualizer = SoundVisualizer(800, 600)
        
        # 运行状态
        self.is_running = False
        
        # 统计信息
        self.frame_count = 0
        self.fps = 0
        self.last_fps_time = time.time()
    
    def start(self):
        """启动系统"""
        print("启动声音定位系统...")
        
        self.audio_capture.start()
        self.is_running = True
        
        print("系统已启动")
    
    def stop(self):
        """停止系统"""
        self.is_running = False
        self.audio_capture.stop()
        print("系统已停止")
    
    def process_audio(self, audio_data):
        """处理音频数据"""
        # 计算TDOA
        tdoa = self.localizer.calculate_tdoa(audio_data)
        
        # 定位声源
        position = self.localizer.localize_sphere(audio_data)
        
        if position is not None:
            # 更新追踪器
            tracked_position = self.tracker.update(position)
            
            # 计算角度和距离
            angle = self.localizer.calculate_angle(tracked_position)
            distance = self.localizer.estimate_distance(audio_data, tracked_position)
            speed = self.tracker.get_speed()
            
            return {
                'position': tracked_position,
                'angle': angle,
                'distance': distance,
                'speed': speed,
                'raw_position': position
            }
        
        return None
    
    def run(self):
        """主循环"""
        self.start()
        
        print("按 'q' 退出")
        print("按 'c' 清除轨迹")
        print("按 'r' 重置追踪器")
        
        try:
            while self.is_running:
                # 获取音频数据
                audio_data = self.audio_capture.get_latest_chunk()
                
                if audio_data is not None:
                    # 处理音频
                    result = self.process_audio(audio_data)
                    
                    # 创建显示帧
                    frame = np.zeros((600, 800, 3), dtype=np.uint8)
                    
                    # 绘制背景
                    frame = self.visualizer.draw_background(frame)
                    frame = self.visualizer.draw_microphones(frame, self.mic_positions)
                    
                    # 绘制声源
                    if result is not None:
                        frame = self.visualizer.draw_sound_source(
                            frame, result['position'])
                        frame = self.visualizer.draw_info_panel(
                            frame, result['angle'], result['distance'], result['speed'])
                        frame = self.visualizer.draw_angle_indicator(
                            frame, result['angle'])
                    
                    # 显示FPS
                    self.frame_count += 1
                    current_time = time.time()
                    if current_time - self.last_fps_time >= 1.0:
                        self.fps = self.frame_count / (current_time - self.last_fps_time)
                        self.frame_count = 0
                        self.last_fps_time = current_time
                    
                    cv2.putText(frame, f'FPS: {self.fps:.1f}', (10, 590),
                               cv2.FONT_HERSHEY_SIMPLEX, 0.5, (200, 200, 200), 1)
                    
                    # 显示
                    cv2.imshow('Sound Localization', frame)
                
                # 键盘控制
                key = cv2.waitKey(1) & 0xFF
                if key == ord('q'):
                    break
                elif key == ord('c'):
                    self.visualizer.clear_trail()
                elif key == ord('r'):
                    self.tracker = KalmanTracker()
        
        except KeyboardInterrupt:
            print("\n用户中断")
        
        finally:
            self.stop()
            cv2.destroyAllWindows()

def main():
    system = SoundLocalizationSystem()
    system.run()

if __name__ == '__main__':
    main()
```

```python
def calibrate_microphones(audio_capture, localizer):
    """
    麦克风校准
    """
    print("麦克风校准程序")

    known_positions = [
        (1.0, 0.0),   # 右方
        (0.0, 1.0),   # 前方
        (-1.0, 0.0),  # 左方
        (0.0, -1.0),  # 后方
    ]

        print(f"\n请在位置 ({pos[0]}, {pos[1]}) 放置声源")
        input("按Enter继续...")

        chunks = []
        for _ in range(10):
            chunk = audio_capture.get_latest_chunk()
            if chunk is not None:
                chunks.append(chunk)
            time.sleep(0.1)
        
        if chunks:
            # 平均数据
            avg_chunk = [np.mean([c[i] for c in chunks], axis=0) for i in range(4)]
            
            # 计算TDOA
            tdoa = localizer.calculate_tdoa(avg_chunk)
            
            calibration_data.append({
                'position': pos,
                'tdoa': tdoa
            })
    
    # 分析校准数据
    print("\n校准数据分析:")
    for data in calibration_data:
        print(f"位置: {data['position']}")
        print(f"TDOA: {data['tdoa']}")
    
    return calibration_data
```

## 常见问题排查

| 问题 | 可能原因 | 解决方案 |
|------|---------|---------|
| 无法检测声源 | 麦克风未连接 | 检查I2S/USB连接 |
| 定位不准确 | 麦克风位置不准 | 重新测量麦克风位置 |
| 延迟太大 | 计算量过大 | 减小FFT长度 |
| 误检测多 | 噪声干扰 | 增加滤波阈值 |
| 声音失真 | 采样率不匹配 | 检查采样率配置 |

## 总结

本方案使用 Orange Pi 5 实现声音定位系统，主要特点：

1. **多麦克风阵列**：4个MEMS麦克风，I2S接口
2. **TDOA定位算法**：GCC-PHAT互相关 + 球面插值法
3. **实时追踪**：卡尔曼滤波/粒子滤波平滑轨迹
4. **可视化显示**：实时显示声源位置、角度、距离

关键优势：
- Orange Pi 5 的 I2S 接口直接支持多通道音频
- RK3588S 的强大算力保证实时处理
- 丰富的外设接口方便扩展
