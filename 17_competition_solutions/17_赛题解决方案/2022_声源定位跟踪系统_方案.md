## 一、题目分析

### 核心要求
使用麦克风阵列接收声音信号，计算声源方位角和俯仰角，控制云台/转台跟踪声源方向，要求：
- 4路麦克风阵列拾音
- 声源DOA（到达方向）估计
- 实时显示声源方向角（方位角/俯仰角）
- 控制舵机/步进电机跟踪声源
- 跟踪精度：方位角±5°，俯仰角±5°
- 响应时间：<1s

### 技术路线选择
| 方案 | 优点 | 缺点 | 评分 |
|------|------|------|------|
| GCC-PHAT | 精度高，抗噪好 | 计算量大 | ★★★★★ |
| 波束成形 | 可实现声源增强 | 需更多麦克风 | ★★★★ |
| SRP-PHAT | 全空间搜索精度高 | 计算量最大 | ★★★ |
| 相位差法 | 简单快速 | 精度较低 | ★★ |

**选择 GCC-PHAT + 空间网格搜索**：精度高，OPi5算力充足。

## 四、核心算法

### 4.1 GCC-PHAT 声源定位

```python
import numpy as np
from scipy.signal import correlate
from scipy.fft import fft, ifft

class GCCPHATLocalizer:
    """
    GCC-PHAT (广义互相关-相位变换) 声源定位
    基于麦克风阵列的TDOA估计
    """
    
    def __init__(self, mic_positions, sample_rate=48000, speed_of_sound=343.0):
        """
        mic_positions: 麦克风位置坐标 [[x1,y1,z1], [x2,y2,z2], ...]
        sample_rate: 采样率
        """
        self.mic_pos = np.array(mic_positions)
        self.fs = sample_rate
        self.c = speed_of_sound
        self.n_mics = len(mic_positions)
        
        # 预计算麦克风对
        self.mic_pairs = []
        for i in range(self.n_mics):
            for j in range(i+1, self.n_mics):
                self.mic_pairs.append((i, j))
        
        # 预计算空间网格 (单位球面)
        self.azimuth_grid = np.linspace(-180, 180, 361)  # 方位角
        self.elevation_grid = np.linspace(-90, 90, 181)  # 俯仰角
        self.grid_vectors = self._build_grid_vectors()
    
    def _build_grid_vectors(self):
        """构建空间方向向量网格"""
        vectors = []
        for az in self.azimuth_grid:
            for el in self.elevation_grid:
                az_rad = np.radians(az)
                el_rad = np.radians(el)
                # 单位方向向量
                v = np.array([
                    np.cos(el_rad) * np.cos(az_rad),
                    np.cos(el_rad) * np.sin(az_rad),
                    np.sin(el_rad)
                ])
                vectors.append(v)
        return np.array(vectors)
    
    def gcc_phat(self, sig1, sig2, fs=None):
        """
        GCC-PHAT 算法
        返回: 互相关函数
        """
        if fs is None:
            fs = self.fs
        
        n = len(sig1) + len(sig2) - 1
        n_fft = 2 ** int(np.ceil(np.log2(n)))
        
        SIG1 = fft(sig1, n_fft)
        SIG2 = fft(sig2, n_fft)
        
        R = SIG1 * np.conj(SIG2)
        # PHAT加权: 归一化幅度
        magnitude = np.abs(R)
        magnitude[magnitude < 1e-10] = 1e-10
        R_phat = R / magnitude
        
        cc = np.real(ifft(R_phat))
        cc = np.concatenate([cc[-(n_fft//2):], cc[:n_fft//2]])
        
        return cc
    
    def estimate_tdoa(self, signals):
        """
        估计所有麦克风对的TDOA
        signals: (n_mics, n_samples) 音频信号
        返回: tdoa列表
        """
        tdoas = []
        for i, j in self.mic_pairs:
            cc = self.gcc_phat(signals[i], signals[j])
            # 找峰值
            max_idx = np.argmax(np.abs(cc))
            center = len(cc) // 2
            delay_samples = max_idx - center
            delay_time = delay_samples / self.fs
            tdoas.append(delay_time)
        
        return np.array(tdoas)
    
    def estimate_doa_srp(self, signals):
        """
        SRP-PHAT (可控响应功率) 全空间搜索
        精度更高但计算量大
        """
        # 计算所有麦克风对的GCC-PHAT
        cc_dict = {}
        for i, j in self.mic_pairs:
            cc = self.gcc_phat(signals[i], signals[j])
            cc_dict[(i, j)] = cc
        
        # 在空间网格上搜索
        srp = np.zeros(len(self.grid_vectors))
        
        for idx, direction in enumerate(self.grid_vectors):
            power = 0
            for i, j in self.mic_pairs:
                # 理论TDOA
                d_i = np.dot(direction, self.mic_pos[i])
                d_j = np.dot(direction, self.mic_pos[j])
                tau = (d_i - d_j) / self.c
                # 转换为采样点
                tau_samples = int(tau * self.fs)
                
                cc = cc_dict[(i, j)]
                center = len(cc) // 2
                power += cc[center + tau_samples]
            
            srp[idx] = power
        
        # 找最大响应方向
        max_idx = np.argmax(srp)
        best_direction = self.grid_vectors[max_idx]
        
        # 转换为方位角/俯仰角
        azimuth = np.degrees(np.arctan2(best_direction[1], best_direction[0]))
        elevation = np.degrees(np.arcsin(best_direction[2]))
        
        return azimuth, elevation, srp[max_idx]
    
    def estimate_doa_fast(self, signals):
        """
        快速DOA估计 (相位差法)
        适合实时处理
        """
        # 计算所有TDOA
        tdoas = self.estimate_tdoa(signals)
        
        # 最小二乘法求解方向
        A = []
        b = []
        for idx, (i, j) in enumerate(self.mic_pairs):
            diff = self.mic_pos[i] - self.mic_pos[j]
            A.append(diff / self.c)
            b.append(tdoas[idx])
        
        A = np.array(A)
        b = np.array(b)
        
        # 最小二乘解
        direction, _, _, _ = np.linalg.lstsq(A, b, rcond=None)
        
        # 归一化
        norm = np.linalg.norm(direction)
        if norm > 0:
            direction = direction / norm
        
        # 转换为角度
        azimuth = np.degrees(np.arctan2(direction[1], direction[0]))
        elevation = np.degrees(np.arcsin(np.clip(direction[2], -1, 1)))
        
        return azimuth, elevation
```

### 4.2 卡尔曼滤波平滑

```python
class KalmanFilter2D:
    """2D卡尔曼滤波器 (方位角+俯仰角)"""
    
    def __init__(self, dt=0.033, process_noise=1.0, measurement_noise=5.0):
        self.dt = dt
        
        # 状态: [azimuth, elevation, az_velocity, el_velocity]
        self.x = np.zeros(4)
        
        # 状态转移矩阵
        self.F = np.array([
            [1, 0, dt, 0],
            [0, 1, 0, dt],
            [0, 0, 1,  0],
            [0, 0, 0,  1]
        ])
        
        # 观测矩阵
        self.H = np.array([
            [1, 0, 0, 0],
            [0, 1, 0, 0]
        ])
        
        # 过程噪声
        self.Q = np.eye(4) * process_noise
        
        # 观测噪声
        self.R = np.eye(2) * measurement_noise
        
        # 状态协方差
        self.P = np.eye(4) * 100
    
    def predict(self):
        """预测"""
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q
    
    def update(self, measurement):
        """更新"""
        z = np.array(measurement)
        y = z - self.H @ self.x
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        self.x = self.x + K @ y
        self.P = (np.eye(4) - K @ self.H) @ self.P
        
        # 角度归一化
        self.x[0] = self._normalize_angle(self.x[0])
        self.x[1] = np.clip(self.x[1], -90, 90)
        
        return self.x[0], self.x[1]
    
    def _normalize_angle(self, angle):
        while angle > 180:
            angle -= 360
        while angle < -180:
            angle += 360
        return angle
    
    def get_state(self):
        return {
            'azimuth': self.x[0],
            'elevation': self.x[1],
            'az_velocity': self.x[2],
            'el_velocity': self.x[3]
        }
```

### 4.3 声音预处理

```python
import numpy as np
from scipy.signal import butter, lfilter

class AudioPreprocessor:
    """音频信号预处理"""
    
    def __init__(self, sample_rate=48000):
        self.fs = sample_rate
        # 带通滤波器 (300Hz - 4000Hz 人声范围)
        self.b_bp, self.a_bp = butter(4, [300, 4000], btype='band', fs=sample_rate)
    
    def preprocess(self, signals):
        """
        预处理多路音频信号
        signals: (n_mics, n_samples)
        """
        processed = np.zeros_like(signals)
        
        for i in range(signals.shape[0]):
            # 1. 去直流偏移
            sig = signals[i] - np.mean(signals[i])
            
            # 2. 带通滤波
            sig = lfilter(self.b_bp, self.a_bp, sig)
            
            # 3. 归一化
            max_val = np.max(np.abs(sig))
            if max_val > 0:
                sig = sig / max_val
            
            processed[i] = sig
        
        return processed
    
    def detect_voice_activity(self, signal, threshold=0.02):
        """简单的VAD (语音活动检测)"""
        energy = np.sqrt(np.mean(signal**2))
        return energy > threshold
    
    def compute_stft(self, signal, n_fft=1024, hop_length=256):
        """短时傅里叶变换"""
        from scipy.signal import stft
        f, t, Zxx = stft(signal, self.fs, nperseg=n_fft, noverlap=n_fft-hop_length)
        return f, t, Zxx
```

        GPIO.setup(self.TILT_PIN, GPIO.OUT)
        self.pan_pwm = GPIO.PWM(self.PAN_PIN, 50)

        self.pan_pwm.start(7.5)   # 90度初始
        self.tilt_pwm.start(7.5)
    
    def angle_to_duty(self, angle):
        """角度转PWM占空比 (0-180度)"""
        return 2.5 + (angle / 180.0) * 10.0

        # 限制范围
        pan_angle = np.clip(pan_angle + 90, 0, 180)   # -90~90 → 0~180
        tilt_angle = np.clip(tilt_angle + 90, 0, 180)
        
        self.pan_pwm.ChangeDutyCycle(self.angle_to_duty(pan_angle))
        self.tilt_pwm.ChangeDutyCycle(self.angle_to_duty(tilt_angle))

        audio_data = np.frombuffer(in_data, dtype=np.int16)
        # 分离4路信号
        signals = audio_data.reshape(-1, self.CHANNELS).T.astype(np.float32)
        self.audio_buffer.append(signals)
        return (in_data, pyaudio.paContinue)

            input=True,
            input_device_index=device_index,
            frames_per_buffer=self.CHUNK,
            stream_callback=self.audio_callback
        )
        self.stream.start_stream()
    
    def _setup_dual_stereo(self):
        """双立体声声卡方案"""
        # 使用2个USB声卡，每个2通道
        # 需要软件层面同步
        pass
    
    def process_audio(self):
        """处理音频信号，估计声源方向"""
        if len(self.audio_buffer) < 2:
            return
        
        # 合并缓冲区

        self.running = True

### 6.1 GCC-PHAT 改进

```python
def gcc_phat_improved(self, sig1, sig2, fs=None):
    """
    改进的GCC-PHAT
    1. 加窗减少频谱泄漏
    2. 平滑处理减少噪声影响
    """
    # 加汉明窗
    window = np.hamming(len(sig1))
    sig1 = sig1 * window
    sig2 = sig2 * window
    
    # GCC-PHAT
    n = len(sig1) + len(sig2) - 1
    n_fft = 2 ** int(np.ceil(np.log2(n)))
    
    SIG1 = fft(sig1, n_fft)
    SIG2 = fft(sig2, n_fft)
    
    R = SIG1 * np.conj(SIG2)
    
    # 改进的加权函数
    magnitude = np.abs(R)
    magnitude[magnitude < 1e-10] = 1e-10
    
    # SCOT加权 (平滑相干变换)
    P1 = np.abs(SIG1)**2
    P2 = np.abs(SIG2)**2
    weight = 1.0 / np.sqrt(P1 * P2 + 1e-10)
    
    R_weighted = R * weight
    
    cc = np.real(ifft(R_weighted))
    cc = np.concatenate([cc[-(n_fft//2):], cc[:n_fft//2]])
    
    # 二次插值提高精度
    max_idx = np.argmax(np.abs(cc))
    if 0 < max_idx < len(cc) - 1:
        alpha = np.abs(cc[max_idx - 1])
        beta = np.abs(cc[max_idx])
        gamma = np.abs(cc[max_idx + 1])
        if 2*beta - alpha - gamma != 0:
            correction = 0.5 * (alpha - gamma) / (alpha - 2*beta + gamma)
            max_idx = max_idx + correction
    
    return cc
```
