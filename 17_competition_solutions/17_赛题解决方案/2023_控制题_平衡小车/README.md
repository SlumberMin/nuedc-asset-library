## 一、题目分析

### 1.1 题目要求概述

设计运动目标控制与自动追踪系统，要求：
- **运动目标控制**：控制激光笔/光点在指定轨迹上运动
- **自动追踪**：摄像头识别运动目标，控制舵机追踪
- **轨迹绘制**：在白板/屏幕上绘制指定图形
- **速度控制**：调节追踪速度和精度

### 1.2 核心技术指标

| 指标 | 基本要求 | 发挥部分 |
|------|---------|---------|
| 光点定位精度 | ±5mm | ±2mm |
| 追踪响应时间 | <500ms | <200ms |
| 轨迹跟踪精度 | ±10mm | ±5mm |
| 追踪速度 | 0.5m/s | 1.0m/s |
| 支持图形 | 圆形、方形 | 任意曲线 |
| 追踪范围 | 60°×60° | 90°×90° |

### 1.3 难点分析

1. **视觉定位精度**：摄像头分辨率与距离的关系
2. **双轴舵机协调**：X/Y轴同步控制
3. **目标识别稳定性**：光照变化、遮挡处理
4. **轨迹平滑性**：避免运动抖动

## 五、核心算法说明

### 5.1 视觉目标定位算法（OpenMV）

```python
# OpenMV视觉识别代码
import sensor, image, time
from pyb import UART

# 颜色阈值 (LAB色彩空间)
THRESHOLD = (30, 70, 15, 60, 15, 60)  # 红色目标

sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)  # 320x240
sensor.skip_frames(time=2000)

uart = UART(3, 115200)

while True:
    img = sensor.snapshot()
    blobs = img.find_blobs([THRESHOLD], pixels_threshold=100, area_threshold=100)
    
    if blobs:
        largest = max(blobs, key=lambda b: b.pixels())
        # 计算目标中心相对于图像中心的偏移
        error_x = largest.cx() - 160  # 图像中心x=160
        error_y = largest.cy() - 120  # 图像中心y=120
        
        # 发送给STM32
        uart.write("$%d,%d\n" % (error_x, error_y))
        img.draw_cross(largest.cx(), largest.cy(), color=(0, 255, 0))
```

### 5.2 双轴PID追踪算法

> ⚠️ 以下代码基于MSPM0G3507平台，使用实际驱动API，可直接编译。
> 需要链接: `servo.h`, `advanced_pid.h`, PCA9685(双舵机扩展)

```c
/* ═══ 公共宏定义 ═══ */
#include <math.h>
#include "servo.h"         /* Servo_SetAngle — 单舵机PA8 */
#include "advanced_pid.h"  /* PID_Init, PID_Calc */
#include "pca9685.h"       /* PCA9685双舵机扩展(X/Y轴) */

#define PI    3.14159265f
#define CLAMP(x, lo, hi) ((x) < (lo) ? (lo) : ((x) > (hi) ? (hi) : (x)))
```

```c
/* 修复C1: pid_calculate → PID_Calc (使用实际驱动API) */
/* 修复C1: servo_set_angle → Servo_SetAngle / PCA9685_SetAngle */
/* 修复: constrain → CLAMP */
/* 修复: pixels_per_degree=0 除零风险 */

static PID_Controller pid_x, pid_y;
static int pids_inited = 0;
static float current_angle_x = 0.0f, current_angle_y = 0.0f;
static float pixels_per_degree = 5.0f;  /* 标定值：每度对应的像素数 */

/* 舵机角度映射 — 修复: 除零保护 */
float error_to_angle(float error_pixels) {
    if (fabsf(pixels_per_degree) < 0.1f) return 0.0f;  /* 除零保护 */
    return error_pixels / pixels_per_degree;
}

void track_target_init(void) {
    PID_Param px = { .kp = 0.3f, .ki = 0.01f, .kd = 0.15f,
                     .output_min = -45.0f, .output_max = 45.0f,
                     .integral_max = 20.0f, .dead_zone = 2.0f };
    PID_Param py = { .kp = 0.3f, .ki = 0.01f, .kd = 0.15f,
                     .output_min = -30.0f, .output_max = 30.0f,
                     .integral_max = 20.0f, .dead_zone = 2.0f };
    PID_Init(&pid_x, &px);
    PID_Init(&pid_y, &py);
    pids_inited = 1;
}

/* 追踪控制主函数 */
void track_target(int target_x, int target_y) {
    if (!pids_inited) track_target_init();

    /* 使用实际API: PID_Calc(pid, ref, feedback) */
    float angle_x = PID_Calc(&pid_x, 0.0f, (float)target_x);
    float angle_y = PID_Calc(&pid_y, 0.0f, (float)target_y);

    /* 限幅保护 */
    angle_x = CLAMP(angle_x, -45.0f, 45.0f);
    angle_y = CLAMP(angle_y, -30.0f, 30.0f);

    /* 更新舵机角度 */
    current_angle_x += angle_x;
    current_angle_y += angle_y;

    /* 修复F4: 双舵机需PCA9685扩展，单舵机仅PA8 */
#ifdef USE_PCA9685
    PCA9685_SetAngle(0, current_angle_x + 90);  /* PCA9685通道0 → X轴 */
    PCA9685_SetAngle(1, current_angle_y + 90);  /* PCA9685通道1 → Y轴 */
#else
    Servo_SetAngle((uint8_t)(current_angle_x + 90));  /* 仅单舵机PA8 */
#endif
}
```

### 5.3 轨迹生成算法

```c
/* 修复#41: cos → cosf, sin → sinf */

static float target_x = 0.0f, target_y = 0.0f;

/* 圆形轨迹生成 */
void generate_circle(float cx, float cy, float radius, float angle) {
    target_x = cx + radius * cosf(angle);
    target_y = cy + radius * sinf(angle);
}

/* 方形轨迹生成 */
void generate_square(float cx, float cy, float size, float t) {
    if (fabsf(size) < 1e-6f) return;  /* 除零保护 */
    float side = size / 4.0f;
    float total = 4.0f * side;
    if (fabsf(total) < 1e-6f) return;
    float phase = fmodf(t, total);
    
    if (phase < side) {          // 上边
        target_x = cx - size/2.0f + phase;
        target_y = cy - size/2.0f;
    } else if (phase < 2*side) { // 右边
        target_x = cx + size/2.0f;
        target_y = cy - size/2.0f + (phase - side);
    } else if (phase < 3*side) { // 下边
        target_x = cx + size/2.0f - (phase - 2*side);
        target_y = cy + size/2.0f;
    } else {                     // 左边
        target_x = cx - size/2.0f;
        target_y = cy + size/2.0f - (phase - 3*side);
    }
}
```

## 八、调试经验总结

### 8.1 标定流程

1. **摄像头标定**：棋盘格标定消除畸变
2. **像素-角度映射**：多点标定建立映射表
3. **舵机校准**：测试各角度对应PWM值
4. **系统标定**：整体闭环测试修正

### 8.2 PID调参建议

```
X轴和Y轴分开调参：
1. 先调X轴（水平），再调Y轴（垂直）
2. 先P后D，P从小到大直到开始振荡
3. D用于抑制振荡，I用于消除稳态误差
4. 动态和静态参数可能不同，分别调参
```

### 8.3 性能优化技巧

1. **降低通信延迟**：使用高速串口（115200+）
2. **图像处理优化**：只处理ROI区域
3. **预测补偿**：卡尔曼滤波预测目标位置
4. **前馈控制**：根据目标速度前馈补偿
