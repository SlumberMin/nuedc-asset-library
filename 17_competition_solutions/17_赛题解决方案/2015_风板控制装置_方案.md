# 2015年电赛 - 风板控制装置 - Orange Pi 5解决方案

## 题目要求

设计制作风板控制装置，实现：
1. 通过风扇控制风板角度
2. 风板稳定在指定角度
3. 快速响应，小超调
4. 抗外部扰动（如吹气）

## 系统架构

```
┌──────────────┐    CSI/USB     ┌──────────────┐    PWM      ┌──────────────┐
│  摄像头侧视   │──────────────→│  Orange Pi 5 │────────────→│  风扇驱动    │
│  检测风板角度  │              │  RK3588S     │   UART→STM32│  (L298N)     │
└──────────────┘              │  视觉+PID    │             └──────┬───────┘
                               └──────────────┘              ┌────┴────┐
                                                              │ 直流风扇 │
                                                              │ 风板     │
                                                              └─────────┘
```

## 核心设计

### 1. 视觉检测风板角度

```python
import cv2
import numpy as np
import time
import struct
import serial
import math

class WindBoardVision:
    """风板角度视觉检测"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 90)  # 高帧率
        
        # 风板枢轴点（标定）
        self.pivot_point = (320, 350)
        
        # 风板标记点颜色（红色标记在风板边缘）
        self.hsv_lower = np.array([0, 120, 100])
        self.hsv_upper = np.array([10, 255, 255])
        self.hsv_lower2 = np.array([170, 120, 100])
        self.hsv_upper2 = np.array([180, 255, 255])
        
        # 角度历史
        self.angle_history = []
        self.time_history = []
        self.max_history = 20
        
        # 角度偏移（标定）
        self.angle_offset = 0
        
    def detect_board_angle(self):
        """
        检测风板角度
        风板竖直时为0度，向右倾斜为正，向左为负
        返回: 角度(度), 角速度(度/秒)
        """
        ret, frame = self.cap.read()
        if not ret:
            return None, None
        
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 检测风板上的红色标记
        mask1 = cv2.inRange(hsv, self.hsv_lower, self.hsv_upper)
        mask2 = cv2.inRange(hsv, self.hsv_lower2, self.hsv_upper2)
        mask = cv2.bitwise_or(mask1, mask2)
        
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            M = cv2.moments(largest)
            if M["m00"] > 100:
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                
                # 计算角度
                dx = cx - self.pivot_point[0]
                dy = self.pivot_point[1] - cy  # 注意Y轴方向
                angle = math.degrees(math.atan2(dx, dy))
                angle -= self.angle_offset  # 补偿偏移
                
                # 记录历史
                now = time.time()
                self.angle_history.append(angle)
                self.time_history.append(now)
                if len(self.angle_history) > self.max_history:
                    self.angle_history.pop(0)
                    self.time_history.pop(0)
                
                # 估计角速度
                angular_velocity = self._estimate_velocity()
                
                # 可视化
                cv2.circle(frame, (cx, cy), 8, (0, 255, 0), -1)
                cv2.circle(frame, self.pivot_point, 5, (255, 0, 0), -1)
                cv2.line(frame, self.pivot_point, (cx, cy), (0, 255, 0), 2)
                
                # 绘制竖直参考线
                cv2.line(frame, self.pivot_point, 
                        (self.pivot_point[0], self.pivot_point[1]-100), 
                        (128, 128, 128), 1)
                
                cv2.putText(frame, f"Angle: {angle:.1f} deg", (10, 30),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)
                cv2.putText(frame, f"Vel: {angular_velocity:.1f} deg/s", (10, 60),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 0), 2)
                
                cv2.imshow('Wind Board', frame)
                
                return angle, angular_velocity
        
        return None, None
    
    def _estimate_velocity(self):
        """估计角速度"""
        if len(self.angle_history) < 2:
            return 0.0
        dt = self.time_history[-1] - self.time_history[-2]
        if dt < 1e-6:
            return 0.0
        
        if len(self.angle_history) >= 3:
            # 二阶差分，更平滑
            d_angle = (3*self.angle_history[-1] - 4*self.angle_history[-2] + self.angle_history[-3])
            d_time = self.time_history[-1] - self.time_history[-3]
            return d_angle / d_time if d_time > 1e-6 else 0.0
        else:
            return (self.angle_history[-1] - self.angle_history[-2]) / dt
    
    def calibrate_zero(self):
        """标定零位（风板竖直时调用）"""
        angles = []
        for _ in range(30):
            angle, _ = self.detect_board_angle()
            if angle is not None:
                angles.append(angle)
            time.sleep(0.03)
        
        if angles:
            self.angle_offset = np.mean(angles)
            print(f"零位标定完成，偏移: {self.angle_offset:.1f}度")
    
    def release(self):
        self.cap.release()
```

### 2. PID控制器

```python
class WindBoardPID:
    """
    风板PID控制器
    带微分滤波和积分抗饱和
    """
    
    def __init__(self, Kp=3.0, Ki=0.5, Kd=1.0):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None
        
        # 微分低通滤波
        self.d_filter_alpha = 0.3
        self.d_filtered = 0
        
        # 输出限幅
        self.output_max = 999
        self.output_min = 0
        
        # 积分限幅
        self.integral_max = 500
        
        # 死区
        self.deadband = 1.0  # 度
        
    def compute(self, target_angle, current_angle, current_velocity=0):
        """
        PID计算
        
        target_angle: 目标角度(度)
        current_angle: 当前角度(度)
        current_velocity: 当前角速度(度/秒)，用于前馈
        
        返回: PWM值(0~999)
        """
        error = target_angle - current_angle
        
        # 死区处理
        if abs(error) < self.deadband:
            error = 0
        
        now = time.time()
        if self.prev_time is None:
            self.prev_time = now
            self.prev_error = error
            return 0
        
        dt = max(now - self.prev_time, 0.001)
        
        # P项
        p_term = self.Kp * error
        
        # I项（带抗积分饱和）
        self.integral += error * dt
        self.integral = np.clip(self.integral, -self.integral_max, self.integral_max)
        
        # 当误差为零时清除积分
        if abs(error) < 0.5:
            self.integral *= 0.9
        
        i_term = self.Ki * self.integral
        
        # D项（带低通滤波）
        d_raw = (error - self.prev_error) / dt
        self.d_filtered = self.d_filter_alpha * d_raw + (1 - self.d_filter_alpha) * self.d_filtered
        d_term = self.Kd * self.d_filtered
        
        # 速度前馈
        feedforward = -current_velocity * 0.1
        
        # 总输出
        output = p_term + i_term + d_term + feedforward
        
        # 风扇只能吹一个方向，所以需要判断
        # 正角度：风扇从一侧吹；负角度：风扇从另一侧吹
        # 简化：单风扇方案，PWM始终为正，通过风板两侧设计实现双向控制
        pwm = abs(output)
        pwm = np.clip(pwm, self.output_min, self.output_max)
        
        # 方向
        direction = 1 if output >= 0 else -1
        
        self.prev_error = error
        self.prev_time = now
        
        return int(pwm), direction
    
    def reset(self):
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None
        self.d_filtered = 0


class WindBoardController:
    """风板控制主系统"""
    
    STATE_IDLE = 0
    STATE_ANGLE_CONTROL = 1
    STATE_SWING = 2
    
    def __init__(self):
        self.vision = WindBoardVision(camera_id=0)
        self.pid = WindBoardPID(Kp=3.0, Ki=0.3, Kd=1.5)
        self.ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
        
        self.state = self.STATE_IDLE
        self.target_angle = 0
        self.swing_params = None
        
    def send_fan_pwm(self, pwm, direction=1):
        """发送风扇PWM到STM32"""
        data = struct.pack('<Hb', int(pwm), direction)
        self.ser.write(b'\xAA' + data + b'\x55')
    
    def calibrate(self):
        """零位标定"""
        print("请确保风板处于竖直位置，按回车开始标定...")
        input()
        self.vision.calibrate_zero()
    
    def set_target_angle(self, angle):
        """设置目标角度"""
        self.target_angle = angle
        self.state = self.STATE_ANGLE_CONTROL
        self.pid.reset()
        print(f"目标角度: {angle}度")
    
    def set_swing_mode(self, min_angle=-30, max_angle=30, period=3):
        """设置摆动模式"""
        self.swing_params = {
            'min': min_angle,
            'max': max_angle,
            'period': period
        }
        self.state = self.STATE_SWING
        print(f"摆动模式: {min_angle}° ~ {max_angle}°, 周期{period}s")
    
    def run(self):
        """主控制循环"""
        print("风板控制系统启动")
        print("命令: c-标定, a<角度>-定角, s-摆动, q-退出")
        
        self.calibrate()
        
        t_start = time.time()
        
        try:
            while True:
                # 读取用户命令（非阻塞）
                import select
                import sys
                if select.select([sys.stdin], [], [], 0)[0]:
                    cmd = input().strip()
                    if cmd == 'c':
                        self.calibrate()
                    elif cmd.startswith('a'):
                        try:
                            angle = float(cmd[1:])
                            self.set_target_angle(angle)
                        except:
                            print("格式: a<角度>，如 a30")
                    elif cmd == 's':
                        self.set_swing_mode()
                    elif cmd == 'q':
                        break
                
                # 视觉检测
                angle, velocity = self.vision.detect_board_angle()
                
                if angle is not None:
                    if self.state == self.STATE_ANGLE_CONTROL:
                        # 定角控制
                        pwm, direction = self.pid.compute(self.target_angle, angle, velocity)
                        self.send_fan_pwm(pwm, direction)
                        
                        print(f"角度: {angle:.1f}° 目标: {self.target_angle:.1f}° PWM: {pwm}")
                    
                    elif self.state == self.STATE_SWING:
                        # 摆动模式
                        t = time.time() - t_start
                        params = self.swing_params
                        mid = (params['min'] + params['max']) / 2
                        amp = (params['max'] - params['min']) / 2
                        target = mid + amp * math.sin(2 * math.pi * t / params['period'])
                        
                        pwm, direction = self.pid.compute(target, angle, velocity)
                        self.send_fan_pwm(pwm, direction)
                        
                        print(f"角度: {angle:.1f}° 目标: {target:.1f}° PWM: {pwm}")
                    
                    elif self.state == self.STATE_IDLE:
                        self.send_fan_pwm(0, 0)
                
                time.sleep(0.01)  # 100Hz
                
        except KeyboardInterrupt:
            pass
        finally:
            self.send_fan_pwm(0, 0)
            self.vision.release()
            self.ser.close()
            print("系统已停止")


def main():
    controller = WindBoardController()
    controller.run()
```

### 3. STM32端风扇驱动

```c
// 风扇PWM驱动
// 接收Orange Pi 5的PWM值和方向

void Fan_SetPWM(uint16_t pwm, int8_t direction) {
    if (direction >= 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_RESET);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_SET);
    }
    __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, pwm);
}
```

## 接线说明

| Orange Pi 5引脚 | 连接目标 | 说明 |
|-----------------|---------|------|
| CSI/USB | 摄像头 | 侧视检测风板角度 |
| UART3_TX | STM32 RX | 风扇PWM控制指令 |
| 5V | STM32 | 供电 |
| GND | 公共地 | - |

## 调试要点

1. **零位标定**：每次启动前需标定风板竖直位置
2. **PID参数**：Kp不宜过大（会振荡），D项起到阻尼作用
3. **摄像头角度**：侧视安装，确保能看到风板全摆幅
4. **帧率要求**：≥60fps才能保证控制实时性
5. **微分滤波**：避免高频噪声导致风扇抖动
