## 一、题目分析

### 1.1 核心要求
- 设计基于自由摆的平板控制系统
- 摆杆自由摆动时，平板能保持水平
- 平板上放置硬币，摆动时硬币不滑落
- 系统能实时检测摆角并调整平板姿态

### 1.2 技术难点
- 摆角的快速准确检测
- 平板姿态的实时调整
- 抗干扰能力（风、振动等）
- 系统响应速度与稳定性平衡

### 4.2 MPU6050驱动模块

```python
#!/usr/bin/env python3
"""
MPU6050驱动模块 - 读取加速度和陀螺仪数据
"""

import smbus2
import time
import numpy as np

class MPU6050:
    # MPU6050寄存器地址
    PWR_MGMT_1 = 0x6B
    SMPLRT_DIV = 0x19
    CONFIG = 0x1A
    GYRO_CONFIG = 0x1B
    ACCEL_CONFIG = 0x1C
    ACCEL_XOUT_H = 0x3B
    ACCEL_YOUT_H = 0x3D
    ACCEL_ZOUT_H = 0x3F
    GYRO_XOUT_H = 0x43
    GYRO_YOUT_H = 0x45
    GYRO_ZOUT_H = 0x47
    
    def __init__(self, bus_id=1, address=0x68):
        self.bus = smbus2.SMBus(bus_id)
        self.address = address
        
        # 初始化MPU6050
        self.bus.write_byte_data(self.address, self.PWR_MGMT_1, 0x00)  # 唤醒
        time.sleep(0.1)
        
        # 设置采样率 1kHz
        self.bus.write_byte_data(self.address, self.SMPLRT_DIV, 0x07)
        
        # 设置低通滤波 44Hz
        self.bus.write_byte_data(self.address, self.CONFIG, 0x03)
        
        # 设置陀螺仪量程 ±250°/s
        self.bus.write_byte_data(self.address, self.GYRO_CONFIG, 0x00)
        
        # 设置加速度计量程 ±2g
        self.bus.write_byte_data(self.address, self.ACCEL_CONFIG, 0x00)
        
        # 校准偏移量
        self.accel_offset = [0, 0, 0]
        self.gyro_offset = [0, 0, 0]
        self.calibrate()
        
    def read_raw_data(self, addr):
        """读取原始数据"""
        high = self.bus.read_byte_data(self.address, addr)
        low = self.bus.read_byte_data(self.address, addr + 1)
        
        value = (high << 8) + low
        if value > 32768:
            value = value - 65536
        return value
        
    def read_accel(self):
        """读取加速度值（单位：g）"""
        x = self.read_raw_data(self.ACCEL_XOUT_H) / 16384.0
        y = self.read_raw_data(self.ACCEL_YOUT_H) / 16384.0
        z = self.read_raw_data(self.ACCEL_ZOUT_H) / 16384.0
        
        # 应用校准偏移
        x -= self.accel_offset[0]
        y -= self.accel_offset[1]
        z -= self.accel_offset[2]
        
        return np.array([x, y, z])
        
    def read_gyro(self):
        """读取陀螺仪值（单位：°/s）"""
        x = self.read_raw_data(self.GYRO_XOUT_H) / 131.0
        y = self.read_raw_data(self.GYRO_YOUT_H) / 131.0
        z = self.read_raw_data(self.GYRO_ZOUT_H) / 131.0
        
        # 应用校准偏移
        x -= self.gyro_offset[0]
        y -= self.gyro_offset[1]
        z -= self.gyro_offset[2]
        
        return np.array([x, y, z])
        
    def calibrate(self, samples=1000):
        """校准传感器"""
        print("校准MPU6050，请保持静止...")
        
        accel_sum = np.zeros(3)
        gyro_sum = np.zeros(3)
        
        for _ in range(samples):
            accel_sum += self.read_raw_data_vec(self.ACCEL_XOUT_H)
            gyro_sum += self.read_raw_data_vec(self.GYRO_XOUT_H)
            time.sleep(0.001)
            
        # 计算平均偏移
        accel_avg = accel_sum / samples / 16384.0
        gyro_avg = gyro_sum / samples / 131.0
        
        # 加速度Z轴应该为1g
        self.accel_offset = [accel_avg[0], accel_avg[1], accel_avg[2] - 1.0]
        self.gyro_offset = gyro_avg
        
        print(f"校准完成: accel_offset={self.accel_offset}, gyro_offset={self.gyro_offset}")
        
    def read_raw_data_vec(self, base_addr):
        """读取三轴原始数据向量"""
        data = []
        for i in range(3):
            high = self.bus.read_byte_data(self.address, base_addr + i * 2)
            low = self.bus.read_byte_data(self.address, base_addr + i * 2 + 1)
            value = (high << 8) + low
            if value > 32768:
                value = value - 65536
            data.append(value)
        return np.array(data)
        
    def get_angle(self):
        """
        计算倾斜角（使用互补滤波）
        返回: (roll, pitch) 角度（度）
        """
        accel = self.read_accel()
        gyro = self.read_gyro()
        
        # 从加速度计计算角度
        accel_roll = np.arctan2(accel[1], accel[2]) * 180 / np.pi
        accel_pitch = np.arctan2(-accel[0], np.sqrt(accel[1]**2 + accel[2]**2)) * 180 / np.pi
        
        # 互补滤波融合
        alpha = 0.98
        dt = 0.01  # 10ms采样周期
        
        if not hasattr(self, 'roll'):
            self.roll = accel_roll
            self.pitch = accel_pitch
        else:
            self.roll = alpha * (self.roll + gyro[0] * dt) + (1 - alpha) * accel_roll
            self.pitch = alpha * (self.pitch + gyro[1] * dt) + (1 - alpha) * accel_pitch
            
        return self.roll, self.pitch
```

### 4.3 视觉角度检测模块

```python
#!/usr/bin/env python3
"""
视觉角度检测模块 - 使用摄像头检测摆杆角度
"""

import cv2
import numpy as np

class VisionAngleDetector:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 60)
        
        # ArUco标记检测
        self.aruco_dict = cv2.aruco.Dictionary_get(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters_create()
        
        # 颜色检测参数（用于检测摆杆上的彩色标记）
        self.color_ranges = {
            'red': ([0, 100, 100], [10, 255, 255]),
            'blue': ([100, 100, 100], [130, 255, 255])
        }
        
    def detect_angle_aruco(self, marker_id=0):
        """
        使用ArUco标记检测角度
        返回: 角度（度）或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 检测标记
        corners, ids, rejected = cv2.aruco.detectMarkers(
            frame, self.aruco_dict, parameters=self.aruco_params
        )
        
        if ids is not None and marker_id in ids:
            idx = np.where(ids == marker_id)[0][0]
            marker_corners = corners[idx][0]
            
            # 计算标记中心
            center = np.mean(marker_corners, axis=0)
            
            # 计算标记朝向
            # 假设标记的前两个点是"上"方向
            top_center = np.mean(marker_corners[:2], axis=0)
            
            # 计算角度（相对于垂直方向）
            dx = top_center[0] - center[0]
            dy = top_center[1] - center[1]
            angle = np.arctan2(dx, -dy) * 180 / np.pi
            
            return angle
            
        return None
        
    def detect_angle_color(self, color='red'):
        """
        使用颜色标记检测角度
        返回: 角度（度）或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 转换到HSV颜色空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 颜色过滤
        lower, upper = self.color_ranges[color]
        mask = cv2.inRange(hsv, np.array(lower), np.array(upper))
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.erode(mask, kernel, iterations=2)
        mask = cv2.dilate(mask, kernel, iterations=2)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if len(contours) >= 2:
            # 找到两个最大的轮廓（两个标记点）
            contours = sorted(contours, key=cv2.contourArea, reverse=True)[:2]
            
            # 计算中心点
            centers = []
            for contour in contours:
                M = cv2.moments(contour)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    centers.append((cx, cy))
                    
            if len(centers) == 2:
                # 计算两点连线的角度
                dx = centers[1][0] - centers[0][0]
                dy = centers[1][1] - centers[0][1]
                angle = np.arctan2(dx, -dy) * 180 / np.pi
                return angle
                
        return None
        
    def detect_angle_line(self):
        """
        使用直线检测摆杆角度
        返回: 角度（度）或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 灰度化
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 边缘检测
        edges = cv2.Canny(gray, 50, 150)
        
        # 霍夫直线检测
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50, minLineLength=100, maxLineGap=10)
        
        if lines is not None:
            # 找到最长的直线
            max_length = 0
            best_line = None
            
            for line in lines:
                x1, y1, x2, y2 = line[0]
                length = np.sqrt((x2-x1)**2 + (y2-y1)**2)
                if length > max_length:
                    max_length = length
                    best_line = line[0]
                    
            if best_line is not None:
                x1, y1, x2, y2 = best_line
                angle = np.arctan2(x2-x1, -(y2-y1)) * 180 / np.pi
                return angle
                
        return None
        
    def get_frame(self):
        """获取当前帧"""
        ret, frame = self.cap.read()
        return frame if ret else None
        
    def release(self):
        """释放资源"""
        self.cap.release()
```

### 4.4 卡尔曼滤波器

```python
#!/usr/bin/env python3
"""
卡尔曼滤波器 - 用于角度估计
"""

import numpy as np

class KalmanFilter:
    def __init__(self, dt=0.01, process_noise=0.01, measurement_noise=0.1):
        """
        dt: 采样时间
        process_noise: 过程噪声
        measurement_noise: 测量噪声
        """
        self.dt = dt
        
        # 状态向量 [角度, 角速度]
        self.x = np.array([[0], [0]])
        
        # 状态转移矩阵
        self.F = np.array([[1, dt],
                           [0, 1]])
        
        # 测量矩阵
        self.H = np.array([[1, 0]])
        
        # 过程噪声协方差
        self.Q = np.array([[dt**4/4, dt**3/2],
                           [dt**3/2, dt**2]]) * process_noise
        
        # 测量噪声协方差
        self.R = np.array([[measurement_noise]])
        
        # 误差协方差矩阵
        self.P = np.eye(2)
        
    def predict(self):
        """预测步骤"""
        # 状态预测
        self.x = self.F @ self.x
        
        # 协方差预测
        self.P = self.F @ self.P @ self.F.T + self.Q
        
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def update(self, measurement):
        """更新步骤"""
        # 卡尔曼增益
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        # 状态更新
        y = measurement - self.H @ self.x  # 测量残差
        self.x = self.x + K @ y
        
        # 协方差更新
        I = np.eye(2)
        self.P = (I - K @ self.H) @ self.P
        
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def get_state(self):
        """获取当前状态"""
        return self.x[0, 0], self.x[1, 0]  # 角度, 角速度
        
    def reset(self):
        """重置滤波器"""
        self.x = np.array([[0], [0]])
        self.P = np.eye(2)
```

### 4.5 平板控制算法

```python
#!/usr/bin/env python3
"""
平板控制算法 - 保持平板水平
"""

import numpy as np

class PlateController:
    def __init__(self, pid_params=None):
        """
        pid_params: PID参数字典 {'kp': ..., 'ki': ..., 'kd': ...}
        """
        if pid_params is None:
            self.pid_params = {
                'kp': 2.0,
                'ki': 0.1,
                'kd': 0.5
            }
        else:
            self.pid_params = pid_params
            
        # PID状态变量
        self.error_sum = 0
        self.error_prev = 0
        
        # 目标角度（水平时为0）
        self.target_angle = 0
        
        # 输出限制
        self.output_min = -90  # 舵机最小角度
        self.output_max = 90   # 舵机最大角度
        
    def compute(self, current_angle, current_angular_velocity=None):
        """
        计算控制输出
        current_angle: 当前摆角（度）
        current_angular_velocity: 当前角速度（度/秒），可选
        
        返回: 舵机控制角度（度）
        """
        # 计算误差
        error = self.target_angle - current_angle
        
        # PID计算
        # 比例项
        p_term = self.pid_params['kp'] * error
        
        # 积分项（带抗饱和）
        self.error_sum += error
        self.error_sum = np.clip(self.error_sum, -100, 100)  # 抗饱和
        i_term = self.pid_params['ki'] * self.error_sum
        
        # 微分项
        if current_angular_velocity is not None:
            # 使用角速度作为微分项（更准确）
            d_term = -self.pid_params['kd'] * current_angular_velocity
        else:
            # 使用误差差分
            d_term = self.pid_params['kd'] * (error - self.error_prev)
            
        self.error_prev = error
        
        # 计算输出
        output = p_term + i_term + d_term
        
        # 限制输出范围
        output = np.clip(output, self.output_min, self.output_max)
        
        return output
        
    def reset(self):
        """重置控制器状态"""
        self.error_sum = 0
        self.error_prev = 0
        
    def set_target(self, angle):
        """设置目标角度"""
        self.target_angle = angle
        
    def tune_pid(self, kp=None, ki=None, kd=None):
        """调整PID参数"""
        if kp is not None:
            self.pid_params['kp'] = kp
        if ki is not None:
            self.pid_params['ki'] = ki
        if kd is not None:
            self.pid_params['kd'] = kd
```

        # 传感器初始化
        self.imu = MPU6050(bus_id=1, address=0x68)
        self.vision = VisionAngleDetector(camera_id=0)
        
        # 滤波器和控制器初始化
        self.kalman = KalmanFilter(dt=0.01, process_noise=0.01, measurement_noise=0.1)
        self.controller = PlateController(pid_params={
            'kp': 2.5,
            'ki': 0.15,
            'kd': 0.6
        })
        
        # 状态变量
        self.current_angle = 0
        self.current_angular_velocity = 0
        self.servo_angle = 90  # 初始位置（中间）
        
        # 运行标志
        self.running = False
        
        # 统计信息
        self.stats = {
            'max_angle': 0,
            'control_count': 0,
            'start_time': 0
        }
        
        # 数据记录（用于调试）
        self.data_log = {
            'time': [],
            'angle': [],
            'angular_velocity': [],
            'servo_angle': []
        }
        
    def start(self):
        """启动系统"""
        print("=" * 50)
        print("基于自由摆的平板控制系统启动")
        print("=" * 50)
        
        self.running = True
        self.stats['start_time'] = time.time()
        
        # 设置舵机到初始位置
        self._set_servo_angle(90)
        time.sleep(1)
        
        # 启动控制线程
        self.control_thread = threading.Thread(target=self._control_loop)
        self.control_thread.start()
        
        # 启动视觉线程
        self.vision_thread = threading.Thread(target=self._vision_loop)
        self.vision_thread.start()
        
        print("系统就绪，请开始摆动...")
        
    def _control_loop(self):
        """主控制循环（100Hz）"""
        dt = 0.01  # 10ms
        last_time = time.time()
        
        while self.running:
            current_time = time.time()
            
            # 读取IMU数据
            accel = self.imu.read_accel()
            gyro = self.imu.read_gyro()
            
            # 使用加速度计计算角度
            accel_angle = np.arctan2(-accel[0], np.sqrt(accel[1]**2 + accel[2]**2)) * 180 / np.pi
            
            # 使用陀螺仪获取角速度
            angular_velocity = gyro[1]  # 绕Y轴旋转
            
            # 卡尔曼滤波融合
            self.kalman.predict()
            self.current_angle, self.current_angular_velocity = self.kalman.update(accel_angle)
            
            # 计算控制输出
            control_output = self.controller.compute(
                self.current_angle,
                self.current_angular_velocity
            )
            
            # 转换为舵机角度（90度为中心）
            self.servo_angle = 90 + control_output
            self.servo_angle = np.clip(self.servo_angle, 0, 180)
            
            # 设置舵机
            self._set_servo_angle(self.servo_angle)
            
            # 更新统计
            self.stats['control_count'] += 1
            self.stats['max_angle'] = max(self.stats['max_angle'], abs(self.current_angle))
            
            # 记录数据
            elapsed = current_time - self.stats['start_time']
            self.data_log['time'].append(elapsed)
            self.data_log['angle'].append(self.current_angle)
            self.data_log['angular_velocity'].append(self.current_angular_velocity)
            self.data_log['servo_angle'].append(self.servo_angle)
            
            # 等待下一个周期
            elapsed_loop = time.time() - current_time
            if elapsed_loop < dt:
                time.sleep(dt - elapsed_loop)
                
    def _vision_loop(self):
        """视觉检测循环（30Hz）"""
        while self.running:
            # 尝试视觉检测（用于校准和验证）
            angle = self.vision.detect_angle_aruco(marker_id=0)
            if angle is not None:
                # 可以用于校准IMU或作为参考
                pass
            time.sleep(0.033)  # 30Hz
            
    def _set_servo_angle(self, angle):
        """设置舵机角度"""
        # 角度转换为占空比
        # 0度 -> 2.5%, 180度 -> 12.5%
        duty = 2.5 + angle / 18.0
        self.servo_pwm.ChangeDutyCycle(duty)
        
    def get_status(self):
        """获取系统状态"""
        return {
            'angle': self.current_angle,
            'angular_velocity': self.current_angular_velocity,

            'control_count': self.stats['control_count'],
            'elapsed': time.time() - self.stats['start_time']
        }
        
    def save_log(self, filename='pendulum_log.csv'):

        import csv
        with open(filename, 'w', newline='') as f:
            writer = csv.writer(f)
            writer.writerow(['Time', 'Angle', 'AngularVelocity', 'ServoAngle'])

    def plot_data(self):
        """绘制数据曲线"""
        import matplotlib.pyplot as plt
        
        fig, axes = plt.subplots(3, 1, figsize=(10, 8))

    def stop(self):
        """停止系统"""
        self.running = False
        
        # 等待线程结束
        if hasattr(self, 'control_thread'):
            self.control_thread.join()
        if hasattr(self, 'vision_thread'):

def manual_tune_guide():
    """手动PID参数整定指南"""
    print("""
    PID参数整定步骤:
    
    1. 先设置较小的kp值（如0.5），ki=0, kd=0
    2. 逐渐增大kp，直到系统开始振荡
    3. 记录振荡时的kp值（临界增益Ku）和振荡周期Tu
    4. 使用Ziegler-Nichols公式计算：
       - P控制器: kp = 0.5 * Ku
       - PI控制器: kp = 0.45 * Ku, ki = 1.2 * kp / Tu
       - PID控制器: kp = 0.6 * Ku, ki = 2 * kp / Tu, kd = kp * Tu / 8
    5. 根据实际效果微调
    
    注意事项:
    - 摆杆系统是欠阻尼系统，需要较大的kd来抑制振荡
    - ki不宜过大，否则会导致超调
    - 建议先调kp和kd，最后微调ki
    """)

### 5.2 常见问题及解决

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| 平板不响应 | 舵机连接错误 | 检查PWM接线和电源 |
| 平板振荡 | PID参数不合适 | 减小kp，增大kd |
| 平板偏移 | IMU未校准 | 重新校准IMU |
| 响应延迟 | 控制频率太低 | 提高控制频率到100Hz以上 |
| 硬币滑落 | 平板表面太滑 | 增加防滑垫 |

### 5.3 性能优化

1. **提高控制频率**: 使用硬件PWM，减少软件延迟
2. **优化滤波**: 调整卡尔曼滤波器参数
3. **前馈补偿**: 根据摆杆运动预测平板角度
4. **自适应控制**: 根据摆幅自动调整PID参数
