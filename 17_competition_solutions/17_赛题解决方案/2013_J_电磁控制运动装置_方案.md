# 2013年J题：电磁控制运动装置 - Orange Pi 5解决方案

## 题目概述

设计一个电磁控制运动装置，通过控制电磁铁的通断电，使铁质物体在平面上按照预定轨迹运动。需要精确控制电磁铁的吸力和释放时机，实现物体的定点停留和轨迹运动。

---

```
                    ┌─────────────┐
```

**关键参数：**
- 电磁铁类型：直流螺线管电磁铁
- 驱动电压：12V~24V
- 驱动电流：0.5A~2A（根据负载调整）
- PWM频率：1kHz（减少电磁噪声）

### 1. 电磁铁控制核心代码

```python
#!/usr/bin/env python3
"""
电磁控制运动装置 - Orange Pi 5
"""

import gpiod
import time
import threading
from collections import deque

class ElectromagnetController:
    """单个电磁铁控制器"""
    
    def __init__(self, chip, pin_enable, pin_dir1, pin_dir2):
        self.chip = gpiod.Chip(str(chip))
        self.pin_enable = self.chip.get_line(pin_enable)
        self.pin_dir1 = self.chip.get_line(pin_dir1)
        self.pin_dir2 = self.chip.get_line(pin_dir2)
        
        # 配置为输出
        config = gpiod.LineSettings(direction=gpiod.Direction.OUTPUT, output_value=0)
        self.pin_enable.request(consumer="em_ctrl", config=config)
        self.pin_dir1.request(consumer="em_ctrl", config=config)
        self.pin_dir2.request(consumer="em_ctrl", config=config)
        
        self.power = 0  # 0-100
    
    def set_power(self, power):
        """设置电磁铁功率 (0-100)"""
        self.power = max(0, min(100, power))
        # 通过PWM控制功率
        # 注意：实际使用硬件PWM更精确
        if self.power > 0:
            self.pin_dir1.set_value(1)
            self.pin_dir2.set_value(0)
        else:
            self.pin_dir1.set_value(0)
            self.pin_dir2.set_value(0)
    
    def activate(self, power=100):
        """激活电磁铁"""
        self.set_power(power)
        self.pin_enable.set_value(1)
    
    def deactivate(self):
        """关闭电磁铁"""
        self.pin_enable.set_value(0)
        self.set_power(0)
    
    def demagnetize(self):
        """消磁处理 - 反向短脉冲"""
        self.pin_dir1.set_value(0)
        self.pin_dir2.set_value(1)
        self.pin_enable.set_value(1)
        time.sleep(0.01)  # 10ms反向脉冲
        self.pin_enable.set_value(0)
        self.pin_dir2.set_value(0)


class ElectromagnetArray:
    """电磁铁阵列控制器"""
    
    def __init__(self):
        # 初始化4个电磁铁 (4x3 GPIO per magnet)
        self.magnets = [
            ElectromagnetController(1, 24, 25, 26),   # 电磁铁1 - 左上
            ElectromagnetController(1, 27, 28, 29),    # 电磁铁2 - 右上
            ElectromagnetController(1, 30, 31, 32),    # 电磁铁3 - 左下
            ElectromagnetController(1, 33, 34, 35),    # 电磁铁4 - 右下
        ]
        
        # 工作台尺寸 (mm)
        self.workspace_width = 300
        self.workspace_height = 300
        
        # 电磁铁位置 (相对于工作台中心, mm)
        self.magnet_positions = [
            (-120, 120),   # 左上
            (120, 120),    # 右上
            (-120, -120),  # 左下
            (120, -120),   # 右下
        ]
    
    def calculate_forces(self, target_x, target_y):
        """
        计算各电磁铁所需功率
        使用反比定律估算吸力
        """
        powers = []
        for mx, my in self.magnet_positions:
            # 计算距离
            dx = target_x - mx
            dy = target_y - my
            distance = (dx**2 + dy**2) ** 0.5
            
            # 反比关系 (距离越近，功率越大)
            # 添加最小距离保护
            if distance < 20:
                distance = 20
            
            power = min(100, 5000 / distance)
            powers.append(power)
        
        # 归一化到总功率不超过400
        total = sum(powers)
        if total > 400:
            powers = [p * 400 / total for p in powers]
        
        return powers
    
    def move_to(self, x, y, duration=0.5):
        """
        移动物体到目标位置
        """
        powers = self.calculate_forces(x, y)
        
        # 激活相应电磁铁
        for i, power in enumerate(powers):
            if power > 10:  # 阈值过滤
                self.magnets[i].activate(int(power))
            else:
                self.magnets[i].deactivate()
        
        # 保持指定时间
        time.sleep(duration)
    
    def stop_all(self):
        """停止所有电磁铁"""
        for magnet in self.magnets:
            magnet.deactivate()
            magnet.demagnetize()
    
    def emergency_stop(self):
        """紧急停止"""
        for magnet in self.magnets:
            magnet.deactivate()
```

### 2. 视觉定位系统

```python
#!/usr/bin/env python3
"""
视觉定位系统 - 检测铁质物体位置
"""

import cv2
import numpy as np

class VisionTracker:
    """物体位置追踪器"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        # 工作台参数
        self.workspace_corners = None  # 标定后设置
        self.pixel_to_mm_ratio = 1.0
        
        # 物体检测参数 (铁球颜色/大小)
        self.object_color_lower = np.array([0, 0, 100])    # HSV下限
        self.object_color_upper = np.array([180, 50, 255]) # HSV上限
        
        # 历史位置 (用于滤波)
        self.position_history = deque(maxlen=10)
    
    def calibrate_workspace(self, corners):
        """
        标定工作台
        corners: [(x1,y1), (x2,y2), (x3,y3), (x4,y4)] 四个角点
        """
        self.workspace_corners = np.array(corners, dtype=np.float32)
        
        # 计算像素到毫米的比例
        width_pixels = np.linalg.norm(self.workspace_corners[1] - self.workspace_corners[0])
        height_pixels = np.linalg.norm(self.workspace_corners[3] - self.workspace_corners[0])
        
        self.pixel_to_mm_ratio_x = 300 / width_pixels  # 300mm工作台
        self.pixel_to_mm_ratio_y = 300 / height_pixels
    
    def detect_object(self, frame):
        """
        检测铁质物体位置
        返回: (x_mm, y_mm) 或 None
        """
        # 转换到HSV颜色空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 创建掩码
        mask = cv2.inRange(hsv, self.object_color_lower, self.object_color_upper)
        
        # 形态学操作去除噪声
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            # 找到最大轮廓
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if area > 100:  # 最小面积阈值
                # 计算中心
                M = cv2.moments(largest)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    
                    # 转换为工作台坐标 (mm)
                    x_mm = self.pixel_to_mm_ratio_x * (cx - self.workspace_corners[0][0])
                    y_mm = self.pixel_to_mm_ratio_y * (cy - self.workspace_corners[0][1])
                    
                    # 添加到历史记录
                    self.position_history.append((x_mm, y_mm))
                    
                    # 返回滤波后的位置
                    return self._filter_position()
        
        return None
    
    def _filter_position(self):
        """
        对位置进行滤波
        """
        if not self.position_history:
            return None
        
        # 中值滤波
        positions = list(self.position_history)
        x_vals = [p[0] for p in positions]
        y_vals = [p[1] for p in positions]
        
        x_filtered = np.median(x_vals)
        y_filtered = np.median(y_vals)
        
        return (x_filtered, y_filtered)
    
    def get_frame_with_overlay(self, frame, current_pos=None, target_pos=None):
        """
        获取带有叠加信息的帧
        """
        display = frame.copy()
        
        # 绘制工作台边界
        if self.workspace_corners is not None:
            pts = self.workspace_corners.reshape((-1, 1, 2)).astype(np.int32)
            cv2.polylines(display, [pts], True, (0, 255, 0), 2)
        
        # 绘制当前位置
        if current_pos:
            x, y = current_pos
            # 转换回像素坐标
            px = int(x / self.pixel_to_mm_ratio_x + self.workspace_corners[0][0])
            py = int(y / self.pixel_to_mm_ratio_y + self.workspace_corners[0][1])
            cv2.circle(display, (px, py), 10, (0, 0, 255), -1)
            cv2.putText(display, f"({x:.1f}, {y:.1f})", (px+15, py), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 0, 255), 2)
        
        # 绘制目标位置
        if target_pos:
            tx, ty = target_pos
            tpx = int(tx / self.pixel_to_mm_ratio_x + self.workspace_corners[0][0])
            tpy = int(ty / self.pixel_to_mm_ratio_y + self.workspace_corners[0][1])
            cv2.circle(display, (tpx, tpy), 10, (0, 255, 0), 2)
            cv2.putText(display, f"Target({tx:.1f}, {ty:.1f})", (tpx+15, tpy), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
        
        return display
    
    def release(self):
        """释放资源"""
        self.cap.release()
```

### 3. 运动控制系统

```python
#!/usr/bin/env python3
"""
运动控制 - 轨迹规划与PID控制
"""

import time
import numpy as np
from collections import deque

class MotionController:
    """运动控制器"""
    
    def __init__(self, electromagnet_array, vision_tracker):
        self.em_array = electromagnet_array
        self.vision = vision_tracker
        
        # PID参数
        self.kp = 0.5
        self.ki = 0.01
        self.kd = 0.1
        
        # PID状态
        self.error_sum_x = 0
        self.error_sum_y = 0
        self.last_error_x = 0
        self.last_error_y = 0
        
        # 轨迹点队列
        self.trajectory = deque()
        self.current_target = None
        
        # 控制参数
        self.position_tolerance = 5  # mm
        self.max_speed = 50  # mm/s
        
    def set_pid(self, kp, ki, kd):
        """设置PID参数"""
        self.kp = kp
        self.ki = ki
        self.kd = kd
    
    def add_trajectory_point(self, x, y, hold_time=0):
        """添加轨迹点"""
        self.trajectory.append((x, y, hold_time))
    
    def generate_circle_trajectory(self, center_x, center_y, radius, num_points=36):
        """生成圆形轨迹"""
        self.trajectory.clear()
        for i in range(num_points):
            angle = 2 * np.pi * i / num_points
            x = center_x + radius * np.cos(angle)
            y = center_y + radius * np.sin(angle)
            self.trajectory.append((x, y, 0.05))
    
    def generate_line_trajectory(self, start, end, num_points=20):
        """生成直线轨迹"""
        self.trajectory.clear()
        for i in range(num_points + 1):
            t = i / num_points
            x = start[0] + (end[0] - start[0]) * t
            y = start[1] + (end[1] - start[1]) * t
            self.trajectory.append((x, y, 0.05))
    
    def generate_rectangle_trajectory(self, corners, points_per_side=10):
        """生成矩形轨迹"""
        self.trajectory.clear()
        for i in range(len(corners)):
            start = corners[i]
            end = corners[(i + 1) % len(corners)]
            for j in range(points_per_side):
                t = j / points_per_side
                x = start[0] + (end[0] - start[0]) * t
                y = start[1] + (end[1] - start[1]) * t
                self.trajectory.append((x, y, 0.05))
    
    def pid_control(self, current_pos, target_pos, dt):
        """
        PID控制计算
        返回: (force_x, force_y)
        """
        # 计算误差
        error_x = target_pos[0] - current_pos[0]
        error_y = target_pos[1] - current_pos[1]
        
        # 比例项
        p_x = self.kp * error_x
        p_y = self.kp * error_y
        
        # 积分项
        self.error_sum_x += error_x * dt
        self.error_sum_y += error_y * dt
        
        # 积分限幅
        self.error_sum_x = max(-100, min(100, self.error_sum_x))
        self.error_sum_y = max(-100, min(100, self.error_sum_y))
        
        i_x = self.ki * self.error_sum_x
        i_y = self.ki * self.error_sum_y
        
        # 微分项
        d_x = self.kd * (error_x - self.last_error_x) / dt if dt > 0 else 0
        d_y = self.kd * (error_y - self.last_error_y) / dt if dt > 0 else 0
        
        # 更新历史误差
        self.last_error_x = error_x
        self.last_error_y = error_y
        
        # 计算总输出
        force_x = p_x + i_x + d_x
        force_y = p_y + i_y + d_y
        
        return force_x, force_y
    
    def is_at_target(self, current_pos, target_pos):
        """检查是否到达目标位置"""
        if current_pos is None or target_pos is None:
            return False
        
        dx = target_pos[0] - current_pos[0]
        dy = target_pos[1] - current_pos[1]
        distance = (dx**2 + dy**2) ** 0.5
        
        return distance < self.position_tolerance
    
    def update(self):
        """
        主控制循环的一次更新
        返回: True 如果当前轨迹点完成
        """
        # 获取当前位置
        frame = self.vision.cap.read()[1]
        current_pos = self.vision.detect_object(frame)
        
        if current_pos is None:
            return False
        
        # 获取当前目标
        if self.current_target is None:
            if self.trajectory:
                self.current_target = self.trajectory.popleft()
            else:
                return True  # 轨迹完成
        
        target_x, target_y, hold_time = self.current_target
        
        # 检查是否到达
        if self.is_at_target(current_pos, (target_x, target_y)):
            # 到达目标，保持指定时间
            if hold_time > 0:
                time.sleep(hold_time)
            
            # 切换到下一个目标
            self.current_target = None
            return len(self.trajectory) == 0
        
        # PID控制
        dt = 0.02  # 50Hz控制频率
        force_x, force_y = self.pid_control(current_pos, (target_x, target_y), dt)
        
        # 计算目标位置 (当前位置 + 力方向)
        next_x = current_pos[0] + force_x * dt
        next_y = current_pos[1] + force_y * dt
        
        # 限幅到工作台范围
        next_x = max(-150, min(150, next_x))
        next_y = max(-150, min(150, next_y))
        
        # 控制电磁铁
        self.em_array.move_to(next_x, next_y, dt)
        
        return False
    
    def reset(self):
        """重置控制器"""
        self.error_sum_x = 0
        self.error_sum_y = 0
        self.last_error_x = 0
        self.last_error_y = 0
        self.trajectory.clear()
        self.current_target = None
        self.em_array.stop_all()
```

### 4. 主程序

```python
#!/usr/bin/env python3
"""
主程序 - 电磁控制运动装置
"""

import cv2
import time
from electromagnet_controller import ElectromagnetArray
from vision_tracker import VisionTracker
from motion_controller import MotionController

def main():
    # 初始化系统
    print("初始化电磁控制运动装置...")
    
    em_array = ElectromagnetArray()
    vision = VisionTracker(camera_id=0)
    controller = MotionController(em_array, vision)
    
    # 标定工作台
    print("请标定工作台四角 (点击四个角点)...")
    # 实际实现中需要交互式标定
    # vision.calibrate_workspace([(50,50), (590,50), (590,430), (50,430)])
    
    # 设置PID参数
    controller.set_pid(kp=0.5, ki=0.01, kd=0.1)
    
    # 选择运动模式
    print("\n选择运动模式:")
    print("1. 圆形轨迹")
    print("2. 矩形轨迹")
    print("3. 自定义轨迹")
    
    choice = input("请输入选择 (1-3): ")
    
    if choice == '1':
        controller.generate_circle_trajectory(0, 0, 80)
    elif choice == '2':
        corners = [(-100, -100), (100, -100), (100, 100), (-100, 100)]
        controller.generate_rectangle_trajectory(corners)
    elif choice == '3':
        print("输入轨迹点 (x,y)，输入 'done' 完成:")
        while True:
            point = input("点: ")
            if point.lower() == 'done':
                break
            try:
                x, y = map(float, point.split(','))
                controller.add_trajectory_point(x, y)
            except:
                print("格式错误，请重新输入")
    
    # 开始运动
    print("\n开始运动控制...")
    print("按 'q' 退出")
    
    try:
        while True:
            # 更新控制器
            done = controller.update()
            
            # 显示画面
            frame = vision.cap.read()[1]
            if frame is not None:
                current_pos = vision.detect_object(frame)
                target_pos = controller.current_target[:2] if controller.current_target else None
                
                display = vision.get_frame_with_overlay(frame, current_pos, target_pos)
                cv2.imshow('Electromagnet Control', display)
            
            # 检查退出
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
            
            if done:
                print("轨迹完成！")
                break
            
            time.sleep(0.02)  # 50Hz
    
    except KeyboardInterrupt:
        print("\n用户中断")
    
    finally:
        # 清理
        controller.reset()
        vision.release()
        cv2.destroyAllWindows()
        print("系统已关闭")

if __name__ == '__main__':
```

```python
class WorkspaceCalibrator:
    """工作台标定工具"""
    
    def __init__(self, vision):
        self.vision = vision
        self.points = []
        
    def mouse_callback(self, event, x, y, flags, param):
        """鼠标回调"""
        if event == cv2.EVENT_LBUTTONDOWN:
            self.points.append((x, y))
            print(f"添加点 {len(self.points)}: ({x}, {y})")
    
    def calibrate(self, frame):
        """交互式标定"""
        print("请点击工作台的四个角点 (左上、右上、右下、左下)")
        
        cv2.namedWindow('Calibration')
        cv2.setMouseCallback('Calibration', self.mouse_callback)
        
        while len(self.points) < 4:
            display = frame.copy()
            
            # 绘制已点击的点
            for i, point in enumerate(self.points):
                cv2.circle(display, point, 5, (0, 255, 0), -1)
                cv2.putText(display, str(i+1), (point[0]+10, point[1]), 
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)
            
            cv2.imshow('Calibration', display)
            cv2.waitKey(50)
        
        cv2.destroyWindow('Calibration')
        
        # 应用标定
        self.vision.calibrate_workspace(self.points)
        print("标定完成！")
        
        return self.points
```

## 常见问题排查

| 问题 | 可能原因 | 解决方案 |
|------|---------|---------|
| 物体不动 | 电磁铁功率不足 | 增加驱动电压/电流 |
| 物体抖动 | PID参数不当 | 调整Kp, Ki, Kd |
| 定位不准 | 视觉标定误差 | 重新标定工作台 |
| 响应延迟 | 控制频率太低 | 提高到100Hz |
| 物体脱落 | 吸力突然消失 | 添加保持力 |
| 电磁铁过热 | 持续大电流 | 降低占空比，增加散热 |

```python
import threading

class ThreadedController:
    """多线程控制器"""
    
    def __init__(self):
        self.em_array = ElectromagnetArray()
        self.vision = VisionTracker()
        self.running = False
        
        # 视觉线程
        self.vision_thread = None
        self.current_position = None
        
        # 控制线程
        self.control_thread = None
    
    def start(self):
        self.running = True
        
        self.vision_thread = threading.Thread(target=self._vision_loop)
        self.control_thread = threading.Thread(target=self._control_loop)
        
        self.vision_thread.start()
        self.control_thread.start()
    
    def _vision_loop(self):
        while self.running:
            frame = self.vision.cap.read()[1]
            self.current_position = self.vision.detect_object(frame)
            time.sleep(0.01)  # 100Hz
    
    def _control_loop(self):
        while self.running:
            if self.current_position:
                # 执行控制逻辑
                pass
            time.sleep(0.02)  # 50Hz
    
    def stop(self):
        self.running = False
        self.vision_thread.join()
        self.control_thread.join()
```
