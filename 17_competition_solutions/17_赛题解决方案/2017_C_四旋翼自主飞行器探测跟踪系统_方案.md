# 2017年电赛C题 - 四旋翼自主飞行器探测跟踪系统 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：四旋翼飞行器自主探测地面移动目标（小车），锁定后持续跟踪，保持指定距离和高度，全程自主完成。

**关键指标**：
- 目标探测成功率
- 跟踪稳定性
- 距离保持精度
- 跟踪持续时间

### 4.1 目标检测与跟踪模块

```python
# target_tracker.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite
from collections import deque

class TargetTracker:
    """目标检测+跟踪"""
    def __init__(self, model_path='target_yolo.rknn'):
        # RKNN检测模型
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0_1_2)
        
        # KCF跟踪器（备用）
        self.tracker = None
        self.tracking = False
        
        # 目标状态
        self.target_pos = None          # 图像坐标
        self.target_size = None         # 目标大小
        self.target_lost_count = 0      # 丢失计数
        self.target_history = deque(maxlen=30)  # 历史轨迹
        
        # 颜色模板（可选：目标小车颜色）
        self.target_color_lower = (0, 100, 100)  # HSV
        self.target_color_upper = (10, 255, 255)
    
    def detect_target(self, frame):
        """RKNN检测目标"""
        img = cv2.resize(frame, (640, 640))
        img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        outputs = self.rknn.inference(inputs=[img_rgb])
        
        detections = self._parse_output(outputs)
        
        # 选择置信度最高的目标
        if detections:
            best = max(detections, key=lambda d: d['confidence'])
            return best
        return None
    
    def track_target(self, frame, detection=None):
        """多策略融合跟踪"""
        h, w = frame.shape[:2]
        
        # 策略1：RKNN检测
        det_result = None
        if detection is None:
            det_result = self.detect_target(frame)
        else:
            det_result = detection
        
        # 策略2：KCF/NCS跟踪器
        track_result = None
        if self.tracking and self.tracker is not None:
            success, bbox = self.tracker.update(frame)
            if success:
                x, y, bw, bh = [int(v) for v in bbox]
                track_result = {
                    'center': (x + bw//2, y + bh//2),
                    'size': (bw, bh),
                    'bbox': (x, y, x+bw, y+bh)
                }
        
        # 策略3：颜色追踪（辅助）
        color_result = self._color_track(frame)
        
        # 融合结果
        if det_result is not None:
            self.target_pos = det_result['center']
            self.target_size = det_result['size']
            self.target_lost_count = 0
            
            # 初始化/更新跟踪器
            bbox = det_result['bbox']
            self.tracker = cv2.TrackerKCF_create()
            self.tracker.init(frame, (bbox[0], bbox[1], 
                                       bbox[2]-bbox[0], bbox[3]-bbox[1]))
            self.tracking = True
            
        elif track_result is not None:
            self.target_pos = track_result['center']
            self.target_size = track_result['size']
            self.target_lost_count = 0
            
        elif color_result is not None:
            self.target_pos = color_result['center']
            self.target_lost_count = 0
            
        else:
            self.target_lost_count += 1
            # 预测位置（卡尔曼滤波）
            if len(self.target_history) >= 3:
                self.target_pos = self._predict_position()
        
        # 记录历史
        if self.target_pos is not None:
            self.target_history.append(self.target_pos)
        
        return {
            'found': self.target_lost_count < 10,
            'position': self.target_pos,
            'size': self.target_size,
            'lost_count': self.target_lost_count
        }
    
    def _color_track(self, frame):
        """颜色追踪"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.target_color_lower, self.target_color_upper)
        
        # 红色需处理环绕
        mask2 = cv2.inRange(hsv, (170, 100, 100), (180, 255, 255))
        mask = cv2.bitwise_or(mask, mask2)
        
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            if cv2.contourArea(largest) > 200:
                M = cv2.moments(largest)
                if M['m00'] > 0:
                    cx = int(M['m10']/M['m00'])
                    cy = int(M['m01']/M['m00'])
                    return {'center': (cx, cy)}
        return None
    
    def _predict_position(self):
        """卡尔曼预测目标位置"""
        history = list(self.target_history)
        if len(history) < 2:
            return history[-1]
        
        # 简单线性外推
        dx = history[-1][0] - history[-2][0]
        dy = history[-1][1] - history[-2][1]
        
        pred_x = history[-1][0] + dx
        pred_y = history[-1][1] + dy
        
        return (pred_x, pred_y)
```

### 4.2 云台控制

```python
# gimbal_controller.py
import time

class GimbalController:
    """2轴云台控制"""
    def __init__(self, pan_pin=12, tilt_pin=13):
        import OPi.GPIO as GPIO
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(pan_pin, GPIO.OUT)
        GPIO.setup(tilt_pin, GPIO.OUT)
        
        self.pan_pwm = GPIO.PWM(pan_pin, 50)   # 50Hz
        self.tilt_pwm = GPIO.PWM(tilt_pin, 50)
        
        self.pan_angle = 90   # 中位
        self.tilt_angle = 60  # 俯视角度
        
        self.pan_pwm.start(self._angle_to_duty(self.pan_angle))
        self.tilt_pwm.start(self._angle_to_duty(self.tilt_angle))
        
        # PID
        self.pid_pan = PIDController(kp=0.08, ki=0.001, kd=0.02)
        self.pid_tilt = PIDController(kp=0.08, ki=0.001, kd=0.02)
    
    def track_target(self, target_x, target_y, frame_w, frame_h):
        """云台跟踪目标"""
        # 计算图像中心偏差
        err_x = target_x - frame_w / 2
        err_y = target_y - frame_h / 2
        
        # PID计算角度修正
        dpan = self.pid_pan.compute(err_x)
        dtilt = self.pid_tilt.compute(err_y)
        
        self.pan_angle = max(0, min(180, self.pan_angle - dpan))
        self.tilt_angle = max(20, min(120, self.tilt_angle + dtilt))
        
        self.pan_pwm.ChangeDutyCycle(self._angle_to_duty(self.pan_angle))
        self.tilt_pwm.ChangeDutyCycle(self._angle_to_duty(self.tilt_angle))
    
    def get_gimbal_offset(self):
        """获取云台偏移角（用于飞行控制补偿）"""
        dpan = self.pan_angle - 90
        dtilt = self.tilt_angle - 60
        return dpan, dtilt
    
    def center(self):
        """云台归中"""
        self.pan_angle = 90
        self.tilt_angle = 60
        self.pan_pwm.ChangeDutyCycle(self._angle_to_duty(90))
        self.tilt_pwm.ChangeDutyCycle(self._angle_to_duty(60))
    
    def _angle_to_duty(self, angle):
        return 2.5 + angle / 18.0


class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
    
    def compute(self, error):
        self.integral += error
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
```

### 4.3 跟踪飞行控制器

```python
# follow_controller.py
import math
import numpy as np

class FollowController:
    """跟踪飞行控制器"""
    def __init__(self, flight_ctrl):
        self.flight = flight_ctrl
        
        # 跟踪参数
        self.follow_distance = 3.0  # 跟踪距离 m
        self.follow_height = 3.0    # 跟踪高度 m
        self.max_speed = 2.0        # 最大跟踪速度 m/s
        
        # PID
        self.pid_forward = PIDController(kp=0.5, ki=0.05, kd=0.2)
        self.pid_lateral = PIDController(kp=0.5, ki=0.05, kd=0.2)
        self.pid_vertical = PIDController(kp=0.3, ki=0.02, kd=0.1)
        
        # 目标大小参考（用于距离估计）
        self.ref_size = 5000          # 参考像素面积
        self.ref_distance = 3.0       # 参考距离
    
    def compute_follow_velocity(self, target_info, gimbal_offset, depth_estimate):
        """计算跟踪速度"""
        if not target_info['found']:
            return 0, 0, 0, 0  # vx, vy, vz, yaw_rate
        
        # 目标在图像中的位置
        tx, ty = target_info['position']
        frame_w, frame_h = 640, 480
        
        # 1. 横向控制（左右跟随）
        lateral_err = (tx - frame_w/2) / (frame_w/2)  # 归一化
        vy = self.pid_lateral.compute(-lateral_err)
        
        # 2. 距离控制（前后跟随）
        if depth_estimate > 0:
            forward_err = depth_estimate - self.follow_distance
        else:
            # 用目标面积估算距离
            target_area = target_info.get('size', (0, 0))
            area = target_area[0] * target_area[1] if target_area else 0
            if area > 0:
                estimated_dist = self.ref_distance * math.sqrt(self.ref_size / area)
                forward_err = estimated_dist - self.follow_distance
            else:
                forward_err = 0
        
        vx = self.pid_forward.compute(forward_err)
        
        # 3. 高度控制
        current_alt = self.flight.get_altitude()
        height_err = current_alt - self.follow_height
        vz = self.pid_vertical.compute(-height_err)
        
        # 4. 偏航（始终面向目标）
        yaw_rate = -lateral_err * 30  # deg/s
        
        # 限速
        speed = math.sqrt(vx**2 + vy**2)
        if speed > self.max_speed:
            scale = self.max_speed / speed
            vx *= scale
            vy *= scale
        
        return vx, vy, vz, yaw_rate
```

### 4.4 任务主流程

```python
# main.py
import cv2
import time

def main():
    camera = cv2.VideoCapture(0)
    tracker = TargetTracker('target_yolo.rknn')
    gimbal = GimbalController(pan_pin=12, tilt_pin=13)
    flight = MAVLinkController('/dev/ttyS5')
    follow = FollowController(flight)
    
    state = 'TAKEOFF'
    follow_start_time = 0
    follow_duration = 300  # 最长跟踪5分钟
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        # 1. 目标检测/跟踪
        target_info = tracker.track_target(frame)
        
        # 2. 云台跟踪
        if target_info['found']:
            tx, ty = target_info['position']
            gimbal.track_target(tx, ty, frame.shape[1], frame.shape[0])
        
        # 3. 跟踪飞行
        gimbal_offset = gimbal.get_gimbal_offset()
        
        if state == 'TAKEOFF':
            flight.takeoff(follow.follow_height)
            time.sleep(8)
            state = 'SEARCH'
            
        elif state == 'SEARCH':
            if target_info['found']:
                state = 'FOLLOW'
                follow_start_time = time.time()
                print("目标锁定，开始跟踪")
            else:
                # 盘旋搜索
                flight.set_velocity_body(0, 0.5, 0)
                
        elif state == 'FOLLOW':
            if target_info['found']:
                vx, vy, vz, yaw_rate = follow.compute_follow_velocity(
                    target_info, gimbal_offset, depth_estimate=0)
                flight.set_velocity_body(vx, vy, vz, yaw_rate)
                
                print(f"跟踪: vx={vx:.2f} vy={vy:.2f} lost={target_info['lost_count']}")
            else:
                # 目标丢失，原地悬停等待
                if target_info['lost_count'] > 30:
                    state = 'SEARCH'
                    print("目标丢失，重新搜索")
            
            # 超时停止
            if time.time() - follow_start_time > follow_duration:
                state = 'LAND'
                
        elif state == 'LAND':
            flight.land()
            break
        
        time.sleep(0.03)  # 30Hz

if __name__ == '__main__':
    main()
```

## 五、关键技术要点

### 5.1 多策略融合跟踪
| 策略 | 优先级 | 适用场景 |
|------|--------|----------|
| RKNN检测 | 最高 | 首次发现/重新捕获 |
| KCF跟踪器 | 高 | 持续跟踪（低延迟） |
| 颜色追踪 | 中 | 检测器失败时辅助 |
| 卡尔曼预测 | 低 | 短暂丢失时外推 |

### 5.2 跟踪距离估算
- **目标面积法**：面积∝距离²，简单有效
- **激光测距**：TFmini直接测高，配合角度算斜距
- **双目视差**：可选方案，精度更高
