## 二、视觉方案

### 2.1 侧视相机检测摆杆角度
```python
import cv2
import numpy as np

class PendulumAngleDetector:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 120)
        
        # 旋转轴位置（标定获取）
        self.pivot_pixel = None  # (px, py)
        self.pixels_per_mm = None
        
        # 摆杆颜色（贴红色标记）
        self.lower_hsv = np.array([0, 120, 150])
        self.upper_hsv = np.array([10, 255, 255])
    
    def calibrate_pivot(self, frame, pivot_pixel):
        """标定旋转轴的像素位置"""
        self.pivot_pixel = np.array(pivot_pixel, dtype=np.float32)
    
    def detect_pendulum_angle(self, frame):
        """
        检测摆杆角度
        返回: theta (rad), 0=正上方, 正=逆时针
        """
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.lower_hsv, self.upper_hsv)
        
        # 在摆杆远端找标记点
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None
        
        # 取最大轮廓的质心作为摆杆端点
        c = max(contours, key=cv2.contourArea)
        M = cv2.moments(c)
        if M["m00"] == 0:
            return None
        
        end_pixel = np.array([M["m10"]/M["m00"], M["m01"]/M["m00"]])
        
        # 计算角度（相对正上方）
        dx = end_pixel[0] - self.pivot_pixel[0]
        dy = self.pivot_pixel[1] - end_pixel[1]  # 图像Y轴翻转
        
        theta = np.arctan2(dx, dy)  # 正上方为0，逆时针正
        
        return theta
    
    def detect_arm_angle(self, frame):
```

## 三、控制方案

### 3.1 状态空间模型

旋转倒立摆状态向量：**x** = [θ_arm, θ_pend, ω_arm, ω_pend]^T

```
状态方程:  x_dot = A*x + B*u
输出方程:  y = C*x

其中:
θ_arm   = 旋转臂角度
θ_pend  = 摆杆角度（正上方为0）
ω_arm   = 旋转臂角速度
ω_pend  = 摆杆角速度
u       = 电机控制量
```

### 3.2 LQR控制器
```python
import numpy as np
from scipy import linalg

class LQRController:
    def __init__(self):
        # 系统参数（需实测辨识）
        self.M_arm = 0.15      # 旋转臂质量(kg)
        self.L_arm = 0.25      # 旋转臂长度(m)
        self.m_pend = 0.05     # 摆锤质量(kg)
        self.L_pend = 0.30     # 摆杆长度(m)
        self.g = 9.81
        self.J_arm = self.M_arm * self.L_arm**2 / 3
        self.J_pend = self.m_pend * self.L_pend**2 / 3
        
        # 线性化后的状态空间（在倒立平衡点附近）
        self.A = np.array([
            [0, 0, 1, 0],
            [0, 0, 0, 1],
            [0, -self.m_pend*self.g/self.J_arm, 0, 0],
            [0, (self.M_arm+self.m_pend)*self.g/(self.J_pend), 0, 0]
        ])
        
        self.B = np.array([
            [0],
            [0],
            [1/self.J_arm],
            [-1/self.J_pend]
        ])
        
        # LQR权重矩阵
        self.Q = np.diag([100, 1000, 10, 50])  # 摆杆角度权重最大
        self.R = np.array([[1]])                # 控制量权重
        
        # 计算最优增益K
        self.K = self._solve_lqr()
    
    def _solve_lqr(self):
        """求解代数Riccati方程"""
        P = linalg.solve_continuous_are(self.A, self.B, self.Q, self.R)
        K = np.linalg.inv(self.R) @ self.B.T @ P
        return K
    
    def compute(self, state):
        """
        state: [theta_arm, theta_pend, omega_arm, omega_pend]
        返回: 控制量 u
        """
        u = -self.K @ state
        return np.clip(u[0], -1.0, 1.0)  # 归一化控制量
```

### 3.3 能量摆起控制
```python
class SwingUpController:
    def __init__(self):
        self.target_energy = 0  # 倒立时的势能
        self.E_threshold = 0.05  # 切换到LQR的能量阈值
    
    def compute_energy(self, theta_pend, omega_pend, m, L, g):
        """计算摆杆总能量（以旋转轴为参考点）"""
        # 动能
        KE = 0.5 * m * (L * omega_pend)**2
        # 势能（正上方为0势能点）
        PE = m * g * L * (1 - np.cos(theta_pend))
        return KE + PE
    
    def compute(self, theta_pend, omega_pend, m, L, g):
        """
        能量泵控制
        当摆杆在下半区时施加力矩增加能量
        """
        E = self.compute_energy(theta_pend, omega_pend, m, L, g)
        dE = self.target_energy - E
        
        # 能量泵策略：在下半区且向下摆时施加同方向力矩
        if np.abs(theta_pend) > np.pi/2:  # 下半区
            # 力矩方向与角速度方向相同（泵入能量）
            u = 0.5 * np.sign(omega_pend) * np.tanh(dE)
        else:
            u = 0
        
        return u
```

### 3.4 卡尔曼滤波（状态估计）
```python
class KalmanFilter:
    def __init__(self, dt=0.008):
        self.dt = dt
        
        # 状态: [theta_arm, theta_pend, omega_arm, omega_pend]
        self.x = np.zeros(4)
        self.P = np.eye(4) * 0.1
        
        # 过程噪声
        self.Q = np.diag([0.001, 0.001, 0.01, 0.01])
        
        # 观测噪声（视觉角度噪声较大）
        self.R = np.diag([0.005, 0.005])  # 角度观测噪声
        
        # 观测矩阵（只观测角度，不直接观测角速度）
        self.H = np.array([
            [1, 0, 0, 0],
            [0, 1, 0, 0]
        ])
    
    def predict(self, A, B, u):
        """预测步骤"""
        self.x = A @ self.x + B.flatten() * u
        self.P = A @ self.P @ A.T + self.Q
    
    def update(self, z):
        """
        更新步骤
        z: 观测向量 [theta_arm, theta_pend]
        """
        y = z - self.H @ self.x
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        self.x = self.x + K @ y
        self.P = (np.eye(4) - K @ self.H) @ self.P
        
        return self.x
    
    def get_state(self):
        """返回估计状态"""
        return self.x.copy()
```

### 3.5 主控制循环
```python
import time

class InvertedPendulumSystem:
    def __init__(self):
        self.vision = PendulumAngleDetector(0)
        self.lqr = LQRController()
        self.swing_up = SwingUpController()
        self.kf = KalmanFilter(dt=0.008)
        
        self.mode = 'swing_up'  # 'swing_up' or 'balance'
    
    def run(self):
        prev_time = time.time()
        
        while True:
            frame = self.vision.get_frame()
            now = time.time()
            dt = now - prev_time
            prev_time = now
            
            # 视觉检测
            theta_pend = self.vision.detect_pendulum_angle(frame)
            theta_arm = self.vision.detect_arm_angle(frame)
            
            if theta_pend is None:
                continue
            
            # 卡尔曼滤波（融合视觉+模型）
            z = np.array([theta_arm, theta_pend])
            self.kf.predict(self.lqr.A, self.lqr.B, self.last_u)
            state = self.kf.update(z)
            
            # 模式切换
            E = self.swing_up.compute_energy(
                state[1], state[3], 
                self.lqr.m_pend, self.lqr.L_pend, self.lqr.g
            )
            E_max = self.lqr.m_pend * self.lqr.g * self.lqr.L_pend * 2
            
            if abs(state[1]) < 0.2 and abs(state[3]) < 1.0:
                self.mode = 'balance'
            
            # 计算控制量
            if self.mode == 'swing_up':
                u = self.swing_up.compute(
                    state[1], state[3],
                    self.lqr.m_pend, self.lqr.L_pend, self.lqr.g
                )
            else:
                u = self.lqr.compute(state)
            
            self.last_u = u
            self.motor.set_speed(u)
```
