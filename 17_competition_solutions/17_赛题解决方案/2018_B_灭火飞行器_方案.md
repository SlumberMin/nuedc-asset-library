# 2018年B题 — 灭火飞行器

## 题目概述
设计一个能够自主飞行、识别火焰并执行灭火任务的飞行器。

## 飞行控制器接口

```python
import serial
import struct
import time

class FlightController:
    """与STM32飞控协处理器通信"""
    
    # 协议定义
    CMD_SET_ATTITUDE = 0x01
    CMD_SET_THRUST = 0x02
    CMD_GET_STATE = 0x10
    CMD_ARM = 0xA0
    CMD_DISARM = 0xA1
    
    def __init__(self, port='/dev/ttyS3', baudrate=115200):
        self.ser = serial.Serial(port, baudrate, timeout=0.1)
        self.armed = False
        
        # 飞行状态
        self.state = {
            'roll': 0, 'pitch': 0, 'yaw': 0,
            'altitude': 0, 'battery': 0,
            'armed': False
        }
    
    def _send_cmd(self, cmd, data=b''):
        """发送指令"""
        packet = struct.pack('<BB', cmd, len(data)) + data
        crc = sum(packet) & 0xFF
        self.ser.write(packet + struct.pack('<B', crc))
    
    def _read_state(self):
        """读取飞控状态"""
        self._send_cmd(self.CMD_GET_STATE)
        resp = self.ser.read(20)
        if len(resp) >= 18:
            self.state['roll'] = struct.unpack('<f', resp[0:4])[0]
            self.state['pitch'] = struct.unpack('<f', resp[4:8])[0]
            self.state['yaw'] = struct.unpack('<f', resp[8:12])[0]
            self.state['altitude'] = struct.unpack('<f', resp[12:16])[0]
            self.state['battery'] = struct.unpack('<H', resp[16:18])[0] / 1000
    
    def arm(self):
        """解锁电机"""
        self._send_cmd(self.CMD_ARM)
        self.armed = True
        time.sleep(1)
    
    def disarm(self):
        """锁定电机"""
        self._send_cmd(self.CMD_DISARM)
        self.armed = False
    
    def set_attitude(self, roll=0, pitch=0, yaw_rate=0, throttle=0):
        """设置姿态指令
        roll: 横滚角(-30~30度)
        pitch: 俯仰角(-30~30度)
        yaw_rate: 偏航角速度(-180~180 deg/s)
        throttle: 油门(0~100%)
        """
        data = struct.pack('<ffff', roll, pitch, yaw_rate, throttle)
        self._send_cmd(self.CMD_SET_ATTITUDE, data)
    
    def hover(self, altitude=1.0):
        """定高悬停"""
        self._read_state()
        alt_error = altitude - self.state['altitude']
        throttle = 50 + alt_error * 20  # 简单P控制
        throttle = max(0, min(100, throttle))
        self.set_attitude(throttle=throttle)
    
    def move_to(self, dx, dy, altitude=1.0):
        """移动到相对位置"""
        # 将位移转换为姿态角
        pitch = max(-20, min(20, dx * 10))  # 前后
        roll = max(-20, min(20, dy * 10))   # 左右
        self.hover(altitude)
        self.set_attitude(roll=roll, pitch=pitch)
```

## 火焰检测系统

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class FlameDetector:
    """火焰检测器（NPU + 红外融合）"""
    
    def __init__(self):
        # NPU模型（YOLOv5s火焰检测）
        self.rknn = RKNNLite()
        self.rknn.load_rknn('flame_detector.rknn')
        self.rknn.init_runtime()
        
        # 红外火焰传感器（模拟量）
        import spidev
        self.spi = spidev.SpiDev()
        self.spi.open(0, 0)
    
    def detect_flame_visual(self, frame):
        """视觉火焰检测"""
        img = cv2.resize(frame, (640, 640))
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        
        outputs = self.rknn.inference(inputs=[img])
        
        flames = []
        for det in outputs[0]:
            x, y, w, h, conf = det[:5]
            if conf > 0.6:
                cx, cy = int(x + w/2), int(y + h/2)
                flames.append({
                    'center': (cx, cy),
                    'bbox': (int(x), int(y), int(w), int(h)),
                    'confidence': float(conf),
                    'area': w * h
                })
        
        # 按置信度排序
        flames.sort(key=lambda f: f['confidence'], reverse=True)
        return flames
    
    def detect_flame_ir(self):
        """红外火焰传感器检测（4方向）"""
        ir_values = []
        for ch in range(4):
            cmd = [1, (8 + ch) << 4, 0]
            result = self.spi.xfer2(cmd)
            value = ((result[1] & 0x03) << 8) | result[2]
            ir_values.append(value)
        
        # 判断火焰方向（值越大越近/强）
        # 通道: 0=前, 1=右, 2=后, 3=左
        if max(ir_values) > 500:
            direction = np.argmax(ir_values)
            return True, direction, ir_values
        return False, -1, ir_values
    
    def get_flame_offset(self, flame, frame_shape):
        """计算火焰相对画面中心的偏移"""
        h, w = frame_shape[:2]
        cx, cy = flame['center']
        return (cx - w//2, cy - h//2)
    
    def estimate_distance(self, flame):
        """根据火焰面积估算距离（经验公式）"""
        # 面积越大越近
        area = flame['area']
        if area > 50000:
            return 0.5  # 0.5m
        elif area > 20000:
            return 1.0
        elif area > 5000:
            return 2.0
        else:
            return 3.0
```

## 灭火飞行器主控

```python
import cv2
import time
import threading

class FirefightingDrone:
    """灭火飞行器主控"""
    
    # 飞行状态
    STATE_IDLE = 0
    STATE_TAKEOFF = 1
    STATE_SEARCH = 2
    STATE_APPROACH = 3
    STATE_EXTINGUISH = 4
    STATE_RETURN = 5
    STATE_LAND = 6
    
    def __init__(self):
        self.camera = cv2.VideoCapture(0)
        self.flight = FlightController()
        self.flame_detector = FlameDetector()
        self.suppression = FireSuppression()
        
        self.state = self.STATE_IDLE
        self.target_altitude = 1.0  # 悬停高度(m)
        self.home_position = (0, 0)
    
    def search_flame(self):
        """搜索火焰"""
        ret, frame = self.camera.read()
        if not ret:
            return None
        
        # 视觉检测
        flames = self.flame_detector.detect_flame_visual(frame)
        if flames:
            return flames[0]
        
        # 红外检测辅助
        ir_detected, direction, _ = self.flame_detector.detect_flame_ir()
        if ir_detected:
            # 转向红外指示方向
            yaw_map = {0: 0, 1: 90, 2: 180, 3: -90}
            self.flight.set_attitude(yaw_rate=yaw_map.get(direction, 0))
            time.sleep(0.5)
        
        return None
    
    def approach_flame(self, flame):
        """接近火焰"""
        offset = self.flame_detector.get_flame_offset(flame, 
                    (480, 640))  # 假设帧大小
        distance = self.flame_detector.estimate_distance(flame)
        
        # 调整飞行方向
        dx = offset[0] * 0.001  # 像素到米的转换系数
        dy = offset[1] * 0.001
        
        if distance > 1.0:
            # 还需接近
            self.flight.move_to(dx, dy, self.target_altitude)
            return False
        else:
            # 到达灭火距离
            return True
    
    def extinguish(self):
        """执行灭火"""
        print("开始灭火！")
        
        # 对准火焰喷水
        for _ in range(3):  # 多次喷射
            ret, frame = self.camera.read()
            if ret:
                flames = self.flame_detector.detect_flame_visual(frame)
                if flames:
                    # 微调对准
                    offset = self.flame_detector.get_flame_offset(flames[0], frame.shape)
                    # ...
            
            self.suppression.spray(duration=2.0)
            time.sleep(1)
            
            # 检查火焰是否熄灭
            ret, frame = self.camera.read()
            if ret:
                flames = self.flame_detector.detect_flame_visual(frame)
                if not flames:
                    print("火焰已熄灭！")
                    return True
        
        return False
    
    def run(self):
        """主循环"""
        print("灭火飞行器启动")
        
        while True:
            if self.state == self.STATE_IDLE:
                # 等待起飞指令
                cmd = input("输入 'start' 起飞: ")
                if cmd == 'start':
                    self.flight.arm()
                    self.state = self.STATE_TAKEOFF
            
            elif self.state == self.STATE_TAKEOFF:
                self.flight.hover(self.target_altitude)
                time.sleep(3)
                self.state = self.STATE_SEARCH
            
            elif self.state == self.STATE_SEARCH:
                flame = self.search_flame()
                if flame:
                    print(f"发现火焰！置信度: {flame['confidence']:.2f}")
                    self.state = self.STATE_APPROACH
            
            elif self.state == self.STATE_APPROACH:
                flame = self.search_flame()
                if flame:
                    reached = self.approach_flame(flame)
                    if reached:
                        self.state = self.STATE_EXTINGUISH
                else:
                    self.state = self.STATE_SEARCH
            
            elif self.state == self.STATE_EXTINGUISH:
                success = self.extinguish()
                self.state = self.STATE_RETURN
            
            elif self.state == self.STATE_RETURN:
                # 返回起点
                self.flight.move_to(-self.flight.state['altitude'], 0, 
                                   self.target_altitude)
                time.sleep(5)
                self.state = self.STATE_LAND
            
            elif self.state == self.STATE_LAND:
                self.flight.disarm()
                print("任务完成，已降落")
                break
            
            time.sleep(0.1)  # 10Hz控制频率

if __name__ == '__main__':
    drone = FirefightingDrone()
    drone.run()
```

## 安全机制

```python
class SafetyMonitor:
    """飞行安全监控"""
    
    LOW_BATTERY = 10.5  # V
    MAX_TILT = 45       # 度
    MAX_ALTITUDE = 5.0  # 米
    
    def check(self, flight_state):
        """安全检查"""
        if flight_state['battery'] < self.LOW_BATTERY:
            return 'LOW_BATTERY'
        
        if abs(flight_state['roll']) > self.MAX_TILT or \
           abs(flight_state['pitch']) > self.MAX_TILT:
            return 'EXCESSIVE_TILT'
        
        if flight_state['altitude'] > self.MAX_ALTITUDE:
            return 'TOO_HIGH'
        
        return None
    
    def emergency_land(self, flight_controller):
        """紧急降落"""
        print("执行紧急降落！")
        for throttle in range(50, 0, -5):
            flight_controller.set_attitude(throttle=throttle)
            time.sleep(0.5)
        flight_controller.disarm()
```
