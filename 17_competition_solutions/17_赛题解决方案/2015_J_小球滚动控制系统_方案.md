# 2015年电赛J题 - 小球滚动控制系统 Orange Pi 5解决方案

## 一、题目分析

在一根导轨上放置小球，通过控制导轨的倾斜角度，使小球能够在导轨上精确定位，或按照预设轨迹运动。

### 核心要求
1. 实时检测小球在导轨上的位置
2. 控制导轨倾斜使小球到达指定位置
3. 小球位置控制精度 ≤ 1cm
4. 响应速度快、无超调

---

## 二、系统总体方案

```
┌─────────────────────────────────────────────┐
│              Orange Pi 5 (主控)              │
│  ┌──────────┐ ┌──────────┐ ┌──────────────┐ │
│  │ 视觉检测 │ │ PID控制器│ │ 状态机管理  │ │
│  │ 小球位置 │ │ 串级PID  │ │             │ │
│  └────┬─────┘ └────┬─────┘ └──────┬───────┘ │
│       │            │              │          │
│  ┌────▼────────────▼──────────────▼───────┐  │
│  │         UART / PWM 通信层              │  │
│  └────┬─────────────┬──────────────┬──────┘  │
└───────┼─────────────┼──────────────┼─────────┘
        │             │              │
   ┌────▼────┐   ┌────▼────┐   ┌────▼────┐
   │STM32    │   │ MPU6050 │   │ 角度/位 │
   │舵机驱动 │   │ 角度反馈│   │ 置传感器│
   └─────────┘   └─────────┘   └─────────┘
```

### 控制策略
- **外环**: 位置PID (小球位置 → 目标角度)
- **内环**: 角度PID (目标角度 → 舵机PWM)
- **视觉反馈**: 60fps高速检测小球位置

## 四、软件方案

### 4.1 小球视觉检测 (高速)

```python
import cv2
import numpy as np
import time

class BallDetector:
    """高速小球位置检测器 - 目标120fps"""

    def __init__(self, roi=None):
        # 限定检测区域提高速度
        self.roi = roi  # (x, y, w, h)

        # 小球颜色范围 (HSV) - 根据实际小球颜色调整
        # 红色小球
        self.lower_color1 = np.array([0, 100, 100])
        self.upper_color1 = np.array([10, 255, 255])
        self.lower_color2 = np.array([160, 100, 100])
        self.upper_color2 = np.array([180, 255, 255])

        # 导轨参考线 (用于坐标映射)
        self.rail_start = None  # 导轨起点像素坐标
        self.rail_end = None    # 导轨终点像素坐标
        self.rail_length = 0.5  # 导轨实际长度(m)

        # 卡尔曼滤波器
        self.kalman = cv2.KalmanFilter(4, 2)
        self.kalman.measurementMatrix = np.array(
            [[1, 0, 0, 0], [0, 1, 0, 0]], np.float32)
        self.kalman.transitionMatrix = np.array(
            [[1, 0, 1, 0], [0, 1, 0, 1],
             [0, 0, 1, 0], [0, 0, 0, 1]], np.float32)
        self.kalman.processNoiseCov = np.eye(4, dtype=np.float32) * 0.03

        self.last_pos = None
        self.last_time = time.time()

    def calibrate_rail(self, frame):
        """标定导轨位置 (手动或自动)"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        edges = cv2.Canny(gray, 50, 150)
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 100,
                                minLineLength=100, maxLineGap=10)
        if lines is not None:
            # 选择最长的水平线作为导轨
            longest = max(lines, key=lambda l:
                         abs(l[0][2] - l[0][0]))
            self.rail_start = (longest[0][0], longest[0][1])
            self.rail_end = (longest[0][2], longest[0][3])
            print(f"导轨标定: {self.rail_start} -> {self.rail_end}")

    def detect(self, frame):
        """
        检测小球位置
        返回: (position_m, velocity_mps, pixel_pos)
        position_m: 小球在导轨上的位置(m), 中心为0
        velocity_mps: 小球速度(m/s)
        """
        t_now = time.time()
        dt = t_now - self.last_time
        self.last_time = t_now

        # ROI裁剪
        if self.roi:
            x, y, w, h = self.roi
            roi_frame = frame[y:y+h, x:x+w]
        else:
            roi_frame = frame

        # 颜色检测 (HSV)
        hsv = cv2.cvtColor(roi_frame, cv2.COLOR_BGR2HSV)

        # 二值化
        mask1 = cv2.inRange(hsv, self.lower_color1, self.upper_color1)
        mask2 = cv2.inRange(hsv, self.lower_color2, self.upper_color2)
        mask = mask1 | mask2

        # 形态学操作
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)

        # 查找轮廓
        contours, _ = cv2.findContours(
            mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        pixel_pos = None
        if contours:
            # 选择最大轮廓
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)

            if 50 < area < 5000:  # 面积过滤
                M = cv2.moments(largest)
                if M['m00'] > 0:
                    cx = int(M['m10'] / M['m00'])
                    cy = int(M['m01'] / M['m00'])

                    if self.roi:
                        cx += self.roi[0]
                        cy += self.roi[1]

                    pixel_pos = (cx, cy)

                    # 卡尔曼滤波
                    measurement = np.array([[cx], [cy]], np.float32)
                    self.kalman.correct(measurement)

        # 卡尔曼预测
        prediction = self.kalman.predict()
        pred_x, pred_y = int(prediction[0]), int(prediction[1])

        if pixel_pos is None:
            pixel_pos = (pred_x, pred_y)  # 使用预测值

        # 像素坐标转实际位置
        position_m = self._pixel_to_position(pixel_pos)

        # 计算速度
        velocity = 0.0
        if self.last_pos is not None and dt > 0:
            velocity = (position_m - self.last_pos) / dt
        self.last_pos = position_m

        return position_m, velocity, pixel_pos

    def _pixel_to_position(self, pixel_pos):
        """像素坐标转换为导轨上的实际位置(m)"""
        if self.rail_start is None or self.rail_end is None:
            # 未标定, 使用简单映射
            return (pixel_pos[0] - 320) / 640.0 * self.rail_length

        # 投影到导轨线上
        px, py = pixel_pos
        sx, sy = self.rail_start
        ex, ey = self.rail_end

        # 计算投影比例
        rail_vec = np.array([ex - sx, ey - sy])
        point_vec = np.array([px - sx, py - sy])
        rail_len = np.linalg.norm(rail_vec)
        rail_unit = rail_vec / rail_len

        projection = np.dot(point_vec, rail_unit)
        ratio = projection / rail_len  # 0~1

        # 转换为实际位置, 中心为0
        position = (ratio - 0.5) * self.rail_length
        return position
```

### 4.2 串级PID控制器

```python
import time

class CascadePIDController:
    """串级PID控制器: 位置环 -> 角度环"""

    def __init__(self):
        # 外环: 位置PID
        self.pos_pid = PID(
            Kp=3.0, Ki=0.01, Kd=2.0,
            output_min=-0.3, output_max=0.3  # 最大倾斜角度(rad)
        )

        # 内环: 角度PID (由STM32执行)
        # self.angle_pid 在STM32上运行

        # 前馈补偿
        self.ff_gain = 0.1  # 速度前馈增益

        # 状态
        self.target_pos = 0.0  # 目标位置(m)
        self.mode = 'POSITION'  # POSITION / TRACKING

    def compute(self, current_pos, current_vel, current_angle, dt):
        """
        计算控制输出
        返回: 目标倾斜角度(rad)
        """
        if self.mode == 'POSITION':
            # 位置PID
            angle_cmd = self.pos_pid.compute(
                self.target_pos, current_pos, dt)

            # 速度前馈阻尼
            angle_cmd += self.ff_gain * current_vel

        elif self.mode == 'TRACKING':
            # 轨迹跟踪模式 (预留)
            angle_cmd = 0

        return angle_cmd


class PID:
    """通用PID控制器"""
    def __init__(self, Kp=0, Ki=0, Kd=0,
                 output_min=-1, output_max=1,
                 integral_max=1.0):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.output_min = output_min
        self.output_max = output_max
        self.integral_max = integral_max

        self.integral = 0.0
        self.prev_error = 0.0
        self.first = True

    def compute(self, target, current, dt):
        error = target - current

        # 积分 (带抗饱和)
        self.integral += error * dt
        self.integral = max(-self.integral_max,
                           min(self.integral_max, self.integral))

        # 微分
        if self.first:
            derivative = 0
            self.first = False
        else:
            derivative = (error - self.prev_error) / dt
        self.prev_error = error

        # 输出
        output = (self.Kp * error +
                  self.Ki * self.integral +
                  self.Kd * derivative)

        return max(self.output_min, min(self.output_max, output))

    def reset(self):
        self.integral = 0
        self.prev_error = 0
        self.first = True
```

### 4.3 主程序

```python
import cv2
import time
import serial
import struct
from ball_detector import BallDetector
from cascade_pid import CascadePIDController, PID

class BallControlSystem:
    def __init__(self):
        # 摄像头配置 (高速模式)
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 120)  # 尝试120fps

        self.detector = BallDetector(roi=(80, 150, 480, 180))
        self.controller = CascadePIDController()

        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)

        # 性能监控
        self.fps = 0
        self.frame_count = 0
        self.fps_time = time.time()

        # 控制模式
        self.target_pos = 0.0  # 目标位置(m)
        self.mode = 'STOP'  # STOP / POSITION / DEMO

    def send_angle_cmd(self, angle_x, angle_y=0):
        """发送角度指令到STM32"""
        buf = bytearray(10)
        buf[0] = 0xAA
        struct.pack_into('<f', buf, 1, angle_x)
        struct.pack_into('<f', buf, 5, angle_y)
        buf[9] = 0x55
        self.ser.write(buf)

    def read_feedback(self):
        """读取STM32反馈"""
        if self.ser.in_waiting >= 10:
            data = self.ser.read(10)
            if data[0] == 0xBB and data[9] == 0x55:
                angle, pos = struct.unpack('<ff', data[1:9])
                return angle, pos
        return None, None

    def run(self):
        print("=== 小球滚动控制系统 ===")
        print("指令: s=停止, p<位置>=设目标位置, d=演示模式, c=标定")

        self.detector.calibrate_rail(
            self.cap.read()[1] if self.cap.read()[0] else None
        )

        last_time = time.time()

        while True:
            ret, frame = self.cap.read()
            if not ret:
                continue

            current_time = time.time()
            dt = current_time - last_time
            last_time = current_time

            # 检测小球位置
            position, velocity, pixel_pos = self.detector.detect(frame)

            # 读取角度反馈
            angle_fb, _ = self.read_feedback()

            if self.mode == 'POSITION':
                # 串级PID控制
                angle_cmd = self.controller.compute(
                    position, velocity,
                    angle_fb if angle_fb else 0,
                    dt
                )
                self.send_angle_cmd(angle_cmd)

            elif self.mode == 'DEMO':
                # 演示模式: 小球来回运动
                import math
                self.target_pos = 0.15 * math.sin(current_time * 0.5)
                self.controller.target_pos = self.target_pos
                angle_cmd = self.controller.compute(
                    position, velocity,
                    angle_fb if angle_fb else 0, dt
                )
                self.send_angle_cmd(angle_cmd)

            else:  # STOP
                self.send_angle_cmd(0)

            # 性能监控
            self.frame_count += 1
            if current_time - self.fps_time >= 1.0:
                self.fps = self.frame_count
                self.frame_count = 0
                self.fps_time = current_time

            # 可视化
            self._draw(frame, position, velocity, pixel_pos)

            # 键盘输入
            key = cv2.waitKey(1) & 0xFF
            if key == ord('q'):
                break
            elif key == ord('s'):
                self.mode = 'STOP'
                self.controller.pos_pid.reset()
            elif key == ord('p'):
                self.mode = 'POSITION'
                self.target_pos = 0.1  # 示例目标
                self.controller.target_pos = self.target_pos
            elif key == ord('d'):
                self.mode = 'DEMO'

        self.send_angle_cmd(0)
        self.cap.release()
        cv2.destroyAllWindows()

    def _draw(self, frame, position, velocity, pixel_pos):
        """绘制可视化信息"""
        # 小球位置
        if pixel_pos:
            cv2.circle(frame, pixel_pos, 15, (0, 0, 255), 3)

        # 导轨参考线
        if self.detector.rail_start and self.detector.rail_end:
            cv2.line(frame, self.detector.rail_start,
                    self.detector.rail_end, (0, 255, 0), 2)

        # 信息文字
        cv2.putText(frame, f"FPS: {self.fps}", (10, 25),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        cv2.putText(frame, f"Pos: {position*100:.1f}cm", (10, 50),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)
        cv2.putText(frame, f"Vel: {velocity*100:.1f}cm/s", (10, 75),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 0), 2)
        cv2.putText(frame, f"Target: {self.target_pos*100:.1f}cm", (10, 100),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 255), 2)
        cv2.putText(frame, f"Mode: {self.mode}", (10, 125),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 200, 0), 2)

        cv2.imshow('Ball Control', frame)

if __name__ == '__main__':
    system = BallControlSystem()
    system.run()
```

## 五、调试要点

### 5.1 PID参数整定
| 参数 | 推荐范围 | 说明 |
|------|---------|------|
| 位置环Kp | 1.0~5.0 | 响应速度 |
| 位置环Ki | 0~0.05 | 消除稳态误差 |
| 位置环Kd | 0.5~3.0 | 阻尼, 减少超调 |
| 前馈增益 | 0.05~0.2 | 速度阻尼 |

### 5.2 性能指标
- 检测帧率: ≥ 60fps (640×480)
- 位置精度: ≤ 5mm
- 响应时间: ≤ 0.5s
- 稳态误差: ≤ 1cm

### 5.3 关键技巧
1. **ROI限定**: 缩小检测区域提高帧率
2. **颜色标定**: 现场光照下重新标定HSV范围
3. **卡尔曼滤波**: 平滑位置检测, 消除噪声
4. **前馈阻尼**: 用速度信息提前制动, 减少超调
