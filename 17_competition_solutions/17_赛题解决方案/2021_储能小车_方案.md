# 2021 储能小车 - Orange Pi 5 解决方案

## 题目分析

小车在赛道上行驶，通过特定方式（如光电/电磁/无线）采集赛道上的能量并存储，在需要时利用存储的能量驱动行驶。要求尽可能多地通过赛道上的能量区域，同时高效利用采集的能量。

### 方案一：光伏采集
```python
class SolarHarvester:
    """太阳能板能量采集"""
    def __init__(self):
        import spidev
        self.adc = spidev.SpiDev()
        self.adc.open(0, 0)
        self.panel_voltage = 0
        self.panel_current = 0

    def read_panel(self):
        """读取光伏板输出"""
        # ADC通道0: 电压(分压后)
        # ADC通道1: 电流(采样电阻+运放)
        v_raw = self._read_adc(0)
        i_raw = self._read_adc(1)
        self.panel_voltage = v_raw * 5.0  # 分压还原
        self.panel_current = i_raw * 2.0  # 电流放大还原
        return self.panel_voltage, self.panel_current

    def get_power(self):
        v, i = self.read_panel()
        return v * i

    def _read_adc(self, ch):
        resp = self.adc.xfer2([0x01, (0x80 | (ch << 4)), 0x00])
        val = ((resp[1] & 0x03) << 8) | resp[2]
        return val / 1023.0 * 3.3
```

## 能量管理系统

```python
import time
import numpy as np

class EnergyStorageManager:
    """储能管理 - 超级电容SOC估算与功率分配"""
    def __init__(self, cap_voltage_max=5.5, cap_voltage_min=2.0):
        self.v_max = cap_voltage_max
        self.v_min = cap_voltage_min
        self.capacity_j = 100  # 电容组储能(焦耳)估算值
        self.soc = 0.5  # 初始50%

        import spidev
        self.adc = spidev.SpiDev()
        self.adc.open(0, 1)  # SPI0.1 电容电压采集

    def read_cap_voltage(self):
        resp = self.adc.xfer2([0x01, 0x80, 0x00])
        val = ((resp[1]&0x03)<<8) | resp[2]
        return val / 1023.0 * 3.3 * 2.0  # 分压还原

    def update_soc(self):
        """更新SOC"""
        v = self.read_cap_voltage()
        self.soc = (v**2 - self.v_min**2) / (self.v_max**2 - self.v_min**2)
        self.soc = np.clip(self.soc, 0, 1)
        return self.soc

    def get_available_power(self):
        """可用功率预算(W)"""
        v = self.read_cap_voltage()
        # 最大放电功率 = V² / (4*ESR)
        esr = 0.5  # 等效串联电阻(Ω)
        p_max = v**2 / (4 * esr)
        return p_max * min(self.soc, 0.8)  # 保留20%底量

    def energy_strategy(self, soc, speed_demand, harvesting):
        """
        能量策略决策
        soc: 电容SOC
        speed_demand: 速度需求(0~1)
        harvesting: 是否正在采集能量
        """
        if harvesting:
            # 能量采集区：减速最大化充电
            return {
                'speed_factor': 0.5,
                'charge_priority': 'HIGH',
                'mode': 'HARVEST'
            }
        elif soc > 0.7:
            # 高电量：正常行驶
            return {
                'speed_factor': 1.0,
                'charge_priority': 'LOW',
                'mode': 'NORMAL'
            }
        elif soc > 0.3:
            # 中等电量：节能行驶
            return {
                'speed_factor': 0.7,
                'charge_priority': 'MEDIUM',
                'mode': 'ECO'
            }
        else:
            # 低电量：最低速行驶
            return {
                'speed_factor': 0.4,
                'charge_priority': 'HIGH',
                'mode': 'SURVIVAL'
            }
```

## 视觉循迹

```python
import cv2
import numpy as np

class EnergyTrackNavigator:
    """储能小车循迹+能量区域识别"""
    def __init__(self):
        self.roi_top = 0.4

    def detect_lane_and_zones(self, frame):
        """同时检测赛道线和能量采集区域"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        h, w = frame.shape[:2]

        # 黑色赛道
        black = cv2.inRange(hsv, (0,0,0), (180,255,60))
        roi_top = int(h * self.roi_top)
        roi = black[roi_top:, :]

        # 赛道中线检测
        deviation = 0
        center_x = w // 2
        for y in range(roi.shape[0]-1, 0, -5):
            row = roi[y, :]
            left = right = None
            for x in range(center_x, 0, -1):
                if row[x] > 0: left = x; break
            for x in range(center_x, w):
                if row[x] > 0: right = x; break
            if left and right:
                center_x = (left + right) // 2
                break

        deviation = (center_x - w/2) / (w/2)

        # 能量采集区域检测(蓝色标记)
        blue = cv2.inRange(hsv, (100,100,100), (130,255,255))
        harvest_zone = cv2.countNonZero(blue) > 1000

        # 无障碍检测
        green = cv2.inRange(hsv, (35,60,60), (85,255,255))
        obstacle = cv2.countNonZero(green) > 500

        return {
            'deviation': deviation,
            'harvest_zone': harvest_zone,
            'obstacle_ahead': obstacle
        }
```

## 主控制逻辑

```python
class EnergyStorageCar:
    def __init__(self):
        self.nav = EnergyTrackNavigator()
        self.energy = EnergyStorageManager()
        self.motors = MotorDriver()  # 复用之前的

        self.Kp = 2.0
        self.Ki = 0.1
        self.Kd = 0.3
        self.integral = 0
        self.last_error = 0

    def run(self):
        camera = cv2.VideoCapture(0)
        base_speed = 0.4  # m/s

        while True:
            ret, frame = camera.read()
            if not ret:
                continue

            # 1. 更新SOC
            soc = self.energy.update_soc()

            # 2. 视觉检测
            info = self.nav.detect_lane_and_zones(frame)

            # 3. 能量策略
            strategy = self.energy.energy_strategy(
                soc, 1.0, info['harvest_zone']
            )

            # 4. 速度计算
            speed = base_speed * strategy['speed_factor']

            # 能量采集区减速
            if info['harvest_zone']:
                speed *= 0.5

            # 5. PID转向
            error = info['deviation']
            self.integral += error * 0.02
            self.integral = np.clip(self.integral, -1, 1)
            deriv = (error - self.last_error) / 0.02
            self.last_error = error

            steer = self.Kp * error + self.Ki * self.integral + self.Kd * deriv

            L = 0.15
            v_l = speed - steer * L
            v_r = speed + steer * L

            self.motors.set_motors(v_l, v_r)

            # 6. 状态输出
            print(f"SOC:{soc:.1%} Mode:{strategy['mode']} "
                  f"Speed:{speed:.2f} Dev:{error:.3f}")

            time.sleep(0.02)
```

## Orange Pi 5 接口分配

| 接口 | 功能 | 说明 |
|------|------|------|
| MIPI CSI-0 | 循迹摄像头 | OV5647 |

| SPI0.1 | 电容电压ADC | SOC监测 |
| PWM0 | 左电机 | TB6612 |
| PWM1 | 右电机 | TB6612 |
| GPIO PA10~13 | 电机方向 | TB6612方向 |

## 调试要点

1. **电容标定**：测量实际容量和ESR
2. **充电效率**：不同采集方式的充电功率测试
3. **SOC-电压曲线**：实测建立精确对照表
4. **能量预算**：计算一圈总消耗 vs 总采集
5. **策略调优**：在采集区降速幅度需要权衡时间成本
