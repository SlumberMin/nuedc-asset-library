# 2017年M题 管道内钢珠运动测量装置 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个测量装置，用于测量管道内钢珠的运动参数。管道倾斜放置，钢珠在管道内滚动或滑动，需要测量钢珠的位置、速度、加速度等运动参数。

### 1.2 关键指标
- 管道长度：约1m，内径略大于钢珠直径
- 测量参数：位置、速度、加速度、运动时间
- 测量精度：位置±1mm，速度±0.01m/s
- 显示方式：实时显示运动曲线

### 1.3 技术难点
1. **非接触测量**：不能影响钢珠运动
2. **高精度定位**：毫米级位置精度
3. **实时性**：需实时跟踪快速运动的钢珠
4. **遮挡问题**：管道壁可能遮挡视线

## 四、视觉检测算法

### 4.1 钢珠检测与追踪
```python
import cv2
import numpy as np
from collections import deque

class SteelBallTracker:
    def __init__(self, pipe_roi, pixels_per_mm=2.0):
        """
        pipe_roi: 管道区域 (x, y, w, h)
        pixels_per_mm: 像素/毫米比例
        """
        self.pipe_roi = pipe_roi
        self.pixels_per_mm = pixels_per_mm
        
        # 钢珠检测参数
        self.ball_color_lower = np.array([0, 0, 40])  # 钢珠颜色（银灰色）
        self.ball_color_upper = np.array([180, 50, 200])
        self.min_radius = 5   # 最小半径（像素）
        self.max_radius = 20  # 最大半径（像素）
        
        # 运动历史
        self.position_history = deque(maxlen=100)
        self.time_history = deque(maxlen=100)
    
    def detect_ball(self, frame):
        """检测钢珠位置"""
        # 裁剪ROI区域
        x, y, w, h = self.pipe_roi
        roi = frame[y:y+h, x:x+w]
        
        # 转换色彩空间
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        
        # 方法1：基于亮度的钢珠检测（钢珠反光）
        # 钢珠通常比管道壁亮
        _, bright_mask = cv2.threshold(gray, 150, 255, cv2.THRESH_BINARY)
        
        # 方法2：基于颜色的检测
        color_mask = cv2.inRange(hsv, self.ball_color_lower, self.ball_color_upper)
        
        # 合并掩码
        mask = cv2.bitwise_or(bright_mask, color_mask)
        
        # 形态学操作
        kernel = np.ones((3, 3), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        
        # 筛选钢珠
        best_ball = None
        best_score = 0
        
        for contour in contours:
            area = cv2.contourArea(contour)
            if area < 50:
                continue
            
            # 拟合圆
            (cx, cy), radius = cv2.minEnclosingCircle(contour)
            
            # 检查半径范围
            if self.min_radius <= radius <= self.max_radius:
                # 计算圆度
                circularity = area / (np.pi * radius**2)
                
                if circularity > 0.7:  # 足够圆
                    score = circularity * area
                    if score > best_score:
                        best_score = score
                        best_ball = {
                            'center': (int(cx) + x, int(cy) + y),  # 转换回原图坐标
                            'radius': int(radius),
                            'circularity': circularity
                        }
        
        return best_ball
    
    def update_tracking(self, ball_info, timestamp):
        """更新追踪状态"""
        if ball_info is not None:
            # 计算管道方向的位置（像素）
            position_px = self.get_pipe_position(ball_info['center'])
            
            # 转换为毫米
            position_mm = position_px / self.pixels_per_mm
            
            self.position_history.append(position_mm)
            self.time_history.append(timestamp)
    
    def get_pipe_position(self, ball_center):
        """将钢珠位置转换为沿管道方向的坐标"""
        # 假设管道水平放置
        # 需要根据实际管道方向调整
        x, y, w, h = self.pipe_roi
        
        # 使用x坐标作为沿管道位置
        pipe_position = ball_center[0] - x
        
        return pipe_position
```

### 4.2 运动参数计算
```python
class MotionAnalyzer:
    def __init__(self, fps=60):
        self.fps = fps
        self.dt = 1.0 / fps
    
    def calculate_velocity(self, positions, timestamps):
        """计算速度"""
        velocities = []
        
        for i in range(1, len(positions)):
            dt = timestamps[i] - timestamps[i-1]
            if dt > 0:
                dx = positions[i] - positions[i-1]
                v = dx / dt
                velocities.append(v)
            else:
                velocities.append(0)
        
        return velocities
    
    def calculate_acceleration(self, velocities, timestamps):
        """计算加速度"""
        accelerations = []
        
        for i in range(1, len(velocities)):
            dt = timestamps[i] - timestamps[i-1]
            if dt > 0:
                dv = velocities[i] - velocities[i-1]
                a = dv / dt
                accelerations.append(a)
            else:
                accelerations.append(0)
        
        return accelerations
    
    def fit_trajectory(self, positions, timestamps):
        """
        拟合运动轨迹
        返回：位置-时间函数参数
        """
        if len(positions) < 3:
            return None
        
        t = np.array(timestamps) - timestamps[0]
        x = np.array(positions)
        
        # 二次多项式拟合: x = a*t² + b*t + c
        coeffs = np.polyfit(t, x, 2)
        
        # 物理含义
        # x = 0.5*a*t² + v0*t + x0
        # a = 2*coeffs[0]  (加速度)
        # v0 = coeffs[1]   (初速度)
        # x0 = coeffs[2]   (初始位置)
        
        acceleration = 2 * coeffs[0]
        initial_velocity = coeffs[1]
        initial_position = coeffs[2]
        
        # 计算拟合误差
        x_fit = np.polyval(coeffs, t)
        rmse = np.sqrt(np.mean((x - x_fit)**2))
        
        return {
            'acceleration': acceleration,
            'initial_velocity': initial_velocity,
            'initial_position': initial_position,
            'coefficients': coeffs,
            'rmse': rmse
        }
    
    def analyze_physics(self, fit_result, pipe_angle_deg=30):
        """分析物理参数"""
        if fit_result is None:
            return None
        
        g = 9810  # 重力加速度 mm/s²
        angle_rad = np.radians(pipe_angle_deg)
        
        # 理论加速度（无摩擦）
        theoretical_acc = g * np.sin(angle_rad)
        
        # 实测加速度
        measured_acc = fit_result['acceleration']
        
        # 计算摩擦系数
        # a = g*sin(θ) - μ*g*cos(θ)
        # μ = (g*sin(θ) - a) / (g*cos(θ))
        if abs(np.cos(angle_rad)) > 0.001:
            friction = (theoretical_acc - measured_acc) / (g * np.cos(angle_rad))
        else:
            friction = 0
        
        return {
            'theoretical_acceleration': theoretical_acc,
            'measured_acceleration': measured_acc,
            'friction_coefficient': friction,
            'energy_loss_percent': (1 - measured_acc/theoretical_acc) * 100 if theoretical_acc != 0 else 0
        }
```

### 4.3 像素标定
```python
def calibrate_pixel_scale(self, frame, known_length_mm=200):
    """
    标定像素与实际尺寸的映射关系
    known_length_mm: 已知长度（如管道上标记的200mm间距）
    """
    # 在图像中标记两个已知点
    # 方法1：自动检测标记点
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    
    # 检测标记点（高亮标记）
    _, mask = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
    contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                    cv2.CHAIN_APPROX_SIMPLE)
    
    if len(contours) >= 2:
        # 获取两个标记点的中心
        centers = []
        for contour in contours[:2]:
            M = cv2.moments(contour)
            if M['m00'] > 0:
                cx = int(M['m10'] / M['m00'])
                cy = int(M['m01'] / M['m00'])
                centers.append((cx, cy))
        
        if len(centers) == 2:
            # 计算像素距离
            pixel_distance = np.sqrt((centers[0][0] - centers[1][0])**2 + 
                                      (centers[0][1] - centers[1][1])**2)
            
            # 计算比例
            pixels_per_mm = pixel_distance / known_length_mm
            
            return pixels_per_mm
    
    return None
```

        self.ax_vel.set_ylabel('速度 (mm/s)')
        self.ax_vel.set_title('速度-时间曲线')
        self.line_vel, = self.ax_vel.plot([], [], 'r-o', markersize=3)
        
        # 加速度-时间曲线
        self.ax_acc = self.axes[2]
        self.ax_acc.set_xlabel('时间 (s)')
        self.ax_acc.set_ylabel('加速度 (mm/s²)')
        self.ax_acc.set_title('加速度-时间曲线')
        self.line_acc, = self.ax_acc.plot([], [], 'g-o', markersize=3)
        
        plt.tight_layout()
    
    def update(self, timestamps, positions, velocities, accelerations):
        """更新曲线"""
        # 位置曲线
        self.line_pos.set_data(timestamps, positions)
        self.ax_pos.relim()
        self.ax_pos.autoscale_view()
        
        # 速度曲线
        if len(velocities) > 0:
            vel_times = timestamps[1:len(velocities)+1]
            self.line_vel.set_data(vel_times, velocities)
            self.ax_vel.relim()
            self.ax_vel.autoscale_view()
        
        # 加速度曲线
        if len(accelerations) > 0:
            acc_times = timestamps[2:len(accelerations)+2]
            self.line_acc.set_data(acc_times, accelerations)
            self.ax_acc.relim()
            self.ax_acc.autoscale_view()
        
        plt.pause(0.01)

### 6.1 完整测量流程
```python
import time

class PipeBallMeasurement:
    def __init__(self):
        # 初始化模块
        self.camera = CameraCapture(width=1280, height=720, fps=60)
        self.tracker = SteelBallTracker(pipe_roi=(100, 300, 1000, 100))
        self.analyzer = MotionAnalyzer(fps=60)
        self.visualizer = MotionVisualizer()
        self.display = DataDisplay()
        
        # 测量状态
        self.measuring = False
        self.start_time = None
        
        # 标定
        self.pixels_per_mm = 2.0  # 需要标定
    
    def calibrate(self):
        """系统标定"""
        print("开始标定...")
        print("请在管道上放置已知长度的标定物")
        
        # 采集标定图像
        frame = self.camera.capture()
        
        # 计算像素比例
        scale = self.tracker.calibrate_pixel_scale(frame, known_length_mm=200)
        
        if scale:
            self.pixels_per_mm = scale
            self.tracker.pixels_per_mm = scale
            print(f"标定完成: {scale:.3f} 像素/毫米")
        else:
            print("标定失败，使用默认值")
    
    def start_measurement(self):
        """开始测量"""
        print("开始测量...")
        print("请释放钢珠")
        
        self.measuring = True
        self.start_time = time.time()
        
        # 清空历史数据
        self.tracker.position_history.clear()
        self.tracker.time_history.clear()
    
    def run(self):
        """主运行循环"""
        # 标定
        self.calibrate()
        
        print("按's'开始测量，按'q'退出")
        
        while True:
            # 获取图像
            frame = self.camera.capture()
            
            # 检测钢珠
            ball = self.tracker.detect_ball(frame)
            
            if ball and self.measuring:
                # 更新追踪
                current_time = time.time() - self.start_time
                self.tracker.update_tracking(ball, current_time)
                
                # 在图像上标记钢珠
                cv2.circle(frame, ball['center'], ball['radius'], (0, 255, 0), 2)
            
            # 显示图像
            cv2.imshow('Pipe Ball Measurement', frame)
            
            # 处理按键
            key = cv2.waitKey(1)
            if key == ord('s'):
                self.start_measurement()
            elif key == ord('q'):
                break
            
            # 如果正在测量，更新分析
            if self.measuring and len(self.tracker.position_history) > 5:
                positions = list(self.tracker.position_history)
                timestamps = list(self.tracker.time_history)
                
                # 计算速度和加速度
                velocities = self.analyzer.calculate_velocity(positions, timestamps)
                accelerations = self.analyzer.calculate_acceleration(velocities, timestamps[1:])
                
                # 拟合轨迹
                fit_result = self.analyzer.fit_trajectory(positions, timestamps)
                
                # 物理分析
                physics_result = self.analyzer.analyze_physics(fit_result)
                
                # 更新显示
                self.visualizer.update(timestamps, positions, velocities, accelerations)
                self.display.display_results(fit_result, physics_result)
                
                # 检测钢珠是否停止
                if len(positions) > 10:
                    recent_positions = positions[-10:]
                    if max(recent_positions) - min(recent_positions) < 1:  # 1mm内
                        print("\n钢珠已停止运动")
                        self.measuring = False
        
        cv2.destroyAllWindows()

if __name__ == '__main__':
    measurement = PipeBallMeasurement()
    measurement.run()
```

```python
class PhotoGateTimer:
    def __init__(self, gate_positions_mm, serial_port='/dev/ttyS3'):
        """
        gate_positions_mm: 各光电门位置 (mm)
        """
        self.gate_positions = gate_positions_mm
        self.ser = serial.Serial(serial_port, 115200, timeout=0.1)
        self.trigger_times = [None] * len(gate_positions_mm)
    
    def read_gates(self):
        """读取光电门状态"""
        if self.ser.in_waiting >= 8:
            data = self.ser.read(8)
            
            # 协议解析
            if data[0] == 0xAA and data[7] == 0x55:
                gate_index = data[1]
                timestamp = (data[2] << 24) | (data[3] << 16) | (data[4] << 8) | data[5]
                
                if gate_index < len(self.trigger_times):
                    self.trigger_times[gate_index] = timestamp / 1000000.0  # 转换为秒
        
        return self.trigger_times
    
    def calculate_gate_velocities(self):
        """根据光电门数据计算各段速度"""
        velocities = []
        
        for i in range(1, len(self.trigger_times)):
            if self.trigger_times[i] is not None and self.trigger_times[i-1] is not None:
                dt = self.trigger_times[i] - self.trigger_times[i-1]
                dx = self.gate_positions[i] - self.gate_positions[i-1]
                
                if dt > 0:
                    v = dx / dt
                    velocities.append(v)
        
        return velocities
```
