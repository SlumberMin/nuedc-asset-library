# 2019 模拟电磁曲射炮 - Orange Pi 5 解决方案

## 题目分析

电磁炮发射弹丸，要求通过视觉识别目标位置，自动调整发射角度和充电量，精确命中目标。核心挑战：弹道计算、电磁发射能量精确控制、视觉测距。

### 目标检测与测距
```python
import cv2
import numpy as np

class TargetDetector:
    """目标检测与距离估计"""
    def __init__(self):
        self.camera_mtx = np.load('cam_mtx.npy')
        self.target_known_width = 0.10  # 目标靶宽度10cm
        self.focal_length = self.camera_mtx[0, 0]

    def detect_target(self, frame):
        """检测红色/黑色靶标"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 红色靶标
        mask = cv2.inRange(hsv, (0,80,80), (10,255,255))
        mask |= cv2.inRange(hsv, (160,80,80), (180,255,255))

        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL,
                                        cv2.CHAIN_APPROX_SIMPLE)
        if not contours:
            return None

        c = max(contours, key=cv2.contourArea)
        if cv2.contourArea(c) < 200:
            return None

        rect = cv2.minAreaRect(c)
        bbox = cv2.boxPoints(rect)
        center = tuple(map(int, rect[0]))
        width_px = min(rect[1])

        # 像素坐标→归一化角偏移
        cx, cy = frame.shape[1]//2, frame.shape[0]//2
        dx_px = center[0] - cx
        dy_px = center[1] - cy

        # 角度偏移(度)
        fx = self.focal_length
        yaw_offset = np.degrees(np.arctan2(dx_px, fx))
        pitch_offset = np.degrees(np.arctan2(dy_px, fx))

        # 距离估计(基于目标已知尺寸)
        distance = (self.target_known_width * fx) / width_px

        return {
            'center': center,
            'distance': distance,      # 米
            'yaw_offset': yaw_offset,  # 度(正=右偏)
            'pitch_offset': pitch_offset,  # 度(正=下偏)
            'width_px': width_px
        }
```

## 弹道计算

```python
class BallisticComputer:
    """弹道解算器 - 考虑重力和弹丸初速"""
    def __init__(self):
        self.g = 9.81
        self.muzzle_height = 0.25  # 炮口离地高度(m)
        self.v0_table = self._build_velocity_table()

    def _build_velocity_table(self):
        """建立充电电压→初速对照表(需标定)"""
        # 格式: {电压: 初速}
        return {
            50: 2.0,   # 50V充电 → 2m/s初速
            100: 3.5,
            150: 5.0,
            200: 6.5,
            250: 8.0,
            300: 9.5
        }

    def compute_launch_params(self, distance, target_height=0.0):
        """
        计算发射参数
        distance: 目标距离(m)
        target_height: 目标高度(m)相对炮口
        返回: (发射角°, 充电电压V, 飞行时间s)
        """
        best_params = None
        min_error = float('inf')

        for v0 in np.linspace(2.0, 10.0, 50):
            for theta in np.linspace(5, 60, 200):
                theta_rad = np.radians(theta)
                vx = v0 * np.cos(theta_rad)
                vy = v0 * np.sin(theta_rad)

                # 飞行时间(到达目标距离)
                t = distance / vx
                # 弹丸高度
                y = vy * t - 0.5 * self.g * t**2 + self.muzzle_height

                error = abs(y - target_height - 0.5)  # 目标中心高度
                if error < min_error:
                    min_error = error
                    if error < 0.05:  # 5cm精度
                        v_charge = self._voltage_for_speed(v0)
                        best_params = (theta, v_charge, t)

        return best_params

    def _voltage_for_speed(self, target_v0):
        """反查充电电压"""
        voltages = sorted(self.v0_table.keys())
        for i in range(len(voltages)-1):
            v1, v2 = voltages[i], voltages[i+1]
            s1, s2 = self.v0_table[v1], self.v0_table[v2]
            if s1 <= target_v0 <= s2:
                ratio = (target_v0 - s1) / (s2 - s1)
                return v1 + ratio * (v2 - v1)
        return voltages[-1]
```

### 充电控制

```python
import gpiod

class CoilDriver:
    """电磁线圈发射控制"""
    def __init__(self):
        self.chip = gpiod.Chip('0')
        self.charge_pin = self.chip.get_line(5)   # PA5: 充电继电器
        self.fire_pin = self.chip.get_line(6)     # PA6: 发射MOSFET
        self.charge_pin.request(consumer='coil', type=gpiod.LINE_REQ_DIR_OUT)
        self.fire_pin.request(consumer='coil', type=gpiod.LINE_REQ_DIR_OUT)

        # ADC读取电容电压
        import spidev
        self.adc = spidev.SpiDev()
        self.adc.open(0, 0)
        self.adc.max_speed_hz = 1000000

    def charge_to_voltage(self, target_v):
        """充电到目标电压"""
        self.charge_pin.set_value(1)  # 开启充电
        while True:
            v_cap = self.read_voltage()
            if v_cap >= target_v * 0.98:
                break
            time.sleep(0.01)
        self.charge_pin.set_value(0)  # 关闭充电

    def fire(self, duration_us=500):
        """发射 - 导通线圈指定时间"""
        self.fire_pin.set_value(1)
        time.sleep(duration_us / 1e6)
        self.fire_pin.set_value(0)

    def read_voltage(self):
        """读取电容电压(通过分压+ADC)"""
        resp = self.adc.xfer2([0x01, 0x80, 0x00])
        adc_val = ((resp[1] & 0x03) << 8) | resp[2]
        voltage = adc_val / 1023.0 * 3.3  # ADC原始
        return voltage * 20.0  # 分压比20:1
```

## 云台控制

```python
class GimbalController:
    """舵机云台 - 俯仰+偏航"""
    def __init__(self):
        # 使用PWM控制两个舵机
        self.pitch = 90.0  # 初始俯仰角
        self.yaw = 90.0    # 初始偏航角
        self.Kp = 0.5      # 视觉伺服增益

    def aim(self, yaw_offset, pitch_offset):
        """根据视觉偏差调整瞄准"""
        self.yaw -= self.Kp * yaw_offset
        self.pitch += self.Kp * pitch_offset

        self.yaw = np.clip(self.yaw, 30, 150)
        self.pitch = np.clip(self.pitch, 30, 120)

        self._set_servo('yaw', self.yaw)
        self._set_servo('pitch', self.pitch)

    def set_pitch_abs(self, angle):
        """设置绝对俯仰角(弹道计算结果)"""
        self.pitch = np.clip(angle, 30, 120)
        self._set_servo('pitch', self.pitch)

    def _set_servo(self, channel, angle):
        duty = 2.5 + angle / 180.0 * 10.0  # 50Hz, 2.5%~12.5%
        # 写入PWM
        period = 20000000  # 20ms
        duty_ns = int(period * duty / 100)
        base = f'/sys/class/pwm/pwmchip0/pwm{0 if channel=="yaw" else 1}'
        with open(f'{base}/duty_cycle', 'w') as f:
            f.write(str(duty_ns))
```

## 主控制流程

```python
class ElectromagneticCannon:
    def __init__(self):
        self.detector = TargetDetector()
        self.ballistic = BallisticComputer()
        self.coil = CoilDriver()
        self.gimbal = GimbalController()
        self.state = 'IDLE'

    def run_cycle(self, frame):
        """单次瞄准-射击循环"""
        # 1. 检测目标
        target = self.detector.detect_target(frame)
        if target is None:
            return "目标未检测到"

        # 2. 视觉伺服瞄准(粗瞄)
        self.gimbal.aim(target['yaw_offset'], target['pitch_offset'])

        # 3. 弹道计算
        params = self.ballistic.compute_launch_params(target['distance'])
        if params is None:
            return "超出射程"
        launch_angle, charge_voltage, flight_time = params

        # 4. 精确调角
        self.gimbal.set_pitch_abs(launch_angle + target['pitch_offset'])

        # 5. 充电
        self.coil.charge_to_voltage(charge_voltage)

        # 6. 确认瞄准后发射
        time.sleep(0.3)  # 等待稳定
        self.coil.fire(duration_us=600)

        return f"发射完成: 距离{target['distance']:.2f}m 角度{launch_angle:.1f}° 电压{charge_voltage:.0f}V"
```

## 调试流程

1. **标定阶段**
   - 相机内参标定(棋盘格)
   - 炮管-相机外参标定
   - 充电电压→初速关系标定(多次射击取平均)

2. **射击精度调优**
   - 静态目标：固定距离逐步调参
   - 多距离测试：建立修正表
   - 重复精度：同一参数连续射击10次统计散布

3. **安全注意**
   - 高压电容放电安全（泄放电阻）
   - 弹丸发射方向安全防护
   - 充电电压上限保护
