# 2017年电赛 - 自动泊车系统 - Orange Pi 5解决方案

## 题目要求

设计制作自动泊车系统，实现：
1. 视觉识别停车位
2. 路径规划
3. 自动泊入车位
4. 侧方停车和倒车入库

## 系统架构

```
┌──────────────┐    CSI/USB     ┌──────────────┐    UART     ┌──────────────┐
│  车载摄像头   │──────────────→│  Orange Pi 5 │────────────→│  STM32主控   │
│  + 超声波     │              │  RK3588S     │             │  电机+舵机   │
└──────────────┘              │  视觉+规划   │             └──────┬───────┘
                               └──────────────┘              ┌────┴────┐
                                                              │ 小车底盘 │
                                                              │ 前轮转向 │
                                                              │ 后轮驱动 │
                                                              └─────────┘
```

## 核心设计

### 1. 车位检测

```python
import cv2
import numpy as np
import time
import struct
import serial
import math

class ParkingVision:
    """自动泊车视觉系统"""
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        # 车位标线颜色（白色/黄色）
        self.line_hsv_lower = np.array([0, 0, 180])
        self.line_hsv_upper = np.array([180, 30, 255])
        
        # 超声波距离（通过UART从STM32读取）
        self.front_dist = 0
        self.rear_dist = 0
        self.left_dist = 0
        self.right_dist = 0
        
    def detect_parking_space(self, frame=None):
        """
        检测停车位
        返回: 车位位置和类型
        """
        if frame is None:
            ret, frame = self.cap.read()
            if not ret:
                return None
        
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 检测车位标线
        mask = cv2.inRange(hsv, self.line_hsv_lower, self.line_hsv_upper)
        
        # 边缘检测
        edges = cv2.Canny(mask, 50, 150)
        
        # 霍夫直线检测
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50, minLineLength=50, maxLineGap=20)
        
        if lines is None:
            return None
        
        # 分析直线特征，判断车位类型
        vertical_lines = []
        horizontal_lines = []
        
        for line in lines:
            x1, y1, x2, y2 = line[0]
            angle = abs(math.degrees(math.atan2(y2-y1, x2-x1)))
            
            if angle < 30 or angle > 150:  # 水平线
                horizontal_lines.append(line[0])
            elif 60 < angle < 120:  # 垂直线
                vertical_lines.append(line[0])
        
        # 判断车位类型
        parking_type = None
        parking_rect = None
        
        if len(vertical_lines) >= 2:
            # 侧方停车位（两条竖线）
            parking_type = 'parallel'
            vertical_lines.sort(key=lambda l: l[0])
            left_line = vertical_lines[0]
            right_line = vertical_lines[-1]
            
            x = min(left_line[0], right_line[0])
            w = abs(right_line[0] - left_line[0])
            y = min(left_line[1], left_line[3], right_line[1], right_line[3])
            h = max(left_line[1], left_line[3], right_line[1], right_line[3]) - y
            
            parking_rect = (x, y, w, h)
            
        elif len(horizontal_lines) >= 2:
            # 垂直停车位（两条横线）
            parking_type = 'perpendicular'
            horizontal_lines.sort(key=lambda l: l[1])
            top_line = horizontal_lines[0]
            bottom_line = horizontal_lines[-1]
            
            x = min(top_line[0], top_line[2], bottom_line[0], bottom_line[2])
            w = max(top_line[0], top_line[2], bottom_line[0], bottom_line[2]) - x
            y = min(top_line[1], bottom_line[1])
            h = abs(bottom_line[1] - top_line[1])
            
            parking_rect = (x, y, w, h)
        
        if parking_rect:
            x, y, w, h = parking_rect
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
            cv2.putText(frame, f"{parking_type} parking", (x, y-10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        
        cv2.imshow('Parking Detection', frame)
        
        return {
            'type': parking_type,
            'rect': parking_rect,
            'lines': lines
        }
    
    def detect_lane_lines(self, frame=None):
        """检测车道线"""
        if frame is None:
            ret, frame = self.cap.read()
            if not ret:
                return None
        
        # 只处理下半部分（近处车道）
        h, w = frame.shape[:2]
        roi = frame[h//2:, :]
        
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.line_hsv_lower, self.line_hsv_upper)
        
        edges = cv2.Canny(mask, 50, 150)
        lines = cv2.HoughLinesP(edges, 1, np.pi/180, 50, minLineLength=30, maxLineGap=15)
        
        left_lines = []
        right_lines = []
        
        if lines is not None:
            for line in lines:
                x1, y1, x2, y2 = line[0]
                if x2 - x1 == 0:
                    continue
                slope = (y2 - y1) / (x2 - x1)
                
                if slope < -0.3:  # 左车道线
                    left_lines.append(line[0])
                elif slope > 0.3:  # 右车道线
                    right_lines.append(line[0])
        
        return left_lines, right_lines
    
    def read_ultrasonic(self, ser):
        """从STM32读取超声波数据"""
        if ser.in_waiting >= 8:
            data = ser.read(8)
            if data[0] == 0xAA and data[7] == 0x55:
                self.front_dist = struct.unpack('<H', data[1:3])[0]
                self.rear_dist = struct.unpack('<H', data[3:5])[0]
                self.left_dist = struct.unpack('<H', data[5:7])[0]
    
    def release(self):
        self.cap.release()
```

### 2. 路径规划

```python
class PathPlanner:
    """
    泊车路径规划器
    支持: 侧方停车、垂直停车
    """
    
    def __init__(self, car_length=300, car_width=150, wheelbase=200):
        self.car_length = car_length  # 车长(mm)
        self.car_width = car_width    # 车宽(mm)
        self.wheelbase = wheelbase    # 轴距(mm)
        self.min_turn_radius = 300    # 最小转弯半径(mm)
        
    def plan_parallel_parking(self, space_length, space_width, car_x, car_y, car_theta):
        """
        侧方停车路径规划
        
        space: 车位尺寸
        car: 当前车辆位姿
        返回: 路径点列表 [(x, y, theta), ...]
        """
        path = []
        
        # 第一阶段：前进到起始位置
        start_x = car_x + space_length + 200
        start_y = car_y
        path.append((start_x, start_y, 0))
        
        # 第二阶段：倒车打满右轮
        # 圆弧路径
        R = self.min_turn_radius
        cx = start_x
        cy = start_y - R
        
        for angle in np.linspace(0, -np.pi/3, 20):
            x = cx + R * math.sin(angle)
            y = cy + R * math.cos(angle)
            theta = angle
            path.append((x, y, theta))
        
        # 第三阶段：倒车打满左轮
        mid_point = path[-1]
        cx2 = mid_point[0]
        cy2 = mid_point[1] + R
        
        for angle in np.linspace(-np.pi/3, 0, 20):
            x = cx2 - R * math.sin(angle)
            y = cy2 + R * math.cos(angle) - R
            theta = angle
            path.append((x, y, theta))
        
        # 第四阶段：调整到车位中心
        final_x = car_x + space_length / 2
        final_y = car_y
        path.append((final_x, final_y, 0))
        
        return path
    
    def plan_perpendicular_parking(self, space_width, space_depth, car_x, car_y):
        """垂直停车路径规划"""
        path = []
        
        # 简单的两段弧线路径
        R = self.min_turn_radius
        
        # 第一段弧线（右转）
        cx = car_x + space_width / 2
        cy = car_y - R
        
        for angle in np.linspace(0, -np.pi/2, 15):
            x = cx + R * math.sin(angle)
            y = cy + R * math.cos(angle)
            path.append((x, y, angle))
        
        # 直线倒车
        last = path[-1]
        for d in np.linspace(0, space_depth - self.car_length, 10):
            path.append((last[0], last[1] - d, -np.pi/2))
        
        return path
    
    def get_steering_angle(self, current, target, wheelbase):
        """根据路径点计算前轮转角"""
        dx = target[0] - current[0]
        dy = target[1] - current[1]
        
        target_angle = math.atan2(dy, dx)
        angle_diff = target_angle - current[2]
        
        # 阿克曼转向模型
        if abs(angle_diff) < 0.01:
            return 0
        
        R = wheelbase / math.tan(abs(angle_diff))
        steering = math.atan(wheelbase / R)
        
        return steering * np.sign(angle_diff)


class ParkingController:
    """泊车主控制器"""
    
    STATE_SEARCHING = 0
    STATE_ALIGNING = 1
    STATE_PARKING = 2
    STATE_ADJUSTING = 3
    STATE_DONE = 4
    
    def __init__(self):
        self.vision = ParkingVision(camera_id=0)
        self.planner = PathPlanner()
        self.ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
        
        self.state = self.STATE_SEARCHING
        self.path = []
        self.path_idx = 0
        
        # 车辆位姿估计
        self.car_x = 0
        self.car_y = 0
        self.car_theta = 0
        
    def send_car_command(self, speed, steering):
        """发送控制指令到STM32"""
        spd = int(np.clip(speed, -100, 100))
        steer = int(np.clip(steering * 180 / np.pi, -45, 45))
        data = struct.pack('<bb', spd, steer)
        self.ser.write(b'\xAA' + data + b'\x55')
    
    def run(self):
        """主控制循环"""
        print("自动泊车系统启动")
        
        try:
            while True:
                # 视觉检测
                parking_info = self.vision.detect_parking_space()
                self.vision.read_ultrasonic(self.ser)
                
                if self.state == self.STATE_SEARCHING:
                    if parking_info and parking_info['type']:
                        print(f"发现{parking_info['type']}停车位")
                        
                        # 规划路径
                        rect = parking_info['rect']
                        if parking_info['type'] == 'parallel':
                            self.path = self.planner.plan_parallel_parking(
                                rect[2], rect[3], self.car_x, self.car_y, self.car_theta)
                        else:
                            self.path = self.planner.plan_perpendicular_parking(
                                rect[2], rect[3], self.car_x, self.car_y)
                        
                        self.path_idx = 0
                        self.state = self.STATE_PARKING
                        print(f"路径规划完成，{len(self.path)}个路径点")
                    else:
                        # 慢速前进搜索车位
                        self.send_car_command(30, 0)
                
                elif self.state == self.STATE_PARKING:
                    if self.path_idx < len(self.path):
                        target = self.path[self.path_idx]
                        
                        # 计算转向角
                        steering = self.planner.get_steering_angle(
                            (self.car_x, self.car_y, self.car_theta),
                            target, self.planner.wheelbase)
                        
                        # 计算速度
                        dist = math.sqrt((target[0]-self.car_x)**2 + (target[1]-self.car_y)**2)
                        speed = 30 if dist > 50 else 15
                        
                        self.send_car_command(-speed, steering)  # 倒车
                        
                        # 更新位姿估计
                        self.car_x += speed * math.cos(self.car_theta) * 0.01
                        self.car_y += speed * math.sin(self.car_theta) * 0.01
                        self.car_theta += steering * 0.01
                        
                        if dist < 30:
                            self.path_idx += 1
                    else:
                        self.state = self.STATE_ADJUSTING
                
                elif self.state == self.STATE_ADJUSTING:
                    # 微调位置
                    front = self.vision.front_dist
                    rear = self.vision.rear_dist
                    
                    if abs(front - rear) < 20:
                        self.state = self.STATE_DONE
                        self.send_car_command(0, 0)
                        print("泊车完成！")
                    elif front > rear:
                        self.send_car_command(-10, 0)
                    else:
                        self.send_car_command(10, 0)
                
                elif self.state == self.STATE_DONE:
                    pass
                
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break
                    
        finally:
            self.send_car_command(0, 0)
            self.vision.release()
            self.ser.close()


def main():
    controller = ParkingController()
    controller.run()
```

## 接线说明

| Orange Pi 5引脚 | 连接目标 | 说明 |
|-----------------|---------|------|
| CSI/USB | 摄像头 | 前方/后方视野 |
| UART3_TX | STM32 RX | 车辆控制指令 |
| UART3_RX | STM32 TX | 超声波数据反馈 |
| 5V | STM32 | 供电 |
| GND | 公共地 | - |

## 调试要点

1. **摄像头安装位置**：影响车位检测视野
2. **超声波布局**：前后左右至少4个超声波
3. **转弯半径标定**：精确测量实际最小转弯半径
4. **位姿估计误差**：长时间累积会偏移，需视觉修正
