## 能量管理系统

```python
import smbus
import time
from datetime import datetime

class EnergyManager:
    """能量管理系统"""
    
    # 电池参数
    BATTERY_CAPACITY = 2000   # mAh
    CELL_FULL = 4.2           # V
    CELL_NOMINAL = 3.7        # V
    CELL_EMPTY = 3.0          # V
    
    # 充电参数
    CHARGE_CURRENT_MAX = 1000  # mA
    CHARGE_VOLTAGE = 8.4       # V (2S)
    
    def __init__(self):
        self.bus = smbus.SMBus(1)
        
        # INA219电流/电压传感器
        self.ina_addr = 0x40
        self._init_ina219()
        
        # 状态
        self.voltage = 0
        self.current = 0
        self.power = 0
        self.soc = 50  # 电量百分比
        self.is_charging = False
        
        # 能量历史记录
        self.energy_log = []
        self.total_energy_used = 0  # mWh
        self.total_energy_charged = 0
    
    def _init_ina219(self):
        """初始化INA219"""
        # 配置寄存器
        config = 0x399F  # 32V, 320mA, 连续采样
        self.bus.write_word_data(self.ina_addr, 0x00, config)
    
    def read_voltage(self):
        """读取总线电压"""
        raw = self.bus.read_word_data(self.ina_addr, 0x02)
        voltage = (raw >> 3) * 4  # mV
        self.voltage = voltage / 1000  # V
        return self.voltage
    
    def read_current(self):
        """读取电流"""
        raw = self.bus.read_word_data(self.ina_addr, 0x04)
        if raw > 32767:
            raw -= 65536
        # 分流电阻 0.1Ω, 增益 320mA
        current = raw * 0.1  # mA (需根据实际分流电阻校准)
        self.current = current
        return self.current
    
    def read_power(self):
        """读取功率"""
        self.read_voltage()
        self.read_current()
        self.power = self.voltage * self.current / 1000  # mW
        return self.power
    
    def estimate_soc(self):
        """估算电量百分比"""
        self.read_voltage()
        
        # 基于电压的简单估算（2S电池）
        total_voltage = self.voltage
        cell_voltage = total_voltage / 2
        
        if cell_voltage >= 4.2:
            self.soc = 100
        elif cell_voltage >= 3.7:
            self.soc = 20 + (cell_voltage - 3.7) / 0.5 * 80
        elif cell_voltage >= 3.3:
            self.soc = 5 + (cell_voltage - 3.3) / 0.4 * 15
        elif cell_voltage >= 3.0:
            self.soc = (cell_voltage - 3.0) / 0.3 * 5
        else:
            self.soc = 0
        
        return int(self.soc)
    
    def log_energy(self):
        """记录能量使用"""
        entry = {
            'time': datetime.now(),
            'voltage': self.voltage,
            'current': self.current,
            'power': self.power,
            'soc': self.soc,
            'charging': self.is_charging
        }
        self.energy_log.append(entry)
    
    def is_low_battery(self, threshold=20):
        """电量是否过低"""
        return self.estimate_soc() < threshold
    
    def is_critical(self, threshold=10):
        """电量是否危急"""
        return self.estimate_soc() < threshold
    
    def get_range_estimate(self, avg_speed_mps=0.5):
        """估算剩余续航(m)"""
        remaining_mah = self.BATTERY_CAPACITY * self.soc / 100
        avg_current = 500  # mA, 典型行驶电流
        remaining_hours = remaining_mah / avg_current
        return remaining_hours * avg_speed_mps * 3600
```

## 无线充电控制

```python
import OPi.GPIO as GPIO

class WirelessCharger:
    """无线充电控制"""
    
    CHARGE_DETECT_PIN = 36  # 检测充电状态
    ALIGN_LED_PIN = 38      # 对准指示LED
    
    def __init__(self):
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.CHARGE_DETECT_PIN, GPIO.IN, pull_up_down=GPIO.PUD_DOWN)
        GPIO.setup(self.ALIGN_LED_PIN, GPIO.OUT)
        
        self.is_aligned = False
        self.charge_power = 0
    
    def detect_receiver(self):
        """检测是否在充电区域"""
        return GPIO.input(self.CHARGE_DETECT_PIN) == GPIO.HIGH
    
    def align_charging_pad(self, camera, line_detector):
        """自动对准充电板"""
        # 视觉引导对准
        while True:
            ret, frame = camera.read()
            if not ret:
                continue
            
            # 检测充电板标记（十字标或二维码）
            pad_center = self._detect_pad_marker(frame)
            
            if pad_center:
                frame_center = (frame.shape[1]//2, frame.shape[0]//2)
                dx = pad_center[0] - frame_center[0]
                dy = pad_center[1] - frame_center[1]
                
                if abs(dx) < 15 and abs(dy) < 15:
                    self.is_aligned = True
                    GPIO.output(self.ALIGN_LED_PIN, GPIO.HIGH)
                    print("充电板对准完成")
                    return True
                
                # 微调位置
                return False, dx, dy
            
            return False, 0, 0
    
    def _detect_pad_marker(self, frame):
        """检测充电板标记"""
        import cv2
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        _, thresh = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        
        contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        if contours:
            c = max(contours, key=cv2.contourArea)
            if cv2.contourArea(c) > 500:
                M = cv2.moments(c)
                if M['m00'] > 0:
                    return (int(M['m10']/M['m00']), int(M['m01']/M['m00']))
        return None
```

## 视觉循迹系统

```python
import cv2
import numpy as np

class LineTracker:
    """视觉循迹系统"""
    
    def __init__(self):
        # 循迹线颜色范围 (HSV)
        self.line_color_lower = np.array([0, 0, 0])      # 黑线
        self.line_color_upper = np.array([180, 255, 50])
        
        # ROI区域
        self.roi_top_ratio = 0.5    # 从画面50%处开始
        self.roi_bottom_ratio = 1.0 # 到画面底部
    
    def detect_line(self, frame):
        """检测循迹线"""
        h, w = frame.shape[:2]
        
        # 裁剪ROI
        y1 = int(h * self.roi_top_ratio)
        y2 = int(h * self.roi_bottom_ratio)
        roi = frame[y1:y2, :]
        
        # 转HSV
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.line_color_lower, self.line_color_upper)
        
        # 形态学处理
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        
        # 分三个区域检测线位置
        h_roi = y2 - y1
        regions = [
            mask[0:h_roi//3, :],          # 近处
            mask[h_roi//3:2*h_roi//3, :],  # 中间
            mask[2*h_roi//3:, :],          # 远处
        ]
        
        line_positions = []
        for region in regions:
            M = cv2.moments(region)
            if M['m00'] > 0:
                cx = int(M['m10'] / M['m00'])
                line_positions.append(cx)
            else:
                line_positions.append(w // 2)  # 默认中间
        
        # 计算偏差和方向
        avg_offset = np.mean(line_positions) - w // 2
        curvature = line_positions[-1] - line_positions[0]  # 远处-近处
        
        return {
            'offset': avg_offset,          # 偏差(像素)
            'curvature': curvature,        # 曲率
            'positions': line_positions,   # 各区域位置
            'detected': len([p for p in line_positions if p != w//2]) >= 2
        }
    
    def detect_intersection(self, mask):
        """检测路口"""
        # 检测水平线
        h, w = mask.shape
        top_half = mask[0:h//2, :]
        
        # 检测横向连续像素
        row_sums = np.sum(top_half, axis=1)
        if np.max(row_sums) > w * 0.6 * 255:
            return True
        return False
    
    def detect_cross(self, mask):
        """检测十字路口"""
        h, w = mask.shape
        mid_h = h // 2
        
        # 横向检测
        horizontal = np.sum(mask[mid_h-10:mid_h+10, :], axis=0)
        has_horizontal = np.sum(horizontal > 128) > w * 0.5
        
        # 纵向检测
        vertical = np.sum(mask[:, w//2-10:w//2+10], axis=1)
        has_vertical = np.sum(vertical > 128) > h * 0.3
        
        return has_horizontal and has_vertical
```

## 运动控制

```python
class CarController:
    """小车运动控制"""
    
    def __init__(self, energy_manager: EnergyManager):
        import smbus
        self.bus = smbus.SMBus(1)
        self.pca_addr = 0x40
        self._init_pca9685()
        
        self.energy = energy_manager
        
        # PID参数
        self.kp = 0.5
        self.ki = 0.01
        self.kd = 0.2
        self.prev_error = 0
        self.integral = 0
        
        self.base_speed = 40  # 基础速度
        self.max_speed = 70   # 最大速度
    
    def _init_pca9685(self):
        self.bus.write_byte_data(self.pca_addr, 0xFE, 0x1E)
        self.bus.write_byte_data(self.pca_addr, 0x00, 0x20)
    
    def set_motor(self, channel, speed):
        """设置电机速度 (-100 ~ 100)"""
        speed = max(-100, min(100, speed))
        if speed >= 0:
            pwm = int(speed / 100 * 4095)
            self.bus.write_word_data(self.pca_addr, 0x06 + channel*4, 0)
            self.bus.write_word_data(self.pca_addr, 0x08 + channel*4, pwm)
        else:
            pwm = int(-speed / 100 * 4095)
            self.bus.write_word_data(self.pca_addr, 0x06 + channel*4, pwm)
            self.bus.write_word_data(self.pca_addr, 0x08 + channel*4, 0)
    
    def follow_line(self, line_info):
        """PID循迹"""
        offset = line_info['offset']
        
        # 自适应速度：电量低时降低速度
        soc = self.energy.estimate_soc()
        if soc < 20:
            speed = self.base_speed * 0.6
        elif soc < 40:
            speed = self.base_speed * 0.8
        else:
            speed = self.base_speed
        
        # PID计算
        self.integral += offset
        self.integral = max(-1000, min(1000, self.integral))
        derivative = offset - self.prev_error
        self.prev_error = offset
        
        correction = self.kp * offset + self.ki * self.integral + self.kd * derivative
        
        left_speed = speed + correction
        right_speed = speed - correction
        
        self.set_motor(0, int(left_speed))
        self.set_motor(1, int(right_speed))
    
    def stop(self):
        self.set_motor(0, 0)
        self.set_motor(1, 0)
    
    def turn_left(self, duration=0.5):
        self.set_motor(0, -30)
        self.set_motor(1, 30)
        time.sleep(duration)
        self.stop()
    
    def turn_right(self, duration=0.5):
        self.set_motor(0, 30)
        self.set_motor(1, -30)
        time.sleep(duration)
        self.stop()
```

## 主控系统

```python
import cv2
import time

class WirelessChargingCar:
    """无线充电小车主控"""
    
    STATE_RUNNING = 0
    STATE_LOW_BATTERY = 1
    STATE_CHARGING = 2
    STATE_CHARGED = 3
    
    def __init__(self):
        self.camera = cv2.VideoCapture(0)
        self.camera.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        self.energy = EnergyManager()
        self.charger = WirelessCharger()
        self.tracker = LineTracker()
        self.car = CarController(self.energy)
        
        self.state = self.STATE_RUNNING
        self.lap_count = 0
        self.total_distance = 0
    
    def run(self):
        """主循环"""
        print("无线充电小车启动")
        print(f"初始电量: {self.energy.estimate_soc()}%")
        
        while True:
            try:
                if self.state == self.STATE_RUNNING:
                    self.run_driving()
                elif self.state == self.STATE_LOW_BATTERY:
                    self.run_return_to_charger()
                elif self.state == self.STATE_CHARGING:
                    self.run_charging()
                elif self.state == self.STATE_CHARGED:
                    self.state = self.STATE_RUNNING
            except KeyboardInterrupt:
                break
        
        self.cleanup()
    
    def run_driving(self):
        """循迹行驶"""
        ret, frame = self.camera.read()
        if not ret:
            return
        
        # 循迹检测
        line_info = self.tracker.detect_line(frame)
        
        if line_info['detected']:
            self.car.follow_line(line_info)
            
            # 检测路口
            hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
            mask = cv2.inRange(hsv, self.tracker.line_color_lower, 
                              self.tracker.line_color_upper)
            
            if self.tracker.detect_cross(mask):
                self.lap_count += 1
                print(f"完成第{self.lap_count}圈")
        
        # 电量监控
        soc = self.energy.estimate_soc()
        if soc < 15:
            print(f"电量过低({soc}%)，返回充电！")
            self.state = self.STATE_LOW_BATTERY
        
        self.energy.log_energy()
    
    def run_return_to_charger(self):
        """返回充电点"""
        self.car.stop()
        print("正在返回充电点...")
        
        # 简单策略：沿循迹线反向行驶到充电点
        # 实际可加入路径规划
        self.car.turn_left(1.0)  # 掉头
        time.sleep(0.5)
        
        # 沿线行驶直到检测到充电板
        while not self.charger.detect_receiver():
            ret, frame = self.camera.read()
            if ret:
                line_info = self.tracker.detect_line(frame)
                if line_info['detected']:
                    self.car.follow_line(line_info)
            time.sleep(0.05)
        
        self.car.stop()
        self.state = self.STATE_CHARGING
    
    def run_charging(self):
        """无线充电"""
        print("开始无线充电...")
        self.car.stop()
        
        # 对准充电板
        while not self.charger.is_aligned:
            result = self.charger.align_charging_pad(self.camera, self.tracker)
            if isinstance(result, tuple) and not result[0]:
                _, dx, dy = result
                # 微调位置
                if abs(dx) > 15:
                    self.car.set_motor(0, 20 if dx > 0 else -20)
                    self.car.set_motor(1, -20 if dx > 0 else 20)
                    time.sleep(0.2)
                    self.car.stop()
            time.sleep(0.1)
        
        # 等待充电完成
        while True:
            soc = self.energy.estimate_soc()
            print(f"充电中... 电量: {soc}%")
            
            if soc >= 90:
                print("充电完成！")
                self.state = self.STATE_CHARGED
                break
            
            time.sleep(5)
    
    def cleanup(self):
        self.car.stop()
        self.camera.release()
        cv2.destroyAllWindows()

if __name__ == '__main__':
    car = WirelessChargingCar()
    car.run()
```

## 能量优化策略

| 策略 | 说明 |
|------|------|
| 自适应速度 | 电量低时自动降低行驶速度 |
| 直线加速 | 直线段适当加速，弯道减速 |
| 制动能量回收 | 减速时电机反接发电（需硬件支持） |
| 充电路径规划 | 电量不足时选择最短路径回充电点 |
| 休眠模式 | 等待时降低系统功耗 |

## 评分要点

| 评分项 | 策略 |
|--------|------|
| 循迹精度 | 视觉+红外双模循迹，PID精调 |
| 能量管理 | INA219精确监测+自适应策略 |
| 无线充电 | 视觉辅助对准，提高充电效率 |
| 续航里程 | 多策略能量优化 |
| 充电效率 | 线圈对准优化+充电管理IC |

## 调试技巧

1. **循迹调参**：先用红外传感器调通PID，再切换视觉
2. **能量校准**：校准INA219分流电阻，提高电流测量精度
3. **充电对准**：增大接收线圈面积，降低对准精度要求
4. **电池保护**：设置低压保护，避免过放损坏电池
