# 2019年J题 模拟电磁曲射炮 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个模拟电磁曲射炮系统。通过电磁线圈加速弹丸（钢珠），使其射出后命中目标。系统需自动测量目标位置，计算发射角度和储能电压，实现自动瞄准和发射。

### 1.2 关键指标
- 弹丸：钢珠（直径约12mm）
- 发射管：电磁线圈加速管
- 目标：距离1~3m处的靶子
- 命中精度：越接近靶心越好
- 发射角度：可调（30°~75°）

### 1.3 技术难点
1. **视觉测距**：准确测量目标距离和方位角
2. **弹道计算**：考虑重力、空气阻力的抛物线弹道
3. **电磁加速**：多级线圈时序控制，优化加速效率
4. **角度控制**：精确控制发射仰角

## 四、视觉定位算法

### 4.1 目标检测与测距
```python
import cv2
import numpy as np

class TargetDetector:
    def __init__(self, camera_matrix, dist_coeffs):
        self.camera_matrix = camera_matrix
        self.dist_coeffs = dist_coeffs
        
        # 目标颜色范围（红色靶子）
        self.lower_red1 = np.array([0, 100, 100])
        self.upper_red1 = np.array([10, 255, 255])
        self.lower_red2 = np.array([160, 100, 100])
        self.upper_red2 = np.array([180, 255, 255])
        
        # 靶子实际尺寸（mm）
        self.target_real_radius = 50  # 靶子半径50mm
    
    def detect_target(self, frame):
        """检测靶子位置"""
        # 畸变校正
        frame = cv2.undistort(frame, self.camera_matrix, self.dist_coeffs)
        
        # HSV色彩空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 红色靶子检测
        mask1 = cv2.inRange(hsv, self.lower_red1, self.upper_red1)
        mask2 = cv2.inRange(hsv, self.lower_red2, self.upper_red2)
        mask = cv2.bitwise_or(mask1, mask2)
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None
        
        # 找最大轮廓（靶子）
        largest = max(contours, key=cv2.contourArea)
        area = cv2.contourArea(largest)
        
        if area < 500:  # 太小，可能是噪声
            return None
        
        # 拟合圆
        (x, y), radius = cv2.minEnclosingCircle(largest)
        
        return {
            'center': (int(x), int(y)),
            'radius': int(radius),
            'area': area
        }
    
    def calculate_distance(self, target_info):
        """根据靶子在图像中的大小计算距离"""
        if target_info is None:
            return None
        
        pixel_radius = target_info['radius']
        
        # 使用相似三角形原理计算距离
        # distance = (real_radius * focal_length) / pixel_radius
        focal_length = self.camera_matrix[0, 0]  # fx
        
        distance_mm = (self.target_real_radius * focal_length) / pixel_radius
        
        return distance_mm / 1000.0  # 转换为米
    
    def calculate_angle(self, target_info, frame_shape):
        """计算靶子相对于光轴的方位角"""
        if target_info is None:
            return None, None
        
        h, w = frame_shape[:2]
        cx, cy = target_info['center']
        
        # 图像中心（光轴）
        img_center_x = w // 2
        img_center_y = h // 2
        
        # 像素偏移
        dx = cx - img_center_x
        dy = cy - img_center_y
        
        # 焦距
        fx = self.camera_matrix[0, 0]
        fy = self.camera_matrix[1, 1]
        
        # 计算角度（弧度转角度）
        azimuth = np.degrees(np.arctan2(dx, fx))  # 水平角
        elevation = np.degrees(np.arctan2(-dy, fy))  # 俯仰角（图像y轴向下）
        
        return azimuth, elevation
```

```python
def calibrate_camera():
    """相机标定"""
    # 准备棋盘格角点
    objp = np.zeros((6*9, 3), np.float32)
    objp[:, :2] = np.mgrid[0:9, 0:6].T.reshape(-1, 2) * 25  # 25mm格子
    
    obj_points = []
    img_points = []
    
    cap = cv2.VideoCapture(0)
    
    print("按空格采集标定图片，按q退出")
    
    while True:
        ret, frame = cap.read()
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        ret_cb, corners = cv2.findChessboardCorners(gray, (9, 6), None)
        
        if ret_cb:
            cv2.drawChessboardCorners(frame, (9, 6), corners, ret_cb)
        
        cv2.imshow('Calibration', frame)
        key = cv2.waitKey(1)
        
        if key == ord(' ') and ret_cb:
            obj_points.append(objp)
            img_points.append(corners)
            print(f"已采集 {len(img_points)} 张")
        elif key == ord('q'):
            break
    
    cap.release()
    cv2.destroyAllWindows()
    
    # 标定
    ret, camera_matrix, dist_coeffs, rvecs, tvecs = cv2.calibrateCamera(
        obj_points, img_points, gray.shape[::-1], None, None)
    
    print(f"标定误差: {ret}")
    print(f"相机矩阵:\n{camera_matrix}")
    print(f"畸变系数:\n{dist_coeffs}")
    
    return camera_matrix, dist_coeffs
```

## 五、弹道计算

### 5.1 弹道模型
```python
import numpy as np
from scipy.optimize import fsolve

class BallisticCalculator:
    def __init__(self):
        self.g = 9.81  # 重力加速度
        self.air_density = 1.225  # 空气密度 kg/m³
        self.drag_coeff = 0.47  # 钢珠阻力系数
        self.ball_radius = 0.006  # 钢珠半径 6mm
        self.ball_mass = 0.0088  # 钢珠质量 约8.8g (12mm钢珠)
        self.muzzle_height = 0.15  # 炮口高度 15cm
    
    def calculate_launch_params(self, distance, height_diff=0):
        """
        计算发射参数
        distance: 目标距离 (m)
        height_diff: 目标与炮口的高度差 (m)
        返回: (发射角度°, 发射速度m/s)
        """
        # 不考虑空气阻力的简化模型
        # 先计算最优角度（约45°附近）
        
        best_angle = 45.0
        min_error = float('inf')
        
        # 遍历角度寻找最优解
        for angle_deg in np.arange(30, 75, 0.5):
            angle_rad = np.radians(angle_deg)
            
            # 所需初速度
            v0 = self._solve_initial_velocity(distance, height_diff, angle_rad)
            
            if v0 is not None and v0 < min_error:
                min_error = v0
                best_angle = angle_deg
        
        # 计算该角度下的所需速度
        best_v0 = self._solve_initial_velocity(
            distance, height_diff, np.radians(best_angle))
        
        return best_angle, best_v0
    
    def _solve_initial_velocity(self, distance, height_diff, angle_rad):
        """求解所需初速度"""
        # 抛物线方程: y = x*tan(θ) - g*x²/(2*v0²*cos²(θ))
        # 当y=height_diff, x=distance时求v0
        
        cos_angle = np.cos(angle_rad)
        sin_angle = np.sin(angle_rad)
        tan_angle = np.tan(angle_rad)
        
        # v0² = g * distance² / (2 * cos²(θ) * (distance*tan(θ) - height_diff))
        denominator = distance * tan_angle - height_diff
        
        if denominator <= 0:
            return None  # 无法命中
        
        v0_squared = self.g * distance**2 / (2 * cos_angle**2 * denominator)
        
        if v0_squared <= 0:
            return None
        
        return np.sqrt(v0_squared)
    
    def calculate_muzzle_energy(self, v0):
        """计算炮口动能"""
        return 0.5 * self.ball_mass * v0**2
    
    def energy_to_voltage(self, energy_joules, efficiency=0.3):
        """
        将所需动能转换为电容电压
        E = 0.5 * C * V² * efficiency
        """
        capacitance = 4e-3  # 4mF (4个1000μF串联)
        
        # V = sqrt(2 * E / (C * efficiency))
        voltage = np.sqrt(2 * energy_joules / (capacitance * efficiency))
        
        # 限幅
        voltage = max(50, min(450, voltage))
        
        return voltage
```

### 5.2 带空气阻力的精确弹道
```python
def simulate_trajectory(self, v0, angle_deg, dt=0.001):
    """模拟弹道轨迹（考虑空气阻力）"""
    angle_rad = np.radians(angle_deg)
    
    # 初始状态
    x = 0
    y = self.muzzle_height
    vx = v0 * np.cos(angle_rad)
    vy = v0 * np.sin(angle_rad)
    
    trajectory = [(x, y)]
    
    while y >= 0:  # 直到落地
        # 速度
        v = np.sqrt(vx**2 + vy**2)
        
        # 空气阻力
        area = np.pi * self.ball_radius**2
        drag_force = 0.5 * self.air_density * v**2 * self.drag_coeff * area
        
        # 加速度
        ax = -drag_force * vx / (v * self.ball_mass)
        ay = -self.g - drag_force * vy / (v * self.ball_mass)
        
        # 更新状态
        vx += ax * dt
        vy += ay * dt
        x += vx * dt
        y += vy * dt
        
        trajectory.append((x, y))
    
    return trajectory
```

## 六、电磁加速控制

### 6.1 多级线圈时序控制
```python
class CoilController:
    def __init__(self, num_stages=3):
        self.num_stages = num_stages
        self.coil_spacing = 0.05  # 线圈间距 5cm
        self.photo_gates = [False] * (num_stages + 1)  # 光电门状态
        
    def calculate_timing(self, target_velocity, stage):
        """
        计算各级线圈触发时机
        target_velocity: 目标速度 m/s
        stage: 当前级数 (0-indexed)
        """
        # 钢珠从当前位置到下一级的时间
        time_to_next = self.coil_spacing / target_velocity
        
        # 提前触发（预充电时间）
        trigger_advance = 0.002  # 2ms提前量
        
        return max(0, time_to_next - trigger_advance)
    
    def on_photo_gate_trigger(self, gate_index, timestamp):
        """光电门触发回调"""
        self.photo_gates[gate_index] = True
        
        if gate_index < self.num_stages:
            # 计算下一级触发时间
            speed = self.estimate_speed(gate_index, timestamp)
            delay = self.calculate_timing(speed, gate_index)
            
            # 延时触发下一级
            self.schedule_trigger(gate_index + 1, delay)
    
    def estimate_speed(self, gate_index, timestamp):
        """根据光电门时间差估算钢珠速度"""
        if gate_index > 0 and hasattr(self, 'last_trigger_time'):
            dt = timestamp - self.last_trigger_time
            if dt > 0:
                speed = self.coil_spacing / dt
                self.last_trigger_time = timestamp
                return speed
        
        self.last_trigger_time = timestamp
        return 10.0  # 默认估计速度 10m/s
```

### 6.2 STM32底层控制
```c
// STM32 电磁炮控制代码
#include "stm32f1xx_hal.h"

#define NUM_COILS 3
#define CAPACITOR_VOLTAGE_MAX 450

// MOSFET驱动引脚
GPIO_TypeDef* coil_ports[NUM_COILS] = {GPIOA, GPIOA, GPIOA};
uint16_t coil_pins[NUM_COILS] = {GPIO_PIN_0, GPIO_PIN_1, GPIO_PIN_2};

// 光电门引脚
GPIO_TypeDef* gate_ports[NUM_COILS+1] = {GPIOB, GPIOB, GPIOB, GPIOB};
uint16_t gate_pins[NUM_COILS+1] = {GPIO_PIN_0, GPIO_PIN_1, GPIO_PIN_2, GPIO_PIN_3};

volatile uint32_t gate_timestamps[NUM_COILS+1];
volatile uint8_t gate_triggered[NUM_COILS+1] = {0};

void fire_coil(int coil_index, uint32_t duration_us) {
    // 开启MOSFET
    HAL_GPIO_WritePin(coil_ports[coil_index], coil_pins[coil_index], GPIO_PIN_SET);
    
    // 延时
    delay_us(duration_us);
    
    // 关闭MOSFET
    HAL_GPIO_WritePin(coil_ports[coil_index], coil_pins[coil_index], GPIO_PIN_RESET);
}

void EXTI0_IRQHandler(void) {
    // 光电门0中断
    if (__HAL_GPIO_EXTI_GET_IT(GPIO_PIN_0)) {
        __HAL_GPIO_EXTI_CLEAR_IT(GPIO_PIN_0);
        gate_timestamps[0] = HAL_GetTick() * 1000 + SysTick->VAL;
        gate_triggered[0] = 1;
    }
}

void charge_capacitor(uint16_t target_voltage) {
    // 控制升压模块充电到目标电压
    while (get_capacitor_voltage() < target_voltage) {
        enable_boost_converter();
        HAL_Delay(10);
    }
    disable_boost_converter();
}

uint16_t get_capacitor_voltage(void) {
    // 读取ADC，计算电容电压
    uint16_t adc_value = HAL_ADC_GetValue(&hadc1);
    // 分压比 100k:1k，ADC参考3.3V
    uint32_t voltage = (uint32_t)adc_value * 3300 / 4096 * 101 / 1;
    return (uint16_t)voltage;
}
```

## 七、主控程序

### 7.1 瞄准与发射流程
```python
import time
import serial

class ElectromagneticMortar:
    def __init__(self):
        # 初始化模块
        self.camera = CameraCapture(width=1280, height=720, fps=30)
        self.detector = TargetDetector(camera_matrix, dist_coeffs)
        self.ballistic = BallisticCalculator()
        self.stm32 = STM32Comm(port='/dev/ttyS3')
        
        # 舵机参数
        self.elevation_servo = ServoController(channel=0, min_angle=30, max_angle=75)
        self.azimuth_servo = ServoController(channel=1, min_angle=-45, max_angle=45)
    
    def aim_and_fire(self):
        """瞄准并发射"""
        print("开始瞄准...")
        
        # 1. 检测目标
        frame = self.camera.capture()
        target = self.detector.detect_target(frame)
        
        if target is None:
            print("未检测到目标！")
            return False
        
        # 2. 计算距离和角度
        distance = self.detector.calculate_distance(target)
        azimuth, elevation = self.detector.calculate_angle(target, frame.shape)
        
        print(f"目标距离: {distance:.2f}m")
        print(f"方位角: {azimuth:.1f}°, 仰角: {elevation:.1f}°")
        
        # 3. 计算发射参数
        launch_angle, launch_speed = self.ballistic.calculate_launch_params(distance)
        
        print(f"发射角度: {launch_angle:.1f}°")
        print(f"发射速度: {launch_speed:.1f}m/s")
        
        # 4. 计算所需电压
        energy = self.ballistic.calculate_muzzle_energy(launch_speed)
        voltage = self.ballistic.energy_to_voltage(energy)
        
        print(f"充电电压: {voltage:.0f}V")
        
        # 5. 调整炮管角度
        self.elevation_servo.set_angle(launch_angle)
        self.azimuth_servo.set_angle(azimuth)
        
        time.sleep(0.5)  # 等待舵机到位
        
        # 6. 充电
        print("充电中...")
        self.stm32.charge_capacitor(int(voltage))
        
        # 7. 发射
        print("发射！")
        self.stm32.fire_all_coils()
        
        return True
    
    def auto_engage(self):
        """自动交战模式"""
        while True:
            # 检测目标
            frame = self.camera.capture()
            target = self.detector.detect_target(frame)
            
            if target:
                print("检测到目标，准备发射...")
                success = self.aim_and_fire()
                
                if success:
                    print("发射完成，等待5秒...")
                    time.sleep(5)
            else:
                time.sleep(0.1)

if __name__ == '__main__':
    mortar = ElectromagneticMortar()
    mortar.auto_engage()
```
