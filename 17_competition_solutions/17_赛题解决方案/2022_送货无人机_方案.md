## 一、题目分析

### 核心要求
无人机从起点起飞，按指定路线飞行至目标区域，识别投递点（靶标），完成精准投递后返回。要求：
- 自主识别起降标志
- 按路线飞行（避障/绕杆）
- 识别靶标颜色/形状/坐标
- 精准投递（物块落点准确）
- 全程自主，不得遥控

### 技术挑战
| 挑战 | 难度 | OPi5 解决方案 |
|------|------|---------------|
| 实时目标检测 | ★★★★★ | NPU 3TOPS 跑 YOLOv5-nano |
| 精准悬停定位 | ★★★★ | 双目/光流 + IMU 融合 |
| 投递点坐标解算 | ★★★★ | 单/双目 PnP 解算 |
| 飞行控制 | ★★★ | MAVROS 串口通信 |

## 五、核心算法

### 5.1 目标检测与坐标解算

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class TargetDetector:
    def __init__(self, model_path, camera_matrix, dist_coeffs):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime()
        self.K = camera_matrix
        self.D = dist_coeffs
        self.class_names = ['red_target', 'blue_target', 'green_target',
                           'landing_mark', 'obstacle']
        
    def detect(self, frame):
        """检测目标并返回像素坐标+类别"""
        img = cv2.resize(frame, (640, 640))
        outputs = self.rknn.inference(inputs=[img])
        detections = self.post_process(outputs, conf_thres=0.5)
        return detections
    
    def get_target_3d(self, detection, depth=None, target_real_size=None):
        """
        计算目标3D坐标
        方法1: 已知目标尺寸的PnP解算
        方法2: 双目视差测距
        方法3: 激光测距辅助
        """
        bbox = detection['bbox']  # [x1,y1,x2,y2]
        cls = detection['class']
        
        # 方法1: 单目PnP (已知靶标尺寸)
        if target_real_size:
            corners = self.get_target_corners(bbox)
            # 世界坐标 (靶标4个角点)
            obj_pts = np.array([
                [-target_real_size/2, -target_real_size/2, 0],
                [ target_real_size/2, -target_real_size/2, 0],
                [ target_real_size/2,  target_real_size/2, 0],
                [-target_real_size/2,  target_real_size/2, 0]
            ], dtype=np.float32)
            
            success, rvec, tvec = cv2.solvePnP(
                obj_pts, corners.astype(np.float32),
                self.K, self.D
            )
            return tvec.flatten()  # [X, Y, Z] 相机坐标系
        
        return None
    
    def get_center_offset(self, detection, frame_shape):
        """计算目标中心相对于图像中心的偏移 (像素)"""
        bbox = detection['bbox']
        cx = (bbox[0] + bbox[2]) / 2
        cy = (bbox[1] + bbox[3]) / 2
        h, w = frame_shape[:2]
        offset_x = cx - w / 2
        offset_y = cy - h / 2
        return offset_x, offset_y
```

            (2, 0, 1.5),    # 航点1
            (4, 2, 1.5),    # 航点2 绕杆
            (6, 0, 1.5),    # 航点3
            (8, 0, 1.5),    # 投递区域
        ]
        self.current_wp = 0
        
        # PID 参数
        self.pid_x = PIDController(kp=1.0, ki=0.01, kd=0.3)
        self.pid_y = PIDController(kp=1.0, ki=0.01, kd=0.3)
        self.pid_z = PIDController(kp=1.5, ki=0.02, kd=0.5)
        self.pid_yaw = PIDController(kp=2.0, ki=0.0, kd=0.5)
        
    def navigate_to_waypoint(self, target):
        """飞向指定航点"""
        target_pose = PoseStamped()
        target_pose.pose.position.x = target[0]
        target_pose.pose.position.y = target[1]
        target_pose.pose.position.z = target[2]
        self.local_pos_pub.publish(target_pose)
    
    def precision_approach(self, target_3d, offset_2d):
        """
        精准接近投递点
        target_3d: 目标3D坐标 (米)
        offset_2d: 像素偏移
        """
        # 使用视觉伺服 (Visual Servoing)
        cmd = Twist()
        cmd.linear.x = self.pid_x.update(target_3d[0])
        cmd.linear.y = self.pid_y.update(target_3d[1])
        cmd.linear.z = self.pid_z.update(target_3d[2] - 0.5)  # 保持0.5m高度
        cmd.angular.z = self.pid_yaw.update(offset_2d[0])  # 航向修正
        return cmd

class DeliveryController:
    def __init__(self, servo_pin=15):
        self.servo_pin = servo_pin
        self.setup_gpio()
        
    def setup_gpio(self):
        """初始化舵机PWM"""
        import OPi.GPIO as GPIO
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.servo_pin, GPIO.OUT)
        self.pwm = GPIO.PWM(self.servo_pin, 50)  # 50Hz
        self.pwm.start(0)
    
    def open_gate(self):
        """打开投递门"""
        self.set_angle(90)
        
    def close_gate(self):
        """关闭投递门"""
        self.set_angle(0)
    
    def set_angle(self, angle):
        """设置舵机角度"""
        duty = 2.5 + (angle / 180.0) * 10
        self.pwm.ChangeDutyCycle(duty)
    
    def deliver(self, hold_time=0.5):
        """执行投递动作"""
        self.open_gate()
        time.sleep(hold_time)
        self.close_gate()

### 6.1 降落标志检测

```python
class LandingDetector:
    """检测地面降落标志"""
    
    def detect_landing_mark(self, frame):
        """
        检测十字/圆圈降落标志
        返回: 中心像素坐标, 偏移角度
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        _, thresh = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        
        # 检测轮廓
        contours, _ = cv2.findContours(thresh, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        
        for cnt in contours:
            area = cv2.contourArea(cnt)
            if area < 1000:
                continue
                
            # 检测十字形
            approx = cv2.approxPolyDP(cnt, 0.02 * cv2.arcLength(cnt, True), True)
            if len(approx) == 12:  # 十字形12个顶点
                M = cv2.moments(cnt)
                cx = int(M['m10'] / M['m00'])
                cy = int(M['m01'] / M['m00'])
                return (cx, cy), 0
            
            # 检测圆形
            (cx, cy), radius = cv2.minEnclosingCircle(cnt)
            circularity = area / (np.pi * radius**2)
            if circularity > 0.85:
                return (int(cx), int(cy)), 0
        
        return None, None
    
    def calc_offset(self, mark_center, frame_size):
        """计算降落偏移"""
        h, w = frame_size[:2]
        dx = mark_center[0] - w // 2
        dy = mark_center[1] - h // 2
        return dx, dy  # 像素偏移
```

### 6.2 靶标识别 (YOLOv5 + NPU)

```python
class TargetClassifier:
    """识别投递靶标颜色和位置"""
    
    COLORS = {
        'red': {'lower': (0, 100, 100), 'upper': (10, 255, 255)},
        'blue': {'lower': (100, 100, 100), 'upper': (130, 255, 255)},
        'green': {'lower': (35, 100, 100), 'upper': (85, 255, 255)},
    }
    
    def classify_target(self, roi):
        """识别靶标颜色"""
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        max_score = 0
        best_color = None
        
        for color_name, bounds in self.COLORS.items():
            mask = cv2.inRange(hsv, bounds['lower'], bounds['upper'])
            score = cv2.countNonZero(mask) / mask.size
            if score > max_score:
                max_score = score
                best_color = color_name
        
        return best_color, max_score
    
    def get_delivery_zone(self, frame, yolo_detections):
        """
        综合YOLO检测和颜色分析，确定投递区域
        返回: [(color, center_3d, confidence), ...]
        """
        zones = []
        for det in yolo_detections:
            x1, y1, x2, y2 = det['bbox']
            roi = frame[y1:y2, x1:x2]
            color, score = self.classify_target(roi)
            if color and score > 0.3:
                zones.append({
                    'color': color,
                    'center': ((x1+x2)//2, (y1+y2)//2),
                    'confidence': score * det['conf'],
                    'bbox': det['bbox']
                })
        return zones
```

### 7.1 UART 直连方案

```python
import serial
import struct
import time

class MAVLinkDirect:
    """OPi5 直接通过 UART 与 PX4 飞控通信"""
    
    def __init__(self, port='/dev/ttyS2', baud=921600):
        self.ser = serial.Serial(port, baud, timeout=0.1)
        self.msg_seq = 0
        
    def send_heartbeat(self):
        """发送心跳包"""
        msg = self.pack_msg(0, struct.pack('<BBBBB', 
            0, 0, 0, 0, 0))  # type, autopilot, base_mode, custom_mode, system_status
        self.ser.write(msg)
    
    def send_setpoint_position(self, x, y, z, yaw=0):
        """发送目标位置 (NED坐标系)"""
        payload = struct.pack('<fffffBB',
            x, y, z, yaw,  # x, y, z, yaw
            0,              # type_mask
            0, 0)           # target_system, target_component
        msg = self.pack_msg(84, payload)  # SET_POSITION_TARGET_LOCAL_NED
        self.ser.write(msg)
    
    def send_command_long(self, command, param1=0, param2=0):
        """发送命令 (解锁/锁定/模式切换)"""
        payload = struct.pack('<fffffffHBBB',
            param1, param2, 0, 0, 0, 0, 0,
            command, 0, 0, 0)  # target_system, target_component, confirmation
        msg = self.pack_msg(76, payload)
        self.ser.write(msg)
    
    def arm(self):
        """解锁电机"""
        self.send_command_long(400, 1)  # MAV_CMD_COMPONENT_ARM_DISARM
        
    def set_mode(self, mode='OFFBOARD'):
        """切换飞行模式"""
        modes = {
            'OFFBOARD': 216,
            'AUTO.LAND': 21,
            'AUTO.RTL': 20,
        }
        if mode in modes:
            self.send_command_long(176, modes[mode])
    
    def pack_msg(self, msg_id, payload):
        """打包 MAVLink v2 消息"""
        header = 0xFD
        length = len(payload)
        # 简化版打包，实际需完整 CRC 计算
        msg = struct.pack('<BBBBBBH', header, length, 0, 0, 0, 0, msg_id)
        msg += payload
        msg += struct.pack('<H', 0)  # CRC placeholder
        return msg
```

## 八、完整主程序

```python
#!/usr/bin/env python3
"""
送货无人机主程序
Orange Pi 5 + PX4 + ROS2
"""

import rclpy
from rclpy.node import Node
import cv2
import numpy as np
import time
from enum import Enum

from sensor_msgs.msg import Image
from geometry_msgs.msg import PoseStamped, Twist
from mavros_msgs.msg import State
from std_msgs.msg import String
from cv_bridge import CvBridge

from target_detector import TargetDetector
from navigator import Navigator
from delivery_controller import DeliveryController

class MissionState(Enum):
    IDLE = 0
    TAKEOFF = 1
    NAVIGATE = 2
    SEARCH_TARGET = 3
    APPROACH = 4
    DELIVER = 5
    RETURN = 6
    LANDING = 7
    DONE = 8

class DeliveryDrone(Node):
    def __init__(self):
        super().__init__('delivery_drone')
        
        # 初始化组件
        self.bridge = CvBridge()
        self.detector = TargetDetector('delivery.rknn')
        self.navigator = Navigator()
        self.delivery = DeliveryController()
        
        # 状态
        self.state = MissionState.IDLE
        self.target_color = 'red'  # 题目指定目标
        self.current_target = None
        self.position = (0, 0, 0)
        
        # 订阅/发布
        self.front_cam_sub = self.create_subscription(
            Image, '/camera_front', self.front_cam_cb, 10)
        self.down_cam_sub = self.create_subscription(
            Image, '/camera_down', self.down_cam_cb, 10)
        self.state_sub = self.create_subscription(
            State, '/mavros/state', self.state_cb, 10)
        
        # 定时器
        self.main_timer = self.create_timer(0.033, self.main_loop)  # 30Hz
        
        self.get_logger().info('送货无人机系统启动')
    
    def main_loop(self):
        """主控制循环"""
        if self.state == MissionState.IDLE:
            self.handle_idle()
        elif self.state == MissionState.TAKEOFF:
            self.handle_takeoff()
        elif self.state == MissionState.NAVIGATE:
            self.handle_navigate()
        elif self.state == MissionState.SEARCH_TARGET:
            self.handle_search()
        elif self.state == MissionState.APPROACH:
            self.handle_approach()
        elif self.state == MissionState.DELIVER:
            self.handle_deliver()
        elif self.state == MissionState.RETURN:
            self.handle_return()
        elif self.state == MissionState.LANDING:
            self.handle_landing()
    
    def handle_idle(self):
        """待机状态"""
        if self.navigator.is_connected():
            self.state = MissionState.TAKEOFF
    
    def handle_takeoff(self):
        """起飞到指定高度"""
        self.navigator.arm()
        time.sleep(1)
        self.navigator.set_mode('OFFBOARD')
        self.navigator.goto(0, 0, 1.5)  # 起飞到1.5m
        time.sleep(3)
        self.state = MissionState.NAVIGATE
    
    def handle_navigate(self):
        """沿路线飞行"""
        if self.navigator.reached_waypoint():
            if self.navigator.has_more_waypoints():
                self.navigator.next_waypoint()
            else:
                self.state = MissionState.SEARCH_TARGET
    
    def handle_search(self):
        """搜索投递靶标"""
        if self.front_frame is not None:
            detections = self.detector.detect(self.front_frame)
            for det in detections:
                if det['class_name'] == f'{self.target_color}_target':
                    self.current_target = det
                    self.state = MissionState.APPROACH
                    break
        
        # 搜索机动: 小范围平移
        self.navigator.search_pattern()
    
    def handle_approach(self):
        """精准接近靶标"""
        if self.current_target:
            target_3d = self.detector.get_target_3d(self.current_target)
            if target_3d is not None:
                if np.linalg.norm(target_3d) < 0.3:  # 30cm内
                    self.state = MissionState.DELIVER
                else:
                    self.navigator.precision_approach(target_3d)
        else:
            self.state = MissionState.SEARCH_TARGET
    
    def handle_deliver(self):
        """执行投递"""
        self.get_logger().info('执行投递...')
        self.delivery.deliver()
        time.sleep(1)
        self.state = MissionState.RETURN
    
    def handle_return(self):
        """返回起点"""
        self.navigator.goto(0, 0, 1.5)
        if self.navigator.reached_waypoint():
            self.state = MissionState.LANDING
    
    def handle_landing(self):
        """降落"""
        if self.down_frame is not None:
            mark_center, _ = self.landing_detector.detect_landing_mark(self.down_frame)
            if mark_center:
                dx, dy = self.landing_detector.calc_offset(mark_center, self.down_frame.shape)
                if abs(dx) < 20 and abs(dy) < 20:
                    self.navigator.land()
                    self.state = MissionState.DONE
                else:
                    self.navigator.adjust_position(dx, dy)

def main():
    rclpy.init()
    node = DeliveryDrone()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

## 十、总结

| 指标 | 目标值 | 实现方法 |
|------|--------|---------|
| 检测帧率 | ≥30fps | NPU INT8量化 |
| 定位精度 | ±5cm | 激光测距+PnP |
| 投递精度 | ±10cm | 视觉伺服闭环 |
| 全程时间 | <3min | 优化航点路线 |
| 系统重量 | <2kg | 轻量化选型 |

**关键优势**：OPi5 的 6TOPS NPU 可同时运行目标检测和视觉定位，RK3588S 的 8 核 CPU 足够处理 ROS2 节点通信和飞控指令，无需额外计算模块。
