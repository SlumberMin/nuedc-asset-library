# 2020年电赛 - 无线运动传感器节点设计 方案

> 基于 Orange Pi 5 + IMU传感器 + 无线传输的运动检测系统

## 一、赛题分析

设计一个无线运动传感器节点，能采集加速度、角速度等运动数据，通过无线方式传输至上位机进行显示和分析。关键指标：数据采样率、传输可靠性、功耗控制。

## 二、系统架构

```
┌─────────────┐    SPI/I2C    ┌──────────────┐    WiFi/2.4G    ┌─────────────┐

```

```python
#!/usr/bin/env python3
"""IMU数据采集与运动检测"""
import smbus2
import time
import numpy as np
import json
import socket

class IMUNode:
    MPU6050_ADDR = 0x68
    REG_PWR_MGMT = 0x6B
    REG_ACCEL_X  = 0x3B
    REG_GYRO_X   = 0x43

    def __init__(self, bus_num=2, sample_rate=100):
        self.bus = smbus2.SMBus(bus_num)
        self.sample_rate = sample_rate
        self._init_sensor()
        # 滤波参数
        self.alpha = 0.98  # 互补滤波系数
        self.roll = 0.0
        self.pitch = 0.0

    def _init_sensor(self):
        self.bus.write_byte_data(self.MPU6050_ADDR, self.REG_PWR_MGMT, 0x00)
        time.sleep(0.1)
        # 设置采样率: 1kHz / (1 + divider)
        divider = int(1000 / self.sample_rate) - 1
        self.bus.write_byte_data(self.MPU6050_ADDR, 0x19, divider)
        # 加速度量程 ±4g
        self.bus.write_byte_data(self.MPU6050_ADDR, 0x1C, 0x08)
        # 陀螺仪量程 ±500°/s
        self.bus.write_byte_data(self.MPU6050_ADDR, 0x1B, 0x08)
        # DLPF 低通滤波
        self.bus.write_byte_data(self.MPU6050_ADDR, 0x1A, 0x03)

    def read_raw(self, reg):
        high = self.bus.read_byte_data(self.MPU6050_ADDR, reg)
        low = self.bus.read_byte_data(self.MPU6050_ADDR, reg + 1)
        val = (high << 8) | low
        return val - 65536 if val > 32767 else val

    def read_accel(self):
        ax = self.read_raw(self.REG_ACCEL_X) / 8192.0      # ±4g
        ay = self.read_raw(self.REG_ACCEL_X + 2) / 8192.0
        az = self.read_raw(self.REG_ACCEL_X + 4) / 8192.0
        return ax, ay, az

    def read_gyro(self):
        gx = self.read_raw(self.REG_GYRO_X) / 65.5         # ±500°/s
        gy = self.read_raw(self.REG_GYRO_X + 2) / 65.5
        gz = self.read_raw(self.REG_GYRO_X + 4) / 65.5
        return gx, gy, gz

    def complementary_filter(self, dt):
        ax, ay, az = self.read_accel()
        gx, gy, gz = self.read_gyro()
        # 加速度计角度
        accel_roll = np.degrees(np.arctan2(ay, az))
        accel_pitch = np.degrees(np.arctan2(-ax, np.sqrt(ay**2 + az**2)))
        # 互补滤波
        self.roll = self.alpha * (self.roll + gx * dt) + (1 - self.alpha) * accel_roll
        self.pitch = self.alpha * (self.pitch + gy * dt) + (1 - self.alpha) * accel_pitch
        return self.roll, self.pitch

    def detect_motion(self, threshold=0.3):
        """检测运动状态：静止/走动/跑步/跌倒"""
        ax, ay, az = self.read_accel()
        gx, gy, gz = self.read_gyro()
        acc_mag = np.sqrt(ax**2 + ay**2 + az**2)
        gyro_mag = np.sqrt(gx**2 + gy**2 + gz**2)

        if acc_mag > 3.0:   # 3g以上
            return "FALL"   # 跌倒
        elif acc_mag > 1.5 or gyro_mag > 100:
            return "RUN"    # 跑步
        elif acc_mag > 1.1 or gyro_mag > 20:
            return "WALK"   # 走动
        else:
            return "STILL"  # 静止

class WiFiTransmitter:
    def __init__(self, server_ip="192.168.1.100", port=8888):
        self.addr = (server_ip, port)
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)

    def send(self, data_dict):
        msg = json.dumps(data_dict).encode()
        self.sock.sendto(msg, self.addr)

def main():
    imu = IMUNode(bus_num=2, sample_rate=100)
    tx = WiFiTransmitter()
    dt = 1.0 / imu.sample_rate

    print("IMU节点启动，采样率: {}Hz".format(imu.sample_rate))

    while True:
        t0 = time.time()
        roll, pitch = imu.complementary_filter(dt)
        acc = imu.read_accel()
        gyro = imu.read_gyro()
        motion = imu.detect_motion()

        packet = {
            "ts": time.time(),
            "acc": [round(a, 4) for a in acc],
            "gyro": [round(g, 4) for g in gyro],
            "roll": round(roll, 2),
            "pitch": round(pitch, 2),
            "motion": motion
        }
        tx.send(packet)

        # 控制采样率
        elapsed = time.time() - t0
        if elapsed < dt:
            time.sleep(dt - elapsed)

if __name__ == "__main__":
    main()
```

### 4.2 上位机接收与可视化（Orange Pi 5同时做上位机）

```python
#!/usr/bin/env python3
"""UDP接收端 + 实时可视化"""
import socket
import json
import matplotlib.pyplot as plt
from collections import deque
import threading

class DataReceiver:
    def __init__(self, port=8888):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock.bind(("0.0.0.0", port))
        self.data = deque(maxlen=500)
        self.running = True

    def listen(self):
        while self.running:
            msg, addr = self.sock.recvfrom(1024)
            pkt = json.loads(msg.decode())
            self.data.append(pkt)

    def get_latest(self, n=100):
        return list(self.data)[-n:]

def plot_realtime(receiver):
    plt.ion()
    fig, (ax1, ax2) = plt.subplots(2, 1)
    while True:
        data = receiver.get_latest(100)
        if not data:
            plt.pause(0.1)
            continue
        rolls = [d["roll"] for d in data]
        pitches = [d["pitch"] for d in data]
        ax1.clear(); ax2.clear()
        ax1.plot(rolls, label="Roll")
        ax2.plot(pitches, label="Pitch", color="orange")
        ax1.legend(); ax2.legend()
        ax1.set_ylabel("Roll (°)")
        ax2.set_ylabel("Pitch (°)")
        ax2.set_xlabel("Sample")
        plt.pause(0.1)

if __name__ == "__main__":
    recv = DataReceiver(port=8888)
    t = threading.Thread(target=recv.listen, daemon=True)
    t.start()
    plot_realtime(recv)
```

### 4.3 运动模式实时识别（NPU加速版）

```python
#!/usr/bin/env python3
"""基于NPU的运动模式分类"""
import rknnlite
import numpy as np

class MotionClassifier:
    """使用RKNN NPU加速运动分类"""
    def __init__(self, model_path="motion_classifier.rknn"):
        self.rknn = rknnlite.RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime()
        self.labels = ["STILL", "WALK", "RUN", "FALL", "UPSTAIRS", "DOWNSTAIRS"]

    def classify(self, imu_window):
        """输入: (128, 6)的IMU滑动窗口 → 分类结果"""
        input_data = imu_window.astype(np.float32).reshape(1, 128, 6)
        outputs = self.rknn.inference(inputs=[input_data])
        pred_idx = np.argmax(outputs[0])
        return self.labels[pred_idx], float(outputs[0][0][pred_idx])
```

## 五、优化策略

1. **低功耗**: 动态调整采样率，静止时降至10Hz
2. **数据压缩**: 差分编码减少传输数据量
3. **丢包重传**: 关键数据(跌倒检测)增加冗余发送
4. **温度补偿**: 记录温度数据校正陀螺仪漂移

## 六、评分要点

| 项目 | 方案对应 |
|------|---------|
| 数据采集精度 | BMI160 + 校准 + DLPF |
| 无线传输可靠性 | UDP + 重传机制 |
| 运动识别准确率 | NPU推理 + 互补滤波 |
| 实时性 | 100Hz采样 + WiFi直传 |
| 功耗 | 动态采样率 + 休眠模式 |
