# 2019年全国大学生电子设计竞赛 - 巡线机器人

## 题目分析
设计一个巡线机器人，能够在包含直角弯、锐角弯、十字路口、断线等复杂路径上自主循迹行驶，要求高速稳定通过。

## 系统架构

```
┌─────────────┐  CSI/USB  ┌──────────────────┐   GPIO/I2C  ┌──────────────┐
│  前视摄像头   │──────────→│   Orange Pi 5    │────────────→│  电机驱动     │
│  + 下视摄像头 │          │  视觉循迹+预测    │             │  (双H桥)     │
```

### 核心代码

#### 1. 双摄像头循迹 (dual_camera_tracker.py)
```python
import cv2
import numpy as np
from dataclasses import dataclass
from enum import Enum

class PathType(Enum):
    STRAIGHT = "straight"
    LEFT_TURN = "left_turn"
    RIGHT_TURN = "right_turn"
    SHARP_LEFT = "sharp_left"
    SHARP_RIGHT = "sharp_right"
    CROSS = "cross"
    T_JUNCTION = "t_junction"
    GAP = "gap"           # 断线
    END = "end"           # 终点

@dataclass
class TrackResult:
    path_type: PathType
    offset: float         # 归一化偏移(-1到1)
    angle: float          # 线角度(度)
    curvature: float      # 曲率
    confidence: float     # 置信度
    speed_factor: float   # 速度因子(0-1)

class DualCameraTracker:
    """双摄像头循迹系统"""

    def __init__(self, config):
        # 前视相机(远距离前瞻)
        self.forward_roi = config.get('forward_roi', (0.0, 0.5))  # 上半部分
        # 下视相机(近距离精确)
        self.down_roi = config.get('down_roi', (0.3, 1.0))       # 下半部分

        self.threshold = config.get('threshold', 100)
        self.line_width = config.get('line_width', 30)

        # 路径历史(用于平滑)
        self.offset_history = []
        self.angle_history = []

    def process(self, forward_frame, down_frame) -> TrackResult:
        """融合双摄像头结果"""
        # 前视: 检测远距离路径
        fwd_result = self._analyze_frame(forward_frame, self.forward_roi, "forward")
        # 下视: 检测近距离路径
        dwn_result = self._analyze_frame(down_frame, self.down_roi, "down")

        # 融合策略
        if fwd_result is None and dwn_result is None:
            return TrackResult(PathType.GAP, 0, 0, 0, 0, 0.3)

        if fwd_result is not None and dwn_result is not None:
            # 加权融合
            offset = 0.3 * fwd_result['offset'] + 0.7 * dwn_result['offset']
            angle = 0.3 * fwd_result['angle'] + 0.7 * dwn_result['angle']
            curvature = fwd_result.get('curvature', 0)
        elif fwd_result is not None:
            offset = fwd_result['offset']
            angle = fwd_result['angle']
            curvature = fwd_result.get('curvature', 0)
        else:
            offset = dwn_result['offset']
            angle = dwn_result['angle']
            curvature = dwn_result.get('curvature', 0)

        # 平滑处理
        self.offset_history.append(offset)
        if len(self.offset_history) > 5:
            self.offset_history.pop(0)
        offset = np.mean(self.offset_history)

        # 路径分类
        path_type = self._classify_path(offset, angle, curvature)
        speed_factor = self._calc_speed_factor(path_type, curvature)

        return TrackResult(
            path_type=path_type,
            offset=offset,
            angle=angle,
            curvature=curvature,
            confidence=1.0 if dwn_result else 0.6,
            speed_factor=speed_factor
        )

    def _analyze_frame(self, frame, roi_range, camera_type):
        """分析单帧"""
        if frame is None:
            return None

        h, w = frame.shape[:2]
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

        # ROI
        y1 = int(h * roi_range[0])
        y2 = int(h * roi_range[1])
        roi = gray[y1:y2, :]

        # 自适应二值化
        binary = cv2.adaptiveThreshold(
            roi, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 15, 3)

        # 形态学
        kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (3, 3))
        binary = cv2.morphologyEx(binary, cv2.MORPH_OPEN, kernel)

        # 多行采样
        roi_h, roi_w = binary.shape
        offsets = []
        for i in range(8):
            y = int(roi_h * (i + 1) / 9)
            row = binary[y, :]
            pixels = np.where(row > 0)[0]
            if len(pixels) > 3:
                center = np.mean(pixels)
                offsets.append((center - roi_w / 2) / (roi_w / 2))

        if not offsets:
            return None

        offset = np.mean(offsets)

        # 角度估计
        if len(offsets) >= 3:
            x = np.arange(len(offsets))
            angle = np.polyfit(x, offsets, 1)[0] * 30  # 近似角度
        else:
            angle = 0

        # 曲率估计
        if len(offsets) >= 4:
            coeffs = np.polyfit(np.arange(len(offsets)), offsets, 2)
            curvature = abs(coeffs[0]) * 100
        else:
            curvature = 0

        return {'offset': offset, 'angle': angle, 'curvature': curvature}

    def _classify_path(self, offset, angle, curvature):
        """路径分类"""
        if curvature < 0.5 and abs(angle) < 5:
            return PathType.STRAIGHT
        elif curvature > 3.0:
            if angle < 0:
                return PathType.SHARP_LEFT
            else:
                return PathType.SHARP_RIGHT
        elif abs(angle) > 5:
            if angle < 0:
                return PathType.LEFT_TURN
            else:
                return PathType.RIGHT_TURN
        return PathType.STRAIGHT

    def _calc_speed_factor(self, path_type, curvature):
        """计算速度因子"""
        factors = {
            PathType.STRAIGHT: 1.0,
            PathType.LEFT_TURN: 0.7,
            PathType.RIGHT_TURN: 0.7,
            PathType.SHARP_LEFT: 0.4,
            PathType.SHARP_RIGHT: 0.4,
            PathType.CROSS: 0.5,
            PathType.GAP: 0.3,
        }
        base = factors.get(path_type, 0.5)
        # 根据曲率进一步调整
        return max(0.2, base - curvature * 0.05)
```

#### 2. 速度规划 (speed_planner.py)
```python
from dataclasses import dataclass

@dataclass
class SpeedCommand:
    left_speed: float
    right_speed: float
    path_type: str

class SpeedPlanner:
    """速度规划器"""

    def __init__(self, config):
        self.base_speed = config.get('base_speed', 60)
        self.max_speed = config.get('max_speed', 100)
        self.min_speed = config.get('min_speed', 20)

        # PID参数
        self.kp = config.get('kp', 1.5)
        self.ki = config.get('ki', 0.05)
        self.kd = config.get('kd', 0.8)

        self.integral = 0
        self.prev_error = 0

    def compute(self, track_result) -> SpeedCommand:
        """计算电机速度"""
        error = track_result.offset

        # PID
        self.integral += error
        self.integral = max(-50, min(50, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error

        correction = (self.kp * error +
                     self.ki * self.integral +
                     self.kd * derivative)

        # 速度规划
        speed = self.base_speed * track_result.speed_factor
        speed = max(self.min_speed, min(self.max_speed, speed))

        left = speed + correction * speed
        right = speed - correction * speed

        # 限幅
        left = max(-self.max_speed, min(self.max_speed, left))
        right = max(-self.max_speed, min(self.max_speed, right))

        return SpeedCommand(
            left_speed=left,
            right_speed=right,
            path_type=track_result.path_type.value
        )

    def reset(self):
        self.integral = 0
        self.prev_error = 0
```

#### 3. 主程序 (main.py)
```python
import cv2
import yaml
import time
from camera import Camera
from dual_camera_tracker import DualCameraTracker
from speed_planner import SpeedPlanner
from motor_control import MotorController

def main():
    with open('config.yaml', 'r') as f:
        config = yaml.safe_load(f)

    fwd_camera = Camera(config['forward_camera'])
    dwn_camera = Camera(config['down_camera'])
    tracker = DualCameraTracker(config['tracker'])
    planner = SpeedPlanner(config['planner'])
    motors = MotorController(config['motor'])

    print("=== 巡线机器人启动 ===")

    frame_count = 0
    start_time = time.time()

    try:
        while True:
            fwd_frame = fwd_camera.capture()
            dwn_frame = dwn_camera.capture()

            if fwd_frame is None and dwn_frame is None:
                continue

            # 循迹处理
            result = tracker.process(fwd_frame, dwn_frame)

            # 速度规划
            cmd = planner.compute(result)

            # 驱动电机
            motors.set_motors(cmd.left_speed, cmd.right_speed)

            # 帧率统计
            frame_count += 1
            if frame_count % 100 == 0:
                fps = frame_count / (time.time() - start_time)
                print(f"FPS: {fps:.1f} | Path: {result.path_type.value} | "
                      f"Offset: {result.offset:.2f} | "
                      f"Speed: L={cmd.left_speed:.0f} R={cmd.right_speed:.0f}")

            # 调试
            if config.get('debug', False):
                display = fwd_frame if fwd_frame is not None else dwn_frame
                if display is not None:
                    cv2.putText(display, f"Path: {result.path_type.value}",
                               (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
                    cv2.putText(display, f"Offset: {result.offset:.2f}",
                               (10, 60), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
                    cv2.imshow('Line Follower', display)
                    if cv2.waitKey(1) & 0xFF == ord('q'):
                        break

    finally:
        motors.cleanup()
        fwd_camera.release()
        dwn_camera.release()
        cv2.destroyAllWindows()

if __name__ == '__main__':
    main()
```

## 复杂路径处理策略

### 1. 断线处理
```
检测到断线 → 保持上一次的偏移方向 → 低速前进 → 恢复检测
```

### 2. 直角弯处理
```
前瞻检测到直角 → 减速 → 编码器计数转弯距离 → 完成转弯 → 恢复速度
```

### 3. 十字路口
```
检测到十字 → 保持上一次方向 → 不做特殊处理
```

### 4. 锐角弯
```
前瞻检测到锐角 → 急减速 → 原地差速转弯 → 对准后加速
```
