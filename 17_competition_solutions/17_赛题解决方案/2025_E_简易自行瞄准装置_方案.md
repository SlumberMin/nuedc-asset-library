## 二、视觉方案

### 2.1 靶心识别
```python
import cv2
import numpy as np

class TargetDetector:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1280)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 720)
        self.cap.set(cv2.CAP_PROP_FPS, 90)
        
        self.frame_center = (640, 360)
        
        # 相机内参（标定后填入）
        self.camera_matrix = None
        self.dist_coeffs = None
        
        # 运动预测器
        self.position_history = []
        self.max_history = 30
    
    def calibrate_camera(self, images, pattern_size=(9,6)):
        """相机标定"""
        obj_points = []
        img_points = []
        
        objp = np.zeros((pattern_size[0]*pattern_size[1], 3), np.float32)
        objp[:, :2] = np.mgrid[0:pattern_size[0], 0:pattern_size[1]].T.reshape(-1, 2)
        
        for img in images:
            gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
            ret, corners = cv2.findChessboardCorners(gray, pattern_size, None)
            if ret:
                obj_points.append(objp)
                img_points.append(corners)
        
        ret, self.camera_matrix, self.dist_coeffs, _, _ = cv2.calibrateCamera(
            obj_points, img_points, gray.shape[::-1], None, None
        )
        return ret
    
    def detect_target(self, frame):
        """
        检测靶心位置
        返回: (center_x, center_y, radius, confidence)
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 霍夫圆检测靶心同心圆
        circles = cv2.HoughCircles(
            gray, cv2.HOUGH_GRADIENT, 
            dp=1.2, minDist=50,
            param1=100, param2=40,
            minRadius=10, maxRadius=200
        )
        
        if circles is None:
            # 备选：轮廓检测
            return self._detect_by_contour(frame)
        
        circles = np.uint16(np.around(circles))
        
        # 选最内层的圆（靶心）
        best_circle = None
        best_radius = float('inf')
        
        for c in circles[0]:
            cx, cy, r = c
            if r < best_radius:
                # 验证：检查圆心区域是否为深色（靶心通常是黑色）
                roi = gray[max(0,cy-5):cy+5, max(0,cx-5):cx+5]
                if roi.size > 0 and np.mean(roi) < 80:
                    best_circle = (cx, cy, r)
                    best_radius = r
        
        if best_circle is None and len(circles[0]) > 0:
            c = circles[0][0]
            best_circle = (c[0], c[1], c[2])
        
        if best_circle:
            return (best_circle[0], best_circle[1], best_circle[2], 0.9)
        
        return None
    
    def _detect_by_contour(self, frame):
        """轮廓法检测靶心"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        blur = cv2.GaussianBlur(gray, (5, 5), 0)
        _, thresh = cv2.threshold(blur, 50, 255, cv2.THRESH_BINARY_INV)
        
        contours, _ = cv2.findContours(thresh, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
        
        # 找最内层的圆
        for c in sorted(contours, key=cv2.contourArea)[:5]:
            area = cv2.contourArea(c)
            perimeter = cv2.arcLength(c, True)
            if perimeter == 0:
                continue
            circularity = 4 * np.pi * area / (perimeter ** 2)
            
            if circularity > 0.7 and area > 100:
                M = cv2.moments(c)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    r = int(np.sqrt(area / np.pi))
                    return (cx, cy, r, 0.7)
        
        return None
    
    def pixel_to_angle(self, pixel_x, pixel_y):
        """像素坐标 → 云台角度偏移"""
        # 简化模型：线性映射
        # 实际应使用相机内参做精确投影
        dx = pixel_x - self.frame_center[0]
        dy = pixel_y - self.frame_center[1]
        
        # 像素/度 系数（标定获取）
        pixels_per_degree_x = 25.0  # 1280分辨率 / ~50°视场角
        pixels_per_degree_y = 25.0
        
        angle_x = dx / pixels_per_degree_x
        angle_y = dy / pixels_per_degree_y
        
        return angle_x, angle_y
```

### 2.2 运动靶预测
```python
class MotionPredictor:
    def __init__(self):
        self.history = []  # [(t, x, y), ...]
        self.max_history = 30
        self.predict_time = 0.15  # 预测150ms后的位置（系统延迟补偿）
    
    def update(self, t, x, y):
        self.history.append((t, x, y))
        if len(self.history) > self.max_history:
            self.history.pop(0)
    
    def predict_position(self):
        """线性外推预测未来位置"""
        if len(self.history) < 3:
            return None
        
        # 最近N个点做线性回归
        recent = self.history[-10:]
        times = np.array([p[0] for p in recent])
        xs = np.array([p[1] for p in recent])
        ys = np.array([p[2] for p in recent])
        
        # 线性拟合
        vx, cx = np.polyfit(times, xs, 1)
        vy, cy = np.polyfit(times, ys, 1)
        
        # 预测
        t_future = times[-1] + self.predict_time
        pred_x = vx * t_future + cx
        pred_y = vy * t_future + cy
        
        return (pred_x, pred_y)
    
    def predict_with_kalman(self):
        """卡尔曼滤波预测（更鲁棒）"""
        # 状态: [x, y, vx, vy]
        # 简化实现
        if len(self.history) < 2:
            return None
        
        dt = self.history[-1][0] - self.history[-2][0]
        vx = (self.history[-1][1] - self.history[-2][1]) / dt
        vy = (self.history[-1][2] - self.history[-2][2]) / dt
        
        pred_x = self.history[-1][1] + vx * self.predict_time
        pred_y = self.history[-1][2] + vy * self.predict_time
        
        return (pred_x, pred_y)
```

## 三、控制方案

### 3.1 二维云台PID
```python
class GimbalController:
    def __init__(self):
        # 水平轴PID
        self.pid_pan = PID(Kp=1.5, Ki=0.02, Kd=0.4)
        # 垂直轴PID
        self.pid_tilt = PID(Kp=1.5, Ki=0.02, Kd=0.4)
        
        # 当前云台角度
        self.pan_angle = 0.0   # 水平角度
        self.tilt_angle = 0.0  # 垂直角度
        
        # 限位
        self.pan_limit = 90    # ±90°
        self.tilt_limit = 45   # ±45°
        
        # 激光偏移补偿（激光与相机的安装偏移）
        self.laser_offset_x = 2.0  # 度
        self.laser_offset_y = 1.0  # 度
    
    def track_target(self, error_x, error_y, dt):
        """
        error_x, error_y: 像素误差
        返回: (pan_delta, tilt_delta)
        """
        # PID计算角度调整量
        delta_pan = self.pid_pan.compute(error_x, dt)
        delta_tilt = self.pid_tilt.compute(error_y, dt)
        
        # 更新云台角度
        self.pan_angle += delta_pan
        self.tilt_angle += delta_tilt
        
        # 限位
        self.pan_angle = np.clip(self.pan_angle, -self.pan_limit, self.pan_limit)
        self.tilt_angle = np.clip(self.tilt_angle, -self.tilt_limit, self.tilt_limit)
        
        return self.pan_angle, self.tilt_angle
    
    def aim_with_compensation(self, target_pixel, current_pixel, velocity, dt):
        """带前馈补偿的瞄准"""
        error_x = target_pixel[0] - current_pixel[0]
        error_y = target_pixel[1] - current_pixel[1]
        
        # 前馈补偿（云台跟随速度补偿）
        ff_x = 0.05 * velocity[0]
        ff_y = 0.05 * velocity[1]
        
        # PID + 前馈
        delta_pan = self.pid_pan.compute(error_x, dt) + ff_x
        delta_tilt = self.pid_tilt.compute(error_y, dt) + ff_y
        
        self.pan_angle += delta_pan
        self.tilt_angle += delta_tilt
        
        self.pan_angle = np.clip(self.pan_angle, -self.pan_limit, self.pan_limit)
        self.tilt_angle = np.clip(self.tilt_angle, -self.tilt_limit, self.tilt_limit)
        
        return self.pan_angle + self.laser_offset_x, self.tilt_angle + self.laser_offset_y
```

### 3.2 PID类（通用）
```python
class PID:
    def __init__(self, Kp, Ki, Kd, output_limit=50, integral_limit=100):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.output_limit = output_limit
        self.integral_limit = integral_limit
        self.integral = 0
        self.prev_error = 0
    
    def compute(self, error, dt):
        self.integral += error * dt
        self.integral = np.clip(self.integral, -self.integral_limit, self.integral_limit)
        derivative = (error - self.prev_error) / dt if dt > 0 else 0
        self.prev_error = error
        
        output = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        return np.clip(output, -self.output_limit, self.output_limit)
```

### 3.3 主控制程序
```python
import time

class AutoAimingSystem:
    def __init__(self):
        self.detector = TargetDetector(0)
        self.predictor = MotionPredictor()
        self.gimbal = GimbalController()
        
        self.mode = 'fixed'  # 'fixed' 或 'moving'
        self.is_aimed = False
        self.aim_threshold = 10  # 像素
    
    def run(self):
        while True:
            frame = self.detector.get_frame()
            t = time.time()
            
            # 检测靶心
            result = self.detector.detect_target(frame)
            if result is None:
                continue
            
            cx, cy, r, conf = result
            
            if self.mode == 'moving':
                # 运动靶：预测
                self.predictor.update(t, cx, cy)
                predicted = self.predictor.predict_position()
                if predicted:
                    target_x, target_y = predicted
                else:
                    target_x, target_y = cx, cy
            else:
                target_x, target_y = cx, cy
            
            # 计算误差
            error_x = target_x - self.detector.frame_center[0]
            error_y = target_y - self.detector.frame_center[1]
            error_pixels = np.sqrt(error_x**2 + error_y**2)
            
            # 判断是否瞄准
            if error_pixels < self.aim_threshold:
                self.is_aimed = True
                self.laser_on()
            else:
                self.is_aimed = False
                self.laser_off()
            
            # 云台控制
            dt = 0.011
            if self.mode == 'moving':
                velocity = self.predictor.get_velocity()
                pan, tilt = self.gimbal.aim_with_compensation(
                    (target_x, target_y), self.detector.frame_center, velocity, dt
                )
            else:
                pan, tilt = self.gimbal.track_target(error_x, error_y, dt)
            
            self.send_gimbal_command(pan, tilt)
```

## 七、调试要点
1. **标定优先**：相机标定 + 云台零位标定 + 激光偏移补偿
2. **PID调参**：先P后D最后I，固定靶调好后再调运动靶
3. **预测验证**：运动靶先不开预测，对比有无预测的效果差异
4. **死区设置**：误差小于阈值时停止调整，避免抖动
5. **激光安全**：使用Class II激光，功率<1mW，避免直视
