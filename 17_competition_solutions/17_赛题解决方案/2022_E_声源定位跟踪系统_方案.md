# 2022年E题：声源定位跟踪系统 — Orange Pi 5 专属方案

## 题目概述
设计声源定位与跟踪系统，通过麦克风阵列定位声源方向，控制舵机/云台使摄像头对准声源并跟踪移动。

## 系统架构

```
[MEMS麦克风阵列(4路)] → [I2S采集] → Orange Pi 5 → [舵机PWM] → [PTZ云台]
[OV5647摄像头] ────→ [MIPI-CSI] ─↗       ↓
                              [声源定位结果 + 视频跟踪]
```

## 核心方案

### 1. 声源定位模块
- **硬件**：4路MEMS麦克风（INMP441），I2S接口连接
- **算法**：GCC-PHAT（广义互相关-相位变换）
  - 计算各麦克风对的时延差TDOA
  - 基于TDOA的球面交叉定位
  - Orange Pi 5上用NumPy向量化计算，4路麦克风6个TDOA对

### 2. 视觉跟踪模块
- **摄像头**：OV5647，640×480@30fps
- **跟踪算法**：
  - 声源定位给出初始ROI（感兴趣区域）
  - CAMShift / KCF跟踪器锁定声源对应视觉目标
  - 视觉反馈修正声源定位误差

### 3. PTZ云台控制
- **硬件**：2路舵机（水平+俯仰），PWM控制
- **算法**：PID闭环
  - 误差 = 声源方位角 - 当前云台角度
  - 视觉目标偏离图像中心的像素误差作为辅助修正

## Orange Pi 5 代码实现

```python
#!/usr/bin/env python3
"""2022E 声源定位跟踪系统 - Orange Pi 5"""

import numpy as np
import cv2
import time
import spidev
from periphery import GPIO, PWM
from collections import deque

# ============ 声源定位 ============
class SoundLocalizer:
    def __init__(self, mic_positions, sample_rate=16000, speed_of_sound=343):
        self.mic_pos = np.array(mic_positions)  # 4路麦克风坐标(m)
        self.fs = sample_rate
        self.c = speed_of_sound
        self.n_mics = len(mic_positions)
        self.n_pairs = self.n_mics * (self.n_mics - 1) // 2

    def gcc_phat(self, sig1, sig2):
        """GCC-PHAT时延估计"""
        n = len(sig1) + len(sig2) - 1
        n_padded = 1 << (n - 1).bit_length()
        SIG1 = np.fft.rfft(sig1, n_padded)
        SIG2 = np.fft.rfft(sig2, n_padded)
        R = SIG1 * np.conj(SIG2)
        R /= (np.abs(R) + 1e-10)
        cc = np.fft.irfft(R)
        max_shift = int(0.5 * self.fs * np.linalg.norm(
            self.mic_pos[0] - self.mic_pos[1]) / self.c) + 10
        cc = np.concatenate([cc[-max_shift:], cc[:max_shift + 1]])
        shift = np.argmax(cc) - max_shift
        tau = shift / self.fs
        return tau

    def estimate_angle(self, signals):
        """基于TDOA估计声源方位角(水平角和俯仰角)"""
        # 计算所有麦克风对的TDOA
        taus = []
        for i in range(self.n_mics):
            for j in range(i + 1, self.n_mics):
                tau = self.gcc_phat(signals[i], signals[j])
                taus.append(tau)

        # 简化2D定位: 假设声源在远场
        # 用左右麦克风对求水平角，上下麦克风对求俯仰角
        # mic0(左), mic1(右), mic2(前), mic3(后)
        d_left_right = np.linalg.norm(self.mic_pos[0] - self.mic_pos[1])
        d_front_back = np.linalg.norm(self.mic_pos[2] - self.mic_pos[3])

        # TDOA对: (0,1), (0,2), (0,3), (1,2), (1,3), (2,3)
        tau_lr = taus[0]  # mic0-mic1
        tau_fb = taus[5]  # mic2-mic3

        azimuth = np.arcsin(np.clip(tau_lr * self.c / d_left_right, -1, 1))
        elevation = np.arcsin(np.clip(tau_fb * self.c / d_front_back, -1, 1))

        return np.degrees(azimuth), np.degrees(elevation)


# ============ 舵机云台控制 ============
class PanTiltController:
    def __init__(self, pan_ch=0, tilt_ch=1):
        self.pan_pwm = PWM(pan_ch, 0)
        self.tilt_pwm = PWM(tilt_ch, 0)
        self.pan_pwm.frequency = 50
        self.tilt_pwm.frequency = 50
        self.pan_pwm.enable = True
        self.tilt_pwm.enable = True
        self.pan_angle = 90.0
        self.tilt_angle = 90.0
        # PID参数
        self.kp_pan, self.ki_pan, self.kd_pan = 0.8, 0.01, 0.3
        self.kp_tilt, self.ki_tilt, self.kd_tilt = 0.8, 0.01, 0.3
        self.err_pan_sum = 0.0
        self.err_tilt_sum = 0.0
        self.err_pan_prev = 0.0
        self.err_tilt_prev = 0.0

    def angle_to_duty(self, angle):
        """角度转PWM占空比 (0.5ms~2.5ms对应0~180度)"""
        pulse = 0.5 + (angle / 180.0) * 2.0  # ms
        return pulse / 20.0  # 50Hz周期20ms

    def move_to(self, pan, tilt):
        self.pan_angle = np.clip(pan, 0, 180)
        self.tilt_angle = np.clip(tilt, 0, 180)
        self.pan_pwm.duty_cycle = self.angle_to_duty(self.pan_angle)
        self.tilt_pwm.duty_cycle = self.angle_to_duty(self.tilt_angle)

    def track(self, err_azimuth, err_elevation, dt):
        """PID跟踪"""
        self.err_pan_sum += err_azimuth * dt
        self.err_tilt_sum += err_elevation * dt

        d_pan = (err_azimuth - self.err_pan_prev) / max(dt, 0.001)
        d_tilt = (err_elevation - self.err_tilt_prev) / max(dt, 0.001)

        pan_adj = self.kp_pan * err_azimuth + self.ki_pan * self.err_pan_sum + self.kd_pan * d_pan
        tilt_adj = self.kp_tilt * err_elevation + self.ki_tilt * self.err_tilt_sum + self.kd_tilt * d_tilt

        self.err_pan_prev = err_azimuth
        self.err_tilt_prev = err_elevation

        self.move_to(self.pan_angle + pan_adj, self.tilt_angle + tilt_adj)


# ============ 视觉跟踪 ============
class VisualTracker:
    def __init__(self):
        self.tracker = None
        self.bbox = None

    def init_roi(self, frame, center_x, center_y, size=80):
        """根据声源方位初始化跟踪ROI"""
        h, w = frame.shape[:2]
        x = max(0, int(center_x - size / 2))
        y = max(0, int(center_y - size / 2))
        self.bbox = (x, y, size, size)
        self.tracker = cv2.TrackerCSRT_create()
        self.tracker.init(frame, self.bbox)
        return self.bbox

    def update(self, frame):
        """更新跟踪"""
        if self.tracker is None:
            return None, False
        ok, bbox = self.tracker.update(frame)
        if ok:
            self.bbox = bbox
            cx = bbox[0] + bbox[2] / 2
            cy = bbox[1] + bbox[3] / 2
            return (cx, cy), True
        return None, False


# ============ 主系统 ============
def main():
    # 麦克风阵列坐标 (正方形, 边长5cm)
    mic_pos = [
        [-0.025, 0.025, 0],   # mic0 左上
        [0.025, 0.025, 0],    # mic1 右上
        [0.025, -0.025, 0],   # mic2 右下
        [-0.025, -0.025, 0],  # mic3 左下
    ]

    localizer = SoundLocalizer(mic_pos)
    controller = PanTiltController(pan_ch=0, tilt_ch=1)
    tracker = VisualTracker()

    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    frame_w, frame_h = 640, 480
    # FOV假设: 水平60度, 垂直45度
    hfov, vfov = 60.0, 45.0

    print("声源定位跟踪系统启动...")
    last_time = time.time()

    while True:
        ret, frame = cap.read()
        if not ret:
            continue

        now = time.time()
        dt = now - last_time
        last_time = now

        # 读取麦克风数据 (4通道, 每帧1024样本)
        signals = read_mic_data(4, 1024)  # 需实现硬件读取

        # 声源定位
        azimuth, elevation = localizer.estimate_angle(signals)

        # 声源方位转像素坐标
        src_x = frame_w / 2 + (azimuth / hfov) * frame_w
        src_y = frame_h / 2 - (elevation / vfov) * frame_h

        # 视觉跟踪
        center, tracked = tracker.update(frame)
        if not tracked:
            # 声源方位引导初始化
            tracker.init_roi(frame, src_x, src_y)
            center = (src_x, src_y)

        if center:
            # 视觉修正误差
            err_pixel_x = center[0] - frame_w / 2
            err_pixel_y = center[1] - frame_h / 2
            err_az = err_pixel_x / frame_w * hfov
            err_el = err_pixel_y / frame_h * vfov

            # PID跟踪
            controller.track(err_az, err_el, dt)

        # 显示
        if center:
            cv2.circle(frame, (int(center[0]), int(center[1])), 20, (0, 255, 0), 2)
        cv2.putText(frame, f"AZ:{azimuth:.1f} EL:{elevation:.1f}",
                    (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
        cv2.imshow("Sound Tracker", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    cap.release()
    cv2.destroyAllWindows()


def read_mic_data(n_channels, n_samples):
    """读取I2S麦克风阵列数据 - 需根据实际硬件实现"""
    # 方案1: 使用pyaudio读取ALSA设备
    # 方案2: 使用SPI读取ADC
    # 方案3: 使用I2S驱动直接读取
    signals = []
    for ch in range(n_channels):
        signals.append(np.random.randn(n_samples))  # 占位
    return signals


if __name__ == "__main__":
    main()
```

## 硬件接线

| 模块 | Orange Pi 5引脚 | 说明 |
|------|-----------------|------|
| INMP441 #0 SDA | GPIO1_A0 (I2S0_SDI) | 麦克风数据 |
| INMP441 SCK | GPIO1_A2 (I2S0_SCLK) | I2S时钟 |
| INMP441 WS | GPIO1_A3 (I2S0_LRCK) | 字选择 |
| 水平舵机 | GPIO1_B4 (PWM14) | 50Hz PWM |
| 俯仰舵机 | GPIO1_B5 (PWM15) | 50Hz PWM |
| OV5647 | MIPI-CSI接口 | 摄像头 |

## 系统参数

| 参数 | 值 |
|------|-----|
| 麦克风数量 | 4 (正方形阵列, 边长5cm) |
| 采样率 | 16kHz |
| 定位精度 | ±5° |
| 跟踪延迟 | <100ms |
| 云台角度范围 | 0~180° |
| 帧率 | 30fps |
