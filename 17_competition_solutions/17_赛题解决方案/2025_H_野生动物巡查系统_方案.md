# 2025年H题：野生动物巡查系统 — Orange Pi 5 专属方案

## 题目概述
设计野外自主巡查系统，能够在指定区域自主巡逻，通过视觉识别野生动物，记录并上报发现的目标。

## 系统架构

```
[OV5647摄像头] → [MIPI-CSI] → Orange Pi 5 → [GPS模块] → 路径规划
                    ↓               ↓
            [MNN推理引擎]    [电机驱动] → [小车底盘]
                    ↓
            [野生动物检测+分类]
                    ↓
            [SD卡存储 + 无线上报]
```

## 核心方案

### 1. 野生动物视觉识别
- **模型**：MobileNet-SSD / YOLOv5n 量化后MNN推理
- **训练数据集**：自建动物数据集（兔、鸟、蛇、鹿等常见物种）
- **推理流程**：摄像头采集 → 缩放至320×320 → MNN前向推理 → NMS后处理
- **帧率**：Orange Pi 5 NPU上可达15~20fps

### 2. 自主巡查路径
- **GPS + IMU** 定位
- **路径规划**：预设航点 + A*避障
- **巡航模式**：按航点依次到达，每个航点停留检测

### 3. 数据记录与上报
- 检测到动物 → 拍照存SD卡（含GPS坐标、时间戳）
- WiFi/4G模块上报到监控中心
- 本地SQLite数据库记录

## Orange Pi 5 代码实现

```python
#!/usr/bin/env python3
"""2025H 野生动物巡查系统 - Orange Pi 5"""

import numpy as np
import cv2
import time
import json
import sqlite3
import os
from datetime import datetime
from periphery import GPIO, PWM

# ============ MNN推理引擎 ============
class AnimalDetector:
    def __init__(self, model_path="animal_detect.mnn", labels=None):
        import MNN
        self.interpreter = MNN.Interpreter(model_path)
        self.session = self.interpreter.createSession()
        self.input_tensor = self.interpreter.getSessionInput(self.session)
        self.labels = labels or ["background", "rabbit", "bird", "snake",
                                  "deer", "wild_boar", "fox", "monkey"]
        self.conf_threshold = 0.5
        self.nms_threshold = 0.4
        self.input_size = 320

    def preprocess(self, frame):
        img = cv2.resize(frame, (self.input_size, self.input_size))
        img = img.astype(np.float32) / 255.0
        img = img[:, :, ::-1]  # BGR→RGB
        img = np.transpose(img, (2, 0, 1))  # CHW
        img = np.expand_dims(img, 0)
        return img

    def detect(self, frame):
        h, w = frame.shape[:2]
        input_data = self.preprocess(frame)

        # MNN推理
        tmp_input = MNN.Tensor((1, 3, self.input_size, self.input_size),
                                MNN.Halide_Type_Float, input_data,
                                MNN.Tensor_DimensionType_Caffe)
        self.input_tensor.copyFrom(tmp_input)
        self.interpreter.runSession(self.session)

        # 解析输出
        output_tensor = self.interpreter.getSessionOutput(self.session)
        output = output_tensor.getData().reshape(-1, 6)  # [x1,y1,x2,y2,score,class]

        # NMS
        detections = []
        for det in output:
            score = det[4]
            if score < self.conf_threshold:
                continue
            cls_id = int(det[5])
            x1 = int(det[0] * w)
            y1 = int(det[1] * h)
            x2 = int(det[2] * w)
            y2 = int(det[3] * h)
            label = self.labels[cls_id] if cls_id < len(self.labels) else "unknown"
            detections.append({
                "bbox": [x1, y1, x2, y2],
                "score": float(score),
                "class_id": cls_id,
                "label": label
            })

        # 简单NMS
        detections = self.nms(detections)
        return detections

    def nms(self, dets):
        if not dets:
            return []
        dets.sort(key=lambda x: x["score"], reverse=True)
        keep = []
        while dets:
            best = dets.pop(0)
            keep.append(best)
            dets = [d for d in dets if self.iou(best["bbox"], d["bbox"]) < self.nms_threshold
                    or d["class_id"] != best["class_id"]]
        return keep

    @staticmethod
    def iou(box1, box2):
        x1 = max(box1[0], box2[0])
        y1 = max(box1[1], box2[1])
        x2 = min(box1[2], box2[2])
        y2 = min(box1[3], box2[3])
        inter = max(0, x2-x1) * max(0, y2-y1)
        area1 = (box1[2]-box1[0]) * (box1[3]-box1[1])
        area2 = (box2[2]-box2[0]) * (box2[3]-box2[1])
        union = area1 + area2 - inter
        return inter / max(union, 1e-6)


# ============ GPS模块 ============
class GPSModule:
    def __init__(self, port="/dev/ttyS1", baud=9600):
        import serial
        self.ser = serial.Serial(port, baud, timeout=1)

    def read_position(self):
        """读取GPS定位"""
        while True:
            line = self.ser.readline().decode('ascii', errors='replace').strip()
            if line.startswith('$GPGGA') or line.startswith('$GNGGA'):
                fields = line.split(',')
                if len(fields) > 5 and fields[2]:
                    lat = self.parse_coord(fields[2], fields[3])
                    lon = self.parse_coord(fields[4], fields[5])
                    alt = float(fields[9]) if fields[9] else 0
                    return lat, lon, alt
        return None, None, None

    @staticmethod
    def parse_coord(val, direction):
        deg = int(val[:2])
        minutes = float(val[2:])
        coord = deg + minutes / 60.0
        if direction in ('S', 'W'):
            coord = -coord
        return coord


# ============ IMU模块 ============
class IMUModule:
    def __init__(self, bus=1, addr=0x68):
        import smbus2
        self.bus = smbus2.SMBus(bus)
        self.addr = addr
        # 唤醒MPU6050
        self.bus.write_byte_data(self.addr, 0x6B, 0)

    def read_yaw(self):
        """读取航向角(简化)"""
        data = self.bus.read_i2c_block_data(self.addr, 0x47, 2)
        gz = (data[0] << 8) | data[1]
        if gz > 32767:
            gz -= 65536
        return gz / 131.0  # deg/s


# ============ 小车底盘控制 ============
class RoverChassis:
    def __init__(self):
        self.motor_left = PWM(0, 0)
        self.motor_right = PWM(1, 0)
        self.motor_left.frequency = 1000
        self.motor_right.frequency = 1000
        self.motor_left.enable = True
        self.motor_right.enable = True

        self.dir_l1 = GPIO((1, 20), "out")
        self.dir_l2 = GPIO((1, 21), "out")
        self.dir_r1 = GPIO((1, 22), "out")
        self.dir_r2 = GPIO((1, 23), "out")

    def drive(self, left_speed, right_speed):
        """左右轮速度 -1.0~1.0"""
        # 左轮
        self.dir_l1.write(left_speed >= 0)
        self.dir_l2.write(left_speed < 0)
        self.motor_left.duty_cycle = min(abs(left_speed), 1.0)
        # 右轮
        self.dir_r1.write(right_speed >= 0)
        self.dir_r2.write(right_speed < 0)
        self.motor_right.duty_cycle = min(abs(right_speed), 1.0)

    def stop(self):
        self.drive(0, 0)


# ============ 数据记录 ============
class DataRecorder:
    def __init__(self, db_path="patrol_data.db", photo_dir="photos"):
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS detections (
                id INTEGER PRIMARY KEY,
                timestamp TEXT,
                latitude REAL,
                longitude REAL,
                species TEXT,
                confidence REAL,
                photo_path TEXT
            )
        """)
        self.photo_dir = photo_dir
        os.makedirs(photo_dir, exist_ok=True)

    def record(self, lat, lon, species, confidence, frame):
        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
        photo_name = f"{ts}_{species}.jpg"
        photo_path = os.path.join(self.photo_dir, photo_name)
        cv2.imwrite(photo_path, frame)

        self.conn.execute(
            "INSERT INTO detections (timestamp, latitude, longitude, species, confidence, photo_path) "
            "VALUES (?, ?, ?, ?, ?, ?)",
            (ts, lat, lon, species, confidence, photo_path)
        )
        self.conn.commit()
        return photo_path


# ============ 路径规划 ============
class WaypointNavigator:
    def __init__(self, gps, chassis, imu):
        self.gps = gps
        self.chassis = chassis
        self.imu = imu
        self.waypoints = []
        self.current_wp = 0
        self.arrival_threshold = 0.0001  # ~10m

    def load_waypoints(self, waypoints):
        """waypoints: [(lat, lon), ...]"""
        self.waypoints = waypoints
        self.current_wp = 0

    def navigate_to_next(self):
        """导航到下一个航点，返回是否到达"""
        if self.current_wp >= len(self.waypoints):
            return True  # 全部完成

        target = self.waypoints[self.current_wp]
        pos = self.gps.read_position()
        if pos[0] is None:
            self.chassis.stop()
            return False

        lat, lon = pos[0], pos[1]
        dlat = target[0] - lat
        dlon = target[1] - lon
        dist = np.sqrt(dlat**2 + dlon**2)

        if dist < self.arrival_threshold:
            self.chassis.stop()
            self.current_wp += 1
            return self.current_wp >= len(self.waypoints)

        # 计算目标方位角
        target_bearing = np.degrees(np.arctan2(dlon, dlat))
        # 简单比例导航
        error = target_bearing  # 需要减去当前航向
        steer = np.clip(error * 0.02, -0.5, 0.5)
        self.chassis.drive(0.4 - steer, 0.4 + steer)
        return False


# ============ 主程序 ============
def main():
    # 初始化各模块
    detector = AnimalDetector(model_path="animal_detect.mnn")
    gps = GPSModule(port="/dev/ttyS1")
    imu = IMUModule()
    chassis = RoverChassis()
    recorder = DataRecorder()
    navigator = WaypointNavigator(gps, chassis, imu)

    # 设置巡查航点(示例)
    navigator.load_waypoints([
        (30.12345, 120.12345),
        (30.12355, 120.12355),
        (30.12365, 120.12345),
        (30.12345, 120.12345),  # 返回起点
    ])

    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    detect_interval = 3  # 每3帧检测一次
    frame_count = 0
    last_detections = []
    patrol_log = []

    print("野生动物巡查系统启动...")
    patrol_start = datetime.now()

    while True:
        ret, frame = cap.read()
        if not ret:
            continue

        frame_count += 1

        # 导航
        arrived = navigator.navigate_to_next()
        if arrived:
            print("巡查完成!")
            break

        # 检测
        if frame_count % detect_interval == 0:
            detections = detector.detect(frame)
            last_detections = detections

            for det in detections:
                x1, y1, x2, y2 = det["bbox"]
                label = det["label"]
                score = det["score"]

                # 绘制检测框
                color = (0, 255, 0) if score > 0.7 else (0, 255, 255)
                cv2.rectangle(frame, (x1, y1), (x2, y2), color, 2)
                cv2.putText(frame, f"{label} {score:.2f}",
                            (x1, y1-10), cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)

                # 记录发现
                if score > 0.7:
                    pos = gps.read_position()
                    lat, lon = pos[0] or 0, pos[1] or 0
                    photo_path = recorder.record(lat, lon, label, score, frame)
                    patrol_log.append({
                        "time": datetime.now().isoformat(),
                        "species": label,
                        "confidence": score,
                        "lat": lat,
                        "lon": lon,
                        "photo": photo_path
                    })
                    print(f"[发现] {label} 置信度:{score:.2f} 位置:({lat:.5f},{lon:.5f})")

        # 绘制上一帧结果
        for det in last_detections:
            x1, y1, x2, y2 = det["bbox"]
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)

        # 状态信息
        pos = gps.read_position()
        wp = navigator.current_wp
        cv2.putText(frame, f"WP:{wp}/{len(navigator.waypoints)} Pos:{pos[0]:.5f},{pos[1]:.5f}",
                    (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 0), 1)
        cv2.putText(frame, f"Detections: {len(patrol_log)}",
                    (10, 55), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 0), 1)

        cv2.imshow("Wildlife Patrol", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    # 保存巡查日志
    patrol_end = datetime.now()
    log_path = f"patrol_{patrol_start.strftime('%Y%m%d_%H%M%S')}.json"
    with open(log_path, 'w') as f:
        json.dump({
            "start": patrol_start.isoformat(),
            "end": patrol_end.isoformat(),
            "detections": patrol_log,
            "waypoints": navigator.waypoints,
            "total_detections": len(patrol_log)
        }, f, indent=2, ensure_ascii=False)

    print(f"巡查日志已保存: {log_path}")
    print(f"总发现: {len(patrol_log)} 只动物")
    chassis.stop()
    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

## 巡查策略

| 策略 | 说明 |
|------|------|
| 固定航点 | 预设GPS坐标依次到达 |
| 区域覆盖 | 往返式扫描覆盖整片区域 |
| 重点巡查 | 对历史高发现区域加密巡查 |
| 响应模式 | 检测到动物后原地停留记录 |
| 节能模式 | 低速巡航，检测到目标时加速接近 |

## 优化建议
1. 使用RKNN替代MNN，充分利用NPU硬件加速
2. 加入声音检测（麦克风）辅助发现隐蔽动物
3. 热成像摄像头用于夜间巡查
4. 蜂群模式：多台巡查车协同覆盖大区域
5. 边缘AI + 云端分析：本地快速检测，云端精确分类
