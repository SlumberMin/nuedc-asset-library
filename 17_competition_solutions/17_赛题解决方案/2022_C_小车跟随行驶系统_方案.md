# 2022年C题：小车跟随行驶系统 — Orange Pi 5 专属方案

## 题目概述
设计两辆小车，前车自主行驶，后车通过视觉识别前车标志并保持一定距离跟随行驶，实现转弯、避障等功能。

## 系统架构

```
后车系统:
[OV5647摄像头] → [MIPI-CSI] → Orange Pi 5 → [电机驱动L298N] → [直流电机×2]
                                  ↓
[超声波HC-SR04] → [GPIO] ───→ [避障+距离控制]
[编码器] ────────→ [GPIO中断] → [里程计]
```

## 核心方案

### 1. 前车视觉识别
- **标志识别**：前车尾部放置ArUco标记（字典DICT_4X4_50）
- **检测流程**：灰度化 → 自适应阈值 → 轮廓检测 → ArUco解码
- **提取信息**：标记ID、中心像素坐标、包围框面积（估算距离）

### 2. 距离估算
- **单目测距公式**：`D = (实际标记宽度 × 焦距) / 像素宽度`
- 标定焦距f后，通过ArUco标记像素宽度实时计算距离
- 超声波辅助校验

### 3. 路径跟随
- 标记中心偏离图像中心 → 计算偏转角 → PID转向
- 距离控制：保持设定距离，PID调速

## Orange Pi 5 代码实现

```python
#!/usr/bin/env python3
"""2022C 小车跟随行驶系统 - Orange Pi 5"""

import numpy as np
import cv2
import time
from periphery import GPIO, PWM

# ============ ArUco标记检测 ============
class ArucoDetector:
    def __init__(self, marker_id=0, marker_size_m=0.08):
        self.marker_id = marker_id
        self.marker_size = marker_size_m
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        # 相机内参(需标定)
        self.camera_matrix = np.array([
            [600, 0, 320],
            [0, 600, 240],
            [0, 0, 1]
        ], dtype=np.float64)
        self.dist_coeffs = np.zeros(5)
        self.focal_length = 600.0

    def detect(self, frame):
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, _ = self.detector.detectMarkers(gray)
        if ids is None:
            return None
        for i, mid in enumerate(ids.flatten()):
            if mid == self.marker_id:
                c = corners[i][0]
                cx = np.mean(c[:, 0])
                cy = np.mean(c[:, 1])
                # 像素宽度
                pixel_w = np.linalg.norm(c[0] - c[1])
                # 距离估算
                distance = (self.marker_size * self.focal_length) / max(pixel_w, 1)
                # 偏转角(水平偏移)
                img_cx = frame.shape[1] / 2
                angle = np.degrees(np.arctan2(cx - img_cx, self.focal_length))
                return {
                    "center": (cx, cy),
                    "distance": distance,
                    "angle": angle,
                    "corners": c,
                    "pixel_width": pixel_w
                }
        return None


# ============ 电机驱动 ============
class MotorDriver:
    def __init__(self):
        # L298N: 左电机(ENA, IN1, IN2), 右电机(ENB, IN3, IN4)
        self.pwm_left = PWM(0, 0)    # PWM通道0
        self.pwm_right = PWM(1, 0)   # PWM通道1
        self.pwm_left.frequency = 1000
        self.pwm_right.frequency = 1000
        self.pwm_left.enable = True
        self.pwm_right.enable = True

        self.in1 = GPIO((1, 20), "out")
        self.in2 = GPIO((1, 21), "out")
        self.in3 = GPIO((1, 22), "out")
        self.in4 = GPIO((1, 23), "out")

    def set_left(self, speed):
        """speed: -1.0 ~ 1.0"""
        if speed >= 0:
            self.in1.write(True)
            self.in2.write(False)
        else:
            self.in1.write(False)
            self.in2.write(True)
        self.pwm_left.duty_cycle = min(abs(speed), 1.0)

    def set_right(self, speed):
        if speed >= 0:
            self.in3.write(True)
            self.in4.write(False)
        else:
            self.in3.write(False)
            self.in4.write(True)
        self.pwm_right.duty_cycle = min(abs(speed), 1.0)

    def drive(self, linear, angular):
        """差速控制: linear [-1,1], angular [-1,1]"""
        left = linear - angular * 0.5
        right = linear + angular * 0.5
        left = np.clip(left, -1, 1)
        right = np.clip(right, -1, 1)
        self.set_left(left)
        self.set_right(right)

    def stop(self):
        self.set_left(0)
        self.set_right(0)


# ============ 超声波测距 ============
class UltrasonicSensor:
    def __init__(self, trig_pin, echo_pin):
        self.trig = GPIO(trig_pin, "out")
        self.echo = GPIO(echo_pin, "in")

    def measure(self):
        self.trig.write(False)
        time.sleep(0.00002)
        self.trig.write(True)
        time.sleep(0.00001)
        self.trig.write(False)

        timeout = time.time() + 0.04
        while not self.echo.read():
            if time.time() > timeout:
                return -1
        t_start = time.time()

        while self.echo.read():
            if time.time() > timeout:
                return -1
        t_end = time.time()

        return (t_end - t_start) * 343 / 2  # meters


# ============ PID控制器 ============
class PID:
    def __init__(self, kp, ki, kd, output_limit=1.0):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.limit = output_limit
        self.integral = 0.0
        self.prev_error = 0.0

    def compute(self, error, dt):
        self.integral += error * dt
        self.integral = np.clip(self.integral, -self.limit, self.limit)
        derivative = (error - self.prev_error) / max(dt, 0.001)
        self.prev_error = error
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        return np.clip(output, -self.limit, self.limit)


# ============ 主程序 ============
def main():
    detector = ArucoDetector(marker_id=0, marker_size_m=0.08)
    motor = MotorDriver()
    sonar = UltrasonicSensor(trig_pin=(1, 24), echo_pin=(1, 25))

    # PID控制器
    pid_distance = PID(kp=0.8, ki=0.02, kd=0.3, output_limit=0.6)
    pid_steering = PID(kp=0.01, ki=0.0005, kd=0.005, output_limit=0.8)

    target_distance = 0.5  # 目标跟随距离50cm
    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    print("小车跟随系统启动...")
    last_time = time.time()
    lost_count = 0

    while True:
        ret, frame = cap.read()
        if not ret:
            continue

        now = time.time()
        dt = now - last_time
        last_time = now

        # 检测前车标记
        result = detector.detect(frame)

        if result:
            lost_count = 0
            distance = result["distance"]
            angle = result["angle"]

            # 距离PID
            dist_error = distance - target_distance
            linear_speed = pid_distance.compute(dist_error, dt)

            # 转向PID
            steer_output = pid_steering.compute(angle, dt)

            # 差速驱动
            motor.drive(linear_speed, steer_output)

            # 显示
            cx, cy = result["center"]
            cv2.circle(frame, (int(cx), int(cy)), 10, (0, 255, 0), 2)
            cv2.putText(frame, f"D:{distance:.2f}m A:{angle:.1f}deg",
                        (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        else:
            lost_count += 1
            if lost_count > 30:  # 1秒未检测到
                motor.stop()  # 停车
                # 可加入搜索模式(原地旋转)

        # 超声波避障
        obstacle_dist = sonar.measure()
        if 0 < obstacle_dist < 0.2:
            motor.stop()
            time.sleep(0.5)
            # 紧急避障
            motor.drive(-0.3, 0.5)
            time.sleep(0.5)

        cv2.imshow("Follower", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    motor.stop()
    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

## 硬件接线

| 模块 | Orange Pi 5引脚 | 说明 |
|------|-----------------|------|
| OV5647 | MIPI-CSI | 摄像头 |
| L298N ENA | PWM0 | 左电机使能 |
| L298N IN1 | GPIO1_B4 | 左电机方向 |
| L298N IN2 | GPIO1_B5 | 左电机方向 |
| L298N ENB | PWM1 | 右电机使能 |
| L298N IN3 | GPIO1_B6 | 右电机方向 |
| L298N IN4 | GPIO1_B7 | 右电机方向 |
| HC-SR04 TRIG | GPIO1_C0 | 超声波触发 |
| HC-SR04 ECHO | GPIO1_C1 | 超声波回响 |
| 编码器A相 | GPIO1_C2 | 速度反馈(可选) |

## 关键参数

| 参数 | 值 |
|------|-----|
| 跟随距离 | 0.5m |
| 最大速度 | 0.5m/s |
| 标记尺寸 | 8cm×8cm ArUco |
| 检测帧率 | 30fps |
| 避障距离 | 0.2m紧急停车 |
| 距离精度 | ±3cm |

## 优化建议
1. 加入前车运动预测（卡尔曼滤波），提升跟随平滑度
2. 使用MobileNet-SSD检测前车本体，不依赖ArUco标记时更鲁棒
3. 编码器里程计融合视觉测距，提升距离估计精度
4. 多级速度规划：远距离快速接近，近距离慢速精细跟随
