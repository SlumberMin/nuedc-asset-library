# 2016年电赛 - 物品分拣搬送装置 - Orange Pi 5解决方案

## 题目要求

设计制作物品分拣搬送装置，实现：
1. 视觉识别不同颜色/形状的物品
2. 机械臂抓取物品
3. 分拣到指定位置
4. 快速准确分拣

## 系统架构

```
┌──────────────┐    CSI/USB     ┌──────────────┐    UART     ┌──────────────┐
│  摄像头俯视   │──────────────→│  Orange Pi 5 │────────────→│  STM32主控   │
│  识别物品     │              │  RK3588S     │             │  舵机/步进驱动│
└──────────────┘              │  视觉+运动规划│             └──────┬───────┘
                               └──────────────┘              ┌────┴────┐
                                                              │4DOF机械臂│
                                                              │吸盘/夹爪 │
                                                              └─────────┘
```

## 核心设计

### 1. 视觉识别物品

```python
import cv2
import numpy as np
import time
import struct
import serial

class ItemRecognitionVision:
    """物品识别视觉系统"""
    
    # 物品类型定义
    ITEM_RED = 1
    ITEM_GREEN = 2
    ITEM_BLUE = 3
    ITEM_YELLOW = 4
    
    # 颜色HSV范围
    COLOR_RANGES = {
        'red1': ([0, 100, 100], [10, 255, 255]),
        'red2': ([170, 100, 100], [180, 255, 255]),
        'green': ([35, 100, 100], [85, 255, 255]),
        'blue': ([100, 100, 100], [130, 255, 255]),
        'yellow': ([20, 100, 100], [35, 255, 255]),
    }
    
    COLOR_NAMES = {
        1: 'red', 2: 'green', 3: 'blue', 4: 'yellow'
    }
    
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        
        # 工作区域标定（传送带区域）
        self.work_area = np.float32([[50, 50], [590, 50], [590, 430], [50, 430]])
        
        # 透视变换
        self.dst = np.float32([[0,0], [400,0], [400,300], [0,300]])
        self.M = cv2.getPerspectiveTransform(self.work_area, self.dst)
        
        # 像素到mm转换
        self.scale_x = 300.0 / 400  # mm/pixel
        self.scale_y = 200.0 / 300
        
        # 物品最小面积
        self.min_area = 500
        
    def detect_items(self):
        """检测所有物品，返回物品列表 [(x_mm, y_mm, type, angle), ...]"""
        ret, frame = self.cap.read()
        if not ret:
            return []
        
        # 透视变换
        warped = cv2.warpPerspective(frame, self.M, (400, 300))
        hsv = cv2.cvtColor(warped, cv2.COLOR_BGR2HSV)
        
        items = []
        
        # 分别检测每种颜色
        for color_name, (lower, upper) in self.COLOR_RANGES.items():
            mask = cv2.inRange(hsv, np.array(lower), np.array(upper))
            
            kernel = np.ones((5, 5), np.uint8)
            mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
            mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
            
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
            
            for cnt in contours:
                area = cv2.contourArea(cnt)
                if area < self.min_area:
                    continue
                
                # 位置
                M = cv2.moments(cnt)
                if M["m00"] == 0:
                    continue
                cx = int(M["m10"] / M["m00"])
                cy = int(M["m01"] / M["m00"])
                
                # 转换为实际坐标
                x_mm = cx * self.scale_x
                y_mm = cy * self.scale_y
                
                # 朝向角度（最小外接矩形）
                rect = cv2.minAreaRect(cnt)
                angle = rect[2]
                
                # 形状判断
                shape = self._classify_shape(cnt)
                
                # 物品类型
                if 'red' in color_name:
                    item_type = self.ITEM_RED
                elif 'green' in color_name:
                    item_type = self.ITEM_GREEN
                elif 'blue' in color_name:
                    item_type = self.ITEM_BLUE
                else:
                    item_type = self.ITEM_YELLOW
                
                items.append((x_mm, y_mm, item_type, angle, shape))
                
                # 可视化
                cv2.drawContours(warped, [cnt], -1, (0, 255, 0), 2)
                cv2.circle(warped, (cx, cy), 4, (0, 0, 255), -1)
                cv2.putText(warped, f"{color_name} {shape}", (cx-20, cy-15),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.4, (255, 255, 255), 1)
        
        cv2.imshow('Detection', warped)
        return items
    
    def _classify_shape(self, contour):
        """形状分类"""
        peri = cv2.arcLength(contour, True)
        approx = cv2.approxPolyDP(contour, 0.04 * peri, True)
        vertices = len(approx)
        
        if vertices == 3:
            return 'triangle'
        elif vertices == 4:
            return 'rectangle'
        else:
            area = cv2.contourArea(contour)
            circularity = 4 * np.pi * area / (peri * peri)
            if circularity > 0.8:
                return 'circle'
            return 'polygon'
    
    def release(self):
        self.cap.release()
```

### 2. 机械臂运动规划

```python
import math

class RoboticArmKinematics:
    """
    4DOF机械臂运动学（简化版）
    关节: 底座旋转 + 大臂 + 小臂 + 夹爪
    """
    
    def __init__(self):
        # 连杆长度(mm)
        self.L1 = 100  # 大臂
        self.L2 = 100  # 小臂
        self.L3 = 50   # 夹爪
        
    def inverse_kinematics(self, x, y, z):
        """
        逆运动学求解
        输入: 目标位置(mm)
        输出: 关节角度(度) [base, joint1, joint2, gripper]
        """
        # 底座旋转角
        base_angle = math.degrees(math.atan2(y, x))
        
        # 平面内距离
        r = math.sqrt(x**2 + y**2)
        
        # 二连杆逆运动学
        D = (r**2 + (z - 0)**2 - self.L1**2 - self.L2**2) / (2 * self.L1 * self.L2)
        D = np.clip(D, -1, 1)
        
        joint2 = math.degrees(math.atan2(math.sqrt(1 - D**2), D))
        
        joint1 = math.degrees(math.atan2(z, r) - math.atan2(
            self.L2 * math.sin(math.radians(joint2)),
            self.L1 + self.L2 * math.cos(math.radians(joint2))
        ))
        
        return base_angle, joint1, joint2, 0


class SortingPlanner:
    """分拣规划器"""
    
    # 分拣目标位置
    SORT_POSITIONS = {
        1: (200, 100, 50),   # 红色物品放置位置
        2: (200, -100, 50),  # 绿色
        3: (-200, 100, 50),  # 蓝色
        4: (-200, -100, 50), # 黄色
    }
    
    def __init__(self):
        self.arm = RoboticArmKinematics()
        self.pick_height = 30  # 抓取高度(mm)
        self.place_height = 60  # 放置高度
        
    def plan_pick_and_place(self, item):
        """
        规划抓取和放置动作序列
        item: (x_mm, y_mm, type, angle, shape)
        返回: 动作序列
        """
        x, y, item_type, angle, shape = item
        
        actions = []
        
        # 1. 移动到物品上方
        angles = self.arm.inverse_kinematics(x, y, self.pick_height + 50)
        actions.append(('move', angles))
        
        # 2. 下降到抓取高度
        angles = self.arm.inverse_kinematics(x, y, self.pick_height)
        actions.append(('move', angles))
        
        # 3. 抓取
        actions.append(('grip', True))
        
        # 4. 提起
        angles = self.arm.inverse_kinematics(x, y, self.pick_height + 50)
        actions.append(('move', angles))
        
        # 5. 移动到目标位置上方
        target = self.SORT_POSITIONS.get(item_type, (0, 0, 50))
        angles = self.arm.inverse_kinematics(*target)
        actions.append(('move', angles))
        
        # 6. 下降放置
        angles = self.arm.inverse_kinematics(target[0], target[1], self.place_height)
        actions.append(('move', angles))
        
        # 7. 释放
        actions.append(('grip', False))
        
        # 8. 提起
        angles = self.arm.inverse_kinematics(target[0], target[1], self.place_height + 50)
        actions.append(('move', angles))
        
        # 9. 回到待命位
        actions.append(('move', (0, 45, -45, 0)))
        
        return actions


class SortingSystem:
    """分拣系统主控制器"""
    
    def __init__(self):
        self.vision = ItemRecognitionVision(camera_id=0)
        self.planner = SortingPlanner()
        self.ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
        
        # 状态
        self.current_action_idx = 0
        self.current_actions = None
        self.state = 'IDLE'  # IDLE, PICKING, MOVING, PLACING
        
    def send_arm_command(self, base, joint1, joint2, gripper):
        """发送机械臂指令到STM32"""
        data = struct.pack('<hhhh', 
                          int(base * 10), int(joint1 * 10),
                          int(joint2 * 10), int(gripper))
        self.ser.write(b'\xAA' + data + b'\x55')
    
    def send_gripper(self, grip):
        """控制夹爪"""
        cmd = 1 if grip else 0
        data = struct.pack('<B', cmd)
        self.ser.write(b'\xBB' + data + b'\x55')
    
    def run(self):
        """主循环"""
        print("物品分拣系统启动")
        
        try:
            while True:
                # 检测物品
                items = self.vision.detect_items()
                
                if items and self.state == 'IDLE':
                    # 选择最近的物品
                    items.sort(key=lambda i: math.sqrt(i[0]**2 + i[1]**2))
                    target_item = items[0]
                    
                    print(f"发现物品: 类型={target_item[2]}, "
                          f"位置=({target_item[0]:.0f},{target_item[1]:.0f})")
                    
                    # 规划动作
                    self.current_actions = self.planner.plan_pick_and_place(target_item)
                    self.current_action_idx = 0
                    self.state = 'EXECUTING'
                
                if self.state == 'EXECUTING':
                    if self.current_action_idx < len(self.current_actions):
                        action = self.current_actions[self.current_action_idx]
                        
                        if action[0] == 'move':
                            base, j1, j2, grip = action[1]
                            self.send_arm_command(base, j1, j2, grip)
                            time.sleep(1)  # 等待运动完成
                        elif action[0] == 'grip':
                            self.send_gripper(action[1])
                            time.sleep(0.5)
                        
                        self.current_action_idx += 1
                    else:
                        self.state = 'IDLE'
                        print("分拣完成")
                
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break
                    
        finally:
            self.vision.release()
            self.ser.close()


def main():
    system = SortingSystem()
    system.run()
```

## 接线说明

| Orange Pi 5引脚 | 连接目标 | 说明 |
|-----------------|---------|------|
| CSI接口 | 摄像头 | 俯视检测物品 |
| UART3_TX | STM32 RX | 机械臂控制指令 |
| UART3_RX | STM32 TX | 状态反馈 |
| 5V | STM32 | 供电 |
| GND | 公共地 | - |

## 调试要点

1. **颜色标定**：在实际光照下标定各颜色HSV范围
2. **机械臂标定**：准确测量连杆长度和零位偏移
3. **抓取精度**：吸盘/夹爪的对准和释放时机
4. **传送带同步**：如物品在传送带上，需补偿运动
