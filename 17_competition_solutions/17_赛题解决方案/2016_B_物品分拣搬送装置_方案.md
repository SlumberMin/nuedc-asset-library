# 2016年B题 物品分拣搬送装置 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个物品分拣搬送装置。能够识别不同颜色/形状的物品，将其分拣到指定位置。装置需自动完成物品识别、抓取、搬运和放置全过程。

### 1.2 关键指标
- 物品类型：3~5种不同颜色/形状
- 分拣速度：越快越好
- 识别准确率：>95%
- 搬运距离：30~50cm
- 放置精度：±5mm

### 1.3 技术难点
1. **快速识别**：运动中的物品需快速准确识别
2. **精准抓取**：机械臂需准确定位物品位置
3. **协调控制**：视觉、运动、抓取需协调配合
4. **多物品处理**：需处理多个物品的队列管理

## 四、视觉识别算法

### 4.1 物品检测与分类
```python
import cv2
import numpy as np

class ItemDetector:
    def __init__(self):
        # 颜色定义 (HSV范围)
        self.color_ranges = {
            'red': {
                'lower1': np.array([0, 100, 100]),
                'upper1': np.array([10, 255, 255]),
                'lower2': np.array([160, 100, 100]),
                'upper2': np.array([180, 255, 255])
            },
            'blue': {
                'lower': np.array([100, 100, 100]),
                'upper': np.array([130, 255, 255])
            },
            'green': {
                'lower': np.array([35, 100, 100]),
                'upper': np.array([85, 255, 255])
            },
            'yellow': {
                'lower': np.array([20, 100, 100]),
                'upper': np.array([35, 255, 255])
            }
        }
        
        # 形状检测参数
        self.min_area = 500
        self.max_area = 50000
    
    def detect_items(self, frame):
        """检测传送带上的物品"""
        # 转换色彩空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 创建传送带区域掩码（排除边缘）
        h, w = frame.shape[:2]
        roi_mask = np.zeros((h, w), dtype=np.uint8)
        cv2.rectangle(roi_mask, (50, 100), (w-50, h-100), 255, -1)
        
        items = []
        
        # 检测每种颜色
        for color_name, ranges in self.color_ranges.items():
            # 创建颜色掩码
            if color_name == 'red':
                mask1 = cv2.inRange(hsv, ranges['lower1'], ranges['upper1'])
                mask2 = cv2.inRange(hsv, ranges['lower2'], ranges['upper2'])
                mask = cv2.bitwise_or(mask1, mask2)
            else:
                mask = cv2.inRange(hsv, ranges['lower'], ranges['upper'])
            
            # 应用ROI掩码
            mask = cv2.bitwise_and(mask, roi_mask)
            
            # 形态学操作
            kernel = np.ones((5, 5), np.uint8)
            mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
            mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
            
            # 查找轮廓
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                            cv2.CHAIN_APPROX_SIMPLE)
            
            # 筛选物品
            for contour in contours:
                area = cv2.contourArea(contour)
                if self.min_area < area < self.max_area:
                    # 计算中心点
                    M = cv2.moments(contour)
                    if M['m00'] > 0:
                        cx = int(M['m10'] / M['m00'])
                        cy = int(M['m01'] / M['m00'])
                        
                        # 检测形状
                        shape = self.detect_shape(contour)
                        
                        items.append({
                            'color': color_name,
                            'shape': shape,
                            'center': (cx, cy),
                            'area': area,
                            'contour': contour
                        })
        
        return items
    
    def detect_shape(self, contour):
        """检测物品形状"""
        # 计算轮廓周长
        perimeter = cv2.arcLength(contour, True)
        
        # 多边形逼近
        epsilon = 0.04 * perimeter
        approx = cv2.approxPolyDP(contour, epsilon, True)
        
        # 根据顶点数判断形状
        vertices = len(approx)
        
        if vertices == 3:
            return 'triangle'
        elif vertices == 4:
            # 判断是否为正方形
            x, y, w, h = cv2.boundingRect(contour)
            aspect_ratio = float(w) / h
            if 0.85 < aspect_ratio < 1.15:
                return 'square'
            else:
                return 'rectangle'
        elif vertices == 5:
            return 'pentagon'
        elif vertices > 5:
            # 判断是否为圆形
            area = cv2.contourArea(contour)
            circularity = 4 * np.pi * area / (perimeter * perimeter)
            if circularity > 0.85:
                return 'circle'
        
        return 'unknown'
    
    def get_item_type(self, item):
        """获取物品类型编码"""
        # 颜色编码
        color_codes = {'red': 0, 'blue': 1, 'green': 2, 'yellow': 3}
        # 形状编码
        shape_codes = {'circle': 0, 'square': 1, 'triangle': 2, 'rectangle': 3}
        
        color_code = color_codes.get(item['color'], 4)
        shape_code = shape_codes.get(item['shape'], 4)
        
        # 综合类型（可根据题目要求调整）
        return color_code  # 或者 color_code * 4 + shape_code
```

### 4.2 物品位置追踪
```python
class ItemTracker:
    def __init__(self):
        self.items = {}  # 已跟踪物品
        self.next_id = 0
        self.max_distance = 50  # 最大匹配距离（像素）
    
    def update(self, detected_items):
        """更新物品追踪状态"""
        # 为每个检测到的物品分配ID
        matched_ids = set()
        
        for item in detected_items:
            best_id = None
            best_distance = self.max_distance
            
            # 尝试匹配已跟踪物品
            for item_id, tracked in self.items.items():
                if item_id in matched_ids:
                    continue
                
                # 计算距离
                dx = item['center'][0] - tracked['center'][0]
                dy = item['center'][1] - tracked['center'][1]
                distance = np.sqrt(dx**2 + dy**2)
                
                if distance < best_distance:
                    best_distance = distance
                    best_id = item_id
            
            if best_id is not None:
                # 匹配成功，更新位置
                self.items[best_id].update(item)
                self.items[best_id]['last_seen'] = cv2.getTickCount()
                matched_ids.add(best_id)
            else:
                # 新物品
                item['id'] = self.next_id
                item['first_seen'] = cv2.getTickCount()
                item['last_seen'] = cv2.getTickCount()
                self.items[self.next_id] = item
                self.next_id += 1
        
        # 清除长时间未见的物品
        current_time = cv2.getTickCount()
        to_remove = []
        for item_id, item in self.items.items():
            if (current_time - item['last_seen']) / cv2.getTickFrequency() > 2.0:
                to_remove.append(item_id)
        
        for item_id in to_remove:
            del self.items[item_id]
        
        return self.items
    
    def predict_position(self, item_id, dt):
        """预测物品未来位置"""
        if item_id not in self.items:
            return None
        
        item = self.items[item_id]
        
        # 如果有速度信息，进行预测
        if 'velocity' in item:
            predicted_x = item['center'][0] + item['velocity'][0] * dt
            predicted_y = item['center'][1] + item['velocity'][1] * dt
            return (int(predicted_x), int(predicted_y))
        
        return item['center']
```

```python
import numpy as np
import math

class RobotArm:
    def __init__(self, link_lengths=[120, 100, 80]):
        """
        link_lengths: 各连杆长度 (mm)
        """
        self.L1, self.L2, self.L3 = link_lengths
        
        # 舵机角度范围
        self.servo_limits = {
            'base': (-90, 90),
            'shoulder': (0, 180),
            'elbow': (0, 180),
            'gripper': (0, 90)
        }
    
    def inverse_kinematics(self, target_x, target_y, target_z):
        """
        逆运动学求解
        target: 目标位置 (x, y, z) mm
        返回: 各关节角度 (base, shoulder, elbow)
        """
        # 底座角度
        base_angle = math.degrees(math.atan2(target_y, target_x))
        
        # 在垂直平面内求解
        r = math.sqrt(target_x**2 + target_y**2)
        z = target_z
        
        # 考虑夹爪长度
        r_eff = r - self.L3 * math.cos(math.radians(30))  # 假设夹爪角度30°
        z_eff = z - self.L3 * math.sin(math.radians(30))
        
        # 二连杆逆运动学
        D = (r_eff**2 + z_eff**2 - self.L1**2 - self.L2**2) / (2 * self.L1 * self.L2)
        
        if abs(D) > 1:
            return None  # 无解
        
        # 肘关节角度
        elbow_angle = math.degrees(math.acos(D))
        
        # 肩关节角度
        alpha = math.atan2(z_eff, r_eff)
        beta = math.atan2(self.L2 * math.sin(math.radians(elbow_angle)),
                          self.L1 + self.L2 * math.cos(math.radians(elbow_angle)))
        shoulder_angle = math.degrees(alpha + beta)
        
        return base_angle, shoulder_angle, elbow_angle
    
    def move_to_position(self, x, y, z, gripper_angle=30):

                   min(self.servo_limits['base'][1], base))

                    min(self.servo_limits['elbow'][1], elbow))
        
        # 发送舵机指令
```

```python
class CoordinateCalibrator:
    def __init__(self):
        self.camera_to_world_matrix = None
        self.world_to_arm_matrix = None
    
    def calibrate_camera_to_world(self, image_points, world_points):
        """
        标定相机坐标系到世界坐标系的转换
        image_points: 图像中的点 [(x, y), ...]
        world_points: 实际世界坐标 [(X, Y), ...]
        """
        # 使用仿射变换
        image_pts = np.float32(image_points[:3])
        world_pts = np.float32(world_points[:3])
        
        self.camera_to_world_matrix = cv2.getAffineTransform(image_pts, world_pts)
        
        return self.camera_to_world_matrix
    
    def image_to_world(self, image_point):
        """将图像坐标转换为世界坐标"""
        if self.camera_to_world_matrix is None:
            return None
        
        pt = np.array([image_point[0], image_point[1], 1])
        world = self.camera_to_world_matrix.dot(pt)
        
        return (world[0], world[1])
    
    def calibrate_with_chessboard(self, frame, square_size_mm=20):
        """使用棋盘格进行标定"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 查找棋盘格角点
        ret, corners = cv2.findChessboardCorners(gray, (7, 6), None)
        
        if ret:
            # 亚像素精确化
            criteria = (cv2.TERM_CRITERIA_EPS + cv2.TERM_CRITERIA_MAX_ITER, 30, 0.001)
            corners = cv2.cornerSubPix(gray, corners, (11, 11), (-1, -1), criteria)
            
            # 生成世界坐标
            world_points = []
            for i in range(6):
                for j in range(7):
                    world_points.append((j * square_size_mm, i * square_size_mm))
            
            # 标定
            image_points = [tuple(c[0]) for c in corners]
            self.calibrate_camera_to_world(image_points, world_points)
            
            return True
        
        return False
```

```python
from queue import PriorityQueue
import threading

class SortingScheduler:
    def __init__(self, arm_controller, num_bins=5):
        self.arm = arm_controller
        self.num_bins = num_bins
        
        # 分拣箱位置
        self.bin_positions = {
            0: (200, -150, 50),   # 红色
            1: (200, -75, 50),    # 蓝色
            2: (200, 0, 50),      # 绿色
            3: (200, 75, 50),     # 黄色
            4: (200, 150, 50)     # 其他
        }
        
        # 任务队列
        self.task_queue = PriorityQueue()
        self.running = False
        
        # 统计
        self.sort_count = {i: 0 for i in range(num_bins)}
    
    def add_task(self, item_type, item_position, priority=0):
        """添加分拣任务"""
        self.task_queue.put((priority, item_type, item_position))
    
    def start(self):
        """启动调度器"""
        self.running = True
        self.thread = threading.Thread(target=self._run_scheduler)
        self.thread.start()
    
    def stop(self):
        """停止调度器"""
        self.running = False
        self.thread.join()
    
    def _run_scheduler(self):
        """调度器主循环"""
        while self.running:
            if not self.task_queue.empty():
                # 获取任务
                priority, item_type, item_pos = self.task_queue.get()
                
                # 执行分拣
                success = self._execute_sort(item_type, item_pos)
                
                if success:
                    self.sort_count[item_type] += 1
                    print(f"分拣成功: 类型{item_type}, 当前计数: {self.sort_count}")
                else:
                    print(f"分拣失败: 类型{item_type}")
            
            time.sleep(0.01)
    
    def _execute_sort(self, item_type, item_position):
        """执行单次分拣任务"""
        # 获取目标位置
        if item_type not in self.bin_positions:
            item_type = self.num_bins - 1  # 默认放到最后的箱子
        
        target_pos = self.bin_positions[item_type]
        
        # 转换坐标（图像坐标 → 机械臂坐标）
        arm_pos = self.camera_to_arm_coords(item_position)
        
        if arm_pos is None:
            return False
        
        # 执行抓取和放置
        try:
            # 抓取
            self.arm.grab_item(arm_pos[0], arm_pos[1], arm_pos[2])
            
            # 移动到目标
            self.arm.place_item(target_pos[0], target_pos[1], target_pos[2])
            
            return True
        except Exception as e:
            print(f"执行错误: {e}")
            return False
    
    def camera_to_arm_coords(self, camera_point):
        """将相机坐标转换为机械臂坐标"""
        # 需要根据标定结果实现
        # 这里使用简化的线性映射
        
        # 假设相机视野中心对应机械臂工作空间中心
        # 需要实际标定
        x_scale = 0.5  # mm/pixel
        y_scale = 0.5
        
        x = camera_point[0] * x_scale
        y = camera_point[1] * y_scale
        z = 10  # 固定抓取高度
        
        return (x, y, z)
```

```python
        self.detector = ItemDetector()
        self.tracker = ItemTracker()
        self.arm = RobotArm()
        self.scheduler = SortingScheduler(self.arm)
        
        # 标定
        self.calibrated = False
    
    def calibrate_system(self):
        """系统标定"""
        print("开始系统标定...")
        
        # 采集标定图像
        frame = self.camera.capture()
        
        # 标定相机到世界坐标
        calibrator = CoordinateCalibrator()
        success = calibrator.calibrate_with_chessboard(frame)
        
        if success:
            self.calibrated = True
            print("标定完成")
        else:
            print("标定失败，请检查棋盘格")
    
    def run(self):
        """主运行循环"""
        # 标定
        self.calibrate_system()
        
        # 启动调度器
        self.scheduler.start()
        
        print("分拣系统启动，按'q'退出")
        
        while True:
            # 获取图像
            frame = self.camera.capture()
            
            # 检测物品
            items = self.detector.detect_items(frame)
            
            # 更新追踪
            tracked_items = self.tracker.update(items)
            
            # 在图像上绘制检测结果
            for item_id, item in tracked_items.items():
                # 绘制边界框
                if 'contour' in item:
                    cv2.drawContours(frame, [item['contour']], -1, (0, 255, 0), 2)
                
                # 绘制中心点
                cv2.circle(frame, item['center'], 5, (0, 0, 255), -1)
                
                # 绘制标签
                label = f"{item['color']} {item['shape']}"
                cv2.putText(frame, label, 
                           (item['center'][0] - 20, item['center'][1] - 20),
                           cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 2)
                
                # 如果物品进入抓取区域，添加到任务队列
                if self.is_in_grab_zone(item['center']):
                    item_type = self.detector.get_item_type(item)
                    self.scheduler.add_task(item_type, item['center'])
            
            # 显示统计信息
            self.draw_statistics(frame)
            
            # 显示图像
            cv2.imshow('Sorting System', frame)
            
            # 处理按键
            key = cv2.waitKey(1)
            if key == ord('q'):
                break
            elif key == ord('c'):
                self.calibrate_system()
        
        # 清理
        self.scheduler.stop()
        cv2.destroyAllWindows()
    
    def is_in_grab_zone(self, position):
        """检查物品是否在抓取区域内"""
        x, y = position
        
        # 定义抓取区域
        grab_zone = (400, 200, 800, 500)  # (x1, y1, x2, y2)
        
        return grab_zone[0] <= x <= grab_zone[2] and grab_zone[1] <= y <= grab_zone[3]
    
    def draw_statistics(self, frame):
        """绘制统计信息"""
        h, w = frame.shape[:2]
        
        # 背景
        cv2.rectangle(frame, (10, 10), (200, 150), (0, 0, 0), -1)
        
        # 显示计数
        y_offset = 30
        for i, count in enumerate(self.scheduler.sort_count.items()):
            text = f"Type {count[0]}: {count[1]}"
            cv2.putText(frame, text, (20, y_offset), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 1)
            y_offset += 25

if __name__ == '__main__':
    system = SortingSystem()
    system.run()
```
