## 二、视觉方案（Orange Pi 5）

### 2.1 多策略循线算法

```python
import cv2
import numpy as np
from picamera2 import Picamera2

class VisualLineFollower:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # 循线参数
        self.roi_y = 320  # ROI起始行
        self.roi_height = 100
        self.threshold = 80  # 二值化阈值
        
        # 历史数据
        self.line_history = []
        self.lost_count = 0
        
    def preprocess_frame(self, frame):
        """图像预处理"""
        # ROI裁剪
        roi = frame[self.roi_y:self.roi_y + self.roi_height, :]
        
        # 转灰度
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        
        # 高斯模糊去噪
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        
        # 自适应阈值（适应不同光照）
        binary = cv2.adaptiveThreshold(
            blurred, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 11, 2
        )
        
        # 形态学处理
        kernel = np.ones((3, 3), np.uint8)
        binary = cv2.morphologyEx(binary, cv2.MORPH_CLOSE, kernel)
        
        return roi, binary
        
    def detect_line_sliding_window(self, binary):
        """滑动窗口法检测车道线"""
        histogram = np.sum(binary[binary.shape[0]//2:, :], axis=0)
        
        # 找左右峰值
        midpoint = len(histogram) // 2
        left_peak = np.argmax(histogram[:midpoint])
        right_peak = np.argmax(histogram[midpoint:]) + midpoint
        
        # 滑动窗口跟踪
        window_height = binary.shape[0] // 4
        left_points = []
        right_points = []
        
        left_x = left_peak
        right_x = right_peak
        
        for i in range(4):
            y_start = binary.shape[0] - (i + 1) * window_height
            y_end = y_start + window_height
            
            # 左窗口
            left_window = binary[y_start:y_end, max(0, left_x-50):left_x+50]
            left_nonzero = np.nonzero(left_window)
            if len(left_nonzero[1]) > 0:
                left_x = max(0, left_x-50) + int(np.mean(left_nonzero[1]))
                left_points.append((left_x, (y_start + y_end) // 2))
            
            # 右窗口
            right_window = binary[y_start:y_end, max(0, right_x-50):right_x+50]
            right_nonzero = np.nonzero(right_window)
            if len(right_nonzero[1]) > 0:
                right_x = max(0, right_x-50) + int(np.mean(right_nonzero[1]))
                right_points.append((right_x, (y_start + y_end) // 2))
        
        return left_points, right_points
        
    def detect_line_simple(self, binary):
        """简单逐行扫描法"""
        mid_row = binary[binary.shape[0] * 3 // 4, :]
        line_positions = np.where(mid_row > 0)[0]
        
        if len(line_positions) > 10:
            # 找到连续的黑线段
            segments = []
            start = line_positions[0]
            for i in range(1, len(line_positions)):
                if line_positions[i] - line_positions[i-1] > 5:
                    segments.append((start, line_positions[i-1]))
                    start = line_positions[i]
            segments.append((start, line_positions[-1]))
            
            if len(segments) == 1:
                # 单线
                line_center = (segments[0][0] + segments[0][1]) // 2
                return 'SINGLE', line_center - 320
            elif len(segments) >= 2:
                # 双线（十字路口或弯道）
                line_center = (segments[0][1] + segments[-1][0]) // 2
                return 'DOUBLE', line_center - 320
        
        return 'LOST', 0
        
    def detect_intersection(self, binary):
        """检测路口类型"""
        h, w = binary.shape
        
        # 水平扫描线
        h_line = binary[h//2, :]
        h_count = np.sum(h_line > 0)
        
        # 垂直扫描线
        v_left = binary[:, w//4]
        v_right = binary[:, 3*w//4]
        v_left_count = np.sum(v_left > 0)
        v_right_count = np.sum(v_right > 0)
        
        has_forward = h_count > w * 0.6
        has_left = v_left_count > h * 0.4
        has_right = v_right_count > h * 0.4
        
        if has_forward and has_left and has_right:
            return 'CROSS'  # 十字路口
        elif has_forward and (has_left or has_right):
            return 'T'      # T字路口
        elif has_left or has_right:
            return 'L'      # L弯
        else:
            return 'NONE'
            
    def get_steering(self):
        """获取转向控制量"""
        frame = self.picam2.capture_array()
        roi, binary = self.preprocess_frame(frame)
        
        line_type, offset = self.detect_line_simple(binary)
        
        if line_type == 'LOST':
            self.lost_count += 1
            if self.lost_count > 10:
                # 使用历史数据
                if self.line_history:
                    offset = self.line_history[-1]
                else:
                    return 0, False, 'LOST'
        else:
            self.lost_count = 0
            self.line_history.append(offset)
            if len(self.line_history) > 10:
                self.line_history.pop(0)
        
        intersection = self.detect_intersection(binary)
        
        return offset, line_type != 'LOST', intersection
```

### 2.2 弯道检测与减速

```python
class CurveDetector:
    def __init__(self):
        self.prev_offsets = []
        
    def detect_curve(self, current_offset):
        """检测弯道"""
        self.prev_offsets.append(current_offset)
        if len(self.prev_offsets) > 20:
            self.prev_offsets.pop(0)
            
        if len(self.prev_offsets) < 10:
            return False, 0
            
        # 计算偏移量变化率
        offsets = np.array(self.prev_offsets)
        derivative = np.diff(offsets)
        
        # 变化率持续增大表示弯道
        avg_derivative = np.mean(np.abs(derivative[-5:]))
        
        is_curve = avg_derivative > 5  # 阈值
        curve_intensity = min(avg_derivative / 20, 1.0)
        
        return is_curve, curve_intensity
```

### 3.1 STM32运动控制

```c
// PID控制器
typedef struct {
    float kp, ki, kd;
    float integral, prev_error;
    float max_integral;
} PID_t;

int16_t PID_Compute(PID_t *pid, float error) {
    pid->integral += error;
    if (pid->integral > pid->max_integral) pid->integral = pid->max_integral;
    if (pid->integral < -pid->max_integral) pid->integral = -pid->max_integral;
    
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    return (int16_t)(pid->kp * error + pid->ki * pid->integral + pid->kd * derivative);
}

// 运动控制
void Motion_Control(int16_t forward, int16_t steering) {
    int16_t left = forward + steering;
    int16_t right = forward - steering;
    
    left = CLAMP(left, -MAX_SPEED, MAX_SPEED);
    right = CLAMP(right, -MAX_SPEED, MAX_SPEED);
    
    Motor_SetSpeed(left, right);
}

// 路口处理
void Intersection_Handler(IntersectionType_t type) {
```

### 3.2 自适应速度控制

```python
class SpeedController:
    def __init__(self):
        self.base_speed = 120
        self.max_speed = 200
        self.min_speed = 60
        
    def get_speed(self, offset, is_curve, curve_intensity):
        """根据路况自适应调速"""
        # 基础速度
        speed = self.base_speed
        
        # 弯道减速
        if is_curve:
            speed = speed * (1 - curve_intensity * 0.5)
            
        # 偏移量大时减速
        offset_factor = abs(offset) / 200  # 归一化
        speed = speed * (1 - offset_factor * 0.3)
        
        return int(np.clip(speed, self.min_speed, self.max_speed))
```

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2016年C题 - 自动循迹小车 主程序（视觉方案）
"""
import cv2
import numpy as np
import serial
import time

class AutoTracingCar:
    def __init__(self):
        self.line_follower = VisualLineFollower()
        self.curve_detector = CurveDetector()
        self.speed_ctrl = SpeedController()
        
        self.ser = serial.Serial('/dev/ttyS5', 115200, timeout=0.01)
        
        # PID参数
        self.kp = 0.6
        self.ki = 0.005
        self.kd = 0.3
        self.integral = 0
        self.prev_error = 0
        
        self.running = True
        
    def send_command(self, left, right):
        """发送电机控制命令"""
        left = int(np.clip(left, -255, 255))
        right = int(np.clip(right, -255, 255))
        
        # 编码：0xAA, left_sign, left_abs, right_sign, right_abs, 0x55
        cmd = bytes([
            0xAA,
            0x01 if left >= 0 else 0x00,
            abs(left),
            0x01 if right >= 0 else 0x00,
            abs(right),
            0x55
        ])
        self.ser.write(cmd)
        
    def pid_compute(self, error):
        """PID计算"""
        self.integral += error
        self.integral = np.clip(self.integral, -100, 100)
        derivative = error - self.prev_error
        self.prev_error = error
        
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        return output
        
    def handle_intersection(self, intersection_type):
        """处理路口"""
        if intersection_type == 'CROSS':
            # 十字路口处理逻辑
            print("检测到十字路口")
            # 可根据任务需求决定直行/转弯
            time.sleep(0.3)
        elif intersection_type == 'T':
            print("检测到T字路口")
            time.sleep(0.3)
        elif intersection_type == 'L':
            print("检测到L弯")
            
    def run(self):
        print("自动循迹小车启动...")
        
        while self.running:
            # 获取循迹数据
            offset, detected, intersection = self.line_follower.get_steering()
            
            if detected:
                # PID计算转向
                steering = self.pid_compute(offset)
                
                # 检测弯道
                is_curve, curve_intensity = self.curve_detector.detect_curve(offset)
                
                # 自适应速度
                forward = self.speed_ctrl.get_speed(offset, is_curve, curve_intensity)
                
                # 计算左右轮速
                left = forward + steering
                right = forward - steering
                
                self.send_command(left, right)
                
                # 处理路口
                if intersection != 'NONE':
                    self.handle_intersection(intersection)
                    
                print(f"偏移:{offset:.0f} 转向:{steering:.1f} "
                      f"速度:{forward} 弯道:{is_curve}")
            else:
                # 丢失线，慢速直行搜索
                self.send_command(40, 40)
                self.integral = 0
                print("丢失目标，慢速搜索...")
                
            time.sleep(0.02)  # 50Hz控制频率

if __name__ == '__main__':
    car = AutoTracingCar()
    try:
        car.run()
    except KeyboardInterrupt:
        car.send_command(0, 0)
        print("循迹小车停止")
```

## 七、调试要点

1. **阈值调整**：不同地面颜色和光照条件下测试二值化
2. **ROI选择**：调整扫描区域高度，平衡前瞻性和稳定性
3. **PID整定**：先调Kp保证基本循线，再调Kd改善动态响应
4. **弯道策略**：提前减速，避免冲出赛道
5. **丢失恢复**：设置合理的搜索策略，快速找回线路
