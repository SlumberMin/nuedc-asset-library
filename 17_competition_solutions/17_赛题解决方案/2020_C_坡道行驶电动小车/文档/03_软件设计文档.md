## 二、核心模块

### 2.1 坡度检测（互补滤波）

```c
typedef struct {
    float pitch;       // 俯仰角(坡度)
    float roll;        // 横滚角
    float accel_x, accel_y, accel_z;
    float gyro_x, gyro_y, gyro_z;
} IMU_Data_t;

float ComplementaryFilter(IMU_Data_t *imu, float dt) {
    // 加速度计计算俯仰角
    float accel_pitch = atan2(imu->accel_y, 
        sqrt(imu->accel_x * imu->accel_x + 
             imu->accel_z * imu->accel_z));
    accel_pitch *= 180.0f / M_PI;
    
    // 互补滤波: 98%陀螺仪 + 2%加速度计
    static float filtered_pitch = 0;
    filtered_pitch = 0.98f * (filtered_pitch + imu->gyro_y * dt) 
                   + 0.02f * accel_pitch;
    
    return filtered_pitch;
}
```

### 2.2 速度控制PID

```c
typedef struct {
    float Kp, Ki, Kd;
    float integral, prev_error;
    float integral_max;
} PID_t;

// 增量式PID (更适合速度控制)
float PID_Incremental(PID_t *pid, float target, float actual) {
    float error = target - actual;
    
    float increment = pid->Kp * (error - pid->prev_error)
                    + pid->Ki * error
                    + pid->Kd * (error - 2 * pid->prev_error + pid->integral);
    
    pid->prev_error = error;
    pid->integral = increment;
    
    return increment;
}
```

### 2.3 坡度自适应控制

```c
float GetAdaptiveTarget(float base_speed, float slope_angle) {
    float target = base_speed;
    
    // 上坡补偿: 每度增加2%速度
    if (slope_angle > 3.0f) {
        target += slope_angle * 2.0f;
    }
    // 下坡减速: 每度减少1%速度
    else if (slope_angle < -3.0f) {
        target += slope_angle * 1.0f;
    }
    
    // 限幅
    if (target > 200.0f) target = 200.0f;
    if (target < 50.0f) target = 50.0f;
    
    return target;
}
```

### 2.4 防滑控制

```c
typedef struct {
    float target_speed;
    float actual_speed;
    float slope_angle;
    uint8_t slip_detected;
} AntiSlip_t;

void AntiSlipControl(AntiSlip_t *ctrl) {
    ctrl->slip_detected = 0;
    
    // 检测打滑: 目标速度>实际速度×1.5 且 在坡道上
    if (ctrl->target_speed > ctrl->actual_speed * 1.5f 
        && fabs(ctrl->slope_angle) > 5.0f) {
        ctrl->slip_detected = 1;
        
        // 降低目标速度
        ctrl->target_speed *= 0.7f;
        
        // 限制加速度
        Motor_SetAcceleration(LOW_ACCEL);
    }
}
```

### 2.5 主程序

```c
PID_t g_speed_pid = {5.0f, 1.0f, 0.5f, 0, 0, 200};
PID_t g_track_pid = {2.0f, 0.3f, 1.0f, 0, 0, 100};

volatile float g_target_speed = 0.5f;  // m/s

int main(void) {
    HAL_Init(); SystemClock_Config();
    GPIO_Init(); TIM_Init(); Encoder_Init();
    OLED_Init(); Motor_Init(); MPU6050_Init();
    
    while(1) {
        // 1. 编码器测速
        float actual_speed = Encoder_GetSpeed();
        
        // 2. MPU6050读取坡度
        IMU_Data_t imu;
        MPU6050_ReadAll(&imu);
        float slope = ComplementaryFilter(&imu, 0.01f);
        
        // 3. 坡度自适应速度
        float target = GetAdaptiveTarget(g_target_speed, slope);
        
        // 4. 防滑检测
        AntiSlip_t anti_slip = {target, actual_speed, slope, 0};
        AntiSlipControl(&anti_slip);
        target = anti_slip.target_speed;
        
        // 5. 速度PID
        float speed_cmd = PID_Calc(&g_speed_pid, target, actual_speed);
        
        // 6. 循线PID
        int16_t pos = Vision_GetLineOffset();
        float track_cmd = PID_Calc(&g_track_pid, 0, pos);
        
        // 7. 合成输出
        int16_t left = (int16_t)speed_cmd + (int16_t)track_cmd;
        int16_t right = (int16_t)speed_cmd - (int16_t)track_cmd;
        Motor_SetSpeed(left, right);
        
        // 8. OLED显示
        char buf[32];
        sprintf(buf, "Spd:%.1f Slp:%.1f", actual_speed, slope);
        OLED_ShowString(0, 2, buf);
        
        HAL_Delay(10);
    }
}
```

## 三、PID参数整定

### 3.1 速度环PID

```
步骤1: Ki=Kd=0, Kp从1.0开始
步骤2: 增大Kp至速度出现振荡, 取60%
步骤3: 加入Ki消除稳态误差
步骤4: 加入Kd抑制超调

推荐初始值: Kp=5.0, Ki=1.0, Kd=0.5
```

### 3.2 方向环PID

```
循线PID, 响应应快于速度环
推荐初始值: Kp=2.0, Ki=0.3, Kd=1.0
```
