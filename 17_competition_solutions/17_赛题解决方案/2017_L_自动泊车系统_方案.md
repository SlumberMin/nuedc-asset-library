# 2017年电赛L题 - 自动泊车系统 Orange Pi 5解决方案

## 一、题目分析

自动泊车系统需要小车在停车场环境中自主识别空闲车位，规划最优泊车路径，并执行精确的自动泊车动作。

### 核心要求
1. 视觉识别车位（空闲/占用状态判断）
2. 路径规划（平行泊车/垂直泊车）
3. 精确运动控制（前进/后退/转向协同）
4. 入库精度控制

### 4.1 视觉车位识别

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class ParkingSpotDetector:
    """基于MIPI摄像头的车位检测器"""

    def __init__(self):
        self.rknn = RKNNLite()
        self.rknn.load_rknn('parking_yolov5s.rknn')
        self.rknn.init_runtime()

        # 相机标定参数
        self.camera_matrix = np.load('camera_matrix.npy')
        self.dist_coeffs = np.load('dist_coeffs.npy')
        # 逆透视变换矩阵 (鸟瞰图)
        self.M_inv = self._compute_ipm_matrix()

    def _compute_ipm_matrix(self):
        """计算逆透视变换(IPM)矩阵"""
        # 原图四点 (地面停车位标记)
        src = np.float32([[100, 300], [540, 300], [620, 480], [20, 480]])
        # 目标鸟瞰图四点
        dst = np.float32([[100, 50], [540, 50], [540, 430], [100, 430]])
        return cv2.getPerspectiveTransform(src, dst)

    def get_bird_eye_view(self, frame):
        """获取鸟瞰图"""
        h, w = frame.shape[:2]
        return cv2.warpPerspective(frame, self.M_inv, (w, h))

    def detect_parking_spots(self, frame):
        """
        检测车位状态
        返回: [(x, y, w, h, status, confidence), ...]
        status: 'empty' / 'occupied'
        """
        bird_view = self.get_bird_eye_view(frame)

        # 方案A: YOLOv5s RKNN推理
        input_data = cv2.resize(bird_view, (640, 640))
        input_data = np.expand_dims(input_data, axis=0)
        outputs = self.rknn.inference(inputs=[input_data])
        detections = self._postprocess_yolo(outputs)

        # 方案B: 传统视觉 - 车位线检测 (备用)
        line_spots = self._detect_by_lines(bird_view)

        # 融合结果
        return self._merge_results(detections, line_spots)

    def _detect_by_lines(self, bird_view):
        """传统方法: 基于车位线的车位检测"""
        gray = cv2.cvtColor(bird_view, cv2.COLOR_BGR2GRAY)
        # 自适应阈值二值化
        binary = cv2.adaptiveThreshold(
            gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
            cv2.THRESH_BINARY_INV, 11, 2
        )
        # 检测车位矩形区域
        contours, _ = cv2.findContours(
            binary, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE
        )

        spots = []
        for cnt in contours:
            area = cv2.contourArea(cnt)
            if 5000 < area < 50000:  # 过滤面积
                x, y, w, h = cv2.boundingRect(cnt)
                aspect = w / h
                if 0.4 < aspect < 2.5:  # 车位长宽比
                    roi = bird_view[y:y+h, x:x+w]
                    # 判断是否空闲 (内部颜色/纹理分析)
                    status = self._check_spot_empty(roi)
                    spots.append((x, y, w, h, status, 0.8))
        return spots

    def _check_spot_empty(self, roi):
        """判断车位是否空闲"""
        hsv = cv2.cvtColor(roi, cv2.COLOR_BGR2HSV)
        # 空车位地面颜色较均匀(灰色/白色)
        # 有车时颜色多样、纹理复杂
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        texture_var = cv2.Laplacian(gray, cv2.CV_64F).var()
        color_std = np.std(hsv[:, :, 0])

        if texture_var < 500 and color_std < 30:
            return 'empty'
        return 'occupied'
```

### 4.2 路径规划

```python
import numpy as np
import math

class ParkingPathPlanner:
    """泊车路径规划器"""

    def __init__(self, car_length=0.30, car_width=0.18,
                 wheel_base=0.22, max_steer=35):
        self.L = car_length
        self.W = car_width
        self.L_wb = wheel_base
        self.max_steer = math.radians(max_steer)

    def plan_parallel_parking(self, spot, car_pose):
        """
        平行泊车路径规划 (三段弧法)
        spot: (x, y, w, h) 车位位置
        car_pose: (x, y, theta) 当前位姿
        返回: 路径点列表 [(x, y, theta, steer, speed), ...]
        """
        sx, sy, sw, sh = spot
        cx, cy, ctheta = car_pose

        # 车位中心点
        spot_center_x = sx + sw / 2
        spot_center_y = sy + sh / 2

        # 三段弧规划
        R_min = self.L_wb / math.tan(self.max_steer)

        # 第一段: 前进右转, 接近车位
        path = []
        R1 = R_min * 1.2
        arc1 = self._generate_arc(
            cx, cy, ctheta, R1, -1,
            math.radians(45), direction=1
        )
        path.extend(arc1)

        # 第二段: 后退左转, 进入车位
        end_x, end_y, end_theta = arc1[-1][:3]
        R2 = R_min * 1.1
        arc2 = self._generate_arc(
            end_x, end_y, end_theta + math.pi, R2, 1,
            math.radians(60), direction=-1
        )
        path.extend(arc2)

        # 第三段: 后退微调, 摆正车身
        end_x, end_y, end_theta = arc2[-1][:3]
        arc3 = self._generate_arc(
            end_x, end_y, end_theta, R2, -1,
            math.radians(25), direction=-1
        )
        path.extend(arc3)

        return path

    def plan_perpendicular_parking(self, spot, car_pose):
        """垂直泊车路径规划"""
        sx, sy, sw, sh = spot
        cx, cy, ctheta = car_pose

        path = []
        R_min = self.L_wb / math.tan(self.max_steer)

        # 第一段: 前进斜向车位
        R = R_min * 1.5
        angle = math.radians(70)
        arc = self._generate_arc(cx, cy, ctheta, R, -1, angle, direction=1)
        path.extend(arc)

        # 第二段: 直行入库
        end_x, end_y, end_theta = arc[-1][:3]
        for i in range(20):
            t = i / 19
            px = end_x + t * 0.15 * math.cos(end_theta)
            py = end_y + t * 0.15 * math.sin(end_theta)
            path.append((px, py, end_theta, 0, 0.3))

        return path

    def _generate_arc(self, x0, y0, theta0, R, direction,
                      total_angle, direction_forward=1):
        """生成圆弧路径段"""
        path = []
        n_points = max(10, int(total_angle / 0.05))

        for i in range(n_points + 1):
            angle = i * total_angle / n_points
            if direction > 0:  # 左转
                cx = x0 - R * math.sin(theta0)
                cy = y0 + R * math.cos(theta0)
            else:  # 右转
                cx = x0 + R * math.sin(theta0)
                cy = y0 - R * math.cos(theta0)

            if direction > 0:
                px = cx + R * math.sin(theta0 + angle)
                py = cy - R * math.cos(theta0 + angle)
            else:
                px = cx - R * math.sin(theta0 - angle)
                py = cy + R * math.cos(theta0 - angle)

            ptheta = theta0 + direction * angle
            steer = math.atan(self.L_wb / R) * direction
            speed = 0.3 * direction_forward
            path.append((px, py, ptheta, steer, speed))

        return path
```

### 4.3 运动控制器 (Pure Pursuit)

```python
import math

class PurePursuitController:
    """纯追踪控制器"""

    def __init__(self, wheel_base=0.22, ld_gain=0.3, ld_min=0.15):
        self.L = wheel_base
        self.k_ld = ld_gain
        self.ld_min = ld_min

    def compute(self, path, car_pose, car_speed):
        """
        计算控制量
        返回: (steer_angle, target_speed)
        """
        cx, cy, ctheta = car_pose

        # 自适应前瞻距离
        ld = max(self.ld_min, self.k_ld * abs(car_speed))

        # 寻找最近路径点
        min_dist = float('inf')
        nearest_idx = 0
        for i, (px, py, _, _, _) in enumerate(path):
            dist = math.hypot(px - cx, py - cy)
            if dist < min_dist:
                min_dist = dist
                nearest_idx = i

        # 寻找前瞻点
        target = None
        for i in range(nearest_idx, len(path)):
            px, py, _, _, sp = path[i]
            dist = math.hypot(px - cx, py - cy)
            if dist >= ld:
                target = (px, py, sp)
                break

        if target is None:
            target = (path[-1][0], path[-1][1], path[-1][4])

        tx, ty, t_speed = target

        # 坐标变换到车体坐标系
        dx = tx - cx
        dy = ty - cy
        local_x = dx * math.cos(ctheta) + dy * math.sin(ctheta)
        local_y = -dx * math.sin(ctheta) + dy * math.cos(ctheta)

        # 纯追踪公式
        if ld < 0.01:
            steer = 0
        else:
            curvature = 2 * local_y / (ld * ld)
            steer = math.atan(curvature * self.L)

        # 限幅
        max_steer = math.radians(35)
        steer = max(-max_steer, min(max_steer, steer))

        return steer, t_speed
```

### 4.4 STM32 协处理器固件

```c
// stm32_parking_ctrl.c - 电机与舵机底层控制
#include "stm32f1xx_hal.h"

// PWM定时器
TIM_HandleTypeDef htim2;  // 电机PWM
TIM_HandleTypeDef htim3;  // 舵机PWM

// 编码器
volatile int32_t encoder_left = 0;
volatile int32_t encoder_right = 0;

typedef struct {
    float target_speed;    // m/s
    float target_steer;    // rad
    float current_speed;
    float current_steer;
} CtrlCmd;

CtrlCmd cmd = {0};

void Motor_SetDuty(int16_t left_duty, int16_t right_duty) {
    // 左电机
    if (left_duty >= 0) {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_RESET);
    } else {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_SET);
        left_duty = -left_duty;
    }
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, left_duty);

    // 右电机
    if (right_duty >= 0) {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_3, GPIO_PIN_RESET);
    } else {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_3, GPIO_PIN_SET);
        right_duty = -right_duty;
    }
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, right_duty);
}

void Servo_SetAngle(float angle_rad) {
    // 舵机: 0.5ms~2.5ms 对应 -45°~+45°
    float angle_deg = angle_rad * 180.0f / 3.14159f;
    float pulse = 1.5f + angle_deg / 45.0f * 1.0f;  // ms
    uint32_t ccr = (uint32_t)(pulse / 20.0f * 20000);  // ARR=19999
    __HAL_TIM_SET_COMPARE(&htim3, TIM_CHANNEL_1, ccr);
}

// UART接收解析
void UART_RxCallback(uint8_t *buf, uint16_t len) {
    // 协议: [0xAA][steer_f32][speed_f32][checksum][0x55]
    if (buf[0] == 0xAA && buf[13] == 0x55) {
        memcpy(&cmd.target_steer, &buf[1], 4);
        memcpy(&cmd.target_speed, &buf[5], 4);
    }
}

// 速度PID
typedef struct {
    float Kp, Ki, Kd;
    float integral, prev_error;
} PID;

float PID_Compute(PID *pid, float error, float dt) {
    pid->integral += error * dt;
    pid->integral = fmaxf(-1.0f, fminf(1.0f, pid->integral));
    float derivative = (error - pid->prev_error) / dt;
    pid->prev_error = error;
    return pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;
}

void Control_Loop(void) {
    static PID pid_speed = {2.0, 0.5, 0.1, 0, 0};
    float dt = 0.01f;  // 10ms

    // 读取编码器速度
    float v_left = encoder_left * 0.001f / dt;   // 转换为m/s
    float v_right = encoder_right * 0.001f / dt;
    float v_avg = (v_left + v_right) / 2.0f;
    encoder_left = 0;
    encoder_right = 0;

    // 速度PID
    float speed_error = cmd.target_speed - v_avg;
    float duty = PID_Compute(&pid_speed, speed_error, dt);
    int16_t duty_int = (int16_t)(duty * 1000);

    // 差速转向补偿
    float steer_diff = cmd.target_steer * 50.0f;
    Motor_SetDuty(duty_int + (int16_t)steer_diff,
                  duty_int - (int16_t)steer_diff);

    Servo_SetAngle(cmd.target_steer);
}
```

### 4.5 主程序

```python
import cv2
import time
import serial
import struct
import math
import numpy as np

from parking_detector import ParkingSpotDetector
from path_planner import ParkingPathPlanner
from pure_pursuit import PurePursuitController

class AutoParkingSystem:
    def __init__(self):
        self.detector = ParkingSpotDetector()
        self.planner = ParkingPathPlanner()
        self.controller = PurePursuitController()

        self.cap = cv2.VideoCapture(0)
        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)

        self.car_x = 0.0
        self.car_y = 0.0
        self.car_theta = 0.0
        self.car_speed = 0.0

        self.state = 'SCANNING'  # SCANNING -> PLANNING -> PARKING -> DONE
        self.target_spot = None
        self.current_path = None
        self.path_idx = 0

    def send_cmd(self, steer, speed):
        """发送控制指令到STM32"""
        buf = bytearray(14)
        buf[0] = 0xAA
        struct.pack_into('<f', buf, 1, steer)
        struct.pack_into('<f', buf, 5, speed)
        buf[13] = 0x55
        self.ser.write(buf)

    def update_odometry(self):
        """里程计更新 (结合IMU)"""
        # 读取STM32反馈的编码器数据
        if self.ser.in_waiting >= 12:
            data = self.ser.read(12)
            vl, vr = struct.unpack('<ff', data[:8])
            dt = 0.01
            v_avg = (vl + vr) / 2.0
            self.car_speed = v_avg
            self.car_theta += (vr - vl) / 0.18 * dt  # 差速模型
            self.car_x += v_avg * math.cos(self.car_theta) * dt
            self.car_y += v_avg * math.sin(self.car_theta) * dt

    def run(self):
        print("=== 自动泊车系统启动 ===")

        while True:
            ret, frame = self.cap.read()
            if not ret:
                continue

            self.update_odometry()

            if self.state == 'SCANNING':
                spots = self.detector.detect_parking_spots(frame)
                for spot in spots:
                    x, y, w, h, status, conf = spot
                    if status == 'empty' and conf > 0.7:
                        self.target_spot = (x, y, w, h)
                        self.state = 'PLANNING'
                        print(f"找到空车位: ({x},{y},{w},{h})")
                        break

                # 未找到车位, 低速巡游
                self.send_cmd(0, 0.2)

            elif self.state == 'PLANNING':
                sx, sy, sw, sh = self.target_spot
                spot_area = sw * sh

                if spot_area > 20000:  # 大车位 -> 垂直泊车
                    self.current_path = self.planner.plan_perpendicular_parking(
                        self.target_spot,
                        (self.car_x, self.car_y, self.car_theta)
                    )
                    print("执行垂直泊车")
                else:  # 小车位 -> 平行泊车
                    self.current_path = self.planner.plan_parallel_parking(
                        self.target_spot,
                        (self.car_x, self.car_y, self.car_theta)
                    )
                    print("执行平行泊车")

                self.path_idx = 0
                self.state = 'PARKING'

            elif self.state == 'PARKING':
                if self.path_idx >= len(self.current_path):
                    self.send_cmd(0, 0)
                    self.state = 'DONE'
                    print("泊车完成!")
                    continue

                steer, speed = self.controller.compute(
                    self.current_path[self.path_idx:],
                    (self.car_x, self.car_y, self.car_theta),
                    self.car_speed
                )
                self.send_cmd(steer, speed)

                # 检查是否到达当前路径点
                target = self.current_path[self.path_idx]
                dist = math.hypot(
                    target[0] - self.car_x,
                    target[1] - self.car_y
                )
                if dist < 0.03:
                    self.path_idx += 1

                # 安全检查 - 超声波避障
                if self._check_obstacle():
                    self.send_cmd(0, 0)
                    print("检测到障碍物, 暂停")

            elif self.state == 'DONE':
                self.send_cmd(0, 0)

            # 可视化
            self._draw_overlay(frame)
            cv2.imshow('Auto Parking', frame)
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break

        self.cap.release()
        self.send_cmd(0, 0)

    def _check_obstacle(self):
        """读取超声波传感器"""
        # GPIO读取前后超声波距离
        return False  # 简化

    def _draw_overlay(self, frame):
        """绘制可视化信息"""
        if self.target_spot:
            x, y, w, h = self.target_spot
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
        cv2.putText(frame, f"State: {self.state}", (10, 30),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 255), 2)
        cv2.putText(frame, f"Pos: ({self.car_x:.2f},{self.car_y:.2f})",
                    (10, 60), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 2)

if __name__ == '__main__':
    system = AutoParkingSystem()
    system.run()
```

### 6.1 关键参数
| 参数 | 推荐值 | 说明 |
|------|--------|------|
| 最小转弯半径 | 0.35m | 由轴距和舵角决定 |
| 泊车速度 | 0.2~0.3 m/s | 低速精确 |
| 巡游速度 | 0.3~0.5 m/s | 寻找车位 |
| 前瞻距离增益 | 0.3 | Pure Pursuit参数 |

### 6.2 常见问题
1. **车位识别不准**: 增加训练数据, 调整置信度阈值
2. **泊车轨迹偏差**: 增加里程计校准频率, 用视觉修正位姿
3. **转向不精确**: 舵机PID校准, 添加死区补偿
