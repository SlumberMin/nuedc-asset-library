# 2022年I题：声源定位跟踪系统 — Orange Pi 5 专属方案

## 题目概述
设计可移动声源定位与跟踪系统，要求系统能够实时定位并跟踪移动中的声源，控制执行机构使指示器始终对准声源。

## 系统架构

```
[MEMS麦克风阵列(4路I2S)] → Orange Pi 5 → [步进电机X/Y轴] → [激光指示器]
[OV5647摄像头] ──────────→↗        ↓
                        [声源定位+视觉确认跟踪]
```

## 核心方案

### 1. 声源定位（与E题差异：移动声源场景）
- **GCC-PHAT + 粒子滤波**：对移动声源做运动估计
- 粒子滤波预测声源下一位置，缩小搜索范围
- 音量阈值检测，过滤噪声干扰

### 2. 视觉辅助跟踪
- 摄像头捕捉场景，检测可能的声源目标（人、扬声器等）
- 声源方位与视觉目标关联匹配
- 遮挡时纯声源跟踪，恢复后视觉修正

### 3. 执行机构（步进电机+激光）
- 2轴步进电机控制激光指向
- 细分驱动提高指向精度

## Orange Pi 5 代码实现

```python
#!/usr/bin/env python3
"""2022I 声源定位跟踪系统 - Orange Pi 5 (移动声源+步进电机)"""

import numpy as np
import cv2
import time
from periphery import GPIO

# ============ 粒子滤波器 ============
class ParticleFilter2D:
    def __init__(self, n_particles=200):
        self.n = n_particles
        self.particles = np.random.uniform(-90, 90, (n_particles, 2))  # [azimuth, elevation]
        self.weights = np.ones(n_particles) / n_particles

    def predict(self, dt, process_noise=2.0):
        self.particles += np.random.randn(self.n, 2) * process_noise * np.sqrt(dt)
        self.particles = np.clip(self.particles, -90, 90)

    def update(self, azimuth, elevation, meas_noise=5.0):
        likelihood = np.exp(-0.5 * (
            ((self.particles[:, 0] - azimuth) / meas_noise) ** 2 +
            ((self.particles[:, 1] - elevation) / meas_noise) ** 2
        ))
        self.weights *= likelihood
        self.weights += 1e-300
        self.weights /= self.weights.sum()

    def resample(self):
        indices = np.random.choice(self.n, self.n, p=self.weights)
        self.particles = self.particles[indices]
        self.weights = np.ones(self.n) / self.n

    def estimate(self):
        return np.average(self.particles, weights=self.weights, axis=0)

    def effective_n(self):
        return 1.0 / np.sum(self.weights ** 2)


# ============ 步进电机控制 ============
class StepperMotor:
    def __init__(self, step_pin, dir_pin, steps_per_rev=200, microstep=16):
        self.step_gpio = GPIO(step_pin, "out")
        self.dir_gpio = GPIO(dir_pin, "out")
        self.steps_per_rev = steps_per_rev * microstep
        self.angle_per_step = 360.0 / self.steps_per_rev
        self.position = 0  # 当前步数

    def step(self, direction, delay_us=500):
        self.dir_gpio.write(direction)
        self.step_gpio.write(True)
        time.sleep(delay_us / 1e6)
        self.step_gpio.write(False)
        time.sleep(delay_us / 1e6)
        self.position += 1 if direction else -1

    def move_to_angle(self, target_angle, speed_rps=1.0):
        target_steps = int(target_angle / self.angle_per_step)
        diff = target_steps - self.position
        direction = diff > 0
        delay = int(1.0 / (self.steps_per_rev * speed_rps) * 1e6)
        for _ in range(abs(diff)):
            self.step(direction, delay)

    def get_angle(self):
        return self.position * self.angle_per_step


# ============ 声源定位（复用GCC-PHAT） ============
class SoundLocalizer:
    def __init__(self, mic_positions, fs=16000):
        self.mic_pos = np.array(mic_positions)
        self.fs = fs
        self.c = 343.0

    def gcc_phat(self, sig1, sig2):
        n = len(sig1) + len(sig2) - 1
        n_pad = 1 << (n - 1).bit_length()
        S1 = np.fft.rfft(sig1, n_pad)
        S2 = np.fft.rfft(sig2, n_pad)
        R = S1 * np.conj(S2)
        R /= (np.abs(R) + 1e-10)
        cc = np.fft.irfft(R)
        max_shift = int(0.5 * self.fs * 0.05 / self.c) + 10
        cc = np.concatenate([cc[-max_shift:], cc[:max_shift + 1]])
        shift = np.argmax(cc) - max_shift
        return shift / self.fs

    def estimate(self, signals):
        taus = []
        pairs = [(0,1),(0,2),(0,3),(1,2),(1,3),(2,3)]
        for i, j in pairs:
            taus.append(self.gcc_phat(signals[i], signals[j]))
        d_lr = np.linalg.norm(self.mic_pos[0] - self.mic_pos[1])
        d_fb = np.linalg.norm(self.mic_pos[2] - self.mic_pos[3])
        az = np.degrees(np.arcsin(np.clip(taus[0] * self.c / d_lr, -1, 1)))
        el = np.degrees(np.arcsin(np.clip(taus[5] * self.c / d_fb, -1, 1)))
        return az, el


# ============ 主程序 ============
def main():
    mic_pos = [[-0.025,0.025,0],[0.025,0.025,0],[0.025,-0.025,0],[-0.025,-0.025,0]]
    localizer = SoundLocalizer(mic_pos)
    pf = ParticleFilter2D(200)

    # 步进电机: X轴(GPIO1_B4), Y轴(GPIO1_B6)
    motor_x = StepperMotor(step_pin=(1,28), dir_pin=(1,29))
    motor_y = StepperMotor(step_pin=(1,30), dir_pin=(1,31))

    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    print("移动声源定位跟踪系统启动...")
    last_time = time.time()

    while True:
        now = time.time()
        dt = now - last_time
        last_time = now

        signals = read_mic_data(4, 1024)
        raw_az, raw_el = localizer.estimate(signals)

        # 音量检测
        volume = np.mean([np.sqrt(np.mean(s**2)) for s in signals])
        if volume < 0.01:
            continue

        # 粒子滤波平滑
        pf.predict(dt)
        pf.update(raw_az, raw_el)
        if pf.effective_n() < pf.n / 2:
            pf.resample()
        est = pf.estimate()

        # 移动步进电机
        motor_x.move_to_angle(est[0], speed_rps=2.0)
        motor_y.move_to_angle(est[1], speed_rps=2.0)

        # 视觉确认
        ret, frame = cap.read()
        if ret:
            cv2.putText(frame, f"AZ:{est[0]:.1f} EL:{est[1]:.1f} Vol:{volume:.3f}",
                        (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0,255,0), 2)
            cv2.imshow("Tracker", frame)
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break

    cap.release()

def read_mic_data(n_ch, n_samp):
    return [np.random.randn(n_samp) for _ in range(n_ch)]

if __name__ == "__main__":
    main()
```

## 硬件接线

| 模块 | Orange Pi 5引脚 | 说明 |
|------|-----------------|------|
| INMP441 I2S | GPIO1_A0~A3 | I2S0接口 |
| X轴步进 STEP | GPIO1_B4 | 脉冲 |
| X轴步进 DIR | GPIO1_B5 | 方向 |
| Y轴步进 STEP | GPIO1_B6 | 脉冲 |
| Y轴步进 DIR | GPIO1_B7 | 方向 |
| 激光模块 | GPIO1_C0 | 开关控制 |
| OV5647 | MIPI-CSI | 摄像头 |

## 与E题差异
| 对比项 | E题 | I题 |
|--------|-----|-----|
| 声源特征 | 静态/慢速 | 移动声源 |
| 跟踪算法 | PID直接跟踪 | 粒子滤波+PID |
| 执行机构 | 舵机PTZ | 步进电机+激光 |
| 指向精度要求 | 一般 | 更高(激光指示) |
