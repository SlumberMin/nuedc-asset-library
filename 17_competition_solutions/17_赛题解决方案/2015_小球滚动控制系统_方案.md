# 2015年电赛 - 小球滚动控制系统 - Orange Pi 5解决方案

## 题目要求

设计制作小球滚动控制系统，实现：
1. 在平板上控制小球运动到指定位置
2. 小球按指定轨迹运动（直线、圆等）
3. 快速到达，小超调，精确定位

## 系统架构

```
┌──────────────┐    CSI/USB     ┌──────────────┐    UART     ┌──────────────┐
│  摄像头俯视   │──────────────→│  Orange Pi 5 │────────────→│  STM32主控   │
│  检测小球位置  │              │  RK3588S     │             │  舵机/步进驱动│
└──────────────┘              │  视觉+双环PID│             └──────┬───────┘
                               └──────────────┘              ┌────┴────┐
                                                              │X/Y轴舵机│
                                                              │倾斜平台  │
                                                              └─────────┘
```

## 核心设计

### 1. 视觉检测小球位置

```python
import cv2
import numpy as np
import time
import struct
import serial

class BallTrackingVision:
    """小球跟踪视觉系统"""
    
    def __init__(self, camera_id=0, board_corners=None):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 90)
        
        # 棋盘/平台四个角点（像素坐标，需标定）
        # 用于透视变换
        if board_corners is None:
            self.board_corners = np.float32([
                [100, 50], [540, 50], [540, 430], [100, 430]
            ])
        else:
            self.board_corners = np.float32(board_corners)
        
        # 平台实际尺寸(mm)
        self.board_width = 300
        self.board_height = 300
        
        # 透视变换目标
        self.dst_corners = np.float32([
            [0, 0], [self.board_width, 0],
            [self.board_width, self.board_height], [0, self.board_height]
        ])
        self.M_persp = cv2.getPerspectiveTransform(self.board_corners, self.dst_corners)
        
        # 小球颜色（红色）
        self.hsv_lower1 = np.array([0, 100, 100])
        self.hsv_upper1 = np.array([10, 255, 255])
        self.hsv_lower2 = np.array([170, 100, 100])
        self.hsv_upper2 = np.array([180, 255, 255])
        
        # 位置历史
        self.pos_history = []
        self.time_history = []
        self.max_history = 15
        
    def detect_ball(self):
        """检测小球位置，返回平台坐标系下的(mm)坐标"""
        ret, frame = self.cap.read()
        if not ret:
            return None, None
            
        # 透视变换
        warped = cv2.warpPerspective(frame, self.M_persp, 
                                     (self.board_width, self.board_height))
        
        hsv = cv2.cvtColor(warped, cv2.COLOR_BGR2HSV)
        
        # 红色小球检测
        mask1 = cv2.inRange(hsv, self.hsv_lower1, self.hsv_upper1)
        mask2 = cv2.inRange(hsv, self.hsv_lower2, self.hsv_upper2)
        mask = cv2.bitwise_or(mask1, mask2)
        
        kernel = np.ones((3, 3), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        # 霍夫圆检测（更鲁棒）
        circles = cv2.HoughCircles(
            cv2.cvtColor(warped, cv2.COLOR_BGR2GRAY),
            cv2.HOUGH_GRADIENT, dp=1, minDist=30,
            param1=100, param2=30, minRadius=5, maxRadius=30
        )
        
        if circles is not None:
            circles = np.uint16(np.around(circles))
            cx, cy, r = circles[0][0]
            
            # 小球中心坐标(mm)
            x_mm = float(cx)
            y_mm = float(cy)
            
            # 记录历史
            now = time.time()
            self.pos_history.append((x_mm, y_mm))
            self.time_history.append(now)
            if len(self.pos_history) > self.max_history:
                self.pos_history.pop(0)
                self.time_history.pop(0)
            
            # 可视化
            cv2.circle(warped, (cx, cy), r, (0, 255, 0), 2)
            cv2.circle(warped, (cx, cy), 3, (0, 0, 255), -1)
            cv2.putText(warped, f"({x_mm:.1f},{y_mm:.1f})", (10, 30),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            
            cv2.imshow('Ball Tracking', warped)
            
            return x_mm, y_mm
        
        return None, None
    
    def get_velocity(self):
        """估计小球速度(mm/s)"""
        if len(self.pos_history) < 2:
            return 0.0, 0.0
        dt = self.time_history[-1] - self.time_history[-2]
        if dt < 1e-6:
            return 0.0, 0.0
        vx = (self.pos_history[-1][0] - self.pos_history[-2][0]) / dt
        vy = (self.pos_history[-1][1] - self.pos_history[-2][1]) / dt
        return vx, vy
    
    def release(self):
        self.cap.release()
```

### 2. 双环PID控制器

```python
class DualLoopPID:
    """
    双环PID控制器（位置环+速度环）
    外环：位置PID → 输出目标速度
    内环：速度PID → 输出舵机角度
    """
    
    def __init__(self):
        # X轴
        self.pos_pid_x = PID(Kp=0.8, Ki=0.02, Kd=0.5, out_max=50)
        self.vel_pid_x = PID(Kp=3.0, Ki=0.0, Kd=0.5, out_max=30)
        
        # Y轴
        self.pos_pid_y = PID(Kp=0.8, Ki=0.02, Kd=0.5, out_max=50)
        self.vel_pid_y = PID(Kp=3.0, Ki=0.0, Kd=0.5, out_max=30)
        
    def compute(self, target_x, target_y, current_x, current_y, vx, vy):
        """计算X和Y轴的舵机偏转角度"""
        # 外环：位置→速度指令
        vel_cmd_x = self.pos_pid_x.compute(target_x, current_x)
        vel_cmd_y = self.pos_pid_y.compute(target_y, current_y)
        
        # 内环：速度→舵机角度
        angle_x = self.vel_pid_x.compute(vel_cmd_x, vx)
        angle_y = self.vel_pid_y.compute(vel_cmd_y, vy)
        
        return angle_x, angle_y


class PID:
    def __init__(self, Kp, Ki, Kd, out_max=100):
        self.Kp, self.Ki, self.Kd = Kp, Ki, Kd
        self.out_max = out_max
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None
        
    def compute(self, target, current):
        error = target - current
        now = time.time()
        if self.prev_time is None:
            self.prev_time = now
            self.prev_error = error
            return 0
        dt = max(now - self.prev_time, 1e-6)
        
        p = self.Kp * error
        self.integral = np.clip(self.integral + error * dt, -100, 100)
        i = self.Ki * self.integral
        d = self.Kd * (error - self.prev_error) / dt
        
        output = np.clip(p + i + d, -self.out_max, self.out_max)
        self.prev_error = error
        self.prev_time = now
        return output
    
    def reset(self):
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None


class BallControlSystem:
    """小球控制系统主程序"""
    
    def __init__(self):
        self.vision = BallTrackingVision(camera_id=0)
        self.pid = DualLoopPID()
        self.ser = serial.Serial('/dev/ttyS6', 115200, timeout=0.01)
        
        # 舵机中立角度
        self.servo_center_x = 90
        self.servo_center_y = 90
        
        # 平台中心
        self.center_x = 150
        self.center_y = 150
        
    def send_servo_angle(self, angle_x, angle_y):
        """发送舵机角度到STM32"""
        sx = int(np.clip(self.servo_center_x + angle_x, 30, 150))
        sy = int(np.clip(self.servo_center_y + angle_y, 30, 150))
        data = struct.pack('<BB', sx, sy)
        self.ser.write(b'\xAA' + data + b'\x55')
    
    def run_point_mode(self, target_x, target_y):
        """定点模式"""
        print(f"目标位置: ({target_x}, {target_y})mm")
        
        while True:
            x, y = self.vision.detect_ball()
            if x is not None:
                vx, vy = self.vision.get_velocity()
                angle_x, angle_y = self.pid.compute(target_x, target_y, x, y, vx, vy)
                self.send_servo_angle(angle_x, angle_y)
                
                error = math.sqrt((x-target_x)**2 + (y-target_y)**2)
                if error < 5:  # 5mm内视为到达
                    print("到达目标位置！")
                    break
                    
                print(f"位置:({x:.1f},{y:.1f}) 误差:{error:.1f}mm 角度:({angle_x:.1f},{angle_y:.1f})")
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
    
    def run_line_mode(self, angle_deg=0, length=200, period=6):
        """直线轨迹模式"""
        print(f"直线轨迹: 角度{angle_deg}°, 长度{length}mm")
        t_start = time.time()
        
        while True:
            t = time.time() - t_start
            x, y = self.vision.detect_ball()
            if x is not None:
                # 生成直线目标
                ratio = (math.sin(2*math.pi*t/period) + 1) / 2
                target_x = self.center_x + length * (ratio - 0.5) * math.cos(math.radians(angle_deg))
                target_y = self.center_y + length * (ratio - 0.5) * math.sin(math.radians(angle_deg))
                
                vx, vy = self.vision.get_velocity()
                angle_x, angle_y = self.pid.compute(target_x, target_y, x, y, vx, vy)
                self.send_servo_angle(angle_x, angle_y)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
    
    def run_circle_mode(self, radius=80, period=5):
        """画圆轨迹模式"""
        print(f"圆形轨迹: 半径{radius}mm")
        t_start = time.time()
        
        while True:
            t = time.time() - t_start
            x, y = self.vision.detect_ball()
            if x is not None:
                target_x = self.center_x + radius * math.cos(2*math.pi*t/period)
                target_y = self.center_y + radius * math.sin(2*math.pi*t/period)
                
                vx, vy = self.vision.get_velocity()
                angle_x, angle_y = self.pid.compute(target_x, target_y, x, y, vx, vy)
                self.send_servo_angle(angle_x, angle_y)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
    
    def cleanup(self):
        self.send_servo_angle(0, 0)
        self.vision.release()
        self.ser.close()


import math

def main():
    system = BallControlSystem()
    
    print("小球滚动控制系统")
    print("1: 定点模式  2: 直线模式  3: 画圆模式")
    choice = input("选择模式: ")
    
    try:
        if choice == '1':
            tx = float(input("目标X(mm): "))
            ty = float(input("目标Y(mm): "))
            system.run_point_mode(tx, ty)
        elif choice == '2':
            system.run_line_mode()
        elif choice == '3':
            system.run_circle_mode()
    finally:
        system.cleanup()
```

## 接线说明

| Orange Pi 5引脚 | 连接目标 | 说明 |
|-----------------|---------|------|
| CSI接口 | 摄像头 | 俯视拍摄平台 |
| UART3_TX | STM32 RX | 舵机控制指令 |
| 5V | STM32/舵机 | 供电 |
| GND | 公共地 | - |

## 调试要点

1. **透视变换标定**：准确标定平台四角像素坐标
2. **颜色标定**：在实际光照下标定小球HSV范围
3. **双环PID调参**：先调位置环P，再调速度环PD
4. **摩擦力补偿**：小球静摩擦会导致抖动，需死区处理
