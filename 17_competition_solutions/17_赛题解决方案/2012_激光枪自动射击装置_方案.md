### 4.2 目标检测模块

```python
#!/usr/bin/env python3
"""
目标检测模块 - 支持ArUco标记和颜色检测
"""

import cv2
import numpy as np

class TargetDetector:
    def __init__(self, camera_id=0):
        self.cap = cv2.VideoCapture(camera_id)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 60)
        
        # ArUco标记检测
        self.aruco_dict = cv2.aruco.Dictionary_get(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters_create()
        
        # 颜色检测参数
        self.color_ranges = {
            'red': ([0, 100, 100], [10, 255, 255]),
            'green': ([35, 100, 100], [85, 255, 255]),
            'blue': ([100, 100, 100], [130, 255, 255])
        }
        
        # 图像中心
        self.frame_center = (320, 240)
        
    def detect_aruco(self, marker_id=None):
        """
        检测ArUco标记
        marker_id: 指定标记ID，None表示检测所有
        返回: [(x, y, id), ...] 或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 检测标记
        corners, ids, rejected = cv2.aruco.detectMarkers(
            frame, self.aruco_dict, parameters=self.aruco_params
        )
        
        if ids is None:
            return None
            
        targets = []
        for i, marker_id_detected in enumerate(ids.flatten()):
            if marker_id is not None and marker_id_detected != marker_id:
                continue
                
            # 计算中心点
            center = np.mean(corners[i][0], axis=0)
            targets.append((int(center[0]), int(center[1]), marker_id_detected))
            
        return targets if targets else None
        
    def detect_color(self, color='red'):
        """
        检测颜色目标
        返回: [(x, y, area), ...] 或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 转换到HSV颜色空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 颜色过滤
        lower, upper = self.color_ranges[color]
        mask = cv2.inRange(hsv, np.array(lower), np.array(upper))
        
        # 形态学操作
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.erode(mask, kernel, iterations=2)
        mask = cv2.dilate(mask, kernel, iterations=2)
        
        # 查找轮廓
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if not contours:
            return None
            
        targets = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if area > 100:  # 最小面积阈值
                M = cv2.moments(contour)
                if M["m00"] > 0:
                    cx = int(M["m10"] / M["m00"])
                    cy = int(M["m01"] / M["m00"])
                    targets.append((cx, cy, area))
                    
        return targets if targets else None
        
    def detect_circle(self, min_radius=20, max_radius=100):
        """
        检测圆形目标
        返回: [(x, y, radius), ...] 或 None
        """
        ret, frame = self.cap.read()
        if not ret:
            return None
            
        # 灰度化
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        
        # 高斯模糊
        blurred = cv2.GaussianBlur(gray, (9, 9), 2)
        
        # 霍夫圆检测
        circles = cv2.HoughCircles(
            blurred, cv2.HOUGH_GRADIENT, dp=1, minDist=50,
            param1=50, param2=30,
            minRadius=min_radius, maxRadius=max_radius
        )
        
        if circles is None:
            return None
            
        targets = []
        for circle in circles[0]:
            x, y, r = circle
            targets.append((int(x), int(y), int(r)))
            
        return targets if targets else None
        
    def get_target_offset(self, target_pos):
        """
        计算目标相对于图像中心的偏移
        返回: (offset_x, offset_y)
        """
        if target_pos is None:
            return None
            
        offset_x = target_pos[0] - self.frame_center[0]
        offset_y = target_pos[1] - self.frame_center[1]
        
        return (offset_x, offset_y)
        
    def get_frame(self):
        """获取当前帧"""
        ret, frame = self.cap.read()
        return frame if ret else None
        
    def get_frame_with_detection(self, targets=None):
        """获取带有检测结果的图像帧"""
        frame = self.get_frame()
        if frame is None:
            return None
            
        # 绘制中心十字
        cv2.line(frame, (self.frame_center[0] - 20, self.frame_center[1]),
                 (self.frame_center[0] + 20, self.frame_center[1]), (0, 255, 0), 2)
        cv2.line(frame, (self.frame_center[0], self.frame_center[1] - 20),
                 (self.frame_center[0], self.frame_center[1] + 20), (0, 255, 0), 2)
        
        # 绘制检测到的目标
        if targets:
            for target in targets:
                x, y = target[0], target[1]
                cv2.circle(frame, (x, y), 5, (0, 0, 255), -1)
                cv2.line(frame, self.frame_center, (x, y), (0, 0, 255), 2)
                
        return frame
        
    def release(self):
        """释放资源"""
        self.cap.release()
```

```python
#!/usr/bin/env python3
"""
云台控制模块 - 双舵机控制
"""

import RPi.GPIO as GPIO
import time

class GimbalController:
    def __init__(self, pan_pin, tilt_pin):
        """
        pan_pin: 水平舵机引脚
        tilt_pin: 垂直舵机引脚
        """
        self.pan_pin = pan_pin
        self.tilt_pin = tilt_pin
        
        # GPIO设置
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(pan_pin, GPIO.OUT)
        GPIO.setup(tilt_pin, GPIO.OUT)
        
        # PWM初始化（50Hz）
        self.pan_pwm = GPIO.PWM(pan_pin, 50)
        self.tilt_pwm = GPIO.PWM(tilt_pin, 50)
        self.pan_pwm.start(0)
        self.tilt_pwm.start(0)
        
        # 当前角度
        self.pan_angle = 90   # 水平角度（0-180）
        self.tilt_angle = 90  # 垂直角度（0-180）
        
        # 角度限制
        self.pan_min = 0
        self.pan_max = 180
        self.tilt_min = 30
        self.tilt_max = 150
        
        # 初始位置
        self.set_position(90, 90)
        
    def _set_servo_angle(self, pwm, angle):
        """设置舵机角度"""
        # 角度转换为占空比
        # 0度 -> 2.5%, 180度 -> 12.5%
        duty = 2.5 + angle / 18.0
        pwm.ChangeDutyCycle(duty)
        
    def set_pan_angle(self, angle):
        """设置水平角度"""
        angle = max(self.pan_min, min(self.pan_max, angle))
        self.pan_angle = angle
        self._set_servo_angle(self.pan_pwm, angle)
        
    def set_tilt_angle(self, angle):
        """设置垂直角度"""
        angle = max(self.tilt_min, min(self.tilt_max, angle))
        self.tilt_angle = angle
        self._set_servo_angle(self.tilt_pwm, angle)
        
    def set_position(self, pan, tilt):
        """设置云台位置"""
        self.set_pan_angle(pan)
        self.set_tilt_angle(tilt)
        
    def adjust(self, delta_pan, delta_tilt):
        """调整云台位置（增量控制）"""
        self.set_pan_angle(self.pan_angle + delta_pan)
        self.set_tilt_angle(self.tilt_angle + delta_tilt)
        
    def center(self):
        """回到中心位置"""
        self.set_position(90, 90)
        
    def get_position(self):
        """获取当前位置"""
        return (self.pan_angle, self.tilt_angle)
        
    def cleanup(self):
        """清理GPIO"""
        self.center()
        time.sleep(0.5)
        self.pan_pwm.stop()
        self.tilt_pwm.stop()
```

```python
"""
激光控制模块 - 激光开关和调制
"""

import RPi.GPIO as GPIO

class LaserController:
    def __init__(self, laser_pin):
        """
        laser_pin: 激光控制引脚
        """
        self.laser_pin = laser_pin
        
        # GPIO设置
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(laser_pin, GPIO.OUT)
        
        # 初始状态：关闭
        GPIO.output(laser_pin, GPIO.LOW)
        self.is_on = False
        
    def on(self):
        """打开激光"""
        GPIO.output(self.laser_pin, GPIO.HIGH)
        self.is_on = True
        
    def off(self):
        """关闭激光"""
        GPIO.output(self.laser_pin, GPIO.LOW)
        self.is_on = False
        
    def toggle(self):
        """切换激光状态"""
        if self.is_on:
            self.off()
        else:
            self.on()
            
    def blink(self, times=3, interval=0.1):
        """闪烁激光"""
        for _ in range(times):
            self.on()
            time.sleep(interval)
            self.off()
            time.sleep(interval)
            
    def cleanup(self):
        """清理GPIO"""
        self.off()
```

### 4.5 瞄准控制算法

```python
#!/usr/bin/env python3
"""
瞄准控制算法 - PID控制
"""

import numpy as np

class AimController:
    def __init__(self, pid_params=None):
        """
        pid_params: PID参数字典 {'kp': ..., 'ki': ..., 'kd': ...}
        """
        if pid_params is None:
            self.pid_params = {
                'pan': {'kp': 0.5, 'ki': 0.01, 'kd': 0.1},
                'tilt': {'kp': 0.5, 'ki': 0.01, 'kd': 0.1}
            }
        else:
            self.pid_params = pid_params
            
        # PID状态变量
        self.pan_error_sum = 0
        self.pan_error_prev = 0
        self.tilt_error_sum = 0
        self.tilt_error_prev = 0
        
        # 目标位置（图像坐标）
        self.target_pos = None
        
        # 瞄准阈值（像素）
        self.aim_threshold = 10
        
        # 输出限制
        self.output_min = -5  # 最小调整量（度）
        self.output_max = 5   # 最大调整量（度）
        
    def compute(self, current_pos, target_pos=None):
        """
        计算控制输出
        current_pos: 当前目标位置 (x, y) 图像坐标
        target_pos: 目标位置 (x, y) 图像坐标，默认为图像中心
        
        返回: (delta_pan, delta_tilt) 舵机调整量（度）
        """
        if target_pos is None:
            target_pos = (320, 240)  # 图像中心
            
        # 计算误差
        error_x = target_pos[0] - current_pos[0]
        error_y = target_pos[1] - current_pos[1]
        
        # PID计算（水平）
        p_pan = self.pid_params['pan']['kp'] * error_x
        self.pan_error_sum += error_x
        self.pan_error_sum = np.clip(self.pan_error_sum, -100, 100)
        i_pan = self.pid_params['pan']['ki'] * self.pan_error_sum
        d_pan = self.pid_params['pan']['kd'] * (error_x - self.pan_error_prev)
        self.pan_error_prev = error_x
        
        delta_pan = p_pan + i_pan + d_pan
        
        # PID计算（垂直）
        p_tilt = self.pid_params['tilt']['kp'] * error_y
        self.tilt_error_sum += error_y
        self.tilt_error_sum = np.clip(self.tilt_error_sum, -100, 100)
        i_tilt = self.pid_params['tilt']['ki'] * self.tilt_error_sum
        d_tilt = self.pid_params['tilt']['kd'] * (error_y - self.tilt_error_prev)
        self.tilt_error_prev = error_y
        
        delta_tilt = p_tilt + i_tilt + d_tilt
        
        # 限制输出范围
        delta_pan = np.clip(delta_pan, self.output_min, self.output_max)
        delta_tilt = np.clip(delta_tilt, self.output_min, self.output_max)
        
        return delta_pan, delta_tilt
        
    def is_aimed(self, current_pos, target_pos=None):
        """
        判断是否瞄准目标
        返回: True/False
        """
        if target_pos is None:
            target_pos = (320, 240)
            
        error_x = abs(target_pos[0] - current_pos[0])
        error_y = abs(target_pos[1] - current_pos[1])
        
        return error_x < self.aim_threshold and error_y < self.aim_threshold
        
    def reset(self):
        """重置控制器状态"""
        self.pan_error_sum = 0
        self.pan_error_prev = 0
        self.tilt_error_sum = 0
        self.tilt_error_prev = 0
        
    def set_target(self, pos):
        """设置目标位置"""
        self.target_pos = pos
        
    def tune_pid(self, axis, kp=None, ki=None, kd=None):
        """调整PID参数"""
        if kp is not None:
            self.pid_params[axis]['kp'] = kp
        if ki is not None:
            self.pid_params[axis]['ki'] = ki
        if kd is not None:
            self.pid_params[axis]['kd'] = kd
```

```python
        print("系统就绪")
        print("命令: a=自动模式, m=手动模式, s=射击, q=退出")
        
    def _detect_loop(self):
        """目标检测循环（30Hz）"""
        while self.running:
            if self.auto_mode:
                # 检测目标
                if self.detection_mode == 'aruco':
                    targets = self.detector.detect_aruco(marker_id=self.target_marker_id)
                elif self.detection_mode == 'color':
                    targets = self.detector.detect_color(color=self.target_color)
                elif self.detection_mode == 'circle':
                    targets = self.detector.detect_circle()
                else:
                    targets = None
                    
                if targets:
                    # 选择最近的目标（面积最大或距离中心最近）
                    self.current_target = targets[0]
                    self.stats['targets_detected'] += 1
                else:
                    self.current_target = None
                    
            time.sleep(0.033)  # 30Hz
            
    def _control_loop(self):
        """控制循环（50Hz）"""
        while self.running:
            if self.auto_mode and self.current_target:
                # 获取目标位置
                target_pos = (self.current_target[0], self.current_target[1])
                
                # 计算控制输出
                delta_pan, delta_tilt = self.aim_controller.compute(target_pos)
                
                # 调整云台
                self.gimbal.adjust(delta_pan, delta_tilt)
                
                # 检查是否瞄准
                if self.aim_controller.is_aimed(target_pos):
                    if not self.is_shooting:
                        self._shoot()
                        
            time.sleep(0.02)  # 50Hz
            
    def _shoot(self):
        """射击"""
        self.is_shooting = True
        self.laser.on()
        self.stats['shots_fired'] += 1
        
        # 保持激光开启一小段时间
        time.sleep(0.5)
        
        self.laser.off()
        self.is_shooting = False
        
        print(f"射击！总计: {self.stats['shots_fired']}")
        
    def set_auto_mode(self, enable):
        """设置自动模式"""
        self.auto_mode = enable
        if enable:
            print("已切换到自动模式")
        else:
            print("已切换到手动模式")
            self.current_target = None
            
    def manual_shoot(self):
        """手动射击"""
        self._shoot()
        
    def manual_adjust(self, delta_pan, delta_tilt):
        """手动调整云台"""
        if not self.auto_mode:
            self.gimbal.adjust(delta_pan, delta_tilt)
            
    def set_detection_mode(self, mode, **kwargs):
        """设置检测模式"""
        self.detection_mode = mode
        if 'color' in kwargs:
            self.target_color = kwargs['color']
        if 'marker_id' in kwargs:
            self.target_marker_id = kwargs['marker_id']
        print(f"检测模式: {mode}")
        
    def get_status(self):
        """获取系统状态"""
        return {
            'auto_mode': self.auto_mode,
            'target': self.current_target,
            'gimbal_pos': self.gimbal.get_position(),
            'laser_on': self.laser.is_on,
            'stats': self.stats
        }
        
    def stop(self):
        """停止系统"""
        self.running = False
        self.auto_mode = False
        
        # 等待线程结束
        if hasattr(self, 'detect_thread'):
            self.detect_thread.join()
        if hasattr(self, 'control_thread'):
            self.control_thread.join()
            
        # 关闭激光
        self.laser.cleanup()
        
        # 清理云台
        self.gimbal.cleanup()
        
        # 释放摄像头
        self.detector.release()
        
        print("系统已停止")
```

```python
def calibrate_camera(detector):
    """
    相机标定
    1. 拍摄多张棋盘格图像
    2. 检测角点
    3. 计算相机内参和畸变系数
    """
    import glob
    
    # 棋盘格参数
    board_size = (9, 6)
    square_size = 0.025  # 25mm
    
    # 准备3D点
    objp = np.zeros((board_size[0] * board_size[1], 3), np.float32)
    objp[:, :2] = np.mgrid[0:board_size[0], 0:board_size[1]].T.reshape(-1, 2) * square_size
    
    obj_points = []
    img_points = []
    
    # 读取图像
    images = glob.glob('calibration_images/*.jpg')
    
    for fname in images:
        img = cv2.imread(fname)
        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        
        # 查找棋盘格角点
        ret, corners = cv2.findChessboardCorners(gray, board_size, None)
        
        if ret:
            obj_points.append(objp)
            img_points.append(corners)
            
    # 标定
    ret, camera_matrix, dist_coeffs, rvecs, tvecs = cv2.calibrateCamera(
        obj_points, img_points, gray.shape[::-1], None, None
    )
    
    print(f"标定完成: 误差={ret}")
    print(f"相机矩阵:\n{camera_matrix}")
    print(f"畸变系数: {dist_coeffs}")
    
    return camera_matrix, dist_coeffs
```

### 5.3 常见问题及解决

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| 目标检测失败 | 光线不好 | 调整光线或使用补光灯 |
| 瞄准不准 | 相机未标定 | 进行相机标定 |
| 响应慢 | 控制频率太低 | 提高控制频率 |
| 激光不亮 | 接线错误 | 检查激光模块接线 |
| 云台抖动 | PID参数不合适 | 调整PID参数 |
