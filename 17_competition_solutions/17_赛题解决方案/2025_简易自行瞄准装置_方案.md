## 一、题目分析

### 核心要求
设计一个简易自行瞄准装置：
- 装置可自动识别目标（靶标/特定标志）
- 控制激光笔瞄准目标中心
- 要求瞄准精度高、响应快
- 支持不同距离的目标瞄准
- 全程自主，无遥控

### 系统组成
```
┌─────────────────────────────────────┐
│          简易自行瞄准装置             │
│                                     │
│  摄像头 → Orange Pi 5 → 激光笔控制   │
│     │         │              │      │

│  (YOLOv5)  (PID控制)      (舵机)    │
└─────────────────────────────────────┘
```

## 四、核心算法

### 4.1 目标检测 (YOLOv5-NPU)

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class TargetDetector:
    """目标检测器 (靶标/标志)"""
    
    def __init__(self, model_path='target.rknn'):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime()
        
        # 目标类别
        self.classes = ['target_center', 'target_ring', 'bullseye']
        
        # 检测参数
        self.conf_threshold = 0.6
        self.nms_threshold = 0.45
        self.input_size = (640, 640)
    
    def detect(self, frame):
        """
        检测目标
        返回: [{'class': str, 'confidence': float, 'bbox': [x1,y1,x2,y2], 'center': (cx,cy)}, ...]
        """
        # 预处理
        img = cv2.resize(frame, self.input_size)
        img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
        
        # NPU推理
        outputs = self.rknn.inference(inputs=[img])
        
        # 后处理 (NMS)
        detections = self.post_process(outputs)
        
        # 转换坐标到原始图像尺寸
        h_ratio = frame.shape[0] / self.input_size[1]
        w_ratio = frame.shape[1] / self.input_size[0]
        
        for det in detections:
            det['bbox'][0] *= w_ratio
            det['bbox'][1] *= h_ratio
            det['bbox'][2] *= w_ratio
            det['bbox'][3] *= h_ratio
            
            # 计算中心点
            cx = (det['bbox'][0] + det['bbox'][2]) / 2
            cy = (det['bbox'][1] + det['bbox'][3]) / 2
            det['center'] = (cx, cy)
        
        return detections
    
    def post_process(self, outputs):
        """后处理: 置信度过滤 + NMS"""
        boxes = []
        scores = []
        class_ids = []
        
        # 解析YOLOv5输出 (根据实际模型调整)
        for output in outputs:
            for detection in output:
                confidence = detection[4]
                if confidence > self.conf_threshold:
                    class_scores = detection[5:]
                    class_id = np.argmax(class_scores)
                    class_score = class_scores[class_id]
                    
                    if class_score > self.conf_threshold:
                        x, y, w, h = detection[:4]
                        x1 = x - w/2
                        y1 = y - h/2
                        x2 = x + w/2
                        y2 = y + h/2
                        
                        boxes.append([x1, y1, x2, y2])
                        scores.append(confidence * class_score)
                        class_ids.append(class_id)
        
        # NMS
        if boxes:
            indices = cv2.dnn.NMSBoxes(boxes, scores, 
                                        self.conf_threshold, 
                                        self.nms_threshold)
            
            detections = []
            for i in indices:
                detections.append({
                    'class': self.classes[class_ids[i]],
                    'confidence': scores[i],
                    'bbox': boxes[i]
                })
            return detections
        
        return []
    
    def detect_target_center_fallback(self, frame):
        """
        备用方案: 传统视觉检测靶标中心
        适用于简单靶标 (同心圆/十字)
        """
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        blurred = cv2.GaussianBlur(gray, (9, 9), 2)
        
        # 霍夫圆检测
        circles = cv2.HoughCircles(
            blurred, cv2.HOUGH_GRADIENT, dp=1, minDist=50,
            param1=100, param2=40,
            minRadius=20, maxRadius=200
        )
        
        if circles is not None:
            circles = np.uint16(np.around(circles))
            # 取最大圆的圆心
            best = max(circles[0], key=lambda c: c[2])
            return (best[0], best[1]), best[2]
        
        return None, None
```

### 4.2 瞄准PID控制器

```python
import numpy as np
import time

class AimingPID:
    """瞄准PID控制器"""
    
    def __init__(self, image_width=640, image_height=480):
        self.img_w = image_width
        self.img_h = image_height
        
        # PID参数 (需要调试)
        self.Kp = 0.3
        self.Ki = 0.005
        self.Kd = 0.15
        
        # 状态
        self.prev_error_x = 0
        self.prev_error_y = 0
        self.integral_x = 0
        self.integral_y = 0
        self.prev_time = time.time()
        
        # 瞄准精度阈值 (像素)
        self.accuracy_threshold = 5  # 5像素内算瞄准成功
        
        # 死区 (避免抖动)
        self.deadzone = 2  # 像素
    
    def compute(self, target_x, target_y):
        """
        计算舵机调整量
        target_x, target_y: 目标中心在图像中的像素坐标
        返回: (pan_adjust, tilt_adjust) 角度调整量
        """
        current_time = time.time()
        dt = current_time - self.prev_time
        if dt <= 0:
            dt = 0.001
        self.prev_time = current_time
        
        # 计算误差 (目标与图像中心的偏差)
        error_x = target_x - self.img_w / 2
        error_y = target_y - self.img_h / 2
        
        # 死区处理
        if abs(error_x) < self.deadzone:
            error_x = 0
        if abs(error_y) < self.deadzone:
            error_y = 0
        
        # 积分项 (带限幅和积分分离)
        if abs(error_x) < 50:  # 误差较小时才积分
            self.integral_x += error_x * dt
        if abs(error_y) < 50:
            self.integral_y += error_y * dt
        
        self.integral_x = np.clip(self.integral_x, -200, 200)
        self.integral_y = np.clip(self.integral_y, -200, 200)
        
        # 微分项
        derivative_x = (error_x - self.prev_error_x) / dt
        derivative_y = (error_y - self.prev_error_y) / dt
        
        # PID输出
        output_x = (self.Kp * error_x + 
                    self.Ki * self.integral_x + 
                    self.Kd * derivative_x)
        output_y = (self.Kp * error_y + 
                    self.Ki * self.integral_y + 
                    self.Kd * derivative_y)
        
        self.prev_error_x = error_x
        self.prev_error_y = error_y
        
        # 限幅
        output_x = np.clip(output_x, -30, 30)  # 最大30度/秒
        output_y = np.clip(output_y, -30, 30)
        
        return output_x, output_y
    
    def is_aimed(self, target_x, target_y):
        """判断是否已瞄准"""
        error_x = abs(target_x - self.img_w / 2)
        error_y = abs(target_y - self.img_h / 2)
        error = np.sqrt(error_x**2 + error_y**2)
        return error < self.accuracy_threshold
    
    def reset(self):
        """重置PID状态"""
        self.prev_error_x = 0
        self.prev_error_y = 0
        self.integral_x = 0
        self.integral_y = 0
```

```python
import numpy as np

class CameraGimbalCalibration:
    """相机-云台坐标标定"""
    
    def __init__(self):
        # 像素到角度的映射 (标定后填入)
        # 简单线性模型
        self.pan_scale = 0.15   # 度/像素 (水平)
        self.tilt_scale = 0.12  # 度/像素 (垂直)
        
        # 偏移 (激光笔与摄像头的物理偏移)
        self.pan_offset = 0     # 度
        self.tilt_offset = 0    # 度
        
        # 多项式校正系数 (更精确)
        self.pan_poly = [0, 0.15, 0.0001]   # a*x^2 + b*x + c
        self.tilt_poly = [0, 0.12, 0.00008]
    
    def pixel_to_angle(self, pixel_error_x, pixel_error_y):
        """
        像素误差转换为舵机角度调整
        """
        # 线性映射
        # pan_adjust = pixel_error_x * self.pan_scale
        # tilt_adjust = pixel_error_y * self.tilt_scale
        
        # 多项式校正 (更精确)
        pan_adjust = (self.pan_poly[0] * pixel_error_x**2 + 
                     self.pan_poly[1] * pixel_error_x + 
                     self.pan_poly[2])
        tilt_adjust = (self.tilt_poly[0] * pixel_error_y**2 + 
                      self.tilt_poly[1] * pixel_error_y + 
                      self.tilt_poly[2])
        
        return pan_adjust, tilt_adjust
    
    def calibrate(self, calibration_points):
        """
        标定: 多点标定法
        calibration_points: [(pixel_x, pixel_y, pan_angle, tilt_angle), ...]
        """
        pixels_x = np.array([p[0] for p in calibration_points])
        pixels_y = np.array([p[1] for p in calibration_points])
        pan_angles = np.array([p[2] for p in calibration_points])
        tilt_angles = np.array([p[3] for p in calibration_points])
        
        # 多项式拟合
        self.pan_poly = np.polyfit(pixels_x, pan_angles, 2)
        self.tilt_poly = np.polyfit(pixels_y, tilt_angles, 2)
        
        print(f"标定完成:")
        print(f"Pan: {self.pan_poly}")
        print(f"Tilt: {self.tilt_poly}")
```

```python
#!/usr/bin/env python3
"""
简易自行瞄准装置主程序
Orange Pi 5 + STM32 + YOLOv5
"""

import cv2
import numpy as np
import time
import serial
import threading

from target_detector import TargetDetector
from aiming_pid import AimingPID
from calibration import CameraGimbalCalibration

class AimingSystem:
    def __init__(self):
        # 初始化组件
        self.detector = TargetDetector('target.rknn')
        self.pid = AimingPID(image_width=640, image_height=480)
        self.calibration = CameraGimbalCalibration()
        
        # 摄像头
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
        self.cap.set(cv2.CAP_PROP_FPS, 60)
        
        # 串口连接STM32
        self.ser = serial.Serial('/dev/ttyS2', 115200, timeout=0.01)
        
        # 舵机状态
        self.pan_angle = 90.0
        self.tilt_angle = 90.0
        
        # 状态
        self.mode = 'search'  # search/track/aimed
        self.target_locked = False
        self.aiming_start_time = 0
        self.running = False
        
        # 性能统计
        self.fps = 0
        self.frame_count = 0
        self.fps_timer = time.time()
        
        # 按键
        import OPi.GPIO as GPIO
        GPIO.setmode(GPIO.BOARD)
        for pin in [11, 12, 13]:
            GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)
    
    def send_command(self, cmd):
        """发送指令到STM32"""
        self.ser.write(f'{cmd}\n'.encode())
    
    def set_servo(self, pan, tilt):
        """设置舵机角度"""
        pan = np.clip(pan, 0, 180)
        tilt = np.clip(tilt, 0, 180)
        self.pan_angle = pan
        self.tilt_angle = tilt
        self.send_command(f'P{pan:.1f}T{tilt:.1f}')
    
    def set_laser(self, on=True):
        """控制激光笔"""
        self.send_command(f'L{1 if on else 0}')
    
    def search_target(self, frame):
        """搜索目标"""
        # YOLOv5检测
        detections = self.detector.detect(frame)
        
        if detections:
            # 取置信度最高的
            best = max(detections, key=lambda d: d['confidence'])
            target_center = best['center']
            
            self.mode = 'track'
            return target_center
        
        # 备用: 传统视觉检测
        center, radius = self.detector.detect_target_center_fallback(frame)
        if center:
            self.mode = 'track'
            return center
        
        # 搜索机动
        self.search_pattern()
        return None
    
    def search_pattern(self):
        """搜索模式: 画网格搜索"""
        t = time.time() % 8
        if t < 2:
            self.set_servo(70, 90)
        elif t < 4:
            self.set_servo(110, 90)
        elif t < 6:
            self.set_servo(90, 70)
        else:
            self.set_servo(90, 110)
    
    def track_target(self, frame, target_center):
        """追踪目标"""
        # 计算PID
        pan_adj, tilt_adj = self.pid.compute(target_center[0], target_center[1])
        
        # 更新舵机
        self.set_servo(self.pan_angle + pan_adj, self.tilt_angle + tilt_adj)
        
        # 检查是否瞄准
        if self.pid.is_aimed(target_center[0], target_center[1]):
            if not self.target_locked:
                self.target_locked = True
                self.aiming_start_time = time.time()
            
            # 持续瞄准0.5秒后开启激光
            if time.time() - self.aiming_start_time > 0.5:
                self.set_laser(True)
                self.mode = 'aimed'
                print("瞄准成功!")
        else:
            self.target_locked = False
            self.set_laser(False)
    
    def draw_debug(self, frame, target_center=None):
        """绘制调试信息"""
        h, w = frame.shape[:2]
        
        # 画十字准星
        cv2.line(frame, (w//2-30, h//2), (w//2+30, h//2), (0, 255, 0), 1)
        cv2.line(frame, (w//2, h//2-30), (w//2, h//2+30), (0, 255, 0), 1)
        cv2.circle(frame, (w//2, h//2), self.pid.accuracy_threshold, (0, 255, 0), 1)
        
        # 画目标位置
        if target_center:
            cv2.circle(frame, (int(target_center[0]), int(target_center[1])), 
                      15, (0, 0, 255), 2)
            # 连线
            cv2.line(frame, (w//2, h//2), 
                    (int(target_center[0]), int(target_center[1])), 
                    (0, 255, 255), 1)
            
            # 显示误差
            error = np.sqrt((target_center[0]-w//2)**2 + (target_center[1]-h//2)**2)
            cv2.putText(frame, f"Error: {error:.1f}px", (10, 60), 
                       cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        
        # 显示状态
        status_text = {
            'search': 'SEARCHING...',
            'track': 'TRACKING',
            'aimed': 'AIMED!'
        }
        color = {'search': (0, 165, 255), 'track': (0, 255, 255), 'aimed': (0, 255, 0)}
        
        cv2.putText(frame, status_text[self.mode], (10, 30), 
                   cv2.FONT_HERSHEY_SIMPLEX, 0.8, color[self.mode], 2)
        cv2.putText(frame, f"FPS: {self.fps}", (10, 90), 
                   cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        cv2.putText(frame, f"Pan: {self.pan_angle:.1f} Tilt: {self.tilt_angle:.1f}", 
                   (10, 120), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
        
        return frame
    
    def check_buttons(self):
        """检查按键"""
        import OPi.GPIO as GPIO
        
        if not GPIO.input(11):  # KEY1: 启动瞄准
            self.mode = 'search'
            self.set_laser(False)
            self.pid.reset()
            time.sleep(0.2)
        
        if not GPIO.input(12):  # KEY2: 模式切换
            time.sleep(0.2)
        
        if not GPIO.input(13):  # KEY3: 复位归中
            self.set_servo(90, 90)
            self.set_laser(False)
            self.mode = 'search'
            self.pid.reset()
            time.sleep(0.2)
    
    def main_loop(self):
        """主循环"""
        self.running = True
        
        # 初始化: 归中
        self.set_servo(90, 90)
        time.sleep(1)
        
        print("简易自行瞄准装置启动")
        print("按 KEY1 开始瞄准, KEY3 复位归中")
        
        while self.running:
            ret, frame = self.cap.read()
            if not ret:
                continue
            
            # 根据模式处理
            if self.mode == 'search':
                target_center = self.search_target(frame)
                if target_center:
                    self.track_target(frame, target_center)
            
            elif self.mode == 'track':
                # 检测目标
                detections = self.detector.detect(frame)
                if detections:
                    best = max(detections, key=lambda d: d['confidence'])
                    target_center = best['center']
                    self.track_target(frame, target_center)
                else:
                    # 丢失目标，回到搜索
                    self.set_laser(False)
                    self.mode = 'search'
                    target_center = None
            
            elif self.mode == 'aimed':
                # 持续保持瞄准
                detections = self.detector.detect(frame)
                if detections:
                    best = max(detections, key=lambda d: d['confidence'])
                    target_center = best['center']
                    
                    # 微调保持瞄准
                    pan_adj, tilt_adj = self.pid.compute(
                        target_center[0], target_center[1])
                    self.set_servo(self.pan_angle + pan_adj, 
                                  self.tilt_angle + tilt_adj)
                else:
                    self.set_laser(False)
                    self.mode = 'search'
            
            # 绘制调试信息
            frame = self.draw_debug(frame, target_center if 'target_center' in dir() else None)
            
            # 显示
            cv2.imshow('Aiming System', frame)
            
            # 检查按键
            self.check_buttons()
            
            # FPS统计
            self.frame_count += 1
            if time.time() - self.fps_timer > 1.0:
                self.fps = self.frame_count
                self.frame_count = 0
                self.fps_timer = time.time()
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        self.cleanup()
    
    def cleanup(self):
        """清理"""
        self.running = False
        self.set_laser(False)
        self.set_servo(90, 90)
        self.cap.release()
        cv2.destroyAllWindows()
        self.ser.close()
        import OPi.GPIO as GPIO
        GPIO.cleanup()

def main():
    system = AimingSystem()
    system.main_loop()

if __name__ == '__main__':
    main()
```

```c
// STM32F103 舵机+激光控制
#include "stm32f10x.h"
#include <string.h>
#include <stdlib.h>

volatile float pan_angle = 90.0;
volatile float tilt_angle = 90.0;
volatile int laser_on = 0;

char rx_buffer[64];
volatile int rx_index = 0;
volatile int rx_complete = 0;

void TIM2_PWM_Init(void) {
    RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
    RCC->APB2ENR |= RCC_APB2ENR_IOPAEN;
    
    // PA0, PA1 AF push-pull
    GPIOA->CRL &= ~(0xFF);
    GPIOA->CRL |= 0x0B;
    GPIOA->CRL |= 0xB0;
    
    TIM2->PSC = 72 - 1;
    TIM2->ARR = 20000 - 1;  // 50Hz
    TIM2->CCR1 = 1500;
    TIM2->CCR2 = 1500;
    TIM2->CCMR1 = 0x6060;
    TIM2->CCER = 0x0011;
    TIM2->CR1 |= TIM_CR1_CEN;
}

void GPIO_Laser_Init(void) {
    // PA2 激光笔控制
    GPIOA->CRL &= ~(0xF << 8);
    GPIOA->CRL |= (0x3 << 8);  // PA2 push-pull output
    GPIOA->ODR &= ~(1 << 2);   // 初始关闭
}

void USART1_Init(void) {
    RCC->APB2ENR |= RCC_APB2ENR_USART1EN | RCC_APB2ENR_IOPAEN;
    
    GPIOA->CRH &= ~(0xFF << 4);
    GPIOA->CRH |= (0x0B << 4);
    GPIOA->CRH |= (0x04 << 8);
    
    USART1->BRR = 72000000 / 115200;
    USART1->CR1 = USART_CR1_TE | USART_CR1_RE | USART_CR1_RXNEIE;
    USART1->CR1 |= USART_CR1_UE;
    NVIC_EnableIRQ(USART1_IRQn);
}

void USART1_IRQHandler(void) {
    if (USART1->SR & USART_SR_RXNE) {
        char c = USART1->DR;
        if (c == '$') {
            rx_index = 0;
        } else if (c == '\n') {
            rx_buffer[rx_index] = '\0';
            rx_complete = 1;
        } else if (rx_index < 63) {
            rx_buffer[rx_index++] = c;
        }
    }
}

void parse_command(char* cmd) {
    if (cmd[0] == 'P') {
        // 舵机指令: P<pan>T<tilt>
        char* t_ptr = strchr(cmd, 'T');
        if (t_ptr) {
            pan_angle = atof(cmd + 1);
            tilt_angle = atof(t_ptr + 1);
            
            pan_angle = (pan_angle < 0) ? 0 : (pan_angle > 180) ? 180 : pan_angle;
            tilt_angle = (tilt_angle < 0) ? 0 : (tilt_angle > 180) ? 180 : tilt_angle;
            
            TIM2->CCR1 = 500 + (int)(pan_angle / 180.0 * 2000);
            TIM2->CCR2 = 500 + (int)(tilt_angle / 180.0 * 2000);
        }
    } else if (cmd[0] == 'L') {
        // 激光指令: L<0或1>
        laser_on = atoi(cmd + 1);
        if (laser_on) {
            GPIOA->ODR |= (1 << 2);
        } else {
            GPIOA->ODR &= ~(1 << 2);
        }
    }
}

int main(void) {
    TIM2_PWM_Init();
    GPIO_Laser_Init();
    USART1_Init();
    
    while (1) {
        if (rx_complete) {
            parse_command(rx_buffer);
            rx_complete = 0;
        }
    }
}
```

## 九、总结

| 指标 | 目标值 | 实现方法 |
|------|--------|---------|
| 检测帧率 | ≥60fps | YOLOv5-NPU |
| 瞄准精度 | ±3像素 | PID+死区 |
| 瞄准时间 | <2s | 快速搜索+跟踪 |
| 支持距离 | 1-5m | 自适应检测 |

**OPi5优势**：
- NPU 实时目标检测 (60fps)
- 多核CPU并行处理视觉和控制
- 丰富接口连接摄像头、STM32、传感器
- HDMI实时显示瞄准效果

**关键创新点**：
1. YOLOv5-NPU 实时检测，传统视觉作为备用
2. 积分分离PID，避免大误差时积分饱和
3. 多项式标定，提高像素-角度映射精度
4. 搜索-跟踪-瞄准三状态机，稳定可靠
