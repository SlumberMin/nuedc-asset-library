### 1.2 任务调度

采用**裸机前后台架构**：

```
中断服务(ISR):
  TIM2 Update  → 编码器计数更新 (1kHz)
  TIM3 Update  → PWM更新 (20kHz)
  UART1 RX     → 串口数据接收
  
主循环(while 1):
  1. 视觉处理 (30fps, ~33ms)
  2. 距离/方向PID计算 (10ms)
  3. 电机控制输出 (10ms)
  4. NRF24L01通信 (50ms)
  5. OLED显示更新 (100ms)
  6. 调试信息输出 (200ms)
```

```c
/* pid_controller.h */
#ifndef __PID_CONTROLLER_H
#define __PID_CONTROLLER_H

typedef struct {
    float Kp;              // 比例系数
    float Ki;              // 积分系数
    float Kd;              // 微分系数
    float integral;        // 积分累积
    float prev_error;      // 上次误差
    float integral_max;    // 积分限幅
    float output_max;      // 输出限幅
    float dead_zone;       // 死区
} PID_t;

// 初始化
void PID_Init(PID_t *pid, float kp, float ki, float kd);

// 计算PID输出
float PID_Calc(PID_t *pid, float target, float actual);

// 重置PID状态
void PID_Reset(PID_t *pid);

#endif
```

```c
/* pid_controller.c */
#include "pid_controller.h"

void PID_Init(PID_t *pid, float kp, float ki, float kd) {
    pid->Kp = kp;
    pid->Ki = ki;
    pid->Kd = kd;
    pid->integral = 0;
    pid->prev_error = 0;
    pid->integral_max = 500.0f;
    pid->output_max = 500.0f;
    pid->dead_zone = 2.0f;
}

float PID_Calc(PID_t *pid, float target, float actual) {
    float error = target - actual;
    
    // 死区处理
    if (fabs(error) < pid->dead_zone) {
        error = 0;
    }
    
    // 积分累积
    pid->integral += error;
    if (pid->integral > pid->integral_max)
        pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max)
        pid->integral = -pid->integral_max;
    
    // 微分计算
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    // PID输出
    float output = pid->Kp * error + 
                   pid->Ki * pid->integral + 
                   pid->Kd * derivative;
    
    // 输出限幅
    if (output > pid->output_max)
        output = pid->output_max;
    if (output < -pid->output_max)
        output = -pid->output_max;
    
    return output;
}

void PID_Reset(PID_t *pid) {
    pid->integral = 0;
    pid->prev_error = 0;
}
```

### 2.2 视觉处理模块

```python
# vision_processor.py (运行在Orange Pi 5)
import cv2
import numpy as np
from picamera2 import Picamera2

class VisionProcessor:
    """ArUco标记检测与距离估算"""
    
    def __init__(self):
        # 相机初始化
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # ArUco检测器
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(
            cv2.aruco.DICT_4X4_50
        )
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(
            self.aruco_dict, self.aruco_params
        )
        
        # 标定参数
        self.marker_size_cm = 10.0
        self.focal_length = 650.0  # 标定得到
        
    def detect_and_estimate(self):
        """检测ArUco标记并估算距离和偏移"""
        frame = self.picam2.capture_array()
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)
        
        corners, ids, _ = self.detector.detectMarkers(gray)
        
        if ids is not None and len(ids) > 0:
            # 取最大标记（最近的）
            best_idx = 0
            best_area = 0
            for i, corner in enumerate(corners):
                area = cv2.contourArea(corner[0])
                if area > best_area:
                    best_area = area
                    best_idx = i
            
            corner = corners[best_idx][0]
            
            # 中心点计算
            cx = int(np.mean(corner[:, 0]))
            cy = int(np.mean(corner[:, 1]))
            
            # 距离估算
            pixel_width = np.linalg.norm(corner[0] - corner[1])
            distance_cm = (self.marker_size_cm * self.focal_length) / pixel_width
            
            # 水平偏移
            frame_center = 320
            offset_x = cx - frame_center
            
            return {
                'detected': True,
                'distance_cm': distance_cm,
                'offset_x': offset_x,
                'area': best_area,
                'id': int(ids[best_idx][0])
            }
        
        return {'detected': False}
```

```c
/* motor_control.c */
#include "motor_control.h"

extern TIM_HandleTypeDef htim2;

void Motor_Init(void) {
    // PWM输出: TIM2 CH1(左), CH2(右)
    HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_1);
    HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_2);
    
    // 方向控制: PB12(左IN1), PB13(左IN2), PB14(右IN3), PB15(右IN4)
    __HAL_RCC_GPIOB_CLK_ENABLE();
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    GPIO_InitStruct.Pin = GPIO_PIN_12 | GPIO_PIN_13 | GPIO_PIN_14 | GPIO_PIN_15;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
    HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);
    
    Motor_Stop();
}

void Motor_SetSpeed(int16_t left, int16_t right) {
    // 限幅
    if (left > 100) left = 100;
    if (left < -100) left = -100;
    if (right > 100) right = 100;
    if (right < -100) right = -100;
    
    // 左电机
    if (left >= 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, left * 40);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, (-left) * 40);
    }
    
    // 右电机
    if (right >= 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_14, GPIO_PIN_SET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_15, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, right * 40);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_14, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_15, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, (-right) * 40);
    }
}

void Motor_Stop(void) {
    Motor_SetSpeed(0, 0);
}

void Motor_Brake(void) {
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12 | GPIO_PIN_13 |
                              GPIO_PIN_14 | GPIO_PIN_15, GPIO_PIN_SET);
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, 4095);
    __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, 4095);
}
```

### 2.5 主程序

```c
/* main.c - 跟随车主程序 */
#include "stm32f4xx_hal.h"
#include "motor_control.h"
#include "nrf24l01_comm.h"
#include "pid_controller.h"
#include <stdio.h>

// PID实例
PID_t g_distance_pid;  // 距离PID
PID_t g_direction_pid; // 方向PID

// 系统状态
volatile uint8_t g_target_distance = 50;  // 目标距离(cm)
volatile uint8_t g_car_state = 0;         // 0=跟随, 1=超车, 2=领跑

int main(void) {
    HAL_Init();
    SystemClock_Config();  // 168MHz
    
    // 外设初始化
    GPIO_Init();
    TIM_Init();      // PWM输出
    UART_Init();     // 调试串口
    SPI_Init();      // NRF24L01
    OLED_Init();     // 显示
    
    // 电机初始化
    Motor_Init();
    
    // PID初始化
    PID_Init(&g_distance_pid, 0.8f, 0.01f, 0.3f);
    PID_Init(&g_direction_pid, 0.5f, 0.005f, 0.2f);
    
    // NRF24L01初始化
    NRF24_Init();
    
    OLED_ShowString(0, 0, "Follow Car System");
    OLED_ShowString(0, 2, "Waiting...");
    
    uint32_t last_print_tick = 0;
    
    while(1) {
        // 1. 从Orange Pi 5接收视觉处理结果
        // (通过UART接收距离和偏移量)
        float distance = Vision_GetDistance();  // mm
        float offset_x = Vision_GetOffsetX();  // 像素
        
        // 2. 距离PID计算
        float forward = PID_Calc(&g_distance_pid, 
                                 (float)g_target_distance, 
                                 distance);
        
        // 3. 方向PID计算
        float steering = PID_Calc(&g_direction_pid, 0.0f, offset_x);
        
        // 4. 合成左右轮速度
        int16_t left_speed = (int16_t)(forward + steering);
        int16_t right_speed = (int16_t)(forward - steering);
        
        // 5. 电机控制
        Motor_SetSpeed(left_speed, right_speed);
        
        // 6. OLED显示（100ms更新一次）
        if (HAL_GetTick() - last_print_tick > 100) {
            char buf[32];
            sprintf(buf, "Dist:%.0fcm", distance);
            OLED_ShowString(0, 4, buf);
            sprintf(buf, "L:%d R:%d", left_speed, right_speed);
            OLED_ShowString(0, 6, buf);
            last_print_tick = HAL_GetTick();
        }
        
        HAL_Delay(10);  // 100Hz控制频率
    }
}
```

## 三、关键算法

### 3.1 距离PID参数整定方法

```
步骤1: 先调Kp
  - Ki=0, Kd=0
  - Kp从0.1开始，每次增加0.1
  - 观察系统响应，直到出现轻微振荡
  - 取振荡Kp的60%作为最终Kp

步骤2: 再调Ki
  - 保持Kp不变
  - Ki从0.001开始，每次增加0.001
  - 直到稳态误差消除且无明显振荡

步骤3: 最后调Kd
  - 保持Kp、Ki不变
  - Kd从0.01开始，每次增加0.01
  - 抑制超调和振荡，提高响应速度
```

## 四、通信协议

### 4.1 NRF24L01数据包格式

```
┌────────┬──────────┬───────────┬─────────┬──────────┐
│ Header │ LeftSpd  │ RightSpd  │  State  │ Checksum │
│ 0xAA   │ int8     │ int8      │ uint8   │ uint8    │
│ 1 byte │ 1 byte   │ 1 byte    │ 1 byte  │ 1 byte   │
└────────┴──────────┴───────────┴─────────┴──────────┘

Checksum = (Header + LeftSpd + RightSpd + State) & 0xFF
```

### 4.2 UART调试协议

```
上位机 → MCU:
  'D' + uint16 → 设置目标距离(cm)
  'S' + int8   → 手动设置速度
  'R'          → 重置PID

MCU → 上位机:
  "[Dist:%.1f][L:%d][R:%d]\n"  // 实时数据
```
