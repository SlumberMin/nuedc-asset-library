# 2021年电赛I题 - 具有发电功能的储能小车 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：小车沿赛道行驶，利用下坡段重力势能发电储能、上坡段释放电能驱动，完成全程行驶的同时实现能量回收与管理。

**关键指标**：
- 全程行驶完成度
- 能量回收效率
- 储能管理策略
- 行驶速度与稳定性

### 4.1 赛道识别模块

```python
# track_detector.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class TrackDetector:
    """赛道识别+坡道检测"""
    def __init__(self, model_path='track_seg.rknn'):
        self.rknn = RKNNLite()
        self.rknn.load_rknn(model_path)
        self.rknn.init_runtime(core_mask=RKNNLite.NPU_CORE_0)
        
    def analyze_track(self, frame):
        """分析赛道特征"""
        # 色彩分割
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 黑色赛道
        track_mask = cv2.inRange(hsv, (0, 0, 0), (180, 255, 50))
        
        # 白色引导线
        line_mask = cv2.inRange(hsv, (0, 0, 200), (180, 30, 255))
        
        # 分析底部区域（近处赛道）
        h, w = frame.shape[:2]
        roi = track_mask[int(h*0.6):h, :]
        
        # 计算赛道中心偏移
        contours, _ = cv2.findContours(roi, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        result = {
            'track_detected': len(contours) > 0,
            'center_offset': 0,
            'slope_angle': 0,
            'curvature': 0,
            'track_width': 0
        }
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            M = cv2.moments(largest)
            if M['m00'] > 0:
                cx = int(M['m10'] / M['m00'])
                result['center_offset'] = cx - w // 2
            
            # 赛道宽度
            x, y, bw, bh = cv2.boundingRect(largest)
            result['track_width'] = bw
        
        return result
    
    def detect_slope(self, imu_data, track_analysis):
        """融合IMU和视觉判断坡道"""
        pitch = imu_data['pitch']
        
        if abs(pitch) > 5:
            if pitch > 0:
                return 'UPHILL'
            else:
                return 'DOWNHILL'
        
        # 视觉辅助判断（赛道宽度变化）
        width_change = track_analysis.get('width_change_rate', 0)
        if width_change < -0.1:  # 赛道变窄→上坡
            return 'UPHILL'
        elif width_change > 0.1:  # 赛道变宽→下坡
            return 'DOWNHILL'
        
        return 'FLAT'
    
    def detect_curve(self, frame):
        """检测弯道"""
        h, w = frame.shape[:2]
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        track_mask = cv2.inRange(hsv, (0, 0, 0), (180, 255, 50))
        
        # 上半部分ROI
        roi_upper = track_mask[int(h*0.3):int(h*0.6), :]
        roi_lower = track_mask[int(h*0.6):h, :]
        
        # 比较上下区域中心
        moments_upper = cv2.moments(roi_upper)
        moments_lower = cv2.moments(roi_lower)
        
        if moments_upper['m00'] > 0 and moments_lower['m00'] > 0:
            cx_upper = moments_upper['m10'] / moments_upper['m00']
            cx_lower = moments_lower['m10'] / moments_lower['m00']
            deviation = cx_upper - cx_lower
            
            if abs(deviation) > 50:
                return 'LEFT_CURVE' if deviation < 0 else 'RIGHT_CURVE'
        
        return 'STRAIGHT'
```

### 4.2 能量管理系统

```python
# energy_manager.py
import time
from enum import Enum

class DriveMode(Enum):
    POWER_DRIVE = 1      # 电机驱动（平坦/上坡）
    REGEN_BRAKE = 2      # 再生制动发电（下坡）
    COAST = 3            # 滑行（节能）
    BOOST = 4            # 超级电容放电助力（陡坡）

class EnergyManager:
    """能量回收与管理"""
    def __init__(self):
        self.battery_voltage = 7.4    # 电池电压
        self.cap_voltage = 0.0        # 超级电容电压
        self.cap_max = 33.0           # 6串5.5V
        self.cap_min = 6.0            # 最低工作电压
        
        self.energy_stored = 0.0      # 累计回收能量 J
        self.energy_consumed = 0.0    # 累计消耗能量 J
        
        # 能量阈值
        self.boost_threshold = 0.3    # 电容电量>30%时可用
        self.regen_threshold = 0.9    # 电容电量<90%时可充电
        
    def get_cap_soc(self):
        """超级电容荷电状态"""
        if self.cap_max == self.cap_min:
            return 0
        return (self.cap_voltage - self.cap_min) / (self.cap_max - self.cap_min)
    
    def decide_mode(self, slope, speed, cap_soc):
        """根据坡道、速度、电量决定驱动模式"""
        if slope == 'DOWNHILL':
            if cap_soc < self.regen_threshold:
                return DriveMode.REGEN_BRAKE  # 下坡回收能量
            else:
                return DriveMode.COAST         # 电容满了就滑行
        
        elif slope == 'UPHILL':
            if cap_soc > self.boost_threshold:
                return DriveMode.BOOST          # 电容助力上坡
            else:
                return DriveMode.POWER_DRIVE    # 电池驱动
        
        else:  # FLAT
            if speed < 0.3:
                return DriveMode.POWER_DRIVE    # 速度太低，驱动
            elif speed > 1.0:
                return DriveMode.COAST          # 速度够了，滑行
            else:
                return DriveMode.POWER_DRIVE
    
    def regen_brake(self, motor_ctrl, target_current=2.0):
        """再生制动发电"""
        motor_ctrl.set_mode('regen')
        motor_ctrl.set_regen_current(target_current)
    
    def boost_drive(self, motor_ctrl, cap_discharge, power_boost=20):
        """超级电容助力"""
        cap_discharge.enable(power_boost)  # W
        motor_ctrl.set_mode('drive')
```

### 4.3 电机控制器

```python
# motor_controller.py
import serial
import struct

class MotorController:
    """双向BLDC电机控制"""
    def __init__(self, uart_port='/dev/ttyS3'):
        self.ser = serial.Serial(uart_port, 115200)
        self.current_mode = 'stop'
    
    def set_mode(self, mode):
        """设置工作模式: drive/regen/brake/coast"""
        self.current_mode = mode
        cmd = {'drive': 0x01, 'regen': 0x02, 'brake': 0x03, 'coast': 0x04}
        self.ser.write(struct.pack('BB', 0xAA, cmd.get(mode, 0)))
    
    def set_speed(self, speed_pct):
        """设置速度 (-100 ~ 100%)"""
        speed_pct = max(-100, min(100, speed_pct))
        self.ser.write(struct.pack('<Bb', 0xBB, int(speed_pct)))
    
    def set_regen_current(self, current_a):
        """设置再生制动电流"""
        current_ma = int(current_a * 1000)
        self.ser.write(struct.pack('<BH', 0xCC, current_ma))
    
    def get_speed(self):
        """读取当前速度"""
        self.ser.write(b'\xDD')
        data = self.ser.read(4)
        if len(data) == 4:
            return struct.unpack('<f', data)[0]
        return 0.0
    
    def get_current(self):
        """读取电机电流"""
        self.ser.write(b'\xDE')
        data = self.ser.read(4)
        if len(data) == 4:
            return struct.unpack('<f', data)[0]
        return 0.0
```

### 4.4 路径跟踪控制器

```python
# path_tracker.py
import numpy as np

class PathTracker:
    """赛道跟踪控制器"""
    def __init__(self):
        self.pid_steer = PIDController(kp=1.5, ki=0.1, kd=0.5)
        self.pid_speed = PIDController(kp=2.0, ki=0.5, kd=0.1)
        
        self.target_speed = 1.0      # 目标速度 m/s
        self.curve_speed = 0.5       # 弯道减速
        self.climb_speed = 0.6       # 爬坡减速
        
    def compute(self, track_result, slope, curve):
        """计算控制输出"""
        # 转向控制
        steer = self.pid_steer.compute(track_result['center_offset'])
        steer = np.clip(steer, -100, 100)
        
        # 速度控制
        if curve != 'STRAIGHT':
            target = self.curve_speed
        elif slope == 'UPHILL':
            target = self.climb_speed
        else:
            target = self.target_speed
        
        speed = self.pid_speed.compute(target - self.current_speed)
        speed = np.clip(speed, 0, 100)
        
        return steer, speed


class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
    
    def compute(self, error):
        self.integral += error
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
```

```python
# main.py
import cv2
import time

def main():
    camera = cv2.VideoCapture(0)
    track_det = TrackDetector('track_seg.rknn')
    energy = EnergyManager()
    motor = MotorController('/dev/ttyS3')
    tracker = PathTracker()
    imu = IMUReader('/dev/ttyS4')
    
    lap_count = 0
    start_time = time.time()
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        # 1. 赛道分析
        track = track_det.analyze_track(frame)
        curve = track_det.detect_curve(frame)
        imu_data = imu.read()
        slope = track_det.detect_slope(imu_data, track)
        
        # 2. 能量决策
        cap_soc = energy.get_cap_soc()
        drive_mode = energy.decide_mode(slope, motor.get_speed(), cap_soc)
        
        # 3. 执行驱动模式
        if drive_mode == DriveMode.REGEN_BRAKE:
            energy.regen_brake(motor, target_current=2.0)
            print(f"[REGEN] 回收能量, 电容SOC={cap_soc:.1%}")
            
        elif drive_mode == DriveMode.BOOST:
            energy.boost_drive(motor, cap_discharge=None, power_boost=20)
            print(f"[BOOST] 电容助力, 电容SOC={cap_soc:.1%}")
            
        elif drive_mode == DriveMode.POWER_DRIVE:
            motor.set_mode('drive')
            
        elif drive_mode == DriveMode.COAST:
            motor.set_mode('coast')
        
        # 4. 路径跟踪
        steer, speed = tracker.compute(track, slope, curve)
        motor.set_speed(int(speed))
        # 差速转向
        left_speed = speed + steer * 0.3
        right_speed = speed - steer * 0.3
        
        # 5. 统计能量
        current = motor.get_current()
        voltage = energy.battery_voltage
        power = current * voltage
        if drive_mode == DriveMode.REGEN_BRAKE:
            energy.energy_stored += power * 0.1
        else:
            energy.energy_consumed += power * 0.1
        
        # 6. 圈数检测
        if track.get('lap_marker_detected'):
            lap_count += 1
            print(f"[LAP] 第{lap_count}圈完成")
        
        time.sleep(0.05)  # 20Hz

if __name__ == '__main__':
    main()
```

## 五、关键技术要点

### 5.1 能量管理策略
| 场景 | 策略 | 说明 |
|------|------|------|
| 下坡 | 再生制动 | BLDC反转发电→超级电容 |
| 上坡(有储能) | 电容+电池联合 | 双电源Boost输出 |
| 上坡(无储能) | 电池驱动 | 降速爬坡 |
| 平路(高速) | 滑行 | 利用惯性 |
| 弯道 | 减速+差速 | 降低能量浪费 |

### 5.2 坡道检测融合
- **IMU**：加速度计直接测倾角
- **视觉**：赛道宽度变化率、透视消失点位移
- **编码器**：速度变化率推断坡度
