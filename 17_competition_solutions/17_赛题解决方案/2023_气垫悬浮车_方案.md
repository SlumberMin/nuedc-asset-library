## 一、题目分析

### 核心要求
设计制作一个气垫悬浮车，在光滑平面上自主行驶：
- 通过风扇产生气垫实现悬浮（减少摩擦）
- 循指定轨迹线行驶（黑色/白色线条）
- 避开障碍物
- 记录行驶轨迹和时间
- 全程自主，无遥控

### 技术挑战
| 挑战 | 难度 | OPi5 解决方案 |
|------|------|---------------|
| 循线行驶 | ★★★★ | OpenCV + NPU 边缘检测 |
| 避障 | ★★★ | 超声波 + 视觉融合 |
| 气垫稳定性 | ★★★ | PID 调节风扇转速 |
| 低摩擦转向 | ★★★ | 差速/舵机转向 |

### 4.1 循线算法

```python
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class LineFollower:
    """视觉循线控制器"""
    
    def __init__(self, camera_matrix=None, camera_dist=None):
        # 相机参数
        self.K = camera_matrix
        self.D = camera_dist
        
        # 循线参数
        self.line_color = 'black'  # 黑线
        self.roi_top_ratio = 0.4   # ROI顶部比例
        self.roi_bottom_ratio = 1.0
        
        # PID参数
        self.Kp = 0.8
        self.Ki = 0.01
        self.Kd = 0.3
        self.prev_error = 0
        self.integral = 0
        
        # 状态
        self.line_detected = False
        self.line_center = 0
        self.line_angle = 0
        self.crossroad_detected = False
    
    def preprocess_frame(self, frame):
        """图像预处理"""
        # 裁剪ROI (只看下方区域)
        h, w = frame.shape[:2]
        roi_top = int(h * self.roi_top_ratio)
        roi = frame[roi_top:, :]
        
        # 转灰度
        gray = cv2.cvtColor(roi, cv2.COLOR_BGR2GRAY)
        
        # 高斯模糊
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        
        return roi, blurred
    
    def detect_line(self, frame):
        """
        检测黑线
        返回: 线中心x坐标, 线角度
        """
        roi, gray = self.preprocess_frame(frame)
        h, w = gray.shape[:2]
        
        # 自适应阈值 (适应不同光照)
        if self.line_color == 'black':
            thresh = cv2.adaptiveThreshold(
                gray, 255, cv2.ADAPTIVE_THRESH_GAUSSIAN_C,
                cv2.THRESH_BINARY_INV, 11, 2
            )
        else:
            _, thresh = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        
        # 形态学处理
        kernel = np.ones((3, 3), np.uint8)
        thresh = cv2.morphologyEx(thresh, cv2.MORPH_CLOSE, kernel, iterations=2)
        thresh = cv2.morphologyEx(thresh, cv2.MORPH_OPEN, kernel, iterations=1)
        
        # 分行扫描找线中心
        row_centers = []
        scan_rows = [int(h * r) for r in [0.2, 0.4, 0.6, 0.8]]
        
        for row_y in scan_rows:
            row = thresh[row_y, :]
            # 找黑色像素的中心
            black_pixels = np.where(row < 128)[0]
            if len(black_pixels) > 10:  # 至少10个黑像素
                center_x = np.mean(black_pixels)
                row_centers.append((center_x, row_y))
        
        if len(row_centers) >= 2:
            self.line_detected = True
            
            # 计算线中心 (加权平均)
            centers = [c[0] for c in row_centers]
            self.line_center = np.mean(centers)
            
            # 计算线角度
            if len(row_centers) >= 2:
                pts = np.array(row_centers)
                vx, vy, x0, y0 = cv2.fitLine(pts.astype(np.float32), 
                                               cv2.DIST_L2, 0, 0.01, 0.01)
                self.line_angle = np.degrees(np.arctan2(vy, vx))
            
            # 检测十字路口
            self.crossroad_detected = self._detect_crossroad(thresh)
            
            return self.line_center, self.line_angle
        
        self.line_detected = False
        return None, None
    
    def _detect_crossroad(self, thresh):
        """检测十字路口"""
        h, w = thresh.shape[:2]
        # 在图像下半部分检测水平线
        roi = thresh[int(h*0.6):, :]
        horizontal_kernel = cv2.getStructuringElement(cv2.MORPH_RECT, (25, 1))
        horizontal = cv2.morphologyEx(roi, cv2.MORPH_OPEN, horizontal_kernel)
        
        # 如果水平线足够长，认为是十字路口
        contours, _ = cv2.findContours(horizontal, cv2.RETR_EXTERNAL, 
                                        cv2.CHAIN_APPROX_SIMPLE)
        for cnt in contours:
            if cv2.contourArea(cnt) > 500:
                return True
        return False
    
    def compute_steering(self, frame_width):
        """
        计算转向控制量
        返回: steering (-1.0 到 1.0)
        """
        if not self.line_detected:
            return 0  # 丢失线时直行
        
        # 计算偏差 (归一化)
        error = (self.line_center - frame_width / 2) / (frame_width / 2)
        
        # PID计算
        self.integral += error
        self.integral = np.clip(self.integral, -1.0, 1.0)  # 积分限幅
        derivative = error - self.prev_error
        
        steering = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        self.prev_error = error
        
        return np.clip(steering, -1.0, 1.0)
```

### 4.2 避障控制器

```python
import time
import OPi.GPIO as GPIO

class UltrasonicSensor:
    """超声波传感器"""
    
    def __init__(self, trig_pin, echo_pin):
        self.trig = trig_pin
        self.echo = echo_pin
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.trig, GPIO.OUT)
        GPIO.setup(self.echo, GPIO.IN)
    
    def measure_distance(self, timeout=0.05):
        """测量距离 (cm)"""
        # 发送触发脉冲
        GPIO.output(self.trig, True)
        time.sleep(0.00001)
        GPIO.output(self.trig, False)
        
        start_time = time.time()
        stop_time = time.time()
        
        # 等待回波
        timeout_start = time.time()
        while GPIO.input(self.echo) == 0:
            start_time = time.time()
            if time.time() - timeout_start > timeout:
                return -1  # 超时
        
        timeout_start = time.time()
        while GPIO.input(self.echo) == 1:
            stop_time = time.time()
            if time.time() - timeout_start > timeout:
                return -1  # 超时
        
        # 计算距离
        elapsed = stop_time - start_time
        distance = (elapsed * 34300) / 2  # cm
        
        return distance


class ObstacleAvoider:
    """避障控制器"""
    
    def __init__(self):
        # 三个超声波
        self.front_sensor = UltrasonicSensor(trig_pin=9, echo_pin=10)
        self.left_sensor = UltrasonicSensor(trig_pin=11, echo_pin=12)
        self.right_sensor = UltrasonicSensor(trig_pin=13, echo_pin=14)
        
        # 避障阈值
        self.danger_distance = 20   # cm
        self.warning_distance = 35  # cm
        
        # 避障状态
        self.obstacle_front = False
        self.obstacle_left = False
        self.obstacle_right = False
    
    def scan(self):
        """扫描障碍物"""
        self.dist_front = self.front_sensor.measure_distance()
        self.dist_left = self.left_sensor.measure_distance()
        self.dist_right = self.right_sensor.measure_distance()
        
        self.obstacle_front = 0 < self.dist_front < self.danger_distance
        self.obstacle_left = 0 < self.dist_left < self.warning_distance
        self.obstacle_right = 0 < self.dist_right < self.warning_distance
        
        return self.dist_front, self.dist_left, self.dist_right
    
    def get_avoidance_action(self):
        """
        获取避障动作
        返回: (steering, speed_factor)
        """
        if self.obstacle_front:
            # 前方有障碍，选择左右绕行
            if self.dist_left > self.dist_right:
                return -0.8, 0.3  # 向左转，减速
            else:
                return 0.8, 0.3   # 向右转，减速
        
        elif self.obstacle_left:
            return 0.3, 0.7  # 轻微右转
        
        elif self.obstacle_right:
            return -0.3, 0.7  # 轻微左转
        
        return 0, 1.0  # 无障碍，正常行驶
```

```python
import OPi.GPIO as GPIO

class HoverController:
    """气垫悬浮控制"""
    
    def __init__(self):
        # 风扇PWM引脚
        self.LEFT_FAN_PIN = 1
        self.RIGHT_FAN_PIN = 4
        self.THRUST_PIN = 7
        
        GPIO.setmode(GPIO.BOARD)
        GPIO.setup(self.LEFT_FAN_PIN, GPIO.OUT)
        GPIO.setup(self.RIGHT_FAN_PIN, GPIO.OUT)
        GPIO.setup(self.THRUST_PIN, GPIO.OUT)
        
        self.left_pwm = GPIO.PWM(self.LEFT_FAN_PIN, 25000)  # 25kHz
        self.right_pwm = GPIO.PWM(self.RIGHT_FAN_PIN, 25000)
        self.thrust_pwm = GPIO.PWM(self.THRUST_PIN, 25000)
        
        self.left_pwm.start(0)
        self.right_pwm.start(0)
        self.thrust_pwm.start(0)
        
        # 悬浮参数
        self.hover_duty = 60  # 悬浮基础占空比%
        self.max_duty = 95
        
        # PID for 倾斜补偿
        self.Kp_tilt = 2.0
        self.Ki_tilt = 0.0
        self.Kd_tilt = 0.5
        self.prev_tilt_error = 0
    
    def set_hover_power(self, left_duty, right_duty):
        """设置悬浮风扇功率"""
        left_duty = np.clip(left_duty, 0, self.max_duty)
        right_duty = np.clip(right_duty, 0, self.max_duty)
        
        self.left_pwm.ChangeDutyCycle(left_duty)
        self.right_pwm.ChangeDutyCycle(right_duty)
    
    def set_thrust(self, duty):
        """设置推进风扇功率"""
        duty = np.clip(duty, 0, self.max_duty)
        self.thrust_pwm.ChangeDutyCycle(duty)
    
    def compensate_tilt(self, roll_angle):
        """
        根据MPU6050检测到的倾斜角补偿
        roll_angle: 横滚角 (度)
        """
        error = roll_angle
        self.prev_tilt_error = error
        
        # 倾斜补偿: 增大低侧风扇功率
        compensation = self.Kp_tilt * error + self.Kd_tilt * (error - self.prev_tilt_error)
        
        left_duty = self.hover_duty + compensation
        right_duty = self.hover_duty - compensation
        
        self.set_hover_power(left_duty, right_duty)
    
    def adjust_for_steering(self, steering):
        """
        转向时差速悬浮
        steering: -1.0 到 1.0
        """
        differential = steering * 10  # 最大10%差速
        
        left_duty = self.hover_duty + differential
        right_duty = self.hover_duty - differential
        
        self.set_hover_power(left_duty, right_duty)
```

```python
#!/usr/bin/env python3
"""
气垫悬浮车主程序
Orange Pi 5 视觉循线 + 避障 + 气垫控制
"""

import cv2
import numpy as np
import time
import threading
from enum import Enum

from line_follower import LineFollower
from obstacle_avoider import ObstacleAvoider
from hover_controller import HoverController
from mpu6050_reader import MPU6050Reader

import OPi.GPIO as GPIO

class VehicleState(Enum):
    IDLE = 0
    HOVERING = 1
    FOLLOWING = 2
    AVOIDING = 3
    CROSSROAD = 4
    STOPPED = 5

class HoverVehicle:
    def __init__(self):
        # 初始化组件
        self.line_follower = LineFollower()
        self.obstacle_avoider = ObstacleAvoider()
        self.hover_ctrl = HoverController()
        self.imu = MPU6050Reader()
        
        # 摄像头
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)
        self.cap.set(cv2.CAP_PROP_FPS, 30)
        
        # 转向舵机
        self.STEER_PIN = 8
        GPIO.setup(self.STEER_PIN, GPIO.OUT)
        self.steer_pwm = GPIO.PWM(self.STEER_PIN, 50)
        self.steer_pwm.start(7.5)
        
        # 状态
        self.state = VehicleState.IDLE
        self.running = False
        self.lap_count = 0
        self.start_time = 0
        
        # 帧率
        self.fps = 0
        self.frame_count = 0
        self.fps_timer = time.time()
    
    def set_steering(self, value):
        """
        设置转向
        value: -1.0 (左) 到 1.0 (右)
        """
        duty = 7.5 + value * 2.5  # 5.0 ~ 10.0
        duty = np.clip(duty, 5.0, 10.0)
        self.steer_pwm.ChangeDutyCycle(duty)
    
    def start(self):
        """启动系统"""
        self.running = True
        self.state = VehicleState.HOVERING
        self.start_time = time.time()
        
        # 启动悬浮
        self.hover_ctrl.set_hover_power(60, 60)
        time.sleep(2)  # 等待气垫建立
        
        self.main_loop()
    
    def main_loop(self):
        """主控制循环"""
        while self.running:
            try:
                # 读取摄像头
                ret, frame = self.cap.read()
                if not ret:
                    continue
                
                # 读取IMU姿态
                accel, gyro = self.imu.read()
                roll = self.imu.compute_roll(accel)
                
                # 超声波避障扫描 (异步)
                dist_f, dist_l, dist_r = self.obstacle_avoider.scan()
                
                # 根据状态处理
                if self.state == VehicleState.HOVERING:
                    self.handle_hovering(roll)
                    
                elif self.state == VehicleState.FOLLOWING:
                    self.handle_following(frame, roll, dist_f, dist_l, dist_r)
                    
                elif self.state == VehicleState.AVOIDING:
                    self.handle_avoiding(frame, roll)
                    
                elif self.state == VehicleState.CROSSROAD:
                    self.handle_crossroad(frame)
                
                # 更新FPS
                self.update_fps()
                
                # 显示 (调试用)
                cv2.imshow('Camera', frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break
                
            except KeyboardInterrupt:
                break
            except Exception as e:
                print(f"Error: {e}")
        
        self.stop()
    
    def handle_hovering(self, roll):
        """悬浮状态处理"""
        # 倾斜补偿
        self.hover_ctrl.compensate_tilt(roll)
        
        # 稳定后切换到循线模式
        if abs(roll) < 5:  # 倾斜小于5度
            self.state = VehicleState.FOLLOWING
            print("进入循线模式")
    
    def handle_following(self, frame, roll, dist_f, dist_l, dist_r):
        """循线状态处理"""
        # 检查障碍物
        if self.obstacle_avoider.obstacle_front:
            self.state = VehicleState.AVOIDING
            print("检测到障碍物，进入避障模式")
            return
        
        # 循线检测
        line_center, line_angle = self.line_follower.detect_line(frame)
        
        if self.line_follower.line_detected:
            # 计算转向
            steering = self.line_follower.compute_steering(frame.shape[1])
            
            # 检测十字路口
            if self.line_follower.crossroad_detected:
                self.state = VehicleState.CROSSROAD
                self.lap_count += 1
                print(f"检测到十字路口，圈数: {self.lap_count}")
                return
            
            # 执行转向
            self.set_steering(steering)
            
            # 气垫差速补偿
            self.hover_ctrl.adjust_for_steering(steering)
            
            # 推进
            speed_factor = 1.0 - abs(steering) * 0.3  # 转弯减速
            self.hover_ctrl.set_thrust(50 * speed_factor)
        else:
            # 丢失线，减速慢行
            self.hover_ctrl.set_thrust(30)
            
            # 尝试寻线 (小幅摆动搜索)
            self.search_line()
    
    def handle_avoiding(self, frame, roll):
        """避障状态处理"""
        steering, speed_factor = self.obstacle_avoider.get_avoidance_action()
        
        self.set_steering(steering)
        self.hover_ctrl.adjust_for_steering(steering)
        self.hover_ctrl.set_thrust(40 * speed_factor)
        
        # 持续检测直到障碍消失
        self.obstacle_avoider.scan()
        if not self.obstacle_avoider.obstacle_front:
            self.state = VehicleState.FOLLOWING
            print("障碍物已避开，恢复循线")
    
    def handle_crossroad(self, frame):
        """十字路口处理"""
        # 直行通过路口
        self.set_steering(0)
        self.hover_ctrl.set_thrust(50)
        time.sleep(0.5)  # 通过路口
        
        self.state = VehicleState.FOLLOWING
    
    def search_line(self):
        """丢失线路时搜索"""
        # 小幅摆动搜索
        t = time.time() % 2
        if t < 1:
            self.set_steering(0.3)
        else:
            self.set_steering(-0.3)
    
    def update_fps(self):
        """更新帧率统计"""
        self.frame_count += 1
        if time.time() - self.fps_timer > 1.0:
            self.fps = self.frame_count
            self.frame_count = 0
            self.fps_timer = time.time()
            print(f"FPS: {self.fps}, 状态: {self.state.name}")
    
    def stop(self):
        """停止系统"""
        self.running = False
        self.hover_ctrl.set_hover_power(0, 0)
        self.hover_ctrl.set_thrust(0)
        self.set_steering(0)
        self.cap.release()
        cv2.destroyAllWindows()
        GPIO.cleanup()
        
        elapsed = time.time() - self.start_time
        print(f"总时间: {elapsed:.1f}s, 圈数: {self.lap_count}")

def main():
    vehicle = HoverVehicle()
    try:
        vehicle.start()
    except KeyboardInterrupt:
        vehicle.stop()

if __name__ == '__main__':
    main()
```

## 七、总结

| 指标 | 目标值 | 实现方法 |
|------|--------|---------|
| 循线精度 | ±2cm | 自适应阈值+PID |
| 避障距离 | 20cm检测 | 超声波三角测距 |
| 最高车速 | 1m/s | 优化PID响应 |
| 视觉帧率 | ≥30fps | NPU加速处理 |

**OPi5优势**：NPU 实时处理循线视觉，CPU 多核并行处理传感器数据和电机控制，4GB 内存足够。
