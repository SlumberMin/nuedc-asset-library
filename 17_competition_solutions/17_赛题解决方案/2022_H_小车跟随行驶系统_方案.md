## 二、视觉方案（Orange Pi 5）

### 2.1 前车标识系统

```python
# 前车标识：ArUco标记 + 顶部颜色标识
import cv2
import numpy as np
from picamera2 import Picamera2

class FrontCarMarker:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # ArUco检测器
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        
        # 标记尺寸（cm）
        self.marker_size_cm = 10.0
        
    def detect_front_car(self):
        """检测前车位置和距离"""
        frame = self.picam2.capture_array()
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)
        
        corners, ids, _ = self.detector.detectMarkers(gray)
        
        if ids is not None and len(ids) > 0:
            # 取最大标记（最近/最显著）
            best_idx = 0
            best_area = 0
            for i, corner in enumerate(corners):
                area = cv2.contourArea(corner[0])
                if area > best_area:
                    best_area = area
                    best_idx = i
            
            corner = corners[best_idx][0]
            
            # 计算中心点
            cx = int(np.mean(corner[:, 0]))
            cy = int(np.mean(corner[:, 1]))
            
            # 基于标记像素宽度估算距离
            pixel_width = np.linalg.norm(corner[0] - corner[1])
            focal_length = 650  # 标定得到
            distance_cm = (self.marker_size_cm * focal_length) / pixel_width
            
            # 水平偏移量（像素）
            frame_center = 320
            offset_x = cx - frame_center
            
            return {
                'detected': True,
                'cx': cx, 'cy': cy,
                'distance_cm': distance_cm,
                'offset_x': offset_x,
                'area': best_area,
                'id': int(ids[best_idx][0])
            }
        
        return {'detected': False}
```

### 2.2 距离PID控制

```python
class FollowController:
    def __init__(self, target_distance=50.0):
        self.target_distance = target_distance  # cm
        
        # 距离PID
        self.dist_kp = 0.8
        self.dist_ki = 0.01
        self.dist_kd = 0.3
        self.dist_integral = 0
        self.dist_prev_error = 0
        
        # 方向PID
        self.dir_kp = 0.5
        self.dir_ki = 0.005
        self.dir_kd = 0.2
        self.dir_integral = 0
        self.dir_prev_error = 0
        
        self.max_speed = 200
        self.min_speed = -100
        
    def compute(self, distance_cm, offset_x):
        """计算左右轮速度"""
        # 距离PID
        dist_error = distance_cm - self.target_distance
        self.dist_integral += dist_error
        dist_derivative = dist_error - self.dist_prev_error
        self.dist_prev_error = dist_error
        
        forward = (self.dist_kp * dist_error + 
                   self.dist_ki * self.dist_integral +
                   self.dist_kd * dist_derivative)
        
        # 方向PID（居中跟随）
        dir_error = offset_x
        self.dir_integral += dir_error
        dir_derivative = dir_error - self.dir_prev_error
        self.dir_prev_error = dir_error
        
        steering = (self.dir_kp * dir_error + 
                    self.dir_ki * self.dir_integral +
                    self.dir_kd * dir_derivative)
        
        # 限幅
        forward = np.clip(forward, self.min_speed, self.max_speed)
        steering = np.clip(steering, -80, 80)
        
        left_speed = int(np.clip(forward + steering, self.min_speed, self.max_speed))
        right_speed = int(np.clip(forward - steering, self.min_speed, self.max_speed))
        
        return left_speed, right_speed
```

### 2.3 超车与领跑模式

```python
class OvertakeController:
    def __init__(self):
        self.state = 'FOLLOW'  # FOLLOW, OVERTAKE, LEAD
        self.overtake_phase = 0
        
    def check_overtake_condition(self, front_car_info):
        """检测超车条件"""
        if not front_car_info['detected']:
            return False
        # 前车速度过慢或距离过近
        return front_car_info['distance_cm'] < 30
        
    def execute_overtake(self, front_car_info):
        """执行超车动作序列"""
        if self.overtake_phase == 0:
            # 向左偏移
            return 150, 50  # 左转
        elif self.overtake_phase == 1:
            # 直行超越
            return 180, 180
        elif self.overtake_phase == 2:
            # 向右回到原车道
            return 50, 150
        else:
            self.state = 'LEAD'
            return 0, 0
```

void Motor_SetSpeed(int16_t left, int16_t right) {
    // 左电机
    if (left >= 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, left * 10);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_12, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_1, -left * 10);
    }
    // 右电机
    if (right >= 0) {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_SET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, right * 10);
    } else {
        HAL_GPIO_WritePin(GPIOB, GPIO_PIN_13, GPIO_PIN_RESET);
        __HAL_TIM_SET_COMPARE(&htim2, TIM_CHANNEL_2, -right * 10);
    }
}

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2022年H题 - 小车跟随行驶系统 主程序（后车）
"""
import cv2
import numpy as np
import serial
import time
from picamera2 import Picamera2

class FollowCarSystem:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # ArUco
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.detector = cv2.aruco.ArucoDetector(
            self.aruco_dict, cv2.aruco.DetectorParameters()
        )
        
        # 串口
        self.ser = serial.Serial('/dev/ttyS5', 115200, timeout=0.01)
        
        # PID参数
        self.target_dist = 50.0  # cm
        self.dist_pid = PIDController(kp=0.8, ki=0.01, kd=0.3)
        self.dir_pid = PIDController(kp=0.5, ki=0.005, kd=0.2)
        
        # 超车状态机
        self.state = 'FOLLOW'
        self.focal_length = 650
        self.marker_size = 10.0
        
    def detect_front_car(self):
        frame = self.picam2.capture_array()
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)
        corners, ids, _ = self.detector.detectMarkers(gray)
        
        if ids is not None and len(ids) > 0:
            corner = corners[0][0]
            cx = int(np.mean(corner[:, 0]))
            pixel_width = np.linalg.norm(corner[0] - corner[1])
            distance = (self.marker_size * self.focal_length) / pixel_width
            offset = cx - 320
            return True, distance, offset
        return False, 0, 0
        
    def send_command(self, left, right):
        cmd = bytes([0xAA, left & 0xFF, right & 0xFF, 0x55])
        self.ser.write(cmd)
        
    def run(self):
        print("跟随行驶系统启动...")
        while True:
            detected, distance, offset = self.detect_front_car()
            
            if detected:
                forward = self.dist_pid.compute(self.target_dist, distance)
                steering = self.dir_pid.compute(0, offset)
                
                left = int(np.clip(forward + steering, -200, 200))
                right = int(np.clip(forward - steering, -200, 200))
                
                self.send_command(left, right)
                print(f"距离:{distance:.1f}cm 偏移:{offset}px L:{left} R:{right}")
            else:
                # 丢失目标，减速停车
                self.send_command(0, 0)
                self.dist_pid.reset()
                self.dir_pid.reset()
                
            time.sleep(0.03)  # ~30fps

class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
        
    def compute(self, target, current):
        error = target - current
        self.integral += error
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative
        
    def reset(self):
        self.integral = 0
        self.prev_error = 0

if __name__ == '__main__':
    system = FollowCarSystem()
    system.run()
```

## 七、调试要点

1. **ArUco标定**：确保不同距离、角度下稳定识别
2. **PID整定**：先调距离环，再调方向环
3. **通信可靠性**：NRF24L01丢包率需<1%
4. **超车逻辑**：状态机转换条件需严格测试
5. **光照适应**：不同环境下调整曝光参数
