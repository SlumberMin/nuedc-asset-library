### 4.1 姿态与位置估计

```python
# state_estimator.py
import numpy as np
import math

class ExtendedKalmanFilter:
    """扩展卡尔曼滤波器 - 飞行器状态估计"""
    def __init__(self):
        # 状态: [x, y, z, vx, vy, vz, roll, pitch, yaw]
        self.x = np.zeros(9)
        self.P = np.eye(9) * 1.0
        
        # 过程噪声
        self.Q = np.eye(9) * 0.01
        self.Q[0:3, 0:3] *= 0.1   # 位置
        self.Q[3:6, 3:6] *= 0.5   # 速度
        self.Q[6:9, 6:9] *= 0.01  # 姿态
        
        # 测量噪声
        self.R_visual = np.eye(3) * 0.05   # 视觉位置
        self.R_imu = np.eye(3) * 0.01      # IMU姿态
        self.R_alt = np.array([[0.02]])     # 高度
        self.R_flow = np.eye(2) * 0.1      # 光流
        
        self.dt = 0.05  # 20Hz
    
    def predict(self, accel_body, gyro_body, attitude):
        """预测步骤"""
        roll, pitch, yaw = attitude
        
        # 旋转矩阵（机体系→世界系）
        R = self._euler_to_rotation(roll, pitch, yaw)
        
        # 加速度转换到世界系（去除重力）
        accel_world = R @ accel_body + np.array([0, 0, 9.81])
        
        # 状态转移
        self.x[3:6] += accel_world * self.dt          # 速度更新
        self.x[0:3] += self.x[3:6] * self.dt          # 位置更新
        self.x[6:9] = attitude                         # 姿态直接用IMU
        
        # 协方差预测
        F = np.eye(9)
        F[0:3, 3:6] = np.eye(3) * self.dt
        self.P = F @ self.P @ F.T + self.Q
    
    def update_visual(self, z_visual):
        """视觉位置更新"""
        H = np.zeros((3, 9))
        H[0:3, 0:3] = np.eye(3)
        
        y = z_visual - H @ self.x
        S = H @ self.P @ H.T + self.R_visual
        K = self.P @ H.T @ np.linalg.inv(S)
        
        self.x += K @ y
        self.P = (np.eye(9) - K @ H) @ self.P
    
    def update_altitude(self, z_alt):
        """高度更新"""
        H = np.zeros((1, 9))
        H[0, 2] = 1
        
        y = z_alt - H @ self.x
        S = H @ self.P @ H.T + self.R_alt
        K = self.P @ H.T @ np.linalg.inv(S)
        
        self.x += K @ y
        self.P = (np.eye(9) - K @ H) @ self.P
    
    def update_flow(self, z_flow):
        """光流更新（水平位移）"""
        H = np.zeros((2, 9))
        H[0, 0] = 1
        H[1, 1] = 1
        
        y = z_flow - H @ self.x
        S = H @ self.P @ H.T + self.R_flow
        K = self.P @ H.T @ np.linalg.inv(S)
        
        self.x += K @ y
        self.P = (np.eye(9) - K @ H) @ self.P
    
    def get_state(self):
        """获取当前状态估计"""
        return {
            'position': self.x[0:3].copy(),
            'velocity': self.x[3:6].copy(),
            'attitude': self.x[6:9].copy()
        }
    
    def _euler_to_rotation(self, roll, pitch, yaw):
        cr, sr = math.cos(roll), math.sin(roll)
        cp, sp = math.cos(pitch), math.sin(pitch)
        cy, sy = math.cos(yaw), math.sin(yaw)
        
        R = np.array([
            [cy*cp, cy*sp*sr-sy*cr, cy*sp*cr+sy*sr],
            [sy*cp, sy*sp*sr+cy*cr, sy*sp*cr-cy*sr],
            [-sp,   cp*sr,          cp*cr]
        ])
        return R
```

### 4.2 姿态控制器

```python
# attitude_controller.py
import numpy as np
import math

class AttitudeController:
    """姿态控制器（内环）"""
    def __init__(self):
        # PID参数
        self.pid_roll = PIDController(kp=4.0, ki=0.2, kd=2.0)
        self.pid_pitch = PIDController(kp=4.0, ki=0.2, kd=2.0)
        self.pid_yaw = PIDController(kp=3.0, ki=0.0, kd=1.0)
        self.pid_alt = PIDController(kp=2.0, ki=0.5, kd=1.0)
        
        # 限幅
        self.max_tilt = math.radians(25)  # 最大倾斜角
        self.max_yaw_rate = 100            # deg/s
        
    def compute_attitude(self, desired_attitude, current_attitude, current_yaw_rate):
        """姿态控制"""
        roll_des, pitch_des, yaw_des = desired_attitude
        roll_cur, pitch_cur, yaw_cur = current_attitude
        
        # 限幅
        roll_des = max(-self.max_tilt, min(self.max_tilt, roll_des))
        pitch_des = max(-self.max_tilt, min(self.max_tilt, pitch_des))
        
        # PID
        roll_out = self.pid_roll.compute(roll_des - roll_cur)
        pitch_out = self.pid_pitch.compute(pitch_des - pitch_cur)
        
        yaw_err = yaw_des - yaw_cur
        yaw_err = (yaw_err + math.pi) % (2 * math.pi) - math.pi
        yaw_out = self.pid_yaw.compute(yaw_err)
        
        return roll_out, pitch_out, yaw_out
    
    def compute_altitude(self, desired_alt, current_alt, current_vz):
        """高度控制"""
        alt_err = desired_alt - current_alt
        throttle = self.pid_alt.compute(alt_err)
        throttle = max(0, min(100, throttle))
        return throttle


class PositionController:
    """位置控制器（外环 → 姿态指令）"""
    def __init__(self):
        self.pid_x = PIDController(kp=1.5, ki=0.05, kd=0.8)
        self.pid_y = PIDController(kp=1.5, ki=0.05, kd=0.8)
        
        self.max_tilt_cmd = math.radians(15)
    
    def compute(self, desired_pos, current_pos, current_yaw):
        """位置→姿态指令"""
        dx = desired_pos[0] - current_pos[0]
        dy = desired_pos[1] - current_pos[1]
        
        # 世界系误差→机体系
        cos_yaw = math.cos(current_yaw)
        sin_yaw = math.sin(current_yaw)
        
        body_dx = cos_yaw * dx + sin_yaw * dy
        body_dy = -sin_yaw * dx + cos_yaw * dy
        
        # PID → 姿态指令
        pitch_cmd = self.pid_x.compute(body_dx)   # 前进→pitch
        roll_cmd = self.pid_y.compute(body_dy)     # 侧移→roll
        
        # 限幅
        pitch_cmd = max(-self.max_tilt_cmd, min(self.max_tilt_cmd, pitch_cmd))
        roll_cmd = max(-self.max_tilt_cmd, min(self.max_tilt_cmd, roll_cmd))
        
        return roll_cmd, pitch_cmd


class PIDController:
    def __init__(self, kp, ki, kd, limit=float('inf')):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.limit = limit
        self.integral = 0
        self.prev_error = 0
    
    def compute(self, error):
        self.integral += error
        self.integral = max(-self.limit, min(self.limit, self.integral))
        derivative = error - self.prev_error
        self.prev_error = error
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        return max(-self.limit, min(self.limit, output))
    
    def reset(self):
        self.integral = 0
        self.prev_error = 0
```

### 4.3 地面标记系统

```python
# ground_marker.py
import cv2
import numpy as np

class GroundMarkerSystem:
    """地面标记检测系统"""
    def __init__(self, camera_matrix, dist_coeffs):
        self.camera_matrix = camera_matrix
        self.dist_coeffs = dist_coeffs
        
        # ArUco设置
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        
        # 标记尺寸
        self.marker_size = 0.20  # 20cm
        
        # 已知标记位置
        self.known_markers = {
            0: {'pos': (0, 0), 'type': 'landing'},
            1: {'pos': (2, 0), 'type': 'waypoint'},
            2: {'pos': (2, 2), 'type': 'waypoint'},
            3: {'pos': (0, 2), 'type': 'waypoint'},
        }
    
    def detect_and_localize(self, frame):
        """检测标记并估计位置"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, _ = self.detector.detectMarkers(gray)
        
        result = {
            'markers': [],
            'position': None,
            'heading': None,
            'target_visible': False
        }
        
        if ids is None:
            return result
        
        for i, mid in enumerate(ids.flatten()):
            # 位姿估计
            rvec, tvec, _ = cv2.aruco.estimatePoseSingleMarkers(
                corners[i], self.marker_size,
                self.camera_matrix, self.dist_coeffs)
            
            marker_info = {
                'id': int(mid),
                'rvec': rvec[0][0],
                'tvec': tvec[0][0],
                'corners': corners[i][0]
            }
            result['markers'].append(marker_info)
            
            # 起降标记
            if mid == 0:
                result['target_visible'] = True
        
        # 位置估计
        if result['markers']:
            m = result['markers'][0]
            tvec = m['tvec']
            
            # 简化：使用标记位置+偏移
            mid = m['id']
            if mid in self.known_markers:
                known_pos = self.known_markers[mid]['pos']
                result['position'] = np.array([
                    known_pos[0] - tvec[0],
                    known_pos[1] - tvec[1]
                ])
            
            # 航向
            rmat, _ = cv2.Rodrigues(m['rvec'])
            result['heading'] = np.arctan2(rmat[1, 0], rmat[0, 0])
        
        return result
    
    def draw_debug(self, frame, result):
        """绘制调试图像"""
        for m in result['markers']:
            cv2.aruco.drawDetectedMarkers(frame, [m['corners']])
            cv2.drawFrameAxes(frame, self.camera_matrix, self.dist_coeffs,
                            m['rvec'], m['tvec'], 0.1)
        
        if result['position'] is not None:
            pos = result['position']
            cv2.putText(frame, f"Pos: ({pos[0]:.2f}, {pos[1]:.2f})",
                       (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
        
        return frame
```

### 4.4 任务主流程

```python
# main.py
import cv2
import time
import math

def main():
    # 初始化
    camera = cv2.VideoCapture(0)
    camera.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    camera.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
    
    # 摄像头标定参数
    camera_matrix = load_camera_calibration('calib.yaml')
    dist_coeffs = load_distortion('calib.yaml')
    
    markers = GroundMarkerSystem(camera_matrix, dist_coeffs)
    estimator = ExtendedKalmanFilter()
    att_ctrl = AttitudeController()
    pos_ctrl = PositionController()
    flight = MAVLinkController('/dev/ttyS5')
    
    # 任务航点
    mission = [
        {'type': 'takeoff', 'alt': 2.0},
        {'type': 'waypoint', 'pos': (2.0, 0.0, 2.0)},
        {'type': 'waypoint', 'pos': (2.0, 2.0, 2.0)},
        {'type': 'waypoint', 'pos': (0.0, 2.0, 2.0)},
        {'type': 'land', 'pos': (0.0, 0.0)},
    ]
    
    state = 'TAKEOFF'
    mission_idx = 0
    takeoff_start = time.time()
    
    while True:
        ret, frame = camera.read()
        if not ret:
            continue
        
        # 1. 视觉定位
        vis_result = markers.detect_and_localize(frame)
        
        # 2. 状态估计融合
        imu_data = flight.get_imu()
        alt = flight.get_altitude()
        flow = flight.get_flow()
        
        estimator.predict(
            accel_body=imu_data['accel'],
            gyro_body=imu_data['gyro'],
            attitude=imu_data['attitude']
        )
        
        if vis_result['position'] is not None:
            estimator.update_visual(vis_result['position'])
        
        if alt is not None:
            estimator.update_altitude(alt)
        
        if flow is not None:
            estimator.update_flow(flow)
        
        state_est = estimator.get_state()
        pos = state_est['position']
        att = state_est['attitude']
        
        # 3. 任务执行
        current_mission = mission[mission_idx]
        
        if state == 'TAKEOFF':
            if time.time() - takeoff_start < 1.0:
                flight.arm()
            else:
                flight.takeoff(current_mission['alt'])
                if pos[2] > current_mission['alt'] * 0.9:
                    mission_idx += 1
                    state = 'NAVIGATE'
                    print("起飞完成")
        
        elif state == 'NAVIGATE':
            target = current_mission['pos']
            
            # 外环：位置→姿态指令
            roll_cmd, pitch_cmd = pos_ctrl.compute(
                desired_pos=target,
                current_pos=pos,
                current_yaw=att[2]
            )
            
            # 偏航保持
            yaw_cmd = 0
            
            # 内环：姿态控制
            roll_out, pitch_out, yaw_out = att_ctrl.compute_attitude(
                desired_attitude=(roll_cmd, pitch_cmd, yaw_cmd),
                current_attitude=att,
                current_yaw_rate=imu_data.get('yaw_rate', 0)
            )
            
            # 高度控制
            throttle = att_ctrl.compute_altitude(target[2], pos[2], state_est['velocity'][2])
            
            # 发送控制指令
            flight.set_attitude_target(roll_out, pitch_out, yaw_out, throttle)
            
            # 到达判断
            dist = math.sqrt((pos[0]-target[0])**2 + (pos[1]-target[1])**2)
            if dist < 0.3:
                mission_idx += 1
                print(f"到达航点 {mission_idx}: {target}")
                if mission_idx >= len(mission):
                    state = 'DONE'
                elif mission[mission_idx]['type'] == 'land':
                    state = 'LAND'
        
        elif state == 'LAND':
            target = current_mission['pos']
            
            # 精确降落
            if vis_result['target_visible']:
                # 视觉引导对准
                marker_pos = vis_result['position']
                if marker_pos is not None:
                    dist_to_target = math.sqrt(marker_pos[0]**2 + marker_pos[1]**2)
                    if dist_to_target < 0.15:
                        flight.land()
                        print("精确降落")
                    else:
                        # 对准后下降
                        flight.set_position_ned(
                            target[0], target[1], pos[2] - 0.1
                        )
            else:
                # 无标记，直接降落
                if pos[2] < 0.3:
                    flight.disarm()
                    state = 'DONE'
                else:
                    flight.set_position_ned(target[0], target[1], pos[2] - 0.2)
        
        elif state == 'DONE':
            print("任务完成！")
            break
        
        # 调试显示
        debug_frame = markers.draw_debug(frame, vis_result)
        cv2.imshow('Debug', debug_frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break
        
        time.sleep(0.05)  # 20Hz

if __name__ == '__main__':
    main()
```

### 5.1 控制回路架构
```
目标航点
    ↓
[位置控制器] → 姿态指令 (roll_cmd, pitch_cmd)
    ↓
[姿态控制器] → 电机混合指令
    ↓
[电机驱动] → 四个电机PWM
    ↑
[状态估计器] ← IMU + 视觉 + 光流 + 气压计
```

### 5.2 多源状态融合
| 传感器 | 信息 | 频率 | 精度 |
|--------|------|------|------|
| IMU | 姿态/角速度/加速度 | 400Hz | 姿态±1° |
| 视觉(ArUco) | 位置/航向 | 30Hz | ±5cm |
| 光流 | 水平位移 | 50Hz | ±10cm |
| 气压计 | 高度 | 50Hz | ±20cm |
| 激光测距 | 精确高度 | 100Hz | ±1cm |
