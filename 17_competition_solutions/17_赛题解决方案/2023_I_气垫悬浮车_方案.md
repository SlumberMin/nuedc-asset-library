## 二、视觉方案（Orange Pi 5）

### 2.1 气垫车循线与避障

```python
import cv2
import numpy as np
from picamera2 import Picamera2

class HovercraftVision:
    def __init__(self):
        self.picam2 = Picamera2()
        config = self.picam2.create_preview_configuration(
            main={"size": (640, 480), "format": "RGB888"}
        )
        self.picam2.configure(config)
        self.picam2.start()
        
        # 避障检测区域
        self.obstacle_rois = [
            (200, 100, 240, 200),   # 左前方
            (280, 100, 360, 200),   # 正前方
            (400, 100, 440, 200),   # 右前方
        ]
        
    def detect_line(self):
        """检测地面引导线"""
        frame = self.picam2.capture_array()
        
        # 底部ROI
        roi = frame[380:480, :]
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        
        # 二值化
        _, binary = cv2.threshold(gray, 80, 255, cv2.THRESH_BINARY_INV)
        
        # 扫描中线
        mid_row = binary[40, :]
        line_pos = np.where(mid_row > 0)[0]
        
        if len(line_pos) > 10:
            center = int(np.mean(line_pos))
            offset = center - 320
            return True, offset
        
        return False, 0
        
    def detect_obstacles(self):
        """检测前方障碍物"""
        frame = self.picam2.capture_array()
        hsv = cv2.cvtColor(frame, cv2.COLOR_RGB2HSV)
        
        obstacles = []
        
        # 检测彩色障碍物
        color_ranges = {
            'red': ([0, 100, 100], [10, 255, 255]),
            'blue': ([100, 100, 100], [130, 255, 255]),
            'yellow': ([20, 100, 100], [35, 255, 255]),
        }
        
        for color_name, (lower, upper) in color_ranges.items():
            mask = cv2.inRange(hsv, np.array(lower), np.array(upper))
            
            contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, 
                                           cv2.CHAIN_APPROX_SIMPLE)
            
            for cnt in contours:
                area = cv2.contourArea(cnt)
                if 1000 < area < 50000:
                    x, y, w, h = cv2.boundingRect(cnt)
                    
                    # 判断在哪个ROI区域
                    region = 'center'
                    cx = x + w // 2
                    if cx < 250:
                        region = 'left'
                    elif cx > 390:
                        region = 'right'
                    
                    # 估算距离（基于面积）
                    distance = 100000 / max(area, 1)
                    
                    obstacles.append({
                        'color': color_name,
                        'region': region,
                        'center': (cx, y + h // 2),
                        'area': area,
                        'distance': min(distance, 5),
                        'bbox': (x, y, w, h)
                    })
        
        return obstacles
        
    def detect_boundaries(self):
        """检测赛道边界"""
        frame = self.picam2.capture_array()
        roi = frame[350:480, :]
        
        # 检测白色边界线
        gray = cv2.cvtColor(roi, cv2.COLOR_RGB2GRAY)
        _, binary = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        
        # 左右边界
        left_boundary = np.where(binary[50, :200] > 0)[0]
        right_boundary = np.where(binary[50, 440:] > 0)[0]
        
        left_x = left_boundary[-1] if len(left_boundary) > 0 else 0
        right_x = right_boundary[0] + 440 if len(right_boundary) > 0 else 640
        
        return left_x, right_x
```

### 2.2 气垫高度视觉估计

```python
class HeightEstimator:
    def __init__(self):
        self.picam2 = Picamera2()
        self.ref_pattern_size = 5  # 参考图案实际尺寸 cm
        
    def estimate_height_from_ground_pattern(self, frame):
        """通过地面图案尺寸估算高度"""
        # 检测地面上的参考标记
        gray = cv2.cvtColor(frame, cv2.COLOR_RGB2GRAY)
        _, binary = cv2.threshold(gray, 200, 255, cv2.THRESH_BINARY)
        
        contours, _ = cv2.findContours(binary, cv2.RETR_EXTERNAL, 
                                       cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            area = cv2.contourArea(largest)
            
            if area > 100:
                # 面积越大说明越近（高度越低）
                height = 500 / max(np.sqrt(area), 1)  # 简化公式
                return np.clip(height, 2, 30)
        
        return 10  # 默认高度
```

typedef struct {
    // 升力电机（4个）
    uint16_t lift_motors[4];
    // 推进电机（2个，左右）
    int16_t thrust_left;
    int16_t thrust_right;
    // 舵面/方向控制
    int16_t direction_servo;
} HoverControl_t;

// PID控制器
typedef struct {
    float kp, ki, kd;
    float integral, prev_error;
    float output;
} PID_t;

// 高度控制
float Height_PID(PID_t *pid, float target, float current) {
    float error = target - current;
    pid->integral += error;
    pid->integral = CLAMP(pid->integral, -100, 100);
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    pid->output = pid->kp * error + pid->ki * pid->integral + pid->kd * derivative;
    return pid->output;
}

// 姿态稳定控制
void Attitude_Control(HoverControl_t *ctrl, float roll, float pitch, float yaw_rate) {

### 3.2 推进与转向控制

```python
class HovercraftController:
    def __init__(self):
        self.ser = serial.Serial('/dev/ttyS5', 115200)
        
        # 高度PID
        self.height_pid = PIDController(kp=2.0, ki=0.1, kd=0.5)
        # 方向PID
        self.direction_pid = PIDController(kp=0.8, ki=0.01, kd=0.3)
        
        self.target_height = 10  # cm
        self.base_thrust = 120
        
    def compute_control(self, line_offset, obstacles, height):
        """计算控制输出"""
        # 高度控制
        height_output = self.height_pid.compute(self.target_height, height)
        lift = int(self.base_thrust + height_output)
        
        # 方向控制
        steering = self.direction_pid.compute(0, line_offset)
        
        # 避障逻辑
        for obs in obstacles:
            if obs['distance'] < 1.5:  # 危险距离
                if obs['region'] == 'left':
                    steering += 50  # 右转避障
                elif obs['region'] == 'right':
                    steering -= 50  # 左转避障
                elif obs['region'] == 'center':
                    # 减速并选择方向
                    if line_offset < 0:
                        steering += 80
                    else:
                        steering -= 80
        
        # 计算推进
        left_thrust = int(self.base_thrust + steering)
        right_thrust = int(self.base_thrust - steering)
        
        return {
            'lift': np.clip(lift, 80, 255),
            'left_thrust': np.clip(left_thrust, 0, 255),
            'right_thrust': np.clip(right_thrust, 0, 255)
        }
        
    def send_command(self, control):
        """发送控制命令"""
        cmd = bytes([
            0xAA,
            control['lift'],
            control['left_thrust'],
            control['right_thrust'],
            0x55
        ])
        self.ser.write(cmd)
```

## 五、完整主程序

```python
#!/usr/bin/env python3
"""
2023年I题 - 气垫悬浮车 主程序
"""
import cv2
import numpy as np
import serial
import time

class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.integral = 0
        self.prev_error = 0
        
    def compute(self, target, current):
        error = target - current
        self.integral += error
        self.integral = np.clip(self.integral, -100, 100)
        derivative = error - self.prev_error
        self.prev_error = error
        return self.kp * error + self.ki * self.integral + self.kd * derivative

class HovercraftSystem:
    def __init__(self):
        self.vision = HovercraftVision()
        self.controller = HovercraftController()
        
        self.ser = serial.Serial('/dev/ttyS5', 115200, timeout=0.01)
        
        self.state = 'HOVER'  # HOVER, NAVIGATE, AVOID
        self.height = 10
        self.running = True
        
    def read_sensors(self):
        """读取传感器数据"""
        if self.ser.in_waiting >= 10:
            data = self.ser.read(10)
            if data[0] == 0xBB:
                # 解析高度、姿态等
                self.height = data[1] / 10.0  # cm
                return True
        return False
        
    def run(self):
        print("气垫悬浮车启动...")
        
        # 初始化：先升力到悬浮状态
        print("正在建立气垫...")
        self.controller.send_command({'lift': 180, 'left_thrust': 0, 'right_thrust': 0})
        time.sleep(3)
        
        while self.running:
            # 读取传感器
            self.read_sensors()
            
            # 视觉检测
            line_detected, line_offset = self.vision.detect_line()
            obstacles = self.vision.detect_obstacles()
            left_bound, right_bound = self.vision.detect_boundaries()
            
            if line_detected:
                # 计算控制量
                control = self.controller.compute_control(
                    line_offset, obstacles, self.height
                )
                
                self.controller.send_command(control)
                
                print(f"高度:{self.height:.1f}cm 偏移:{line_offset:.0f} "
                      f"升力:{control['lift']} 推进:{control['left_thrust']}/{control['right_thrust']}")
                
                # 避障警告
                for obs in obstacles:
                    if obs['distance'] < 2:
                        print(f"⚠️ 障碍物警告: {obs['color']} {obs['region']} "
                              f"距离:{obs['distance']:.1f}m")
            else:
                # 丢失目标，悬停
                self.controller.send_command({
                    'lift': 150, 'left_thrust': 0, 'right_thrust': 0
                })
                print("丢失目标，悬停等待...")
                
            time.sleep(0.02)  # 50Hz

if __name__ == '__main__':
    hovercraft = HovercraftSystem()
    try:
        hovercraft.run()
    except KeyboardInterrupt:
        hovercraft.controller.send_command({'lift': 80, 'left_thrust': 0, 'right_thrust': 0})
        print("气垫车停止")
```

## 七、调试要点

1. **气垫密封**：裙边与地面接触良好，漏气量最小
2. **重心平衡**：电池和设备位置对称，避免偏移
3. **升力校准**：4个电机升力一致，避免翻转
4. **PID顺序**：先高度稳定，再姿态，最后位置控制
5. **避障响应**：提前检测，预留足够反应距离
