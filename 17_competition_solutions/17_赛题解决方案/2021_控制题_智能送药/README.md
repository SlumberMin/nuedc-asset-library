# 2021年全国大学生电子设计竞赛 - 控制题（F题）智能送药小车 完整解决方案

## 一、题目分析

### 1.1 题目要求概述

设计智能送药小车系统，要求：
- **病房识别**：识别病房号（数字标识）
- **路径规划**：沿红线自动寻径到达目标病房
- **送药操作**：到达病房后执行送药动作
- **自动返回**：送药完成后自动返回药房
- **双车协同（发挥部分）**：两车协同送药

### 1.2 核心技术指标

| 指标 | 基本要求 | 发挥部分 |
|------|---------|---------|
| 数字识别率 | ≥90% | ≥98% |
| 循线精度 | ±2cm | ±1cm |
| 转弯精度 | ±10° | ±5° |
| 送药准确率 | 100% | 100% |
| 任务完成时间 | <5min | <3min |
| 协同送药 | 无 | 双车无冲突 |

### 1.3 难点分析

1. **数字识别鲁棒性**：不同角度、光照下的识别
2. **红线循线精度**：红色在图像中较难提取
3. **路径规划效率**：多病房时的最优路径
4. **双车协同调度**：避免路径冲突

## 五、核心算法说明

### 5.1 红线循线算法（OpenMV）

```python
# OpenMV红线循线代码
import sensor, image, time

# 红色LAB阈值
RED_THRESHOLD = (20, 60, 30, 80, 10, 70)

sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)
sensor.skip_frames(time=2000)

while True:
    img = sensor.snapshot()
    
    # 提取红色线条
    blobs = img.find_blobs([RED_THRESHOLD], pixels_threshold=200, 
                           area_threshold=200, merge=True)
    
    if blobs:
        largest = max(blobs, key=lambda b: b.pixels())
        # 计算中心偏移
        error = largest.cx() - 160  # 图像中心
        print("偏移:", error)
        
        # 发送给STM32
        uart.write("$L,%d\n" % error)
```

### 5.2 数字识别算法（OpenMV）

```python
# 数字模板匹配
import image

# 加载模板
templates = {}
for i in range(1, 9):  # 病房号1-8
    templates[i] = image.Template(image.Image("/templates/%d.pgm" % i))

def recognize_number(img):
    best_match = 0
    best_score = 0
    
    for num, template in templates.items():
        result = img.find_template(template, 0.7, step=4, search=image.SEARCH_EX)
        if result and result[4] > best_score:
            best_score = result[4]
            best_match = num
    
    return best_match if best_score > 0.8 else 0
```

### 5.3 A*路径规划算法

> ⚠️ 以下A*算法为教学伪代码，辅助函数(is_valid/is_in_closed/heuristic等)需根据实际地图实现。

```c
#define MAP_SIZE 10
#define MAX_NODES (MAP_SIZE * MAP_SIZE)

typedef struct {
    int x, y;
} Point;

/* 修复: Node结构体添加in_open/in_closed标志，避免O(n)查找 */
typedef struct {
    Point pos;
    int g, h, f;
    Point parent;
    uint8_t in_open;
    uint8_t in_closed;
} Node;

/* 曼哈顿距离启发函数 */
static int heuristic(Point a, Point b) {
    return abs(a.x - b.x) + abs(a.y - b.y);
}

/* 地图障碍物标记 (需根据实际赛道配置) */
static uint8_t obstacle_map[MAP_SIZE][MAP_SIZE];

static int is_valid(Point p) {
    if (p.x < 0 || p.x >= MAP_SIZE || p.y < 0 || p.y >= MAP_SIZE) return 0;
    return !obstacle_map[p.x][p.y];
}

/* 修复: 使用Node网格数组替代线性搜索open/closed list */
static Node grid[MAX_NODES];

int astar(Point start, Point end, Point *path, int max_path_len) {
    /* 清空网格 */
    for (int i = 0; i < MAX_NODES; i++) {
        grid[i].in_open = 0;
        grid[i].in_closed = 0;
        grid[i].g = 9999;
    }

    int sx = start.x, sy = start.y;
    grid[sy * MAP_SIZE + sx] = (Node){start, 0, heuristic(start, end), 0, {-1, -1}, 1, 0};

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};
    int path_len = 0;

    while (1) {
        /* 找open list中f值最小的节点 */
        int min_f = 9999, min_idx = -1;
        for (int i = 0; i < MAX_NODES; i++) {
            if (grid[i].in_open && !grid[i].in_closed && grid[i].f < min_f) {
                min_f = grid[i].f;
                min_idx = i;
            }
        }
        if (min_idx < 0) return -1;  /* 无路径 */

        Node *cur = &grid[min_idx];
        cur->in_open = 0;
        cur->in_closed = 1;

        /* 到达终点 */
        if (cur->pos.x == end.x && cur->pos.y == end.y) {
            /* 回溯路径 */
            Point tmp[MAX_NODES];
            int n = 0;
            Node *p = cur;
            while (p->parent.x >= 0) {
                tmp[n++] = p->pos;
                p = &grid[p->parent.y * MAP_SIZE + p->parent.x];
            }
            tmp[n++] = start;
            /* 反转 */
            for (int i = 0; i < n && i < max_path_len; i++)
                path[i] = tmp[n - 1 - i];
            return n;
        }

        /* 扩展邻居节点 */
        for (int d = 0; d < 4; d++) {
            Point nb = {cur->pos.x + dx[d], cur->pos.y + dy[d]};
            if (!is_valid(nb)) continue;

            int ni = nb.y * MAP_SIZE + nb.x;
            int g = cur->g + 1;
            if (g < grid[ni].g) {
                grid[ni].pos = nb;
                grid[ni].g = g;
                grid[ni].h = heuristic(nb, end);
                grid[ni].f = g + grid[ni].h;
                grid[ni].parent = cur->pos;
                grid[ni].in_open = 1;
            }
        }
    }
}
```
