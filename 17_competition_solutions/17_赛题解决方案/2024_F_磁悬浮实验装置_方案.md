# 2024年F题：磁悬浮实验装置 — Orange Pi 5 专属方案

## 题目概述
设计磁悬浮实验装置，通过电磁铁控制悬浮物体的稳定悬浮高度，要求能够在设定高度稳定悬浮，并具备抗干扰能力。

## 系统架构

```
[OV5647摄像头(俯视)] → [MIPI-CSI] → Orange Pi 5 → [DAC/电机驱动] → [电磁铁线圈]
                                      ↓
[霍尔传感器] ────→ [ADC(ADS1115)] ─→ [高度反馈+视觉融合]
[红外测距] ─────→ [ADC(ADS1115)] ─↗
```

## 核心方案

### 1. 悬浮高度检测（视觉方案）
- **摄像头俯视**拍摄悬浮物体
- **颜色/轮廓检测**：悬浮磁球为特定颜色（如银色金属球）
- **像素面积法**测距：球的像素面积与高度成反比关系
- **标定公式**：`H = k / sqrt(pixel_area) + b`

### 2. 辅助传感器
- **霍尔传感器**：检测磁场强度，间接反映高度
- **红外测距**：GP2Y0A21，直接测量距离
- **传感器融合**：加权平均提升可靠性

### 3. 电磁控制
- **PID闭环**：设定高度 - 实际高度 → PID → PWM占空比 → 电磁铁电流
- **难点**：磁悬浮系统本质不稳定，需要高频率控制(>100Hz)
- **前馈补偿**：抵消重力的基础电流 + PID微调

## Orange Pi 5 代码实现

```python
#!/usr/bin/env python3
"""2024F 磁悬浮实验装置 - Orange Pi 5"""

import numpy as np
import cv2
import time
from periphery import GPIO, PWM

# ============ ADS1115 ADC ============
class ADS1115Reader:
    def __init__(self, bus=1, addr=0x48):
        import smbus2
        self.bus = smbus2.SMBus(bus)
        self.addr = addr
        self.gain = 1  # ±4.096V

    def read_channel(self, ch):
        """读取指定通道(0-3)"""
        # 配置寄存器
        config = 0x8000  # OS=1(单次转换)
        config |= (4 + ch) << 12  # MUX: 单端输入
        config |= self.gain << 9   # PGA
        config |= 0x0100           # 单次模式
        config |= 0x0080           # 128SPS

        self.bus.write_i2c_block_data(self.addr, 0x01,
                                      [(config >> 8) & 0xFF, config & 0xFF])
        time.sleep(0.01)

        data = self.bus.read_i2c_block_data(self.addr, 0x00, 2)
        raw = (data[0] << 8) | data[1]
        if raw > 32767:
            raw -= 65536
        # 转换为电压
        voltage = raw * 4.096 / 32767
        return voltage


# ============ 视觉高度检测 ============
class VisionHeightSensor:
    def __init__(self):
        # 标定参数
        self.k = 500.0   # 标定系数
        self.b = 0.0     # 偏移
        # 颜色范围 (HSV) - 银色金属球
        self.lower_color = np.array([0, 0, 180])
        self.upper_color = np.array([180, 30, 255])

    def calibrate(self, known_heights, pixel_areas):
        """标定: 已知高度和对应像素面积"""
        # H = k / sqrt(A) + b
        inv_sqrt_a = 1.0 / np.sqrt(np.array(pixel_areas))
        coeffs = np.polyfit(inv_sqrt_a, known_heights, 1)
        self.k = coeffs[0]
        self.b = coeffs[1]
        print(f"标定完成: k={self.k:.2f}, b={self.b:.2f}")

    def detect_height(self, frame):
        """视觉检测悬浮物体高度"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.lower_color, self.upper_color)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((5,5), np.uint8))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, np.ones((5,5), np.uint8))

        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        if not contours:
            return None, None

        # 找最大轮廓
        c = max(contours, key=cv2.contourArea)
        area = cv2.contourArea(c)
        if area < 50:
            return None, None

        # 计算高度
        height = self.k / np.sqrt(area) + self.b
        M = cv2.moments(c)
        if M["m00"] > 0:
            cx = int(M["m10"] / M["m00"])
            cy = int(M["m01"] / M["m00"])
        else:
            cx, cy = 0, 0

        return height, (cx, cy, area)


# ============ 霍尔传感器高度检测 ============
class HallHeightSensor:
    def __init__(self, adc: ADS1115Reader, channel=0):
        self.adc = adc
        self.ch = channel
        # 标定参数: 磁场强度与高度关系
        self.coeffs = [1.0, -2.0, 5.0]  # 二次多项式

    def read_height(self):
        voltage = self.adc.read_channel(self.ch)
        # 电压 → 高度映射
        height = np.polyval(self.coeffs, voltage)
        return max(0, height)

    def calibrate(self, heights, voltages):
        self.coeffs = np.polyfit(voltages, heights, 2)


# ============ 红外测距传感器 ============
class IRDistanceSensor:
    def __init__(self, adc: ADS1115Reader, channel=1):
        self.adc = adc
        self.ch = channel

    def read_distance(self):
        """GP2Y0A21: 电压转距离(cm)"""
        voltage = self.adc.read_channel(self.ch)
        if voltage < 0.3:
            return 80.0  # 超出范围
        distance = 27.86 / (voltage - 0.1)  # 近似公式
        return np.clip(distance, 10, 80)


# ============ 电磁铁驱动 ============
class ElectromagnetDriver:
    def __init__(self):
        # PWM驱动电磁铁 (通过MOSFET)
        self.pwm = PWM(0, 0)
        self.pwm.frequency = 20000  # 20kHz PWM
        self.pwm.enable = True
        self.base_current = 0.5  # 基础电流(抵消重力)

    def set_power(self, duty):
        """设置电磁铁功率 0.0~1.0"""
        self.pwm.duty_cycle = np.clip(duty, 0, 1.0)

    def get_base_power(self):
        return self.base_current


# ============ 高级PID控制器 ============
class AntiWindupPID:
    def __init__(self, kp, ki, kd, output_min=0, output_max=1.0):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.out_min = output_min
        self.out_max = output_max
        self.integral = 0.0
        self.prev_error = 0.0
        self.prev_output = 0.0

    def compute(self, error, dt):
        P = self.kp * error
        self.integral += error * dt
        # Anti-windup
        I = self.ki * self.integral
        D = self.kd * (error - self.prev_error) / max(dt, 0.0001)
        self.prev_error = error

        output = P + I + D
        output = np.clip(output, self.out_min, self.out_max)

        # Anti-windup: 如果饱和则回退积分
        if output >= self.out_max or output <= self.out_min:
            self.integral -= error * dt

        self.prev_output = output
        return output


# ============ 传感器融合 ============
class SensorFusion:
    def __init__(self):
        self.weights = {"vision": 0.5, "hall": 0.3, "ir": 0.2}

    def fuse(self, vision_h, hall_h, ir_h):
        """加权平均融合"""
        values = []
        weights = []
        if vision_h is not None:
            values.append(vision_h)
            weights.append(self.weights["vision"])
        if hall_h is not None:
            values.append(hall_h)
            weights.append(self.weights["hall"])
        if ir_h is not None:
            values.append(ir_h)
            weights.append(self.weights["ir"])

        if not values:
            return None
        weights = np.array(weights) / sum(weights)
        return np.average(values, weights=weights)


# ============ 主程序 ============
def main():
    # 初始化
    adc = ADS1115Reader(bus=1, addr=0x48)
    vision = VisionHeightSensor()
    hall = HallHeightSensor(adc, channel=0)
    ir_sensor = IRDistanceSensor(adc, channel=1)
    em_driver = ElectromagnetDriver()
    fusion = SensorFusion()

    # PID参数 - 磁悬浮需要精细调参
    pid = AntiWindupPID(
        kp=0.15,
        ki=0.5,
        kd=0.02,
        output_min=0.0,
        output_max=1.0
    )

    target_height = 3.0  # 目标悬浮高度(cm)
    cap = cv2.VideoCapture(0)
    cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
    cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    # 标定阶段(可选)
    # vision.calibrate(known_heights=[2, 3, 4, 5], pixel_areas=[8000, 4500, 3000, 2000])

    print("磁悬浮实验装置启动...")
    print(f"目标高度: {target_height} cm")
    last_time = time.time()
    control_dt = 0.01  # 100Hz控制周期

    while True:
        now = time.time()
        dt = now - last_time
        if dt < control_dt:
            time.sleep(0.001)
            continue
        last_time = now

        # 读取视觉高度
        ret, frame = cap.read()
        vision_h = None
        if ret:
            vision_h, blob_info = vision.detect_height(frame)

        # 读取霍尔传感器
        hall_h = hall.read_height()

        # 读取红外测距
        ir_h = ir_sensor.read_distance() / 100.0  # cm → 对应单位

        # 传感器融合
        measured_h = fusion.fuse(vision_h, hall_h, ir_h)

        if measured_h is not None:
            # PID控制
            error = target_height - measured_h
            pid_output = pid.compute(error, dt)

            # 前馈 + PID
            total_power = em_driver.get_base_power() + pid_output
            em_driver.set_power(total_power)

            # 显示
            if ret:
                if blob_info:
                    cx, cy, area = blob_info
                    cv2.circle(frame, (cx, cy), 10, (0, 0, 255), 2)
                cv2.putText(frame, f"Target: {target_height:.1f}cm",
                            (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
                cv2.putText(frame, f"Measured: {measured_h:.2f}cm",
                            (10, 60), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
                cv2.putText(frame, f"Power: {total_power:.2f}",
                            (10, 90), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
                cv2.putText(frame, f"V:{vision_h:.2f} H:{hall_h:.2f} IR:{ir_h:.2f}" if vision_h else "V:N/A",
                            (10, 120), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (200, 200, 0), 1)
                cv2.imshow("Maglev", frame)

        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    em_driver.set_power(0)
    cap.release()
    cv2.destroyAllWindows()

if __name__ == "__main__":
    main()
```

## 硬件接线

| 模块 | Orange Pi 5引脚 | 说明 |
|------|-----------------|------|
| OV5647 | MIPI-CSI | 俯视摄像头 |
| ADS1115 SDA | GPIO1_B0 (I2C3_SDA) | ADC数据 |
| ADS1115 SCL | GPIO1_B1 (I2C3_SCL) | ADC时钟 |
| 电磁铁MOSFET | PWM0 (GPIO1_B4) | 20kHz PWM驱动 |
| 霍尔传感器 | ADS1115 CH0 | 磁场检测 |
| IR测距GP2Y0A21 | ADS1115 CH1 | 红外测距 |
| 限位保护 | GPIO1_C2 | 下限位开关 |

## 控制参数

| 参数 | 值 | 说明 |
|------|-----|------|
| 控制频率 | 100Hz | 高频保持稳定 |
| PID Kp | 0.15 | 比例系数 |
| PID Ki | 0.5 | 积分系数(抗重力) |
| PID Kd | 0.02 | 微分系数(阻尼) |
| PWM频率 | 20kHz | 避免电磁铁噪音 |
| 目标高度 | 3cm | 默认值 |
| 高度精度 | ±0.5mm | 视觉+融合 |

## 系统调试要点

1. **先调基础电流**：找到刚好平衡重力的电流值，作为前馈
2. **PID调参顺序**：先P到临界振荡 → 加D抑制振荡 → 加I消除稳态误差
3. **Anti-windup必须**：磁悬浮积分饱和会导致飞走
4. **视觉检测需要稳定光源**：避免环境光变化影响
5. **安全保护**：限位开关断电保护，防止物体掉落或撞击电磁铁
