# 2019年A题 电动小车动态无线充电系统 — Orange Pi 5解决方案

## 一、题目分析

### 1.1 核心任务
设计一个电动小车动态无线充电系统。小车在圆形跑道上运行，跑道下方铺设发射线圈，小车底部安装接收线圈，实现**边跑边充电**，同时小车需自主保持在跑道上行驶。

### 1.2 关键指标
- 跑道直径约50~70cm，小车宽度约15~20cm
- 发射线圈沿跑道铺设，接收线圈安装在小车底部
- 充电效率要求：接收功率/发射功率
- 小车需自主循迹，不得跑出跑道
- 连续运行时间越长越好

### 1.3 技术难点
1. **动态耦合**：小车运动时接收线圈与发射线圈的相对位置不断变化，耦合系数波动大
2. **视觉循迹**：需实时识别跑道边界，保持小车在跑道中央
3. **能量管理**：充电功率有限，需平衡充电速率与消耗速率
4. **效率优化**：需实时调整发射/接收电路参数以维持高效率

## 四、视觉循迹算法

### 4.1 跑道检测流程
```python
import cv2
import numpy as np

class TrackFollower:
    def __init__(self):
        # 跑道颜色范围（黑色跑道在白色背景上）
        self.lower_bound = np.array([0, 0, 0])
        self.upper_bound = np.array([180, 255, 50])
        self.pid = PIDController(Kp=0.8, Ki=0.01, Kd=0.3)
    
    def detect_track(self, frame):
        """检测跑道边界"""
        # 转换到HSV色彩空间
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 提取跑道区域（深色跑道）
        mask = cv2.inRange(hsv, self.lower_bound, self.upper_bound)
        
        # 形态学操作去噪
        kernel = np.ones((5, 5), np.uint8)
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)
        mask = cv2.morphologyEx(mask, cv2.MORPH_OPEN, kernel)
        
        return mask
    
    def get_track_center(self, mask, row_ratio=0.7):
        """在指定行提取跑道中心线"""
        h, w = mask.shape
        row = int(h * row_ratio)
        
        # 提取该行的跑道像素
        line = mask[row, :]
        
        # 找到跑道边界
        track_pixels = np.where(line > 0)[0]
        
        if len(track_pixels) < 10:
            return None  # 未检测到跑道
        
        # 计算跑道中心
        left_edge = track_pixels[0]
        right_edge = track_pixels[-1]
        center = (left_edge + right_edge) // 2
        
        # 计算偏移量（相对于图像中心）
        offset = center - w // 2
        
        return offset, left_edge, right_edge
    
    def compute_steering(self, offset):
        """PID计算转向量"""
        if offset is None:
            return 0  # 丢失跑道，停车
        
        # PID控制
        steering = self.pid.update(offset)
        
        # 限幅
        steering = max(-1.0, min(1.0, steering / 100.0))
        
        return steering

class PIDController:
    def __init__(self, Kp, Ki, Kd):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.prev_error = 0
        self.integral = 0
    
    def update(self, error):
        self.integral += error
        self.integral = max(-1000, min(1000, self.integral))  # 积分限幅
        derivative = error - self.prev_error
        self.prev_error = error
        
        return self.Kp * error + self.Ki * self.integral + self.Kd * derivative
```

### 4.2 弯道预判
```python
def predict_curve(self, mask, rows=[0.5, 0.6, 0.7, 0.8]):
    """通过多行分析预判弯道"""
    centers = []
    for ratio in rows:
        h, w = mask.shape
        row = int(h * ratio)
        line = mask[row, :]
        pixels = np.where(line > 0)[0]
        if len(pixels) > 10:
            center = (pixels[0] + pixels[-1]) // 2
            centers.append(center)
    
    if len(centers) < 2:
        return 0  # 无法预判
    
    # 计算曲率（中心点偏移趋势）
    curvature = (centers[-1] - centers[0]) / len(centers)
    
    # 提前减速
    speed_factor = max(0.4, 1.0 - abs(curvature) / 200.0)
    
    return curvature, speed_factor
```

## 五、无线充电控制

### 5.1 谐振频率跟踪
```python
class WirelessChargingController:
    def __init__(self):
        self.target_voltage = 8.4V  # 充电目标电压
        self.max_power = 5W
        self.frequency = 150e3  # 初始频率150kHz
    
    def adjust_frequency(self, voltage, current):
        """根据反馈调整工作频率，维持谐振"""
        # 计算当前阻抗
        impedance = voltage / max(current, 0.001)
        
        # 阻抗最小点即为谐振点
        if impedance < self.prev_impedance:
            self.frequency += 1e3  # 向阻抗减小方向调频
        else:
            self.frequency -= 1e3
        
        self.prev_impedance = impedance
        
        # 限幅
        self.frequency = max(100e3, min(200e3, self.frequency))
        
        return self.frequency
    
    def calculate_efficiency(self, input_power, output_power):
        """计算传输效率"""
        if input_power <= 0:
            return 0
        return output_power / input_power * 100
```

### 5.2 能量平衡管理
```python
class EnergyManager:
    def __init__(self, battery_capacity_mah=2000):
        self.battery_capacity = battery_capacity_mah
        self.battery_level = battery_capacity_mah  # 初始满电
        self.charging_power = 0  # 当前充电功率(mW)
        self.consumption_power = 0  # 当前消耗功率(mW)
    
    def update(self, charging_mW, consumption_mW, dt):
        """更新电池状态"""
        self.charging_power = charging_mW
        self.consumption_power = consumption_mW
        
        # 净功率
        net_power = charging_mW - consumption_mW
        
        # 更新电量 (mAh = mW * dt(h) / V)
        # 假设平均电压7.4V
        energy_change_mWh = net_power * dt / 3600
        self.battery_level += energy_change_mWh / 7.4
        
        # 限幅
        self.battery_level = max(0, min(self.battery_capacity, self.battery_level))
        
        return self.get_battery_percent()
    
    def get_battery_percent(self):
        return self.battery_level / self.battery_capacity * 100
    
    def should_reduce_speed(self):
        """电量低时降低速度以减少功耗"""
        percent = self.get_battery_percent()
        if percent < 20:
            return 0.5  # 降低50%速度
        elif percent < 40:
            return 0.7  # 降低30%速度
        return 1.0
```

import time
import serial

class DynamicChargingCar:
    def __init__(self):
        # 初始化模块
        self.camera = CameraCapture(width=640, height=480, fps=30)
        self.tracker = TrackFollower()
        self.charger = WirelessChargingController()
        self.energy = EnergyManager()
        self.motor = MotorController(serial_port='/dev/ttyS3', baudrate=115200)
        
        # 状态
        self.running = True
        self.base_speed = 200  # 基础PWM值(0~1000)
    
    def run(self):
        """主循环"""
        print("动态无线充电小车启动...")
        
        while self.running:
            start_time = time.time()
            
            # 1. 获取图像
            frame = self.camera.capture()
            
            # 2. 跑道检测
            mask = self.tracker.detect_track(frame)
            result = self.tracker.get_track_center(mask)
            
            # 3. 弯道预判
            curve_info = self.tracker.predict_curve(mask)
            
            # 4. 计算转向
            if result is not None:
                offset, left, right = result
                steering = self.tracker.compute_steering(offset)
                
                # 弯道减速
                speed_factor = curve_info[1] if curve_info != 0 else 1.0
                
                # 电量低减速
                energy_factor = self.energy.should_reduce_speed()
                
                # 最终速度
                speed = int(self.base_speed * speed_factor * energy_factor)
                
                # 5. 差速控制
                left_speed = int(speed * (1 + steering))
                right_speed = int(speed * (1 - steering))
                
                # 限幅
                left_speed = max(0, min(1000, left_speed))
                right_speed = max(0, min(1000, right_speed))
                
                self.motor.set_speed(left_speed, right_speed)
            else:
                # 丢失跑道，停车
                self.motor.stop()
                print("警告：丢失跑道！")
            
            # 6. 更新能量管理
            dt = time.time() - start_time
            # 模拟数据，实际从ADC读取
            self.energy.update(
                charging_mW=self.charger.get_current_power(),
                consumption_mW=self.estimate_consumption(),
                dt=dt
            )
            
            # 7. 控制帧率
            elapsed = time.time() - start_time
            if elapsed < 0.033:  # 30fps
                time.sleep(0.033 - elapsed)
    
    def estimate_consumption(self):
        """估算当前功耗"""
        # OPi5: ~3W, 电机: ~2W, 其他: ~0.5W
        return 5500  # mW

if __name__ == '__main__':
    car = DynamicChargingCar()
    car.run()

### 7.1 Orange Pi 5 ↔ STM32 通信
```python
# UART协议定义
# 帧格式: [HEAD, CMD, DATA0, DATA1, CHECK, TAIL]
# HEAD: 0xAA
# CMD: 命令字
# DATA: 数据
# CHECK: XOR校验
# TAIL: 0x55

CMD_MOTOR_SPEED = 0x01    # 设置电机速度
CMD_SERVO_ANGLE = 0x02    # 设置舵机角度
CMD_CHARGE_CTRL = 0x03    # 充电控制
CMD_GET_BATTERY = 0x10    # 获取电池电压
CMD_GET_CURRENT = 0x11    # 获取充电电流

class STM32Comm:
    def __init__(self, port='/dev/ttyS3', baudrate=115200):
        self.ser = serial.Serial(port, baudrate, timeout=0.1)
    
    def send_command(self, cmd, data):
        """发送命令到STM32"""
        frame = bytes([0xAA, cmd, data[0], data[1], 
                       0xAA ^ cmd ^ data[0] ^ data[1], 0x55])
        self.ser.write(frame)
    
    def set_motor_speed(self, left, right):
        """设置左右电机速度"""
        self.send_command(CMD_MOTOR_SPEED, 
                         [left >> 8 & 0xFF, left & 0xFF])
    
    def get_battery_voltage(self):
        """获取电池电压"""
        self.send_command(CMD_GET_BATTERY, [0, 0])
        response = self.ser.read(6)
        if len(response) == 6 and response[0] == 0xAA:
            voltage = (response[2] << 8) | response[3]
            return voltage / 1000.0  # mV转V
        return None
```

### 8.2 PID参数整定
| 参数 | 建议范围 | 说明 |
|------|---------|------|
| Kp | 0.5~1.5 | 比例系数，影响响应速度 |
| Ki | 0.001~0.05 | 积分系数，消除稳态误差 |
| Kd | 0.1~0.5 | 微分系数，减少超调 |

### 8.3 性能优化策略
1. **图像ROI裁剪**：只处理跑道区域，减少计算量
2. **降低分辨率**：使用320×240处理，提高帧率
3. **NPU加速**：将语义分割模型部署到NPU
4. **多线程**：图像采集、处理、通信分线程
5. **预测控制**：根据曲率提前调整速度和转向
