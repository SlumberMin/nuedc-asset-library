# 2017年全国大学生电子设计竞赛 - 滚球控制系统

## 题目分析
设计一个通过摄像头检测小球在平板上的位置，利用四个舵机调节平板倾斜角度，使小球能够按照指定轨迹运动或停留在指定位置的控制系统。

## 系统架构

```
┌─────────────┐    CSI/USB    ┌──────────────────┐   UART/SPI   ┌──────────────┐
│  摄像头模块   │──────────────→│   Orange Pi 5    │──────────────→│  舵机驱动板   │
│ (OV5647/USB) │              │  RK3588S         │              │ (PCA9685)    │
└─────────────┘              │  视觉处理+PID     │              └──────┬───────┘
                              │  + 坐标变换       │                     │ PWM×4
                              └──────────────────┘              ┌──────┴───────┐
                                                                │  4×舵机      │
                                                                │  控制平台倾斜  │
                                                                └──────────────┘
```

#### 1. 小球检测模块 (detector.py)
```python
import cv2
import numpy as np
from dataclasses import dataclass

@dataclass
class BallState:
    x: float          # 像素坐标x
    y: float          # 像素坐标y
    radius: float     # 像素半径
    found: bool       # 是否检测到
    real_x: float = 0.0  # 实际坐标(mm)
    real_y: float = 0.0  # 实际坐标(mm)

class BallDetector:
    def __init__(self, config):
        # 平板在图像中的四个角点(标定获得)
        self.board_corners = np.array(config['board_corners'], dtype=np.float32)
        # 平板实际尺寸(mm)
        self.board_size = (config['board_width'], config['board_height'])
        # 小球颜色范围(HSV)
        self.lower_color = np.array(config['ball_hsv_lower'])
        self.upper_color = np.array(config['ball_hsv_upper'])
        # 计算透视变换矩阵
        dst = np.array([
            [0, 0],
            [self.board_size[0], 0],
            [self.board_size[0], self.board_size[1]],
            [0, self.board_size[1]]
        ], dtype=np.float32)
        self.M = cv2.getPerspectiveTransform(self.board_corners, dst)
        self.M_inv = cv2.getPerspectiveTransform(dst, self.board_corners)

    def detect(self, frame) -> BallState:
        """检测小球位置并转换为实际坐标"""
        # 方法1: 颜色+圆形检测
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        mask = cv2.inRange(hsv, self.lower_color, self.upper_color)
        mask = cv2.erode(mask, None, iterations=2)
        mask = cv2.dilate(mask, None, iterations=2)

        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if contours:
            # 找最大轮廓
            c = max(contours, key=cv2.contourArea)
            ((x, y), radius) = cv2.minEnclosingCircle(c)

            if radius > 5:  # 过滤噪声
                # 像素坐标转换为实际坐标
                pixel_pt = np.array([[[x, y]]], dtype=np.float32)
                real_pt = cv2.perspectiveTransform(pixel_pt, self.M)
                return BallState(
                    x=x, y=y, radius=radius, found=True,
                    real_x=real_pt[0][0][0],
                    real_y=real_pt[0][0][1]
                )

        # 方法2: 霍夫圆检测(备选)
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        gray = cv2.GaussianBlur(gray, (9, 9), 2)
        circles = cv2.HoughCircles(gray, cv2.HOUGH_GRADIENT, 1, 20,
                                    param1=50, param2=30, minRadius=5, maxRadius=50)
        if circles is not None:
            circles = np.round(circles[0, :]).astype("int")
            x, y, r = circles[0]
            pixel_pt = np.array([[[float(x), float(y)]]], dtype=np.float32)
            real_pt = cv2.perspectiveTransform(pixel_pt, self.M)
            return BallState(
                x=float(x), y=float(y), radius=float(r), found=True,
                real_x=real_pt[0][0][0],
                real_y=real_pt[0][0][1]
            )

        return BallState(x=0, y=0, radius=0, found=False)
```

#### 2. PID控制器 (pid_controller.py)
```python
import time
from dataclasses import dataclass

@dataclass
class PIDParams:
    kp: float
    ki: float
    kd: float
    output_min: float = -15.0  # 最大倾斜角度(度)
    output_max: float = 15.0
    integral_limit: float = 100.0

class PIDController:
    def __init__(self, params: PIDParams):
        self.params = params
        self.integral = 0.0
        self.prev_error = 0.0
        self.prev_time = time.time()

    def update(self, error: float) -> float:
        now = time.time()
        dt = now - self.prev_time
        if dt <= 0:
            dt = 0.001

        # 比例项
        p = self.params.kp * error

        # 积分项(带限幅)
        self.integral += error * dt
        self.integral = max(-self.params.integral_limit,
                          min(self.params.integral_limit, self.integral))
        i = self.params.ki * self.integral

        # 微分项
        d = self.params.kd * (error - self.prev_error) / dt

        # 输出限幅
        output = p + i + d
        output = max(self.params.output_min, min(self.params.output_max, output))

        self.prev_error = error
        self.prev_time = now

        return output

    def reset(self):
        self.integral = 0.0
        self.prev_error = 0.0
        self.prev_time = time.time()
```

#### 3. 平台运动学 (platform_kinematics.py)
```python
import numpy as np

class PlatformKinematics:
    """
    四舵机平板运动学模型
    平板布局(俯视):
        S1(front-left) ---- S2(front-right)
        |                        |
        |        平板中心         |
        |                        |
        S3(back-left)  ---- S4(back-right)

    舵机通过连杆连接平板四角
    控制输入: 绕X轴倾斜角α, 绕Y轴倾斜角β
    输出: 四个舵机的角度
    """
    def __init__(self, config):
        # 平板尺寸(mm)
        self.length = config['plate_length']  # Y方向
        self.width = config['plate_width']    # X方向
        # 连杆长度(mm)
        self.link_length = config['link_length']
        # 舵机安装半径(mm)
        self.servo_arm = config['servo_arm_length']
        # 舵机中位角度
        self.servo_center = config.get('servo_center', 90)
        # 零位标定偏移
        self.offsets = config.get('servo_offsets', [0, 0, 0, 0])

    def tilt_to_servo_angles(self, alpha_deg, beta_deg):
        """
        将平台倾斜角转换为四个舵机角度
        alpha_deg: 绕X轴倾斜(前后), 正值=前高后低
        beta_deg:  绕Y轴倾斜(左右), 正值=右高左低
        """
        alpha = np.radians(alpha_deg)
        beta = np.radians(beta_deg)

        # 计算各角点的高度偏移(mm)
        half_w = self.width / 2
        half_l = self.length / 2

        # 四角高度变化
        # h = beta * x + alpha * y (小角度近似)
        h1 = -beta * half_w + alpha * half_l   # S1: front-left
        h2 = beta * half_w + alpha * half_l    # S2: front-right
        h3 = -beta * half_w - alpha * half_l   # S3: back-left
        h4 = beta * half_w - alpha * half_l    # S4: back-right

        # 高度转换为舵机角度
        # 角度 = arctan(height / servo_arm)
        angles = []
        for i, h in enumerate([h1, h2, h3, h4]):
            angle = np.degrees(np.arctan2(h, self.servo_arm))
            servo_angle = self.servo_center + angle + self.offsets[i]
            servo_angle = max(0, min(180, servo_angle))
            angles.append(servo_angle)

        return angles

    def compute_gravity_compensation(self, ball_x, ball_y, ball_mass=0.01):
        """
        计算小球重力对平台的影响(前馈补偿)
        ball_x, ball_y: 小球位置(mm), 相对中心
        """
        # 小球重力产生的力矩
        torque_x = ball_mass * 9.8 * ball_y  # 绕X轴
        torque_y = ball_mass * 9.8 * ball_x  # 绕Y轴

        # 转换为角度补偿(简化模型)
        comp_alpha = torque_x * 0.1  # 系数需实验标定
        comp_beta = torque_y * 0.1

        return comp_alpha, comp_beta
```

#### 4. 主程序 (main.py)
```python
import cv2
import time
import yaml
from camera import Camera
from detector import BallDetector
from pid_controller import PIDController, PIDParams
from platform_kinematics import PlatformKinematics
from servo_driver import ServoDriver
from trajectory import TrajectoryPlanner

def main():
    # 加载配置
    with open('config.yaml', 'r') as f:
        config = yaml.safe_load(f)

    # 初始化模块
    camera = Camera(config['camera'])
    detector = BallDetector(config['detector'])

    pid_x = PIDController(PIDParams(**config['pid_x']))
    pid_y = PIDController(PIDParams(**config['pid_y']))

    platform = PlatformKinematics(config['platform'])
    servo = ServoDriver(config['servo'])

    trajectory = TrajectoryPlanner(config.get('trajectory', {}))

    # 运行模式
    mode = config.get('mode', 'hold')  # hold / track / circle
    target_x = config.get('target_x', 0.0)
    target_y = config.get('target_y', 0.0)

    print(f"滚球控制系统启动 - 模式: {mode}")

    try:
        while True:
            # 1. 采集图像
            frame = camera.capture()
            if frame is None:
                continue

            # 2. 检测小球
            ball = detector.detect(frame)

            if not ball.found:
                # 小球丢失, 保持当前状态或缓慢回平
                print("小球未检测到!")
                continue

            # 3. 计算目标位置
            if mode == 'hold':
                goal_x, goal_y = target_x, target_y
            elif mode == 'circle':
                goal_x, goal_y = trajectory.get_circle_target(time.time())
            elif mode == 'track':
                goal_x, goal_y = trajectory.get_custom_target(time.time())

            # 4. PID计算
            error_x = goal_x - ball.real_x
            error_y = goal_y - ball.real_y

            output_alpha = pid_y.update(error_y)  # 前后倾斜
            output_beta = pid_x.update(error_x)   # 左右倾斜

            # 5. 重力补偿(前馈)
            comp_a, comp_b = platform.compute_gravity_compensation(
                ball.real_x, ball.real_y)
            output_alpha += comp_a
            output_beta += comp_b

            # 6. 运动学转换
            servo_angles = platform.tilt_to_servo_angles(output_alpha, output_beta)

            # 7. 驱动舵机
            servo.set_angles(servo_angles)

            # 8. 调试显示
            if config.get('debug', False):
                cv2.putText(frame, f"Ball: ({ball.real_x:.1f}, {ball.real_y:.1f})",
                           (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
                cv2.putText(frame, f"Goal: ({goal_x:.1f}, {goal_y:.1f})",
                           (10, 60), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 255), 2)
                cv2.putText(frame, f"Err: ({error_x:.1f}, {error_y:.1f})",
                           (10, 90), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2)
                cv2.circle(frame, (int(ball.x), int(ball.y)), int(ball.radius), (0, 255, 0), 2)
                cv2.imshow('Ball Control', frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break

    finally:
        servo.cleanup()
        camera.release()
        cv2.destroyAllWindows()

if __name__ == '__main__':
    main()
```

#### 5. 舵机驱动 (servo_driver.py)
```python
import smbus2

class ServoDriver:
    """通过PCA9685驱动4路舵机"""
    PCA9685_ADDR = 0x40
    MODE1 = 0x00
    PRESCALE = 0xFE
    LED0_ON_L = 0x06

    def __init__(self, config):
        self.bus = smbus2.SMBus(config.get('i2c_bus', 3))
        self.channels = config.get('channels', [0, 1, 2, 3])
        self._init_pca9685()

    def _init_pca9685(self):
        self.bus.write_byte_data(self.PCA9685_ADDR, self.MODE1, 0x10)  # Sleep
        prescale = int(25000000.0 / (4096 * 50) - 1)  # 50Hz
        self.bus.write_byte_data(self.PCA9685_ADDR, self.PRESCALE, prescale)
        self.bus.write_byte_data(self.PCA9685_ADDR, self.MODE1, 0x20)  # Auto-increment
        self.bus.write_byte_data(self.PCA9685_ADDR, self.MODE1 | 0x08, 0x20)  # Restart

    def set_angle(self, channel, angle):
        """设置单个舵机角度(0-180度)"""
        pulse = int(500 + angle / 180.0 * 2000)  # 500-2500us
        tick = int(pulse * 4096 / 20000)
        on = 0
        off = tick
        reg = self.LED0_ON_L + 4 * channel
        self.bus.write_word_data(self.PCA9685_ADDR, reg, on)
        self.bus.write_word_data(self.PCA9685_ADDR, reg + 2, off)

    def set_angles(self, angles):
        """批量设置4个舵机角度"""
        for ch, angle in zip(self.channels, angles):
            self.set_angle(ch, angle)

    def cleanup(self):
        """所有舵机回中"""
        self.set_angles([90, 90, 90, 90])
```

## PID调参指南

1. 先调P: 从小值开始增大, 直到小球能快速响应但不剧烈振荡
2. 再调D: 增大D抑制超调和振荡
3. 最后调I: 消除稳态误差, 注意积分饱和

推荐初始值:
- 响应型: Kp=1.0, Ki=0.05, Kd=0.4
- 稳定型: Kp=0.6, Ki=0.03, Kd=0.5

## 性能指标

| 指标 | 目标值 | 实测值 |
|------|--------|--------|
| 帧率 | ≥30fps | ~45fps |
| 检测延迟 | <10ms | ~5ms |
| 控制周期 | <30ms | ~20ms |
| 定位精度 | ±2mm | ±1.5mm |

## 注意事项

1. **摄像头安装**: 俯视安装, 避免透视畸变, 广角镜头需畸变校正
2. **光照条件**: 保持均匀光照, 避免阳光直射, 可加遮光罩
3. **舵机供电**: 独立5V供电, 加大电容滤波, 避免与主控共电源
4. **控制频率**: 保证≥30Hz, 低于20Hz系统不稳定
5. **小球材质**: 推荐钢球(直径15-20mm), 视觉检测一致性好
