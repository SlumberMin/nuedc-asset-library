# 2015年电赛C题 - 多旋翼自主飞行器 - Orange Pi 5解决方案

## 一、题目分析

**核心任务**：多旋翼飞行器实现自主起降、定点悬停、航点飞行、视觉定位，完成指定飞行任务（绕杆、穿环、定点降落等）。

**关键指标**：
- 起降平稳性
- 定点悬停精度
- 航点飞行准确性
- 视觉辅助定位能力

### 4.1 视觉定位模块

```python
# visual_localization.py
import cv2
import numpy as np
from rknnlite.api import RKNNLite

class VisualLocalizer:
    """视觉定位系统"""
    def __init__(self, model_path='marker_det.rknn'):
        # ArUco检测器
        self.aruco_dict = cv2.aruco.getPredefinedDictionary(cv2.aruco.DICT_4X4_50)
        self.aruco_params = cv2.aruco.DetectorParameters()
        self.detector = cv2.aruco.ArucoDetector(self.aruco_dict, self.aruco_params)
        
        # 标记物理尺寸
        self.marker_size = 0.15  # 15cm
        
        # 摄像头内参（标定后填入）
        self.camera_matrix = np.array([
            [500, 0, 320],
            [0, 500, 240],
            [0, 0, 1]
        ], dtype=np.float32)
        self.dist_coeffs = np.zeros(5)
        
        # 已知标记位置（场地坐标系）
        self.marker_positions = {
            0: (0, 0),       # 起飞点
            1: (2, 0),       # 航点1
            2: (2, 3),       # 航点2
            3: (0, 3),       # 航点3
            4: (1, 1.5),     # 中心点
        }
        
        # 融合定位
        self.position = np.array([0.0, 0.0, 0.0])
        self.orientation = np.array([0.0, 0.0, 0.0])
        
    def detect_markers(self, frame):
        """检测ArUco标记"""
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        corners, ids, rejected = self.detector.detectMarkers(gray)
        
        results = []
        if ids is not None:
            for i, marker_id in enumerate(ids.flatten()):
                # 估计位姿
                rvec, tvec, _ = cv2.aruco.estimatePoseSingleMarkers(
                    corners[i], self.marker_size, 
                    self.camera_matrix, self.dist_coeffs)
                
                results.append({
                    'id': int(marker_id),
                    'corners': corners[i],
                    'rvec': rvec[0][0],
                    'tvec': tvec[0][0],
                    'distance': np.linalg.norm(tvec[0][0])
                })
        
        return results
    
    def estimate_position_from_markers(self, markers, height):
        """从标记估计飞行器位置"""
        if not markers:
            return None
        
        # 使用可见标记估计位置
        positions = []
        for m in markers:
            if m['id'] in self.marker_positions:
                # 标记在图像中的位置 → 飞行器相对标记的位置
                marker_world = np.array(self.marker_positions[m['id']])
                
                # 使用平移向量计算
                tvec = m['tvec']
                
                # 飞行器在世界坐标系的位置
                pos = marker_world + np.array([tvec[0], tvec[1]])
                positions.append(pos)
        
        if positions:
            # 多标记融合（加权平均）
            return np.mean(positions, axis=0)
        
        return None
    
    def estimate_position_color(self, frame, ground_truth=None):
        """基于色标的定位（辅助）"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # 红色基准标
        red_mask = cv2.inRange(hsv, (0, 100, 100), (10, 255, 255))
        red_mask2 = cv2.inRange(hsv, (170, 100, 100), (180, 255, 255))
        red_mask = cv2.bitwise_or(red_mask, red_mask2)
        
        contours, _ = cv2.findContours(red_mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        if contours:
            largest = max(contours, key=cv2.contourArea)
            M = cv2.moments(largest)
            if M['m00'] > 0:
                cx = M['m10'] / M['m00']
                cy = M['m01'] / M['m00']
                return (cx, cy)
        
        return None
    
    def get_heading_from_markers(self, markers):
        """从标记获取航向"""
        if markers:
            m = markers[0]
            rvec = m['rvec']
            rmat, _ = cv2.Rodrigues(rvec)
            # 提取yaw角
            yaw = np.arctan2(rmat[1, 0], rmat[0, 0])
            return np.degrees(yaw)
        return None
```

### 4.2 航点飞行控制器

```python
# waypoint_controller.py
import numpy as np
import math

class WaypointController:
    """航点飞行控制器"""
    def __init__(self, flight_ctrl):
        self.flight = flight_ctrl
        
        # PID参数
        self.pid_x = PIDController(kp=1.0, ki=0.05, kd=0.5)
        self.pid_y = PIDController(kp=1.0, ki=0.05, kd=0.5)
        self.pid_z = PIDController(kp=1.5, ki=0.1, kd=0.3)
        self.pid_yaw = PIDController(kp=2.0, ki=0.0, kd=0.5)
        
        # 航点列表
        self.waypoints = []
        self.current_wp = 0
        self.reach_tolerance = 0.2  # m
        
    def set_waypoints(self, waypoints):
        """设置航点列表 [(x, y, z, yaw), ...]"""
        self.waypoints = waypoints
        self.current_wp = 0
    
    def navigate(self, current_pos, current_yaw):
        """航点导航"""
        if self.current_wp >= len(self.waypoints):
            return None, True  # 全部完成
        
        target = self.waypoints[self.current_wp]
        tx, ty, tz = target[0], target[1], target[2]
        tyaw = target[3] if len(target) > 3 else None
        
        # 位置误差
        dx = tx - current_pos[0]
        dy = ty - current_pos[1]
        dz = tz - current_pos[2]
        
        distance = math.sqrt(dx**2 + dy**2 + dz**2)
        
        # 到达判断
        if distance < self.reach_tolerance:
            self.current_wp += 1
            print(f"到达航点 {self.current_wp}: ({tx:.1f}, {ty:.1f}, {tz:.1f})")
            return (0, 0, 0, 0), False
        
        # PID控制
        vx = self.pid_x.compute(dx)
        vy = self.pid_y.compute(dy)
        vz = self.pid_z.compute(dz)
        
        # 偏航控制
        yaw_rate = 0
        if tyaw is not None:
            yaw_err = tyaw - current_yaw
            # 归一化到 -180~180
            yaw_err = (yaw_err + 180) % 360 - 180
            yaw_rate = self.pid_yaw.compute(yaw_err)
        
        # 限速
        speed = math.sqrt(vx**2 + vy**2)
        max_speed = 1.5
        if speed > max_speed:
            scale = max_speed / speed
            vx *= scale
            vy *= scale
        
        return (vx, vy, vz, yaw_rate), False
    
    def circle_waypoint(self, center, radius, altitude, num_points=12):
        """生成绕圈航点"""
        waypoints = []
        for i in range(num_points):
            angle = 2 * math.pi * i / num_points
            x = center[0] + radius * math.cos(angle)
            y = center[1] + radius * math.sin(angle)
            yaw = math.degrees(angle + math.pi)  # 朝向圆心
            waypoints.append((x, y, altitude, yaw))
        return waypoints
    
    def figure_eight(self, center, radius, altitude, num_points=24):
        """生成8字形航点"""
        waypoints = []
        for i in range(num_points):
            t = 2 * math.pi * i / num_points
            x = center[0] + radius * math.sin(t)
            y = center[1] + radius * math.sin(2*t) / 2
            waypoints.append((x, y, altitude))
        return waypoints


class PIDController:
    def __init__(self, kp, ki, kd, limit=2.0):
        self.kp, self.ki, self.kd = kp, ki, kd
        self.limit = limit
        self.integral = 0
        self.prev_error = 0
    
    def compute(self, error):
        self.integral = max(-self.limit, min(self.limit, self.integral + error * 0.05))
        derivative = error - self.prev_error
        self.prev_error = error
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        return max(-self.limit, min(self.limit, output))
```

```python
# formation_controller.py
import numpy as np

class FormationController:
    """简单编队控制器"""
    def __init__(self, role='leader', offset=(0, 0, 0)):
        """
        role: 'leader' 或 'follower'
        offset: 相对领队的偏移 (x, y, z)
        """
        self.role = role
        self.offset = np.array(offset)
        self.leader_pos = None
    
    def update_leader_position(self, leader_pos, leader_yaw):
        """follower接收领队位置"""
        self.leader_pos = np.array(leader_pos)
        self.leader_yaw = leader_yaw
    
    def get_target_position(self):
        """获取本机目标位置"""
        if self.role == 'leader':
            return None  # 领队自行决定
        
        if self.leader_pos is None:
            return None
        
        # 根据领队位置和偏移计算目标
        # 考虑领队航向旋转偏移
        yaw_rad = np.radians(self.leader_yaw)
        rot_matrix = np.array([
            [np.cos(yaw_rad), -np.sin(yaw_rad)],
            [np.sin(yaw_rad), np.cos(yaw_rad)]
        ])
        
        rotated_offset = rot_matrix @ self.offset[:2]
        target = self.leader_pos[:2] + rotated_offset
        target_z = self.leader_pos[2] + self.offset[2]
        
        return np.array([target[0], target[1], target_z])
```

### 4.4 任务主流程

```python
# main.py
import cv2
import time
import numpy as np

def main():
    # 初始化
    cam_down = cv2.VideoCapture(0)   # 下视
    cam_front = cv2.VideoCapture(2)  # 前视
    
    localizer = VisualLocalizer('marker_det.rknn')
    flight = MAVLinkController('/dev/ttyS5')
    wp_ctrl = WaypointController(flight)
    
    # 任务航点（评委指定）
    mission_waypoints = [
        (0, 0, 2.0),      # 起飞
        (2, 0, 2.0),      # 航点1
        (2, 3, 2.5),      # 航点2
        (0, 3, 2.0),      # 航点3
        (0, 0, 2.0),      # 返回
    ]
    wp_ctrl.set_waypoints(mission_waypoints)
    
    # 或生成绕杆航线
    # circle_wps = wp_ctrl.circle_waypoint(center=(2, 2), radius=1.0, altitude=2.0)
    # wp_ctrl.set_waypoints(circle_wps)
    
    state = 'TAKEOFF'
    use_visual = True   # 是否使用视觉定位
    
    while True:
        ret_d, frame_d = cam_down.read()
        ret_f, frame_f = cam_front.read()
        if not ret_d:
            continue
        
        # 1. 视觉定位
        current_pos = np.array([0.0, 0.0, 2.0])
        current_yaw = 0.0
        
        if use_visual:
            markers = localizer.detect_markers(frame_d)
            if markers:
                vis_pos = localizer.estimate_position_from_markers(markers, 2.0)
                if vis_pos is not None:
                    current_pos[:2] = vis_pos
                heading = localizer.get_heading_from_markers(markers)
                if heading is not None:
                    current_yaw = heading
        
        # 也可以用飞控自带定位
        fc_pos = flight.get_position_ned()
        if fc_pos is not None:
            # 融合视觉和飞控定位
            if use_visual and markers:
                alpha = 0.7  # 视觉权重
                current_pos = alpha * current_pos + (1-alpha) * fc_pos
            else:
                current_pos = fc_pos
        
        # 2. 状态机
        if state == 'TAKEOFF':
            flight.takeoff(2.0)
            time.sleep(8)
            state = 'NAVIGATE'
            
        elif state == 'NAVIGATE':
            velocity, done = wp_ctrl.navigate(current_pos, current_yaw)
            
            if velocity is not None:
                vx, vy, vz, yaw_rate = velocity
                flight.set_velocity_body(vx, vy, vz, yaw_rate)
            
            if done:
                print("全部航点完成")
                state = 'LAND'
                
        elif state == 'LAND':
            # 精确降落（视觉引导到标记0）
            markers = localizer.detect_markers(frame_d)
            landing_marker = None
            for m in markers:
                if m['id'] == 0:
                    landing_marker = m
                    break
            
            if landing_marker:
                corners = landing_marker['corners'][0]
                cx = np.mean(corners[:, 0])
                cy = np.mean(corners[:, 1])
                
                err_x = cx - frame_d.shape[1] / 2
                err_y = cy - frame_d.shape[0] / 2
                
                if abs(err_x) < 20 and abs(err_y) < 20:
                    flight.land()
                    break
                else:
                    flight.set_position_ned(
                        current_pos[0] - err_y * 0.001,
                        current_pos[1] + err_x * 0.001,
                        current_pos[2] - 0.1
                    )
            else:
                flight.land()
                break
        
        time.sleep(0.05)  # 20Hz

if __name__ == '__main__':
    main()
```
