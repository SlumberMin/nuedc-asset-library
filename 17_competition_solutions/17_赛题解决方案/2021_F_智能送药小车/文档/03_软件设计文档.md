typedef enum {
    STATE_IDLE = 0,      // 药房等待
    STATE_DELIVER,       // 运送到病房
    STATE_UNLOAD,        // 病房等待卸载
    STATE_RETURN         // 返回药房

### 2.2 A*路径规划

```python
import heapq

class DeliveryMap:
    def __init__(self, grid_size=10):
        self.grid_size = grid_size
        self.grid = np.zeros((grid_size, grid_size), dtype=np.int8)
        # 0=通路, 1=障碍, 2=房间, 3=药房
        self.rooms = {}
        
    def find_path(self, start, goal):
        """A*寻路算法"""
        def heuristic(a, b):
            return abs(a[0] - b[0]) + abs(a[1] - b[1])
        
        open_set = []
        heapq.heappush(open_set, (0, start))
        came_from = {}
        g_score = {start: 0}
        
        while open_set:
            _, current = heapq.heappop(open_set)
            
            if current == goal:
                path = [current]
                while current in came_from:
                    current = came_from[current]
                    path.append(current)
                return path[::-1]
            
            for dx, dy in [(0,1),(0,-1),(1,0),(-1,0)]:
                neighbor = (current[0]+dx, current[1]+dy)
                if (0 <= neighbor[0] < self.grid_size and 
                    0 <= neighbor[1] < self.grid_size and
                    self.grid[neighbor[1], neighbor[0]] != 1):
                    tentative_g = g_score[current] + 1
                    if neighbor not in g_score or tentative_g < g_score[neighbor]:
                        came_from[neighbor] = current
                        g_score[neighbor] = tentative_g
                        f_score = tentative_g + heuristic(neighbor, goal)
                        heapq.heappush(open_set, (f_score, neighbor))
        
        return None
```

### 2.3 循线PID

```c
PID_t g_track_pid = {2.0f, 0.3f, 1.0f, 0, 0};

void FollowLine(void) {
    int16_t pos = Vision_GetRedLinePos();  // OpenMV返回偏移量
    float pid_out = PID_Calc(&g_track_pid, 0, pos);
    
    int16_t base_speed = 400;
    Motor_SetSpeed(base_speed + pid_out, base_speed - pid_out);
}
```

### 2.4 主程序流程

```c
int main(void) {
    HAL_Init(); SystemClock_Config();
    GPIO_Init(); TIM_Init(); UART_Init();
    OLED_Init(); Motor_Init(); Servo_Init();
    Sensor_Init(); MPU6050_Init();
    
    while(1) {
        // OpenMV读取病房号
        uint8_t room = Vision_GetRoomNumber();
        
        // 按键确认
        uint8_t key = Key_Scan();
        if(key == KEY1 && room > 0) {
            g_target_room = room;
            g_has_medicine = 1;
            g_state = STATE_DELIVER;
        }
        
        // 状态机
        switch(g_state) {
            case STATE_IDLE:
                Motor_SetSpeed(0, 0);
                if(Sensor_DetectLoad()) {
                    g_has_medicine = 1;
                }
                break;
                
            case STATE_DELIVER:
                FollowLine();
                if(Vision_CheckRoom(g_target_room)) {
                    Motor_SetSpeed(0, 0);
                    g_state = STATE_UNLOAD;
                }
                break;
                
            case STATE_UNLOAD:
                Motor_SetSpeed(0, 0);
                if(Sensor_DetectUnload()) {
                    g_has_medicine = 0;
                    g_state = STATE_RETURN;
                }
                break;
                
            case STATE_RETURN:
                FollowLine();
                if(Vision_CheckRoom(0)) {
                    Motor_SetSpeed(0, 0);
                    g_state = STATE_IDLE;
                }
                break;
        }
        
        HAL_Delay(10);
    }
}
```

## 三、避障策略

```c
void ObstacleAvoidance(void) {
    float dist_front = Ultrasonic_GetDistance(TRIG_FRONT);
    float dist_left = Ultrasonic_GetDistance(TRIG_LEFT);
    float dist_right = Ultrasonic_GetDistance(TRIG_RIGHT);
    
    if (dist_front < 20.0f) {
        // 前方障碍: 停车→判断→绕行
        Motor_SetSpeed(0, 0);
        
        if (dist_left > dist_right) {
            // 左转绕行
            Motor_SetSpeed(-200, 200);
            HAL_Delay(500);
        } else {
            // 右转绕行
            Motor_SetSpeed(200, -200);
            HAL_Delay(500);
        }
    }
}
```

## 四、通信协议

```
OpenMV → STM32 (UART 115200):
  'R' + uint8 → 病房号 (1~9)
  'P' + int16 → 循线偏移量
  'C' + uint8 → 确认到达

STM32 → OpenMV:
  'S' → 请求识别
  'T' → 触发拍照
```
