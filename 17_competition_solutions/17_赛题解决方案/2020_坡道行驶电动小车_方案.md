# 2020 坡道行驶电动小车 - Orange Pi 5 解决方案

## 题目分析

小车需在含坡道的赛道上循迹行驶，坡道角度变化，要求在坡道上能自适应调节速度、不溜坡、不失速，平路段快速行驶。

## 视觉循迹方案

### 摄像头安装
- **型号**：OV5647，前置俯角15°~20°
- **视野**：覆盖前方1~3m赛道
- **分辨率**：640×480，ROI裁剪加速

### 循迹算法
```python
import cv2
import numpy as np

class SlopeTrackNavigator:
    """坡道循迹导航"""
    def __init__(self):
        self.roi_top_ratio = 0.4    # ROI上界(图像上方)
        self.roi_bottom_ratio = 0.95  # ROI下界

    def detect_lane(self, frame):
        """检测黑色赛道边线"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 黑色赛道
        black_mask = cv2.inRange(hsv, (0, 0, 0), (180, 255, 60))

        # 裁剪ROI
        h, w = black_mask.shape
        roi_top = int(h * self.roi_top_ratio)
        roi_bottom = int(h * self.roi_bottom_ratio)
        roi = black_mask[roi_top:roi_bottom, :]

        # 多行扫描找边线
        lane_points = []
        scan_step = 5
        for y in range(0, roi.shape[0], scan_step):
            row = roi[y, :]
            # 找左右边界
            left_edge = None
            right_edge = None
            for x in range(w//2, 0, -1):
                if row[x] > 0:
                    left_edge = x
                    break
            for x in range(w//2, w):
                if row[x] > 0:
                    right_edge = x
                    break

            if left_edge is not None and right_edge is not None:
                center_x = (left_edge + right_edge) // 2
                real_y = roi_top + y
                lane_points.append((center_x, real_y))

        if len(lane_points) < 3:
            return None, 0

        # 最小二乘拟合中线
        points = np.array(lane_points)
        coeffs = np.polyfit(points[:, 1], points[:, 0], 2)
        poly = np.poly1d(coeffs)

        # 图像底部中线位置
        bottom_center = poly(roi_bottom)
        deviation = (bottom_center - w/2) / (w/2)  # 归一化偏差 [-1, 1]

        return poly, deviation

    def detect_slope_marker(self, frame):
        """检测坡道起止标记（彩色条纹）"""
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        # 绿色坡道入口标记
        green = cv2.inRange(hsv, (35,80,80), (85,255,255))
        if cv2.countNonZero(green) > 500:
            return 'ENTER_SLOPE'
        # 黄色坡道出口标记
        yellow = cv2.inRange(hsv, (20,100,100), (35,255,255))
        if cv2.countNonZero(yellow) > 500:
            return 'EXIT_SLOPE'
        return 'NORMAL'
```

## 坡度检测

### MPU6050融合算法
```python
import smbus
import time
import math

class Inclinometer:
    """坡度检测器 - 互补滤波融合加速度计和陀螺仪"""
    def __init__(self, bus_num=3, addr=0x68):
        self.bus = smbus.SMBus(bus_num)
        self.addr = addr
        self._init_mpu()
        self.pitch = 0.0
        self.alpha = 0.98  # 互补滤波系数
        self.last_time = time.time()

    def _init_mpu(self):
        self.bus.write_byte_data(self.addr, 0x6B, 0x00)  # 唤醒
        self.bus.write_byte_data(self.addr, 0x1C, 0x08)  # ±4g
        self.bus.write_byte_data(self.addr, 0x1B, 0x08)  # ±500°/s

    def read_raw(self):
        data = self.bus.read_iota_data(self.addr, 0x3B, 14)
        ax = (data[0]<<8|data[1]) / 8192.0
        ay = (data[2]<<8|data[3]) / 8192.0
        az = (data[4]<<8|data[5]) / 8192.0
        gx = (data[8]<<8|data[9]) / 65.5
        return ax, ay, az, gx

    def update(self):
        """更新坡度角(度)"""
        ax, ay, az, gx = self.read_raw()

        now = time.time()
        dt = now - self.last_time
        self.last_time = now

        # 加速度计求坡度角
        pitch_acc = math.degrees(math.atan2(ax, math.sqrt(ay**2 + az**2)))

        # 互补滤波
        self.pitch = self.alpha * (self.pitch + gx * dt) + (1-self.alpha) * pitch_acc

        return self.pitch  # 正值=上坡, 负值=下坡

    def get_slope_grade(self):
        """返回坡度等级"""
        angle = abs(self.pitch)
        if angle < 3:
            return 'FLAT'
        elif angle < 10:
            return 'MILD'     # 缓坡
        elif angle < 20:
            return 'MODERATE' # 中坡
        else:
            return 'STEEP'    # 陡坡
```

## 自适应速度控制

```python
class AdaptiveSpeedController:
    """坡道自适应速度控制器"""
    def __init__(self):
        # 基础参数
        self.v_flat = 0.6       # 平路速度(m/s)
        self.v_up_mild = 0.4    # 缓坡上坡
        self.v_up_moderate = 0.3
        self.v_up_steep = 0.2
        self.v_down = 0.35      # 下坡速度

        # PID参数
        self.Kp = 1.5
        self.Ki = 0.1
        self.Kd = 0.3
        self.integral = 0
        self.last_error = 0

        # 前馈增益
        self.Kff_slope = 0.02  # 坡度前馈

    def compute_speed(self, deviation, slope_angle, slope_grade):
        """
        deviation: 赛道偏差 [-1, 1]
        slope_angle: 坡度角(度)
        slope_grade: 坡度等级
        """
        # 1. 根据坡度设定目标速度
        if slope_grade == 'FLAT':
            v_target = self.v_flat
        elif slope_grade == 'MILD':
            v_target = self.v_up_mild if slope_angle > 0 else self.v_down
        elif slope_grade == 'MODERATE':
            v_target = self.v_up_moderate if slope_angle > 0 else self.v_down
        else:
            v_target = self.v_up_steep

        # 2. PID转向修正
        error = deviation
        self.integral += error * 0.02
        self.integral = np.clip(self.integral, -1, 1)
        derivative = (error - self.last_error) / 0.02
        self.last_error = error

        steer = self.Kp * error + self.Ki * self.integral + self.Kd * derivative

        # 3. 坡度前馈补偿(上坡增加扭矩)
        feedforward = self.Kff_slope * slope_angle  # 正=上坡加力

        # 4. 差速输出
        L = 0.15  # 轮距
        v_left = v_target + feedforward - steer * L
        v_right = v_target + feedforward + steer * L

        # 限幅
        v_left = np.clip(v_left, -0.8, 0.8)
        v_right = np.clip(v_right, -0.8, 0.8)

        return v_left, v_right

    def anti_slide(self, slope_angle, current_speed):
        """防溜坡策略"""
        if slope_angle > 10 and current_speed < 0.05:
            # 停在坡上时施加制动
            return 'BRAKE'
        if slope_angle < -10:  # 下坡
            # 下坡限速
            return 'LIMIT_SPEED'
        return 'NORMAL'
```

## 电机驱动

```python
import gpiod

class MotorDriver:
    def __init__(self):
        # GPIO方向控制
        self.chip = gpiod.Chip('0')
        self.in1 = self.chip.get_line(10)  # PA10
        self.in2 = self.chip.get_line(11)  # PA11
        self.in3 = self.chip.get_line(12)  # PA12
        self.in4 = self.chip.get_line(13)  # PA13
        for pin in [self.in1, self.in2, self.in3, self.in4]:
            pin.request(consumer='motor', type=gpiod.LINE_REQ_DIR_OUT)

    def set_motors(self, v_left, v_right):
        """设置左右电机速度和方向"""
        # 左电机
        if v_left >= 0:
            self.in1.set_value(1); self.in2.set_value(0)
        else:
            self.in1.set_value(0); self.in2.set_value(1)
        # 右电机
        if v_right >= 0:
            self.in3.set_value(1); self.in4.set_value(0)
        else:
            self.in3.set_value(0); self.in4.set_value(1)

        # PWM调速
        duty_left = min(abs(v_left) / 0.8, 1.0)
        duty_right = min(abs(v_right) / 0.8, 1.0)
        self._set_pwm(0, duty_left)
        self._set_pwm(1, duty_right)

    def _set_pwm(self, ch, duty):
        period = 1000000  # 1kHz
        duty_ns = int(period * duty)
        base = f'/sys/class/pwm/pwmchip0/pwm{ch}'
        with open(f'{base}/duty_cycle', 'w') as f:
            f.write(str(duty_ns))
```

## 主控制流程

```python
def main():
    camera = Camera('/dev/video0')
    tracker = SlopeTrackNavigator()
    inclino = Inclinometer()
    speed_ctrl = AdaptiveSpeedController()
    motors = MotorDriver()

    while True:
        frame = camera.read()
        poly, deviation = tracker.detect_lane(frame)
        slope_angle = inclino.update()
        slope_grade = inclino.get_slope_grade()

        v_l, v_r = speed_ctrl.compute_speed(deviation, slope_angle, slope_grade)
        motors.set_motors(v_l, v_r)

        time.sleep(0.02)  # 50Hz控制
```

## 调试要点

1. **循迹参数**：先在平路调好，再上坡道微调
2. **IMU校准**：水平放置校准零偏
3. **坡度前馈**：在已知角度坡道上标定前馈增益
4. **防溜坡**：上坡停车→起步过渡要平滑
5. **下坡限速**：避免冲出弯道
