# 2009年I题 — 模拟路灯控制系统

## 题目概述
设计一个模拟路灯控制系统，实现光照检测、定时控制、故障检测等功能。

## 光照检测模块

```python
import smbus
import time

class LightSensor:
    """BH1750光照传感器"""
    
    BH1750_ADDR = 0x23
    CONTINUOUS_HIGH_RES = 0x10
    
    def __init__(self):
        self.bus = smbus.SMBus(1)
        self.bus.write_byte(self.BH1750_ADDR, self.CONTINUOUS_HIGH_RES)
        self.threshold_on = 50    # 开灯阈值(lux)
        self.threshold_off = 200  # 关灯阈值(lux)
    
    def read_lux(self):
        """读取光照度"""
        data = self.bus.read_i2c_block_data(self.BH1750_ADDR, 0x02, 2)
        lux = (data[0] << 8 | data[1]) / 1.2
        return lux
    
    def is_dark(self):
        """是否天暗"""
        return self.read_lux() < self.threshold_on
    
    def is_bright(self):
        """是否天亮"""
        return self.read_lux() > self.threshold_off
```

## 定时控制模块

```python
import smbus
from datetime import datetime, timedelta

class RTCController:
    """DS3231 RTC时钟控制"""
    
    DS3231_ADDR = 0x68
    
    def __init__(self):
        self.bus = smbus.SMBus(1)
        
        # 定时开灯/关灯时间
        self.on_time = datetime.strptime("18:00", "%H:%M").time()
        self.off_time = datetime.strptime("06:00", "%H:%M").time()
    
    def get_time(self):
        """读取当前时间"""
        data = self.bus.read_i2c_block_data(self.DS3231_ADDR, 0x00, 7)
        sec = (data[0] >> 4) * 10 + (data[0] & 0x0F)
        minute = (data[1] >> 4) * 10 + (data[1] & 0x0F)
        hour = ((data[2] >> 4) & 0x03) * 10 + (data[2] & 0x0F)
        return datetime.now().replace(hour=hour, minute=minute, second=sec).time()
    
    def should_be_on(self):
        """根据时间判断是否应开灯"""
        current = self.get_time()
        if self.on_time > self.off_time:
            # 跨天，如18:00 - 06:00
            return current >= self.on_time or current < self.off_time
        else:
            return self.on_time <= current < self.off_time
```

## LED调光控制

```python
import smbus
import OPi.GPIO as GPIO

class StreetLightController:
    """路灯PWM调光控制"""
    
    def __init__(self, num_lights=4):
        self.num_lights = num_lights
        self.bus = smbus.SMBus(1)
        self.pca_addr = 0x40
        self._init_pca9685()
        
        # 各路灯状态
        self.lights = [{'brightness': 0, 'fault': False} for _ in range(num_lights)]
        
        # 自动调光参数
        self.auto_mode = True
        self.brightness_curve = {
            # (光照度, 亮度百分比)
            0: 100,
            10: 80,
            30: 60,
            50: 40,
            100: 20,
            200: 0,
        }
    
    def _init_pca9685(self):
        self.bus.write_byte_data(self.pca_addr, 0xFE, 0x08)  # 1kHz
        self.bus.write_byte_data(self.pca_addr, 0x00, 0x20)
    
    def set_brightness(self, light_id, percent):
        """设置单路灯亮度 (0-100%)"""
        percent = max(0, min(100, percent))
        tick = int(percent / 100 * 4095)
        ch = light_id * 4
        self.bus.write_word_data(self.pca_addr, 0x06 + ch, 0)
        self.bus.write_word_data(self.pca_addr, 0x08 + ch, tick)
        self.lights[light_id]['brightness'] = percent
    
    def set_all(self, percent):
        """设置所有路灯亮度"""
        for i in range(self.num_lights):
            self.set_brightness(i, percent)
    
    def auto_adjust(self, lux):
        """根据光照自动调光"""
        # 插值计算目标亮度
        target = 0
        thresholds = sorted(self.brightness_curve.keys())
        
        for i in range(len(thresholds) - 1):
            if thresholds[i] <= lux < thresholds[i+1]:
                t = (lux - thresholds[i]) / (thresholds[i+1] - thresholds[i])
                target = self.brightness_curve[thresholds[i]] * (1-t) + \
                         self.brightness_curve[thresholds[i+1]] * t
                break
        
        if lux >= thresholds[-1]:
            target = self.brightness_curve[thresholds[-1]]
        
        self.set_all(int(target))
        return int(target)
```

## 故障检测模块

```python
import spidev

class FaultDetector:
    """路灯故障检测"""
    
    def __init__(self, num_lights=4):
        self.spi = spidev.SpiDev()
        self.spi.open(0, 0)
        self.spi.max_speed_hz = 1000000
        
        # 正常电流范围 (mA)
        self.normal_current_min = 50
        self.normal_current_max = 500
        
        # 故障记录
        self.faults = {}
    
    def read_current(self, channel):
        """通过ADC读取电流传感器值"""
        # MCP3008 ADC
        cmd = [1, (8 + channel) << 4, 0]
        result = self.spi.xfer2(cmd)
        value = ((result[1] & 0x03) << 8) | result[2]
        
        # 转换为电流值
        voltage = value * 3.3 / 1024
        current = (voltage - 2.5) / 0.185  # ACS712灵敏度
        return abs(current * 1000)  # mA
    
    def check_fault(self, light_id, is_on):
        """检查路灯故障"""
        current = self.read_current(light_id)
        
        if is_on:
            if current < self.normal_current_min:
                return '开路故障'  # 灯不亮
            elif current > self.normal_current_max:
                return '短路故障'
        else:
            if current > 10:  # 应该关闭但有电流
                return '无法关闭'
        
        return None
    
    def diagnose(self, light_controller):
        """全面诊断"""
        results = {}
        for i in range(light_controller.num_lights):
            is_on = light_controller.lights[i]['brightness'] > 0
            fault = self.check_fault(i, is_on)
            results[f'light_{i}'] = {
                'current_mA': self.read_current(i),
                'fault': fault
            }
            if fault:
                self.faults[i] = fault
        return results
```

## 主控系统

```python
import time
import logging
from datetime import datetime

class StreetLightSystem:
    """路灯控制系统主控"""
    
    def __init__(self):
        self.light_sensor = LightSensor()
        self.rtc = RTCController()
        self.lights = StreetLightController(num_lights=4)
        self.fault_detector = FaultDetector(num_lights=4)
        
        self.mode = 'AUTO'  # AUTO, TIMER, MANUAL
        
        logging.basicConfig(level=logging.INFO, 
                          format='%(asctime)s - %(message)s')
    
    def run(self):
        """主循环"""
        logging.info("路灯控制系统启动")
        
        while True:
            try:
                # 读取环境光照
                lux = self.light_sensor.read_lux()
                
                # 根据模式控制
                if self.mode == 'AUTO':
                    # 综合光照+时间判断
                    should_on = self.light_sensor.is_dark() or self.rtc.should_be_on()
                    
                    if should_on:
                        brightness = self.lights.auto_adjust(lux)
                        logging.info(f"自动模式 - 光照:{lux:.1f}lux, 亮度:{brightness}%")
                    else:
                        self.lights.set_all(0)
                
                elif self.mode == 'TIMER':
                    if self.rtc.should_be_on():
                        self.lights.set_all(100)
                    else:
                        self.lights.set_all(0)
                
                # 故障检测
                faults = self.fault_detector.diagnose(self.lights)
                for name, info in faults.items():
                    if info['fault']:
                        logging.warning(f"{name}: {info['fault']} (电流:{info['current_mA']:.1f}mA)")
                
                time.sleep(5)  # 5秒检测周期
                
            except Exception as e:
                logging.error(f"系统异常: {e}")
                time.sleep(1)
    
    def set_mode(self, mode):
        """切换控制模式"""
        self.mode = mode
        logging.info(f"切换到{mode}模式")
    
    def manual_control(self, light_id, brightness):
        """手动控制单路灯"""
        if self.mode == 'MANUAL':
            self.lights.set_brightness(light_id, brightness)

if __name__ == '__main__':
    system = StreetLightSystem()
    system.run()
```

## Web管理界面

```python
from flask import Flask, jsonify, request

app = Flask(__name__)
system = None

@app.route('/api/status')
def get_status():
    return jsonify({
        'mode': system.mode,
        'lights': system.lights.lights,
        'lux': system.light_sensor.read_lux(),
        'faults': system.fault_detector.faults
    })

@app.route('/api/mode', methods=['POST'])
def set_mode():
    system.set_mode(request.json['mode'])
    return jsonify({'ok': True})

@app.route('/api/light/<int:lid>', methods=['POST'])
def set_light(lid):
    brightness = request.json['brightness']
    system.manual_control(lid, brightness)
    return jsonify({'ok': True})
```

## 评分要点

| 评分项 | 策略 |

| 光照检测精度 | BH1750高精度传感器+滤波算法 |
| 定时准确度 | DS3231温补晶振，年误差<2分钟 |
| 调光平滑度 | 12-bit PWM + 渐变过渡 |
| 故障检测 | 电流阈值判断+多级报警 |
