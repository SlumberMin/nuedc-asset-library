```python
#!/usr/bin/env python3
"""
音频采集模块 - 多通道同步采集
"""

import numpy as np
import pyaudio
import struct
from collections import deque
import threading
import time

class AudioCapture:
    def __init__(self, sample_rate=44100, chunk_size=2048, channels=4):
        """
        sample_rate: 采样率
        chunk_size: 每帧采样点数
        channels: 通道数
        """
        self.sample_rate = sample_rate
        self.chunk_size = chunk_size
        self.channels = channels
        self.audio = pyaudio.PyAudio()
        self.stream = None
        
        # 数据缓冲区
        self.buffer = deque(maxlen=10)
        self.latest_data = None
        
        # 运行标志
        self.running = False
        
    def start(self):
        """启动音频采集"""
        # 查找I2S设备
        device_index = self._find_i2s_device()
        
        self.stream = self.audio.open(
            format=pyaudio.paInt32,
            channels=self.channels,
            rate=self.sample_rate,
            input=True,
            frames_per_buffer=self.chunk_size,
            input_device_index=device_index
        )
        
        self.running = True
        self.capture_thread = threading.Thread(target=self._capture_loop)
        self.capture_thread.start()
        
        print(f"音频采集已启动: {self.sample_rate}Hz, {self.channels}通道")
        
    def _find_i2s_device(self):
        """查找I2S音频设备"""
        for i in range(self.audio.get_device_count()):
            dev_info = self.audio.get_device_info_by_index(i)
            if 'I2S' in dev_info['name'] or 'i2s' in dev_info['name']:
                print(f"找到I2S设备: {dev_info['name']}")
                return i
        print("未找到I2S设备，使用默认设备")
        return 0
        
    def _capture_loop(self):
        """采集循环"""
        while self.running:
            try:
                data = self.stream.read(self.chunk_size, exception_on_overflow=False)
                
                # 解析为32位整数
                samples = struct.unpack(f'{self.chunk_size * self.channels}i', data)
                
                # 分离各通道
                channels_data = {}
                for ch in range(self.channels):
                    ch_data = np.array(samples[ch::self.channels], dtype=np.float64) / (2**31)
                    channels_data[f'ch{ch}'] = ch_data
                    
                self.latest_data = channels_data
                self.buffer.append(channels_data)
                
            except Exception as e:
                print(f"音频采集错误: {e}")
                time.sleep(0.01)
                
    def read_chunk(self):
        """读取一帧数据"""
        return self.latest_data
        
    def read_buffer(self, num_frames=1):
        """读取缓冲区数据"""
        if len(self.buffer) >= num_frames:
            data = []
            for _ in range(num_frames):
                data.append(self.buffer.popleft())
            return data
        return None
        
    def stop(self):
        """停止采集"""
        self.running = False
        if hasattr(self, 'capture_thread'):
            self.capture_thread.join()
        if self.stream:
            self.stream.stop_stream()
            self.stream.close()
        self.audio.terminate()
        print("音频采集已停止")
```

### 4.3 信号处理模块

```python
#!/usr/bin/env python3
"""
信号处理模块 - 滤波、特征提取、互相关
"""

import numpy as np
from scipy import signal
from scipy.fft import fft, fftfreq

class SignalProcessor:
    def __init__(self, sample_rate=44100, target_freq=2000):
        """
        sample_rate: 采样率
        target_freq: 目标频率（声源频率）
        """
        self.sample_rate = sample_rate
        self.target_freq = target_freq
        self.nyquist = sample_rate / 2
        
        # 设计带通滤波器
        low = (target_freq - 300) / self.nyquist
        high = (target_freq + 300) / self.nyquist
        self.b, self.a = signal.butter(4, [low, high], btype='band')
        
    def filter_signal(self, data):
        """带通滤波"""
        return signal.filtfilt(self.b, self.a, data)
        
    def calculate_rms(self, data):
        """计算信号RMS值"""
        filtered = self.filter_signal(data)
        return np.sqrt(np.mean(filtered**2))
        
    def calculate_amplitude(self, data):
        """计算信号幅度"""
        filtered = self.filter_signal(data)
        return np.max(np.abs(filtered))
        
    def cross_correlate(self, sig1, sig2, method='gcc-phat'):
        """
        计算两路信号的互相关
        method: 'direct', 'gcc-phat'
        返回: (延迟时间, 相关系数)
        """
        # 滤波
        sig1_filtered = self.filter_signal(sig1)
        sig2_filtered = self.filter_signal(sig2)
        
        if method == 'direct':
            # 直接互相关
            correlation = signal.correlate(sig1_filtered, sig2_filtered, mode='full')
            lags = signal.correlation_lags(len(sig1_filtered), len(sig2_filtered), mode='full')
            
            # 找到峰值
            peak_idx = np.argmax(np.abs(correlation))
            delay_samples = lags[peak_idx]
            
        elif method == 'gcc-phat':
            # GCC-PHAT（广义互相关-相位变换）
            n = len(sig1_filtered)
            
            # FFT
            X1 = fft(sig1_filtered, n=2*n)
            X2 = fft(sig2_filtered, n=2*n)
            
            # 互功率谱
            R = X1 * np.conj(X2)
            
            # PHAT加权
            magnitude = np.abs(R)
            magnitude[magnitude < 1e-10] = 1e-10
            R_phat = R / magnitude
            
            # IFFT
            correlation = np.real(np.fft.ifft(R_phat))
            
            # 找到峰值
            peak_idx = np.argmax(np.abs(correlation))
            if peak_idx > n:
                peak_idx = peak_idx - 2 * n
            delay_samples = peak_idx
            
        delay_time = delay_samples / self.sample_rate
        
        return delay_time, np.max(np.abs(correlation))
        
    def calculate_tdoa(self, mic_signals):
        """
        计算到达时间差（TDOA）
        mic_signals: {'mic0': data0, 'mic1': data1, ...}
        返回: {(i,j): delay} 字典
        """
        mic_names = sorted(mic_signals.keys())
        tdoa = {}
        
        for i in range(len(mic_names)):
            for j in range(i+1, len(mic_names)):
                delay, _ = self.cross_correlate(
                    mic_signals[mic_names[i]],
                    mic_signals[mic_names[j]]
                )
                tdoa[(i, j)] = delay
                
        return tdoa
        
    def extract_features(self, data):
        """提取声音特征"""
        filtered = self.filter_signal(data)
        
        features = {
            'rms': self.calculate_rms(data),
            'amplitude': self.calculate_amplitude(data),
            'zero_crossings': self._count_zero_crossings(filtered),
            'spectral_centroid': self._spectral_centroid(filtered)
        }
        
        return features
        
    def _count_zero_crossings(self, data):
        """计算过零率"""
        return np.sum(np.diff(np.sign(data)) != 0)
        
    def _spectral_centroid(self, data):
        """计算频谱质心"""
        spectrum = np.abs(fft(data))
        freqs = fftfreq(len(data), 1/self.sample_rate)
        positive_mask = freqs > 0
        return np.sum(freqs[positive_mask] * spectrum[positive_mask]) / np.sum(spectrum[positive_mask])
```

### 4.4 声音定位算法

```python
#!/usr/bin/env python3
"""
声音定位算法 - TDOA定位
"""

import numpy as np
from scipy.optimize import minimize

class SoundLocalizer:
    def __init__(self, mic_positions, sound_speed=343):
        """
        mic_positions: 麦克风位置坐标 [(x1,y1), (x2,y2), ...]
        sound_speed: 声速（m/s）
        """
        self.mic_positions = np.array(mic_positions)
        self.sound_speed = sound_speed
        self.num_mics = len(mic_positions)
        
    def localize_tdoa(self, tdoa):
        """
        基于TDOA的定位
        tdoa: {(i,j): delay} 字典
        返回: (x, y) 声源坐标
        """
        # 转换为距离差
        distance_diffs = {}
        for (i, j), delay in tdoa.items():
            distance_diffs[(i, j)] = delay * self.sound_speed
            
        # 使用最小二乘法求解
        def objective(pos):
            x, y = pos
            error = 0
            
            # 计算到各麦克风的距离
            distances = []
            for mic_pos in self.mic_positions:
                d = np.sqrt((x - mic_pos[0])**2 + (y - mic_pos[1])**2)
                distances.append(d)
                
            # 计算误差
            for (i, j), d_diff in distance_diffs.items():
                predicted_diff = distances[j] - distances[i]
                error += (predicted_diff - d_diff)**2
                
            return error
            
        # 初始猜测（麦克风阵列中心）
        x0 = np.mean(self.mic_positions[:, 0])
        y0 = np.mean(self.mic_positions[:, 1])
        
        # 优化
        result = minimize(objective, [x0, y0], method='Nelder-Mead')
        
        if result.success:
            return result.x
        else:
            return None
            
    def localize_spherical(self, tdoa):
        """
        基于球面插值的定位方法
        更稳定，适合实时系统
        """
        # 选择参考麦克风（第一个）
        ref_mic = self.mic_positions[0]
        
        # 构建方程组 Ax = b
        A = []
        b = []
        
        mic_list = list(self.mic_positions[1:])
        for i, mic_pos in enumerate(mic_list):
            # 计算距离差
            if (0, i+1) in tdoa:
                d_diff = tdoa[(0, i+1)] * self.sound_speed
            elif (i+1, 0) in tdoa:
                d_diff = -tdoa[(i+1, 0)] * self.sound_speed
            else:
                continue
                
            # 构建方程
            xi, yi = mic_pos
            x0, y0 = ref_mic
            
            A.append([2*(xi - x0), 2*(yi - y0), 2*d_diff])
            b.append(d_diff**2 + xi**2 + yi**2 - x0**2 - y0**2)
            
        A = np.array(A)
        b = np.array(b)
        
        # 最小二乘求解
        if len(A) >= 2:
            result = np.linalg.lstsq(A, b, rcond=None)
            if len(result[0]) >= 2:
                return result[0][:2]
                
        return None
        
    def localize_hybrid(self, tdoa, amplitudes=None, alpha=0.8):
        """
        混合定位算法
        alpha: TDOA权重
        """
        pos_tdoa = self.localize_tdoa(tdoa)
        
        if amplitudes is not None and len(amplitudes) == self.num_mics:
            # 基于RSSI的定位（加权质心）
            weights = 1.0 / (np.array(amplitudes) + 1e-6)
            weights = weights / np.sum(weights)
            pos_rssi = np.sum(self.mic_positions * weights[:, np.newaxis], axis=0)
            
            if pos_tdoa is not None:
                return alpha * pos_tdoa + (1 - alpha) * pos_rssi
            else:
                return pos_rssi
                
        return pos_tdoa
        
    def calculate_angle(self, source_pos):
        """计算声源方位角（相对于第一个麦克风）"""
        dx = source_pos[0] - self.mic_positions[0][0]
        dy = source_pos[1] - self.mic_positions[0][1]
        angle = np.arctan2(dy, dx)
        return np.degrees(angle)
        
    def calculate_distance(self, source_pos, ref_mic=0):
        """计算声源距离"""
        dx = source_pos[0] - self.mic_positions[ref_mic][0]
        dy = source_pos[1] - self.mic_positions[ref_mic][1]
        return np.sqrt(dx**2 + dy**2)
```

### 4.5 卡尔曼滤波器

```python
#!/usr/bin/env python3
"""
卡尔曼滤波器 - 用于轨迹平滑
"""

import numpy as np

class KalmanFilter2D:
    def __init__(self, dt=0.1, process_noise=0.1, measurement_noise=1.0):
        """
        dt: 采样时间
        process_noise: 过程噪声
        measurement_noise: 测量噪声
        """
        self.dt = dt
        
        # 状态向量 [x, y, vx, vy]
        self.x = np.array([[0], [0], [0], [0]])
        
        # 状态转移矩阵
        self.F = np.array([
            [1, 0, dt, 0],
            [0, 1, 0, dt],
            [0, 0, 1, 0],
            [0, 0, 0, 1]
        ])
        
        # 测量矩阵
        self.H = np.array([
            [1, 0, 0, 0],
            [0, 1, 0, 0]
        ])
        
        # 过程噪声协方差
        self.Q = np.eye(4) * process_noise
        
        # 测量噪声协方差
        self.R = np.eye(2) * measurement_noise
        
        # 误差协方差矩阵
        self.P = np.eye(4) * 100
        
    def predict(self):
        """预测步骤"""
        self.x = self.F @ self.x
        self.P = self.F @ self.P @ self.F.T + self.Q
        return self.x[:2, 0]
        
    def update(self, measurement):
        """更新步骤"""
        z = np.array(measurement).reshape(2, 1)
        
        # 卡尔曼增益
        S = self.H @ self.P @ self.H.T + self.R
        K = self.P @ self.H.T @ np.linalg.inv(S)
        
        # 状态更新
        y = z - self.H @ self.x
        self.x = self.x + K @ y
        
        # 协方差更新
        I = np.eye(4)
        self.P = (I - K @ self.H) @ self.P
        
        return self.x[:2, 0]
        
    def get_state(self):
        """获取当前状态"""
        return {
            'position': self.x[:2, 0].tolist(),
            'velocity': self.x[2:, 0].tolist()
        }
        
    def reset(self):
        """重置滤波器"""
        self.x = np.array([[0], [0], [0], [0]])
        self.P = np.eye(4) * 100
```

```python
#!/usr/bin/env python3

        self.current_angle = None
        self.current_distance = None
        
        # 运行标志
        self.running = False
        
        # 统计信息
        self.stats = {
            'update_count': 0,
            'valid_count': 0,
            'start_time': 0
        }
        
        # 数据记录
        self.data_log = {
            'time': [],
            'position': [],
            'angle': [],
            'distance': []
        }
        
    def start(self):
        """启动系统"""
        print("=" * 50)
        print("声音定位系统启动")
        print("=" * 50)
        
        # 启动音频采集
        self.audio_capture.start()
        
        self.running = True
        self.stats['start_time'] = time.time()
        
        # 启动处理线程
        self.process_thread = threading.Thread(target=self._process_loop)
        self.process_thread.start()
        
        print("系统就绪，等待声源信号...")
        
    def _process_loop(self):
        """主处理循环"""
        while self.running:
            # 读取音频数据
            audio_data = self.audio_capture.read_chunk()
            if audio_data is None:
                time.sleep(0.01)
                continue
                
            # 提取各通道数据
            mic_signals = {}
            for i in range(4):
                mic_signals[f'mic{i}'] = audio_data[f'ch{i}']
                
            # 计算TDOA
            tdoa = self.signal_processor.calculate_tdoa(mic_signals)
            
            # 计算信号幅度
            amplitudes = []
            for i in range(4):
                amp = self.signal_processor.calculate_amplitude(mic_signals[f'mic{i}'])
                amplitudes.append(amp)
                
            # 检查信号强度
            if max(amplitudes) < 0.01:  # 信号太弱
                time.sleep(0.05)
                continue
                
            # 定位
            position = self.localizer.localize_hybrid(
                tdoa=tdoa,
                amplitudes=amplitudes,
                alpha=0.8
            )
            
            if position is not None:
                # 卡尔曼滤波平滑
                filtered_position = self.kalman.update(position)
                
                # 更新状态
                self.current_position = filtered_position.tolist()
                self.current_angle = self.localizer.calculate_angle(filtered_position)
                self.current_distance = self.localizer.calculate_distance(filtered_position)
                
                # 更新统计
                self.stats['update_count'] += 1
                self.stats['valid_count'] += 1
                
                # 记录数据
                elapsed = time.time() - self.stats['start_time']
                self.data_log['time'].append(elapsed)
                self.data_log['position'].append(self.current_position)
                self.data_log['angle'].append(self.current_angle)
                self.data_log['distance'].append(self.current_distance)
                
                # 显示结果
                print(f"\r位置: ({self.current_position[0]:.3f}, {self.current_position[1]:.3f}) | "
                      f"角度: {self.current_angle:.1f}° | "
                      f"距离: {self.current_distance:.3f}m | "
                      f"更新: {self.stats['valid_count']}", end='')
            else:
                self.stats['update_count'] += 1
                
            time.sleep(0.05)  # 20Hz更新率
            
    def get_status(self):
        """获取系统状态"""
        return {
            'position': self.current_position,
            'angle': self.current_angle,
            'distance': self.current_distance,
            'update_count': self.stats['update_count'],
            'valid_count': self.stats['valid_count'],
            'elapsed': time.time() - self.stats['start_time']
        }
        
    def save_log(self, filename='localization_log.csv'):

        print(f"\n数据已保存到 {filename}")
        
    def plot_data(self):

        """停止系统"""
        self.running = False
        
        # 等待线程结束
        if hasattr(self, 'process_thread'):
            self.process_thread.join()
            
        # 停止音频采集
        self.audio_capture.stop()
        
        print("\n系统已停止")

            if audio_data is not None:

                tdoa = self.signal_processor.calculate_tdoa(mic_signals)

        avg_tdoa = {}

        return avg_tdoa
```

### 5.1 麦克风校准

```python
def calibrate_microphones(audio_capture, signal_processor, num_samples=100):
    """
    麦克风灵敏度校准
    1. 在已知位置播放已知频率的声音
    2. 测量各麦克风的信号幅度
    3. 计算灵敏度校准系数
    """
    print("开始麦克风校准...")
    print("请在中心位置放置声源...")
    input("按Enter开始...")
    
    amplitudes = {f'mic{i}': [] for i in range(4)}
    
    for _ in range(num_samples):
        audio_data = audio_capture.read_chunk()
        if audio_data:
            for i in range(4):
                amp = signal_processor.calculate_amplitude(audio_data[f'ch{i}'])
                amplitudes[f'mic{i}'].append(amp)
        time.sleep(0.05)
        
    # 计算平均幅度
    avg_amplitudes = {}
    for mic, amps in amplitudes.items():
        avg_amplitudes[mic] = np.mean(amps)
        
    # 计算校准系数（以第一个麦克风为基准）
    reference = avg_amplitudes['mic0']
    calibration_factors = {}
    for mic, avg_amp in avg_amplitudes.items():
        calibration_factors[mic] = reference / avg_amp if avg_amp > 0 else 1.0
        
    print(f"校准系数: {calibration_factors}")
    return calibration_factors
```

### 5.2 常见问题及解决

| 问题 | 可能原因 | 解决方案 |
|------|----------|----------|
| 定位不准 | 麦克风未校准 | 进行麦克风校准 |
| 响应慢 | 处理频率太低 | 减小chunk_size，优化算法 |
| 噪声干扰 | 环境噪声大 | 使用带通滤波，提高信噪比 |
| 信号弱 | 声源太远 | 增大声源功率或减小距离 |
| 多径效应 | 反射干扰 | 使用GCC-PHAT算法 |

### 5.3 性能优化

1. **提高采样率**: 使用更高的采样率提高时间分辨率
2. **优化滤波**: 使用更窄的带通滤波器
3. **并行处理**: 使用多线程处理各通道数据
4. **硬件加速**: 使用NEON指令集优化FFT计算
