# 2009年D题：无线环境监测模拟装置 - Orange Pi 5解决方案

## 题目概述

设计一个无线环境监测系统，由多个监测节点和一个主站组成。监测节点采集温度、湿度、光照等环境参数，通过无线方式发送给主站，主站接收并显示各节点数据。

### 1. 监测节点固件 (ESP32-C3)

```python
#!/usr/bin/env python3
"""
监测节点固件 - ESP32-C3 (MicroPython)
"""

import machine
import dht
import time
import json
import network
import espnow

class SensorNode:
    """环境监测节点"""
    
    def __init__(self, node_id=1):
        self.node_id = node_id
        
        # 初始化传感器
        self.dht_sensor = dht.DHT22(machine.Pin(4))
        self.light_sensor = BH1750(machine.Pin(5), machine.Pin(6))  # I2C
        
        # 初始化ESP-NOW
        self.wlan = network.WLAN(network.STA_IF)
        self.wlan.active(True)
        
        self.esp = espnow.ESPNow()
        self.esp.active(True)
        
        # 主站MAC地址（需要预先配置）
        self.master_mac = b'\xff\xff\xff\xff\xff\xff'  # 广播
        self.esp.add_peer(self.master_mac)
        
        # 数据缓存
        self.data_buffer = []
        
        # 采样间隔
        self.sample_interval = 5  # 秒
    
    def read_sensors(self):
        """读取传感器数据"""
        try:
            # 读取DHT22
            self.dht_sensor.measure()
            temperature = self.dht_sensor.temperature()
            humidity = self.dht_sensor.humidity()
            
            # 读取BH1750
            light = self.light_sensor.luminance()
            
            return {
                'node_id': self.node_id,
                'timestamp': time.time(),
                'temperature': round(temperature, 1),
                'humidity': round(humidity, 1),
                'light': round(light, 1),
                'battery': self._read_battery()
            }
        except Exception as e:
            print(f"传感器读取错误: {e}")
            return None
    
    def _read_battery(self):
        """读取电池电压"""
        # 使用ADC读取电池电压
        adc = machine.ADC(machine.Pin(7))
        adc.atten(machine.ADC.ATTN_11DB)
        
        # 读取并转换
        raw = adc.read()
        voltage = raw / 4095 * 3.3 * 2  # 分压电阻
        
        return round(voltage, 2)
    
    def send_data(self, data):
        """发送数据到主站"""
        try:
            message = json.dumps(data)
            self.esp.send(self.master_mac, message.encode())
            return True
        except Exception as e:
            print(f"发送失败: {e}")
            return False
    
    def run(self):
        """主循环"""
        print(f"节点 {self.node_id} 启动")
        
        while True:
            # 读取传感器
            data = self.read_sensors()
            
            if data:
                # 发送数据
                if self.send_data(data):
                    print(f"数据已发送: {data}")
                else:
                    # 缓存数据
                    self.data_buffer.append(data)
            
            # 尝试发送缓存数据
            self._send_buffered_data()
            
            # 等待
            time.sleep(self.sample_interval)
    
    def _send_buffered_data(self):
        """发送缓存数据"""
        if self.data_buffer:
            data = self.data_buffer.pop(0)
            if self.send_data(data):
                print(f"缓存数据已发送")


class BH1750:
    """BH1750光照传感器驱动"""
    
    def __init__(self, sda_pin, scl_pin):
        self.i2c = machine.I2C(0, sda=sda_pin, scl=scl_pin, freq=100000)
        self.addr = 0x23
        
        # 初始化
        self.i2c.writeto(self.addr, bytes([0x10]))  # 连续高分辨率模式
    
    def luminance(self):
        """读取光照强度（lux）"""
        data = self.i2c.readfrom(self.addr, 2)
        raw = (data[0] << 8) | data[1]
        lux = raw / 1.2
        
        return lux


# 节点启动
if __name__ == '__main__':
    node = SensorNode(node_id=1)
    node.run()
```

```python
#!/usr/bin/env python3
"""
主站数据接收 - Orange Pi 5
"""

import serial
import json
import time
import threading
from collections import deque

class DataReceiver:
    """数据接收器"""
    
    def __init__(self, port='/dev/ttyS2', baudrate=115200):
        self.serial = serial.Serial(port, baudrate, timeout=0.1)
        
        # 节点数据缓存
        self.node_data = {}  # node_id -> latest data
        self.data_history = {}  # node_id -> deque of data
        
        # 接收队列
        self.receive_queue = deque(maxsize=1000)
        
        # 运行状态
        self.is_running = False
        self.rx_thread = None
        
        # 回调函数
        self.on_data_received = None
        
        # 统计
        self.packet_count = 0
        self.error_count = 0
    
    def start(self):
        """启动接收"""
        self.is_running = True
        self.rx_thread = threading.Thread(target=self._rx_loop)
        self.rx_thread.daemon = True
        self.rx_thread.start()
    
    def stop(self):
        """停止接收"""
        self.is_running = False
        if self.rx_thread:
            self.rx_thread.join()
        self.serial.close()
    
    def _rx_loop(self):
        """接收循环"""
        buffer = b''
        
        while self.is_running:
            try:
                # 读取数据
                data = self.serial.read(self.serial.in_waiting or 1)
                
                if data:
                    buffer += data
                    
                    # 查找完整消息
                    while b'\n' in buffer:
                        line, buffer = buffer.split(b'\n', 1)
                        
                        try:
                            message = json.loads(line.decode('utf-8'))
                            self._process_message(message)
                        except json.JSONDecodeError:
                            self.error_count += 1
            
            except Exception as e:
                print(f"接收错误: {e}")
                time.sleep(0.1)
    
    def _process_message(self, message):
        """处理接收到的消息"""
        node_id = message.get('node_id')
        
        if node_id is not None:
            # 更新节点数据
            self.node_data[node_id] = message
            
            # 添加到历史
            if node_id not in self.data_history:
                self.data_history[node_id] = deque(maxlen=1000)
            self.data_history[node_id].append(message)
            
            # 添加到队列
            self.receive_queue.put(message)
            
            # 回调
            if self.on_data_received:
                self.on_data_received(message)
            
            self.packet_count += 1
    
    def get_node_data(self, node_id):
        """获取节点最新数据"""
        return self.node_data.get(node_id)
    
    def get_all_data(self):
        """获取所有节点数据"""
        return dict(self.node_data)
    
    def get_history(self, node_id, limit=100):
        """获取节点历史数据"""
        if node_id in self.data_history:
            history = list(self.data_history[node_id])
            return history[-limit:]
        return []
    
    def get_statistics(self):
        """获取统计信息"""
        return {
            'packet_count': self.packet_count,
            'error_count': self.error_count,
            'active_nodes': len(self.node_data)
        }


class WiFiReceiver:
    """WiFi数据接收器"""
    
    def __init__(self, port=8080):
        import socket
        
        self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self.server_socket.bind(('0.0.0.0', port))
        self.server_socket.listen(5)
        
        self.clients = []
        self.node_data = {}
        
        self.is_running = False
    
    def start(self):
        """启动服务器"""
        self.is_running = True
        
        # 接受连接线程
        self.accept_thread = threading.Thread(target=self._accept_loop)
        self.accept_thread.daemon = True
        self.accept_thread.start()
    
    def _accept_loop(self):
        """接受连接循环"""
        while self.is_running:
            try:
                client, addr = self.server_socket.accept()
                print(f"新连接: {addr}")
                self.clients.append(client)
                
                # 启动接收线程
                rx_thread = threading.Thread(target=self._client_rx_loop, args=(client,))
                rx_thread.daemon = True
                rx_thread.start()
            except:
                pass
    
    def _client_rx_loop(self, client):
        """客户端接收循环"""
        buffer = b''
        
        while self.is_running:
            try:
                data = client.recv(1024)
                if not data:
                    break
                
                buffer += data
                
                while b'\n' in buffer:
                    line, buffer = buffer.split(b'\n', 1)
                    message = json.loads(line.decode('utf-8'))
                    
                    node_id = message.get('node_id')
                    if node_id:
                        self.node_data[node_id] = message
                
            except:
                break
        
        self.clients.remove(client)
        client.close()
```

### 3. 数据处理与存储

```python
#!/usr/bin/env python3
"""
数据处理与存储
"""

import sqlite3
import time
import json

class DataStorage:
    """数据存储"""
    
    def __init__(self, db_file='environment_data.db'):
        self.db_file = db_file
        self.conn = None
        
        self._init_db()
    
    def _init_db(self):
        """初始化数据库"""
        self.conn = sqlite3.connect(self.db_file)
        cursor = self.conn.cursor()
        
        # 创建表
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS sensor_data (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                node_id INTEGER,
                timestamp REAL,
                temperature REAL,
                humidity REAL,
                light REAL,
                battery REAL
            )
        ''')
        
        cursor.execute('''
            CREATE TABLE IF NOT EXISTS alerts (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                node_id INTEGER,
                timestamp REAL,
                alert_type TEXT,
                message TEXT
            )
        ''')
        
        self.conn.commit()
    
    def insert_data(self, data):
        """插入数据"""
        cursor = self.conn.cursor()
        
        cursor.execute('''
            INSERT INTO sensor_data (node_id, timestamp, temperature, humidity, light, battery)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (
            data.get('node_id'),
            data.get('timestamp', time.time()),
            data.get('temperature'),
            data.get('humidity'),
            data.get('light'),
            data.get('battery')
        ))
        
        self.conn.commit()
    
    def insert_alert(self, node_id, alert_type, message):
        """插入报警"""
        cursor = self.conn.cursor()
        
        cursor.execute('''
            INSERT INTO alerts (node_id, timestamp, alert_type, message)
            VALUES (?, ?, ?, ?)
        ''', (node_id, time.time(), alert_type, message))
        
        self.conn.commit()
    
    def get_recent_data(self, node_id=None, limit=100):
        """获取最近数据"""
        cursor = self.conn.cursor()
        
        if node_id:
            cursor.execute('''
                SELECT * FROM sensor_data 
                WHERE node_id = ? 
                ORDER BY timestamp DESC 
                LIMIT ?
            ''', (node_id, limit))
        else:
            cursor.execute('''
                SELECT * FROM sensor_data 
                ORDER BY timestamp DESC 
                LIMIT ?
            ''', (limit,))
        
        return cursor.fetchall()
    
    def get_data_range(self, node_id, start_time, end_time):
        """获取时间范围内的数据"""
        cursor = self.conn.cursor()
        
        cursor.execute('''
            SELECT * FROM sensor_data 
            WHERE node_id = ? AND timestamp BETWEEN ? AND ?
            ORDER BY timestamp
        ''', (node_id, start_time, end_time))
        
        return cursor.fetchall()
    
    def get_alerts(self, limit=50):
        """获取报警记录"""
        cursor = self.conn.cursor()
        
        cursor.execute('''
            SELECT * FROM alerts 
            ORDER BY timestamp DESC 
            LIMIT ?
        ''', (limit,))
        
        return cursor.fetchall()
    
    def close(self):
        """关闭数据库"""
        if self.conn:
            self.conn.close()


class DataAnalyzer:
    """数据分析器"""
    
    def __init__(self, storage):
        self.storage = storage
        
        # 报警阈值
        self.thresholds = {
            'temperature': {'min': 10, 'max': 40},
            'humidity': {'min': 20, 'max': 80},
            'light': {'min': 100, 'max': 10000},
            'battery': {'min': 3.3, 'max': 4.2}
        }
    
    def check_alerts(self, data):
        """检查是否需要报警"""
        alerts = []
        
        node_id = data.get('node_id')
        
        # 温度报警
        temp = data.get('temperature')
        if temp is not None:
            if temp < self.thresholds['temperature']['min']:
                alerts.append({
                    'node_id': node_id,
                    'type': 'temperature_low',
                    'message': f'温度过低: {temp}°C'
                })
            elif temp > self.thresholds['temperature']['max']:
                alerts.append({
                    'node_id': node_id,
                    'type': 'temperature_high',
                    'message': f'温度过高: {temp}°C'
                })
        
        # 湿度报警
        humidity = data.get('humidity')
        if humidity is not None:
            if humidity < self.thresholds['humidity']['min']:
                alerts.append({
                    'node_id': node_id,
                    'type': 'humidity_low',
                    'message': f'湿度过低: {humidity}%'
                })
            elif humidity > self.thresholds['humidity']['max']:
                alerts.append({
                    'node_id': node_id,
                    'type': 'humidity_high',
                    'message': f'湿度过高: {humidity}%'
                })
        
        # 电池报警
        battery = data.get('battery')
        if battery is not None:
            if battery < self.thresholds['battery']['min']:
                alerts.append({
                    'node_id': node_id,
                    'type': 'battery_low',
                    'message': f'电池电量低: {battery}V'
                })
        
        # 保存报警
        for alert in alerts:
            self.storage.insert_alert(
                alert['node_id'],
                alert['type'],
                alert['message']
            )
        
        return alerts
    
    def calculate_statistics(self, node_id, hours=24):
        """计算统计信息"""
        end_time = time.time()
        start_time = end_time - hours * 3600
        
        data = self.storage.get_data_range(node_id, start_time, end_time)
        
        if not data:
            return None
        
        temps = [d[3] for d in data if d[3] is not None]
        humids = [d[4] for d in data if d[4] is not None]
        lights = [d[5] for d in data if d[5] is not None]
        
        import numpy as np
        
        stats = {
            'temperature': {
                'mean': np.mean(temps) if temps else None,
                'std': np.std(temps) if temps else None,
                'min': np.min(temps) if temps else None,
                'max': np.max(temps) if temps else None
            },
            'humidity': {
                'mean': np.mean(humids) if humids else None,
                'std': np.std(humids) if humids else None,
                'min': np.min(humids) if humids else None,
                'max': np.max(humids) if humids else None
            },
            'light': {
                'mean': np.mean(lights) if lights else None,
                'std': np.std(lights) if lights else None,
                'min': np.min(lights) if lights else None,
                'max': np.max(lights) if lights else None
            }
        }
        
        return stats
```

```python
                else:
                    indicator_color = self.colors['green']
                
                cv2.circle(frame, (x + panel_width - 20, y + 20), 8, indicator_color, -1)
            
            # 更新位置
            x += panel_width + padding
            
            if x + panel_width > self.width:
                x = start_x
                y += panel_height + padding
    
    def _draw_alerts(self, frame, alerts):
        """绘制报警区域"""
        x = 20
        y = 350
        
        cv2.putText(frame, 'Alerts:', (x, y),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['red'], 1)
        
        y += 25
        
        for alert in alerts[-5:]:  # 显示最近5条
            message = alert.get('message', '')
            node_id = alert.get('node_id', '')
            
            cv2.putText(frame, f'[Node {node_id}] {message}', (x + 10, y),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['yellow'], 1)
            y += 20
    
    def _draw_charts(self, frame):
        """绘制图表"""
        # 温度历史图
        chart_x = 500
        chart_y = 50
        chart_width = 500
        chart_height = 150
        
        # 背景
        cv2.rectangle(frame, (chart_x, chart_y), 
                     (chart_x + chart_width, chart_y + chart_height),
                     self.colors['panel'], -1)
        
        cv2.putText(frame, 'Temperature History', (chart_x + 10, chart_y + 20),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['text'], 1)
        
        # 绘制数据线
        for node_id, history in self.temp_history.items():
            if len(history) > 1:
                points = []
                for i, temp in enumerate(history[-50:]):
                    x = chart_x + 10 + i * (chart_width - 20) / 50
                    y = chart_y + chart_height - 20 - (temp - 10) * (chart_height - 40) / 40
                    points.append((int(x), int(y)))
                
                # 绘制线条
                for i in range(len(points) - 1):
                    cv2.line(frame, points[i], points[i+1], self.colors['green'], 2)
        
        # 湿度历史图
        chart_y += 170
        
        cv2.rectangle(frame, (chart_x, chart_y), 
                     (chart_x + chart_width, chart_y + chart_height),
                     self.colors['panel'], -1)
        
        cv2.putText(frame, 'Humidity History', (chart_x + 10, chart_y + 20),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.5, self.colors['text'], 1)
        
        # 绘制数据线
        for node_id, history in self.humid_history.items():
            if len(history) > 1:
                points = []
                for i, humid in enumerate(history[-50:]):
                    x = chart_x + 10 + i * (chart_width - 20) / 50
                    y = chart_y + chart_height - 20 - humid * (chart_height - 40) / 100
                    points.append((int(x), int(y)))
                
                for i in range(len(points) - 1):
                    cv2.line(frame, points[i], points[i+1], self.colors['blue'], 2)
    
    def _draw_status_bar(self, frame, node_data):
        """绘制状态栏"""
        y = self.height - 30
        
        cv2.line(frame, (0, y), (self.width, y), (100, 100, 100), 1)
        
        # 状态信息
        active_nodes = len(node_data)
        cv2.putText(frame, f'Active Nodes: {active_nodes}', 
                   (20, y + 20), cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
        
        cv2.putText(frame, f'Time: {time.strftime("%H:%M:%S")}', 
                   (200, y + 20), cv2.FONT_HERSHEY_SIMPLEX, 0.4, self.colors['text'], 1)
    
    def update_history(self, node_id, data):
        """更新历史数据"""
        if node_id not in self.temp_history:
            self.temp_history[node_id] = []
            self.humid_history[node_id] = []
            self.light_history[node_id] = []
        
        self.temp_history[node_id].append(data.get('temperature', 0))
        self.humid_history[node_id].append(data.get('humidity', 0))
        self.light_history[node_id].append(data.get('light', 0))
        
        # 保留最近100个数据点
        self.temp_history[node_id] = self.temp_history[node_id][-100:]
        self.humid_history[node_id] = self.humid_history[node_id][-100:]
        self.light_history[node_id] = self.light_history[node_id][-100:]
```

### 5. 主程序

```python
#!/usr/bin/env python3
"""
无线环境监测系统主程序
"""

import cv2
import time
from data_receiver import DataReceiver
from data_storage import DataStorage, DataAnalyzer
from monitoring_ui import MonitoringUI

class EnvironmentMonitor:
    """环境监测系统"""
    
    def __init__(self):
        # 初始化组件
        self.receiver = DataReceiver(port='/dev/ttyS2')
        self.storage = DataStorage()
        self.analyzer = DataAnalyzer(self.storage)
        self.ui = MonitoringUI(1024, 600)
        
        # 运行状态
        self.is_running = False
        
        # 报警列表
        self.alerts = []
        
        # 设置回调
        self.receiver.on_data_received = self._on_data_received
    
    def start(self):
        """启动系统"""
        print("启动环境监测系统...")
        
        self.receiver.start()
        self.is_running = True
        
        print("系统已启动")
    
    def stop(self):
        """停止系统"""
        self.is_running = False
        self.receiver.stop()
        self.storage.close()
        print("系统已停止")
    
    def _on_data_received(self, data):
        """数据接收回调"""
        # 存储数据
        self.storage.insert_data(data)
        
        # 检查报警
        alerts = self.analyzer.check_alerts(data)
        self.alerts.extend(alerts)
        
        # 更新UI历史
        node_id = data.get('node_id')
        self.ui.update_history(node_id, data)
        
        print(f"收到节点 {node_id} 数据: {data}")
    
    def run(self):
        """主循环"""
        self.start()
        
        print("\n按 'q' 退出")
        print("按 's' 显示统计")
        print("按 'c' 清除报警")
        
        try:
            while self.is_running:
                # 获取所有节点数据
                node_data = self.receiver.get_all_data()
                
                # 绘制界面
                frame = self.ui.draw_main_frame(node_data, self.alerts)
                
                # 显示
                cv2.imshow('Environment Monitor', frame)
                
                # 键盘控制
                key = cv2.waitKey(1) & 0xFF
                if key == ord('q'):
                    break
                elif key == ord('s'):
                    self._show_statistics()
                elif key == ord('c'):
                    self.alerts.clear()
                    print("报警已清除")
                
                time.sleep(0.1)  # 10fps
        
        except KeyboardInterrupt:
            print("\n用户中断")
        
        finally:
            self.stop()
            cv2.destroyAllWindows()
    
    def _show_statistics(self):
        """显示统计信息"""
        print("\n=== 统计信息 ===")
        
        # 接收统计
        stats = self.receiver.get_statistics()
        print(f"接收包数: {stats['packet_count']}")
        print(f"错误包数: {stats['error_count']}")
        print(f"活跃节点: {stats['active_nodes']}")
        
        # 节点统计
        for node_id in self.receiver.node_data.keys():
            node_stats = self.analyzer.calculate_statistics(node_id, hours=1)
            if node_stats:
                print(f"\n节点 {node_id} (最近1小时):")
                print(f"  温度: {node_stats['temperature']['mean']:.1f}°C (±{node_stats['temperature']['std']:.1f})")
                print(f"  湿度: {node_stats['humidity']['mean']:.1f}% (±{node_stats['humidity']['std']:.1f})")
                print(f"  光照: {node_stats['light']['mean']:.1f}lux (±{node_stats['light']['std']:.1f})")

def main():
    monitor = EnvironmentMonitor()
    monitor.run()

if __name__ == '__main__':
    main()
```

```python
#!/usr/bin/env python3
"""
节点管理工具
"""

import json
import time

class NodeManager:
    """节点管理器"""
    
    def __init__(self):
        self.nodes = {}
    
    def register_node(self, node_id, mac_address, location=''):
        """注册节点"""
        self.nodes[node_id] = {
            'mac': mac_address,
            'location': location,
            'status': 'offline',
            'last_seen': None
        }
    
    def update_status(self, node_id, data):
        """更新节点状态"""
        if node_id in self.nodes:
            self.nodes[node_id]['status'] = 'online'
            self.nodes[node_id]['last_seen'] = time.time()
    
    def check_offline_nodes(self, timeout=60):
        """检查离线节点"""
        offline_nodes = []
        
        current_time = time.time()
        
        for node_id, info in self.nodes.items():
            if info['last_seen'] is not None:
                if current_time - info['last_seen'] > timeout:
                    info['status'] = 'offline'
                    offline_nodes.append(node_id)
        
        return offline_nodes
    
    def get_node_info(self, node_id):
        """获取节点信息"""
        return self.nodes.get(node_id)
    
    def get_all_nodes(self):
        """获取所有节点"""
        return dict(self.nodes)
    
    def save_config(self, filename='nodes.json'):
        """保存配置"""
        with open(filename, 'w') as f:
            json.dump(self.nodes, f, indent=2)
    
    def load_config(self, filename='nodes.json'):
        """加载配置"""
        try:
            with open(filename, 'r') as f:
                self.nodes = json.load(f)
        except FileNotFoundError:
            print("配置文件不存在")
```
