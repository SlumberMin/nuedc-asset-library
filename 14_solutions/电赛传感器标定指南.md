## 标定基本原理

### 标定的数学模型

**线性标定 (两点标定)**:
```
实际值 = K × 传感器读数 + B
```
其中:
- K = (实际高 - 实际低) / (读数高 - 读数低)
- B = 实际高 - K × 读数高

**多项式标定**:
```
实际值 = a₀ + a₁x + a₂x² + ... + aₙxⁿ
```
适用于非线性传感器，一般二次项足够

### 标定流程

```
1. 收集标定点（至少2个，建议5~10个）
2. 建立数学模型
3. 计算标定参数
4. 验证标定精度
5. 将参数写入代码
```

## 1. ADC传感器标定

### 1.1 基本方法

```c
// ADC读数 → 实际物理量
typedef struct {
    float k;        // 斜率
    float b;        // 偏置
    float offset;   // 零点偏移
} Calibration_t;

float adc_to_real(uint16_t adc_val, Calibration_t *cal) {
    return cal->k * (float)adc_val + cal->b;
}

// 标定过程：记录两组数据 (adc1, real1), (adc2, real2)
void calibrate(Calibration_t *cal, 
               uint16_t adc1, float real1,
               uint16_t adc2, float real2) {
    cal->k = (real2 - real1) / (float)(adc2 - adc1);
    cal->b = real1 - cal->k * (float)adc1;
}
```

### 1.2 多点线性插值标定

```c
#define CAL_POINTS 5
typedef struct {
    uint16_t adc[CAL_POINTS];
    float    real[CAL_POINTS];
} MultiPointCal_t;

float multi_point_lookup(uint16_t adc_val, MultiPointCal_t *cal) {
    // 找到adc_val所在的区间
    if (adc_val <= cal->adc[0]) {
        return cal->real[0];
    }
    for (int i = 1; i < CAL_POINTS; i++) {
        if (adc_val <= cal->adc[i]) {
            float ratio = (float)(adc_val - cal->adc[i-1]) / 
                          (float)(cal->adc[i] - cal->adc[i-1]);
            return cal->real[i-1] + ratio * (cal->real[i] - cal->real[i-1]);
        }
    }
    return cal->real[CAL_POINTS - 1];
}
```

## 2. 陀螺仪与加速度计标定

### 2.1 陀螺仪零偏标定

```c
// 静止状态下采集N个样本求平均
float gyro_offset = 0;
#define GYRO_CAL_SAMPLES 1000

void calibrate_gyro_offset(float *gyro_raw) {
    double sum = 0;
    for (int i = 0; i < GYRO_CAL_SAMPLES; i++) {
        sum += gyro_raw[i];
        delay_ms(2);
    }
    gyro_offset = (float)(sum / GYRO_CAL_SAMPLES);
}

// 使用时减去零偏
float get_gyro_rate(float raw) {
    return (raw - gyro_offset) * GYRO_SENSITIVITY;  // °/s
}
```

### 2.2 加速度计六面标定

**原理**: 加速度计在6个面（±X, ±Y, ±Z）各采集一组数据，求解标定参数

```c
typedef struct {
    float offset[3];    // XYZ零偏
    float scale[3];     // XYZ比例因子
} AccelCal_t;

// 最简标定: 水平放置
// Z轴朝上: 读数应为 (0, 0, +1g)
// Z轴朝下: 读数应为 (0, 0, -1g)
void calibrate_accel_z(AccelCal_t *cal, 
                        float z_up_raw, float z_down_raw) {
    cal->scale[2] = 2.0f / (z_up_raw - z_down_raw);  // 2g范围
    cal->offset[2] = (z_up_raw + z_down_raw) / 2.0f;
}

// 六面完整标定
void calibrate_accel_full(AccelCal_t *cal,
    float x_max, float x_min,
    float y_max, float y_min,
    float z_max, float z_min) {
    cal->scale[0] = 2.0f / (x_max - x_min);
    cal->scale[1] = 2.0f / (y_max - y_min);
    cal->scale[2] = 2.0f / (z_max - z_min);
    cal->offset[0] = (x_max + x_min) / 2.0f;
    cal->offset[1] = (y_max + y_min) / 2.0f;
    cal->offset[2] = (z_max + z_min) / 2.0f;
}

float get_accel_real(float raw, int axis, AccelCal_t *cal) {
    return (raw - cal->offset[axis]) * cal->scale[axis];
}
```

### 2.3 姿态融合标定 (互补滤波)

```c
#define ALPHA 0.98f  // 陀螺仪权重

typedef struct {
    float angle;       // 融合角度
    float gyro_offset; // 陀螺仪零偏
} ComplementaryFilter_t;

float complementary_update(ComplementaryFilter_t *cf,
                           float gyro_raw, float accel_angle, float dt) {
    float gyro_rate = gyro_raw - cf->gyro_offset;
    cf->angle = ALPHA * (cf->angle + gyro_rate * dt) + 
                (1.0f - ALPHA) * accel_angle;
    return cf->angle;
}
```

## 3. 电流传感器标定

### 3.1 ACS712 标定

```c
// ACS712-5A: 灵敏度 185mV/A, 零电流输出 VCC/2
// ACS712-20A: 灵敏度 100mV/A
// ACS712-30A: 灵敏度 66mV/A

typedef struct {
    float vref;         // 参考电压(V)，通常3.3V
    float sensitivity;  // mV/A
    float zero_offset;  // 零电流时的ADC值
    float v_per_adc;    // 每个ADC对应的电压
} CurrentSensorCal_t;

float adc_to_current(uint16_t adc, CurrentSensorCal_t *cal) {
    float voltage = (float)adc * cal->v_per_adc;
    float v_offset = cal->zero_offset * cal->v_per_adc;
    return (voltage - v_offset) / (cal->sensitivity / 1000.0f);
}

// 标定零点: 无负载时采集多个样本取平均
void calibrate_current_zero(CurrentSensorCal_t *cal, uint16_t *samples, int n) {
    float sum = 0;
    for (int i = 0; i < n; i++) sum += samples[i];
    cal->zero_offset = sum / n;
}
```

### 3.2 采样电阻+运放标定

```c
// 采样电阻 Rsense = 0.1Ω, 运放增益 G = 20
// 电流 I = Vadc / (Rsense × G)

float adc_to_current_shunt(uint16_t adc, float vref, int adc_bits,
                           float rsense, float gain) {
    float v_adc = (float)adc / (1 << adc_bits) * vref;
    return v_adc / (rsense * gain);
}
```

## 4. 电压传感器标定

### 4.1 电阻分压标定

```c
// 分压电阻: R1=10k, R2=2k → 分压比 = R2/(R1+R2) = 1/6
// 实际电压 = Vadc × (R1+R2)/R2

typedef struct {
    float ratio;     // 分压比 (R1+R2)/R2
    float vref;      // ADC参考电压
    int adc_max;     // ADC最大值 (如4095)
} VoltageSensorCal_t;

float adc_to_voltage(uint16_t adc, VoltageSensorCal_t *cal) {
    float v_adc = (float)adc / cal->adc_max * cal->vref;
    return v_adc * cal->ratio;
}

// 标定方法: 用万用表测量实际电压，与ADC读数对比修正ratio
void calibrate_voltage_ratio(VoltageSensorCal_t *cal,
                              uint16_t adc_val, float actual_voltage) {
    float v_adc = (float)adc_val / cal->adc_max * cal->vref;
    cal->ratio = actual_voltage / v_adc;
}
```

## 5. 光电编码器标定

### 5.1 脉冲数到角度/距离

```c
typedef struct {
    int ppr;          // 每转脉冲数
    float wheel_dia;  // 轮子直径(mm)
    int count;        // 当前计数
    int prev_count;
} EncoderCal_t;

float encoder_to_angle(EncoderCal_t *cal) {
    return (float)cal->count / (cal->ppr * 4) * 360.0f;  // 4倍频
}

float encoder_to_distance(EncoderCal_t *cal) {
    float circumference = 3.14159f * cal->wheel_dia;  // mm
    return (float)cal->count / (cal->ppr * 4) * circumference;
}

float encoder_get_speed(EncoderCal_t *cal, float dt) {
    int delta = cal->count - cal->prev_count;
    cal->prev_count = cal->count;
    float dist = (float)delta / (cal->ppr * 4) * 
                 (3.14159f * cal->wheel_dia);
    return dist / dt;  // mm/s
}
```

## 6. 超声波传感器标定

### 6.1 HC-SR04 标定

```c
// 距离 = 声速 × 时间 / 2
// 声速受温度影响: v = 331.3 + 0.606 × T (m/s)

typedef struct {
    float sound_speed;  // m/s, 随温度变化
} UltrasonicCal_t;

float pulse_to_distance(float pulse_us, UltrasonicCal_t *cal) {
    // 距离(mm) = 声速(mm/s) × 时间(s) / 2
    float time_s = pulse_us / 1e6f;
    return cal->sound_speed * 1000.0f * time_s / 2.0f;
}

void update_sound_speed(UltrasonicCal_t *cal, float temperature_c) {
    cal->sound_speed = 331.3f + 0.606f * temperature_c;
}

// 标定: 已知距离处测量脉冲宽度，修正sound_speed
void calibrate_ultrasonic(UltrasonicCal_t *cal, 
                           float pulse_us, float known_dist_mm) {
    float time_s = pulse_us / 1e6f;
    cal->sound_speed = known_dist_mm * 2.0f / (time_s * 1000.0f);
}
```

## 7. 红外传感器标定

### 7.1 红外测距 (Sharp GP2Y0A21) 标定

```c
// 典型曲线: distance = a / (V - b)  (非线性)
// 经验公式拟合

typedef struct {
    float a, b;  // 拟合参数
} IRDistanceCal_t;

float adc_to_ir_distance(uint16_t adc, IRDistanceCal_t *cal,
                          float vref, int adc_max) {
    float v = (float)adc / adc_max * vref;
    if (v <= cal->b) return 999.0f;  // 超出范围
    return cal->a / (v - cal->b);    // cm
}

// 多点拟合标定
void calibrate_ir_sensor(IRDistanceCal_t *cal,
    float *voltages, float *distances, int n_points) {
    // 最小二乘法拟合 1/d = (V-b)/a
    // 即 d = a/(V-b), 可转化为线性回归
    // 简化: 用两组数据估算
    float v1 = voltages[0], d1 = distances[0];
    float v2 = voltages[n_points-1], d2 = distances[n_points-1];
    // d1 = a/(v1-b), d2 = a/(v2-b)
    // 联立求解 a, b
    cal->b = (d2*v2 - d1*v1) / (d2 - d1);
    cal->a = d1 * (v1 - cal->b);
}
```

## 8. 温度传感器标定

### 8.1 NTC热敏电阻标定

```c
// Steinhart-Hart方程: 1/T = A + B*ln(R) + C*(ln(R))^3
// 简化B参数方程: 1/T = 1/T0 + (1/B)*ln(R/R0)

typedef struct {
    float R0;    // 25°C时的阻值
    float T0;    // 参考温度(K) = 298.15
    float B;     // B参数(查数据手册)
    float R_ser; // 串联电阻
    float vref;  // 参考电压
    int adc_max;
} NTC_Cal_t;

float ntc_to_temperature(uint16_t adc, NTC_Cal_t *cal) {
    float v_adc = (float)adc / cal->adc_max * cal->vref;
    float R_ntc = cal->R_ser * v_adc / (cal->vref - v_adc);
    float inv_T = 1.0f/cal->T0 + (1.0f/cal->B) * logf(R_ntc / cal->R0);
    return 1.0f / inv_T - 273.15f;  // 转为摄氏度
}
```

## 9. 电位器/角度传感器标定

```c
typedef struct {
    uint16_t adc_min;   // 最小角度对应的ADC值
    uint16_t adc_max;   // 最大角度对应的ADC值
    float angle_min;    // 最小角度(°)
    float angle_max;    // 最大角度(°)
} PotentiometerCal_t;

float pot_to_angle(uint16_t adc, PotentiometerCal_t *cal) {
    float ratio = (float)(adc - cal->adc_min) / 
                  (float)(cal->adc_max - cal->adc_min);
    return cal->angle_min + ratio * (cal->angle_max - cal->angle_min);
}

// 标定: 转到两个极端位置，记录ADC值
void calibrate_potentiometer(PotentiometerCal_t *cal,
                              uint16_t adc_at_min, uint16_t adc_at_max,
                              float angle_min, float angle_max) {
    cal->adc_min = adc_at_min;
    cal->adc_max = adc_at_max;
    cal->angle_min = angle_min;
    cal->angle_max = angle_max;
}
```

## 10. 多传感器融合标定

### 10.1 卡尔曼滤波融合

```c
typedef struct {
    float x;     // 状态估计
    float P;     // 估计协方差
    float Q;     // 过程噪声
    float R;     // 观测噪声
} KalmanFilter_t;

float kalman_update(KalmanFilter_t *kf, float z) {
    // 预测
    float x_pred = kf->x;
    float p_pred = kf->P + kf->Q;
    // 更新
    float K = p_pred / (p_pred + kf->R);
    kf->x = x_pred + K * (z - x_pred);
    kf->P = (1 - K) * p_pred;
    return kf->x;
}

// 标定Q和R: 观察滤波器响应
// Q大 → 信任测量，响应快，噪声大
// R大 → 信任模型，响应慢，平滑
```

### 10.2 标定数据存储

```c
// 使用Flash/EEPROM保存标定参数
#include "stm32f1xx_hal.h"

#define CAL_FLASH_ADDR  0x0800FC00  // Flash最后一页

typedef struct {
    uint32_t magic;      // 0xCA1B2C3D 标识已标定
    float gyro_offset[3];
    float accel_offset[3];
    float accel_scale[3];
    float voltage_ratio;
    float current_offset;
    // ... 其他标定参数
} CalibrationData_t;

void save_calibration(CalibrationData_t *cal) {
    HAL_FLASH_Unlock();
    // 擦除页
    FLASH_EraseInitTypeDef erase;
    erase.TypeErase = FLASH_TYPEERASE_PAGES;
    erase.PageAddress = CAL_FLASH_ADDR;
    erase.NbPages = 1;
    uint32_t error;
    HAL_FLASHEx_Erase(&erase, &error);
    // 写入
    uint32_t *data = (uint32_t*)cal;
    for (int i = 0; i < sizeof(CalibrationData_t)/4; i++) {
        HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, 
                         CAL_FLASH_ADDR + i*4, data[i]);
    }
    HAL_FLASH_Lock();
}
```

## 标定精度参考

| 传感器 | 标定方法 | 典型精度 |
|--------|----------|----------|
| 电阻分压电压 | 单点修正 | ±0.5% |
| ACS712电流 | 零点标定 | ±1% |
| 陀螺仪 | 静态零偏 | ±0.1°/s |
| 加速度计 | 六面标定 | ±0.02g |
| 超声波 | 已知距离修正 | ±2mm |
| NTC温度 | B参数法 | ±0.5°C |
| 编码器 | 脉冲计数 | ±1个脉冲 |
| 电位器 | 两点标定 | ±0.5° |
