## 1. PID控制

### 1.1 标准PID

```
u(t) = Kp·e(t) + Ki·∫e(t)dt + Kd·de(t)/dt
```

### 1.2 增量式PID（推荐）

```c
typedef struct {
    float Kp, Ki, Kd;
    float error[3];      // 当前/上次/上上次
    float output;
    float max_output;
    float deadzone;
} IncrementalPID_t;

float IncrementalPID_Calc(IncrementalPID_t *pid, float target, float current) {
    pid->error[2] = pid->error[1];
    pid->error[1] = pid->error[0];
    pid->error[0] = target - current;
    
    float delta = pid->Kp * (pid->error[0] - pid->error[1])
                + pid->Ki * pid->error[0]
                + pid->Kd * (pid->error[0] - 2.0f * pid->error[1] + pid->error[2]);
    
    pid->output += delta;
    
    // 死区
    if (fabsf(pid->error[0]) < pid->deadzone) pid->output = 0.0f;
    
    // 限幅
    if (pid->output > pid->max_output) pid->output = pid->max_output;
    if (pid->output < -pid->max_output) pid->output = -pid->max_output;
    
    return pid->output;
}
```

### 1.3 位置式PID

```c
typedef struct {
    float Kp, Ki, Kd;
    float integral;
    float prev_error;
    float max_integral;
    float max_output;
} PositionPID_t;

float PositionPID_Calc(PositionPID_t *pid, float target, float current, float dt) {
    float error = target - current;
    
    pid->integral += error * dt;
    // 积分限幅
    if (pid->integral > pid->max_integral) pid->integral = pid->max_integral;
    if (pid->integral < -pid->max_integral) pid->integral = -pid->max_integral;
    
    float derivative = (error - pid->prev_error) / dt;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;
    pid->prev_error = error;
    
    if (output > pid->max_output) output = pid->max_output;
    if (output < -pid->max_output) output = -pid->max_output;
    
    return output;
}
```

### 1.4 调参口诀

| 参数 | 增大效果 | 调参顺序 |
|------|---------|---------|
| Kp | 加快响应，减小稳态误差，可能振荡 | 先调 |
| Ki | 消除稳态误差，可能超调增大 | 再调 |
| Kd | 抑制振荡，改善动态性能 | 最后调 |

**经验法则**：先P后I再D，振荡减P，慢响应加P，有稳态误差加I，超调大加D。

## 2. FFT（快速傅里叶变换）

### 2.1 应用场景

- 音频信号频率分析（麦克风题）
- 振动信号频谱分析
- 信号滤波（频域滤波）
- 频率测量

### 2.2 基数2-FFT（蝶形运算）

```c
#include <math.h>

#define FFT_N 256  // 必须是2的幂

typedef struct {
    float re;
    float im;
} Complex_t;

Complex_t Complex_Mul(Complex_t a, Complex_t b) {
    return (Complex_t){a.re*b.re - a.im*b.im, a.re*b.im + a.im*b.re};
}

Complex_t Complex_Add(Complex_t a, Complex_t b) {
    return (Complex_t){a.re + b.re, a.im + b.im};
}

Complex_t Complex_Sub(Complex_t a, Complex_t b) {
    return (Complex_t){a.re - b.re, a.im - b.im};
}

void FFT_BitReverse(Complex_t *x, int n) {
    int j = 0;
    for (int i = 0; i < n - 1; i++) {
        if (i < j) {
            Complex_t temp = x[i];
            x[i] = x[j];
            x[j] = temp;
        }
        int k = n >> 1;
        while (k <= j) {
            j -= k;
            k >>= 1;
        }
        j += k;
    }
}

void FFT_Compute(Complex_t *x, int n) {
    FFT_BitReverse(x, n);
    
    for (int stage = 1; stage < n; stage <<= 1) {
        float angle = -M_PI / stage;
        Complex_t wn = {cosf(angle), sinf(angle)};
        
        for (int group = 0; group < n; group += (stage << 1)) {
            Complex_t w = {1.0f, 0.0f};
            for (int pair = 0; pair < stage; pair++) {
                Complex_t t = Complex_Mul(w, x[group + pair + stage]);
                Complex_t u = x[group + pair];
                x[group + pair] = Complex_Add(u, t);
                x[group + pair + stage] = Complex_Sub(u, t);
                w = Complex_Mul(w, wn);
            }
        }
    }
}

// 计算幅值谱
void FFT_Magnitude(Complex_t *x, float *mag, int n) {
    for (int i = 0; i < n / 2; i++) {
        mag[i] = sqrtf(x[i].re * x[i].re + x[i].im * x[i].im) / (n / 2);
    }
}

// 寻找基频
float FFT_FindFundamentalFreq(float *mag, int n, float sample_rate) {
    int max_idx = 1;
    float max_val = 0;
    for (int i = 1; i < n / 2; i++) {
        if (mag[i] > max_val) {
            max_val = mag[i];
            max_idx = i;
        }
    }
    return (float)max_idx * sample_rate / (float)n;
}
```

### 2.3 FFT使用要点

| 要点 | 说明 |
|------|------|
| 采样率 | ≥ 2×信号最高频率（奈奎斯特定理） |
| 采样点数 | 2的幂，越大频率分辨率越高 |
| 频率分辨率 | Δf = fs / N |
| 加窗 | 频谱泄漏时加汉宁窗/汉明窗 |
| 幅值校正 | 除以N/2得到真实幅值 |

## 3. 卡尔曼滤波

### 3.1 一维卡尔曼滤波器

```c
typedef struct {
    float x;    // 状态估计
    float P;    // 估计误差协方差
    float Q;    // 过程噪声
    float R;    // 测量噪声
} KalmanFilter1D_t;

void Kalman1D_Init(KalmanFilter1D_t *kf, float Q, float R, float x0, float P0) {
    kf->Q = Q;
    kf->R = R;
    kf->x = x0;
    kf->P = P0;
}

float Kalman1D_Update(KalmanFilter1D_t *kf, float measurement) {
    // 预测
    kf->P += kf->Q;
    
    // 更新
    float K = kf->P / (kf->P + kf->R);  // 卡尔曼增益
    kf->x += K * (measurement - kf->x);
    kf->P *= (1.0f - K);
    
    return kf->x;
}
```

### 3.2 二维卡尔曼（位置-速度）

```c
typedef struct {
    float x[2];      // [位置, 速度]
    float P[2][2];   // 协方差矩阵
    float Q[2][2];   // 过程噪声
    float R;         // 测量噪声
} Kalman2D_t;

void Kalman2D_Predict(Kalman2D_t *kf, float dt) {
    // 状态预测：x = F*x
    float x_new = kf->x[0] + kf->x[1] * dt;
    float v_new = kf->x[1];
    kf->x[0] = x_new;
    kf->x[1] = v_new;
    
    // 协方差预测：P = F*P*F^T + Q
    float P00 = kf->P[0][0] + dt*(kf->P[1][0] + kf->P[0][1]) + dt*dt*kf->P[1][1] + kf->Q[0][0];
    float P01 = kf->P[0][1] + dt*kf->P[1][1] + kf->Q[0][1];
    float P10 = kf->P[1][0] + dt*kf->P[1][1] + kf->Q[1][0];
    float P11 = kf->P[1][1] + kf->Q[1][1];
    kf->P[0][0] = P00; kf->P[0][1] = P01;
    kf->P[1][0] = P10; kf->P[1][1] = P11;
}

float Kalman2D_Update(Kalman2D_t *kf, float z) {
    // 只观测位置 H = [1, 0]
    float S = kf->P[0][0] + kf->R;
    float K0 = kf->P[0][0] / S;
    float K1 = kf->P[1][0] / S;
    
    float innovation = z - kf->x[0];
    kf->x[0] += K0 * innovation;
    kf->x[1] += K1 * innovation;
    
    // 更新协方差
    float P00 = (1.0f - K0) * kf->P[0][0];
    float P01 = (1.0f - K0) * kf->P[0][1];
    float P10 = kf->P[1][0] - K1 * kf->P[0][0];
    float P11 = kf->P[1][1] - K1 * kf->P[0][1];
    kf->P[0][0] = P00; kf->P[0][1] = P01;
    kf->P[1][0] = P10; kf->P[1][1] = P11;
    
    return kf->x[0];
}
```

### 3.3 卡尔曼调参指南

| 参数 | 含义 | 调整方向 |
|------|------|---------|
| Q（过程噪声） | 模型不确定性 | 增大→更信任测量，响应快但噪声大 |
| R（测量噪声） | 传感器噪声 | 增大→更信任模型，平滑但响应慢 |
| P0 | 初始不确定度 | 影响收敛速度，一般取较大值 |

## 5. 通信协议

### 5.1 UART串口通信

#### 帧协议设计

```
┌──────┬──────┬──────┬────────┬──────┬──────┐
│ 帧头  │ 长度  │ 命令  │  数据   │ 校验  │ 帧尾  │
│ 0xAA │ 1B   │ 1B   │ nB     │ 1B   │ 0x55 │
└──────┴──────┴──────┴────────┴──────┴──────┘
校验：数据段异或校验 或 CRC8
```

#### 解析状态机

```c
typedef enum {
    PARSE_IDLE,
    PARSE_LEN,
    PARSE_CMD,
    PARSE_DATA,
    PARSE_CHECK,
    PARSE_TAIL
} ParseState_t;

typedef struct {
    ParseState_t state;
    uint8_t buffer[64];
    uint8_t length;
    uint8_t index;
    uint8_t checksum;
} FrameParser_t;

bool FrameParser_FeedByte(FrameParser_t *p, uint8_t byte) {
    switch (p->state) {
        case PARSE_IDLE:
            if (byte == 0xAA) p->state = PARSE_LEN;
            break;
        case PARSE_LEN:
            p->length = byte;
            p->index = 0;
            p->checksum = 0;
            p->state = PARSE_CMD;
            break;
        case PARSE_CMD:
            p->buffer[0] = byte;
            p->checksum ^= byte;
            p->state = (p->length > 1) ? PARSE_DATA : PARSE_CHECK;
            break;
        case PARSE_DATA:
            p->buffer[p->index++] = byte;
            p->checksum ^= byte;
            if (p->index >= p->length - 1) p->state = PARSE_CHECK;
            break;
        case PARSE_CHECK:
            p->state = PARSE_IDLE;
            return (byte == p->checksum);  // 校验通过返回true
        default:
            p->state = PARSE_IDLE;
            break;
    }
    return false;
}
```

### 5.2 SPI通信要点

```
模式0：CPOL=0, CPHA=0（最常用）
模式3：CPOL=1, CPHA=1

速度：通常 ≤ 外设最大时钟/2
片选：软件控制CS拉低→传输→拉高
```

### 5.3 I2C通信要点

```
起始条件：SDA高→低（SCL高时）
停止条件：SDA低→高（SCL高时）
地址：7位地址 + 1位读写位
应答：每字节后接收方拉低SDA
```

### 5.4 无线通信（NRF24L01示例）

```c
// NRF24L01通道和地址配置
void NRF24L01_Config(void) {
    // 设置通道
    SPI_WriteReg(RF_CH, 40);          // 通道40
    
    // 设置地址（5字节）
    uint8_t addr[5] = {0xE7, 0xE7, 0xE7, 0xE7, 0xE7};
    SPI_WriteBytes(TX_ADDR, addr, 5);
    SPI_WriteBytes(RX_ADDR_P0, addr, 5);
    
    // 设置速率和功率
    SPI_WriteReg(RF_SETUP, 0x0E);     // 2Mbps, 0dBm
    
    // 设置有效数据宽度
    SPI_WriteReg(RX_PW_P0, 32);       // 32字节
}
```

## 6. 综合实战技巧

### 6.1 ADC采样优化

```c
// 多次采样取均值，去除异常值
uint16_t ADC_FilteredRead(ADC_HandleTypeDef *hadc) {
    uint16_t samples[16];
    for (int i = 0; i < 16; i++) {
        HAL_ADC_Start(hadc);
        HAL_ADC_PollForConversion(hadc, 10);
        samples[i] = HAL_ADC_GetValue(hadc);
    }
    
    // 排序取中值（去除最大最小各4个）
    // 简化：取均值
    uint32_t sum = 0;
    for (int i = 0; i < 16; i++) sum += samples[i];
    return (uint16_t)(sum / 16);
}
```

### 6.2 定时器中断配置要点

```
TIM频率 = 时钟频率 / (PSC+1) / (ARR+1)

例：72MHz主频，1ms中断
PSC = 71, ARR = 999
频率 = 72MHz / 72 / 1000 = 1kHz
```

### 6.3 GPIO速度等级选择

| 等级 | 速度 | 适用场景 |
|------|------|---------|
| 低速 | 2MHz | LED、按键 |
| 中速 | 25MHz | I2C、SPI |
| 高速 | 50MHz | 并口、高速SPI |
| 超高速 | 100MHz+ | FSMC、LTDC |
