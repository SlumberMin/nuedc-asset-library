## 2. 阻抗测量

### 2.1 测量方法分类

| 方法 | 原理 | 精度 | 频率范围 | 实现难度 |
|------|------|------|---------|---------|
| 伏安法 | V/I 测量 | 中 | DC~MHz | ⭐⭐ |
| 电桥法 | 平衡比较 | 高 | DC~GHz | ⭐⭐⭐ |
| I-V 变换法 | 运放反馈 | 中 | DC~MHz | ⭐⭐ |
| 谐振法 | LC 谐振频率 | 中 | kHz~GHz | ⭐⭐ |
| 反射法 | 反射系数 | 高 | MHz~GHz | ⭐⭐⭐⭐ |
| 自动平衡电桥 | 伺服反馈 | 高 | DC~110MHz | ⭐⭐⭐ |

### 2.2 伏安法（最常用）

```
原理: Z = V / I

实现:
  信号源 → [已知电阻 Rs] → [被测阻抗 Zx]
              |                    |
           ADC测Vrs            ADC测Vzx
  
  Zx = Rs × Vzx / Vrs
```

**适用场景**：阻值范围 1Ω~1MΩ，频率 DC~1MHz

### 2.3 运放 I-V 变换法（推荐）

```
         Rf
   ┌────[===]────┐
   |              |
   |    ┌────┐    |
信号源──┤-   │    |
        │ OPA├────┴──── ADC (Vout)
Zx ──┤+   │
        └────┘

Vout = -Zx / Rf × Vin
Zx = -Rf × Vout / Vin
```

**优势**：
- 一端接地，测量方便
- 精度取决于 Rf 精度
- 适合自动化测量

### 2.4 电桥法

```
经典惠斯通电桥:

    Vin
    ┌───┐
    │   │
   R1   R3
    │   │
    ├───┼── Vdetector
    │   │
   R2   Zx
    │   │
    └───┘
   GND

平衡条件: Zx = R3 × R2 / R1
```

### 2.5 自动平衡电桥法（高精度 LCR 表原理）

```
        Rr
信号源──[===]──┬── Zx ── GND
              |
         ┌────┴────┐
         │  反相    │
         │ 放大器   ├── V2 (ADC)
         └─────────┘
              |
          ┌───┘
          R (虚地)

Zx = -Rr × V2 / V1

测量 V1 和 V2 的幅度比和相位差即可得到 Zx 的实部和虚部
```

### 2.6 阻抗测量 Python 代码

```python
import numpy as np

class ImpedanceMeter:
    """基于 DFT 的阻抗测量"""
    
    def __init__(self, rs, fs=100000, n_cycles=10):
        """
        rs: 串联采样电阻
        fs: 采样率
        n_cycles: 测量周期数
        """
        self.rs = rs
        self.fs = fs
        self.n_cycles = n_cycles
    
    def measure_dft(self, v_rs, v_zx, f_test):
        """
        通过 DFT 提取基波分量
        v_rs: 采样电阻两端电压序列
        v_zx: 被测阻抗两端电压序列
        f_test: 测试频率
        返回: Z (复数)
        """
        N = len(v_rs)
        n = np.arange(N)
        
        # 测试频率对应的数字频率
        w = 2 * np.pi * f_test / self.fs
        
        # DFT 提取基波 (相关法)
        cos_ref = np.cos(w * n)
        sin_ref = np.sin(w * n)
        
        # V_rs 的基波
        rs_I = 2 * np.mean(v_rs * cos_ref)
        rs_Q = 2 * np.mean(v_rs * sin_ref)
        V_rs = rs_I - 1j * rs_Q
        
        # V_zx 的基波
        zx_I = 2 * np.mean(v_zx * cos_ref)
        zx_Q = 2 * np.mean(v_zx * sin_ref)
        V_zx = zx_I - 1j * zx_Q
        
        # 阻抗计算
        I = V_rs / self.rs
        Z = V_zx / I
        
        return Z
    
    def decompose(self, Z):
        """分解阻抗"""
        R = Z.real
        X = Z.imag
        mag = abs(Z)
        phase = np.angle(Z, deg=True)
        
        Q = abs(R / X) if X != 0 else float('inf')
        
        return {
            'R': R, 'X': X,
            'magnitude': mag, 'phase_deg': phase,
            'Q': Q
        }
    
    def identify_component(self, Z, f_test):
        """识别元件类型"""
        R = Z.real
        X = Z.imag
        
        if abs(X) < abs(R) * 0.05:
            return f'电阻 R = {R:.2f} Ω'
        elif X > 0:
            L = X / (2 * np.pi * f_test)
            return f'电感 L = {L*1e6:.2f} μH, Rs = {R:.2f} Ω'
        else:
            C = -1 / (2 * np.pi * f_test * X)
            return f'电容 C = {C*1e12:.2f} pF, Rs = {R:.2f} Ω'
```

```python
import numpy as np

def spectrum_analysis(signal_data, fs):
    """频谱分析"""
    N = len(signal_data)
    
    # 加窗 (Hanning)
    window = np.hanning(N)
    windowed = signal_data * window
    
    # FFT
    spectrum = np.fft.fft(windowed)
    freqs = np.fft.fftfreq(N, 1/fs)
    
    # 单边幅度谱
    magnitude = 2 * np.abs(spectrum[:N//2]) / np.sum(window)
    freqs = freqs[:N//2]
    
    # 功率谱 (dB)
    power_dB = 20 * np.log10(magnitude + 1e-30)
    
    return freqs, magnitude, power_dB
```

### 3.2 频率精确测量

```python
def precise_frequency_estimate(signal_data, fs, f_approx):
    """
    精确频率估计 - 相位差法 (Phase Difference Method)
    精度可达频率分辨率的 1/100
    """
    N = len(signal_data)
    n = np.arange(N)
    
    # 用近似频率做 DFT
    w_approx = 2 * np.pi * f_approx / fs
    I = 2 * np.mean(signal_data * np.cos(w_approx * n))
    Q = 2 * np.mean(signal_data * np.sin(w_approx * n))
    
    # 频率修正: 利用前后半段相位差
    half = N // 2
    phi1 = np.arctan2(
        np.mean(signal_data[:half] * np.sin(w_approx * n[:half])),
        np.mean(signal_data[:half] * np.cos(w_approx * n[:half]))
    )
    phi2 = np.arctan2(
        np.mean(signal_data[half:] * np.sin(w_approx * n[half:])),
        np.mean(signal_data[half:] * np.cos(w_approx * n[half:]))
    )
    
    delta_phi = phi2 - phi1
    if delta_phi > np.pi: delta_phi -= 2*np.pi
    if delta_phi < -np.pi: delta_phi += 2*np.pi
    
    # 频率修正量
    delta_f = delta_phi * fs / (2 * np.pi * half)
    
    f_precise = f_approx + delta_f
    amplitude = np.sqrt(I**2 + Q**2)
    
    return f_precise, amplitude
```

### 3.3 THD（总谐波失真）测量

```python
def measure_thd(signal_data, fs, f0, n_harmonics=10):
    """
    测量总谐波失真
    signal_data: 输入信号
    fs: 采样率
    f0: 基波频率
    n_harmonics: 谐波数量
    """
    N = len(signal_data)
    n = np.arange(N)
    
    # 测量基波
    w0 = 2 * np.pi * f0 / fs
    I0 = 2 * np.mean(signal_data * np.cos(w0 * n))
    Q0 = 2 * np.mean(signal_data * np.sin(w0 * n))
    V0 = np.sqrt(I0**2 + Q0**2)
    
    # 测量各次谐波
    harmonics = []
    thd_sum = 0
    for k in range(2, n_harmonics + 1):
        wk = 2 * np.pi * k * f0 / fs
        Ik = 2 * np.mean(signal_data * np.cos(wk * n))
        Qk = 2 * np.mean(signal_data * np.sin(wk * n))
        Vk = np.sqrt(Ik**2 + Qk**2)
        harmonics.append((k, Vk, Vk/V0 * 100))
        thd_sum += Vk**2
    
    thd = np.sqrt(thd_sum) / V0 * 100  # 百分比
    
    return {
        'fundamental': V0,
        'harmonics': harmonics,
        'thd_percent': thd
    }
```

### 4.1 频率测量方法

```python
def measure_frequency_zero_cross(signal_data, fs, threshold=0):
    """
    过零检测法测频率
    简单但对噪声敏感
    """
    # 去直流
    signal_ac = signal_data - np.mean(signal_data)
    
    # 找过零点 (上升沿)
    crossings = []
    for i in range(1, len(signal_ac)):
        if signal_ac[i-1] <= threshold and signal_ac[i] > threshold:
            # 线性插值精确定位
            frac = (threshold - signal_ac[i-1]) / (signal_ac[i] - signal_ac[i-1])
            t = (i - 1 + frac) / fs
            crossings.append(t)
    
    if len(crossings) < 2:
        return None
    
    # 计算周期
    periods = np.diff(crossings)
    T = np.mean(periods)
    f = 1 / T
    
    return f

def measure_frequency_dft(signal_data, fs):
    """DFT 法测频率 (精度更高)"""
    N = len(signal_data)
    
    # FFT
    spectrum = np.fft.fft(signal_data * np.hanning(N))
    freqs = np.fft.fftfreq(N, 1/fs)
    
    # 找峰值 (正频率部分)
    half = N // 2
    idx = np.argmax(np.abs(spectrum[1:half])) + 1  # 跳过 DC
    
    # 抛物线插值提高精度
    if 1 < idx < half - 1:
        alpha = np.abs(spectrum[idx - 1])
        beta = np.abs(spectrum[idx])
        gamma = np.abs(spectrum[idx + 1])
        p = 0.5 * (alpha - gamma) / (alpha - 2*beta + gamma)
        f = (idx + p) * fs / N
    else:
        f = freqs[idx]
    
    return abs(f)
```

### 4.2 相位差测量

```python
def measure_phase_difference(sig1, sig2, fs, f_test):
    """
    测量两个同频信号的相位差
    精度可达 0.1°
    """
    N = len(sig1)
    n = np.arange(N)
    w = 2 * np.pi * f_test / fs
    
    # DFT 提取相位
    I1 = 2 * np.mean(sig1 * np.cos(w * n))
    Q1 = 2 * np.mean(sig1 * np.sin(w * n))
    phi1 = np.arctan2(Q1, I1)
    
    I2 = 2 * np.mean(sig2 * np.cos(w * n))
    Q2 = 2 * np.mean(sig2 * np.sin(w * n))
    phi2 = np.arctan2(Q2, I2)
    
    phase_diff = phi2 - phi1  # 弧度
    phase_diff_deg = np.degrees(phase_diff)
    
    # 归一化到 [-180, 180]
    if phase_diff_deg > 180: phase_diff_deg -= 360
    if phase_diff_deg < -180: phase_diff_deg += 360
    
    return phase_diff_deg
```

### 4.3 占空比/脉宽测量

```python
def measure_duty_cycle(signal_data, fs, threshold_ratio=0.5):
    """测量数字信号占空比"""
    # 阈值
    vmax = np.max(signal_data)
    vmin = np.min(signal_data)
    threshold = vmin + threshold_ratio * (vmax - vmin)
    
    # 数字化
    digital = signal_data > threshold
    
    # 找上升沿和下降沿
    rising = np.where(np.diff(digital.astype(int)) == 1)[0]
    falling = np.where(np.diff(digital.astype(int)) == -1)[0]
    
    if len(rising) < 2 or len(falling) < 1:
        return None
    
    # 周期
    T = (rising[1] - rising[0]) / fs
    
    # 高电平时间
    if falling[0] > rising[0]:
        t_high = (falling[0] - rising[0]) / fs
    else:
        t_high = (falling[1] - rising[0]) / fs
    
    duty = t_high / T * 100
    freq = 1 / T
    
    return {'frequency': freq, 'duty_percent': duty, 'period': T}
```

## 5. 频率特性测量（波特图）

### 5.1 系统频率响应测量

```python
import numpy as np

def measure_frequency_response(input_data, output_data, fs, f_test):
    """
    测量系统在特定频率的频率响应 H(f)
    使用正弦激励法
    """
    N = len(input_data)
    n = np.arange(N)
    w = 2 * np.pi * f_test / fs
    
    # 输入信号基波
    I_in = 2 * np.mean(input_data * np.cos(w * n))
    Q_in = 2 * np.mean(input_data * np.sin(w * n))
    V_in = I_in - 1j * Q_in
    
    # 输出信号基波
    I_out = 2 * np.mean(output_data * np.cos(w * n))
    Q_out = 2 * np.mean(output_data * np.sin(w * n))
    V_out = I_out - 1j * Q_out
    
    # 频率响应
    H = V_out / V_in
    
    magnitude_dB = 20 * np.log10(abs(H))
    phase_deg = np.degrees(np.angle(H))
    
    return {'H': H, 'magnitude_dB': magnitude_dB, 'phase_deg': phase_deg}


def sweep_frequency_response(input_data, output_data, fs, f_start, f_stop, n_points=50):
    """
    扫频测量频率响应 (波特图)
    假设输入是扫频信号，需要同步标记
    """
    frequencies = np.logspace(np.log10(f_start), np.log10(f_stop), n_points)
    results = []
    
    # 对每个频率点，用窄带 DFT 测量
    N_seg = len(input_data) // n_points  # 每个频率点的样本数
    
    for i, f in enumerate(frequencies):
        start = i * N_seg
        end = start + N_seg
        if end > len(input_data):
            break
        
        result = measure_frequency_response(
            input_data[start:end],
            output_data[start:end],
            fs, f
        )
        results.append({
            'frequency': f,
            'magnitude_dB': result['magnitude_dB'],
            'phase_deg': result['phase_deg']
        })
    
    return results
```

### 5.2 电赛波特图测量方案

```
方案1: 逐点扫频法
  DDS信号源 → 系统输入 (频率可调正弦波)
  ADC同时采集输入和输出
  逐点计算幅度比和相位差
  优点: 精度高  缺点: 速度慢

方案2: 白噪声激励法
  白噪声 → 系统输入
  ADC采集输入和输出
  FFT计算互谱/自谱 → H(f) = Sxy / Sxx
  优点: 一次测量得到全频段  缺点: 精度略低

方案3: 啁啾信号 (Chirp) 法
  线性/对数扫频信号 → 系统输入
  ADC采集 → FFT分析
  优点: 速度快、信噪比高  缺点: 需要信号发生器支持
```

```python
def measure_transfer_function(vin, vout, fs, n_averages=1):
    """
    用白噪声激励测量传输函数
    vin: 输入信号
    vout: 输出信号
    """
    N = len(vin)
    
    # FFT
    Vin = np.fft.fft(vin)
    Vout = np.fft.fft(vout)
    
    # 互功率谱和自功率谱
    Sxy = Vout * np.conj(Vin)  # 互谱
    Sxx = Vin * np.conj(Vin)   # 自谱
    
    # 频率响应
    H = Sxy / (Sxx + 1e-30)
    
    freqs = np.fft.fftfreq(N, 1/fs)
    mag_dB = 20 * np.log10(np.abs(H[:N//2]) + 1e-30)
    phase = np.degrees(np.angle(H[:N//2]))
    
    return freqs[:N//2], mag_dB, phase

def measure_coherence(vin, vout):
    """
    相干函数: 评估测量质量
    γ² ≈ 1: 测量可靠
    γ² < 0.5: 存在噪声/非线性
    """
    Vin = np.fft.fft(vin)
    Vout = np.fft.fft(vout)
    
    Sxy = Vout * np.conj(Vin)
    Sxx = Vin * np.conj(Vin)
    Syy = Vout * np.conj(Vout)
    
    gamma_sq = np.abs(Sxy)**2 / (Sxx * Syy + 1e-30)
    return gamma_sq
```

### 7.1 过采样提高分辨率

```c
uint16_t oversample_16x(uint16_t *samples) {
    uint32_t sum = 0;
    for (int i = 0; i < 16; i++) {
        sum += samples[i];

    return (uint16_t)(sum >> 4);  // 右移4位 = 除以16
```

### 7.2 锁定放大器原理（微弱信号检测）

```
信号 → [前置放大] → [带通滤波] → [乘法器] → [低通滤波] → 输出
                                    ↑
                              参考信号 (同频同相)

原理: 
  输出 = ½ × A_signal × cos(φ) × G_LPF
  
  只有与参考信号同频的分量能通过
  噪声被低通滤波器抑制
  等效带宽极窄 → 信噪比极高
```

### 7.3 相关测量法（抑制非相干噪声）

```python
def correlated_measurement(signal_in, signal_ref, fs, f_test, n_periods=100):
    """
    相关法测量: 抑制与参考不同频的噪声
    """
    N = len(signal_in)
    n = np.arange(N)
    w = 2 * np.pi * f_test / fs
    
    # 相干检波
    I = 2 * np.mean(signal_in * np.cos(w * n))
    Q = 2 * np.mean(signal_in * np.sin(w * n))
    
    # 等效噪声带宽 = fs / (2N)
    # 采集 N 越大，等效带宽越窄，信噪比越高
    
    magnitude = np.sqrt(I**2 + Q**2)
    phase = np.degrees(np.arctan2(Q, I))
    
    return magnitude, phase
```

### 7.4 自动量程切换

```c
// 自动量程切换逻辑
typedef struct {

} range_t;

#define NUM_RANGES 5
range_t ranges[NUM_RANGES] = {

    {0,    0.01,  100,  0},   // 10mV 量程
    {0,    0.1,   10,   0},   // 100mV 量程
    {0,    1.0,   1,    0},   // 1V 量程
    {0,    10.0,  0.1,  0},   // 10V 量程
};

int current_range = 3;  // 默认 1V 量程

float auto_range_read(float raw_adc) {
    float value = raw_adc * ranges[current_range].gain;
    
    // 过载: 降低增益
    if (raw_adc > ranges[current_range].range_max * 0.9) {
        if (current_range < NUM_RANGES - 1) {
            current_range++;
            set_pga(ranges[current_range].gain);
            return -1;  // 需要重新测量
        }
    }
    // 信号太小: 提高增益
    else if (raw_adc < ranges[current_range].range_max * 0.1) {
        if (current_range > 0) {
            current_range--;
            set_pga(ranges[current_range].gain);
            return -1;
        }
    }
    
    return value;
}
```

## 8. 系统校准

### 8.1 两点校准

```c
// 两点校准: 消除增益误差和偏移误差
typedef struct {
    float gain;
    float offset;
} calibration_t;

calibration_t cal;

void calibrate(float ref_low, float meas_low, float ref_high, float meas_high) {
    // gain = (ref_high - ref_low) / (meas_high - meas_low)
    cal.gain = (ref_high - ref_low) / (meas_high - meas_low);
    // offset = ref_low - gain * meas_low
    cal.offset = ref_low - cal.gain * meas_low;
}

float apply_calibration(float raw) {
    return cal.gain * raw + cal.offset;
}
```

### 8.2 多点校准（分段线性）

```python
def piecewise_calibrate(raw_values, ref_values, raw_measured):
    """
    多点分段线性校准
    raw_values: 校准点的测量值
    ref_values: 校准点的参考值
    raw_measured: 待校准的测量值
    """
    # 线性插值
    return np.interp(raw_measured, raw_values, ref_values)
```

```c
// ADC DMA 采样 + FFT 分析
#include "arm_math.h"  // CMSIS-DSP

#define FFT_SIZE  1024
#define SAMPLE_FS 100000  // 100kHz

float32_t adc_buffer[FFT_SIZE];
float32_t fft_output[FFT_SIZE];
float32_t fft_magnitude[FFT_SIZE/2];

arm_rfft_fast_instance_f32 fft_instance;

void measurement_init(void) {
    arm_rfft_fast_init_f32(&fft_instance, FFT_SIZE);
    ADC_DMA_Init(SAMPLE_FS);  // 初始化 ADC DMA
}

void measurement_process(void) {
    // 等待 DMA 完成
    while (!dma_complete);
    
    // 去直流
    float32_t mean;
    arm_mean_f32(adc_buffer, FFT_SIZE, &mean);
    arm_offset_f32(adc_buffer, -mean, adc_buffer, FFT_SIZE);
    
    // 加窗
    float32_t window[FFT_SIZE];
    arm_hanning_f32(window, FFT_SIZE);
    arm_mult_f32(adc_buffer, window, adc_buffer, FFT_SIZE);
    
    // FFT
    arm_rfft_fast_f32(&fft_instance, adc_buffer, fft_output, 0);
    
    // 幅度谱
    arm_cmplx_mag_f32(fft_output, fft_magnitude, FFT_SIZE/2);
    
    // 找基波频率
    uint32_t peak_idx;
    float32_t peak_val;
    arm_max_f32(fft_magnitude + 1, FFT_SIZE/2 - 1, &peak_val, &peak_idx);
    peak_idx += 1;  // 补偿跳过 DC 的偏移
    
    float f0 = (float)peak_idx * SAMPLE_FS / FFT_SIZE;
    float amplitude = peak_val * 2.0f / FFT_SIZE;
    
    // THD 计算
    float thd_sq = 0;
    float fundamental = fft_magnitude[peak_idx];
    for (int k = 2; k <= 10; k++) {
        int idx = k * peak_idx;
        if (idx < FFT_SIZE/2) {
            float hv = fft_magnitude[idx];
            thd_sq += hv * hv;
        }
    }
    float thd = sqrtf(thd_sq) / fundamental * 100.0f;
    
    // 显示结果
    LCD_Printf("f0 = %.2f Hz\n", f0);
    LCD_Printf("Amplitude = %.4f V\n", amplitude);
    LCD_Printf("THD = %.2f %%\n", thd);
}
```

## 10. 误差分析与提高精度

### 10.1 误差来源

| 误差源 | 典型值 | 减小方法 |
|--------|-------|---------|
| ADC 量化 | ±0.5 LSB | 过采样、提高分辨率 |
| ADC 非线性 | ±1~2 LSB | 校准、选择好 ADC |
| 参考电压 | ±0.1~1% | 精密基准源 |
| 温度漂移 | ±50~200ppm/°C | 温度补偿、选择低温漂器件 |
| 噪声 | 取决于带宽 | 滤波、过采样、屏蔽 |
| 采样时钟抖动 | ps~ns 级 | 精密晶振 |
| 信号调理误差 | ±0.1~1% | 精密电阻、校准 |

### 10.2 精度提升技巧

```
1. 过采样 + 平均: 每提高1位分辨率需要4倍过采样
2. 数字滤波: IIR/FIR 降低带内噪声
3. 相干采样: 采样窗口包含整数个信号周期 → 消除频谱泄漏
4. 自校准: 开机/定期用已知参考校准
5. 温度补偿: 记录温度查表补偿
6. 屏蔽/接地: 减小外部干扰
```
