## 二、ADC采样技术

### 2.1 采样定理与实际选择

```
奈奎斯特：fs ≥ 2 × f_max

实际工程：fs ≥ (5 ~ 10) × f_max
原因：抗混叠滤波器不是理想的
```

### 2.2 STM32 ADC配置

```c
/* 单通道高速采样（DMA方式） */
#define ADC_BUF_SIZE 1024
volatile uint16_t adc_buf[ADC_BUF_SIZE];

void ADC_DMA_Init(void) {
    // ADC时钟: PCLK2/6 = 12MHz, 采样时间: 1.5周期
    // 采样率 = 12MHz / (1.5 + 12.5) = 857kSPS
    
    RCC_ADCCLKConfig(RCC_PCLK2_Div6);
    
    ADC_InitTypeDef adc;
    adc.ADC_Mode = ADC_Mode_Independent;
    adc.ADC_ScanConvMode = DISABLE;
    adc.ADC_ContinuousConvMode = ENABLE;
    adc.ADC_ExternalTrigConv = ADC_ExternalTrigConv_None;
    adc.ADC_DataAlign = ADC_DataAlign_Right;
    adc.ADC_NbrOfChannel = 1;
    ADC_Init(ADC1, &adc);
    
    ADC_RegularChannelConfig(ADC1, ADC_Channel_0, 1, ADC_SampleTime_1Cycles5);
    ADC_DMACmd(ADC1, ENABLE);
    ADC_Cmd(ADC1, ENABLE);
    
    // DMA配置
    DMA_InitTypeDef dma;
    dma.DMA_PeripheralBaseAddr = (uint32_t)&ADC1->DR;
    dma.DMA_MemoryBaseAddr = (uint32_t)adc_buf;
    dma.DMA_DIR = DMA_DIR_PeripheralSRC;
    dma.DMA_BufferSize = ADC_BUF_SIZE;
    dma.DMA_PeripheralInc = DMA_PeripheralInc_Disable;
    dma.DMA_MemoryInc = DMA_MemoryInc_Enable;
    dma.DMA_PeripheralDataSize = DMA_PeripheralDataSize_HalfWord;
    dma.DMA_MemoryDataSize = DMA_MemoryDataSize_HalfWord;
    dma.DMA_Mode = DMA_Mode_Circular;
    dma.DMA_Priority = DMA_Priority_High;
    DMA_Init(DMA1_Channel1, &dma);
    DMA_Cmd(DMA1_Channel1, ENABLE);
    
    ADC_SoftwareStartConvCmd(ADC1, ENABLE);
}
```

### 2.4 过采样提高分辨率

```c
/* 过采样：4倍过采样 = 提高1位分辨率 */
uint16_t ADC_Oversample(void) {
    uint32_t sum = 0;
    for (int i = 0; i < 16; i++) {  // 16倍过采样 = +2位
        sum += ADC_SingleRead();
    }
    return (uint16_t)(sum >> 4);  // 右移4位（÷16）
    // 12位ADC → 14位有效分辨率
}
```

### 2.5 信号调理电路

**运放选择要点：**
```
- 带宽：GBW ≥ 10 × 信号最高频率
- 压摆率：SR ≥ 2π × f × Vpeak
- 输入偏置电流：高阻源选FET输入
- 噪声：< 10nV/√Hz（低噪声应用）
```

**常用运放推荐：**
| 型号 | GBW | SR | 噪声 | 用途 |
|------|-----|-----|------|------|
| OPA2134 | 8MHz | 20V/μs | 8nV | 通用音频 |
| OPA2350 | 38MHz | 22V/μs | 7nV | 高速信号 |
| AD8065 | 145MHz | 180V/μs | 7nV | 高速采集 |
| OPA1612 | 40MHz | 27V/μs | 1.1nV | 超低噪声 |

## 三、FFT频谱分析

### 3.1 FFT基础

```
N点FFT → N/2个频率点
频率分辨率：Δf = fs / N
最高分析频率：f_max = fs / 2
```

### 3.2 定点FFT实现（适合STM32）

```c
/* 基2-FFT，定点Q15格式 */
#include <stdint.h>
#include <math.h>

#define FFT_N 1024
#define LOG2_N 10

// 旋转因子（预计算）
int16_t cos_table[FFT_N/2];
int16_t sin_table[FFT_N/2];

void FFT_Init(void) {
    for (int i = 0; i < FFT_N/2; i++) {
        cos_table[i] = (int16_t)(32767 * cosf(2.0f * 3.14159f * i / FFT_N));
        sin_table[i] = (int16_t)(-32767 * sinf(2.0f * 3.14159f * i / FFT_N));
    }
}

void FFT_Radix2(int16_t *real, int16_t *imag, int n) {
    // 位反转排列
    for (int i = 1, j = 0; i < n; i++) {
        int bit = n >> 1;
        for (; j & bit; bit >>= 1) j ^= bit;
        j ^= bit;
        if (i < j) {
            int16_t tmp;
            tmp = real[i]; real[i] = real[j]; real[j] = tmp;
            tmp = imag[i]; imag[i] = imag[j]; imag[j] = tmp;
        }
    }
    
    // 蝶形运算
    for (int stage = 1; stage <= LOG2_N; stage++) {
        int m = 1 << stage;
        int half_m = m >> 1;
        int step = FFT_N / m;
        
        for (int k = 0; k < n; k += m) {
            for (int j = 0; j < half_m; j++) {
                int idx = j * step;
                int16_t wr = cos_table[idx];
                int16_t wi = sin_table[idx];
                
                int p = k + j;
                int q = k + j + half_m;
                
                // 复数乘法 Q15
                int32_t tr = ((int32_t)real[q] * wr - (int32_t)imag[q] * wi) >> 15;
                int32_t ti = ((int32_t)real[q] * wi + (int32_t)imag[q] * wr) >> 15;
                
                real[q] = real[p] - (int16_t)tr;
                imag[q] = imag[p] - (int16_t)ti;
                real[p] = real[p] + (int16_t)tr;
                imag[p] = imag[p] + (int16_t)ti;
            }
        }
    }
}

/* 计算幅度谱 */
void FFT_Magnitude(int16_t *real, int16_t *imag, uint16_t *mag, int n) {
    for (int i = 0; i < n/2; i++) {
        int32_t r = real[i];
        int32_t m = imag[i];
        mag[i] = (uint16_t)sqrtf(r*r + m*m);
    }
}
```

### 3.3 浮点FFT（精度更高）

```c
#include "arm_math.h"  // CMSIS-DSP

float fft_input[FFT_N * 2];  // 交错复数格式
float fft_output[FFT_N];

arm_cfft_radix4_instance_f32 fft_inst;

void FFT_Float_Init(void) {
    arm_cfft_radix4_init_f32(&fft_inst, FFT_N, 0, 1);
}

void FFT_Float_Run(uint16_t *adc_data) {
    // 填充输入（实部=ADC数据，虚部=0）
    for (int i = 0; i < FFT_N; i++) {
        fft_input[2*i] = (float)adc_data[i] - 2048.0f;  // 去直流
        fft_input[2*i+1] = 0;
    }
    
    arm_cfft_radix4_f32(&fft_inst, fft_input);
    arm_cmplx_mag_f32(fft_input, fft_output, FFT_N);
    
    // fft_output[0]=直流, fft_output[1~N/2-1]=各频率分量
}
```

### 3.4 频率测量

```c
/* 精确频率测量：FFT粗测 + 插值精测 */
float Measure_Frequency(uint16_t *adc_data, float fs) {
    // 1. FFT
    FFT_Float_Run(adc_data);
    
    // 2. 找峰值（排除直流）
    int peak_idx = 1;
    float peak_val = fft_output[1];
    for (int i = 2; i < FFT_N/2; i++) {
        if (fft_output[i] > peak_val) {
            peak_val = fft_output[i];
            peak_idx = i;
        }
    }
    
    // 3. 抛物线插值精测
    float alpha = fft_output[peak_idx - 1];
    float beta = fft_output[peak_idx];
    float gamma = fft_output[peak_idx + 1];
    float delta = 0.5f * (alpha - gamma) / (alpha - 2*beta + gamma);
    
    float freq = (peak_idx + delta) * fs / FFT_N;
    return freq;
}
```

### 3.5 THD（总谐波失真）测量

```c
float Measure_THD(float *mag, float fs) {
    // 找基波频率
    int fund = 1;
    for (int i = 2; i < FFT_N/2; i++) {
        if (mag[i] > mag[fund]) fund = i;
    }
    
    float fund_amp = mag[fund];
    float harm_sum = 0;
    
    // 计算2~10次谐波
    for (int h = 2; h <= 10; h++) {
        int idx = fund * h;
        if (idx < FFT_N/2) {
            harm_sum += mag[idx] * mag[idx];
        }
    }
    
    float thd = sqrtf(harm_sum) / fund_amp * 100.0f;
    return thd;
}
```

## 四、数字滤波器

### 4.1 IIR滤波器

```c
/* 二阶IIR（Biquad） */
typedef struct {
    float b0, b1, b2;  // 分子系数
    float a1, a2;       // 分母系数（a0=1）
    float x1, x2;       // 输入历史
    float y1, y2;       // 输出历史
} Biquad;

float Biquad_Run(Biquad *f, float x) {
    float y = f->b0 * x + f->b1 * f->x1 + f->b2 * f->x2
             - f->a1 * f->y1 - f->a2 * f->y2;
    f->x2 = f->x1; f->x1 = x;
    f->y2 = f->y1; f->y1 = y;
    return y;
}

/* 级联二阶节实现高阶滤波器 */
#define NUM_SECTIONS 3
Biquad filter[NUM_SECTIONS];

float Filter_Run(float x) {
    for (int i = 0; i < NUM_SECTIONS; i++) {
        x = Biquad_Run(&filter[i], x);
    }
    return x;
}
```

### 4.2 滤波器系数计算

```c
/* 巴特沃斯低通滤波器系数生成 */
void Butterworth_LPF(float fc, float fs, Biquad *f) {
    float wc = tanf(3.14159f * fc / fs);
    float wc2 = wc * wc;
    float sqrt2 = 1.41421356f;
    float norm = 1.0f / (1.0f + sqrt2 * wc + wc2);
    
    f->b0 = wc2 * norm;
    f->b1 = 2.0f * f->b0;
    f->b2 = f->b0;
    f->a1 = 2.0f * (wc2 - 1.0f) * norm;
    f->a2 = (1.0f - sqrt2 * wc + wc2) * norm;
}
```

### 4.3 FIR滤波器

```c
/* FIR滤波器（线性相位） */
#define FIR_N 32
float fir_coeff[FIR_N];
float fir_buf[FIR_N];
int fir_idx = 0;

float FIR_Run(float x) {
    fir_buf[fir_idx] = x;
    
    float y = 0;
    int idx = fir_idx;
    for (int i = 0; i < FIR_N; i++) {
        y += fir_coeff[i] * fir_buf[idx];
        idx = (idx == 0) ? FIR_N-1 : idx-1;
    }
    
    fir_idx = (fir_idx + 1) % FIR_N;
    return y;
}

/* 用窗函数法设计FIR系数 */
void Design_FIR_LPF(float fc, float fs, int n, float *coeff) {
    float fc_norm = fc / fs;
    int mid = n / 2;
    
    for (int i = 0; i < n; i++) {
        if (i == mid) {
            coeff[i] = 2.0f * fc_norm;
        } else {
            float x = 3.14159f * (i - mid);
            coeff[i] = sinf(2.0f * fc_norm * x) / x;
        }
        // 汉宁窗
        coeff[i] *= 0.5f * (1.0f - cosf(2.0f * 3.14159f * i / (n-1)));
    }
}
```

### 4.4 滤波器选择指南

| 滤波器 | 相位 | 阶数 | 延迟 | 适用 |
|--------|------|------|------|------|
| IIR Butterworth | 非线性 | 低 | 小 | 实时控制 |
| IIR Chebyshev | 非线性 | 更低 | 小 | 高选择性 |
| FIR | **线性** | 高 | 大 | 音频、测量 |
| 滑动平均 | 线性 | - | 中 | 简单降噪 |
| 中值 | - | - | 中 | 去脉冲噪声 |

## 五、波形生成（DDS）

### 5.1 DDS原理

```
输出频率 = 基准时钟 × 相位增量 / 2^32

相位累加器：每个时钟加一次相位增量
查表：相位 → 幅度
DAC：幅度 → 模拟电压
```

### 5.2 STM32 DDS实现

```c
/* 32位DDS */
#define DDS_CLOCK 72000000   // 72MHz
#define SINE_POINTS 256

const uint16_t sine_lut[SINE_POINTS] = {
    2048, 2098, 2148, 2198, 2248, 2298, 2347, 2397,
    // ... 完整256点正弦表
};

volatile uint32_t phase_acc = 0;
volatile uint32_t phase_inc = 0;

void DDS_SetFreq(float freq) {
    phase_inc = (uint32_t)((double)freq * 4294967296.0 / DDS_CLOCK);
}

/* 定时器中断（DDS时钟） */
void TIM6_IRQHandler(void) {
    phase_acc += phase_inc;
    uint8_t idx = phase_acc >> 24;  // 取高8位作为索引
    DAC_SetValue(sine_lut[idx]);
}
```

### 5.3 多波形DDS

```c
typedef enum { SINE, SQUARE, TRIANGLE, SAW } WaveType;

uint16_t DDS_GetSample(WaveType type) {
    uint8_t idx = phase_acc >> 24;
    
    switch (type) {
    case SINE:
        return sine_lut[idx];
    case SQUARE:
        return (idx < 128) ? 4095 : 0;
    case TRIANGLE:
        return (idx < 128) ? idx * 32 : (255 - idx) * 32;
    case SAW:
        return idx * 16;
    }
    return 0;
}
```

## 六、信号测量技术

### 6.1 频率测量

```c
/* 方法1：输入捕获（精度高） */
volatile uint32_t capture_val = 0;
volatile uint32_t period = 0;

void TIM3_IRQHandler(void) {
    if (TIM_GetITStatus(TIM3, TIM_IT_CC1)) {
        static uint32_t last = 0;
        uint32_t now = TIM_GetCapture1(TIM3);
        period = now - last;
        last = now;
    }
}

float GetFrequency(void) {
    if (period > 0)
        return 72000000.0f / (float)period;
    return 0;
}

/* 方法2：计数法（低频） */
// 在固定时间（如1s）内计脉冲数
float Freq_By_Counting(void) {
    uint32_t count = TIM_GetCounter(TIM2);
    TIM_SetCounter(TIM2, 0);
    return (float)count;  // 1s内的脉冲数 = 频率
}
```

### 6.2 相位差测量

```c
/* 双通道ADC采样 → FFT → 相位差 */
float Measure_Phase(uint16_t *ch1, uint16_t *ch2, float fs) {
    // 对两个通道分别做FFT
    float real1[FFT_N], imag1[FFT_N];
    float real2[FFT_N], imag2[FFT_N];
    
    FFT_Run(ch1, real1, imag1);
    FFT_Run(ch2, real2, imag2);
    
    // 找基波索引
    int fund = FindFundamental(ch1, fs);
    
    // 计算相位差
    float phase1 = atan2f(imag1[fund], real1[fund]);
    float phase2 = atan2f(imag2[fund], real2[fund]);
    float phase_diff = (phase2 - phase1) * 57.2958f;  // 转度
    
    // 归一化到 ±180°
    while (phase_diff > 180) phase_diff -= 360;
    while (phase_diff < -180) phase_diff += 360;
    
    return phase_diff;
}
```

### 6.3 有效值（RMS）测量

```c
float Measure_RMS(uint16_t *adc_data, int n, float vref) {
    float sum = 0;
    for (int i = 0; i < n; i++) {
        float v = (float)adc_data[i] / 4096.0f * vref;
        sum += v * v;
    }
    return sqrtf(sum / n);
}
```

## 七、常见问题与解决

### Q1：FFT频谱泄漏

```
原因：采样不是信号周期的整数倍

解决：
1. 加窗（汉宁窗、海明窗）
2. 整周期采样（同步采样）
3. 增加采样点数
```

```c
// 加汉宁窗
for (int i = 0; i < FFT_N; i++) {
    float w = 0.5f * (1.0f - cosf(2*3.14159f*i/(FFT_N-1)));
    fft_input[2*i] *= w;
}
```

### Q2：ADC采样有噪声

```
解决：
1. 硬件：加RC低通滤波（fc=信号最高频率的5倍）
2. 软件：过采样 + 数字滤波
3. 布局：模拟地与数字地单点连接
4. 电源：ADC参考电压加去耦电容
```

### Q3：DAC输出有毛刺

```
解决：
1. 后级加RC低通滤波
2. 使用带缓冲的DAC
3. DMA传输避免CPU中断延迟
```
