# 进程间通信(IPC)设计

## 1. 技术方案选择

| 方案 | 延迟 | 复杂度 | 适用场景 |
|------|------|--------|----------|
| **POSIX共享内存** | **<0.1ms** | 低 | ✅ 选定 |
| Unix Domain Socket | ~0.5ms | 中 | 备选 |
| 管道 Pipe | ~0.3ms | 低 | 单向少量数据 |
| D-Bus | ~2ms | 高 | ❌ 不适用 |
| TCP Socket | ~1ms | 中 | ❌ 过重 |

**最终方案：POSIX共享内存 + 命名信号量同步**

## 2. 共享内存布局

### 2.1 全局内存段划分

```
/dev/shm/
├── vision_result     # 视觉结果 (1KB)
├── control_command   # 控制指令 (512B)
├── system_status     # 系统状态 (2KB)
├── hmi_params        # 人机参数 (4KB)
└── debug_data        # 调试数据 (8KB)
```

### 2.2 vision_result_t 结构 (视觉→控制)

```c
typedef struct {
    uint64_t timestamp_ns;      // 时间戳 (纳秒)
    uint32_t frame_id;          // 帧序号
    int32_t  target_x;          // 目标X坐标 (0.1mm)
    int32_t  target_y;          // 目标Y坐标 (0.1mm)
    int32_t  target_z;          // 目标Z坐标 (0.1mm)
    float    confidence;        // 置信度 0~1
    uint8_t  target_type;       // 目标类型
    uint8_t  detection_valid;   // 检测有效标志
    uint8_t  reserved[2];       // 对齐填充
} vision_result_t;              // 总计 32 字节
```

### 2.3 system_status_t 结构 (状态共享)


## 3. 同步机制：命名信号量

### 3.1 信号量命名

```
/sem_vision_ready    # 视觉结果就绪 (视觉写→控制读)
/sem_params_updated  # 参数已更新 (HMI写→控制/视觉读)
/sem_status_updated  # 状态已更新 (控制写→HMI读)
```

### 3.2 同步流程

```
视觉进程                        控制进程
   │                               │
   ├── 处理图像                     │
   ├── 写入 vision_result          │
   ├── sem_post(/sem_vision_ready) │
   │                               ├── sem_wait(/sem_vision_ready)
   │                               ├── 读取 vision_result
   │                               ├── PID计算
   │                               ├── 写入 system_status
   │                               └── sem_post(/sem_status_updated)
   │
```

### 3.3 非阻塞读取 (控制进程关键)

```c
// 控制进程使用非阻塞等待，超时则使用上一次数据
struct timespec ts;
clock_gettime(CLOCK_REALTIME, &ts);
ts.tv_nsec += 500000;  // 0.5ms超时

if (sem_timedwait(sem, &ts) == 0) {
    // 新数据到达，读取
    memcpy(&local, shm_vision, sizeof(vision_result_t));
} else {
    // 超时，使用上一次数据 (控制不中断)
}
```

## 4. Python接口封装

```python
import mmap
import posix_ipc
import struct
import time

class SharedMemoryIPC:
    """POSIX共享内存IPC封装"""

    # 结构体格式定义
    VISION_FMT = '=QIiii fBB2x'    # 32 bytes
    STATUS_FMT  = '=QIBhhhh fBBH12x'  # 48 bytes (简化)

    def __init__(self):
        # 创建共享内存段
        self.shm_vision = posix_ipc.SharedMemory(
            '/vision_result', posix_ipc.O_CREAT, size=4096)
        self.shm_status = posix_ipc.SharedMemory(
            '/system_status', posix_ipc.O_CREAT, size=4096)

        # 创建信号量
        self.sem_vision = posix_ipc.Semaphore(
            '/sem_vision_ready', posix_ipc.O_CREAT, initial_value=0)
        self.sem_status = posix_ipc.Semaphore(
            '/sem_status_updated', posix_ipc.O_CREAT, initial_value=0)

        # mmap映射
        self.vision_map = mmap.mmap(self.shm_vision.fd, 4096)
        self.status_map = mmap.mmap(self.shm_status.fd, 4096)

    def write_vision_result(self, target_x, target_y, target_z,
                            confidence, target_type=0):
        """写入视觉结果"""
        data = struct.pack(
            self.VISION_FMT,
            time.monotonic_ns(), 0,
            target_x, target_y, target_z,
            confidence, target_type, 1)
        self.vision_map.seek(0)
        self.vision_map.write(data)
        self.sem_vision.release()  # 通知控制进程

    def read_vision_result(self, timeout=0.001):
        """读取视觉结果 (阻塞等待)"""
        self.sem_vision.acquire(timeout)
        self.vision_map.seek(0)
        data = self.vision_map.read(32)
        return struct.unpack(self.VISION_FMT, data)

    def cleanup(self):
        """清理资源"""
        self.vision_map.close()
        self.status_map.close()
        posix_ipc.unlink_shared_memory('/vision_result')
        posix_ipc.unlink_shared_memory('/system_status')
        self.sem_vision.unlink()
        self.sem_status.unlink()
```

## 5. 延迟实测方法

```bash
# 运行IPC延迟测试
python3 -c "
import time, mmap, posix_ipc, struct, os

shm = posix_ipc.SharedMemory('/ipc_test', posix_ipc.O_CREAT, size=64)
sem_w = posix_ipc.Semaphore('/sem_w', posix_ipc.O_CREAT, initial_value=0)
sem_r = posix_ipc.Semaphore('/sem_r', posix_ipc.O_CREAT, initial_value=0)
mm = mmap.mmap(shm.fd, 64)

pid = os.fork()
if pid == 0:
    # Writer
    for i in range(10000):
        t0 = time.monotonic_ns()
        mm.seek(0)
        mm.write(struct.pack('=Q', t0))
        sem_w.release()
        sem_r.acquire()
        t1 = time.monotonic_ns()
        print(f'Round-trip: {(t1-t0)/1000:.1f} us')
else:
    # Reader (echo)
    for i in range(10000):
        sem_w.acquire()
        mm.seek(0)
        data = mm.read(8)
        sem_r.release()
"
```

## 6. 容错设计

| 场景 | 处理方式 |
|------|----------|
| 视觉进程崩溃 | 控制进程超时后使用最后有效数据，3次后报错 |
| 控制进程崩溃 | watchdog检测，自动重启 |
| 共享内存损坏 | CRC校验 + 时间戳检查 |
| 信号量死锁 | 超时释放 + watchdog重置 |

## 7. 注意事项

1. **内存对齐**：结构体使用 `__attribute__((packed))` 或手动填充
2. **字节序**：全部使用主机字节序 (ARM小端)
3. **原子性**：单次读写 < cache line (64B) 保证原子性
4. **cache一致性**：RK3588S多核共享L3，无需手动flush
5. **启动顺序**：人机交互 → 视觉进程 → 控制进程
