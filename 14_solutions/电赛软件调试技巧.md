### 2. 分级日志系统

```c
// 日志等级定义
#define LOG_LEVEL_NONE  0
#define LOG_LEVEL_ERROR 1
#define LOG_LEVEL_WARN  2
#define LOG_LEVEL_INFO  3
#define LOG_LEVEL_DEBUG 4

#define CURRENT_LOG_LEVEL LOG_LEVEL_INFO  // 发布时改为WARN或ERROR

#define LOG_ERROR(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_ERROR) \
        printf("[ERROR][%s:%d] " fmt "\r\n", __FILE__, __LINE__, ##__VA_ARGS__); \
    } while(0)

#define LOG_WARN(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_WARN) \
        printf("[WARN] " fmt "\r\n", ##__VA_ARGS__); \
    } while(0)

#define LOG_INFO(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_INFO) \
        printf("[INFO] " fmt "\r\n", ##__VA_ARGS__); \
    } while(0)

#define LOG_DEBUG(fmt, ...) \
    do { if (CURRENT_LOG_LEVEL >= LOG_LEVEL_DEBUG) \
        printf("[DEBUG] " fmt "\r\n", ##__VA_ARGS__); \
    } while(0)

// 使用
LOG_INFO("系统启动完成");
LOG_DEBUG("PID输出: Kp=%.2f, Ki=%.2f, Kd=%.2f", kp, ki, kd);
LOG_ERROR("传感器读取失败, err=%d", err);
```

### 3. 带时间戳的调试输出

```c
// 使用SysTick作为时间戳
uint32_t get_tick_ms(void) {
    return HAL_GetTick();
}

#define LOG_TS(fmt, ...) \
    printf("[%lu.%03lu] " fmt "\r\n", \
           get_tick_ms()/1000, get_tick_ms()%1000, ##__VA_ARGS__)

// 使用
LOG_TS("ADC采样完成, 值=%d", adc_val);
// 输出: [12.345] ADC采样完成, 值=2048
```

### 5. printf效率优化

```c
// 问题：printf通过UART发送太慢，影响实时性
// 方案1：DMA发送
char log_buf[256];
int len = snprintf(log_buf, sizeof(log_buf), "val=%d\r\n", val);
HAL_UART_Transmit_DMA(&huart1, (uint8_t*)log_buf, len);

// 方案2：缓冲区异步发送
#define LOG_BUF_SIZE 4096
static char log_buffer[LOG_BUF_SIZE];
static volatile uint16_t log_head = 0;
static volatile uint16_t log_tail = 0;

void log_put(char c) {
    uint16_t next = (log_head + 1) % LOG_BUF_SIZE;
    if (next != log_tail) {
        log_buffer[log_head] = c;
        log_head = next;
    }
}

// 在空闲中断中发送缓冲区数据
```

## 二、断点调试技巧

### 1. Keil MDK调试

```
常用快捷键：
F5        - 运行/停止
F10       - 单步跳过（Step Over）
F11       - 单步进入（Step Into）
F12       - 跳转到光标处
Ctrl+F10  - 运行到光标处

断点类型：
- 普通断点：F9设置，程序执行到此处暂停
- 条件断点：右键断点→设置条件（如 count == 100）
- 数据断点：当某变量值改变时暂停（Watch窗口设置）
```

### 2. 条件断点实战

```
场景：循环1000次中第857次出错

方法1：条件断点
  在循环内设断点，条件设为 i == 857

方法2：计数断点
  右键断点→设置pass count为857
  
方法3：代码辅助
  if (i == 857) {
      __NOP();  // 在这里设断点
  }
```

### 3. 实时变量观察

```
Watch窗口技巧：
1. 添加变量名直接观察
2. 支持表达式：a*b + c
3. 支持类型转换：(float)raw_adc * 3.3f / 4096
4. 格式切换：Hex/Dec/Float/Binary

Logic Analyzer（Keil内置）：
1. Setup → 添加变量
2. 设置采样周期
3. 运行后可看变量波形
4. 适合观察PID输出、ADC值变化趋势
```

### 4. STM32CubeIDE调试

```
调试配置：
1. Run → Debug As → STM32 C/C++ Application
2. 配置SWD/JTAG调试器
3. 设置断点后Step调试

Live Expressions（实时观察）：
- 不暂停CPU也能看变量值
- 适合观察正在运行的系统状态
- 刷新率约1-5Hz

SWV（Serial Wire Viewer）：
- 使用ITM通道输出printf
- 不占用UART资源
- 支持实时波形显示（Profiler）
```

### 5. 常见调试陷阱

```
1. 优化导致变量不可见
   解决：变量加volatile，或调试时用-O0编译

2. 断点影响时序
   解决：用数据断点代替代码断点
   或使用SWV/ETM非侵入式追踪

3. 中断中调试困难
   解决：在中断中设标志位，主循环中处理
   或用Segger SystemView可视化中断

4. 优化后行为不一致
   解决：先用Debug配置调试，确认逻辑后切Release
```

### 2. 堆内存泄漏检测

```c
// 简易内存跟踪器
typedef struct {
    void *ptr;
    size_t size;
    const char *file;
    int line;
} mem_record_t;

#define MAX_RECORDS 100
static mem_record_t records[MAX_RECORDS];
static int record_count = 0;

void* debug_malloc(size_t size, const char *file, int line) {
    void *ptr = malloc(size);
    if (ptr && record_count < MAX_RECORDS) {
        records[record_count].ptr = ptr;
        records[record_count].size = size;
        records[record_count].file = file;
        records[record_count].line = line;
        record_count++;
    }
    printf("[MALLOC] %s:%d size=%d ptr=%p\r\n", file, line, size, ptr);
    return ptr;
}

void debug_free(void *ptr, const char *file, int line) {
    for (int i = 0; i < record_count; i++) {
        if (records[i].ptr == ptr) {
            printf("[FREE] %s:%d ptr=%p size=%d\r\n", file, line, ptr, records[i].size);
            records[i] = records[--record_count];
            break;
        }
    }
    free(ptr);
}

#define malloc(s) debug_malloc(s, __FILE__, __LINE__)
#define free(p)   debug_free(p, __FILE__, __LINE__)

// 打印未释放的内存
void print_leaks(void) {
    printf("=== Memory Leaks ===\r\n");
    for (int i = 0; i < record_count; i++) {
        printf("  LEAK: %s:%d ptr=%p size=%d\r\n",
               records[i].file, records[i].line,
               records[i].ptr, records[i].size);
    }
}
```

### 3. HardFault调试

```c
// HardFault信息打印
void HardFault_Handler(void) {
    __asm volatile (
        "TST LR, #4       \n"
        "ITE EQ           \n"
        "MRSEQ R0, MSP    \n"
        "MRSNE R0, PSP    \n"
        "B hard_fault_handler_c \n"
    );
}

void hard_fault_handler_c(uint32_t *stack) {
    printf("=== HARD FAULT ===\r\n");
    printf("R0  = 0x%08X\r\n", stack[0]);
    printf("R1  = 0x%08X\r\n", stack[1]);
    printf("R2  = 0x%08X\r\n", stack[2]);
    printf("R3  = 0x%08X\r\n", stack[3]);
    printf("R12 = 0x%08X\r\n", stack[4]);
```

## 五、实用调试宏

```c
// 断言（失败时打印位置信息）
#define ASSERT(expr) \
    do { \
        if (!(expr)) { \
            printf("ASSERT FAILED: %s at %s:%d\r\n", \
                   #expr, __FILE__, __LINE__); \
            while(1); \
        } \
    } while(0)

// 调试用LED指示
#define DBG_LED_ON(pin)   HAL_GPIO_WritePin(GPIOC, pin, GPIO_PIN_SET)
#define DBG_LED_OFF(pin)  HAL_GPIO_WritePin(GPIOC, pin, GPIO_PIN_RESET)
#define DBG_LED_TOGGLE(pin) HAL_GPIO_TogglePin(GPIOC, pin)

// 调试用GPIO脉冲（示波器测量用）
#define DBG_PULSE(port, pin) do { \
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_SET); \
    HAL_GPIO_WritePin(port, pin, GPIO_PIN_RESET); \
} while(0)

// 函数执行时间测量
#define MEASURE_START() uint32_t _t0 = DWT->CYCCNT
#define MEASURE_END(label) \
    printf(label " took %lu cycles (%.2f us)\r\n", \
           DWT->CYCCNT - _t0, \
           (float)(DWT->CYCCNT - _t0) / (SystemCoreClock / 1000000))
```

## 六、调试心态与策略

```
1. 二分法定位：注释一半代码，确认问题在前半还是后半
2. 最小复现：用最简单的代码重现问题
3. 一次改一个：每次只修改一个变量
4. 记录日志：每次修改和结果都记录下来
5. 不要假设：用数据验证，不要凭直觉
6. 休息一下：卡住超过30分钟，站起来活动一下
```
