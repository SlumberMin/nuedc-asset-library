## 一、电源类赛题

### 1.1 开关电源设计（Buck变换器）

**考点：** 效率优化、纹波抑制、环路补偿

```c
// Buck变换器参数计算
// Vin=12V, Vout=5V, f=200kHz, Iout=2A

// 电感计算
// L = (Vin - Vout) * D / (f * ΔI)
// D = Vout/Vin = 5/12 = 0.417
// ΔI = 0.3 * Iout = 0.6A
// L = (12-5) * 0.417 / (200000 * 0.6) = 24.3μH → 选22μH

// 输出电容
// C = ΔI / (8 * f * ΔVout)
// ΔVout = 50mV (目标纹波)
// C = 0.6 / (8 * 200000 * 0.05) = 7.5μF → 选22μF×2

// 闭环控制（电压模式PID）
typedef struct {
    float Kp, Ki;
    float integral;
    float duty_min, duty_max;
} BuckPID_t;

float buck_pid_calc(BuckPID_t* pid, float vref, float vout) {
    float err = vref - vout;
    pid->integral += pid->Ki * err;
    
    // 积分限幅
    if (pid->integral > pid->duty_max) pid->integral = pid->duty_max;
    if (pid->integral < pid->duty_min) pid->integral = pid->duty_min;
    
    float duty = pid->Kp * err + pid->integral;
    
    // 输出限幅
    if (duty > pid->duty_max) duty = pid->duty_max;
    if (duty < pid->duty_min) duty = pid->duty_min;
    
    return duty;
}

// 在PWM中断中调用（200kHz = 5μs周期）
void TIM1_UP_IRQHandler(void) {
    TIM1->SR &= ~TIM_SR_UIF;
    
    float vout = read_adc_voltage();  // 读取输出电压
    float duty = buck_pid_calc(&buck_pid, 5.0f, vout);
    
    // 设置PWM占空比
    TIM1->CCR1 = (uint16_t)(duty * TIM1->ARR);
}
```

### 1.2 程控电源设计

```c
// 程控电压源（DAC + 运放 + MOS管）
// 电压设定：DAC输出 → 运放 → MOS管栅极
// 电流采样：采样电阻 → 差分运放 → ADC

typedef struct {
    float v_set;      // 设定电压
    float i_limit;    // 限流值
    float v_now;      // 当前电压
    float i_now;      // 当前电流
    uint8_t cc_mode;  // 0=CV恒压, 1=CC恒流
} PSU_t;

void psu_control(PSU_t* psu) {
    psu->v_now = read_voltage_adc() * VOLTAGE_GAIN;
    psu->i_now = read_current_adc() * CURRENT_GAIN;
    
    if (psu->i_now >= psu->i_limit) {
        // 恒流模式：调节电压使电流不超限
        psu->cc_mode = 1;
        float duty = pid_calc(&i_pid, psu->i_limit, psu->i_now);
        set_dac_voltage(duty);
    } else {
        // 恒压模式
        psu->cc_mode = 0;
        float duty = pid_calc(&v_pid, psu->v_set, psu->v_now);
        set_dac_voltage(duty);
    }
}
```

```c
// 单相逆变器 SPWM
// 50Hz正弦调制，20kHz载波
// 载波比 = 20000/50 = 400

#define CARRIER_FREQ  20000
#define MODULATE_FREQ 50
#define SINE_POINTS   400

uint16_t sine_table[SINE_POINTS];  // 预计算正弦表

void generate_spwm_table(void) {
    for (int i = 0; i < SINE_POINTS; i++) {
        // 0.5 + 0.5*sin(θ) 映射到 0~ARR
        float val = 0.5f + 0.5f * sinf(2.0f * 3.14159f * i / SINE_POINTS);
        sine_table[i] = (uint16_t)(val * TIM_PERIOD);
    }
}

// 在20kHz定时器中断中更新PWM
volatile uint16_t sine_index = 0;
void TIM1_UP_IRQHandler(void) {
    TIM1->SR &= ~TIM_SR_UIF;
    TIM1->CCR1 = sine_table[sine_index];
    sine_index = (sine_index + 1) % SINE_POINTS;
}
```

```c
#define DDS_CLK       1000000  // DDS更新率1MHz
```

### 2.2 频率/相位测量

**考点：** 等精度测频、相位差测量

```c
// 等精度测频法
// 原理：同时计数标准频率和被测频率的脉冲数
// f_x = f_s * N_x / N_s

// TIM2: 外部时钟模式，计数被测信号
// TIM3: 内部时钟，计数基准时间

volatile uint32_t N_x = 0;  // 被测信号脉冲数
volatile uint32_t N_s = 0;  // 基准信号脉冲数

float measure_frequency(void) {
    // 开闸门
    HAL_TIM_Base_Start(&htim2);  // 被测信号计数
    HAL_TIM_Base_Start(&htim3);  // 基准计数
    
    HAL_Delay(100);  // 闸门时间100ms
    
    N_x = TIM2->CNT;
    N_s = TIM3->CNT;
    
    HAL_TIM_Base_Stop(&htim2);
    HAL_TIM_Base_Stop(&htim3);
    
    TIM2->CNT = 0;
    TIM3->CNT = 0;
    
    // f_x = f_s * N_x / N_s
    float f_s = 72000000.0f;  // 基准频率
    return f_s * N_x / N_s;
}

// 相位差测量
// 用两个输入捕获通道，测量上升沿时间差
float measure_phase_diff(void) {
    // TIM2 CH1: 信号A上升沿捕获
    // TIM2 CH2: 信号B上升沿捕获
    uint32_t t1 = HAL_TIM_ReadCapturedValue(&htim2, TIM_CHANNEL_1);
    uint32_t t2 = HAL_TIM_ReadCapturedValue(&htim2, TIM_CHANNEL_2);
    uint32_t period = read_signal_period();
    
    float phase = 360.0f * (t2 - t1) / period;
    if (phase < 0) phase += 360.0f;
    return phase;
}
```

### 2.3 滤波器设计（IIR/FIR）

```c
// 二阶IIR低通滤波器（Butterworth）
// 截止频率1kHz，采样率10kHz
// 系数由MATLAB fdatool 或在线工具生成

typedef struct {
    float b0, b1, b2;  // 分子系数
    float a1, a2;       // 分母系数
    float x1, x2;       // 输入历史
    float y1, y2;       // 输出历史
} IIR_Filter_t;

float iir_filter(IIR_Filter_t* f, float x) {
    float y = f->b0 * x + f->b1 * f->x1 + f->b2 * f->x2
            - f->a1 * f->y1 - f->a2 * f->y2;
    
    f->x2 = f->x1;
    f->x1 = x;
    f->y2 = f->y1;
    f->y1 = y;
    
    return y;
}

// 初始化（1kHz截止，10kHz采样，2阶Butterworth）
IIR_Filter_t lpf = {
    .b0 = 0.0675f, .b1 = 0.1349f, .b2 = 0.0675f,
    .a1 = -1.1430f, .a2 = 0.4128f,
    .x1 = 0, .x2 = 0, .y1 = 0, .y2 = 0
};

// 滑动平均滤波（简单有效）
#define AVG_N 16
float moving_avg(float new_val) {
    static float buf[AVG_N] = {0};
    static int idx = 0;
    static float sum = 0;
    
    sum -= buf[idx];
    buf[idx] = new_val;
    sum += new_val;
    idx = (idx + 1) % AVG_N;
    
    return sum / AVG_N;
}
```

### 3.1 电机调速系统

**考点：** PWM驱动、速度反馈、PID调参

```c
// 直流电机速度闭环
// 编码器反馈 → 速度计算 → PID → PWM输出

// 编码器接口（TIM3编码器模式）
int16_t read_encoder(void) {
    int16_t count = (int16_t)TIM3->CNT;
    TIM3->CNT = 0;
    return count;
}

// 速度计算（每10ms调用）
float calc_speed(void) {
    static float speed_rpm = 0;
    int16_t count = read_encoder();
    // count = 10ms内的脉冲数
    // RPM = count / 编码器线数 / 减速比 * (60s / 0.01s)
    speed_rpm = (float)count / ENCODER_PPR / GEAR_RATIO * 6000.0f;
    return speed_rpm;
}

// 增量式PID
typedef struct {
    float Kp, Ki, Kd;
    float err[3];  // err[0]=当前, err[1]=上次, err[2]=上上次
    float output;
} IncPID_t;

float inc_pid_calc(IncPID_t* pid, float target, float actual) {
    pid->err[2] = pid->err[1];
    pid->err[1] = pid->err[0];
    pid->err[0] = target - actual;
    
    float delta = pid->Kp * (pid->err[0] - pid->err[1])
                + pid->Ki * pid->err[0]
                + pid->Kd * (pid->err[0] - 2*pid->err[1] + pid->err[2]);
    
    pid->output += delta;
    
    // 限幅
    if (pid->output > 1000) pid->output = 1000;
    if (pid->output < -1000) pid->output = -1000;
    
    return pid->output;
}

// 电机驱动（L298N / TB6612）
void motor_set_pwm(int16_t pwm) {
    if (pwm >= 0) {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_SET);   // 正转
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2, GPIO_PIN_RESET);
        TIM1->CCR1 = pwm;
    } else {
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_1, GPIO_PIN_RESET); // 反转
        HAL_GPIO_WritePin(GPIOA, GPIO_PIN_2, GPIO_PIN_SET);
        TIM1->CCR1 = -pwm;
    }
}
```

### 3.2 角度/位置跟踪系统

**考点：** 陀螺仪+加速度计融合、舵机控制、云台稳定

```c
// MPU6050 互补滤波（姿态解算）
typedef struct {
    float angle;   // 融合角度
    float alpha;   // 滤波系数
} ComplementaryFilter_t;

float comp_filter_update(ComplementaryFilter_t* cf, 
                          float gyro_rate, float accel_angle, float dt) {
    // 互补滤波：角速度短期准，加速度长期准
    // angle = α * (angle + gyro * dt) + (1-α) * accel
    cf->angle = cf->alpha * (cf->angle + gyro_rate * dt) 
              + (1.0f - cf->alpha) * accel_angle;
    return cf->angle;
}

// 陀螺仪数据读取
typedef struct {
    int16_t ax, ay, az;
    int16_t gx, gy, gz;
} MPU6050_Data_t;

void mpu6050_read(MPU6050_Data_t* data) {
    uint8_t buf[14];
    i2c_read_bytes(MPU6050_ADDR, 0x3B, buf, 14);
    data->ax = (buf[0] << 8) | buf[1];
    data->ay = (buf[2] << 8) | buf[3];
    data->az = (buf[4] << 8) | buf[5];
    data->gx = (buf[8] << 8) | buf[9];
    data->gy = (buf[10] << 8) | buf[11];
    data->gz = (buf[12] << 8) | buf[13];
}

float calc_accel_angle(MPU6050_Data_t* data) {
    return atan2f(data->ax, data->az) * 57.2958f;  // 弧度→角度
}

// 舵机控制（50Hz PWM，0.5ms~2.5ms对应0~180°）
void servo_set_angle(uint8_t angle) {
    // 50Hz → 周期20ms
    // ARR = 20000-1, PSC = 72-1 (72MHz/72=1MHz)
    // 0° → 500, 180° → 2500
    uint16_t pulse = 500 + (uint16_t)(angle * 2000.0f / 180.0f);
    TIM2->CCR1 = pulse;
}
```

### 3.3 平衡车/倒立摆

**考点：** 姿态融合、双环控制、直立+速度+转向

```c
// 平衡车三环控制
// 外环：速度环（100Hz）
// 中环：角度环（500Hz）  
// 内环：角速度环（500Hz）

typedef struct {
    float angle_target;     // 目标角度（由速度环输出）
    float speed_target;     // 目标速度
    float turn_target;      // 转向偏置
    float angle_Kp;
    float gyro_Kp;
    float speed_Kp, speed_Ki;
    float speed_integral;
} BalanceCtrl_t;

float speed_loop(BalanceCtrl_t* ctrl, float speed_now) {
    float err = ctrl->speed_target - speed_now;
    ctrl->speed_integral += ctrl->speed_Ki * err;
    // 积分限幅
    if (ctrl->speed_integral > 5.0f) ctrl->speed_integral = 5.0f;
    if (ctrl->speed_integral < -5.0f) ctrl->speed_integral = -5.0f;
    return ctrl->speed_Kp * err + ctrl->speed_integral;
}

float balance_loop(BalanceCtrl_t* ctrl, float angle, float gyro) {
    // 角度环 + 角速度环（PD控制）
    float angle_out = ctrl->angle_Kp * (ctrl->angle_target - angle);
    float gyro_out = ctrl->gyro_Kp * (0 - gyro);  // 角速度阻尼
    return angle_out + gyro_out;
}

void balance_task(BalanceCtrl_t* ctrl) {
    float angle = get_angle();       // 互补滤波/卡尔曼
    float gyro = get_gyro();         // 角速度
    float speed = get_speed();       // 编码器速度
    
    ctrl->angle_target = speed_loop(ctrl, speed);  // 速度环输出
    float pwm = balance_loop(ctrl, angle, gyro);
    
    // 转向叠加
    int16_t left_pwm  = (int16_t)(pwm + ctrl->turn_target);
    int16_t right_pwm = (int16_t)(pwm - ctrl->turn_target);
    
    motor_set_left(left_pwm);
    motor_set_right(right_pwm);
}
```

```c
// NRF24L01 数据包协议设计
typedef struct __attribute__((packed)) {
    uint8_t  header;      // 0xAA
    uint8_t  type;        // 数据类型
    uint16_t seq;         // 序列号
    uint8_t  data[28];    // 数据
    uint8_t  crc;         // 校验
} NRF_Packet_t;  // 32字节（NRF24L01最大载荷）

// 发送端
void nrf_send_data(uint8_t type, uint8_t* data, uint8_t len) {
    NRF_Packet_t pkt;
    pkt.header = 0xAA;
    pkt.type = type;
    pkt.seq = seq_num++;
    memset(pkt.data, 0, 28);
    memcpy(pkt.data, data, len > 28 ? 28 : len);
    pkt.crc = calc_crc8((uint8_t*)&pkt, 31);
    
    nrf24l01_send((uint8_t*)&pkt);
}

// 接收端（带重传请求）
void nrf_receive_handler(void) {
    NRF_Packet_t pkt;
    nrf24l01_recv((uint8_t*)&pkt);
    
    // 校验
    if (pkt.header != 0xAA) return;
    if (pkt.crc != calc_crc8((uint8_t*)&pkt, 31)) {
        send_nack(pkt.seq);  // 发送NAK，请求重传
        return;
    }
    
    // 丢包检测
    if (pkt.seq != expected_seq) {
        log_warn("丢包: expected=%d, got=%d", expected_seq, pkt.seq);
    }
    expected_seq = pkt.seq + 1;
    
    // 处理数据
    process_data(pkt.type, pkt.data);
    send_ack(pkt.seq);  // 发送ACK
}
```

### 4.2 多机通信（主从模式）

```c
// 485总线主从通信协议
// 主机轮询，从机应答

// 协议帧格式
typedef struct __attribute__((packed)) {
    uint8_t  addr;      // 从机地址
    uint8_t  cmd;       // 命令码
    uint8_t  len;       // 数据长度
    uint8_t  data[32];  // 数据
    uint16_t crc;       // CRC16
} RS485_Frame_t;

// 主机轮询
void master_poll(void) {
    for (uint8_t addr = 1; addr <= MAX_SLAVE; addr++) {
        RS485_Frame_t tx_frame = {
            .addr = addr,
            .cmd = CMD_READ_DATA,
            .len = 0
        };
        tx_frame.crc = crc16((uint8_t*)&tx_frame, 3 + tx_frame.len);
        
        rs485_send(&tx_frame);
        
        // 等待应答（超时10ms）
        if (rs485_wait_response(10)) {
            RS485_Frame_t rx_frame;
            rs485_recv(&rx_frame);
            if (rx_frame.crc == crc16((uint8_t*)&rx_frame, 3 + rx_frame.len)) {
                process_slave_data(addr, rx_frame.data, rx_frame.len);
            }
        } else {
            log_warn("Slave %d timeout", addr);
        }
    }
}
```

```c
// 频率测量（过零检测）
float measure_frequency_simple(uint16_t* buf, int len, uint16_t mid_level) {
    int rising_count = 0;
    uint32_t first_pos = 0, last_pos = 0;
    
    for (int i = 1; i < len; i++) {
        if (buf[i-1] < mid_level && buf[i] >= mid_level) {
            if (rising_count == 0) first_pos = i;
            last_pos = i;
            rising_count++;
        }
    }
    
    if (rising_count < 2) return 0;
    
    float avg_period = (float)(last_pos - first_pos) / (rising_count - 1);
    return SAMPLE_RATE / avg_period;  // Hz
}
```

### 5.2 RLC 测量仪

**考点：** 阻抗测量、相位法、谐振法

```c
// RLC测量仪设计
// 方法：正弦激励 + 测量电压/电流相位差

typedef enum { COMP_R, COMP_L, COMP_C } CompType_t;

typedef struct {
    CompType_t type;
    float value;      // 电阻Ω / 电感mH / 电容nF
    float Q;          // 品质因数
} RLC_Result_t;

RLC_Result_t rlc_measure(float freq, float v_drive, float v_component, 
                          float phase_diff_deg) {
    RLC_Result_t result;
    
    // 阻抗模值
    float Z = v_component / (v_drive - v_component) * R_REF;
    
    // 相位判断元件类型
    if (fabsf(phase_diff_deg) < 5.0f) {
        // 相位差≈0 → 纯电阻
        result.type = COMP_R;
        result.value = Z;
    } else if (phase_diff_deg > 0) {
        // 电压超前电流 → 电感
        result.type = COMP_L;
        float X_L = Z * sinf(phase_diff_deg * 0.01745f);
        result.value = X_L / (2 * 3.14159f * freq) * 1000;  // mH
        result.Q = X_L / (Z * cosf(phase_diff_deg * 0.01745f));
    } else {
        // 电流超前电压 → 电容
        result.type = COMP_C;
        float X_C = Z * sinf(-phase_diff_deg * 0.01745f);
        result.value = 1.0f / (2 * 3.14159f * freq * X_C) * 1e9;  // nF
        result.Q = X_C / (Z * cosf(-phase_diff_deg * 0.01745f));
    }
    
    return result;
}
```

### 6.2 信号失真度分析仪（2021年）

```c
// THD（总谐波失真）计算
// THD = √(V2² + V3² + ... + Vn²) / V1

// FFT计算（使用ARM DSP库）
#include "arm_math.h"

#define FFT_SIZE  1024
float32_t fft_input[FFT_SIZE * 2];  // 实部+虚部
float32_t fft_output[FFT_SIZE];

arm_cfft_radix4_instance_f32 fft_inst;
arm_cfft_radix4_init_f32(&fft_inst, FFT_SIZE, 0, 1);

void calc_thd(uint16_t* adc_data) {
    // 填充FFT输入
    for (int i = 0; i < FFT_SIZE; i++) {
        fft_input[2*i] = (float)adc_data[i] / 4096.0f;  // 实部
        fft_input[2*i+1] = 0;                              // 虚部
    }
    
    // 执行FFT
    arm_cfft_radix4_f32(&fft_inst, fft_input);
    
    // 计算幅度谱
    arm_cmplx_mag_f32(fft_input, fft_output, FFT_SIZE);
    
    // 找基波（最大分量，排除DC）
    float fund_mag = 0;
    int fund_idx = 1;
    for (int i = 2; i < FFT_SIZE/2; i++) {
        if (fft_output[i] > fund_mag) {
            fund_mag = fft_output[i];
            fund_idx = i;
        }
    }
    
    // 计算谐波能量
    float harm_power = 0;
    for (int n = 2; n <= 10; n++) {
        int idx = fund_idx * n;
        if (idx < FFT_SIZE/2) {
            harm_power += fft_output[idx] * fft_output[idx];
        }
    }
    
    // THD
    float thd = sqrtf(harm_power) / fund_mag * 100.0f;
    printf("THD = %.2f%%\r\n", thd);
}
```
