```c
// ========== 有限状态机 (FSM) ==========
typedef enum {
    STATE_IDLE,
    STATE_READY,
    STATE_RUNNING,
    STATE_PAUSED,
    STATE_ERROR,
    STATE_COUNT
} SystemState;

typedef enum {
    EVENT_START,
    EVENT_STOP,
    EVENT_PAUSE,
    EVENT_RESUME,
    EVENT_ERROR,
    EVENT_RESET,
    EVENT_COUNT
} SystemEvent;

// 状态转移表
typedef SystemState (*StateHandler)(void);

SystemState FSM_Transition(SystemState current, SystemEvent event)
{
    // 状态转移表
    static const SystemState transition[STATE_COUNT][EVENT_COUNT] = {
    //               START         STOP          PAUSE         RESUME        ERROR         RESET
    /* IDLE    */ { STATE_READY,  STATE_IDLE,   STATE_IDLE,   STATE_IDLE,   STATE_ERROR,  STATE_IDLE   },
    /* READY   */ { STATE_READY,  STATE_IDLE,   STATE_IDLE,   STATE_READY,  STATE_ERROR,  STATE_IDLE   },
    /* RUNNING */ { STATE_RUNNING,STATE_IDLE,   STATE_PAUSED, STATE_RUNNING,STATE_ERROR,  STATE_IDLE   },
    /* PAUSED  */ { STATE_PAUSED, STATE_IDLE,   STATE_PAUSED, STATE_RUNNING,STATE_ERROR,  STATE_IDLE   },
    /* ERROR   */ { STATE_ERROR,  STATE_ERROR,  STATE_ERROR,  STATE_ERROR,  STATE_ERROR,  STATE_IDLE   },
    };
    
    return transition[current][event];
}
```

### 2.6 通信协议设计

```c
// ========== 帧格式 ==========
// | 帧头(2B) | 长度(1B) | 命令(1B) | 数据(NB) | CRC16(2B) |
// | 0xAA 0x55|          |          |          |           |

#define FRAME_HEAD      0xAA55
#define FRAME_HEAD_SIZE 2
#define FRAME_LEN_SIZE  1
#define FRAME_CMD_SIZE  1
#define FRAME_CRC_SIZE  2
#define FRAME_OVERHEAD  (FRAME_HEAD_SIZE + FRAME_LEN_SIZE + FRAME_CMD_SIZE + FRAME_CRC_SIZE)
#define FRAME_DATA_MAX  64

typedef struct {
    uint8_t cmd;
    uint8_t data[FRAME_DATA_MAX];
    uint8_t len;
} Protocol_Frame;

// 解析状态机
typedef enum {
    PARSE_HEAD1,
    PARSE_HEAD2,
    PARSE_LEN,
    PARSE_CMD,
    PARSE_DATA,
    PARSE_CRC1,
    PARSE_CRC2
} ParseState;

typedef struct {
    ParseState state;
    Protocol_Frame frame;
    uint8_t dataIdx;
    uint16_t crc;
} Protocol_Parser;

void Protocol_ParseByte(Protocol_Parser* p, uint8_t byte)
{
    switch (p->state)
    {
    case PARSE_HEAD1:
        if (byte == 0xAA) p->state = PARSE_HEAD2;
        break;
    case PARSE_HEAD2:
        p->state = (byte == 0x55) ? PARSE_LEN : PARSE_HEAD1;
        break;
    case PARSE_LEN:
        p->frame.len = byte;
        p->dataIdx = 0;
        p->state = PARSE_CMD;
        break;
    case PARSE_CMD:
        p->frame.cmd = byte;
        p->state = (p->frame.len > 0) ? PARSE_DATA : PARSE_CRC1;
        break;
    case PARSE_DATA:
        p->frame.data[p->dataIdx++] = byte;
        if (p->dataIdx >= p->frame.len) p->state = PARSE_CRC1;
        break;
    case PARSE_CRC1:
        p->crc = byte;
        p->state = PARSE_CRC2;
        break;
    case PARSE_CRC2:
        p->crc |= (byte << 8);
        // CRC校验 + 回调处理
        if (CRC16_Check(&p->frame, p->crc))
            Protocol_HandleFrame(&p->frame);
        p->state = PARSE_HEAD1;
        break;
    }
}
```

## 三、内存管理

### 3.1 静态内存分配（推荐）

```c
// 电赛项目强烈建议静态分配，避免malloc/free
// 优势：无碎片、确定性、无泄漏风险

// 预分配缓冲区
#define RX_BUF_SIZE   256
#define TX_BUF_SIZE   256

static uint8_t g_rxBuf[RX_BUF_SIZE];
static uint8_t g_txBuf[TX_BUF_SIZE];

// 环形缓冲区（静态）
typedef struct {
    uint8_t  buf[RX_BUF_SIZE];
    volatile uint16_t head;
    volatile uint16_t tail;
    uint16_t size;
} RingBuffer;

void RingBuffer_Init(RingBuffer* rb)
{
    rb->head = 0;
    rb->tail = 0;
    rb->size = RX_BUF_SIZE;
}

uint16_t RingBuffer_Available(RingBuffer* rb)
{
    return (rb->head - rb->tail + rb->size) % rb->size;
}

int RingBuffer_Put(RingBuffer* rb, uint8_t data)
{
    uint16_t next = (rb->head + 1) % rb->size;
    if (next == rb->tail) return -1;  // 满
    rb->buf[rb->head] = data;
    rb->head = next;
    return 0;
}

int RingBuffer_Get(RingBuffer* rb, uint8_t* data)
{
    if (rb->head == rb->tail) return -1;  // 空
    *data = rb->buf[rb->tail];
    rb->tail = (rb->tail + 1) % rb->size;
    return 0;
}
```

### 3.2 栈深度估算

```
栈深度估算方法：

1. 最长调用链分析
   main() → App_Control() → Motor_PID() → Kalman_Update() → arm_mat_mult()
   
2. 每层栈使用
   main: 20B + 局部变量
   App_Control: 30B
   Motor_PID: 50B (PID结构体)
   Kalman_Update: 200B (矩阵运算)
   arm_mat_mult: 100B
   
3. 中断嵌套
   最大嵌套深度 × 最大中断栈 = 中断额外开销
   2 × 200B = 400B

4. 总栈 = 最长链(400B) + 中断(400B) + 安全余量(50%) = 1200B
   推荐分配: 2048B (2KB)

// 链接脚本中设置
_Min_Heap_Size = 0x200;   // 512B 堆
_Min_Stack_Size = 0x800;  // 2KB 栈
```

### 4.3 ADC规范

```c
// ========== ADC采样规范 ==========
// 多通道DMA采样 + 软件滤波

#define ADC_CHANNELS   4
#define ADC_OVERSAMPLE 8

typedef struct {
    uint16_t dma_buf[ADC_CHANNELS * ADC_OVERSAMPLE];
    uint16_t filtered[ADC_CHANNELS];
    uint8_t channel_map[ADC_CHANNELS];  // 逻辑通道→物理通道映射
} ADC_Manager;

void ADC_Filter(ADC_Manager* adc)
{
    for (uint8_t ch = 0; ch < ADC_CHANNELS; ch++)
    {
        uint32_t sum = 0;
        for (uint8_t s = 0; s < ADC_OVERSAMPLE; s++)
        {
            sum += adc->dma_buf[s * ADC_CHANNELS + ch];
        }
        adc->filtered[ch] = sum / ADC_OVERSAMPLE;
    }
}

// 滑动平均滤波器
#define MOVING_AVG_SIZE 16

typedef struct {
    uint16_t buf[MOVING_AVG_SIZE];
    uint8_t idx;
    uint32_t sum;
} MovingAvg;

uint16_t MovingAvg_Update(MovingAvg* ma, uint16_t new_val)
{
    ma->sum -= ma->buf[ma->idx];
    ma->sum += new_val;
    ma->buf[ma->idx] = new_val;
    ma->idx = (ma->idx + 1) % MOVING_AVG_SIZE;
    return ma->sum / MOVING_AVG_SIZE;
}
```

### 4.4 UART/DMA收发规范

```c
// ========== UART DMA收发 ==========
#define UART_RX_BUF_SIZE  256
#define UART_TX_BUF_SIZE  256

typedef struct {
    UART_HandleTypeDef* huart;
    uint8_t rxBuf[UART_RX_BUF_SIZE];
    uint8_t txBuf[UART_TX_BUF_SIZE];
    RingBuffer rxRing;
    volatile uint8_t txBusy;
} UART_Driver;

void UART_Driver_Init(UART_Driver* drv, UART_HandleTypeDef* huart)
{
    drv->huart = huart;
    drv->txBusy = 0;
    RingBuffer_Init(&drv->rxRing);
    
    // 开启DMA接收（空闲中断方式）
    HAL_UARTEx_ReceiveToIdle_DMA(drv->huart, drv->rxBuf, UART_RX_BUF_SIZE);
    __HAL_DMA_DISABLE_IT(drv->huart->hdmarx, DMA_IT_HT);  // 关半传输中断
}

// DMA接收完成回调
void HAL_UARTEx_RxEventCallback(UART_HandleTypeDef* huart, uint16_t Size)
{
    UART_Driver* drv = UART_GetDriver(huart);
    if (drv == NULL) return;
    
    // 将接收到的数据放入环形缓冲区
    for (uint16_t i = 0; i < Size; i++)
    {
        RingBuffer_Put(&drv->rxRing, drv->rxBuf[i]);
    }
    
    // 重新开启DMA接收
    HAL_UARTEx_ReceiveToIdle_DMA(drv->huart, drv->rxBuf, UART_RX_BUF_SIZE);
    __HAL_DMA_DISABLE_IT(drv->huart->hdmarx, DMA_IT_HT);
}
```

```c
// ========== ISR编写规范 ==========
// 1. 尽量短小，快速返回
// 2. 不使用浮点（除非FPU已启用）
// 3. 不调用阻塞函数
// 4. 不使用互斥锁
// 5. 共享变量用volatile

// 正确示例：设置标志位
volatile uint8_t g_captureDone;
volatile uint32_t g_captureValue;

void TIM3_IRQHandler(void)
{
    if (TIM3->SR & TIM_SR_CC1IF)
    {
        TIM3->SR &= ~TIM_SR_CC1IF;
        g_captureValue = TIM3->CCR1;
        g_captureDone = 1;  // 设置标志，主循环处理
    }
}

// 错误示例：ISR中做复杂处理
void TIM3_IRQHandler_WRONG(void)
{
    if (TIM3->SR & TIM_SR_CC1IF)
    {
        TIM3->SR &= ~TIM_SR_CC1IF;
        float freq = 1000000.0f / TIM3->CCR1;  // 浮点运算
        OLED_ShowNum(0, 0, (uint32_t)freq);     // 慢速I2C操作
        // 以上都不应在ISR中做！
    }
}
```

```c
// 二进制帧格式（高效）
// | 帧头 0xAA | 通道数 | ch1(L+H) | ch2(L+H) | ... | 校验 |
void Debug_SendChannels(int16_t* data, uint8_t count)
{
    uint8_t buf[32];
    uint8_t idx = 0;
    
    buf[idx++] = 0xAA;
    buf[idx++] = count;
    
    for (uint8_t i = 0; i < count; i++)
    {
        buf[idx++] = data[i] & 0xFF;
        buf[idx++] = (data[i] >> 8) & 0xFF;
    }
    
    // 简单校验
    uint8_t checksum = 0;
    for (uint8_t i = 0; i < idx; i++)
        checksum += buf[i];
    buf[idx++] = checksum;
    
    HAL_UART_Transmit_DMA(&huart1, buf, idx);
}
```

### 7.1 低功耗设计

```c
// ========== 低功耗模式 ==========
typedef enum {
    POWER_RUN,       // 正常运行
    POWER_SLEEP,     // CPU停止，外设运行
    POWER_STOP,      // 大部分时钟停止
    POWER_STANDBY    // 最低功耗，仅备份域
} PowerMode;

void Power_EnterMode(PowerMode mode)
{
    switch (mode)
    {
    case POWER_SLEEP:
        HAL_SuspendTick();
        HAL_PWR_EnterSLEEPMode(PWR_MAINREGULATOR_ON, PWR_SLEEPENTRY_WFI);
        HAL_ResumeTick();
        break;
        
    case POWER_STOP:
        HAL_SuspendTick();
        HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
        SystemClock_Config();  // 唤醒后需重新配置时钟
        HAL_ResumeTick();
        break;
        
    case POWER_STANDBY:
        HAL_PWR_EnterSTANDBYMode();  // 唤醒后执行复位
        break;
    }
}
```

```c
void Emergency_Stop(void)
{
    __disable_irq();
    
    // 1. 关闭所有PWM输出
    TIM1->CCR1 = 0;
    TIM1->CCR2 = 0;
    TIM1->CCR3 = 0;
    TIM1->CCR4 = 0;
    
    // 2. 刹车使能

    HAL_GPIO_WritePin(MOTOR_EN_GPIO_Port, MOTOR_EN_Pin, GPIO_PIN_RESET);
```
