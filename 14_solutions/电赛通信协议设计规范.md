# 电赛通信协议设计规范

> 模块间通信协议标准化设计指南

---

## 一、为什么需要自定义协议

电赛系统通常包含多个模块（主控+显示+传感器+执行器），通用协议（UART/SPI/I2C）只解决物理层，**应用层协议**需要自定义：

```
物理层: UART/SPI/I2C (硬件解决)
传输层: 帧格式/校验/流控 (本文档)
应用层: 命令定义/数据解析 (根据题目定义)
```

## 二、帧格式设计

### 2.1 推荐帧结构

```
┌──────┬──────┬──────┬────────┬──────────┬──────┬──────┐
│ 帧头 │ 长度 │ 命令 │ 序列号 │  数据域  │ CRC  │ 帧尾 │
│ 2B   │ 1B   │ 1B   │  1B    │  0~255B  │ 2B   │ 1B   │
│0xAA55│      │      │        │          │CRC16 │0x0D  │
└──────┴──────┴──────┴────────┴──────────┴──────┴──────┘
```

### 2.2 各字段说明

| 字段 | 长度 | 说明 |
|------|------|------|
| 帧头 | 2B | 固定 0xAA 0x55，用于同步 |
| 长度 | 1B | 数据域字节数（不含帧头/帧尾/CRC） |
| 命令 | 1B | 操作类型（0x01~0xFF） |
| 序列号 | 1B | 0~255循环，用于匹配请求/响应 |
| 数据域 | 0~255B | 变长负载 |
| CRC | 2B | CRC-16/CCITT，覆盖长度到数据域 |
| 帧尾 | 1B | 固定 0x0D |

### 2.3 C实现

```c
#include <stdint.h>
#include <string.h>

#define FRAME_HEAD      0xAA55
#define FRAME_TAIL      0x0D
#define FRAME_MAX_DATA  255

typedef struct __attribute__((packed)) {
    uint16_t head;       // 0xAA55
    uint8_t  length;     // 数据域长度
    uint8_t  cmd;        // 命令字
    uint8_t  seq;        // 序列号
    uint8_t  data[FRAME_MAX_DATA]; // 数据域
    uint16_t crc;        // CRC-16
    uint8_t  tail;       // 0x0D
} Frame_t;
```

## 三、CRC校验

### 3.1 CRC-16/CCITT（推荐）

```c
// 查表法，速度快
static const uint16_t crc16_table[256] = {
    0x0000, 0x1021, 0x2042, 0x3063, 0x4084, 0x50A5, 0x60C6, 0x70E7,
    0x8108, 0x9129, 0xA14A, 0xB16B, 0xC1AC, 0xD1CD, 0xE1EE, 0xF1EF,
    // ... 完整256项表省略，可用工具生成
};

uint16_t crc16_ccitt(const uint8_t *data, int len) {
    uint16_t crc = 0xFFFF; // 初始值
    for (int i = 0; i < len; i++) {
        crc = (crc << 8) ^ crc16_table[((crc >> 8) ^ data[i]) & 0xFF];
    }
    return crc;
}

// 简单实现（无查表，适合资源受限）
uint16_t crc16_simple(const uint8_t *data, int len) {
    uint16_t crc = 0xFFFF;
    for (int i = 0; i < len; i++) {
        crc ^= (uint16_t)data[i] << 8;
        for (int j = 0; j < 8; j++) {
            if (crc & 0x8000) crc = (crc << 1) ^ 0x1021;
            else crc <<= 1;
        }
    }
    return crc;
}
```

### 3.2 校验范围

```
CRC计算范围: [长度] [命令] [序列号] [数据域...]
即从第3字节到第(4+length-1)字节
```

## 四、发送与接收

### 4.1 发送

```c
void frame_send(uint8_t cmd, uint8_t seq,
                const uint8_t *data, uint8_t len) {
    uint8_t buf[FRAME_MAX_DATA + 7];
    int idx = 0;

    buf[idx++] = 0xAA;  // 帧头
    buf[idx++] = 0x55;
    buf[idx++] = len;    // 长度
    buf[idx++] = cmd;    // 命令
    buf[idx++] = seq;    // 序列号
    memcpy(&buf[idx], data, len);
    idx += len;

    uint16_t crc = crc16_simple(&buf[2], 3 + len); // 从长度开始
    buf[idx++] = crc >> 8;
    buf[idx++] = crc & 0xFF;
    buf[idx++] = 0x0D;  // 帧尾

    uart_send(buf, idx);
}
```

### 4.2 接收状态机

```c
typedef enum {
    RX_IDLE,
    RX_HEAD2,
    RX_LENGTH,
    RX_CMD,
    RX_SEQ,
    RX_DATA,
    RX_CRC_H,
    RX_CRC_L,
    RX_TAIL
} RxState;

typedef struct {
    RxState state;
    uint8_t length;
    uint8_t cmd;
    uint8_t seq;
    uint8_t data[FRAME_MAX_DATA];
    uint8_t data_idx;
    uint8_t crc_buf[2];
} RxParser;

void rx_parser_init(RxParser *p) {
    p->state = RX_IDLE;
    p->data_idx = 0;
}

// 每收到1字节调用一次
// 返回1表示收到完整帧，0表示还在接收
int rx_parser_feed(RxParser *p, uint8_t byte) {
    switch (p->state) {
    case RX_IDLE:
        if (byte == 0xAA) p->state = RX_HEAD2;
        break;
    case RX_HEAD2:
        p->state = (byte == 0x55) ? RX_LENGTH : RX_IDLE;
        break;
    case RX_LENGTH:
        p->length = byte;
        p->data_idx = 0;
        p->state = RX_CMD;
        break;
    case RX_CMD:
        p->cmd = byte;
        p->state = RX_SEQ;
        break;
    case RX_SEQ:
        p->seq = byte;
        p->state = (p->length > 0) ? RX_DATA : RX_CRC_H;
        break;
    case RX_DATA:
        p->data[p->data_idx++] = byte;
        if (p->data_idx >= p->length) p->state = RX_CRC_H;
        break;
    case RX_CRC_H:
        p->crc_buf[0] = byte;
        p->state = RX_CRC_L;
        break;
    case RX_CRC_L:
        p->crc_buf[1] = byte;
        p->state = RX_TAIL;
        break;
    case RX_TAIL:
        p->state = RX_IDLE;
        if (byte == 0x0D) {
            // 验证CRC
            uint8_t hdr[3] = {p->length, p->cmd, p->seq};
            uint16_t crc_calc = crc16_simple(hdr, 3);
            // 需要把data也加入CRC计算...
            return 1; // 帧有效
        }
        break;
    }
    return 0;
}
```

## 五、流控机制

### 5.1 软件流控（推荐电赛使用）

```
发送方: 发帧 → 等待ACK → 收到ACK后发下一帧
接收方: 收帧 → 校验 → 发ACK/NAK

ACK帧: cmd=0x00, data=序列号
NAK帧: cmd=0xFF, data=序列号
```

### 5.2 速率控制

```c
// 发送间隔控制（防止接收端溢出）
#define MIN_FRAME_INTERVAL_MS 5
static uint32_t last_send_time = 0;

int can_send_frame(void) {
    return (get_tick_ms() - last_send_time) >= MIN_FRAME_INTERVAL_MS;
}
```

## 六、重传机制

```c
#define MAX_RETRY 3
#define RETRY_TIMEOUT_MS 50

typedef struct {
    uint8_t pending_cmd;
    uint8_t pending_seq;
    uint8_t pending_data[FRAME_MAX_DATA];
    uint8_t pending_len;
    uint8_t retry_cnt;
    uint32_t send_time;
    int pending;
} ReliableSender;

int reliable_send(ReliableSender *s, uint8_t cmd, uint8_t seq,
                  const uint8_t *data, uint8_t len) {
    if (s->pending) return -1; // 上一帧未确认

    s->pending_cmd = cmd;
    s->pending_seq = seq;
    memcpy(s->pending_data, data, len);
    s->pending_len = len;
    s->retry_cnt = 0;
    s->send_time = get_tick_ms();
    s->pending = 1;

    frame_send(cmd, seq, data, len);
    return 0;
}

// 周期调用，处理超时重传
void reliable_poll(ReliableSender *s) {
    if (!s->pending) return;
    if (get_tick_ms() - s->send_time > RETRY_TIMEOUT_MS) {
        if (s->retry_cnt >= MAX_RETRY) {
            s->pending = 0; // 放弃
            return;
        }
        s->retry_cnt++;
        s->send_time = get_tick_ms();
        frame_send(s->pending_cmd, s->pending_seq,
                   s->pending_data, s->pending_len);
    }
}

void reliable_ack_received(ReliableSender *s, uint8_t seq) {
    if (s->pending && seq == s->pending_seq) {
        s->pending = 0;
    }
}
```

## 七、转义机制（帧头/帧尾冲突处理）

当数据域中出现 0xAA/0x55/0x0D 时需转义：

```c
#define ESCAPE_BYTE  0x7D
#define ESCAPE_XOR   0x20

void frame_send_raw(const uint8_t *buf, int len) {
    for (int i = 0; i < len; i++) {
        if (buf[i] == 0xAA || buf[i] == 0x55 ||
            buf[i] == 0x0D || buf[i] == ESCAPE_BYTE) {
            uint8_t esc = ESCAPE_BYTE;
            uart_send(&esc, 1);
            esc = buf[i] ^ ESCAPE_XOR;
            uart_send(&esc, 1);
        } else {
            uart_send(&buf[i], 1);
        }
    }
}
```

## 九、设计检查清单

- [ ] 帧头唯一，数据域中不会误匹配
- [ ] CRC覆盖所有关键字段
- [ ] 有超时机制，不会永久阻塞
- [ ] 接收缓冲区有溢出保护
- [ ] 命令字文档化，双方约定一致
- [ ] 调试口可用十六进制打印原始帧
- [ ] 波特率统一（推荐115200或921600）
