# 电赛嵌入式实时系统设计

> nuedc-asset-library · 问题解决方案库  
> 适用场景：多任务并行、中断驱动、时序严格约束、资源受限系统

---

## 1. 实时系统基础概念

### 1.1 实时 ≠ 快速

```
实时系统的定义: 在确定的时间内完成确定的任务
硬实时: 超时 = 系统失败 (如: 电机换相、保护动作)
软实时: 超时 = 性能下降 (如: 显示刷新、通信上报)
```

### 1.2 关键指标

| 指标 | 定义 | 电赛典型要求 |
|------|------|------------|
| 中断延迟 | 事件发生→ISR开始执行 | < 1μs |
| 上下文切换 | 任务切换开销 | < 10μs |
| 任务周期 | 定时任务执行间隔 | 100μs~10ms |
| 抖动(Jitter) | 周期偏差 | < 周期的5% |
| CPU利用率 | 任务计算时间/总时间 | < 70% |

## 2. RTOS选择与移植

### 2.1 常用RTOS对比

| RTOS | 内核大小 | 许可 | 特点 | 电赛推荐度 |
|------|---------|------|------|-----------|
| FreeRTOS | ~6KB | MIT | 生态好、资料多 | ⭐⭐⭐⭐⭐ |
| RT-Thread | ~3KB | Apache | 国产、组件丰富 | ⭐⭐⭐⭐ |
| uC/OS-III | ~6KB | 商业 | 稳定可靠 | ⭐⭐⭐ |
| Zephyr | ~8KB | Apache | 功能全、较重 | ⭐⭐ |
| 裸机+状态机 | 0 | - | 最简单 | ⭐⭐⭐⭐ |

### 2.2 FreeRTOS快速配置

```c
// FreeRTOSConfig.h 关键配置 (STM32F4, 168MHz)
#define configUSE_PREEMPTION            1       // 抢占式调度
#define configCPU_CLOCK_HZ              168000000
#define configTICK_RATE_HZ              1000    // 1ms系统节拍
#define configMAX_PRIORITIES            8       // 优先级数
#define configMINIMAL_STACK_SIZE        128     // 最小栈(words)
#define configTOTAL_HEAP_SIZE           (32*1024) // 堆大小
#define configMAX_TASK_NAME_LEN         16
#define configUSE_MUTEXES               1       // 使能互斥锁
#define configUSE_COUNTING_SEMAPHORES   1       // 使能计数信号量
#define configUSE_QUEUE_SETS            0
#define configUSE_TASK_NOTIFICATIONS    1       // 任务通知(高效)
#define configCHECK_FOR_STACK_OVERFLOW  2       // 栈溢出检测
```

### 2.3 移植要点

```
STM32 + FreeRTOS移植:
1. 使用CubeMX直接生成RTOS工程 (推荐)
2. 或手动: 复制FreeRTOS源码 → 实现port层 → 配置SysTick为RTOS时钟
3. 注意: SysTick中断优先级设为最低 (15)，PendSV同理
4. 所有外设中断优先级 < configMAX_SYSCALL_INTERRUPT_PRIORITY
```

## 3. 任务设计

### 3.1 任务划分原则

```
按功能模块划分:
┌──────────────┐
│ 传感器采集任务 │  周期: 1ms   优先级: 高
├──────────────┤
│ 控制计算任务   │  周期: 1ms   优先级: 最高
├──────────────┤
│ PWM输出任务    │  周期: 1ms   优先级: 高
├──────────────┤
│ 通信任务       │  周期: 10ms  优先级: 中
├──────────────┤
│ 显示任务       │  周期: 50ms  优先级: 低
├──────────────┤
│ 按键/状态机    │  周期: 20ms  优先级: 低
└──────────────┘
```

### 3.2 任务框架模板

```c
// 任务栈大小定义
#define STACK_SENSOR    (256)   // 传感器任务栈
#define STACK_CONTROL   (256)   // 控制任务栈
#define STACK_DISPLAY   (256)   // 显示任务栈
#define STACK_COMM      (512)   // 通信任务栈

// 任务优先级
#define PRIO_CONTROL    (configMAX_PRIORITIES - 1)  // 最高
#define PRIO_SENSOR     (configMAX_PRIORITIES - 2)
#define PRIO_PWM        (configMAX_PRIORITIES - 2)
#define PRIO_COMM       (configMAX_PRIORITIES - 4)
#define PRIO_DISPLAY    (configMAX_PRIORITIES - 5)

// 任务句柄
TaskHandle_t hTaskSensor, hTaskControl, hTaskDisplay, hTaskComm;

// 传感器采集任务
void vTaskSensor(void *pvParameters) {
    TickType_t xLastWakeTime = xTaskGetTickCount();
    
    for (;;) {
        // 读取ADC/DMA
        float adc_val = Read_ADC_Filtered();
        float voltage = adc_val * 3.3f / 4096.0f;
        
        // 放入队列/更新全局变量
        xQueueOverwrite(xQueueSensorData, &voltage);
        
        // 通知控制任务
        xTaskNotify(hTaskControl, 0, eNoAction);
        
        // 精确周期延时
        vTaskDelayUntil(&xLastWakeTime, pdMS_TO_TICKS(1));
    }
}

// 控制计算任务
void vTaskControl(void *pvParameters) {
    for (;;) {
        // 等待传感器数据就绪通知
        ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
        
        float sensor_val;
        xQueuePeek(xQueueSensorData, &sensor_val, 0);
        
        // 控制算法
        float output = PID_Calculate(&pid, setpoint, sensor_val);
        
        // 输出PWM
        Set_PWM_Duty(output);
    }
}

// main()中创建任务
int main(void) {
    Hardware_Init();
    
    xTaskCreate(vTaskSensor,  "Sensor",  STACK_SENSOR,  NULL, PRIO_SENSOR,  &hTaskSensor);
    xTaskCreate(vTaskControl, "Control", STACK_CONTROL, NULL, PRIO_CONTROL, &hTaskControl);
    xTaskCreate(vTaskDisplay, "Display", STACK_DISPLAY, NULL, PRIO_DISPLAY, &hTaskDisplay);
    xTaskCreate(vTaskComm,    "Comm",    STACK_COMM,    NULL, PRIO_COMM,    &hTaskComm);
    
    vTaskStartScheduler();  // 不会返回
    
    for (;;);  // 异常保护
}
```

## 4. 中断管理

### 4.1 中断优先级配置 (STM32 + FreeRTOS)

```
STM32中断优先级: 0(最高) ~ 15(最低)
  - 抢占优先级: 决定能否嵌套
  - 子优先级: 同级不嵌套时的排队顺序

FreeRTOS要求:
  configMAX_SYSCALL_INTERRUPT_PRIORITY = 5
  → 优先级 0~4 的中断不能调用FreeRTOS API (硬实时保护)
  → 优先级 5~15 的中断可以调用FreeRTOS API

推荐配置:
  优先级 0~1: 硬件保护 (过流/过压) — 不调用RTOS API
  优先级 2~3: 高速外设 (编码器/ADC DMA) — 不调用RTOS API
  优先级 4~5: 通信外设 (UART/SPI) — 可用信号量通知
  优先级 6~7: 定时器 — 可用任务通知
```

### 4.2 ISR设计规范

```c
// ✅ 正确的ISR设计
void ADC_IRQHandler(void) {
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    
    // 1. 快速读取数据 (最短时间)
    uint16_t raw = ADC1->DR;
    
    // 2. 通知任务处理 (非阻塞)
    xTaskNotifyFromISR(hTaskSensor, raw, 
                        eSetValueWithOverwrite,
                        &xHigherPriorityTaskWoken);
    
    // 3. 清除中断标志
    ADC1->SR &= ~ADC_SR_EOC;
    
    // 4. 必要时触发上下文切换
    portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
}

// ❌ 错误做法
void ADC_IRQHandler(void) {
    // 在ISR中做复杂计算 — 延长中断执行时间
    float filtered = Kalman_Filter(raw_data);
    
    // 在ISR中使用printf — 极其耗时
    printf("ADC: %d\n", raw);
    
    // 在ISR中使用阻塞API
    xQueueSend(xQueue, &data, portMAX_DELAY);  // 死锁!
}
```

### 4.3 中断与任务通信

```
方法          | 延迟  | 复杂度 | 数据量 | 适用场景
--------------|-------|--------|--------|--------
任务通知      | 最低  | 最低   | 1个值  | 单一事件/单值
二值信号量    | 低    | 低     | 无     | 事件触发
计数信号量    | 低    | 低     | 无     | 多次触发
消息队列      | 中    | 中     | 多值   | 结构化数据
流/消息缓冲   | 中    | 中     | 大量   | DMA数据流
全局变量+标志 | 最低  | 最低   | 任意   | 简单系统(裸机)
```

## 5. 资源共享与同步

### 5.1 互斥锁 (Mutex)

```c
// 保护共享资源 (如: I2C总线)
SemaphoreHandle_t xI2CMutex;

void vTaskSensor(void *pvParameters) {
    for (;;) {
        // 获取I2C互斥锁
        if (xSemaphoreTake(xI2CMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
            float temp = I2C_ReadTemperature();
            float humi = I2C_ReadHumidity();
            xSemaphoreGive(xI2CMutex);
            
            // 处理数据...
        }
        vTaskDelay(pdMS_TO_TICKS(100));
    }
}

void vTaskDisplay(void *pvParameters) {
    for (;;) {
        if (xSemaphoreTake(xI2CMutex, pdMS_TO_TICKS(10)) == pdTRUE) {
            OLED_WriteCmd(/* ... */);
            xSemaphoreGive(xI2CMutex);
        }
        vTaskDelay(pdMS_TO_TICKS(50));
    }
}
```

### 5.2 优先级反转与解决方案

```
问题:
  低优先级任务L持有资源 → 高优先级任务H等待 → 中优先级任务M抢占L
  → H被M间接阻塞 → 优先级反转!

解决方案:
1. 互斥锁 + 优先级继承 (FreeRTOS Mutex自动支持)
   → H等待时，L临时提升到H的优先级
2. 优先级天花板
   → 任务获取资源时，直接提升到可能的最高等待优先级
3. 禁用中断 (仅限极短临界区)
   taskENTER_CRITICAL() / taskEXIT_CRITICAL()
```

### 5.3 临界区保护

```c
// 方法1: 任务级临界区 (禁用中断，短时间)
void Update_Shared_Data(float new_val) {
    taskENTER_CRITICAL();
    shared_data = new_val;      // 原子操作保护
    taskEXIT_CRITICAL();
}

// 方法2: 调度器挂起 (允许中断，禁止任务切换)
void Update_Complex_Data(Data_t *src) {
    vTaskSuspendAll();          // 挂起调度器
    memcpy(&shared, src, sizeof(Data_t));
    xTaskResumeAll();           // 恢复调度器
}

// 方法3: 原子操作 (最高效)
// 单个变量读写 (32位对齐) 在ARM上天然原子
// 或使用 __atomic 内建函数
```

## 6. 实用设计模式

### 6.1 生产者-消费者模式

```c
// 传感器(生产者) → 队列 → 控制器(消费者)
QueueHandle_t xSensorQueue;

// 生产者
void vTaskSensor(void *p) {
    SensorData_t data;
    for (;;) {
        data.voltage = Read_ADC();
        data.timestamp = xTaskGetTickCount();
        xQueueSend(xSensorQueue, &data, 0);  // 非阻塞
        vTaskDelay(pdMS_TO_TICKS(1));
    }
}

// 消费者
void vTaskControl(void *p) {
    SensorData_t data;
    for (;;) {
        if (xQueueReceive(xSensorQueue, &data, pdMS_TO_TICKS(5)) == pdTRUE) {
            float output = PID_Calculate(&pid, setpoint, data.voltage);
            Set_PWM(output);
        }
    }
}
```

### 6.2 事件驱动模式

```c
// 多事件等待
void vTaskStateMachine(void *p) {
    EventBits_t events;
    for (;;) {
        events = xEventGroupWaitBits(
            xEventGroup,
            EVT_KEY_PRESS | EVT_TIMEOUT | EVT_SENSOR_READY,
            pdTRUE,           // 清除已响应位
            pdFALSE,          // 任意一位即可
            portMAX_DELAY
        );
        
        if (events & EVT_KEY_PRESS)    Handle_Key();
        if (events & EVT_TIMEOUT)      Handle_Timeout();
        if (events & EVT_SENSOR_READY) Handle_Sensor();
    }
}
```

### 6.3 软件定时器模式


## 9. 常见问题与解决方案

| 问题 | 可能原因 | 解决方案 |
|------|---------|---------|
| 任务不执行 | 栈溢出/优先级错误 | 检查栈水印、优先级配置 |
| 周期抖动大 | 高优先级任务阻塞 | 降低阻塞时间、优化任务 |
| 中断丢失 | ISR执行太长 | 移计算到任务中 |
| 死锁 | 多锁顺序不一致 | 统一锁获取顺序 |
| 优先级反转 | 使用Semaphore | 改用Mutex(优先级继承) |
| 内存不足 | 栈/堆过大 | 减小栈大小、优化数据结构 |
| HardFault | 未对齐访问/栈溢出 | 检查数据对齐、增大栈 |
| CPU 100% | 空闲任务被阻塞 | 检查是否有任务不延时 |
