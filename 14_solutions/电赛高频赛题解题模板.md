
### PID调参模板

```c
// 增量式PID（适合执行器）
typedef struct {
    float Kp, Ki, Kd;
    float e[3];  // e[0]=当前, e[1]=上次, e[2]=上上次
} PID_Inc;

float PID_Inc_Calc(PID_Inc *pid, float ref, float fbk) {
    pid->e[2] = pid->e[1];
    pid->e[1] = pid->e[0];
    pid->e[0] = ref - fbk;
    
    float du = pid->Kp * (pid->e[0] - pid->e[1])
             + pid->Ki * pid->e[0]
             + pid->Kd * (pid->e[0] - 2.0f * pid->e[1] + pid->e[2]);
    return du;
}

// 调参口诀：
// 先P后I再D
// P: 大到振荡，减半
// I: 消除稳态误差，从小开始
// D: 抑制超调，谨慎使用
```

### 快速得分策略
- **基础分**：能跑起来，完成基本功能（60%）
- **进阶分**：精度达标，快速响应（80%）
- **高分段**：鲁棒性好，边缘情况处理（90%+）

### Buck变换器设计模板

```c
// Buck参数计算
// Vout = D × Vin
// D = Vout / Vin

// 电感计算（纹波电流20%~40%）
// L = (Vin - Vout) × D / (ΔI × fsw)
// ΔI = 0.3 × Iout_max

// 电容计算（纹波电压要求）
// C = ΔI / (8 × fsw × ΔVout)

typedef struct {
    float Vin, Vout, Iout_max;
    float fsw;           // 开关频率
    float ripple_i;      // 电流纹波率
    float ripple_v;      // 电压纹波
} Buck_Params;

void Buck_CalcParams(Buck_Params *p) {
    float D = p->Vout / p->Vin;
    float dI = p->ripple_i * p->Iout_max;
    
    float L = (p->Vin - p->Vout) * D / (dI * p->fsw);
    float C = dI / (8.0f * p->fsw * p->ripple_v);
    
    // 输出计算结果
    printf("D = %.3f\n", D);
    printf("L = %.1f uH\n", L * 1e6);
    printf("C = %.1f uF\n", C * 1e6);
}
```

### FFT频谱分析模板

```c
#define FFT_SIZE 1024
#define FS 100000  // 采样率100kHz

float adc_buf[FFT_SIZE];
float fft_re[FFT_SIZE], fft_im[FFT_SIZE];
float magnitude[FFT_SIZE / 2];

void Spectrum_Analyze(void) {
    // 1. 采集数据（DMA）
    ADC_StartDMA(adc_buf, FFT_SIZE);
    
    // 2. 加窗
    for (int i = 0; i < FFT_SIZE; i++) {
        float w = 0.54f - 0.46f * cosf(2.0f * M_PI * i / FFT_SIZE);
        fft_re[i] = adc_buf[i] * w;
        fft_im[i] = 0;
    }
    
    // 3. FFT
    FFT_Compute(fft_re, fft_im, FFT_SIZE);
    
    // 4. 计算幅度
    for (int i = 0; i < FFT_SIZE / 2; i++) {
        magnitude[i] = sqrtf(fft_re[i]*fft_re[i] + fft_im[i]*fft_im[i]);
        magnitude[i] *= 2.0f / FFT_SIZE;
    }
    
    // 5. 频率分辨率 = FS/FFT_SIZE = 97.66Hz
    // 找峰值频率
    int peak_idx = 1;
    for (int i = 2; i < FFT_SIZE / 2; i++)
        if (magnitude[i] > magnitude[peak_idx]) peak_idx = i;
    float peak_freq = peak_idx * (float)FS / FFT_SIZE;
}
```

- **基础分**：能显示频谱/生成波形（60%）
- **进阶分**：频率精度<1%，动态范围>40dB（80%）

### 频率测量模板

```c
// 等精度测频法
typedef struct {
    uint32_t gate_time_ms;    // 闸门时间
    uint32_t ref_count;       // 参考时钟计数
    uint32_t signal_count;    // 信号计数
    float frequency;
    float resolution;
} FreqMeter;

float FreqMeter_Measure(FreqMeter *fm) {
    // 开启定时器闸门
    Timer_StartGate(fm->gate_time_ms);
    
    // 等待闸门结束
    while (!Timer_GateDone());
    
    // 读取计数值
    fm->ref_count = Timer_GetRefCount();
    fm->signal_count = Timer_GetSignalCount();
    
    // 计算频率
    // f_sig = f_ref × N_sig / N_ref
    float f_ref = 72000000.0f;  // 参考时钟72MHz
    fm->frequency = f_ref * fm->signal_count / fm->ref_count;
    
    // 分辨率
    fm->resolution = fm->frequency / fm->signal_count;
    
    return fm->frequency;
}
```

### FSK调制解调模板

```c
#define FREQ_MARK   1200   // 传号频率
#define FREQ_SPACE  2200   // 空号频率
#define BAUD_RATE   1200   // 波特率
#define FS          48000  // 采样率

// FSK调制
void FSK_Modulate(uint8_t *data, int len, float *output) {
    int samples_per_bit = FS / BAUD_RATE;
    int idx = 0;
    
    for (int i = 0; i < len; i++) {
        for (int bit = 0; bit < 8; bit++) {
            float freq = (data[i] & (1 << bit)) ? FREQ_MARK : FREQ_SPACE;
            for (int s = 0; s < samples_per_bit; s++) {
                output[idx++] = sinf(2.0f * M_PI * freq * s / FS);
            }
        }
    }
}

// FSK解调（过零检测法）
void FSK_Demodulate(float *input, int len, uint8_t *data) {
    int samples_per_bit = FS / BAUD_RATE;
    int idx = 0;
    
    for (int i = 0; i < len / samples_per_bit / 8; i++) {
        uint8_t byte = 0;
        for (int bit = 0; bit < 8; bit++) {
            // 计算该bit期间的过零次数
            int zero_cross = 0;
            int start = (i * 8 + bit) * samples_per_bit;
            for (int s = 1; s < samples_per_bit; s++) {
                if ((input[start+s] > 0) != (input[start+s-1] > 0))
                    zero_cross++;
            }
            float freq_est = (float)zero_cross * FS / (2 * samples_per_bit);
            if (freq_est > (FREQ_MARK + FREQ_SPACE) / 2)
                byte |= (1 << bit);
        }
        data[i] = byte;
    }
}
```

### 快速得分策略
- **基础分**：能通（60%）
- **进阶分**：误码率<10^-3，距离>5m（80%）
- **高分段**：高速率，远距离，强抗干扰（90%+）
