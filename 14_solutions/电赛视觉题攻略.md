### 2.2 颜色识别（色块检测）

```python
# OpenMV - LAB颜色空间色块检测
import sensor, image, time

sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)  # 320x240
sensor.skip_frames(time=2000)
sensor.set_auto_gain(False)  # 关闭自动增益！
sensor.set_auto_whitebal(False)  # 关闭白平衡！

# LAB颜色阈值 (L_min, L_max, A_min, A_max, B_min, B_max)
# 红色示例（需用工具→机器视觉→阈值编辑器获取）
RED_THRESHOLD = (30, 80, 15, 127, 15, 127)
BLUE_THRESHOLD = (15, 60, -30, 30, -128, -20)
GREEN_THRESHOLD = (30, 80, -60, -15, 10, 60)

while True:
    img = sensor.snapshot()

    # 查找色块
    blobs = img.find_blobs([RED_THRESHOLD],
                           pixels_threshold=100,   # 最小像素数
                           area_threshold=100,      # 最小面积
                           merge=True)              # 合并相邻色块

    if blobs:
        largest = max(blobs, key=lambda b: b.pixels())
        # 画框
        img.draw_rectangle(largest.rect(), color=(255, 0, 0))
        img.draw_cross(largest.cx(), largest.cy(), color=(255, 0, 0))

        # 输出坐标（归一化到-1~1）
        err_x = (largest.cx() - 160) / 160.0
        err_y = (largest.cy() - 120) / 120.0
        print("Target: x=%.2f y=%.2f size=%d" % (err_x, err_y, largest.pixels()))
```

```python
img = sensor.snapshot()
circles = img.find_circles(threshold=3000,
```

### 2.4 AprilTag定位（精确测距）

```python
# AprilTag - 精确定位和测距
import math

TAG_SIZE = 0.04  # 标签实际尺寸(米)

# 使用TAG36H11家族
tags = img.find_apriltags(families=image.TAG36H11)
for tag in tags:

    # 获取位姿
    print("Tag ID:", tag.id())
    print("Translation:", tag.x_translation(),
                         tag.y_translation(),
                         tag.z_translation())  # 单位米
    print("Rotation:", tag.x_rotation(),
                      tag.y_rotation(),
                      tag.z_rotation())

    # 计算距离
    distance = tag.z_translation()
    # x_translation: 水平偏移（右正左负）
    # y_translation: 垂直偏移（上下）
    angle = math.degrees(math.atan2(tag.x_translation(),
                                     tag.z_translation()))
    print("Distance: %.2f m, Angle: %.1f deg" % (distance, angle))
```

### 2.5 与STM32串口通信

```python
# OpenMV发送数据给STM32
import pyb

uart = pyb.UART(3, 115200, timeout_char=1000)

def send_target(cx, cy, area):
    """帧格式: [0xAA][cx_H][cx_L][cy_H][cy_L][area_H][area_L][sum]"""
    buf = bytearray(8)
    buf[0] = 0xAA
    buf[1] = (cx >> 8) & 0xFF
    buf[2] = cx & 0xFF
    buf[3] = (cy >> 8) & 0xFF
    buf[4] = cy & 0xFF
    buf[5] = (area >> 8) & 0xFF
    buf[6] = area & 0xFF
    buf[7] = sum(buf[1:7]) & 0xFF  # 校验和
    uart.write(buf)

# 主循环
while True:
    img = sensor.snapshot()
    blobs = img.find_blobs([RED_THRESHOLD], pixels_threshold=200)
    if blobs:
        b = max(blobs, key=lambda x: x.pixels())
        send_target(b.cx(), b.cy(), b.pixels())
    else:
        send_target(0, 0, 0)  # 无目标
```

```python
import sensor, image, lcd
from machine import UART

sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)
sensor.run(1)
lcd.init()

uart = UART(UART.UART1, 115200)

while True:
    img = sensor.snapshot()

    # 颜色追踪
    blobs = img.find_blobs([(30, 80, 15, 127, 15, 127)],
                           pixels_threshold=100,
                           area_threshold=100,
                           merge=True)
    for b in blobs:
        img.draw_rectangle(b.rect())
        img.draw_cross(b.cx(), b.cy())
        # 发送数据
        uart.write(bytes([0xAA, b.cx(), b.cy()]))

    lcd.display(img)
```

### 3.3 K210数字识别（MNIST）

```python
import KPU as kpu

# 加载模型
task = kpu.load("/sd/mnist.kmodel")

while True:
    img = sensor.snapshot()
    # 裁剪ROI
    roi = (100, 60, 120, 120)
    img_roi = img.copy(roi=roi)

    # 推理
    fmap = kpu.forward(task, img_roi)
    plist = fmap[:]
    digit = plist.index(max(plist))
    confidence = max(plist)

    if confidence > 0.8:
        print("Recognized: %d (%.1f%%)" % (digit, confidence * 100))
```

## 四、树莓派方案（高级题）

### 4.1 OpenCV颜色识别

```python
import cv2
import numpy as np

cap = cv2.VideoCapture(0)
cap.set(cv2.CAP_PROP_FRAME_WIDTH, 320)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 240)

while True:
    ret, frame = cap.read()
    hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

    # 红色HSV范围（两段）
    lower1 = np.array([0, 100, 100])
    upper1 = np.array([10, 255, 255])
    lower2 = np.array([160, 100, 100])
    upper2 = np.array([180, 255, 255])

    mask = cv2.inRange(hsv, lower1, upper1) | cv2.inRange(hsv, lower2, upper2)
    mask = cv2.erode(mask, None, iterations=2)
    mask = cv2.dilate(mask, None, iterations=2)

    contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL,
                                    cv2.CHAIN_APPROX_SIMPLE)
    if contours:
        c = max(contours, key=cv2.contourArea)
        if cv2.contourArea(c) > 500:
            x, y, w, h = cv2.boundingRect(c)
            cx, cy = x + w//2, y + h//2
            cv2.rectangle(frame, (x, y), (x+w, y+h), (0, 255, 0), 2)
            cv2.circle(frame, (cx, cy), 5, (0, 0, 255), -1)
            print(f"Target at ({cx}, {cy}), area={cv2.contourArea(c):.0f}")

    cv2.imshow("frame", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break
```

### 4.2 OpenCV形状识别

```python
def classify_shape(contour):
    peri = cv2.arcLength(contour, True)
    approx = cv2.approxPolyDP(contour, 0.04 * peri, True)
    vertices = len(approx)

    if vertices == 3:
        return "Triangle"
    elif vertices == 4:
        (x, y, w, h) = cv2.boundingRect(approx)
        aspect_ratio = w / float(h)
        if 0.95 <= aspect_ratio <= 1.05:
            return "Square"
        else:
            return "Rectangle"
    elif vertices == 5:
        return "Pentagon"
    else:
        return "Circle"

# 使用
for contour in contours:
    if cv2.contourArea(contour) > 500:
        shape = classify_shape(contour)
        print(f"Detected: {shape}")
```

## 五、视觉+控制联调

### 5.1 视觉追踪小车（OpenMV+STM32）

```python
# OpenMV端 - 发送目标偏差
import sensor, image, time, pyb

sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)
sensor.skip_frames(time=2000)
sensor.set_auto_gain(False)
sensor.set_auto_whitebal(False)

uart = pyb.UART(3, 115200)

RED_THRESH = (30, 80, 15, 127, 15, 127)

while True:
    img = sensor.snapshot()
    blobs = img.find_blobs([RED_THRESH], pixels_threshold=200, merge=True)

    if blobs:
        b = max(blobs, key=lambda x: x.pixels())
        # 偏差：-100 ~ +100
        err_x = b.cx() - 160
        err_y = b.cy() - 120
        area = b.pixels()

        # 发送: [头][err_x][err_y][area_lo][area_hi][校验]
        data = bytearray(6)
        data[0] = 0xAA
        data[1] = err_x & 0xFF
        data[2] = err_y & 0xFF
        data[3] = area & 0xFF
        data[4] = (area >> 8) & 0xFF
        data[5] = sum(data[1:5]) & 0xFF
        uart.write(data)
    else:
        uart.write(bytearray([0xAA, 0, 0, 0, 0, 0xAA]))
```

```c
// STM32端 - 接收并控制
void Vision_TrackControl(void)
{
    if (uart_frame_ready) {
        uart_frame_ready = 0;

        int8_t err_x = (int8_t)rx_buf[1];  // 水平偏差
        int16_t area = rx_buf[3] | (rx_buf[4] << 8);

        // 水平偏差 → 转向
        float turn_pwm = PID_Calc(&turn_pid, 0, err_x);

        // 距离控制（面积越大越近）
        float target_area = 5000;  // 目标面积
        float speed_pwm = PID_Calc(&speed_pid, target_area, area);

        // 差速控制
        Motor_SetPWM(1, speed_pwm + turn_pwm);
        Motor_SetPWM(2, speed_pwm - turn_pwm);
    }
}
```

### 6.1 阈值获取方法

```
OpenMV IDE: 工具 → 机器视觉 → 阈值编辑器
→ 实时拖动滑块查看效果
→ 记录最终LAB阈值

K210 MaixPy IDE: 工具 → 颜色阈值编辑器
```

### 6.2 常见问题

| 问题 | 原因 | 解决 |
|------|------|------|
| 识别不稳定 | 光线变化 | 关闭自动增益/白平衡，固定曝光 |
| 检测不到 | 阈值不对 | 重新标定阈值 |
| 帧率太低 | 分辨率太高 | 用QVGA(320x240)或QQVGA |
| 距离不准 | 没标定 | 用AprilTag或已知尺寸标定 |
| 误检多 | 环境干扰 | 增大pixels_threshold |

### 6.3 性能优化

```
1. 降低分辨率: QVGA→QQVGA 帧率翻倍
2. ROI裁剪: 只处理感兴趣区域
3. 减少find_blobs参数: merge=True减少后处理
4. 避免在循环中创建对象
5. 二值化后再处理（比RGB快得多）
```
