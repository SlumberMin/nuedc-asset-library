## 二、考点1：PID深度解析

### 2.1 增量式PID（电赛最常用）

```c
typedef struct {
    float Kp, Ki, Kd;
    float error[3];     // error[0]=当前, [1]=上次, [2]=上上次
    float output;
    float max_output;
    float dead_zone;    // 死区
} PID_Inc_t;

float PID_IncCalc(PID_Inc_t *pid, float target, float actual)
{
    pid->error[2] = pid->error[1];
    pid->error[1] = pid->error[0];
    pid->error[0] = target - actual;

    // 死区处理
    if (fabsf(pid->error[0]) < pid->dead_zone) {
        pid->error[0] = 0;
    }

    float increment = pid->Kp * (pid->error[0] - pid->error[1])
                    + pid->Ki * pid->error[0]
                    + pid->Kd * (pid->error[0] - 2*pid->error[1] + pid->error[2]);

    pid->output += increment;

    // 限幅
    if (pid->output > pid->max_output) pid->output = pid->max_output;
    if (pid->output < -pid->max_output) pid->output = -pid->max_output;

    return pid->output;
}
```

### 2.2 积分抗饱和（考试常问）

```c
// 方法1：积分限幅
pid->integral += error * dt;
if (pid->integral > integral_max) pid->integral = integral_max;
if (pid->integral < -integral_max) pid->integral = -integral_max;

// 方法2：条件积分（只在误差小时积分）
if (fabsf(error) < threshold) {
    pid->integral += error * dt;
}

// 方法3：Back-Calculation抗饱和
// 当输出饱和时，反馈修正积分项
if (output > max_output || output < -max_output) {
    pid->integral -= Kb * (output - pid->output_clamped) * dt;
}
```

### 2.3 PID调参口诀

```
先P后I再D，先内环后外环
P→响应速度（大则快但振荡）
I→消除稳态误差（大则慢但准）
D→抑制超调（大则敏感，需滤波）

口诀：
"先只加P，振荡就减"
"加I消差，缓慢增加"
"加D减超，注意噪声"
```

## 三、考点2：传感器数据处理

### 3.1 编码器测速（三种方法对比）

```c
/* 方法1：M法（定时采样计数）—— 高速精确 */
float Speed_M_Method(void)
{
    static int32_t last_count = 0;
    int32_t count = TIM_GetCounter(ENCODER_TIM);
    int32_t diff = count - last_count;
    last_count = count;

    // 处理溢出
    if (diff > 32768) diff -= 65536;
    if (diff < -32768) diff += 65536;

    // 转换为RPM
    float rpm = diff * 60.0f / (PPR * 4 * SAMPLE_PERIOD_S);
    return rpm;
}

/* 方法2：T法（脉冲计时）—— 低速精确 */
float Speed_T_Method(void)
{
    static uint32_t last_pulse_time = 0;
    uint32_t now = HAL_GetTick();  // ms

    if (pulse_detected) {
        uint32_t period = now - last_pulse_time;
        last_pulse_time = now;
        if (period > 0) {
            return 60000.0f / (PPR * 4 * period);  // RPM
        }
    }
    return 0;
}

/* 方法3：M/T法（综合）—— 全速域精确 */
float Speed_MT_Method(void)
{
    // 在固定采样周期T内，同时计数脉冲m1和时钟m2
    // speed = m1 * f_clock / (PPR * m2)
    return (float)pulse_count * CLOCK_FREQ / (PPR * 4 * clock_count);
}
```

### 3.2 MPU6050姿态解算

```c
/* 互补滤波融合加速度计和陀螺仪 */
typedef struct {
    float angle;        // 融合角度
    float gyro_bias;    // 陀螺仪零偏
    float alpha;        // 融合系数
} ComplementaryFilter_t;

float ComplementaryFilter_Update(ComplementaryFilter_t *cf,
                                  float accel_angle, float gyro_rate, float dt)
{
    // 陀螺仪积分（短期准确）
    float gyro_angle = cf->angle + (gyro_rate - cf->gyro_bias) * dt;

    // 加速度计角度（长期稳定，但有噪声）
    // accel_angle = atan2(ay, az) * 57.3f

    // 互补融合：alpha越大，越信任陀螺仪
    cf->angle = cf->alpha * gyro_angle + (1 - cf->alpha) * accel_angle;

    // 零偏更新（缓慢）
    cf->gyro_bias += 0.001f * (gyro_rate - cf->gyro_bias);

    return cf->angle;
}
```

```c
/* 卡尔曼滤波（更精确） */
typedef struct {
    float x[2];     // 状态：[角度, 角速度偏差]
    float P[2][2];  // 协方差
    float Q[2];     // 过程噪声
    float R;        // 测量噪声
} Kalman_t;

void Kalman_Init(Kalman_t *kf)
{
    kf->x[0] = 0; kf->x[1] = 0;
    kf->P[0][0] = 1; kf->P[0][1] = 0;
    kf->P[1][0] = 0; kf->P[1][1] = 1;
    kf->Q[0] = 0.001f;  // 角度过程噪声
    kf->Q[1] = 0.003f;  // 偏差过程噪声
    kf->R = 0.5f;        // 加速度计测量噪声
}

float Kalman_Update(Kalman_t *kf, float angle_meas, float gyro_rate, float dt)
{
    // 预测
    kf->x[0] += (gyro_rate - kf->x[1]) * dt;
    kf->P[0][0] += dt * (dt*kf->P[1][1] - kf->P[0][1] - kf->P[1][0] + kf->Q[0]);
    kf->P[0][1] -= dt * kf->P[1][1];
    kf->P[1][0] -= dt * kf->P[1][1];
    kf->P[1][1] += kf->Q[1] * dt;

    // 更新
    float y = angle_meas - kf->x[0];
    float S = kf->P[0][0] + kf->R;
    float K0 = kf->P[0][0] / S;
    float K1 = kf->P[1][0] / S;

    kf->x[0] += K0 * y;
    kf->x[1] += K1 * y;
    kf->P[0][0] *= (1 - K0);
    kf->P[0][1] *= (1 - K0);
    kf->P[1][0] -= K1 * kf->P[0][0];
    kf->P[1][1] -= K1 * kf->P[0][1];

    return kf->x[0];
}
```

## 四、考点3：串级PID控制

### 4.1 经典串级结构

```
目标位置 → [外环位置PID] → 目标速度 → [内环速度PID] → PWM
                ↑                              ↑
            实际位置(编码器)              实际速度(编码器)
```

```c
typedef struct {
    PID_t outer;    // 外环（位置/角度）
    PID_t inner;    // 内环（速度/角速度）
} CascadePID_t;

float CascadePID_Calc(CascadePID_t *cpid, float target_pos, float actual_pos, float actual_vel)
{
    // 外环：位置→目标速度
    float target_vel = PID_Calc(&cpid->outer, target_pos, actual_pos);

    // 内环：速度→PWM
    float pwm = PID_Calc(&cpid->inner, target_vel, actual_vel);

    return pwm;
}
```

### 4.2 平衡小车双环控制

```c
// 角度环（内环）+ 位置环（外环）
float Balance_Control(float target_pos, float angle, float gyro, float pos)
{
    // 外环：位置→目标角度
    float target_angle = PID_Calc(&pos_pid, target_pos, pos);
    target_angle = constrain(target_angle, -15, 15);  // 限幅

    // 内环：角度→PWM
    float pwm = PID_Calc(&angle_pid, target_angle, angle);

    // 角速度阻尼（D项）
    pwm -= Kd_gyro * gyro;

    return pwm;
}
```

## 五、考点4：循迹算法

### 5.1 电磁循迹（5电感方案）

```c
/* 五电感布局：  L2  L1  M  R1  R2  */
float sensor[5];  // 左2, 左1, 中, 右1, 右2

float Calc_TrackError(void)
{
    // 加权求和归一化
    float numerator = -2*sensor[0] + -1*sensor[1] + 0*sensor[2]
                    + 1*sensor[3] + 2*sensor[4];
    float denominator = sensor[0] + sensor[1] + sensor[2]
                      + sensor[3] + sensor[4];

    if (denominator < 10) return 0;  // 脱线处理

    return numerator / denominator;  // 归一化偏差 [-2, 2]
}

// 十字路口检测
int Detect_Crossroad(void)
{
    if (sensor[0] > THRESH && sensor[4] > THRESH) {
        return 1;  // 十字路口
    }
    return 0;
}

// T字路口检测
int Detect_TJunction(void)
{
    if (sensor[0] > THRESH && sensor[4] > THRESH && sensor[2] < THRESH_LOW) {
        return 1;  // T字路口
    }
    return 0;
}
```

### 5.2 光电循迹（8路）

uint8_t sensor_bits = Read_8Sensors();  // 8位，1=黑线

float Calc_Error_Bits(uint8_t bits)
{
    // 位置编码表
    switch (bits) {
        case 0b00011000: return 0;      // 居中
        case 0b00110000: return -0.5f;  // 稍偏左
        case 0b01100000: return -1.0f;  // 偏左
        case 0b11000000: return -2.0f;  // 大偏左
        case 0b00001100: return 0.5f;   // 稍偏右
        case 0b00000110: return 1.0f;   // 偏右
        case 0b00000011: return 2.0f;   // 大偏右
        case 0b00000000: return 99;     // 脱线
        default: return Calc_WeightedAvg(bits);  // 其他情况加权平均
    }
}

## 六、考点5：通信协议

### 6.1 自定义帧协议

```c
/* 帧格式: [帧头][长度][命令][数据...][校验][帧尾] */
#define FRAME_HEAD  0xAA
#define FRAME_TAIL  0x55

typedef struct {
    uint8_t cmd;
    uint8_t data[32];
    uint8_t len;
} Frame_t;

// 发送
void Send_Frame(uint8_t cmd, uint8_t *data, uint8_t len)
{
    UART_SendByte(FRAME_HEAD);
    UART_SendByte(len + 2);     // 长度 = cmd + data + checksum(2)
    UART_SendByte(cmd);
    uint8_t checksum = cmd;
    for (int i = 0; i < len; i++) {
        UART_SendByte(data[i]);
        checksum ^= data[i];
    }
    UART_SendByte(checksum);
    UART_SendByte(FRAME_TAIL);
}
```

## 七、高频面试/测评问题

| 问题 | 答案要点 |
|------|---------|
| PID各参数作用？ | P：响应速度；I：消除稳态误差；D：抑制超调 |
| 增量式vs位置式？ | 增量式无积分饱和，适合执行器；位置式有积分累积 |
| 如何消除积分饱和？ | 积分限幅/条件积分/Back-Calculation |
| 互补滤波原理？ | 高频信任陀螺仪，低频信任加速度计 |
| 串级PID为什么好？ | 内环快速抑制扰动，外环精确跟踪 |
| 编码器M/T法区别？ | M法高速精确，T法低速精确 |
| 滑模控制抖振原因？ | 不连续切换项导致，用饱和函数解决 |
| 滑模面如何选择？ | s = ė + λe，λ决定收敛速度 |
