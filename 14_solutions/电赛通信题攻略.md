## 一、通信题型分析

| 类型 | 典型任务 | 难度 |
|---|---|---|
| 基带传输 | 2FSK/2PSK/4QAM调制解调 | ★★★ |
| 频带传输 | AM/FM/SSB调幅调频 | ★★ |

| 同步系统 | 载波同步/位同步/帧同步 | ★★★★ |
| 综合通信链路 | 完整收发系统 | ★★★★★ |

### 1.2 评审重点

```
1. 通信距离/误码率（核心指标）
2. 传输速率（比特率）
3. 频谱效率
4. 抗噪声/抗干扰能力
5. 系统完整性和创新点
```

## 二、调制解调技术

### 2.1 2FSK调制解调（最常考）

```c
// 2FSK调制：用两个频率表示0和1
#define FREQ_0  2000   // 0对应频率
#define FREQ_1  4000   // 1对应频率
#define FS      48000  // 采样率
#define BIT_RATE 1000  // 比特率

int samples_per_bit = FS / BIT_RATE;  // 48

void FSK_modulate(uint8_t *bits, int n_bits, float *output) {
    for (int i = 0; i < n_bits; i++) {
        float freq = bits[i] ? FREQ_1 : FREQ_0;
        for (int j = 0; j < samples_per_bit; j++) {
            float t = (float)j / FS;
            output[i * samples_per_bit + j] = sinf(2 * M_PI * freq * t);
        }
    }
}

// 2FSK解调：过零检测法（硬件友好）
int FSK_demodulate_zerocross(float *signal, int len) {
    int zero_count = 0;
    for (int i = 1; i < len; i++) {
        if ((signal[i-1] < 0 && signal[i] >= 0) ||
            (signal[i-1] >= 0 && signal[i] < 0)) {
            zero_count++;
        }
    }
    // 过零率 ≈ 2*freq*T = 2*freq/bit_duration
    float freq_est = (float)zero_count / (2.0f * len / FS);
    return (freq_est > (FREQ_0 + FREQ_1) / 2) ? 1 : 0;
}

// 2FSK解调：双带通+包络检波（推荐）
// 用两个带通滤波器分别提取f0和f1分量
// 比较两路包络幅度，大的那路对应发送符号
```

### 2.2 2PSK/BPSK调制解调

```c
// BPSK调制：0→0相位，1→π相位
void BPSK_modulate(uint8_t *bits, int n_bits, float *output, float fc) {
    for (int i = 0; i < n_bits; i++) {
        float phase = bits[i] ? M_PI : 0;
        for (int j = 0; j < samples_per_bit; j++) {
            float t = (float)j / FS;
            output[i * samples_per_bit + j] = cosf(2 * M_PI * fc * t + phase);
        }
    }
}

// BPSK解调：相干解调
// 1. 本地载波恢复（Costas环或平方环）
// 2. 与本地载波相乘 → 低通滤波 → 判决
float BPSK_demodulate_coherent(float *signal, float *local_carrier, int len) {
    float sum = 0;
    for (int i = 0; i < len; i++) {
        sum += signal[i] * local_carrier[i];  // 相乘
    }
    sum /= len;  // 低通（积分+平均）
    return (sum > 0) ? 0 : 1;  // 判决（注意π相位翻转）
}
```

### 2.3 QPSK/4QAM调制解调

```c
// QPSK：每2bit一个符号，4种相位
// 星座图映射：00→(1,1), 01→(-1,1), 11→(-1,-1), 10→(1,-1)
typedef struct { float I; float Q; } Complex;

Complex QPSK_map(uint8_t b0, uint8_t b1) {
    Complex c;
    c.I = b0 ? -1.0f : 1.0f;
    c.Q = b1 ? -1.0f : 1.0f;
    return c;
}

// QPSK调制
void QPSK_modulate(uint8_t *bits, int n_bits, Complex *output, float fc) {
    int n_sym = n_bits / 2;
    for (int i = 0; i < n_sym; i++) {
        Complex sym = QPSK_map(bits[2*i], bits[2*i+1]);
        for (int j = 0; j < samples_per_symbol; j++) {
            float t = (float)j / FS;
            output[i*samples_per_symbol+j].I = sym.I * cosf(2*M_PI*fc*t);
            output[i*samples_per_symbol+j].Q = -sym.Q * sinf(2*M_PI*fc*t);
        }
    }
}
```

### 2.4 调制方式选择指南

| 方式 | 频谱效率 | 抗噪声 | 实现难度 | 推荐场景 |
|---|---|---|---|---|
| 2FSK | 低 | 强 | 简单 | 距离优先、低速率 |
| BPSK | 中 | 中 | 中等 | 均衡方案 |
| QPSK | 高 | 中 | 较难 | 高速率需求 |
| 16QAM | 很高 | 弱 | 难 | 高速率+短距离 |

## 三、信道编码

### 3.1 汉明码(7,4)

```c
// (7,4)汉明码：4bit信息→7bit编码，纠1位错
// 生成矩阵G和校验矩阵H

// 编码
uint8_t hamming74_encode(uint8_t data4) {
    // data4: 4bit数据 d3d2d1d0
    uint8_t d = data4 & 0x0F;
    uint8_t p0 = ((d>>3) ^ (d>>2) ^ (d>>0)) & 1;  // 校验位
    uint8_t p1 = ((d>>3) ^ (d>>1) ^ (d>>0)) & 1;
    uint8_t p2 = ((d>>2) ^ (d>>1) ^ (d>>0)) & 1;
    return (p0<<6) | (p1<<5) | (d>>2<<4) | (p2<<3) | ((d>>1&1)<<2) | ((d&1)<<1) | 0;
    // 简化：按 [p0 p1 d3 p2 d2 d1 d0] 排列
}

// 解码+纠错
uint8_t hamming74_decode(uint8_t code7) {
    uint8_t s0 = ((code7>>6) ^ (code7>>4) ^ (code7>>2) ^ code7) & 1;
    uint8_t s1 = ((code7>>5) ^ (code7>>4) ^ (code7>>1) ^ code7) & 1;
    uint8_t s2 = ((code7>>3) ^ (code7>>2) ^ (code7>>1) ^ code7) & 1;
    uint8_t syndrome = (s2<<2) | (s1<<1) | s0;
    if (syndrome != 0) {
        code7 ^= (1 << (7 - syndrome));  // 纠正错误位
    }
    // 提取数据位
    return ((code7>>4&1)<<3) | ((code7>>2&1)<<2) | ((code7>>1&1)<<1) | (code7&1);
}
```

### 3.2 卷积码 (2,1,3)

```c
// 约束长度K=3，编码率1/2
// 生成多项式：g1=111(八进制7)，g2=101(八进制5)
// 移位寄存器实现

uint8_t conv_state = 0;  // 2bit状态

// 编码（逐bit输入）
uint8_t conv_encode(uint8_t input_bit) {
    conv_state = ((conv_state << 1) | input_bit) & 0x03;
    uint8_t g1 = ((conv_state >> 1) ^ (conv_state & 1) ^ input_bit) & 1;
    uint8_t g2 = ((conv_state >> 1) ^ input_bit) & 1;
    return (g1 << 1) | g2;  // 2bit输出
}

// Viterbi译码（简化版，4状态）
// 完整实现较长，建议用数组存储幸存路径
// 关键：对每个状态计算路径度量，选择最小度量的路径
```

### 3.3 CRC校验

```c
// CRC-16-CCITT: x^16 + x^12 + x^5 + 1
uint16_t crc16_ccitt(uint8_t *data, int len) {
    uint16_t crc = 0xFFFF;
    for (int i = 0; i < len; i++) {
        crc ^= (uint16_t)data[i] << 8;
        for (int j = 0; j < 8; j++) {
            if (crc & 0x8000) crc = (crc << 1) ^ 0x1021;
            else crc <<= 1;
        }
    }
    return crc;
}
```

### 3.4 交织（抗突发错误）

```c
#define ROWS 8
#define COLS 16

// 矩阵交织器
uint8_t interleave_buf[ROWS * COLS];

void interleave(uint8_t *in, uint8_t *out, int len) {
    // 按行写入
    for (int i = 0; i < len; i++) interleave_buf[i] = in[i];
    // 按列读出
    int idx = 0;
    for (int c = 0; c < COLS; c++) {
        for (int r = 0; r < ROWS; r++) {
            if (r * COLS + c < len) {
                out[idx++] = interleave_buf[r * COLS + c];
            }
        }
    }
}

// 解交织：按列写入，按行读出（反操作）
```

## 四、同步技术

### 4.1 位同步（码元定时恢复）

```c
// 超前-滞后门位同步器
// 核心：比较超前和滞后采样点的幅度，调整采样时刻

typedef struct {
    float early_sample;
    float late_sample;
    float prompt_sample;
    int   sample_offset;  // 当前采样偏移
    float loop_bw;        // 环路带宽
} BitSync;

float samples_per_sym = 48;  // 每符号采样数

void bitsync_update(BitSync *bs, float *signal, int sym_idx) {
    int base = sym_idx * (int)samples_per_sym + bs->sample_offset;
    bs->early_sample  = fabsf(signal[base + 1]);
    bs->late_sample   = fabsf(signal[base - 1]);
    bs->prompt_sample = fabsf(signal[base]);

    float error = bs->early_sample - bs->late_sample;
    // 正error→采样偏早，负error→采样偏晚
    bs->sample_offset -= (int)(bs->loop_bw * error);
    // 限制偏移范围
    if (bs->sample_offset > 4) bs->sample_offset = 4;
    if (bs->sample_offset < -4) bs->sample_offset = -4;
}
```

### 4.2 载波同步

```c
// Costas环（用于BPSK/QPSK载波恢复）
typedef struct {
    float phase;       // 估计的载波相位
    float freq;        // 估计的频率偏移
    float alpha;       // 相位环路增益
    float beta;        // 频率环路增益
} CostasLoop;

void costas_update(CostasLoop *cl, float sample, float t) {
    float I = sample * cosf(cl->phase);
    float Q = sample * sinf(cl->phase);

    // 低通滤波（积分清零）
    static float Ilp = 0, Qlp = 0;
    Ilp = 0.9f * Ilp + 0.1f * I;
    Qlp = 0.9f * Qlp + 0.1f * Q;

    // BPSK鉴相器：e = I*Q
    float error = Ilp * Qlp;

    // 环路滤波器
    cl->freq  += cl->beta * error;
    cl->phase += cl->alpha * error + cl->freq;
    cl->phase = fmodf(cl->phase, 2 * M_PI);
}
```

### 4.3 帧同步

```c
// 巴克码帧同步（常用巴克码：1110010 或 11100010010）
uint8_t barker7[] = {1,1,1,0,0,1,0};
#define BARKER_LEN 7
#define SYNC_THRESHOLD 5  // 允许1bit错误

int find_frame_sync(uint8_t *bits, int len) {
    for (int i = 0; i <= len - BARKER_LEN; i++) {
        int match = 0;
        for (int j = 0; j < BARKER_LEN; j++) {
            if (bits[i+j] == barker7[j]) match++;
        }
        if (match >= SYNC_THRESHOLD) {
            return i + BARKER_LEN;  // 帧数据起始位置
        }
    }
    return -1;  // 未找到
}
```

## 五、均衡技术

### 5.1 迫零均衡器（ZF）

```c
// 简单线性均衡，消除码间干扰
// 已知信道响应h[0..L-1]，求均衡器系数w[0..M-1]

#define EQ_LEN  5
float w[EQ_LEN];  // 均衡器系数

// 迫零：解线性方程组 H*w = delta（中间为1其余为0）
// 实际中常用LMS自适应代替
```

### 5.2 LMS自适应均衡器（推荐）

```c
#define EQ_ORDER  11   // 均衡器阶数（奇数）
float eq_coeff[EQ_ORDER] = {0};
float eq_buf[EQ_ORDER] = {0};
float mu = 0.01f;      // 步长（关键参数）

float lms_equalize(float input, float desired) {
    // 移入新样值
    for (int i = EQ_ORDER - 1; i > 0; i--) eq_buf[i] = eq_buf[i-1];
    eq_buf[0] = input;

    // 计算均衡器输出
    float y = 0;
    for (int i = 0; i < EQ_ORDER; i++) y += eq_coeff[i] * eq_buf[i];

    // 计算误差
    float error = desired - y;

    // 更新系数（LMS算法）
    for (int i = 0; i < EQ_ORDER; i++) {
        eq_coeff[i] += mu * error * eq_buf[i];
    }

    return y;
}

// mu取值：0.001~0.05
// 太大→不稳定，太小→收敛慢
// 经验：mu = 0.1 / (EQ_ORDER * signal_power)
```

## 六、完整通信链路设计模板

```
发射端：
  源数据 → 信道编码(CRC+汉明) → 交织 → 调制(BPSK/FSK) → 成形滤波 → DAC → PA

接收端：
  LNA → ADC → 带通滤波 → 载波同步 → 位同步 → 均衡 → 解调 → 解交织 → 信道解码 → 输出
```

### 6.1 电赛推荐方案（中等难度）

```
调制：2FSK（简单可靠）
编码：(7,4)汉明码+CRC-16
同步：过零检测（位同步）+ 巴克码（帧同步）
平台：STM32 + DAC/ADC 或 FPGA
传输速率：1~10kbps
距离：数米~数十米
```

### 6.2 高难度方案

```
调制：QPSK
编码：卷积码(2,1,7) + Viterbi译码 + 交织
同步：Costas环（载波）+ 超前滞后门（位）
均衡：LMS自适应均衡器
平台：FPGA（时序要求高）
传输速率：100kbps+
```

## 七、硬件实现要点

### 7.1 发射端

```
DAC选型：
  - 低速（<100kbps）：STM32内置DAC（12bit, 1MHz）
  - 高速：TLV5618、AD9760

功放：
  - 短距离：直接DAC输出+简单放大
  - 长距离：丙类功放（效率高）或AB类（线性好）

天线：
  - 短距离：PCB天线、弹簧天线
  - 长距离：八木天线、抛物面天线
```

### 7.2 接收端

```
LNA（低噪声放大器）：
  - 前端第一级，噪声系数<3dB
  - 常用：MAX2659、BFP420

ADC选型：
  - 低速：STM32内置ADC（12bit, 1MHz）
  - 高速：ADS8860、AD9226

带通滤波：
  - 有源：运放+RC网络
  - 无源：LC滤波器
  - 数字：FIR滤波器（推荐，灵活可调）
```

## 八、调试与测试

### 8.1 误码率测试

```c
// 发送已知测试序列，统计误码
#define TEST_LEN 10000
uint8_t test_tx[TEST_LEN];
uint8_t test_rx[TEST_LEN];

// 用伪随机序列
void generate_prbs(uint8_t *seq, int len) {
    uint32_t lfsr = 0xACE1u;  // 初始值
    for (int i = 0; i < len; i++) {
        seq[i] = lfsr & 1;
        lfsr = (lfsr >> 1) ^ (-(lfsr & 1) & 0xB400u);
    }
}

float calc_ber(uint8_t *tx, uint8_t *rx, int len) {
    int errors = 0;
    for (int i = 0; i < len; i++) {
        if (tx[i] != rx[i]) errors++;
    }
    return (float)errors / len;
}
```

### 8.2 常见问题

| 现象 | 原因 | 解决 |
|---|---|---|
| 误码率高 | 同步未建立 | 先调同步，加前导码 |
| 间歇性误码 | 突发干扰 | 加交织 |
| 信号幅度波动 | 多径/衰落 | 加均衡器 |
| 载波偏移 | 频偏 | Costas环补偿 |
| 完全无输出 | 帧同步丢失 | 降低同步门限，加冗余同步字 |
