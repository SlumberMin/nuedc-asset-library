## 3. 通信协议设计

### 3.1 传输层选择

| 方案 | 延迟 | 吞吐 | 可靠性 | 适用场景 |
|------|------|------|--------|---------|
| UDP组播 | <1ms | 高 | 低 | 实时状态广播 |
| TCP | ~5ms | 中 | 高 | 关键指令 |
| 共享内存 | <0.01ms | 极高 | 高 | 同机多进程 |
| UART | ~1ms | 低 | 中 | 近距离简单通信 |
| ROS2 DDS | ~2ms | 中 | 高 | 标准化方案 |

### 3.2 消息协议

```python
# 消息格式定义
import struct
import json
from dataclasses import dataclass, asdict
from enum import IntEnum

class MsgType(IntEnum):
    HEARTBEAT = 0x01
    VISION_RESULT = 0x02
    CTRL_CMD = 0x03
    SYNC_TIME = 0x04
    EMERGENCY = 0x05
    CONFIG_UPDATE = 0x06
    STATUS_REPORT = 0x07

# 二进制协议头
# | magic(4) | type(1) | seq(4) | timestamp(8) | src_id(1) | dst_id(1) | len(4) | payload(...) | crc(4) |
HEADER_FMT = '!IBQBBBI'  # 22 bytes
HEADER_SIZE = struct.calcsize(HEADER_FMT)
MAGIC = 0x4F504935  # "OPI5"

@dataclass
class Message:
    msg_type: int
    seq: int
    timestamp_ns: int
    src_id: int
    dst_id: int  # 0xFF = broadcast
    payload: bytes

    def pack(self) -> bytes:
        header = struct.pack(HEADER_FMT,
                             MAGIC, self.msg_type, self.seq,
                             self.timestamp_ns, self.src_id, self.dst_id,
                             len(self.payload))
        data = header + self.payload
        crc = crc32(data)
        return data + struct.pack('I', crc)

    @classmethod
    def unpack(cls, data: bytes) -> 'Message':
        if len(data) < HEADER_SIZE + 4:
            raise ValueError("数据太短")
        magic, msg_type, seq, ts, src, dst, plen = struct.unpack_from(HEADER_FMT, data)
        if magic != MAGIC:
            raise ValueError(f"Magic不匹配: {magic:#x}")
        payload = data[HEADER_SIZE:HEADER_SIZE + plen]
        crc_recv = struct.unpack_from('I', data, HEADER_SIZE + plen)[0]
        crc_calc = crc32(data[:HEADER_SIZE + plen])
        if crc_recv != crc_calc:
            raise ValueError("CRC校验失败")
        return cls(msg_type, seq, ts, src, dst, payload)

def crc32(data: bytes) -> int:
    import binascii
    return binascii.crc32(data) & 0xFFFFFFFF
```

### 3.3 时间同步

```python
# NTP-like 时间同步
class TimeSync:
    """简易时间同步"""

    def __init__(self):
        self.offset_ns = 0
        self.rtt_ns = 0
        self._samples = []

    def sync_request(self, udp_sock, master_addr):
        """从机发起同步"""
        t1 = time.monotonic_ns()
        msg = Message(MsgType.SYNC_TIME, 0, t1, SLAVE_ID, 0xFF, b'')
        udp_sock.sendto(msg.pack(), master_addr)
        return t1

    def sync_response(self, t1_ns, udp_sock, addr):
        """主机回复"""
        t2 = time.monotonic_ns()
        payload = struct.pack('QQ', t1_ns, t2)
        msg = Message(MsgType.SYNC_TIME, 0, t2, MASTER_ID, 0xFF, payload)
        udp_sock.sendto(msg.pack(), addr)

    def sync_complete(self, t1_ns, t2_ns, t4_ns):
        """从机完成同步"""
        t3 = time.monotonic_ns()  # 收到回复时间
        # offset = ((t2-t1) + (t3-t4)) / 2
        offset = ((t2_ns - t1_ns) + (t3_ns - t4_ns)) // 2
        rtt = (t4_ns - t1_ns) - (t3_ns - t2_ns)
        self._samples.append((offset, rtt))
        if len(self._samples) > 10:
            # 取RTT最小的样本的offset
            best = min(self._samples, key=lambda x: x[1])
            self.offset_ns = best[0]
            self.rtt_ns = best[1]
            self._samples = self._samples[-5:]
```

## 4. 多机通信管理器

```python
import socket
import threading
import time
from queue import Queue, Empty

class MultiNodeManager:
    """多机通信管理器"""

    def __init__(self, node_id: int, port: int = 5000):
        self.node_id = node_id
        self.port = port
        self.peers = {}           # {node_id: (ip, port, last_seen)}
        self._recv_queue = Queue(maxsize=1000)
        self._running = False

        # UDP socket
        self._sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self._sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self._sock.bind(('0.0.0.0', port))

        # 组播
        self._mcast_group = '239.255.0.1'
        self._setup_multicast()

    def _setup_multicast(self):
        """设置组播"""
        import struct
        mreq = struct.pack('4sL',
                           socket.inet_aton(self._mcast_group),
                           socket.INADDR_ANY)
        self._sock.setsockopt(socket.IPPROTO_IP,
                              socket.IP_ADD_MEMBERSHIP, mreq)

    def start(self):
        self._running = True
        self._recv_thread = threading.Thread(target=self._recv_loop, daemon=True)
        self._recv_thread.start()
        self._heartbeat_thread = threading.Thread(target=self._heartbeat_loop, daemon=True)
        self._heartbeat_thread.start()

    def stop(self):
        self._running = False
        self._sock.close()

    def send(self, msg: Message, target_id: int = 0xFF):
        """发送消息"""
        msg.src_id = self.node_id
        msg.dst_id = target_id
        data = msg.pack()

        if target_id == 0xFF:
            # 广播
            self._sock.sendto(data, (self._mcast_group, self.port))
        elif target_id in self.peers:
            # 单播
            addr = (self.peers[target_id][0], self.port)
            self._sock.sendto(data, addr)

    def recv(self, timeout: float = 0.1) -> Optional[Message]:
        """接收消息"""
        try:
            return self._recv_queue.get(timeout=timeout)
        except Empty:
            return None

    def _recv_loop(self):
        while self._running:
            try:
                data, addr = self._sock.recvfrom(65536)
                msg = Message.unpack(data)
                # 更新peer
                self.peers[msg.src_id] = (addr[0], addr[1], time.monotonic())
                self._recv_queue.put(msg)
            except Exception:
                pass

    def _heartbeat_loop(self):
        while self._running:
            hb = Message(MsgType.HEARTBEAT, 0, time.monotonic_ns(),
                         self.node_id, 0xFF, b'')
            self.send(hb)
            # 清理超时peer
            now = time.monotonic()
            expired = [nid for nid, (_, _, t) in self.peers.items()
                       if now - t > 5.0]
            for nid in expired:
                del self.peers[nid]
            time.sleep(1.0)

    def get_online_nodes(self) -> list:
        return list(self.peers.keys())
```

## 5. 协同策略

### 5.1 任务分配

```python
class TaskAllocator:
    """多机任务分配"""

    NODE_CAPABILITY = {
        1: ["vision", "decision", "control"],      # 主机
        2: ["vision"],                               # 视觉从机
        3: ["control"],                              # 执行从机
    }

    def allocate(self, task_type: str) -> int:
        for node_id, caps in self.NODE_CAPABILITY.items():
            if task_type in caps:
                return node_id
        return 1  # 默认主机处理

    def load_balance(self, task_type: str, node_loads: dict) -> int:
        """负载均衡"""
        candidates = [nid for nid, caps in self.NODE_CAPABILITY.items()
                      if task_type in caps]
        if not candidates:
            return 1
        return min(candidates, key=lambda nid: node_loads.get(nid, 0))
```

### 5.2 故障转移

```python
class FailoverManager:
    """故障转移管理"""

    def __init__(self, node_id: int, manager: MultiNodeManager):
        self.node_id = node_id
        self.manager = manager
        self._backup_roles = {}  # role -> backup_node_id

    def check_master_alive(self, timeout: float = 3.0) -> bool:
        """检查主机是否存活"""
        master = self.manager.peers.get(1)
        if master is None:
            return False
        return (time.monotonic() - master[2]) < timeout

    def elect_new_master(self) -> int:
        """选举新主机 (简单: 最小ID)"""
        online = self.manager.get_online_nodes()
        if not online:
            return self.node_id
        return min(online)

    def takeover(self):
        """接管主机角色"""
        print(f"节点{self.node_id}接管为主机")
        # 1. 更新角色
        # 2. 接管决策任务
        # 3. 通知其他节点
        msg = Message(MsgType.STATUS_REPORT, 0, time.monotonic_ns(),
                      self.node_id, 0xFF,
                      b'ROLE_CHANGE:MASTER')
        self.manager.send(msg)
```
