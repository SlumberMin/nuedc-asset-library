## 1. PID 控制器实现

### 1.1 增量式PID（最常用，电赛首选）

**来源**: 经典开源PID实现，适用于STM32

```c
/* 增量式PID控制器 - 适合电机速度控制 */
typedef struct {
    float Kp;           // 比例系数
    float Ki;           // 积分系数
    float Kd;           // 微分系数
    float target;       // 目标值
    float error[3];     // error[0]=当前, error[1]=上次, error[2]=上上次
    float output;       // 输出值
    float output_max;   // 输出上限
    float output_min;   // 输出下限
    float dead_zone;    // 死区
} PID_Incremental_t;

void PID_Incremental_Init(PID_Incremental_t *pid, float kp, float ki, float kd) {
    pid->Kp = kp;
    pid->Ki = ki;
    pid->Kd = kd;
    pid->target = 0;
    pid->error[0] = pid->error[1] = pid->error[2] = 0;
    pid->output = 0;
    pid->output_max = 1000;
    pid->output_min = -1000;
    pid->dead_zone = 5;
}

float PID_Incremental_Calc(PID_Incremental_t *pid, float measured) {
    pid->error[2] = pid->error[1];
    pid->error[1] = pid->error[0];
    pid->error[0] = pid->target - measured;
    
    // 死区处理
    if (fabsf(pid->error[0]) < pid->dead_zone) {
        pid->error[0] = 0;
    }
    
    // 增量式PID公式
    float increment = pid->Kp * (pid->error[0] - pid->error[1])
                    + pid->Ki * pid->error[0]
                    + pid->Kd * (pid->error[0] - 2.0f * pid->error[1] + pid->error[2]);
    
    pid->output += increment;
    
    // 输出限幅
    if (pid->output > pid->output_max)  pid->output = pid->output_max;
    if (pid->output < pid->output_min)  pid->output = pid->output_min;
    
    return pid->output;
}
```

### 1.2 位置式PID（适合位置/角度控制）

```c
/* 位置式PID控制器 - 适合舵机角度/位置控制 */
typedef struct {
    float Kp, Ki, Kd;
    float target;
    float integral;
    float integral_max;  // 积分限幅（抗积分饱和）
    float prev_error;
    float output;
    float output_max, output_min;
} PID_Position_t;

float PID_Position_Calc(PID_Position_t *pid, float measured) {
    float error = pid->target - measured;
    
    // 积分累加（带限幅）
    pid->integral += error;
    if (pid->integral > pid->integral_max)  pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max) pid->integral = -pid->integral_max;
    
    // 微分（对误差微分）
    float derivative = error - pid->prev_error;
    
    // 位置式PID公式
    pid->output = pid->Kp * error 
                + pid->Ki * pid->integral 
                + pid->Kd * derivative;
    
    pid->prev_error = error;
    
    // 输出限幅
    if (pid->output > pid->output_max)  pid->output = pid->output_max;
    if (pid->output < pid->output_min)  pid->output = pid->output_min;
    
    return pid->output;
}
```

### 1.3 串级PID（电赛高频使用）

```c
/* 串级PID - 外环位置 + 内环速度 */
typedef struct {
    PID_Position_t outer;   // 外环（位置/角度）
    PID_Incremental_t inner; // 内环（速度）
} PID_Cascade_t;

float PID_Cascade_Calc(PID_Cascade_t *pid, float outer_measured, float inner_measured) {
    // 外环输出作为内环目标
    pid->inner.target = PID_Position_Calc(&pid->outer, outer_measured);
    // 内环输出驱动执行器
    return PID_Incremental_Calc(&pid->inner, inner_measured);
}
```

```c
/* 定时器回调：周期性计算转速 */
// 假设每50ms调用一次，编码器线数=13，减速比=30
float Motor_GetSpeed(TIM_HandleTypeDef *htim) {
    static int16_t last_count = 0;
    int16_t current_count = (int16_t)__HAL_TIM_GET_COUNTER(htim);
    int16_t diff = current_count - last_count;
    last_count = current_count;
    
    // 转速 = 脉冲数 / (编码器线数 × 倍频数 × 减速比) × 采样频率 × 60
    // RPM = diff / (13 * 4 * 30) * (1000/50) * 60
    float rpm = (float)diff / 1560.0f * 20.0f * 60.0f;
    return rpm;
}
```

## 3. TB6612 电机驱动

```c
/* TB6612FNG 双路电机驱动 */
// 来源: STM32电赛通用代码库

typedef enum {
    MOTOR_LEFT = 0,
    MOTOR_RIGHT = 1
} Motor_ID_t;

typedef struct {
    GPIO_TypeDef *in1_port;
    uint16_t in1_pin;
    GPIO_TypeDef *in2_port;
    uint16_t in2_pin;
    TIM_HandleTypeDef *pwm_htim;
    uint32_t pwm_channel;
} Motor_Config_t;

Motor_Config_t motors[2] = {
    {GPIOA, GPIO_PIN_4, GPIOA, GPIO_PIN_5, &htim2, TIM_CHANNEL_1},  // 左电机
    {GPIOA, GPIO_PIN_6, GPIOA, GPIO_PIN_7, &htim2, TIM_CHANNEL_2},  // 右电机
};

void Motor_SetSpeed(Motor_ID_t id, int16_t speed) {
    Motor_Config_t *m = &motors[id];
    
    // 限幅
    if (speed > 1000) speed = 1000;
    if (speed < -1000) speed = -1000;
    
    // 方向控制
    if (speed >= 0) {
        HAL_GPIO_WritePin(m->in1_port, m->in1_pin, GPIO_PIN_SET);
        HAL_GPIO_WritePin(m->in2_port, m->in2_pin, GPIO_PIN_RESET);
    } else {
        HAL_GPIO_WritePin(m->in1_port, m->in1_pin, GPIO_PIN_RESET);
        HAL_GPIO_WritePin(m->in2_port, m->in2_pin, GPIO_PIN_SET);
        speed = -speed;
    }
    
    // PWM占空比
    __HAL_TIM_SET_COMPARE(m->pwm_htim, m->pwm_channel, (uint16_t)speed);
}

void Motor_Stop(Motor_ID_t id) {
    Motor_Config_t *m = &motors[id];
    HAL_GPIO_WritePin(m->in1_port, m->in1_pin, GPIO_PIN_RESET);
    HAL_GPIO_WritePin(m->in2_port, m->in2_pin, GPIO_PIN_RESET);
    __HAL_TIM_SET_COMPARE(m->pwm_htim, m->pwm_channel, 0);
}
```

## 4. MPU6050 DMP 姿态解算

```c
/* MPU6050 DMP 初始化和读取 */
// 来源: invensense/Embedded-MotionDriver

#include "inv_mpu.h"
#include "inv_mpu_dmp_motion_driver.h"

typedef struct {
    float pitch;    // 俯仰角
    float roll;     // 横滚角
    float yaw;      // 偏航角
} EulerAngle_t;

uint8_t MPU6050_DMP_Init(void) {
    if (mpu_init() != 0) return 1;
    if (mpu_set_sensors(INV_XYZ_GYRO | INV_XYZ_ACCEL) != 0) return 2;
    if (mpu_set_sample_rate(100) != 0) return 3;  // 100Hz采样
    if (dmp_load_motion_driver_firmware() != 0) return 4;
    if (dmp_set_feature_inv(DMP_FEATURE_6X_LP_QUAT | DMP_FEATURE_GYRO_CAL) != 0) return 5;
    if (dmp_enable_feature() != 0) return 6;
    if (mpu_set_dmp_state(1) != 0) return 7;
    return 0;
}

uint8_t MPU6050_DMP_Read(EulerAngle_t *angle) {
    short gyro[3], accel[3], sensors;
    unsigned long timestamp;
    long quat[4];
    
    if (dmp_read_fifo(gyro, accel, quat, &timestamp, &sensors, 0) != 0) {
        return 1;  // 数据未就绪
    }
    
    // 四元数转欧拉角
    float q0 = quat[0] / 1073741824.0f;
    float q1 = quat[1] / 1073741824.0f;
    float q2 = quat[2] / 1073741824.0f;
    float q3 = quat[3] / 1073741824.0f;
    
    angle->pitch = asinf(-2.0f * q1 * q3 + 2.0f * q0 * q2) * 57.2958f;
    angle->roll  = atan2f(2.0f * q2 * q3 + 2.0f * q0 * q1, 
                          -2.0f * q1*q1 - 2.0f * q2*q2 + 1.0f) * 57.2958f;
    angle->yaw   = atan2f(2.0f * (q1*q2 + q0*q3), 
                          q0*q0 + q1*q1 - q2*q2 - q3*q3) * 57.2958f;
    return 0;
}
```

```cpp
/* RGA 硬件加速图像缩放 + 色彩转换 */
// 来源: airockchip/librga

#include "im2d.h"
#include "RgaUtils.h"

// RGA硬件缩放（零CPU开销）
int RGA_Resize(cv::Mat& src, cv::Mat& dst, int dst_w, int dst_h) {
    // 分配DMA内存（RGA需要物理连续内存）
    int src_buf_size = src.cols * src.rows * 3;
    int dst_buf_size = dst_w * dst_h * 3;
    
    void *src_buf = malloc(src_buf_size);
    void *dst_buf = malloc(dst_buf_size);
    
    memcpy(src_buf, src.data, src_buf_size);
    
    // 设置RGA源和目标
    rga_buffer_t src_img = wrapbuffer_virtualaddr(src_buf, src.cols, src.rows, 
                                                   RK_FORMAT_RGB_888);
    rga_buffer_t dst_img = wrapbuffer_virtualaddr(dst_buf, dst_w, dst_h, 
                                                   RK_FORMAT_RGB_888);
    
    // 执行硬件缩放
    int ret = imresize(src_img, dst_img);
    if (ret != IM_STATUS_SUCCESS) {
        printf("RGA resize failed: %s\n", imStrError((IM_STATUS)ret));
        free(src_buf);
        free(dst_buf);
        return -1;
    }
    
    dst = cv::Mat(dst_h, dst_w, CV_8UC3, dst_buf);
    free(src_buf);
    // 注意：dst_buf需要在使用完后free
    return 0;
}

// RGA色彩转换（YUV→RGB）
int RGA_CvtColor(void *yuv_buf, int w, int h, cv::Mat& rgb_out) {
    int rgb_size = w * h * 3;
    void *rgb_buf = malloc(rgb_size);
    
    rga_buffer_t src_img = wrapbuffer_virtualaddr(yuv_buf, w, h, 
                                                   RK_FORMAT_YCbCr_420_SP);
    rga_buffer_t dst_img = wrapbuffer_virtualaddr(rgb_buf, w, h, 
                                                   RK_FORMAT_RGB_888);
    
    int ret = imcvtcolor(src_img, dst_img, 
                          RK_FORMAT_YCbCr_420_SP, RK_FORMAT_RGB_888);
    
    rgb_out = cv::Mat(h, w, CV_8UC3, rgb_buf);
    return (ret == IM_STATUS_SUCCESS) ? 0 : -1;
}
```

### 7.2 NEON优化的图像处理

```cpp
/* ARM NEON 优化的灰度转换 */
// 比OpenCV默认实现快2-3倍

#include <arm_neon.h>

void BGR2GRAY_NEON(const uint8_t* bgr, uint8_t* gray, int pixels) {
    // BT.601权重: Y = 0.299R + 0.587G + 0.114B
    // 定点化: Y = (77R + 150G + 29B) >> 8
    const uint8x8_t r_weight = vdup_n_u8(77);
    const uint8x8_t g_weight = vdup_n_u8(150);
    const uint8x8_t b_weight = vdup_n_u8(29);
    
    int i;
    for (i = 0; i < pixels - 7; i += 8) {
        // 加载24字节BGR数据
        uint8x8x3_t bgr_data = vld3_u8(bgr + i * 3);
        
        // 加权求和
        uint16x8_t sum = vmull_u8(bgr_data.val[2], r_weight);  // R
        sum = vmlal_u8(sum, bgr_data.val[1], g_weight);         // + G
        sum = vmlal_u8(sum, bgr_data.val[0], b_weight);         // + B
        
        // 右移8位
        uint8x8_t result = vshrn_n_u16(sum, 8);
        
        // 存储
        vst1_u8(gray + i, result);
    }
    
    // 处理剩余像素
    for (; i < pixels; i++) {
        uint8_t b = bgr[i*3], g = bgr[i*3+1], r = bgr[i*3+2];
        gray[i] = (77*r + 150*g + 29*b) >> 8;
    }
}

/* ARM NEON 优化的阈值二值化 */
void Threshold_NEON(const uint8_t* src, uint8_t* dst, int pixels, uint8_t thresh) {
    uint8x8_t thresh_vec = vdup_n_u8(thresh);
    uint8x8_t max_val = vdup_n_u8(255);
    uint8x8_t zero = vdup_n_u8(0);
    
    int i;
    for (i = 0; i < pixels - 7; i += 8) {
        uint8x8_t data = vld1_u8(src + i);
        uint8x8_t mask = vcgt_u8(data, thresh_vec);
        uint8x8_t result = vbsl_u8(mask, max_val, zero);
        vst1_u8(dst + i, result);
    }
    
    for (; i < pixels; i++) {
        dst[i] = (src[i] > thresh) ? 255 : 0;
    }
}
```

## 8. 串口通信协议（电赛通用）

```c
/* 串口通信协议 - 适用于STM32与OpenMV/树香派通信 */
// 帧格式: [帧头][长度][命令][数据...][校验][帧尾]

#define FRAME_HEADER  0xAA
#define FRAME_TAIL    0x55
#define MAX_DATA_LEN  32

typedef struct {
    uint8_t header;          // 帧头
    uint8_t length;          // 数据长度
    uint8_t command;         // 命令字
    uint8_t data[MAX_DATA_LEN];  // 数据区
    uint8_t checksum;        // 校验和
    uint8_t tail;            // 帧尾
} UART_Frame_t;

// 发送数据帧
void UART_SendFrame(UART_HandleTypeDef *huart, uint8_t cmd, uint8_t *data, uint8_t len) {
    uint8_t buf[MAX_DATA_LEN + 6];
    buf[0] = FRAME_HEADER;
    buf[1] = len;
    buf[2] = cmd;
    
    uint8_t checksum = len ^ cmd;
    for (int i = 0; i < len; i++) {
        buf[3 + i] = data[i];
        checksum ^= data[i];
    }
    buf[3 + len] = checksum;
    buf[4 + len] = FRAME_TAIL;
    
    HAL_UART_Transmit(huart, buf, len + 5, 100);
}

// 接收状态机
typedef enum {
    STATE_HEADER,
    STATE_LENGTH,
    STATE_COMMAND,
    STATE_DATA,
    STATE_CHECKSUM,
    STATE_TAIL
} UART_State_t;

void UART_ParseByte(UART_Frame_t *frame, uint8_t byte) {
    static UART_State_t state = STATE_HEADER;
    static uint8_t data_index = 0;
    static uint8_t checksum = 0;
    
    switch (state) {
        case STATE_HEADER:
            if (byte == FRAME_HEADER) state = STATE_LENGTH;
            break;
        case STATE_LENGTH:
            frame->length = byte;
            checksum = byte;
            state = STATE_COMMAND;
            break;
        case STATE_COMMAND:
            frame->command = byte;
            checksum ^= byte;
            data_index = 0;
            state = (frame->length > 0) ? STATE_DATA : STATE_CHECKSUM;
            break;
        case STATE_DATA:
            frame->data[data_index++] = byte;
            checksum ^= byte;
            if (data_index >= frame->length) state = STATE_CHECKSUM;
            break;
        case STATE_CHECKSUM:
            if (byte == checksum) state = STATE_TAIL;
            else state = STATE_HEADER;  // 校验失败，重新同步
            break;
        case STATE_TAIL:
            if (byte == FRAME_TAIL) {
                // 帧接收成功，处理数据
                // ProcessFrame(frame);
            }
            state = STATE_HEADER;
            break;
    }
}
```

```c
/* STM32 多通道ADC DMA采集 */
// 适用于电赛中的多路模拟传感器读取

#define ADC_CHANNELS  4
#define ADC_SAMPLES   16  // 每通道16次采样取平均

uint16_t adc_raw[ADC_CHANNELS * ADC_SAMPLES];  // DMA缓冲区
uint16_t adc_filtered[ADC_CHANNELS];            // 滤波后结果

void ADC_DMA_Init(void) {
    HAL_ADC_Start_DMA(&hadc1, (uint32_t*)adc_raw, ADC_CHANNELS * ADC_SAMPLES);
}

// 在DMA完成回调中处理
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef *hadc) {
    for (int ch = 0; ch < ADC_CHANNELS; ch++) {
        uint32_t sum = 0;
        for (int s = 0; s < ADC_SAMPLES; s++) {
            sum += adc_raw[s * ADC_CHANNELS + ch];
        }
        adc_filtered[ch] = sum / ADC_SAMPLES;
    }
}

// 将ADC值转换为电压
float ADC_GetVoltage(uint8_t channel) {
    return (float)adc_filtered[channel] / 4096.0f * 3.3f;
}
```

```c
// PWM周期20ms (50Hz)，脉宽0.5ms-2.5ms对应0-180度

#define SERVO_MIN_PULSE  500   // 0.5ms = 500us
#define SERVO_MAX_PULSE  2500  // 2.5ms = 2500us
#define SERVO_PERIOD     20000 // 20ms = 20000us

void Servo_SetAngle(TIM_HandleTypeDef *htim, uint32_t channel, float angle) {
    // 角度限幅
    if (angle < 0) angle = 0;
    if (angle > 180) angle = 180;
    
    // 线性映射: 角度 → 脉宽
    uint16_t pulse = SERVO_MIN_PULSE + 
                     (uint16_t)((angle / 180.0f) * (SERVO_MAX_PULSE - SERVO_MIN_PULSE));
    
    // 设置比较值（假设TIM时钟分频后1us计数一次）
    __HAL_TIM_SET_COMPARE(htim, channel, pulse);
}

// 平滑移动（避免舵机抖动）
void Servo_MoveSmooth(TIM_HandleTypeDef *htim, uint32_t channel, 
                      float current, float target, uint16_t step_ms) {
    float step = (target > current) ? 1.0f : -1.0f;
    while (fabsf(current - target) > 1.0f) {
        current += step;
        Servo_SetAngle(htim, channel, current);
        HAL_Delay(step_ms);
    }
    Servo_SetAngle(htim, channel, target);
}
```

## 11. 按键消抖（状态机方式）

```c
/* 按键状态机消抖 - 适用于电赛菜单切换 */
typedef enum {
    KEY_IDLE,
    KEY_PRESS,
    KEY_CONFIRM,
    KEY_RELEASE
} Key_State_t;

typedef struct {
    GPIO_TypeDef *port;
    uint16_t pin;
    Key_State_t state;
    uint32_t timer;
    uint8_t pressed;       // 确认按下标志
    uint8_t long_pressed;  // 长按标志
} Key_t;

void Key_Scan(Key_t *key) {
    uint8_t pin_state = HAL_GPIO_ReadPin(key->port, key->pin);
    
    switch (key->state) {
        case KEY_IDLE:
            if (pin_state == GPIO_PIN_RESET) {
                key->state = KEY_PRESS;
                key->timer = 0;
            }
            break;
            
        case KEY_PRESS:
            if (pin_state == GPIO_PIN_RESET) {
                key->timer++;
                if (key->timer >= 20) {  // 20ms消抖
                    key->state = KEY_CONFIRM;
                    key->pressed = 1;
                    key->timer = 0;
                }
            } else {
                key->state = KEY_IDLE;
            }
            break;
            
        case KEY_CONFIRM:
            if (pin_state == GPIO_PIN_SET) {
                key->state = KEY_RELEASE;
            } else {
                key->timer++;
                if (key->timer >= 1000) {  // 1s长按
                    key->long_pressed = 1;
                }
            }
            break;
            
        case KEY_RELEASE:
            key->state = KEY_IDLE;
            break;
    }
}
```
