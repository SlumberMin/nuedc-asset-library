### 1.1 2021年F题 智能送药小车 - 全局状态机

```c
/*
 * 智能送药小车 - 任务状态机（国一作品核心框架）
 * 
 * 状态转换图：
 * INIT → STANDBY → LINE_FOLLOW → INTERSECTION_DETECT
 *                                    ↓
 *                              ROOM_IDENTIFY → DELIVER → RETURN → STANDBY
 */

typedef enum {
    TASK_INIT,              // 系统初始化
    TASK_STANDBY,           // 待命状态
    TASK_LINE_FOLLOW,       // 循线行驶
    TASK_INTERSECTION,      // 路口检测
    TASK_ROOM_IDENTIFY,     // 病房识别
    TASK_TURN,              // 转弯执行
    TASK_DELIVER,           // 送药动作
    TASK_RETURN,            // 返回药房
    TASK_ERROR,             // 错误处理
    TASK_COMPLETE           // 任务完成
} TaskState;

typedef struct {
    TaskState state;
    TaskState last_state;
    uint32_t state_timer;       // 状态持续时间
    uint32_t timeout;           // 状态超时时间
    int target_room;            // 目标病房号
    int current_position;       // 当前位置
    int intersection_count;     // 路口计数
    bool medicine_delivered;    // 送药完成标志
} TaskContext;

// 状态处理函数表
typedef void (*StateHandler)(TaskContext* ctx);

void state_init(TaskContext* ctx) {
    // 系统初始化
    motor_init();
    servo_init();
    camera_init();
    ultrasonic_init();
    
    // 等待传感器稳定
    delay_ms(1000);
    
    // 读取目标病房号（按键输入或串口指令）
    ctx->target_room = get_target_room();
    ctx->intersection_count = 0;
    ctx->medicine_delivered = false;
    
    ctx->state = TASK_STANDBY;
}

void state_line_follow(TaskContext* ctx) {
    // 获取循线偏移量
    float offset = get_line_offset();  // 来自OpenMV
    
    // 前瞻PID循线
    float steer = pid_calc_offset(&line_pid, offset);
    
    // 差速控制
    float base_speed = 40.0f;  // 基础速度
    float left_speed = base_speed + steer;
    float right_speed = base_speed - steer;
    
    motor_set_speed(MOTOR_LEFT, left_speed);
    motor_set_speed(MOTOR_RIGHT, right_speed);
    
    // 检测路口（偏移量突变 + 传感器组合）
    if (detect_intersection()) {
        ctx->state = TASK_INTERSECTION;
        ctx->state_timer = 0;
    }
    
    // 超时保护
    if (ctx->state_timer > 30000) {  // 30秒超时
        ctx->state = TASK_ERROR;
    }
}

void state_intersection(TaskContext* ctx) {
    // 路口处理
    static uint8_t turn_direction = 0;
    
    // 减速
    motor_set_speed(MOTOR_LEFT, 20.0f);
    motor_set_speed(MOTOR_RIGHT, 20.0f);
    
    // 识别路口类型
    IntersectionType type = identify_intersection();
    
    switch (type) {
        case INTERSECTION_T_LEFT:
            turn_direction = TURN_LEFT;
            break;
        case INTERSECTION_T_RIGHT:
            turn_direction = TURN_RIGHT;
            break;
        case INTERSECTION_CROSS:
            // 十字路口，根据目标决定方向
            turn_direction = decide_turn_direction(ctx);
            break;
        case INTERSECTION_END:
            // 到达尽头，需要掉头
            turn_direction = TURN_AROUND;
            break;
    }
    
    ctx->intersection_count++;
    ctx->state = TASK_TURN;
}

void state_room_identify(TaskContext* ctx) {
    // 识别病房号
    int room_number = camera_recognize_number();
    
    if (room_number == ctx->target_room) {
        // 找到目标病房
        ctx->state = TASK_DELIVER;
        motor_stop();
    } else {
        // 继续寻找
        ctx->state = TASK_LINE_FOLLOW;
    }
}

void state_deliver(TaskContext* ctx) {
    // 送药动作序列
    motor_stop();
    delay_ms(500);
    
    // 舵机执行送药动作
    servo_set_angle(SERVO_ARM, 90);   // 机械臂伸出
    delay_ms(1000);
    servo_set_angle(SERVO_GRIP, 0);   // 松开夹爪
    delay_ms(500);
    servo_set_angle(SERVO_ARM, 0);    // 机械臂收回
    delay_ms(1000);
    
    ctx->medicine_delivered = true;
    ctx->state = TASK_RETURN;
}

// 状态机主循环
void task_manager_update(TaskContext* ctx) {
    static StateHandler handlers[] = {
        state_init,
        state_standby,
        state_line_follow,
        state_intersection,
        state_room_identify,
        state_turn,
        state_deliver,
        state_return,
        state_error,
        state_complete
    };
    
    // 执行当前状态处理函数
    handlers[ctx->state](ctx);
    
    // 更新状态计时器
    ctx->state_timer++;
}
```

### 1.2 2019年B题 巡线机器人 - 动态前瞻PID

```c
/*
 * 巡线机器人 - 动态前瞻PID控制器
 * 
 * 核心思想：
 * 1. 直道：前瞻距离远，响应平滑，提高速度
 * 2. 弯道：前瞻距离近，响应迅速，保证精度
 * 3. 根据曲率估计动态调整前瞻距离
 */

typedef struct {
    // 基础PID参数
    float kp;
    float ki;
    float kd;
    
    // 前瞻参数
    float preview_min;      // 最小前瞻距离（像素）
    float preview_max;      // 最大前瞻距离（像素）
    float preview_current;  // 当前前瞻距离
    
    // 曲率估计
    float curvature;        // 当前路径曲率
    float curvature_smooth; // 平滑后的曲率
    float curvature_alpha;  // 平滑系数
    
    // PID内部状态
    float error_last;
    float integral;
    float integral_max;
    float output_max;
} PreviewPID;

void preview_pid_init(PreviewPID* pid) {
    pid->kp = 2.5f;
    pid->ki = 0.3f;
    pid->kd = 1.2f;
    pid->preview_min = 30.0f;
    pid->preview_max = 120.0f;
    pid->curvature_alpha = 0.3f;
    pid->integral_max = 1000.0f;
    pid->output_max = 100.0f;
    pid->error_last = 0;
    pid->integral = 0;
}

// 估计路径曲率（基于连续三帧偏移量）
float estimate_curvature(float offset_current, float offset_prev1, float offset_prev2) {
    // 曲率 ≈ 二阶差分
    float d1 = offset_current - offset_prev1;
    float d2 = offset_prev1 - offset_prev2;
    float curvature = (d1 - d2);  // 二阶差分近似曲率
    return curvature;
}

float preview_pid_calc(PreviewPID* pid, float offset_current, 
                       float offset_preview, float curvature) {
    // 1. 更新曲率估计
    pid->curvature = curvature;
    pid->curvature_smooth = pid->curvature_alpha * curvature 
                          + (1 - pid->curvature_alpha) * pid->curvature_smooth;
    
    // 2. 动态调整前瞻距离
    // 直道（曲率小）→ 前瞻远
    // 弯道（曲率大）→ 前瞻近
    float curvature_abs = fabs(pid->curvature_smooth);
    pid->preview_current = pid->preview_max 
                         - (pid->preview_max - pid->preview_min) 
                         * (curvature_abs / 100.0f);  // 100为曲率归一化系数
    
    // 限幅
    if (pid->preview_current < pid->preview_min) 
        pid->preview_current = pid->preview_min;
    if (pid->preview_current > pid->preview_max) 
        pid->preview_current = pid->preview_max;
    
    // 3. 计算加权误差
    // 当前偏移权重：直道0.3/弯道0.7
    // 前瞻偏移权重：直道0.7/弯道0.3
    float weight_current = 0.3f + 0.4f * (curvature_abs / 100.0f);
    float weight_preview = 1.0f - weight_current;
    
    float error = weight_current * offset_current + weight_preview * offset_preview;
    
    // 4. PID计算
    pid->integral += error;
    
    // 积分限幅
    if (pid->integral > pid->integral_max) pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max) pid->integral = -pid->integral_max;
    
    float derivative = error - pid->error_last;
    pid->error_last = error;
    
    float output = pid->kp * error 
                 + pid->ki * pid->integral 
                 + pid->kd * derivative;
    
    // 输出限幅
    if (output > pid->output_max) output = pid->output_max;
    if (output < -pid->output_max) output = -pid->output_max;
    
    return output;
}
```

### 1.3 2017年C题 磁悬浮系统 - LQR+卡尔曼

```c
/*
 * 磁悬浮系统 - LQR最优控制 + 卡尔曼滤波
 * 
 * 状态空间模型：
 * x = [位置, 速度, 电流]^T
 * u = [电压]^T
 * y = [位置, 电流]^T
 * 
 * 状态方程：dx/dt = A*x + B*u
 * 观测方程：y = C*x + D*u
 */

#include <math.h>

#define STATE_DIM 3
#define INPUT_DIM 1
#define OUTPUT_DIM 2

typedef struct {
    float A[STATE_DIM][STATE_DIM];  // 系统矩阵
    float B[STATE_DIM][INPUT_DIM];  // 输入矩阵
    float C[OUTPUT_DIM][STATE_DIM]; // 观测矩阵
    float K[INPUT_DIM][STATE_DIM];  // LQR增益矩阵
    float x[STATE_DIM];             // 状态估计
    float P[STATE_DIM][STATE_DIM];  // 误差协方差
    float Q[STATE_DIM][STATE_DIM];  // 过程噪声
    float R[OUTPUT_DIM][OUTPUT_DIM];// 观测噪声
    float dt;                       // 采样周期
} LQRKalmanSystem;

void system_init(LQRKalmanSystem* sys) {
    // 系统参数（根据实际系统辨识得到）
    float m = 0.05f;    // 小球质量 50g
    float k = 0.001f;   // 电磁力系数
    float c = 0.01f;    // 阻尼系数
    float L = 0.01f;    // 电感
    float R_coil = 1.0f;// 线圈电阻
    
    // 连续系统矩阵
    // dx/dt = [0, 1, 0; 0, -c/m, k/m; 0, -k/L, -R/L] * x + [0; 0; 1/L] * u
    sys->A[0][0] = 0;  sys->A[0][1] = 1;     sys->A[0][2] = 0;
    sys->A[1][0] = 0;  sys->A[1][1] = -c/m;  sys->A[1][2] = k/m;
    sys->A[2][0] = 0;  sys->A[2][1] = -k/L;  sys->A[2][2] = -R_coil/L;
    
    sys->B[0][0] = 0;
    sys->B[1][0] = 0;
    sys->B[2][0] = 1/L;
    
    // 观测矩阵（直接观测位置和电流）
    sys->C[0][0] = 1; sys->C[0][1] = 0; sys->C[0][2] = 0;
    sys->C[1][0] = 0; sys->C[1][1] = 0; sys->C[1][2] = 1;
    
    // 采样周期
    sys->dt = 0.001f;  // 1ms
    
    // 过程噪声（根据实际调试）
    for (int i = 0; i < STATE_DIM; i++)
        for (int j = 0; j < STATE_DIM; j++)
            sys->Q[i][j] = (i == j) ? 0.01f : 0;
    
    // 观测噪声
    sys->R[0][0] = 0.1f;  // 位置观测噪声
    sys->R[0][1] = 0;
    sys->R[1][0] = 0;
    sys->R[1][1] = 0.01f; // 电流观测噪声
    
    // LQR权重矩阵（通过离线计算得到K）
    // Q_lqr = diag(100, 10, 1)  R_lqr = [0.1]
    // 求解Riccati方程得到K
    // 这里直接给出计算好的K值
    sys->K[0][0] = -31.62f;  // 位置反馈增益
    sys->K[0][1] = -12.45f;  // 速度反馈增益
    sys->K[0][2] = -2.15f;   // 电流反馈增益
    
    // 初始化状态
    for (int i = 0; i < STATE_DIM; i++) sys->x[i] = 0;
    for (int i = 0; i < STATE_DIM; i++)
        for (int j = 0; j < STATE_DIM; j++)
            sys->P[i][j] = (i == j) ? 1.0f : 0;
}

// 卡尔曼滤波预测
void kalman_predict(LQRKalmanSystem* sys, float u) {
    float x_pred[STATE_DIM];
    
    // x_pred = A*x + B*u（欧拉离散化）
    for (int i = 0; i < STATE_DIM; i++) {
        x_pred[i] = 0;
        for (int j = 0; j < STATE_DIM; j++) {
            // 离散化: A_d = I + A*dt
            float A_d = (i == j) ? 1.0f + sys->A[i][j] * sys->dt 
                                 : sys->A[i][j] * sys->dt;
            x_pred[i] += A_d * sys->x[j];
        }
        x_pred[i] += sys->B[i][0] * u * sys->dt;
    }
    
    // 更新状态
    for (int i = 0; i < STATE_DIM; i++) sys->x[i] = x_pred[i];
    
    // P_pred = A_d * P * A_d' + Q（简化处理）
    for (int i = 0; i < STATE_DIM; i++)
        for (int j = 0; j < STATE_DIM; j++)
            sys->P[i][j] += sys->Q[i][j] * sys->dt;
}

// 卡尔曼滤波更新
void kalman_update(LQRKalmanSystem* sys, float y[OUTPUT_DIM]) {
    // 计算新息
    float innovation[OUTPUT_DIM];
    for (int i = 0; i < OUTPUT_DIM; i++) {
        innovation[i] = y[i];
        for (int j = 0; j < STATE_DIM; j++)
            innovation[i] -= sys->C[i][j] * sys->x[j];
    }
    
    // 计算卡尔曼增益（简化计算）
    // K = P*C'*(C*P*C' + R)^-1
    float K[STATE_DIM][OUTPUT_DIM];
    for (int i = 0; i < STATE_DIM; i++) {
        for (int j = 0; j < OUTPUT_DIM; j++) {
            K[i][j] = sys->P[i][i] * sys->C[j][i] 
                     / (sys->P[i][i] * sys->C[j][i] * sys->C[j][i] + sys->R[j][j]);
        }
    }
    
    // 更新状态估计
    for (int i = 0; i < STATE_DIM; i++)
        for (int j = 0; j < OUTPUT_DIM; j++)
            sys->x[i] += K[i][j] * innovation[j];
    
    // 更新协方差（Joseph形式，数值稳定）
    for (int i = 0; i < STATE_DIM; i++)
        for (int j = 0; j < STATE_DIM; j++)
            sys->P[i][j] *= (1 - K[i][0] * sys->C[0][j] - K[i][1] * sys->C[1][j]);
}

// LQR控制计算
float lqr_control(LQRKalmanSystem* sys) {
    float u = 0;
    
    // u = -K * x
    for (int i = 0; i < STATE_DIM; i++) {
        u -= sys->K[0][i] * sys->x[i];
    }
    
    // 输出限幅
    float u_max = 12.0f;  // 最大电压
    if (u > u_max) u = u_max;
    if (u < -u_max) u = -u_max;
    
    return u;
}

// 系统主循环（1ms周期）
void control_loop(void) {
    static LQRKalmanSystem sys;
    static int initialized = 0;
    
    if (!initialized) {
        system_init(&sys);
        initialized = 1;
    }
    
    // 读取传感器
    float position = read_position_sensor();  // 位置传感器
    float current = read_current_sensor();    // 电流传感器
    
    // 卡尔曼滤波
    float y[2] = {position, current};
    kalman_predict(&sys, 0);  // 预测（u在上一周期）
    kalman_update(&sys, y);   // 更新
    
    // LQR控制
    float control_voltage = lqr_control(&sys);
    
    // 输出到驱动电路
    set_pwm_voltage(control_voltage);
}
```

### 1.4 2020年E题 简易自动泊车系统 - 路径规划

```c
/*
 * 自动泊车系统 - 路径规划算法
 * 
 * 核心算法：Reeds-Shepp曲线（考虑车辆运动学约束）
 * 简化版：分段圆弧 + 直线组合
 */

typedef struct {
    float x;
    float y;
    float theta;  // 航向角（弧度）
} Pose;

typedef struct {
    Pose start;
    Pose end;
    float R_min;  // 最小转弯半径
} PathPlanner;

// 计算两点之间的距离
float distance(Pose* a, Pose* b) {
    float dx = b->x - a->x;
    float dy = b->y - a->y;
    return sqrt(dx*dx + dy*dy);
}

// 生成平行泊车路径（国一作品核心算法）
typedef struct {
    Pose waypoints[20];
    int count;
    float total_length;
} Path;

Path plan_parallel_parking(PathPlanner* planner, Pose* parking_spot) {
    Path path;
    path.count = 0;
    path.total_length = 0;
    
    // 阶段1：前进到起始位置
    Pose approach;
    approach.x = parking_spot->x - 1.5f;  // 车位前1.5m
    approach.y = parking_spot->y;
    approach.theta = 0;
    path.waypoints[path.count++] = approach;
    
    // 阶段2：倒车入位（圆弧路径）
    float R = planner->R_min * 1.2f;  // 实际转弯半径 = 1.2 × 最小半径
    
    // 计算圆弧中心
    float cx = approach.x;
    float cy = approach.y + R;
    
    // 生成圆弧上的路径点（每10度一个点）
    for (float angle = 180; angle >= 90; angle -= 10) {
        float rad = angle * M_PI / 180.0f;
        Pose wp;
        wp.x = cx + R * cos(rad);
        wp.y = cy + R * sin(rad);
        wp.theta = rad - M_PI / 2;  // 车头朝向
        path.waypoints[path.count++] = wp;
    }
    
    // 阶段3：调整姿态
    Pose final_pose;
    final_pose.x = parking_spot->x;
    final_pose.y = parking_spot->y + 0.1f;  // 车位中心
    final_pose.theta = 0;  // 车头朝前
    path.waypoints[path.count++] = final_pose;
    
    // 计算总路径长度
    for (int i = 1; i < path.count; i++) {
        path.total_length += distance(&path.waypoints[i-1], &path.waypoints[i]);
    }
    
    return path;
}

// 路径跟踪控制器（Pure Pursuit算法）
typedef struct {
    float lookahead_dist;  // 前视距离
    float wheelbase;       // 轴距
} PurePursuit;

float pure_pursuit_steer(PurePursuit* pp, Pose* current, 
                          Path* path, int* target_idx) {
    // 1. 寻找前视点
    for (int i = *target_idx; i < path->count - 1; i++) {
        float dist = distance(current, &path->waypoints[i]);
        if (dist >= pp->lookahead_dist) {
            *target_idx = i;
            break;
        }
    }
    
    Pose target = path->waypoints[*target_idx];
    
    // 2. 计算前视点在车辆坐标系下的位置
    float dx = target.x - current->x;
    float dy = target.y - current->y;
    float target_local_x = dx * cos(current->theta) + dy * sin(current->theta);
    float target_local_y = -dx * sin(current->theta) + dy * cos(current->theta);
    
    // 3. 计算曲率
    float L = pp->lookahead_dist;
    float curvature = 2.0f * target_local_y / (L * L);
    
    // 4. 转向角 = atan(曲率 × 轴距)
    float steer_angle = atan(curvature * pp->wheelbase);
    
    return steer_angle;
}
```

### 2.1 OpenMV颜色循线（2021年F题国一方案）

```python
# OpenMV 红线循线算法（国一作品）
# 核心优化：多ROI分区 + 加权偏移计算

import sensor
import image
import time
import math

# 初始化摄像头
sensor.reset()
sensor.set_pixformat(sensor.RGB565)
sensor.set_framesize(sensor.QVGA)  # 320x240
sensor.skip_frames(time=2000)
sensor.set_auto_gain(False)
sensor.set_auto_whitebal(False)

# 红色阈值（LAB颜色空间）
RED_THRESHOLD = (30, 70, 40, 80, 20, 80)

# 多ROI配置（从近到远）
ROIS = [
    (0, 200, 320, 40, 0.7),    # 近处ROI，权重0.7
    (0, 150, 320, 40, 0.2),    # 中间ROI，权重0.2
    (0, 100, 320, 40, 0.1),    # 远处ROI，权重0.1
]

def line_tracking():
    """多ROI加权循线算法"""
    img = sensor.snapshot()
    
    weighted_sum = 0
    weight_total = 0
    
    for roi in ROIS:
        # 在ROI内查找红色色块
        blobs = img.find_blobs([RED_THRESHOLD], 
                               roi=roi[0:4],
                               pixels_threshold=100,
                               area_threshold=100,
                               merge=True)
        
        if blobs:
            # 找到最大色块
            largest = max(blobs, key=lambda b: b.pixels())
            
            # 计算该ROI的偏移量（相对于图像中心）
            offset = largest.cx() - img.width() // 2
            
            # 加权累加
            weight = roi[4]
            weighted_sum += offset * weight
            weight_total += weight
            
            # 绘制调试信息
            img.draw_rectangle(largest.rect())
            img.draw_cross(largest.cx(), largest.cy())
            img.draw_string(largest.x(), largest.y(), 
                          str(offset), color=(255, 0, 0))
    
    if weight_total > 0:
        # 计算加权平均偏移
        final_offset = weighted_sum / weight_total
        
        # 绘制中心指示线
        img.draw_line(160, 0, 160, 240, color=(0, 255, 0))
        img.draw_line(160 + int(final_offset), 0, 
                      160 + int(final_offset), 240, color=(0, 0, 255))
        
        return final_offset
    
    return None  # 丢失线

def detect_intersection():
    """路口检测算法"""
    img = sensor.snapshot()
    
    # 在底部ROI检测红色色块数量
    blobs = img.find_blobs([RED_THRESHOLD],
                           roi=(0, 200, 320, 40),
                           pixels_threshold=50,
                           area_threshold=50,
                           merge=False)
    
    if len(blobs) >= 2:
        # 多个色块可能是路口
        # 分析色块位置关系
        blob_centers = sorted([b.cx() for b in blobs])
        
        # 检查是否分散在两侧
        if blob_centers[-1] - blob_centers[0] > 100:
            return 'T_INTERSECTION'
        else:
            return 'CROSS'
    
    return None

def identify_room_number():
```

```python
import cv2
import numpy as np

def detect_chessboard(image):
    """
    棋盘检测：定位棋盘边界和格点
    返回：透视变换矩阵、格点坐标
    """
    # 1. 预处理
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    blurred = cv2.GaussianBlur(gray, (5, 5), 0)
    
    # 2. 边缘检测
    edges = cv2.Canny(blurred, 50, 150)
    
    # 3. 查找轮廓
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, 
                                   cv2.CHAIN_APPROX_SIMPLE)
    
    # 4. 找到最大的矩形轮廓（棋盘边界）
    chessboard_contour = None
    max_area = 0
    
    for contour in contours:
        area = cv2.contourArea(contour)
        if area > max_area:
            peri = cv2.arcLength(contour, True)
            approx = cv2.approxPolyDP(contour, 0.02 * peri, True)
            if len(approx) == 4:  # 四边形
                max_area = area
                chessboard_contour = approx
    
    if chessboard_contour is None:
        return None, None
    
    # 5. 排序四个角点（左上、右上、右下、左下）
    corners = chessboard_contour.reshape(4, 2)
    corners = order_corners(corners)
    
    # 6. 计算透视变换矩阵
    board_size = 300  # 输出图像大小
    dst_points = np.array([
        [0, 0],
        [board_size, 0],
        [board_size, board_size],
        [0, board_size]
    ], dtype=np.float32)
    
    M = cv2.getPerspectiveTransform(corners.astype(np.float32), dst_points)
    
    # 7. 计算格点坐标（3×3棋盘的交点）
    grid_points = []
    for row in range(3):
        for col in range(3):
            x = int((col + 0.5) * board_size / 3)
            y = int((row + 0.5) * board_size / 3)
            grid_points.append((x, y))
    
    return M, grid_points

def order_corners(pts):
    """排序角点：左上、右上、右下、左下"""
    rect = np.zeros((4, 2), dtype=np.float32)
    
    # 左上：x+y最小
    s = pts.sum(axis=1)
    rect[0] = pts[np.argmin(s)]
    
    # 右下：x+y最大
    rect[2] = pts[np.argmax(s)]
    
    # 右上：y-x最小
    d = np.diff(pts, axis=1)
    rect[1] = pts[np.argmin(d)]
    
    # 左下：y-x最大
    rect[3] = pts[np.argmax(d)]
    
    return rect

def detect_chess_pieces(image, M, grid_points):
    """
    棋子检测：识别每个格点的状态
    返回：3×3棋盘状态（0=空, 1=黑棋, 2=白棋）
    """
    # 1. 透视变换
    board = cv2.warpPerspective(image, M, (300, 300))
    
    # 2. 转换到HSV空间
    hsv = cv2.cvtColor(board, cv2.COLOR_BGR2HSV)
    
    # 3. 颜色阈值（根据实际棋子颜色调整）
    # 黑色棋子
    black_lower = np.array([0, 0, 0])
    black_upper = np.array([180, 255, 50])
    black_mask = cv2.inRange(hsv, black_lower, black_upper)
    
    # 白色棋子
    white_lower = np.array([0, 0, 200])
    white_upper = np.array([180, 30, 255])
    white_mask = cv2.inRange(hsv, white_lower, white_upper)
    
    # 4. 形态学处理
    kernel = np.ones((3, 3), np.uint8)
    black_mask = cv2.morphologyEx(black_mask, cv2.MORPH_OPEN, kernel)
    white_mask = cv2.morphologyEx(white_mask, cv2.MORPH_OPEN, kernel)
    
    # 5. 检测每个格点的状态
    board_state = [[0]*3 for _ in range(3)]
    
    for idx, (gx, gy) in enumerate(grid_points):
        row = idx // 3
        col = idx % 3
        
        # 在格点周围检测色块
        roi_size = 30
        roi = (gx - roi_size, gy - roi_size, 
               gx + roi_size, gy + roi_size)
        
        # 统计黑色和白色像素
        black_count = cv2.countNonZero(
            black_mask[roi[1]:roi[3], roi[0]:roi[2]])
        white_count = cv2.countNonZero(
            white_mask[roi[1]:roi[3], roi[0]:roi[2]])
        
        # 判断格点状态
        threshold = 100
        if black_count > threshold:
            board_state[row][col] = 1  # 黑棋
        elif white_count > threshold:
            board_state[row][col] = 2  # 白棋
        else:
            board_state[row][col] = 0  # 空
    
    return board_state
```

```python
# OpenMV端（发送）
import struct

def send_command(cmd, data=None):
    """发送命令到STM32"""
    # 帧格式：[0xAA, 0x55, 长度, 命令, 数据..., CRC, 0x0D, 0x0A]
    frame = bytearray()
    frame.append(0xAA)
    frame.append(0x55)
    
    if data is None:
        frame.append(0)      # 数据长度
        frame.append(cmd)    # 命令字
        frame.append(calc_crc(cmd, bytes()))
    else:
        frame.append(len(data))  # 数据长度
        frame.append(cmd)        # 命令字
        frame.extend(data)       # 数据
        frame.append(calc_crc(cmd, data))
    
    frame.append(0x0D)
    frame.append(0x0A)
    
    uart.write(frame)

def send_line_offset(offset):
    """发送循线偏移量"""
    data = struct.pack('<f', offset)
    send_command(0x01, data)

def send_intersection(type_code):
    """发送路口类型"""
    data = struct.pack('<B', type_code)
    send_command(0x02, data)

def send_room_number(number):
    """发送病房号"""
    data = struct.pack('<B', number)
    send_command(0x03, data)

def calc_crc(cmd, data):
    """计算CRC8校验"""
    crc = cmd
    for byte in data:
        crc ^= byte
        for _ in range(8):
            if crc & 0x80:
                crc = (crc << 1) ^ 0x07
            else:
                crc = crc << 1
            crc &= 0xFF
    return crc
```

```c
#include <stdint.h>
#include <stdbool.h>

#define CMD_LINE_OFFSET    0x01
#define CMD_INTERSECTION   0x02
#define CMD_ROOM_NUMBER    0x03

typedef struct {
    float line_offset;
    uint8_t intersection_type;
    uint8_t room_number;
    uint32_t last_update_time;
} VisionData;

VisionData vision_data;

void process_uart_frame(uint8_t* frame, uint8_t length) {
    // 验证帧头帧尾
    if (frame[0] != 0xAA || frame[1] != 0x55) return;
    if (frame[length-2] != 0x0D || frame[length-1] != 0x0A) return;
    
    uint8_t data_len = frame[2];
    uint8_t cmd = frame[3];
    uint8_t* data = &frame[4];
    
    // 验证CRC
    uint8_t received_crc = frame[4 + data_len];
    uint8_t calc_crc = calc_crc8(cmd, data, data_len);
    if (received_crc != calc_crc) return;
    
    // 处理命令
    switch (cmd) {
        case CMD_LINE_OFFSET:
            if (data_len == 4) {
                float offset;
                memcpy(&offset, data, 4);
                vision_data.line_offset = offset;
                vision_data.last_update_time = HAL_GetTick();
            }
            break;
            
        case CMD_INTERSECTION:
            if (data_len == 1) {
                vision_data.intersection_type = data[0];
            }
            break;
            
        case CMD_ROOM_NUMBER:
            if (data_len == 1) {
                vision_data.room_number = data[0];
            }
            break;
    }
}
```
