### 4.2 控制算法核心代码
```c
// 位置式PID（抗积分饱和）
typedef struct {
    float Kp, Ki, Kd;
    float integral;
    float prev_error;
    float out_min, out_max;
    float dt;
} PID_t;

float PID_Compute(PID_t *pid, float setpoint, float measurement) {
    float error = setpoint - measurement;
    pid->integral += error * pid->dt;

    // 抗积分饱和
    float derivative = (error - pid->prev_error) / pid->dt;
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;

    if (output > pid->out_max) {
        output = pid->out_max;
        pid->integral -= error * pid->dt;  // 回退
    } else if (output < pid->out_min) {
        output = pid->out_min;
        pid->integral -= error * pid->dt;
    }

    pid->prev_error = error;
    return output;
}
```

### 4.3 数字滤波
```c
// 滑动平均滤波
#define FILTER_N 8
float filter_buffer[FILTER_N];
uint8_t filter_idx = 0;

float MovingAverage(float new_value) {
    filter_buffer[filter_idx] = new_value;
    filter_idx = (filter_idx + 1) % FILTER_N;
    float sum = 0;
    for (int i = 0; i < FILTER_N; i++) sum += filter_buffer[i];
    return sum / FILTER_N;
}
```
