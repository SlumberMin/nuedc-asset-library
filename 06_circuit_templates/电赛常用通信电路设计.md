## 一、CAN 总线电路设计

### 1.1 CAN 总线概述
```
特点：差分信号、多主结构、强抗干扰、1Mbps@40m / 125kbps@500m
电赛应用：多板通信、电机控制总线、传感器网络
```


### 1.3 分裂终端设计（推荐）

```
改善EMC性能的终端方案：

CAN_H ──┬── 60Ω ──┬── 0.1μF ──┬── 60Ω ──┬── CAN_L
        │         │           │         │
        └─────────┴─────┬─────┴─────────┘
                        │
                       GND

优势：共模滤波效果好，降低辐射
```

### 1.4 PCB布局要点
```
1. CAN收发器靠近STM32放置，走线 < 5cm
2. 终端电阻靠近连接器放置
3. CAN_H/CAN_L 差分走线，间距固定（2倍线宽）
4. 差分对远离高速信号线（SPI/时钟）
5. 地平面完整，不被信号线割裂
```

### 1.5 STM32 CAN 驱动代码

```c
// CAN初始化（500kbps，APB1=42MHz）
void CAN1_Init(void) {
    // 时钟使能
    RCC->APB1ENR |= RCC_APB1ENR_CAN1EN;
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOBEN;
    
    // GPIO配置：PB8=CAN_RX, PB9=CAN_TX
    GPIOB->MODER &= ~(0xF << 16);
    GPIOB->MODER |= (0xA << 16);     // 复用功能
    GPIOB->AFR[1] |= (0x9 << 0);     // PB8=AF9
    GPIOB->AFR[1] |= (0x9 << 4);     // PB9=AF9
    
    // CAN配置：500kbps
    CAN1->MCR = CAN_MCR_INRQ;        // 进入初始化模式
    while (!(CAN1->MSR & CAN_MSR_INAK));
    
    CAN1->MCR |= CAN_MCR_ABOM;       // 自动离线管理
    CAN1->MCR &= ~CAN_MCR_TXFP;      // 发送FIFO优先级
    
    // 波特率：42MHz / (1+6+3) / 6 = 500kbps (建议SJW=1)
    CAN1->BTR = (5 << 24) | (2 << 20) | (5 << 16) | (5 << 0);
    // BRP=5, TS1=6TQ, TS2=3TQ, SJW=2TQ
    
    CAN1->MCR &= ~CAN_MCR_INRQ;      // 退出初始化
    while (CAN1->MSR & CAN_MSR_INAK);
}

// CAN发送
uint8_t CAN1_Send(uint32_t id, uint8_t* data, uint8_t len) {
    if (!(CAN1->TSR & CAN_TSR_TME0)) return 1;  // 邮箱满
    
    CAN1->sTxMailBox[0].TIR = (id << 21);        // 标准ID
    CAN1->sTxMailBox[0].TDTR = len & 0x0F;       // 数据长度
    CAN1->sTxMailBox[0].TDLR = *(uint32_t*)data;  // 低4字节
    CAN1->sTxMailBox[0].TDHR = *(uint32_t*)(data+4); // 高4字节
    CAN1->sTxMailBox[0].TIR |= CAN_TI0R_TXRQ;    // 发送请求
    
    return 0;
}

// CAN接收（FIFO0中断）
void CAN1_RX0_IRQHandler(void) {
    CAN_RxHeaderTypeDef header;
    uint8_t data[8];
    
    header.StdId = (CAN1->sFIFO0MailBox[0].RIR >> 21) & 0x7FF;
    header.DLC = CAN1->sFIFO0MailBox[0].RDTR & 0x0F;
    *(uint32_t*)data = CAN1->sFIFO0MailBox[0].RDLR;
    *(uint32_t*)(data+4) = CAN1->sFIFO0MailBox[0].RDHR;
    
    CAN1->RF0R |= CAN_RF0R_RFOM0;  // 释放邮箱
    
    // 处理接收数据
    process_can_data(header.StdId, data, header.DLC);
}
```

## 二、RS485 电路设计

### 2.1 RS485 总线概述
```
特点：半双工、差分信号、多点通信（最多32/128节点）
电赛应用：远距离传感器通信（>10m）、多节点数据采集
```

### 2.2 完整 RS485 收发电路

```
┌─────────────────────────────────────────────────────┐
│ RS485收发器电路（MAX485/SN75176）                      │
│                                                       │
│  STM32F407         MAX485              RS485总线       │
│  ┌─────────┐      ┌──────────┐                        │
│  │         │      │          │                        │
│  │  PA9 TX ├─┬────┤ DI   A  ├───┬── RS485_A (+)      │
│  │  PA10RX ├─┼─┬──┤ RO   B  ├───┼── RS485_B (-)      │
│  │         │ │ │  │          │   │                    │
│  │  PA8 RE ├─┼─┼──┤ ~RE      │   │  120Ω终端电阻     │
│  │  PA8 DE ├─┘ │  │  DE      │   ├── 120Ω            │
│  │         │   │  │          │   │                    │
│  │    3.3V ├──────┤ VCC  GND │   │                    │
│  │    GND  ├──────┤          │   └── GND（可选）      │
│  └─────────┘      └──────────┘                        │
│                                                       │
│  注意：PA8 同时控制 RE和DE（收发切换）                  │
│  PA8=HIGH → 发送模式                                   │
│  PA8=LOW  → 接收模式                                   │
└─────────────────────────────────────────────────────┘
```

### 2.3 自动方向控制电路（推荐）

```
无需GPIO控制方向，由硬件自动切换：

STM32 TX ──┬── 100Ω ── DI (MAX485)
            │
            └── 470Ω ── DE (MAX485)
                        ~RE 接GND（常使能接收）

原理：TX高电平时，DI=高，DE=高（发送）
      TX低电平时，DI=低，DE=低（接收）

简化版（MC3486等带自动方向）：
STM32 TX ──────── DI
STM32 RX ──────── RO
                   DE = GND（常接收）
                   ~RE = VCC（常接收）
```

### 2.4 PCB布局要点
```
1. MAX485 靠近连接器，走线 < 3cm
2. A/B 差分走线，保持等长
3. 终端电阻在总线两端
4. GND连接到连接器（共地参考）
5. 485总线建议用双绞线（线距1-2mm）
```

### 2.5 RS485 驱动代码

```c
// RS485半双工收发控制
#define RS485_TX_EN()  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_SET)
#define RS485_RX_EN()  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_8, GPIO_PIN_RESET)

// 发送数据
void RS485_Send(uint8_t* data, uint16_t len) {
    RS485_TX_EN();  // 切换到发送模式
    HAL_Delay(1);   // 等待方向稳定
    
    HAL_UART_Transmit(&huart1, data, len, 100);
    
    // 等待最后一字节发送完成
    while (!(USART1->SR & USART_SR_TC));
    
    RS485_RX_EN();  // 切换回接收模式
}

// 接收（DMA方式）
uint8_t rx_buf[256];
void RS485_StartReceive(void) {
    RS485_RX_EN();
    HAL_UART_Receive_DMA(&huart1, rx_buf, 256);
}

// Modbus RTU 帧格式示例
typedef struct __attribute__((packed)) {
    uint8_t  addr;       // 从机地址
    uint8_t  func;       // 功能码
    uint16_t reg_addr;   // 寄存器地址
    uint16_t reg_count;  // 寄存器数量
    uint16_t crc;        // CRC16
} Modbus_ReadReq_t;

// CRC16-Modbus 计算
uint16_t modbus_crc16(uint8_t* data, uint16_t len) {
    uint16_t crc = 0xFFFF;
    for (int i = 0; i < len; i++) {
        crc ^= data[i];
        for (int j = 0; j < 8; j++) {
            if (crc & 1) crc = (crc >> 1) ^ 0xA001;
            else crc >>= 1;
        }
    }
    return crc;
}
```

### 3.4 E32模块驱动（AT指令透传模式）

```c
// E32-433T20DT 配置（M0/M1引脚控制模式）
// M0=0, M1=0: 透传模式（正常通信）
// M0=1, M1=0: WOR模式（唤醒）
// M0=0, M1=1: 配置模式
// M0=1, M1=1: 睡眠模式

void e32_init(void) {
    // M0=PB3, M1=PB4
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_3, GPIO_PIN_RESET);  // M0=0
    HAL_GPIO_WritePin(GPIOB, GPIO_PIN_4, GPIO_PIN_RESET);  // M1=0
    HAL_Delay(100);  // 等待模块启动
}

// 发送数据（定点模式：地址+信道+数据）
void e32_send(uint16_t addr, uint8_t channel, uint8_t* data, uint8_t len) {
    uint8_t header[3];
    header[0] = (addr >> 8) & 0xFF;   // 地址高字节
    header[1] = addr & 0xFF;          // 地址低字节
    header[2] = channel;              // 信道（0-31对应410-441MHz）
    
    HAL_UART_Transmit(&huart2, header, 3, 100);
    HAL_UART_Transmit(&huart2, data, len, 100);
}

// 接收数据（DMA接收）
uint8_t lora_rx_buf[256];
volatile uint16_t lora_rx_len = 0;

void lora_start_receive(void) {
    HAL_UARTEx_ReceiveToIdle_DMA(&huart2, lora_rx_buf, 256);
}

void HAL_UARTEx_RxEventCallback(UART_HandleTypeDef *huart, uint16_t Size) {
    if (huart == &huart2) {
        lora_rx_len = Size;
        process_lora_data(lora_rx_buf, lora_rx_len);
        HAL_UARTEx_ReceiveToIdle_DMA(&huart2, lora_rx_buf, 256);
    }
}
```

### 4.4 BLE 驱动代码

```c
// BLE透传模式（与串口相同，无需特殊协议）
void ble_init(void) {
    // UART2: 9600 波特率（BLE默认）
    huart2.Init.BaudRate = 9600;
    HAL_UART_Init(&huart2);
}

// 发送传感器数据到手机
void ble_send_sensor_data(float temp, float humi, float voltage) {
    char buf[64];
    int len = snprintf(buf, sizeof(buf), 
                       "{\"t\":%.1f,\"h\":%.1f,\"v\":%.2f}\r\n",
                       temp, humi, voltage);
    HAL_UART_Transmit(&huart2, (uint8_t*)buf, len, 100);
}

// 接收手机控制指令
uint8_t ble_rx_buf[64];
volatile uint8_t ble_rx_flag = 0;

void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    if (huart == &huart2) {
        ble_rx_flag = 1;
    }
}

void process_ble_command(void) {
    if (!ble_rx_flag) return;
    ble_rx_flag = 0;
    
    // 解析JSON简单指令：{"cmd":"motor","val":50}
    if (strstr((char*)ble_rx_buf, "motor")) {
        int val = extract_int((char*)ble_rx_buf, "val");
        motor_set_speed(val);
    }
    else if (strstr((char*)ble_rx_buf, "led")) {
        int on = extract_int((char*)ble_rx_buf, "val");
        HAL_GPIO_WritePin(GPIOC, GPIO_PIN_13, on ? GPIO_PIN_RESET : GPIO_PIN_SET);
    }
}
```

## 六、通用UART扩展电路

### 6.1 串口扩展（多路UART）

```
当STM32 UART不够用时：

方案1：软件串口（适合低波特率 < 9600）
方案2：UART扩展芯片（GM8125：1扩5）
方案3：SPI转UART（SC16IS752：1扩2）

GM8125电路：
STM32 UART ── GM8125 ── UART1~UART5
                       (5路独立串口)
```

### 6.2 电平转换电路

```
3.3V ↔ 5V 电平转换（MOSFET法）：

5V ── 10kΩ ──┬── 5V设备
              │
3.3V信号 ──── MOSFET(2N7002) ── 3.3V设备
              │
             GND

或用专用芯片：
TXB0108（8路双向电平转换）
- VCCA=3.3V, VCCB=5V
- A1-A8: 3.3V侧
- B1-B8: 5V侧
```
