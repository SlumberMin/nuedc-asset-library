## 一、代码优化

### 1.1 编译器优化选项

```makefile
# Keil ARMCC
-O1  # 基本优化（调试友好）
-O2  # 推荐：平衡优化
-O3  # 激进优化（可能增大代码体积）
-Os  # 体积优先（电赛推荐）

# GCC ARM
-mcpu=cortex-m4 -mthumb -O2 -flto   # -flto 链接时优化
-ffunction-sections -fdata-sections   # 配合 --gc-sections 消除未用代码
```

### 1.2 关键代码优化技巧

```c
// ✅ 位运算替代乘除
uint32_t val = x * 8;      // 慢
uint32_t val = x << 3;     // 快

uint32_t val = x / 16;     // 慢
uint32_t val = x >> 4;     // 快

uint32_t val = x % 64;     // 慢
uint32_t val = x & 63;     // 快（仅限2的幂）

// ✅ 查表法替代复杂计算
// 正弦表（256点，Q15格式）
const int16_t sin_table[256] = { /* ... */ };
int16_t fast_sin(uint8_t angle) {
    return sin_table[angle];
}

// ✅ 定点数替代浮点数
// Q16.16 格式：高16位整数，低16位小数
typedef int32_t q16_t;
#define Q16_ONE  65536
#define TO_Q16(x) ((q16_t)((x) * Q16_ONE))
#define FROM_Q16(x) ((float)(x) / Q16_ONE)

q16_t q16_mul(q16_t a, q16_t b) {
    return (q16_t)(((int64_t)a * b) >> 16);
}

q16_t q16_div(q16_t a, q16_t b) {
    return (q16_t)(((int64_t)a << 16) / b);
}

// ✅ DMA传输替代CPU搬移
HAL_ADC_Start_DMA(&hadc1, adc_buf, 1024);  // ADC+DMA，零CPU开销
```

### 1.3 循环优化

```c
// ❌ 低效写法
for (int i = 0; i < len; i++) {
    buf[i] = read_reg(i);
}

// ✅ 循环展开（4路展开）
int i;
for (i = 0; i < (len & ~3); i += 4) {
    buf[i]   = read_reg(i);
    buf[i+1] = read_reg(i+1);
    buf[i+2] = read_reg(i+2);
    buf[i+3] = read_reg(i+3);
}
for (; i < len; i++) {
    buf[i] = read_reg(i);
}

// ✅ 减少循环内计算
// ❌
for (i = 0; i < n; i++) {
    arr[i] = base + i * step + offset;
}
// ✅
int val = base + offset;
for (i = 0; i < n; i++) {
    arr[i] = val;
    val += step;
}
```

### 1.4 内联与宏

```c
// 强制内联小函数
static inline uint16_t swap16(uint16_t x) {
    return (x >> 8) | (x << 8);
}

// 编译期计算宏
#define ARRAY_SIZE(a) (sizeof(a) / sizeof((a)[0]))
#define ALIGN_UP(x, a) (((x) + (a)-1) & ~((a)-1))
```

## 二、内存优化

### 2.1 内存布局认知

```
┌─────────────────┐ 高地址
│    Stack (栈)    │ ← 局部变量，向下增长
├─────────────────┤
│    Heap (堆)     │ ← malloc，向上增长
├─────────────────┤
│    BSS (零初始化) │ ← 未初始化全局变量
├─────────────────┤
│    Data (已初始化)│ ← 已初始化全局变量
├─────────────────┤
│    Text (代码)   │ ← 程序代码+常量
└─────────────────┘ 低地址

STM32F407: 128KB SRAM + 1MB Flash
```

### 2.2 内存优化技巧

```c
// ✅ 使用合适的数据类型
uint8_t  flag;      // 而非 int flag;  (省3字节)
uint16_t adc_val;   // 而非 uint32_t;  (省2字节)

// ✅ 使用 packed 结构体（嵌入式通信协议）
typedef struct __attribute__((packed)) {
    uint8_t  header;
    uint16_t data;
    uint8_t  crc;
} Packet_t;  // 4字节，而非对齐的6字节

// ✅ union 共享内存
typedef union {
    float    f;
    uint32_t u;
    uint8_t  bytes[4];
} FloatU_t;

// ✅ static 局部变量替代全局变量（减少全局污染）
void process(void) {
    static uint32_t count = 0;  // 仅函数内可见，但保持值
    count++;
}

// ✅ 避免栈上分配大数组
// ❌ void func(void) { uint8_t buf[2048]; }  // 栈溢出风险
// ✅ static uint8_t buf[2048]; 或用全局
```

### 2.3 堆管理（谨慎使用）

```c
// 电赛建议：尽量避免动态内存分配
// 如必须使用，用内存池替代 malloc

#define POOL_BLOCK_SIZE  64
#define POOL_BLOCK_COUNT 16
static uint8_t pool[POOL_BLOCK_COUNT][POOL_BLOCK_SIZE];
static uint8_t pool_used[POOL_BLOCK_COUNT] = {0};

void* pool_alloc(void) {
    for (int i = 0; i < POOL_BLOCK_COUNT; i++) {
        if (!pool_used[i]) {
            pool_used[i] = 1;
            return pool[i];
        }
    }
    return NULL;
}

void pool_free(void* ptr) {
    for (int i = 0; i < POOL_BLOCK_COUNT; i++) {
        if (pool[i] == ptr) {
            pool_used[i] = 0;
            return;
        }
    }
}
```

## 三、功耗优化

### 3.1 STM32 低功耗模式

```
┌──────────────┬──────────┬──────────┬─────────────┐
│    模式      │ 电流消耗  │ 唤醒时间  │   唤醒源     │
├──────────────┼──────────┼──────────┼─────────────┤
│ Sleep        │ ~mA级    │ 即时      │ 任意中断     │
│ Stop         │ ~μA级    │ ~μs级    │ EXTI/RTC    │
│ Standby      │ ~μA级    │ 复位级    │ WKUP/RTC    │
└──────────────┴──────────┴──────────┴─────────────┘
```

```c
// Sleep 模式（等待中断）
HAL_SuspendTick();
HAL_PWR_EnterSLEEPMode(PWR_MAINREGULATOR_ON, PWR_SLEEPENTRY_WFI);
HAL_ResumeTick();

// Stop 模式（低功耗等待）
HAL_SuspendTick();
HAL_PWR_EnterSTOPMode(PWR_LOWPOWERREGULATOR_ON, PWR_STOPENTRY_WFI);
SystemClock_Config();  // 唤醒后需重新配置时钟
HAL_ResumeTick();

// Standby 模式（最低功耗）
HAL_PWR_EnterSTANDBYMode();  // 唤醒等同复位
```

### 3.2 外设功耗管理

```c
// ✅ 关闭不用的外设时钟
__HAL_RCC_GPIOB_CLK_DISABLE();
__HAL_RCC_USART2_CLK_DISABLE();
__HAL_RCC_TIM3_CLK_DISABLE();

// ✅ GPIO 低功耗配置
// 未用引脚配置为模拟输入（最低功耗）
GPIO_InitStruct.Pin = GPIO_PIN_All;
GPIO_InitStruct.Mode = GPIO_MODE_ANALOG;
HAL_GPIO_Init(GPIOx, &GPIO_InitStruct);

// ✅ 降低系统时钟（无任务时）
void enter_low_power_clock(void) {
    RCC_ClkInitTypeDef clk = {0};
    clk.SYSCLKSource = RCC_SYSCLKSOURCE_HSI;  // 切到16MHz
    clk.AHBCLKDivider = RCC_SYSCLK_DIV8;       // AHB=2MHz
    HAL_RCC_ClockConfig(&clk, FLASH_LATENCY_0);
}

// ✅ 外设间歇工作
// ADC 采样 → 处理 → 关ADC → 延时 → 重复
HAL_ADC_Start(&hadc1);
HAL_ADC_PollForConversion(&hadc1, 10);
uint16_t val = HAL_ADC_GetValue(&hadc1);
HAL_ADC_Stop(&hadc1);
HAL_Delay(1000);  // 1秒采一次
```

### 3.3 功耗估算公式

```
电池续航(h) = 电池容量(mAh) / 平均电流(mA)

例：3000mAh电池，平均电流15mA
续航 = 3000/15 = 200小时 ≈ 8.3天
```

## 四、实时性优化

### 4.1 中断优先级设计（STM32）

```c
// 抢占优先级：高可打断低
// 子优先级：同级不打断，仅排队

// 电赛推荐优先级分配
NVIC_SetPriority(TIM1_UP_IRQn, 0);    // 最高：控制环（电流环）
NVIC_SetPriority(TIM2_IRQn, 1);        // 高：速度/位置环
NVIC_SetPriority(ADC_IRQn, 2);         // 中：ADC采样
NVIC_SetPriority(USART1_IRQn, 3);      // 低：通信
NVIC_SetPriority(SysTick_IRQn, 4);     // 最低：系统滴答
```

### 4.2 中断处理快进快出

```c
// ❌ 中断中做大量计算
void TIM2_IRQHandler(void) {
    // 200行PID计算...
}

// ✅ 中断中设标志，主循环处理
volatile uint8_t pid_flag = 0;
void TIM2_IRQHandler(void) {
    TIM2->SR &= ~TIM_SR_UIF;
    pid_flag = 1;  // 仅设标志
}

int main(void) {
    while (1) {
        if (pid_flag) {
            pid_flag = 0;
            pid_calculate();  // 主循环中处理
        }
    }
}

// ✅ 或用环形缓冲区传递数据
typedef struct {
    volatile uint16_t buf[256];
    volatile uint16_t head, tail;
} RingBuf_t;

void ringbuf_push(RingBuf_t* rb, uint16_t val) {
    rb->buf[rb->head] = val;
    rb->head = (rb->head + 1) & 255;
}

uint16_t ringbuf_pop(RingBuf_t* rb) {
    uint16_t val = rb->buf[rb->tail];
    rb->tail = (rb->tail + 1) & 255;
    return val;
}
```

### 4.3 定时器精确时序

```c
// 使用定时器替代 HAL_Delay（阻塞式）
// 定时器中断实现非阻塞延时

// 1ms 定时器（84MHz 主频）
// TIM2: PSC=83, ARR=999 → 84MHz/84/1000 = 1kHz

// 多任务调度器（裸机轮询）
typedef struct {
    uint32_t period_ms;
    uint32_t last_run;
    void (*task)(void);
} Task_t;

Task_t tasks[] = {
    {  1, 0, pid_task },       // 1ms: PID控制
    { 10, 0, sensor_task },    // 10ms: 传感器读取
    { 50, 0, display_task },   // 50ms: 显示刷新
    {100, 0, comm_task },      // 100ms: 通信处理
};

void scheduler_run(void) {
    uint32_t now = HAL_GetTick();
    for (int i = 0; i < ARRAY_SIZE(tasks); i++) {
        if (now - tasks[i].last_run >= tasks[i].period_ms) {
            tasks[i].last_run = now;
            tasks[i].task();
        }
    }
}
```

### 4.4 关键路径优化

```c
// ✅ 中断中避免浮点运算（FPU开启时除外）
// ✅ 中断中避免除法（周期长）
// ✅ 使用 volatile 告知编译器不要优化掉
volatile uint32_t *reg = (uint32_t*)0x40021000;
```

## 五、优化检查清单

| 项目 | 检查点 |
|------|--------|
| 编译优化 | -O2 / -Os 是否开启 |
| 未用代码 | -ffunction-sections + --gc-sections |
| 数据类型 | 是否用了最小够用的类型 |
| 浮点运算 | 能否用定点数替代 |
| 查表法 | 复杂函数是否可查表 |

| volatile | 共享变量是否正确标记 |
