## 一、断点调试（Keil / STM32）

### 1.1 基础断点类型

```
┌──────────────┬───────────────────────────────┐
│ 断点类型      │ 用途                           │
├──────────────┼───────────────────────────────┤
│ 软件断点      │ 代码行断点（最常用）            │
│ 硬件断点      │ Flash中设置，数量有限（F4有6个）│
│ 数据断点      │ 变量被读/写时停下              │
│ 条件断点      │ 满足条件时才停下               │
└──────────────┴───────────────────────────────┘
```

### 1.2 Keil 断点操作

```
F9          → 设置/取消断点
F5          → 运行到断点
F10         → 单步（Step Over）
F11         → 单步进入（Step Into）
Ctrl+F11    → 跳出函数（Step Out）
```

### 1.3 数据断点（Watchpoint）

```
Keil: Debug → Breakpoints → Watchpoint
设置：变量地址 + 读/写条件

场景：某全局变量被意外修改
1. 在 Watch 窗口记下变量地址（如 0x20000100）
2. 设置数据断点：Write, 0x20000100, Size=4
3. 运行，程序停在修改该变量的代码行
```

### 1.4 条件断点

```
Keil: 右键断点 → Breakpoint Properties → Define condition

例：count == 1000 时停下
表达式: count == 1000
```

### 1.5 SWV（串行线查看器）

```
Keil: Debug → System Viewer → 配置 SWV
- ITM Stimulus Port 0: printf 输出
- 变量实时波形显示
- 不占用额外引脚（SWD接口）
```

## 二、日志系统

### 2.1 轻量级日志框架

```c
// log.h
#ifndef LOG_H
#define LOG_H

#include <stdio.h>

// 日志级别
#define LOG_LEVEL_NONE  0
#define LOG_LEVEL_ERR   1
#define LOG_LEVEL_WARN  2
#define LOG_LEVEL_INFO  3
#define LOG_LEVEL_DEBUG 4

// 修改此值控制日志输出级别
#define LOG_LEVEL LOG_LEVEL_DEBUG

// 颜色定义（串口终端用）
#define CLR_RED    "\033[31m"
#define CLR_YELLOW "\033[33m"
#define CLR_GREEN  "\033[32m"
#define CLR_CYAN   "\033[36m"
#define CLR_RESET  "\033[0m"

#define LOG_ERR(fmt, ...)  do { if(LOG_LEVEL >= LOG_LEVEL_ERR)  \
    printf(CLR_RED    "[ERR ] " fmt CLR_RESET "\r\n", ##__VA_ARGS__); } while(0)
#define LOG_WARN(fmt, ...) do { if(LOG_LEVEL >= LOG_LEVEL_WARN) \
    printf(CLR_YELLOW "[WARN] " fmt CLR_RESET "\r\n", ##__VA_ARGS__); } while(0)
#define LOG_INFO(fmt, ...) do { if(LOG_LEVEL >= LOG_LEVEL_INFO) \
    printf(CLR_GREEN  "[INFO] " fmt CLR_RESET "\r\n", ##__VA_ARGS__); } while(0)
#define LOG_DBG(fmt, ...)  do { if(LOG_LEVEL >= LOG_LEVEL_DEBUG) \
    printf(CLR_CYAN   "[DBG ] " fmt CLR_RESET "\r\n", ##__VA_ARGS__); } while(0)

// 带时间戳版本（需要SysTick）
#define LOG_INFO_TS(fmt, ...) do { if(LOG_LEVEL >= LOG_LEVEL_INFO) \
    printf(CLR_GREEN "[%lu] [INFO] " fmt CLR_RESET "\r\n", HAL_GetTick(), ##__VA_ARGS__); } while(0)

#endif
```

### 2.3 环形缓冲日志（中断安全）

```c
#define LOG_BUF_SIZE 1024
static char log_buf[LOG_BUF_SIZE];
static volatile uint16_t log_head = 0, log_tail = 0;

void log_put(char c) {
    uint16_t next = (log_head + 1) % LOG_BUF_SIZE;
    if (next != log_tail) {
        log_buf[log_head] = c;
        log_head = next;
    }
}

// 在主循环中输出
void log_flush(void) {
    while (log_tail != log_head) {
        while (!(USART1->SR & USART_SR_TXE));
        USART1->DR = log_buf[log_tail];
        log_tail = (log_tail + 1) % LOG_BUF_SIZE;
    }
}
```

```c
// 用GPIO翻转测量代码执行时间
// 将PA0接示波器或逻辑分析仪

#define DBG_PIN_HIGH()  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_SET)
#define DBG_PIN_LOW()   HAL_GPIO_WritePin(GPIOA, GPIO_PIN_0, GPIO_PIN_RESET)

void critical_function(void) {
    DBG_PIN_HIGH();     // 示波器上升沿开始
    // ... 关键代码 ...
    pid_calculate();
    // ... 关键代码 ...
    DBG_PIN_LOW();      // 示波器下降沿结束
    // 脉冲宽度 = 函数执行时间
}

// 测量中断响应延迟
void TIM2_IRQHandler(void) {
    DBG_PIN_HIGH();
    TIM2->SR &= ~TIM_SR_UIF;
    // ... 处理 ...
    DBG_PIN_LOW();
}

```

## 五、性能分析

### 5.1 代码执行时间测量

```c
// 方式1：DWT周期计数器（Cortex-M3/M4/M7）
void dwt_init(void) {
    CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
    DWT->CYCCNT = 0;
    DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
}

uint32_t dwt_get(void) {
    return DWT->CYCCNT;
}

// 使用示例
dwt_init();
uint32_t t0 = dwt_get();
pid_calculate();
uint32_t t1 = dwt_get();
uint32_t cycles = t1 - t0;
float us = (float)cycles / (SystemCoreClock / 1000000);
printf("PID耗时: %.1f us (%lu cycles)\r\n", us, cycles);
```

### 5.2 栈使用分析

```c
// 栈溢出检测（Keil配置）
// Options → Linker → Use Memory Layout → 勾选 Enable Stack Analysis

// 运行时栈检测（填充魔法数字）
void stack_paint(void) {
    extern uint32_t __initial_sp[];
    extern uint32_t Image$$ARM_LIB_STACK$$ZI$$Limit[];
    uint32_t *p = Image$$ARM_LIB_STACK$$ZI$$Limit;
    while (p < __initial_sp) {
        *p++ = 0xDEADBEEF;  // 魔法数字
    }
}

uint32_t stack_usage(void) {
    extern uint32_t Image$$ARM_LIB_STACK$$ZI$$Limit[];
    uint32_t *p = Image$$ARM_LIB_STACK$$ZI$$Limit;
    while (*p == 0xDEADBEEF) p++;
    return (uint32_t)(__initial_sp - p) * 4;  // 字节数
}
```

### 5.3 CPU 使用率统计

```c
// 简易CPU占用率测量
volatile uint32_t idle_count = 0;
volatile uint32_t cpu_usage = 0;

void cpu_usage_task(void) {
    static uint32_t last_idle = 0;
    static uint32_t last_tick = 0;
    
    uint32_t now = HAL_GetTick();
    if (now - last_tick >= 1000) {  // 每秒统计
        cpu_usage = 100 - (idle_count - last_idle) * 100 / 
                    (SystemCoreClock / 1000 / 16);  // 粗略估算
        last_idle = idle_count;
        last_tick = now;
    }
}

// 主循环空闲时递增
int main(void) {
    while (1) {
        scheduler_run();
        idle_count++;  // 空闲计数
        cpu_usage_task();
    }
}
```

### HardFault 定位

```c
// 在启动文件中添加 HardFault Handler
void HardFault_Handler(void) {
    __asm volatile (
        "TST LR, #4      \n"
        "ITE EQ          \n"
        "MRSEQ R0, MSP   \n"
        "MRSNE R0, PSP   \n"
        "B hard_fault_handler_c \n"
    );
}

void hard_fault_handler_c(uint32_t* stack) {
    uint32_t r0  = stack[0];
    uint32_t r1  = stack[1];
    uint32_t r2  = stack[2];
    uint32_t r3  = stack[3];
    uint32_t r12 = stack[4];
    uint32_t lr  = stack[5];   // 链接寄存器（返回地址）
    uint32_t pc  = stack[6];   // 程序计数器（出错位置）
    uint32_t psr = stack[7];
    
    printf("HardFault at PC=0x%08lX, LR=0x%08lX\r\n", pc, lr);
    // 在Keil中：View → Disassembly，跳转到PC地址查看出错指令
    while(1);
}
```
