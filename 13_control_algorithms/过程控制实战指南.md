## 1. PID控制深入

### 1.1 PID三种形式

```c
// 位置式PID
typedef struct {
    float Kp, Ki, Kd;
    float integral;
    float prev_error;
    float output_min, output_max;
    float integral_max; // 积分限幅
} PID_Position;

float pid_position_update(PID_Position *pid, float setpoint, float measured) {
    float error = setpoint - measured;
    
    pid->integral += error;
    // 积分限幅
    if (pid->integral > pid->integral_max) pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max) pid->integral = -pid->integral_max;
    
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;
    
    // 输出限幅
    if (output > pid->output_max) output = pid->output_max;
    if (output < pid->output_min) output = pid->output_min;
    
    return output;
}

// 增量式PID
typedef struct {
    float Kp, Ki, Kd;
    float prev_error;
    float prev2_error;
    float delta_output;
    float output_min, output_max;
} PID_Incremental;

float pid_incremental_update(PID_Incremental *pid, float setpoint, float measured) {
    float error = setpoint - measured;
    
    float delta = pid->Kp * (error - pid->prev_error)
                + pid->Ki * error
                + pid->Kd * (error - 2 * pid->prev_error + pid->prev2_error);
    
    pid->prev2_error = pid->prev_error;
    pid->prev_error = error;
    
    pid->delta_output += delta;
    
    // 输出限幅
    if (pid->delta_output > pid->output_max) pid->delta_output = pid->output_max;
    if (pid->delta_output < pid->output_min) pid->delta_output = pid->output_min;
    
    return pid->delta_output;
}
```

### 1.2 积分抗饱和（Anti-Windup）

```c
// 方法1：积分限幅
float pid_with_integral_clamp(PID_Position *pid, float sp, float pv) {
    float error = sp - pv;
    
    pid->integral += error;
    if (pid->integral > pid->integral_max) pid->integral = pid->integral_max;
    if (pid->integral < -pid->integral_max) pid->integral = -pid->integral_max;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * (error - pid->prev_error);
    pid->prev_error = error;
    
    return CLAMP(output, pid->output_min, pid->output_max);
}

// 方法2：条件积分（当输出饱和时不积分）
float pid_conditional_integration(PID_Position *pid, float sp, float pv) {
    float error = sp - pv;
    float output_unclamped = pid->Kp * error + pid->Ki * pid->integral;
    
    // 只在输出未饱和或误差方向与输出相反时积分
    int output_saturated = (output_unclamped > pid->output_max) || 
                           (output_unclamped < pid->output_min);
    int error_would_increase = (error > 0 && output_unclamped > pid->output_max) ||
                               (error < 0 && output_unclamped < pid->output_min);
    
    if (!output_saturated || !error_would_increase) {
        pid->integral += error;
    }
    
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * derivative;
    return CLAMP(output, pid->output_min, pid->output_max);
}

// 方法3：反计算法（Back-Calculation）
typedef struct {
    float Kp, Ki, Kd;
    float Kb;  // 反计算增益
    float integral;
    float prev_error;
    float output_min, output_max;
} PID_AntiWindup;

float pid_back_calculation(PID_AntiWindup *pid, float sp, float pv) {
    float error = sp - pv;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * (error - pid->prev_error);
    pid->prev_error = error;
    
    float output_clamped = CLAMP(output, pid->output_min, pid->output_max);
    
    // 反计算：将饱和部分反馈回积分项
    pid->integral += error + pid->Kb * (output_clamped - output);
    
    return output_clamped;
}
```

### 1.3 微分滤波

```c
// 微分项一阶低通滤波
typedef struct {
    float Kp, Ki, Kd;
    float alpha;       // 滤波系数 (0~1)
    float filtered_d;  // 滤波后的微分值
    float prev_error;
    float integral;
    float output_min, output_max;
} PID_FilteredD;

float pid_filtered_derivative(PID_FilteredD *pid, float sp, float pv) {
    float error = sp - pv;
    float raw_d = error - pid->prev_error;
    
    // 一阶低通滤波
    pid->filtered_d = pid->alpha * pid->filtered_d + (1 - pid->alpha) * raw_d;
    
    pid->integral += error;
    pid->prev_error = error;
    
    float output = pid->Kp * error + pid->Ki * pid->integral + pid->Kd * pid->filtered_d;
    return CLAMP(output, pid->output_min, pid->output_max);
}
```

### 1.4 变积分PID

```c
// 误差大时减小积分，误差小时增大积分
float pid_variable_integral(PID_Position *pid, float sp, float pv) {
    float error = sp - pv;
    float abs_error = fabsf(error);
    
    // 积分增益随误差变化
    float Ki_effective;
    if (abs_error > pid->error_threshold_large) {
        Ki_effective = 0;  // 大误差时关闭积分
    } else if (abs_error > pid->error_threshold_small) {
        // 线性插值
        float ratio = (pid->error_threshold_large - abs_error) / 
                      (pid->error_threshold_large - pid->error_threshold_small);
        Ki_effective = pid->Ki * ratio;
    } else {
        Ki_effective = pid->Ki;
    }
    
    pid->integral += Ki_effective * error;
    
    float derivative = error - pid->prev_error;
    pid->prev_error = error;
    
    float output = pid->Kp * error + pid->integral + pid->Kd * derivative;
    return CLAMP(output, pid->output_min, pid->output_max);
}
```

### 1.5 PID参数自整定

```c
// 继电反馈法（Relay Feedback）自整定
typedef struct {
    float amplitude;    // 振荡幅度
    float period;       // 振荡周期
    float hysteresis;   // 迟滞宽度
    int state;          // 0=上升, 1=下降
    float peak_high, peak_low;
    uint32_t peak_time_high, peak_time_low;
    int cycle_count;
    float Ku, Tu;       // 临界增益和临界周期
} RelayAutoTune;

void relay_autotune_init(RelayAutoTune *at, float hysteresis) {
    at->hysteresis = hysteresis;
    at->state = 0;
    at->peak_high = -1e9;
    at->peak_low = 1e9;
    at->cycle_count = 0;
}

float relay_autotune_step(RelayAutoTune *at, float pv, uint32_t time_ms) {
    if (at->state == 0) {
        // 输出正向
        if (pv > at->peak_high) {
            at->peak_high = pv;
            at->peak_time_high = time_ms;
        }
        if (pv < at->peak_high - at->hysteresis) {
            at->state = 1;
            at->peak_low = pv;
            at->peak_time_low = time_ms;
        }
        return RELAY_AMPLITUDE;
    } else {
        // 输出负向
        if (pv < at->peak_low) {
            at->peak_low = pv;
            at->peak_time_low = time_ms;
        }
        if (pv > at->peak_low + at->hysteresis) {
            at->state = 0;
            at->cycle_count++;
            
            if (at->cycle_count >= 3) {
                // 计算临界增益和周期
                at->Tu = (at->peak_time_high - at->peak_time_low) * 2.0f / 1000.0f;
                at->amplitude = (at->peak_high - at->peak_low) / 2.0f;
                at->Ku = 4.0f * RELAY_AMPLITUDE / (M_PI * at->amplitude);
                
                // Ziegler-Nichols整定
                // Kp = 0.6*Ku, Ki = 2*Kp/Tu, Kd = Kp*Tu/8
                float Kp = 0.6f * at->Ku;
                float Ki = 2.0f * Kp / at->Tu;
                float Kd = Kp * at->Tu / 8.0f;
                
                // 整定完成
                return -1; // 表示完成
            }
            
            at->peak_high = pv;
            at->peak_time_high = time_ms;
        }
        return -RELAY_AMPLITUDE;
    }
}
```

## 2. 串级控制系统

### 2.1 串级控制基本结构

```
设定值 ──→ [主控制器] ──→ [副控制器] ──→ [执行器] ──→ [副过程] ──→ [主过程] ──→ 输出
              ↑   ──────────────────← 副反馈 ←─────────────────← 主反馈 ←──
              └─────────────────────←───────────────────────────←─────────┘
```

### 2.2 串级控制实现

```c
// 串级PID控制器
typedef struct {
    // 主控制器（外环）
    struct {
        float Kp, Ki, Kd;
        float integral, prev_error;
        float output_min, output_max;
    } master;
    
    // 副控制器（内环）
    struct {
        float Kp, Ki, Kd;
        float integral, prev_error;
        float output_min, output_max;
    } slave;
    
    // 副回路设定值（来自主控制器输出）
    float slave_setpoint;
} CascadePID;

float cascade_pid_update(CascadePID *cascade, float main_sp, 
                          float main_pv, float slave_pv) {
    // 主控制器计算
    float main_error = main_sp - main_pv;
    cascade->master.integral += main_error;
    float main_derivative = main_error - cascade->master.prev_error;
    cascade->master.prev_error = main_error;
    
    float main_output = cascade->master.Kp * main_error 
                       + cascade->master.Ki * cascade->master.integral 
                       + cascade->master.Kd * main_derivative;
    main_output = CLAMP(main_output, cascade->master.output_min, cascade->master.output_max);
    
    // 主控制器输出作为副控制器设定值
    cascade->slave_setpoint = main_output;
    
    // 副控制器计算
    float slave_error = cascade->slave_setpoint - slave_pv;
    cascade->slave.integral += slave_error;
    float slave_derivative = slave_error - cascade->slave.prev_error;
    cascade->slave.prev_error = slave_error;
    
    float slave_output = cascade->slave.Kp * slave_error 
                        + cascade->slave.Ki * cascade->slave.integral 
                        + cascade->slave.Kd * slave_derivative;
    slave_output = CLAMP(slave_output, cascade->slave.output_min, cascade->slave.output_max);
    
    return slave_output;
}
```

### 2.3 温度串级控制案例

```c
// 烤箱温度串级控制
// 主回路：炉温控制（慢过程）
// 副回路：加热丝功率控制（快过程）

typedef struct {
    CascadePID pid;
    float temp_setpoint;      // 温度设定值
    float current_temp;       // 当前温度
    float heater_power;       // 加热丝功率
    float heater_current;     // 加热丝电流反馈
} OvenCascadeControl;

void oven_cascade_init(OvenCascadeControl *oven, float temp_sp) {
    oven->temp_setpoint = temp_sp;
    
    // 主控制器参数（温度环，响应慢）
    oven->pid.master.Kp = 5.0f;
    oven->pid.master.Ki = 0.02f;
    oven->pid.master.Kd = 1.0f;
    oven->pid.master.output_min = 0;
    oven->pid.master.output_max = 100; // 最大功率百分比
    
    // 副控制器参数（功率环，响应快）
    oven->pid.slave.Kp = 2.0f;
    oven->pid.slave.Ki = 0.5f;
    oven->pid.slave.Kd = 0;
    oven->pid.slave.output_min = 0;
    oven->pid.slave.output_max = 100;
}

void oven_cascade_update(OvenCascadeControl *oven) {
    // 读取传感器
    oven->current_temp = read_temperature_sensor();
    oven->heater_current = read_heater_current();
    
    // 串级控制计算
    float pwm_duty = cascade_pid_update(&oven->pid, oven->temp_setpoint, 
                                         oven->current_temp, oven->heater_current);
    
    // 输出到加热丝
    set_heater_pwm(pwm_duty);
    oven->heater_power = pwm_duty;
}
```

### 2.4 串级控制参数整定原则

```
整定顺序：先副环，后主环

1. 副环整定：
   - 将主控制器设为比例控制，Kp=1
   - 使用Ziegler-Nichols法整定副控制器
   - 副环应比主环快3~10倍

2. 主环整定：
   - 副环闭合后，整定主控制器
   - 主环可使用更大的比例增益
   - 主环积分时间应大于副环响应时间
```

## 3. 比值控制系统

### 3.1 比值控制基本原理

```
主流量Q1 ──→ [比值器K] ──→ [副流量控制器] ──→ [调节阀] ──→ Q2
                               ↑                              |
                               └──── 流量测量 ←───────────────┘
                               
要求：Q2 = K × Q1
```

### 3.2 单闭环比值控制

```c
// 单闭环比值控制
typedef struct {
    float ratio;           // 比值系数 K
    float main_flow;       // 主流量（开环测量）
    float slave_flow_sp;   // 副流量设定值
    float slave_flow_pv;   // 副流量测量值
    PID_Incremental slave_pid;
    float output;          // 控制输出
} SingleLoopRatio;

void ratio_control_init(SingleLoopRatio *rc, float K) {
    rc->ratio = K;
    rc->slave_pid.Kp = 2.0f;
    rc->slave_pid.Ki = 0.1f;
    rc->slave_pid.Kd = 0;
}

float ratio_control_update(SingleLoopRatio *rc, float main_flow, float slave_flow) {
    rc->main_flow = main_flow;
    rc->slave_flow_pv = slave_flow;
    
    // 计算副流量设定值
    rc->slave_flow_sp = rc->ratio * main_flow;
    
    // 副PID控制
    rc->output = pid_incremental_update(&rc->slave_pid, 
                                         rc->slave_flow_sp, 
                                         rc->slave_flow_pv);
    return rc->output;
}
```

```c
typedef struct {
    float ratio;
    float main_flow_sp;
    float main_flow_pv;
    float slave_flow_sp;
    float slave_flow_pv;

    PID_Incremental slave_pid;

                                float main_sp, float main_pv, float slave_pv) {

    drc->slave_flow_sp = drc->ratio * main_output;
    
    // 副流量闭环控制
    float slave_output = pid_incremental_update(&drc->slave_pid, 
                                                 drc->slave_flow_sp, slave_pv);
    return slave_output;
}
```

## 4. 前馈控制系统

### 4.1 前馈控制原理

```
扰动d ──→ [前馈控制器Gf] ──┐
                              ├──→ [执行器] ──→ [过程] ──→ 输出
设定值 ──→ [反馈控制器Gc] ──┘        ↑                       |
                                      └──── 测量 ←────────────┘
```

前馈控制的目的是在扰动影响输出之前就进行补偿。

### 4.2 静态前馈

```c
// 静态前馈：输出与扰动成比例
typedef struct {
    float Kf;           // 前馈增益
    float Kp;           // 过程增益（扰动到输出）
    float Kv;           // 控制通道增益
    PID_Position feedback_pid;
} StaticFeedForward;

void static_ff_init(StaticFeedForward *ff, float Kp, float Kv) {
    ff->Kp = Kp;
    ff->Kv = Kv;
    ff->Kf = -Kp / Kv;  // 静态前馈系数
}

float static_ff_update(StaticFeedForward *ff, float sp, float pv, float disturbance) {
    // 前馈补偿
    float ff_output = ff->Kf * disturbance;
    
    // 反馈控制
    float fb_output = pid_position_update(&ff->feedback_pid, sp, pv);
    
    // 总输出 = 前馈 + 反馈
    return ff_output + fb_output;
}
```

### 4.3 动态前馈

```c
// 动态前馈：考虑扰动到输出的动态特性
typedef struct {
    float Kf;           // 前馈增益
    float Tf;           // 前馈时间常数
    float Td;           // 前馈微分时间
    float prev_disturbance;
    float filtered_ff;
    float alpha;        // 滤波系数
    PID_Position feedback_pid;
} DynamicFeedForward;

void dynamic_ff_init(DynamicFeedForward *ff, float Kf, float Tf, float Td, float dt) {
    ff->Kf = Kf;
    ff->Tf = Tf;
    ff->Td = Td;
    ff->alpha = Tf / (Tf + dt);
}

float dynamic_ff_update(DynamicFeedForward *ff, float sp, float pv, 
                         float disturbance, float dt) {
    // 动态前馈：Kf * (1 + Td*s) / (1 + Tf*s)
    // 离散化为一阶差分
    
    float d_dist = (disturbance - ff->prev_disturbance) / dt;
    ff->prev_disturbance = disturbance;
    
    // 前馈输出（带一阶滤波）
    float ff_raw = ff->Kf * (disturbance + ff->Td * d_dist);
    ff->filtered_ff = ff->alpha * ff->filtered_ff + (1 - ff->alpha) * ff_raw;
    
    // 反馈控制
    float fb_output = pid_position_update(&ff->feedback_pid, sp, pv);
    
    return ff->filtered_ff + fb_output;
}
```

### 4.4 前馈-反馈复合控制

```c
// 前馈-反馈复合控制完整实现
typedef struct {
    // 前馈通道
    float ff_gain;
    float ff_T1, ff_T2;  // 前馈动态参数
    float ff_state;      // 前馈滤波状态
    
    // 反馈通道
    PID_FilteredD fb_pid;
    
    // 输出限幅
    float output_min, output_max;
} FeedForwardFeedback;

float fffb_update(FeedForwardFeedback *ctrl, float sp, float pv, 
                   float disturbance, float dt) {
    // 前馈计算
    float ff_output = ctrl->ff_gain * disturbance;
    // 一阶惯性滤波
    float alpha = ctrl->ff_T1 / (ctrl->ff_T1 + dt);
    ctrl->ff_state = alpha * ctrl->ff_state + (1 - alpha) * ff_output;
    
    // 反馈计算
    float fb_output = pid_filtered_derivative(&ctrl->fb_pid, sp, pv);
    
    // 合成
    float total = ctrl->ff_state + fb_output;
    return CLAMP(total, ctrl->output_min, ctrl->output_max);
}
```

### 4.5 前馈控制典型应用

```c
// 温度前馈控制（补偿环境温度变化）
typedef struct {
    FeedForwardFeedback ctrl;
    float ambient_temp;      // 环境温度
    float prev_ambient;
} TempFeedForward;

float temp_ff_update(TempFeedForward *tff, float sp, float pv, float dt) {
    // 环境温度变化作为扰动
    float disturbance = tff->ambient_temp - tff->prev_ambient;
    tff->prev_ambient = tff->ambient_temp;
    
    return fffb_update(&tff->ctrl, sp, pv, disturbance, dt);
}

// 流量前馈控制（补偿压力变化）
typedef struct {
    FeedForwardFeedback ctrl;
    float supply_pressure;   // 供气压力
    float prev_pressure;
} FlowFeedForward;

float flow_ff_update(FlowFeedForward *fff, float sp, float pv, float dt) {
    float disturbance = fff->supply_pressure - fff->prev_pressure;
    fff->prev_pressure = fff->supply_pressure;
    
    return fffb_update(&fff->ctrl, sp, pv, disturbance, dt);
}
```

## 5. 选择性控制系统

### 5.1 选择性控制原理

```
当某个变量超出安全范围时，切换到安全控制器。

正常工况：[控制器1] ──→ 选择器 ──→ [执行器]
异常工况：[控制器2] ──→ 选择器 ──→ [执行器]
                     ↑
                 [选择逻辑]
```

### 5.2 高选/低选控制

```c
// 高选器（选择较大值）
float high_select(float a, float b) {
    return (a > b) ? a : b;
}

// 低选器（选择较小值）
float low_select(float a, float b) {
    return (a < b) ? a : b;
}

// 选择性控制器
typedef struct {
    PID_Position normal_pid;    // 正常工况控制器
    PID_Position safety_pid;    // 安全工况控制器
    float safety_limit_high;    // 安全上限
    float safety_limit_low;     // 安全下限
    int safety_active;          // 安全控制器激活标志
} SelectiveControl;

float selective_control_update(SelectiveControl *sc, float sp, float pv, 
                                float safety_pv) {
    float normal_output = pid_position_update(&sc->normal_pid, sp, pv);
    float safety_output = pid_position_update(&sc->safety_pid, 
                                               sc->safety_limit_high, safety_pv);
    
    // 选择逻辑：如果安全变量超限，使用安全控制器
    if (safety_pv > sc->safety_limit_high) {
        sc->safety_active = 1;
        return low_select(normal_output, safety_output); // 取较小值（减少输出）
    } else if (safety_pv < sc->safety_limit_low) {
        sc->safety_active = 1;
        return high_select(normal_output, safety_output); // 取较大值（增加输出）
    } else {
        sc->safety_active = 0;
        return normal_output;
    }
}
```

### 5.3 多路选择性控制

```c
// 多路选择性控制（多变量约束）
typedef struct {
    PID_Position pid[4];      // 4个控制器
    float limits[4];          // 各变量限制值
    int active_controller;    // 当前激活的控制器
    int num_controllers;
} MultiSelectControl;

float multi_select_update(MultiSelectControl *msc, float sp, float *pv_array) {
    float outputs[4];
    int min_idx = 0;
    float min_output = 1e9;
    
    // 计算所有控制器输出
    for (int i = 0; i < msc->num_controllers; i++) {
        if (i == 0) {
            // 主控制器
            outputs[i] = pid_position_update(&msc->pid[i], sp, pv_array[i]);
        } else {
            // 约束控制器
            outputs[i] = pid_position_update(&msc->pid[i], msc->limits[i], pv_array[i]);
        }
        
        // 选择最小输出（低选逻辑）
        if (outputs[i] < min_output) {
            min_output = outputs[i];
            min_idx = i;
        }
    }
    
    msc->active_controller = min_idx;
    return min_output;
}
```

```c
    set_feedwater_valve(feedwater_cmd);
```

## 6. 过程控制工程实践

### 6.1 控制系统设计流程

```
1. 工艺分析
   ├── 确定被控变量
   ├── 确定操纵变量
   ├── 分析扰动因素
   └── 确定控制目标

2. 控制方案选择
   ├── 简单回路 → 单回路PID
   ├── 主要扰动可测 → 前馈+反馈
   ├── 多个变量相关 → 串级/比值
   ├── 有安全约束 → 选择性控制
   └── 大滞后过程 → Smith预估/内模控制

3. 设备选型
   ├── 传感器/变送器
   ├── 执行器（阀门/电机）
   ├── 控制器（PLC/DCS/嵌入式）
   └── 通信接口

4. 参数整定
   ├── 理论计算初始值
   ├── 现场调试优化
   └── 记录最终参数
```

```c
// 阀门特性补偿
typedef enum {
    VALVE_LINEAR,       // 线性
    VALVE_EQUAL_PERCENT,// 等百分比
    VALVE_QUICK_OPEN    // 快开
} ValveCharacteristic;

float valve_linearize(ValveCharacteristic type, float command) {
    // command: 0~100%
    switch (type) {
        case VALVE_LINEAR:
            return command;
        case VALVE_EQUAL_PERCENT:
            // 等百分比：R^(x/100) - 1) / (R - 1)
            return (powf(30, command / 100.0f) - 1) / 29.0f * 100.0f;
        case VALVE_QUICK_OPEN:
            return sqrtf(command / 100.0f) * 100.0f;
        default:
            return command;
    }
}
```

### 6.3 通信与数据记录

```c
// Modbus通信读写（RS485）
typedef struct {
    UART_HandleTypeDef *huart;
    uint8_t slave_addr;
    uint16_t holding_regs[100];
} ModbusRTU;

int modbus_read_holding(ModbusRTU *modbus, uint16_t start_addr, uint16_t count, 
                         uint16_t *data) {
    uint8_t tx_buf[8];
    tx_buf[0] = modbus->slave_addr;
    tx_buf[1] = 0x03; // 读保持寄存器
    tx_buf[2] = start_addr >> 8;
    tx_buf[3] = start_addr & 0xFF;
    tx_buf[4] = count >> 8;
    tx_buf[5] = count & 0xFF;
    
    uint16_t crc = modbus_crc16(tx_buf, 6);
    tx_buf[6] = crc & 0xFF;
    tx_buf[7] = crc >> 8;
    
    HAL_UART_Transmit(modbus->huart, tx_buf, 8, 100);
    
    // 接收响应
    uint8_t rx_buf[256];
    HAL_UART_Receive(modbus->huart, rx_buf, 3 + 2 * count + 2, 100);
    
    // 解析数据
    for (int i = 0; i < count; i++) {
        data[i] = (rx_buf[3 + 2*i] << 8) | rx_buf[3 + 2*i + 1];
    }
    
    return 0;
}
```

```c
typedef struct {
    CascadePID pid;        // 串级PID
    float temp_sp;         // 温度设定值
    float current_temp;    // 当前温度
    float heater_power;    // 加热功率

    float fan_speed;       // 风扇转速
} TemperatureControlSystem;

void temp_sys_init(TemperatureControlSystem *sys, float sp) {
    sys->temp_sp = sp;
    
    // 加热PID
    sys->pid.master.Kp = 8.0f;
    sys->pid.master.Ki = 0.05f;
    sys->pid.master.Kd = 2.0f;
    
    // 风扇PID
    sys->fan_pid.Kp = 3.0f;
    sys->fan_pid.Ki = 0.1f;
    sys->fan_pid.Kd = 0.5f;
    sys->fan_pid.alpha = 0.8f;
}

void temp_sys_update(TemperatureControlSystem *sys) {
    sys->current_temp = read_temperature();
    
    float error = sys->temp_sp - sys->current_temp;
    
    if (error > 0) {
        // 需要加热
        sys->heater_power = pid_position_update(&sys->pid.master, 
                                                 sys->temp_sp, sys->current_temp);
        sys->fan_speed = 0;
    } else {
        // 需要降温
        sys->heater_power = 0;
        sys->fan_speed = pid_filtered_derivative(&sys->fan_pid, 
                                                   -sys->temp_sp, -sys->current_temp);
    }
    
    set_heater_pwm(sys->heater_power);
    set_fan_pwm(sys->fan_speed);
}
```

```c
typedef struct {
    CascadePID cascade;    // 串级控制
    FeedForwardFeedback ff; // 前馈补偿
    float level_sp;        // 液位设定值

float level_sys_update(LevelControlSystem *sys, float dt) {
    sys->tank_level = read_level_sensor();
    sys->inflow = read_inflow_sensor();
    
    // 串级控制（液位→阀门开度）
    float cascade_out = cascade_pid_update(&sys->cascade, 
                                            sys->level_sp, 
                                            sys->tank_level, 
                                            sys->outflow);
    
    // 前馈补偿（入口流量变化）
    float ff_out = fffb_update(&sys->ff, sys->level_sp, 
                                sys->tank_level, sys->inflow, dt);
    
    // 合成输出
    float valve_cmd = cascade_out + ff_out;
    set_valve_position(CLAMP(valve_cmd, 0, 100));
    
    return valve_cmd;
}
```

```c
float motor_dual_loop_update(MotorDualLoop *motor, float speed_sp, 
                              float speed_fb, float current_fb) {
    motor->speed_sp = speed_sp;
    motor->speed_pv = speed_fb;
    motor->current_pv = current_fb;
    
    // 速度环（外环）
    float speed_out = pid_position_update(&motor->speed_pid, speed_sp, speed_fb);
    speed_out = CLAMP(speed_out, -motor->max_current, motor->max_current);
    
    // 电流环（内环）
    motor->current_sp = speed_out;
    float current_out = pid_position_update(&motor->current_pid, 
                                             motor->current_sp, current_fb);
    current_out = CLAMP(current_out, -motor->max_voltage, motor->max_voltage);
    
    return current_out;
}
```

## 8. 调试技巧与注意事项

### 8.1 参数整定速查表

| 控制类型 | Kp | Ti | Td | 适用场景 |
|---------|-----|-----|-----|---------|
| 温度控制 | 2~10 | 60~300s | 15~60s | 大惯性、大滞后 |
| 液位控制 | 1~5 | 30~120s | 5~20s | 积分过程 |
| 流量控制 | 0.5~2 | 5~30s | 0~5s | 快速响应 |
| 压力控制 | 1~4 | 10~60s | 5~15s | 中等响应 |
| 速度控制 | 0.5~3 | 10~50s | 2~10s | 电机控制 |

### 8.2 常见问题排查

| 问题 | 可能原因 | 解决方法 |
|------|---------|---------|
| 持续振荡 | Kp过大 | 减小比例增益 |
| 响应太慢 | Ki太小 | 增大积分增益 |
| 超调太大 | Kd太小或Ki太大 | 增大微分或减小积分 |
| 稳态误差 | Ki为0或太小 | 增加积分作用 |
| 输出饱和 | 限幅设置不当 | 检查限幅参数 |
| 测量噪声 | 传感器问题 | 增加滤波 |
| 阀门抖动 | 控制器过于灵敏 | 增加死区或滤波 |

```c
    }
}
```
