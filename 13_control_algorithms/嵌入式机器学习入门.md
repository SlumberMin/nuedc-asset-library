# 嵌入式机器学习入门（Embedded ML / TinyML）

> 适用于电赛中需要在MCU上实现智能感知、分类、预测的场景

---

## 1. 概述

嵌入式机器学习（TinyML）是指在资源受限的微控制器上运行机器学习模型，实现边缘端智能推理。

### 典型电赛应用场景
| 场景 | 说明 | 典型模型 |
|------|------|----------|
| 手势识别 | 加速度计/陀螺仪数据分类 | 1D-CNN / DNN |
| 语音关键词检测 | "开"/"关"等简单命令 | DS-CNN / LSTM |
| 异常检测 | 电机振动/电流异常 | Autoencoder |
| 图像分类 | 简单物体识别（限时题） | MobileNet / SqueezeNet |
| 信号分类 | 调制方式识别/故障诊断 | FC / CNN |
| 预测控制 | 温度/电压趋势预测 | LSTM / GRU |

### 资源约束参考
| MCU | Flash | RAM | 主频 | 推理能力 |
|-----|-------|-----|------|----------|
| STM32F103 | 64-512KB | 20-64KB | 72MHz | 极简单模型 |
| STM32F407 | 512KB-1MB | 192KB | 168MHz | 小型DNN/CNN |
| STM32H743 | 2MB | 1MB | 480MHz | 中型CNN |
| ESP32 | 4MB Flash | 520KB | 240MHz | 小型DNN |
| nRF52840 | 1MB | 256KB | 64MHz | 微型模型 |

## 2. TensorFlow Lite for Microcontrollers

### 2.1 架构概览

```
┌─────────────────────────────────┐
│      TensorFlow (Python)        │  ← 训练
├─────────────────────────────────┤
│  TF Lite Converter (.tflite)    │  ← 量化转换
├─────────────────────────────────┤
│  TF Lite for Microcontrollers   │  ← MCU推理引擎
├─────────────────────────────────┤
│  MCU (Cortex-M / ESP32)         │  ← 硬件
└─────────────────────────────────┘
```

### 2.2 Python端：模型训练与量化

```python
import tensorflow as tf
import numpy as np

# ===== 1. 定义模型 =====
model = tf.keras.Sequential([
    tf.keras.layers.Dense(64, activation='relu', input_shape=(10,)),
    tf.keras.layers.Dense(32, activation='relu'),
    tf.keras.layers.Dense(3, activation='softmax')  # 3分类
])

model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# 假设已有训练数据 X_train, y_train
# model.fit(X_train, y_train, epochs=50, validation_split=0.2)

# ===== 2. 转换为TFLite（全整数量化） =====
def representative_dataset():
    """提供校准数据集（100-500样本）"""
    for i in range(100):
        yield [X_train[i:i+1].astype(np.float32)]

converter = tf.lite.TFLiteConverter.from_keras_model(model)
converter.optimizations = [tf.lite.Optimize.DEFAULT]
converter.representative_dataset = representative_dataset
converter.target_spec.supported_ops = [tf.lite.OpsSet.TFLITE_BUILTINS_INT8]
converter.inference_input_type = tf.int8
converter.inference_output_type = tf.int8

tflite_model = converter.convert()

# 保存
with open('model_int8.tflite', 'wb') as f:
    f.write(tflite_model)

# ===== 3. 转换为C数组 =====
def convert_to_c_array(tflite_file, var_name='g_model'):
    with open(tflite_file, 'rb') as f:
        data = f.read()
    
    hex_lines = []
    for i in range(0, len(data), 12):
        chunk = data[i:i+12]
        hex_str = ', '.join(f'0x{b:02x}' for b in chunk)
        hex_lines.append(f'  {hex_str}')
    
    c_code = f'#include <cstdint>\n\n'
    c_code += f'alignas(16) const unsigned char {var_name}[] = {{\n'
    c_code += ',\n'.join(hex_lines)
    c_code += f'\n}};\n'
    c_code += f'const unsigned int {var_name}_len = {len(data)};\n'
    return c_code

with open('model_data.cc', 'w') as f:
    f.write(convert_to_c_array('model_int8.tflite'))
print("模型大小:", len(tflite_model), "bytes")
```

### 2.3 MCU端推理代码（STM32 HAL）

```cpp
// main.cpp - STM32 + TFLite Micro
#include "tensorflow/lite/micro/micro_interpreter.h"
#include "tensorflow/lite/micro/micro_mutable_op_resolver.h"
#include "tensorflow/lite/schema/schema_generated.h"
#include "model_data.h"  // 模型C数组

// 运算符注册（只注册用到的算子，节省Flash）
constexpr int kTensorArenaSize = 16 * 1024;
alignas(16) static uint8_t tensor_arena[kTensorArenaSize];

const tflite::Model* model = nullptr;
tflite::MicroInterpreter* interpreter = nullptr;
TfLiteTensor* input = nullptr;
TfLiteTensor* output = nullptr;

void ml_init() {
    // 加载模型
    model = tflite::GetModel(g_model);
    
    // 注册算子（按模型实际需要）
    static tflite::MicroMutableOpResolver<8> resolver;
    resolver.AddFullyConnected();
    resolver.AddRelu();
    resolver.AddSoftmax();
    resolver.AddReshape();
    
    // 创建解释器
    static tflite::MicroInterpreter static_interpreter(
        model, resolver, tensor_arena, kTensorArenaSize);
    interpreter = &static_interpreter;
    
    // 分配内存
    if (interpreter->AllocateTensors() != kTfLiteOk) {
        Error_Handler();
    }
    
    input = interpreter->input(0);
    output = interpreter->output(0);
}

// 推理
int ml_predict(float* features, int num_features) {
    // 量化输入（int8模型）
    float input_scale = input->params.scale;
    int input_zero_point = input->params.zero_point;
    
    for (int i = 0; i < num_features; i++) {
        int8_t quantized = (int8_t)(features[i] / input_scale + input_zero_point);
        input->data.int8[i] = quantized;
    }
    
    // 执行推理
    if (interpreter->Invoke() != kTfLiteOk) {
        return -1;
    }
    
    // 找最大输出
    int max_idx = 0;
    int8_t max_val = output->data.int8[0];
    for (int i = 1; i < output->bytes; i++) {
        if (output->data.int8[i] > max_val) {
            max_val = output->data.int8[i];
            max_idx = i;
        }
    }
    
    return max_idx;
}

// 使用示例
int main() {
    HAL_Init();
    SystemClock_Config();
    ml_init();
    
    while (1) {
        float sensor_data[10];
        // 采集传感器数据...
        int result = ml_predict(sensor_data, 10);
        // result = 0: 类别A, 1: 类别B, 2: 类别C
    }
}
```

**Keil MDK配置要点：**
- C++11以上
- 添加 `tensorflow/lite/micro/` 到Include路径
- Flash至少256KB，RAM至少64KB
- 堆大小设为 `kTensorArenaSize` + 预留

## 3. CMSIS-NN 加速

### 3.1 概述
ARM CMSIS-NN库为Cortex-M处理器优化的神经网络算子，可显著加速推理。

### 3.2 性能对比（Cortex-M4, 100MHz）
| 算子 | 朴素C实现 | CMSIS-NN | 加速比 |
|------|-----------|----------|--------|
| FC (256→128) | 12ms | 3ms | 4x |
| Conv 3x3 | 45ms | 12ms | 3.75x |
| MaxPool | 2ms | 0.8ms | 2.5x |

### 3.3 使用方法

```cpp
#include "arm_nnfunctions.h"

// 量化全连接层
arm_fully_connected_q7(
    pInput,           // int8输入
    pWeight,          // int8权重
    input_size,       // 输入维度
    output_size,      // 输出维度
    bias_shift,       // 偏置移位
    output_shift,     // 输出移位
    bias,             // 偏置
    pOut,             // int8输出
    scratch_buf       // 临时缓冲区
);

// 量化卷积层
arm_convolve_HWC_q7_basic(
    Im_in,            // 输入特征图
    dim_im_in,        // 输入尺寸
    ch_im_in,         // 输入通道
    wt,               // 权重
    ch_im_out,        // 输出通道
    dim_kernel,       // 卷积核大小
    padding,          // 填充
    stride,           // 步长
    bias,             // 偏置
    bias_shift,
    out_shift,
    Im_out,           // 输出
    dim_im_out,       // 输出尺寸
    bufferA,          // 缓冲区
    bufferB           // 缓冲区
);
```

## 5. 模型优化技巧

### 5.1 量化策略
| 方案 | 精度损失 | 大小缩减 | 速度提升 | 适用场景 |
|------|----------|----------|----------|----------|
| Float32 → Float16 | 极小 | 2x | 1-2x | 对精度敏感 |
| Float32 → Int8 | 小 | 4x | 2-4x | **推荐默认** |
| Float32 → Int4 | 中等 | 8x | 3-5x | 极端资源受限 |

### 5.2 模型压缩
```python
# 剪枝（Pruning）
import tensorflow_model_optimization as tfmot

pruning_schedule = tfmot.sparsity.keras.PolynomialDecay(
    initial_sparsity=0.30,
    final_sparsity=0.80,  # 去掉80%权重
    begin_step=0,
    end_step=end_step
)

model_for_pruning = tfmot.sparsity.keras.prune_low_magnitude(
    model, pruning_schedule=pruning_schedule
)
```

### 5.3 输入预处理优化
```cpp
// 避免浮点运算，直接用int8
// 坏的做法：先转float再转int8
// 好的做法：直接量化传感器原始值

int8_t quantize_raw(int16_t adc_value, float scale, int zero_point) {
    return (int8_t)((adc_value * scale) + zero_point);
}
```

## 6. 常见问题排查

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| AllocateTensors失败 | arena太小 | 增大kTensorArenaSize |
| 推理结果全为0 | 输入量化参数错误 | 检查scale和zero_point |
| 精度下降严重 | 量化损失 | 用更多校准数据/混合量化 |
| Flash溢出 | 模型太大 | 剪枝+量化/换更大MCU |
| 推理太慢 | 未启用CMSIS-NN | 检查CMSIS-NN集成 |
| 编译报错 | 算子未注册 | 在OpResolver中添加缺失算子 |

## 7. 电赛实战建议

1. **优先用Edge Impulse**：图形化界面，快速迭代，适合比赛限时
2. **模型尽量小**：Flash<100KB，Arena<32KB，确保资源余量
3. **Int8量化必用**：在Cortex-M上Int8比Float快3-5倍
4. **提前准备数据集**：比赛前训练好基础模型，现场微调
5. **打印推理时间**：用 `HAL_GetTick()` 计时，确保满足实时性
6. **备用方案**：准备简单的阈值/规则方法，ML模型训练不理想时回退
