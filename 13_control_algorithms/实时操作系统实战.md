# 实时操作系统（RTOS）实战指南

> FreeRTOS 在电赛中的完整应用：任务管理、同步通信、内存管理

---

## 1. 为什么需要 RTOS

### 1.1 裸机 vs RTOS

| 对比项 | 裸机（轮询/中断） | RTOS |
|--------|-------------------|------|
| 复杂度 | 简单任务适合 | 多任务并行适合 |
| 实时性 | 依赖中断优先级 | 任务优先级保证 |
| 可维护性 | 状态机复杂 | 模块化清晰 |
| 资源占用 | 极小 | RAM多用2-10KB |
| 调试难度 | 串行好调试 | 需要RTOS调试工具 |

### 1.2 电赛典型RTOS场景
- 同时采集多路传感器 + PID控制 + 显示更新 + 通信
- 多协议通信（UART + SPI + I2C并行处理）
- 复杂状态机（自动题多模式切换）
- 低功耗管理（空闲任务自动休眠）

## 2. FreeRTOS 快速上手

### 2.1 STM32CubeMX 配置

```
1. Pinout → SYS → Debug: Serial Wire
2. Pinout → SYS → Timebase Source: TIM6（不能用SysTick！）
3. Middleware → FREERTOS → Interface: CMSIS_V2
4. Config Parameters:
   - TOTAL_HEAP_SIZE: 8192~32768（根据RAM调整）
   - MAX_PRIORITIES: 56
   - TICK_RATE_HZ: 1000
5. Tasks and Queues → 添加任务
```

### 2.3 基本任务创建

```c
#include "FreeRTOS.h"
#include "task.h"

// 任务句柄
TaskHandle_t hTaskSensor;
TaskHandle_t hTaskControl;
TaskHandle_t hTaskDisplay;

// 任务函数
void TaskSensor(void *pvParameters) {
    for (;;) {
        // 读取传感器
        float voltage = ADC_Read();
        float current = ACS712_Read();
        
        // 发送到队列（非阻塞）
        SensorData_t data = {voltage, current, HAL_GetTick()};
        xQueueSend(xQueueSensor, &data, 0);
        
        vTaskDelay(pdMS_TO_TICKS(10));  // 100Hz采样
    }
}

void TaskControl(void *pvParameters) {
    SensorData_t data;
    for (;;) {
        // 等待传感器数据
        if (xQueueReceive(xQueueSensor, &data, pdMS_TO_TICKS(50)) == pdPASS) {
            // PID计算
            float output = PID_Calculate(&pid, data.voltage, target);
            PWM_SetDuty(output);
        }
    }
}

void TaskDisplay(void *pvParameters) {
    for (;;) {
        OLED_Clear();
        OLED_Printf(0, 0, "V:%.2f V", global_voltage);
        OLED_Printf(0, 2, "I:%.2f A", global_current);
        OLED_Update();
        vTaskDelay(pdMS_TO_TICKS(200));  // 5Hz刷新
    }
}

// main
int main(void) {
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_ADC1_Init();
    MX_TIM1_Init();
    MX_USART1_Init();
    
    // 创建任务
    xTaskCreate(TaskSensor,  "Sensor",  256, NULL, 3, &hTaskSensor);
    xTaskCreate(TaskControl, "Control", 256, NULL, 4, &hTaskControl);
    xTaskCreate(TaskDisplay, "Display", 256, NULL, 1, &hTaskDisplay);
    
    // 启动调度器
    vTaskStartScheduler();
    
    // 不应到达此处
    while (1) {}
}
```

## 3. 任务调度详解

### 3.1 任务优先级设计原则

```
优先级分配（数字越大优先级越高）：

P5 (最高) - 安全保护/紧急停止
P4        - 实时控制（PID/电机）
P3        - 传感器采集
P2        - 通信处理（UART/CAN）
P1 (最低) - 显示/日志/非关键任务
```

### 3.2 任务状态机

```
                  ┌──────────────┐
                  │   Running    │ ← CPU正在执行
                  └──────┬───────┘
                         │
              ┌──────────┼──────────┐
              ▼          ▼          ▼
     ┌──────────┐  ┌──────────┐  ┌──────────┐
     │  Ready   │  │ Blocked  │  │ Suspended│
     │(等CPU)   │  │(等事件)  │  │(被挂起)  │
     └──────────┘  └──────────┘  └──────────┘
```

### 3.3 任务管理API

```c
// 创建任务
BaseType_t xTaskCreate(
    TaskFunction_t pvTaskCode,

    TaskHandle_t *pxCreatedTask

```

### 3.4 时间管理

```c
// 延时
vTaskDelay(pdMS_TO_TICKS(100));           // 相对延时100ms
vTaskDelayUntil(&xLastWakeTime, period);  // 绝对周期延时（更精确）

// 精确周期任务示例（100Hz）
void Task_100Hz(void *pvParameters) {
    TickType_t xLastWakeTime = xTaskGetTickCount();
    const TickType_t xPeriod = pdMS_TO_TICKS(10);  // 10ms
    
    for (;;) {
        // 执行任务...
        PID_Task();
        
        // 等到下一个周期（补偿执行时间）
        vTaskDelayUntil(&xLastWakeTime, xPeriod);
    }
}

// 获取当前tick
TickType_t tick = xTaskGetTickCount();
uint32_t ms = tick * portTICK_PERIOD_MS;
```

## 4. 同步与通信机制

### 4.1 二值信号量（Binary Semaphore）

```c
// 用于中断通知任务
SemaphoreHandle_t xSemADC;

void main_init() {
    xSemADC = xSemaphoreCreateBinary();
}

// ADC完成中断
void HAL_ADC_ConvCpltCallback(ADC_HandleTypeDef* hadc) {
    BaseType_t xHigherPriorityTaskWoken = pdFALSE;
    xSemaphoreGiveFromISR(xSemADC, &xHigherPriorityTaskWoken);
    portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
}

// 任务等待ADC完成
void Task_ADC(void *pvParameters) {
    for (;;) {
        if (xSemaphoreTake(xSemADC, pdMS_TO_TICKS(100)) == pdPASS) {
            uint16_t adc_val = HAL_ADC_GetValue(&hadc1);
            // 处理数据...
        }
    }
}
```

### 4.3 计数信号量（Counting Semaphore）

```c
        }
    }
```

### 4.4 消息队列（Queue）

```c
// 最常用的通信方式
QueueHandle_t xQueueSensor;
QueueHandle_t xQueueCmd;

typedef struct {
    float voltage;
    float current;
    uint32_t timestamp;
} SensorData_t;

typedef struct {
    uint8_t cmd;
    int32_t param;
} Cmd_t;

void main_init() {
    xQueueSensor = xQueueCreate(10, sizeof(SensorData_t));  // 深度10
    xQueueCmd    = xQueueCreate(5, sizeof(Cmd_t));
}

// 生产者
void Task_Sensor(void *pvParameters) {
    SensorData_t data;
    for (;;) {
        data.voltage = ADC_ReadVoltage();
        data.current = ADC_ReadCurrent();
        data.timestamp = HAL_GetTick();
        
        xQueueSend(xQueueSensor, &data, pdMS_TO_TICKS(10));
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}

// 消费者
void Task_Process(void *pvParameters) {
    SensorData_t data;
    for (;;) {
        if (xQueueReceive(xQueueSensor, &data, pdMS_TO_TICKS(100)) == pdPASS) {
            // 处理数据
            PID_Update(data.voltage);
        }
    }
}

// UART命令解析
void Task_UART(void *pvParameters) {
    Cmd_t cmd;
    uint8_t rx_buf[32];
    for (;;) {
        int len = UART_Receive(rx_buf, sizeof(rx_buf), 100);
        if (len > 0) {
            cmd.cmd = rx_buf[0];
            cmd.param = *(int32_t*)&rx_buf[1];
            xQueueSend(xQueueCmd, &cmd, 0);
        }
    }
}
```

### 4.5 事件组（Event Group）

```c
// 多条件同步
EventHandle_t xEvents;

#define EVT_SENSOR_READY  (1 << 0)
#define EVT_UART_READY    (1 << 1)
#define EVT_TIMEOUT       (1 << 2)

void main_init() {
    xEvents = xEventGroupCreate();
}

// 设置事件
void Task_Sensor(void *pvParameters) {
    // 传感器初始化完成
    xEventGroupSetBits(xEvents, EVT_SENSOR_READY);
}

// 等待多个事件
void Task_Control(void *pvParameters) {
    EventBits_t bits = xEventGroupWaitBits(
        xEvents,
        EVT_SENSOR_READY | EVT_UART_READY,  // 等待这些位
        pdTRUE,    // 清除匹配位
        pdTRUE,    // 等待所有位（AND）
        pdMS_TO_TICKS(5000)  // 超时5s
    );
    
    if ((bits & (EVT_SENSOR_READY | EVT_UART_READY)) == 
        (EVT_SENSOR_READY | EVT_UART_READY)) {
        // 所有条件满足
        Start_Control_Loop();
    } else {
        // 超时
        Error_Handler();
    }
}
```

## 5. 内存管理

### 5.1 FreeRTOS 内存分配方案

| 方案 | 特点 | 适用场景 |
|------|------|----------|
| heap_1 | 只分配不释放 | 任务数固定 |
| heap_2 | 释放但不合并碎片 | **推荐大多数场景** |
| heap_3 | 包装标准malloc | 有RTOS安全的malloc |
| heap_4 | 释放+合并相邻空闲块 | 频繁分配/释放 |
| heap_5 | 跨不连续内存区域 | 多段RAM的MCU |

```
在FreeRTOSConfig.h中选择：
#define configUSE_HEAP_SCHEME  2  // 使用heap_2
```

```c
// 方法：先设大一点，运行后查高水位
void Task_Sensor(void *pvParameters) {
    for (;;) {
        // 正常运行...
        UBaseType_t hw = uxTaskGetStackHighWaterMark(NULL);
        printf("Sensor stack free: %d words\n", hw);
        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}
```

## 6. 常见设计模式

### 6.1 生产者-消费者模式

```
[传感器任务] --Queue--> [处理任务] --Queue--> [显示任务]
   (生产)                (消费+生产)            (消费)
```

### 6.2 请求-响应模式

```c
// 使用队列实现同步请求
QueueHandle_t xReqQueue;

typedef struct {
    uint8_t cmd;
    QueueHandle_t xRespQueue;  // 响应队列（深度1）
    int32_t response;
} Request_t;

// 客户端
int32_t SendRequest(uint8_t cmd) {
    Request_t req;
    req.cmd = cmd;
    req.xRespQueue = xQueueCreate(1, sizeof(int32_t));
    
    xQueueSend(xReqQueue, &req, portMAX_DELAY);
    
    int32_t resp;
    xQueueReceive(req.xRespQueue, &resp, pdMS_TO_TICKS(1000));
    vQueueDelete(req.xRespQueue);
    return resp;
}

// 服务端
void Task_Server(void *pvParameters) {
    Request_t req;
    for (;;) {
        if (xQueueReceive(xReqQueue, &req, portMAX_DELAY) == pdPASS) {
            int32_t result = ProcessCommand(req.cmd);
            xQueueSend(req.xRespQueue, &result, 0);
        }
    }
}
```

### 6.3 状态机任务

```c
typedef enum {
    STATE_IDLE,
    STATE_RUNNING,
    STATE_ERROR,
    STATE_STOP
} SystemState_t;

void Task_StateMachine(void *pvParameters) {
    SystemState_t state = STATE_IDLE;
    Cmd_t cmd;
    
    for (;;) {
        switch (state) {
        case STATE_IDLE:
            if (xQueueReceive(xQueueCmd, &cmd, portMAX_DELAY) == pdPASS) {
                if (cmd.cmd == CMD_START) state = STATE_RUNNING;
            }
            break;
            
        case STATE_RUNNING:
            Do_Control();
            if (xQueueReceive(xQueueCmd, &cmd, 0) == pdPASS) {
                if (cmd.cmd == CMD_STOP) state = STATE_IDLE;
                if (cmd.cmd == CMD_ERROR) state = STATE_ERROR;
            }
            break;
            
        case STATE_ERROR:
            Safe_Stop();
            vTaskDelay(pdMS_TO_TICKS(1000));
            state = STATE_IDLE;
            break;
            
        case STATE_STOP:
            Emergency_Stop();
            vTaskSuspend(NULL);  // 自挂起
            break;
        }
        
        vTaskDelay(pdMS_TO_TICKS(10));
    }
}
```

### 7.3 常见问题

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| HardFault | 栈溢出/非法内存 | 增大栈空间，检查指针 |
| 任务不运行 | 优先级被抢占 | 检查优先级分配 |
| 死锁 | 互斥量嵌套 | 统一加锁顺序/超时 |
| 内存不足 | 堆太小 | 增大TOTAL_HEAP_SIZE |
| SysTick冲突 | Timebase用了SysTick | 改用TIM6/7 |
| 中断中用API | ISR不应用阻塞API | 用FromISR版本 |

## 8. 完整电赛工程模板

```c
// ===== 全局资源 =====
QueueHandle_t xQueueADC;
QueueHandle_t xQueueUART;
SemaphoreHandle_t xMutexDisplay;
EventHandle_t xSysEvents;

// ===== 任务定义 =====
void Task_ADC(void *p) {
    ADC_Data_t data;
    TickType_t lastWake = xTaskGetTickCount();
    for (;;) {
        data.ch0 = ADC_Read(0);
        data.ch1 = ADC_Read(1);
        data.ts  = HAL_GetTick();
        xQueueSend(xQueueADC, &data, 0);
        vTaskDelayUntil(&lastWake, pdMS_TO_TICKS(1));  // 1kHz
    }
}

void Task_PID(void *p) {
    ADC_Data_t data;
    for (;;) {
        if (xQueueReceive(xQueueADC, &data, pdMS_TO_TICKS(5)) == pdPASS) {
            float out = PID_Update(&pid, data.ch0);
            PWM_Set(out);
        }
    }
}

void Task_UART(void *p) {
    uint8_t buf[64];
    for (;;) {
        int n = UART_ReadLine(buf, sizeof(buf), 100);
        if (n > 0) {
            Parse_Command(buf);
        }
    }
}

void Task_OLED(void *p) {
    for (;;) {
        xSemaphoreTake(xMutexDisplay, portMAX_DELAY);
        OLED_Printf(0, 0, "PID: %.2f", pid.output);
        OLED_Printf(0, 2, "V: %.3f", global_voltage);
        OLED_Update();
        xSemaphoreGive(xMutexDisplay);
        vTaskDelay(pdMS_TO_TICKS(100));
    }
}

// ===== main =====
int main(void) {
    HAL_Init();
    SystemClock_Config();
    Peripherals_Init();
    
    xQueueADC     = xQueueCreate(16, sizeof(ADC_Data_t));
    xQueueUART    = xQueueCreate(4, 64);
    xMutexDisplay = xSemaphoreCreateMutex();
    xSysEvents    = xEventGroupCreate();
    
    xTaskCreate(Task_ADC,   "ADC",   128, NULL, 4, NULL);
    xTaskCreate(Task_PID,   "PID",   256, NULL, 5, NULL);
    xTaskCreate(Task_UART,  "UART",  256, NULL, 2, NULL);
    xTaskCreate(Task_OLED,  "OLED",  256, NULL, 1, NULL);
    
    vTaskStartScheduler();
    while (1);
}
```
