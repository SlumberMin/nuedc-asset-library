## 1. 混杂系统概述

混杂系统（Hybrid System）是同时包含连续动态和离散事件的系统。在电赛中，几乎所有实际控制问题都是混杂系统。

### 1.1 典型场景

| 场景 | 连续部分 | 离散部分 |
|------|---------|---------|
| 小车状态切换 | 电机速度控制 | 巡线/避障/寻迹模式切换 |
| 电机控制 | PWM连续调节 | 正转/反转/制动状态 |
| 温控系统 | PID温度调节 | 加热/保温/冷却模式 |
| 四旋翼 | 姿态PID | 起飞/悬停/降落/失控保护 |

### 1.2 混杂系统的一般模型

```
ẋ = f(x, u, q)     连续动态（q为当前离散状态）
q⁺ = g(x, q)        离散跳变（guard条件触发）
y = h(x, q)         输出
```

## 2. 有限状态机（FSM）

### 2.1 基础FSM实现

```c
typedef enum {
    STATE_IDLE = 0,
    STATE_RUNNING,
    STATE_SEARCHING,
    STATE_AVOIDING,
    STATE_STOPPED,
    STATE_FAULT
} SystemState_t;

typedef enum {
    EVENT_NONE = 0,
    EVENT_START,
    EVENT_OBSTACLE_DETECTED,
    EVENT_OBSTACLE_CLEARED,
    EVENT_TARGET_FOUND,
    EVENT_TIMEOUT,
    EVENT_FAULT
} SystemEvent_t;

typedef struct {
    SystemState_t current_state;
    SystemState_t previous_state;
    uint32_t state_enter_time;
    uint32_t state_duration;
} FSM_t;

// 状态转移表
typedef struct {
    SystemState_t current;
    SystemEvent_t event;
    SystemState_t next;
} StateTransition_t;

static const StateTransition_t transitions[] = {
    {STATE_IDLE,      EVENT_START,              STATE_RUNNING},
    {STATE_RUNNING,   EVENT_OBSTACLE_DETECTED,  STATE_AVOIDING},
    {STATE_RUNNING,   EVENT_TARGET_FOUND,       STATE_STOPPED},
    {STATE_RUNNING,   EVENT_FAULT,              STATE_FAULT},
    {STATE_AVOIDING,  EVENT_OBSTACLE_CLEARED,   STATE_RUNNING},
    {STATE_SEARCHING, EVENT_TARGET_FOUND,       STATE_RUNNING},
    {STATE_SEARCHING, EVENT_TIMEOUT,            STATE_STOPPED},
    {STATE_FAULT,     EVENT_START,              STATE_IDLE},  // 复位
};

#define NUM_TRANSITIONS (sizeof(transitions) / sizeof(transitions[0]))

SystemState_t FSM_ProcessEvent(FSM_t *fsm, SystemEvent_t event) {
    for (int i = 0; i < NUM_TRANSITIONS; i++) {
        if (transitions[i].current == fsm->current_state &&
            transitions[i].event == event) {
            fsm->previous_state = fsm->current_state;
            fsm->current_state = transitions[i].next;
            fsm->state_enter_time = HAL_GetTick();
            return fsm->current_state;
        }
    }
    return fsm->current_state;  // 无匹配转移，保持当前状态
}

void FSM_Update(FSM_t *fsm) {
    fsm->state_duration = HAL_GetTick() - fsm->state_enter_time;
}
```

### 2.2 带动作的FSM

```c
typedef void (*StateAction_t)(void);

typedef struct {
    SystemState_t state;
    StateAction_t on_enter;      // 进入状态时执行
    StateAction_t on_run;        // 状态中持续执行
    StateAction_t on_exit;       // 离开状态时执行
} StateConfig_t;

void State_Idle_Enter(void)    { /* 初始化 */ }
void State_Idle_Run(void)      { /* 等待启动指令 */ }
void State_Idle_Exit(void)     { /* 清理 */ }

void State_Running_Enter(void) { /* 启动电机 */ }
void State_Running_Run(void)   { /* PID控制 */ }
void State_Running_Exit(void)  { /* 减速 */ }

void State_Avoiding_Enter(void) { /* 记录避障方向 */ }
void State_Avoiding_Run(void)   { /* 避障算法 */ }
void State_Avoiding_Exit(void)  { /* 恢复航向 */ }

static const StateConfig_t state_configs[] = {
    {STATE_IDLE,     State_Idle_Enter,     State_Idle_Run,     State_Idle_Exit},
    {STATE_RUNNING,  State_Running_Enter,  State_Running_Run,  State_Running_Exit},
    {STATE_AVOIDING, State_Avoiding_Enter, State_Avoiding_Run, State_Avoiding_Exit},
};

void FSM_RunWithActions(FSM_t *fsm, SystemEvent_t event) {
    SystemState_t old_state = fsm->current_state;
    SystemState_t new_state = FSM_ProcessEvent(fsm, event);
    
    if (new_state != old_state) {
        // 执行旧状态的退出动作
        for (int i = 0; i < sizeof(state_configs)/sizeof(state_configs[0]); i++) {
            if (state_configs[i].state == old_state && state_configs[i].on_exit)
                state_configs[i].on_exit();
        }
        // 执行新状态的进入动作
        for (int i = 0; i < sizeof(state_configs)/sizeof(state_configs[0]); i++) {
            if (state_configs[i].state == new_state && state_configs[i].on_enter)
                state_configs[i].on_enter();
        }
    }
    
    // 执行当前状态的运行动作
    for (int i = 0; i < sizeof(state_configs)/sizeof(state_configs[0]); i++) {
        if (state_configs[i].state == new_state && state_configs[i].on_run)
            state_configs[i].on_run();
    }
}
```

## 3. 切换系统控制

### 3.1 切换线性系统

```
ẋ = Aᵢx + Bᵢu,  i ∈ {1, 2, ..., N}  （i为当前模式）

切换规则 σ(t) → i：决定当前使用哪个子系统
```

### 3.2 多模型自适应控制（MMAC）

```c
// 多模型并行运行，根据性能选择最优模型
#define NUM_MODELS 3

typedef struct {
    float A, B;          // 模型参数
    float state;         // 模型状态
    float cost;          // 累积代价
    float output;        // 模型预测输出
    float weight;        // 混合权重
} ModelCandidate_t;

typedef struct {
    ModelCandidate_t models[NUM_MODELS];
    float learning_rate;
} MultiModelController_t;

void MMAC_Init(MultiModelController_t *mmc) {
    // 初始化多个候选模型（覆盖不同工况）
    mmc->models[0].A = 0.9f; mmc->models[0].B = 0.1f;  // 低速模型
    mmc->models[1].A = 0.8f; mmc->models[1].B = 0.2f;  // 中速模型
    mmc->models[2].A = 0.7f; mmc->models[2].B = 0.3f;  // 高速模型
    mmc->learning_rate = 0.01f;
    
    for (int i = 0; i < NUM_MODELS; i++) {
        mmc->models[i].cost = 0.0f;
        mmc->models[i].weight = 1.0f / NUM_MODELS;
    }
}

float MMAC_Update(MultiModelController_t *mmc, float measurement, float control) {
    float total_weight = 0.0f;
    float weighted_output = 0.0f;
    
    for (int i = 0; i < NUM_MODELS; i++) {
        ModelCandidate_t *m = &mmc->models[i];
        
        // 各模型独立预测
        m->output = m->A * m->state + m->B * control;
        
        // 计算预测误差
        float error = measurement - m->output;
        
        // 更新代价（指数加权）
        m->cost = 0.9f * m->cost + error * error;
        
        // 更新权重（代价越小权重越大）
        m->weight = expf(-5.0f * m->cost);
        
        total_weight += m->weight;
        weighted_output += m->weight * m->output;
        
        // 更新模型状态
        m->state = measurement;
    }
    
    // 归一化权重
    for (int i = 0; i < NUM_MODELS; i++) {
        mmc->models[i].weight /= total_weight;
    }
    
    return weighted_output / total_weight;
}
```

### 3.3 基于规则的切换控制

```c
// 根据工作区间切换不同控制器参数
typedef struct {
    float range_min, range_max;
    float Kp, Ki, Kd;
} ControllerRegion_t;

typedef struct {
    ControllerRegion_t regions[4];
    int num_regions;
} GainSchedulingController_t;

float GainScheduling_Calc(GainSchedulingController_t *gs, float error, float d_error, float i_error, float input) {
    for (int i = 0; i < gs->num_regions; i++) {
        if (input >= gs->regions[i].range_min && input < gs->regions[i].range_max) {
            return gs->regions[i].Kp * error + 
                   gs->regions[i].Ki * i_error + 
                   gs->regions[i].Kd * d_error;
        }
    }
    // 默认使用最后一个区域
    int last = gs->num_regions - 1;
    return gs->regions[last].Kp * error + 
           gs->regions[last].Ki * i_error + 
           gs->regions[last].Kd * d_error;
}
```

## 4. 混合自动机

### 4.1 混合自动机形式化

```
H = (Q, X, f, Init, D, G, R)

Q: 离散状态集合
X: 连续状态空间
f: 连续向量场 ẋ = f(q, x, u)
Init: 初始条件
D: 不变量（每个状态的合法区域）
G: 守卫条件（触发离散跳变）
R: 重置映射（跳变时状态重置）
```

### 4.2 混合自动机实现模板

```c
typedef struct {
    int discrete_state;
    float continuous_state;
    float (*flow)(int q, float x, float u);       // 连续动态
    bool (*guard)(int q, float x);                  // 守卫条件
    float (*reset)(int q_from, int q_to, float x); // 重置映射
    int (*transition)(int q, float x);              // 离散跳变
} HybridAutomaton_t;

float HybridAutomaton_Step(HybridAutomaton_t *ha, float u, float dt) {
    // 1. 检查守卫条件
    int new_q = ha->transition(ha->discrete_state, ha->continuous_state);
    
    // 2. 离散跳变
    if (new_q != ha->discrete_state) {
        ha->continuous_state = ha->reset(ha->discrete_state, new_q, ha->continuous_state);
        ha->discrete_state = new_q;
    }
    
    // 3. 连续演化（欧拉法）
    float dx = ha->flow(ha->discrete_state, ha->continuous_state, u);
    ha->continuous_state += dx * dt;
    
    return ha->continuous_state;
}
```

### 5.2 电源管理混杂系统

```c
typedef enum {
    POWER_OFF,
    POWER_STANDBY,
    POWER_NORMAL,
    POWER_LOW_BATTERY,
    POWER_CRITICAL
} PowerState_t;

PowerState_t PowerFSM_Update(PowerState_t current, float battery_voltage, bool power_btn) {
    switch (current) {
        case POWER_OFF:
            return power_btn ? POWER_STANDBY : POWER_OFF;
            
        case POWER_STANDBY:
            if (power_btn) return POWER_NORMAL;
            return POWER_STANDBY;
            
        case POWER_NORMAL:
            if (battery_voltage < 3.3f) return POWER_LOW_BATTERY;
            if (battery_voltage < 3.0f) return POWER_CRITICAL;
            return POWER_NORMAL;
            
        case POWER_LOW_BATTERY:
            if (battery_voltage < 3.0f) return POWER_CRITICAL;
            if (battery_voltage > 3.5f) return POWER_NORMAL;  // 恢复
            return POWER_LOW_BATTERY;
            
        case POWER_CRITICAL:
            return POWER_OFF;  // 立即关机保护
            
        default:
            return POWER_OFF;
    }
}
```

### 5.3 多模式小车控制器

```c
// 三种驾驶模式的连续控制器 + 模式切换逻辑
typedef enum {
    MODE_PRECISE,   // 精确模式：低速高增益
    MODE_NORMAL,    // 正常模式：标准PID
    MODE_FAST       // 快速模式：高速低增益
} DriveMode_t;

typedef struct {
    float Kp, Ki, Kd;
    float max_speed;
    float deadzone;
} ModeParams_t;

static const ModeParams_t mode_params[] = {
    {2.0f, 0.5f, 0.3f, 0.3f, 0.005f},  // PRECISE
    {1.5f, 0.3f, 0.2f, 0.6f, 0.010f},  // NORMAL
    {1.0f, 0.1f, 0.5f, 1.0f, 0.020f},  // FAST
};

float MultiMode_Control(DriveMode_t mode, float error, float d_error, float i_error) {
    const ModeParams_t *p = &mode_params[mode];
    float output = p->Kp * error + p->Ki * i_error + p->Kd * d_error;
    
    // 死区处理
    if (fabsf(error) < p->deadzone) output = 0.0f;
    
    // 限幅
    if (output > p->max_speed) output = p->max_speed;
    if (output < -p->max_speed) output = -p->max_speed;
    
    return output;
}
```

## 6. 设计要点

| 要点 | 说明 |
|------|------|
| 状态完备性 | 每个状态都要有明确的进入/退出条件 |
| 无死锁 | 确保不存在无法离开的状态 |
| 无活锁 | 避免状态快速来回切换（加滞环） |
| 超时保护 | 每个状态设置最大停留时间 |
| 默认安全 | 未定义转移默认进入安全状态 |
| 滞环设计 | 切换条件加滞环，避免抖动 |
