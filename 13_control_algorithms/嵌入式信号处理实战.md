## 1. FIR滤波器

### 1.1 基本FIR实现

```c
#define FIR_ORDER 32

typedef struct {
    float coeff[FIR_ORDER + 1];  // 滤波器系数
    float buffer[FIR_ORDER + 1]; // 延迟缓冲
    uint8_t index;               // 环形索引
} FIR_Filter;

void FIR_Init(FIR_Filter *fir, const float *coeffs) {
    for (int i = 0; i <= FIR_ORDER; i++) {
        fir->coeff[i] = coeffs[i];
        fir->buffer[i] = 0.0f;
    }
    fir->index = 0;
}

float FIR_Process(FIR_Filter *fir, float input) {
    fir->buffer[fir->index] = input;
    float output = 0.0f;
    uint8_t idx = fir->index;
    
    for (int i = 0; i <= FIR_ORDER; i++) {
        output += fir->coeff[i] * fir->buffer[idx];
        idx = (idx == 0) ? FIR_ORDER : idx - 1;
    }
    
    fir->index = (fir->index + 1) % (FIR_ORDER + 1);
    return output;
}
```

### 1.2 低通FIR系数生成（窗函数法）

```c
// 生成低通FIR系数 (截止频率fc, 采样率fs, 阶数N)
void FIR_GenLowPass(float *coeff, int N, float fc, float fs) {
    float wc = 2.0f * M_PI * fc / fs;
    int M = N / 2;
    
    for (int n = 0; n <= N; n++) {
        if (n == M)
            coeff[n] = wc / M_PI;
        else
            coeff[n] = sinf(wc * (n - M)) / (M_PI * (n - M));
        // 汉宁窗
        coeff[n] *= 0.5f * (1.0f - cosf(2.0f * M_PI * n / N));
    }
}
```

### 1.3 带通FIR系数

```c
void FIR_GenBandPass(float *coeff, int N, float f_low, float f_high, float fs) {
    float wl = 2.0f * M_PI * f_low / fs;
    float wh = 2.0f * M_PI * f_high / fs;
    int M = N / 2;
    
    for (int n = 0; n <= N; n++) {
        if (n == M)
            coeff[n] = (wh - wl) / M_PI;
        else
            coeff[n] = (sinf(wh * (n - M)) - sinf(wl * (n - M))) / (M_PI * (n - M));
        coeff[n] *= 0.54f - 0.46f * cosf(2.0f * M_PI * n / N); // Hamming窗
    }
}
```

## 2. IIR滤波器

### 2.1 二阶IIR（Biquad）实现

```c
typedef struct {
    float b0, b1, b2;  // 分子系数
    float a1, a2;       // 分母系数（a0归一化为1）
    float x1, x2;       // 输入历史
    float y1, y2;       // 输出历史
} IIR_Biquad;

void IIR_Init(IIR_Biquad *f, float b0, float b1, float b2, float a1, float a2) {
    f->b0 = b0; f->b1 = b1; f->b2 = b2;
    f->a1 = a1; f->a2 = a2;
    f->x1 = f->x2 = f->y1 = f->y2 = 0.0f;
}

float IIR_Process(IIR_Biquad *f, float input) {
    float output = f->b0 * input + f->b1 * f->x1 + f->b2 * f->x2
                   - f->a1 * f->y1 - f->a2 * f->y2;
    f->x2 = f->x1; f->x1 = input;
    f->y2 = f->y1; f->y1 = output;
    return output;
}
```

### 2.2 级联IIR（高阶滤波器）

```c
#define IIR_CASCADE_MAX 8

typedef struct {
    IIR_Biquad stages[IIR_CASCADE_MAX];
    int num_stages;
} IIR_Cascade;

float IIR_Cascade_Process(IIR_Cascade *c, float input) {
    float x = input;
    for (int i = 0; i < c->num_stages; i++) {
```

### 2.3 双线性变换法求IIR系数

```c
// 二阶Butterworth低通 → IIR系数
void IIR_DesignButterworthLP(float fc, float fs, 
                              float *b0, float *b1, float *b2,
                              float *a1, float *a2) {
    float wc = tanf(M_PI * fc / fs);
    float wc2 = wc * wc;
    float sqrt2 = 1.41421356f;
    
    float K = 1.0f / (1.0f + sqrt2 * wc + wc2);
    
    *b0 = wc2 * K;
    *b1 = 2.0f * wc2 * K;
    *b2 = wc2 * K;
    *a1 = 2.0f * (wc2 - 1.0f) * K;
    *a2 = (1.0f - sqrt2 * wc + wc2) * K;
}
```

## 3. FFT（快速傅里叶变换）

### 3.1 基2-FFT（蝶形运算）

```c
#include <math.h>

#define FFT_N 256  // 必须为2的幂

typedef struct {
    float re[FFT_N];
    float im[FFT_N];
} FFT_Data;

// 位反转置换
static void bit_reverse(float *re, float *im, int n) {
    int j = 0;
    for (int i = 0; i < n - 1; i++) {
        if (i < j) {
            float tmp;
            tmp = re[i]; re[i] = re[j]; re[j] = tmp;
            tmp = im[i]; im[i] = im[j]; im[j] = tmp;
        }
        int k = n >> 1;
        while (k <= j) { j -= k; k >>= 1; }
        j += k;
    }
}

void FFT_Compute(FFT_Data *fft, int n) {
    bit_reverse(fft->re, fft->im, n);
    
    for (int len = 2; len <= n; len <<= 1) {
        float angle = -2.0f * M_PI / len;
        float wr = cosf(angle), wi = sinf(angle);
        
        for (int i = 0; i < n; i += len) {
            float tr = 1.0f, ti = 0.0f;
            for (int j = 0; j < len / 2; j++) {
                int u = i + j;
                int v = i + j + len / 2;
                
                float p_re = tr * fft->re[v] - ti * fft->im[v];
                float p_im = tr * fft->im[v] + ti * fft->re[v];
                
                fft->re[v] = fft->re[u] - p_re;
                fft->im[v] = fft->im[u] - p_im;
                fft->re[u] += p_re;
                fft->im[u] += p_im;
                
                float new_tr = tr * wr - ti * wi;
                ti = tr * wi + ti * wr;
                tr = new_tr;
            }
        }
    }
}

// 计算幅度谱
void FFT_Magnitude(FFT_Data *fft, float *mag, int n) {
    for (int i = 0; i < n / 2; i++) {
        mag[i] = sqrtf(fft->re[i] * fft->re[i] + fft->im[i] * fft->im[i]) * 2.0f / n;
    }
}
```

### 3.2 FFT求频率/相位/THD

```c
// 找基波频率和相位
void FFT_FindFundamental(float *mag, float *re, float *im, 
                          int n, float fs, 
                          float *freq, float *phase, float *amplitude) {
    int max_idx = 1;
    for (int i = 2; i < n / 2; i++) {
        if (mag[i] > mag[max_idx]) max_idx = i;
    }
    *freq = (float)max_idx * fs / n;
    *amplitude = mag[max_idx];
    *phase = atan2f(im[max_idx], re[max_idx]) * 180.0f / M_PI;
}

// 计算THD（总谐波失真）
float FFT_CalcTHD(float *mag, int fund_idx, int max_harmonic, int n) {
    float v1 = mag[fund_idx];
    float sum_sq = 0.0f;
    for (int h = 2; h <= max_harmonic; h++) {
        int idx = fund_idx * h;
        if (idx < n / 2) sum_sq += mag[idx] * mag[idx];
    }
    return sqrtf(sum_sq) / v1 * 100.0f;
}
```

## 4. 抽取（Decimation）

### 4.1 整数倍抽取

```c
typedef struct {
    FIR_Filter anti_alias;  // 抗混叠滤波器
    uint8_t decim_factor;   // 抽取因子
    uint8_t counter;        // 计数器
} Decimator;

void Decimator_Init(Decimator *d, uint8_t factor, const float *coeffs, int order) {
    FIR_Init(&d->anti_alias, coeffs);
    d->decim_factor = factor;
    d->counter = 0;
}

// 返回1表示有新输出
int Decimator_Process(Decimator *d, float input, float *output) {
    float filtered = FIR_Process(&d->anti_alias, input);
    d->counter++;
    if (d->counter >= d->decim_factor) {
        d->counter = 0;
        *output = filtered;
        return 1;
    }
    return 0;
}
```

### 4.2 CIC抽取滤波器（零开销）

```c
#define CIC_ORDER 3
#define CIC_DECIM 16

typedef struct {
    int32_t integrator[CIC_ORDER];
    int32_t comb[CIC_ORDER];
    int32_t comb_prev[CIC_ORDER];
    uint8_t decim_cnt;
} CIC_Filter;

int32_t CIC_Process(CIC_Filter *cic, int32_t input) {
    // 积分器
    cic->integrator[0] += input;
    for (int i = 1; i < CIC_ORDER; i++)
        cic->integrator[i] += cic->integrator[i-1];
    
    // 抽取 + 梳状
    cic->decim_cnt++;
    if (cic->decim_cnt >= CIC_DECIM) {
        cic->decim_cnt = 0;
        int32_t x = cic->integrator[CIC_ORDER - 1];
        for (int i = 0; i < CIC_ORDER; i++) {
            int32_t tmp = x;
            x -= cic->comb_prev[i];
            cic->comb_prev[i] = tmp;
        }
        return x;
    }
    return 0; // 无新输出
}
```

### 5.2 三次插值（Catmull-Rom）

```c
float cubic_interp(float y0, float y1, float y2, float y3, float frac) {
    float a = -0.5f * y0 + 1.5f * y1 - 1.5f * y2 + 0.5f * y3;
    float b = y0 - 2.5f * y1 + 2.0f * y2 - 0.5f * y3;
    float c = -0.5f * y0 + 0.5f * y2;
    float d = y1;
    return ((a * frac + b) * frac + c) * frac + d;
}
```

## 6. 正交解调（I/Q解调）

### 6.1 同步正交解调

```c
typedef struct {
    float lo_phase;       // 本振相位
    float lo_freq;        // 本振频率（归一化，cycles/sample）
    IIR_Biquad lpf_i;     // I路低通
    IIR_Biquad lpf_q;     // Q路低通
} QuadDemod;

void QuadDemod_Init(QuadDemod *qd, float freq, float lpf_cutoff, float fs) {
    qd->lo_phase = 0.0f;
    qd->lo_freq = freq / fs;
    
    float b0, b1, b2, a1, a2;
    IIR_DesignButterworthLP(lpf_cutoff, fs, &b0, &b1, &b2, &a1, &a2);
    IIR_Init(&qd->lpf_i, b0, b1, b2, a1, a2);
    IIR_Init(&qd->lpf_q, b0, b1, b2, a1, a2);
}

void QuadDemod_Process(QuadDemod *qd, float input, float *out_i, float *out_q) {
    float lo_i = cosf(2.0f * M_PI * qd->lo_phase);
    float lo_q = sinf(2.0f * M_PI * qd->lo_phase);
    
    qd->lo_phase += qd->lo_freq;
    if (qd->lo_phase >= 1.0f) qd->lo_phase -= 1.0f;
    
    float mixed_i = input * lo_i;
    float mixed_q = input * lo_q;
    
    *out_i = IIR_Process(&qd->lpf_i, mixed_i);
    *out_q = IIR_Process(&qd->lpf_q, mixed_q);
}

// 计算瞬时幅度和频率
void QuadDemod_GetAMFM(float I, float Q, float fs, 
                        float *amplitude, float *instant_freq) {
    static float prev_phase = 0.0f;
    *amplitude = sqrtf(I * I + Q * Q);
    float phase = atan2f(Q, I);
    float dphase = phase - prev_phase;
    // 相位展开
    if (dphase > M_PI) dphase -= 2.0f * M_PI;
    if (dphase < -M_PI) dphase += 2.0f * M_PI;
    *instant_freq = dphase * fs / (2.0f * M_PI);
    prev_phase = phase;
}
```

### 6.2 解调应用：AM/FM解调

```c
// AM解调 - 包络检波
float AM_Demodulate(QuadDemod *qd, float input) {
    float I, Q;
    QuadDemod_Process(qd, input, &I, &Q);
    return sqrtf(I * I + Q * Q);
}

// FM解调 - 瞬时频率
float FM_Demodulate(QuadDemod *qd, float input, float fs) {
    float I, Q;
    QuadDemod_Process(qd, input, &I, &Q);
    float amp, freq;
    QuadDemod_GetAMFM(I, Q, fs, &amp, &freq);
    return freq;
}
```

## 7. 实用信号处理工具函数

### 7.1 滑动平均滤波器

```c
#define MA_LEN 16

typedef struct {
    float buf[MA_LEN];
    uint8_t idx;
    float sum;
} MovingAvg;

float MA_Process(MovingAvg *ma, float input) {
    ma->sum -= ma->buf[ma->idx];
    ma->buf[ma->idx] = input;
    ma->sum += input;
    ma->idx = (ma->idx + 1) % MA_LEN;
    return ma->sum / MA_LEN;
}
```

### 7.2 中值滤波器

```c
#define MED_LEN 9

float MedianFilter(float *buf, int len) {
    // 简单排序找中值
    float sorted[MED_LEN];
    for (int i = 0; i < len; i++) sorted[i] = buf[i];
    for (int i = 0; i < len - 1; i++)
        for (int j = i + 1; j < len; j++)
            if (sorted[i] > sorted[j]) {
                float t = sorted[i]; sorted[i] = sorted[j]; sorted[j] = t;
            }
    return sorted[len / 2];
}
```

### 7.3 陷波滤波器（50Hz工频陷波）

```c
// 50Hz陷波，Q=30
void IIR_DesignNotch50Hz(float fs, float *b0, float *b1, float *b2,
                          float *a1, float *a2) {
    float f0 = 50.0f;
    float w0 = 2.0f * M_PI * f0 / fs;
    float Q = 30.0f;
    float alpha = sinf(w0) / (2.0f * Q);
    
    float a0_inv = 1.0f / (1.0f + alpha);
    *b0 = 1.0f * a0_inv;
    *b1 = -2.0f * cosf(w0) * a0_inv;
    *b2 = 1.0f * a0_inv;
    *a1 = -2.0f * cosf(w0) * a0_inv;
    *a2 = (1.0f - alpha) * a0_inv;
}
```

## 8. 窗函数库

```c
typedef enum { WIN_RECT, WIN_HANN, WIN_HAMMING, WIN_BLACKMAN, WIN_KAISER } WinType;

void Window_Generate(float *win, int N, WinType type, float beta) {
    for (int n = 0; n < N; n++) {
        switch (type) {
        case WIN_RECT:    win[n] = 1.0f; break;
        case WIN_HANN:    win[n] = 0.5f * (1.0f - cosf(2.0f * M_PI * n / (N - 1))); break;
        case WIN_HAMMING: win[n] = 0.54f - 0.46f * cosf(2.0f * M_PI * n / (N - 1)); break;
        case WIN_BLACKMAN:
            win[n] = 0.42f - 0.5f * cosf(2.0f * M_PI * n / (N - 1))
                     + 0.08f * cosf(4.0f * M_PI * n / (N - 1));
            break;
        case WIN_KAISER:
            // 近似Kaiser窗
            {
                float x = 2.0f * n / (N - 1) - 1.0f;
                win[n] = coshf(beta * sqrtf(1.0f - x * x)) / coshf(beta);
            }
            break;
        }
    }
}
```

### 9.1 定点FIR（Q15）

```c
int16_t FIR_Process_Q15(int16_t *coeff, int16_t *buf, int16_t input, int order, uint8_t *idx) {
    buf[*idx] = input;
    int32_t acc = 0;
    uint8_t p = *idx;
    for (int i = 0; i <= order; i++) {
        acc += (int32_t)coeff[i] * buf[p];
        p = (p == 0) ? order : p - 1;
    }
    *idx = (*idx + 1) % (order + 1);
    return (int16_t)(acc >> 15);
}
```
