## 1. FOC（磁场定向控制）总览

```
┌─────────────────────────────────────────────────────┐
│                    FOC 控制框架                       │
│                                                     │
│  速度给定 → [速度环PI] → iq给定 ──┐                 │
│                                    ↓                 │
│  ┌──────────────────────────────────────────┐       │
│  │         电流环 (dq解耦)                    │       │
│  │  iq给定 → [PI] → Vq ─┐                  │       │
│  │  id给定=0 → [PI] → Vd ┘→ Park逆变换      │       │
│  │                          ↓                │       │
│  │                    Vα,Vβ → SVPWM → 6路PWM │       │
│  └──────────────────────────────────────────┘       │
│                                                     │
│  三相电流 → Clarke → Park → id,iq (反馈)            │
│  编码器/观测器 → θ (电角度)                           │
└─────────────────────────────────────────────────────┘
```

## 2. 坐标变换

### 2.1 Clarke变换（abc → αβ）

```c
// 输入: ia, ib, ic (三相电流)
// 输出: alpha, beta
void Clarke_Transform(float ia, float ib, float ic, float *alpha, float *beta) {
    *alpha = ia;
    *beta = (ia + 2.0f * ib) / 1.7320508f;  // 1/√3
}

// 简化版（只用两相，ia+ib+ic=0）
void Clarke_2Phase(float ia, float ib, float *alpha, float *beta) {
    *alpha = ia;
    *beta = (ia + 2.0f * ib) / 1.7320508f;
}
```

### 2.2 Park变换（αβ → dq）

```c
void Park_Transform(float alpha, float beta, float theta,
                     float *d, float *q) {
    float cos_t = cosf(theta);
    float sin_t = sinf(theta);
    *d = alpha * cos_t + beta * sin_t;
    *q = -alpha * sin_t + beta * cos_t;
}

void Park_InverseTransform(float d, float q, float theta,
                            float *alpha, float *beta) {
    float cos_t = cosf(theta);
    float sin_t = sinf(theta);
    *alpha = d * cos_t - q * sin_t;
    *beta = d * sin_t + q * cos_t;
}
```

## 3. SVPWM（空间矢量PWM）

### 3.1 基本实现

```c
#define SQRT3 1.7320508f
#define TWO_THIRDS 0.6666667f

// 输入: Valpha, Vbeta, Vdc
// 输出: Ta, Tb, Tc (三相占空比 0~1)
void SVPWM_Calc(float Valpha, float Vbeta, float Vdc,
                 float *Ta, float *Tb, float *Tc) {
    // 1. 确定扇区
    float V1 = Vbeta;
    float V2 = SQRT3 / 2.0f * Valpha - Vbeta / 2.0f;
    float V3 = -SQRT3 / 2.0f * Valpha - Vbeta / 2.0f;
    
    int sector = 0;
    if (V1 > 0) sector += 1;
    if (V2 > 0) sector += 2;
    if (V3 > 0) sector += 4;
    
    // 2. 计算基本矢量作用时间
    float Ts = 1.0f;  // PWM周期归一化
    float T1, T2;
    
    switch (sector) {
    case 3:  // 扇区I
        T1 = (SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha - Vbeta / 2.0f);
        T2 = (SQRT3 * Ts / Vdc) * Vbeta;
        break;
    case 1:  // 扇区II
        T1 = (SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha + Vbeta / 2.0f);
        T2 = -(SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha - Vbeta / 2.0f);
        break;
    case 5:  // 扇区III
        T1 = (SQRT3 * Ts / Vdc) * Vbeta;
        T2 = -(SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha + Vbeta / 2.0f);
        break;
    case 4:  // 扇区IV
        T1 = -(SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha - Vbeta / 2.0f);
        T2 = -(SQRT3 * Ts / Vdc) * Vbeta;
        break;
    case 6:  // 扇区V
        T1 = -(SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha + Vbeta / 2.0f);
        T2 = (SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha - Vbeta / 2.0f);
        break;
    case 2:  // 扇区VI
        T1 = -(SQRT3 * Ts / Vdc) * Vbeta;
        T2 = (SQRT3 * Ts / Vdc) * (SQRT3 / 2.0f * Valpha + Vbeta / 2.0f);
        break;
    default:
        T1 = T2 = 0;
        break;
    }
    
    // 3. 过调制处理
    if (T1 + T2 > Ts) {
        float ratio = Ts / (T1 + T2);
        T1 *= ratio;
        T2 *= ratio;
    }
    
    // 4. 七段式对称PWM分配
    float T0 = Ts - T1 - T2;
    float Tcm1, Tcm2, Tcm3;
    
    switch (sector) {
    case 3: Tcm1 = T0/4;      Tcm2 = T0/4 + T1/2;  Tcm3 = T0/4 + T1/2 + T2/2; break;
    case 1: Tcm1 = T0/4 + T2/2; Tcm2 = T0/4;         Tcm3 = T0/4 + T1/2 + T2/2; break;
    case 5: Tcm1 = T0/4 + T1/2 + T2/2; Tcm2 = T0/4;   Tcm3 = T0/4 + T1/2; break;
    case 4: Tcm1 = T0/4 + T1/2 + T2/2; Tcm2 = T0/4 + T2/2; Tcm3 = T0/4; break;
    case 6: Tcm1 = T0/4 + T1/2; Tcm2 = T0/4 + T1/2 + T2/2; Tcm3 = T0/4 + T2/2; break;
    case 2: Tcm1 = T0/4 + T1/2; Tcm2 = T0/4 + T1/2 + T2/2; Tcm3 = T0/4; break;
    default: Tcm1 = Tcm2 = Tcm3 = 0.5f; break;
    }
    
    *Ta = Tcm1 / Ts;
    *Tb = Tcm2 / Ts;
    *Tc = Tcm3 / Ts;
}
```

### 3.2 SVPWM的母线电压利用率

| 调制方式 | 最大输出线电压 | 相比SPWM |
|----------|--------------|----------|
| SPWM | 0.866 × Vdc | 100% |
| SVPWM | 1.0 × Vdc | 115.5% |

## 4. 电流环PI控制器

### 4.1 带解耦的电流环

```c
typedef struct {
    float Kp, Ki;
    float integral;
    float out_min, out_max;
    float dt;
} PI_Controller;

float PI_Update(PI_Controller *pi, float ref, float fbk) {
    float err = ref - fbk;
    pi->integral += err * pi->dt;
    
    // 积分限幅
    float int_max = pi->out_max / pi->Ki;
    if (pi->integral > int_max) pi->integral = int_max;
    if (pi->integral < -int_max) pi->integral = -int_max;
    
    float out = pi->Kp * err + pi->Ki * pi->integral;
    
    // 输出限幅
    if (out > pi->out_max) out = pi->out_max;
    if (out < pi->out_min) out = pi->out_min;
    
    return out;
}

typedef struct {
    PI_Controller pi_d;
    PI_Controller pi_q;
    float Ld, Lq;       // dq轴电感
    float flux;          // 永磁体磁链
    float we;            // 电角速度
} CurrentLoop;

void CurrentLoop_Update(CurrentLoop *cl, 
                         float id_ref, float iq_ref,
                         float id_fbk, float iq_fbk,
                         float *Vd, float *Vq) {
    // PI控制
    float Vd_pi = PI_Update(&cl->pi_d, id_ref, id_fbk);
    float Vq_pi = PI_Update(&cl->pi_q, iq_ref, iq_fbk);
    
    // 前馈解耦
    *Vd = Vd_pi - cl->we * cl->Lq * iq_fbk;
    *Vq = Vq_pi + cl->we * cl->Ld * id_fbk + cl->we * cl->flux;
}
```

### 4.2 电流环PI参数整定

```
带宽设计法（适用于表贴式电机 PMSM）：

Kp = Ls × ωc
Ki = Rs × ωc

其中：
  Ls = 定子电感 (H)
  Rs = 定子电阻 (Ω)
  ωc = 期望电流环带宽 (rad/s)，通常取 PWM频率的 1/10~1/5

示例（Ls=1mH, Rs=0.5Ω, PWM=20kHz, ωc=2000π rad/s）：
  Kp = 0.001 × 6283 = 6.28
  Ki = 0.5 × 6283 = 3142
```

## 5. 无感控制（无位置传感器）

### 5.1 反电动势观测器（滑模观测器）

```c
typedef struct {
    float alpha_est, beta_est;  // 估计的反电动势
    float x_alpha, x_beta;     // 观测器状态
    float Kslide;               // 滑模增益
    float Kfilter;              // 低通滤波系数
    float Ls;                   // 电感
    float Rs;                   // 电阻
    float Ts;                   // 采样周期
    float theta_est;            // 估计电角度
} SMO_Observer;

void SMO_Init(SMO_Observer *smo, float Rs, float Ls, float Ts, float Kslide) {
    smo->Rs = Rs;
    smo->Ls = Ls;
    smo->Ts = Ts;
    smo->Kslide = Kslide;
    smo->Kfilter = Ts / (Ts + 0.001f);  // 低通截止
    smo->alpha_est = smo->beta_est = 0;
    smo->x_alpha = smo->x_beta = 0;
    smo->theta_est = 0;
}

void SMO_Update(SMO_Observer *smo,
                float ialpha, float ibeta,
                float valpha, float vbeta) {
    // 电流观测
    float ialpha_hat = smo->x_alpha + smo->Ts / smo->Ls * 
                       (valpha - smo->Rs * smo->x_alpha);
    float ibeta_hat = smo->x_beta + smo->Ts / smo->Ls * 
                      (vbeta - smo->Rs * smo->x_beta);
    
    // 滑模切换
    float z_alpha = smo->Kslide * (ialpha_hat - ialpha > 0 ? 1.0f : -1.0f);
    float z_beta = smo->Kslide * (ibeta_hat - ibeta > 0 ? 1.0f : -1.0f);
    
    // 更新状态
    smo->x_alpha += smo->Ts * (valpha - smo->Rs * ialpha - z_alpha) / smo->Ls;
    smo->x_beta += smo->Ts * (vbeta - smo->Rs * ibeta - z_beta) / smo->Ls;
    
    // 低通滤波提取反电动势
    smo->alpha_est += smo->Kfilter * (z_alpha - smo->alpha_est);
    smo->beta_est += smo->Kfilter * (z_beta - smo->beta_est);
    
    // 计算电角度
    smo->theta_est = atan2f(-smo->alpha_est, smo->beta_est);
}
```

### 5.2 锁相环（PLL）提取角度

```c
typedef struct {
    float Kp_pll, Ki_pll;
    float integral;
    float omega_est;    // 估计角速度
    float theta_est;    // 估计角度
    float Ts;
} PLL_Estimator;

void PLL_Init(PLL_Estimator *pll, float Kp, float Ki, float Ts) {
    pll->Kp_pll = Kp;
    pll->Ki_pll = Ki;
    pll->integral = 0;
    pll->omega_est = 0;
    pll->theta_est = 0;
    pll->Ts = Ts;
}

float PLL_Update(PLL_Estimator *pll, float e_alpha, float e_beta) {
    // 误差信号
    float err = -e_alpha * cosf(pll->theta_est) - e_beta * sinf(pll->theta_est);
    
    // PI环
    pll->integral += pll->Ki_pll * err * pll->Ts;
    pll->omega_est = pll->Kp_pll * err + pll->integral;
    
    // 积分得角度
    pll->theta_est += pll->omega_est * pll->Ts;
    if (pll->theta_est > M_PI) pll->theta_est -= 2.0f * M_PI;
    if (pll->theta_est < -M_PI) pll->theta_est += 2.0f * M_PI;
    
    return pll->theta_est;
}
```

### 5.3 高频注入法（低速/零速）

```c
typedef struct {
    float Vh;           // 注入电压幅值
    float fh;           // 注入频率 (Hz)
    float phase;        // 注入相位
    IIR_Biquad bpf;     // 带通滤波器
    IIR_Biquad lpf;     // 低通滤波器
    float Ts;
    float theta_est;
} HFI_Controller;

void HFI_Init(HFI_Controller *hfi, float Vh, float fh, float fs) {
    hfi->Vh = Vh;
    hfi->fh = fh;
    hfi->Ts = 1.0f / fs;
    hfi->phase = 0;
}

float HFI_Process(HFI_Controller *hfi, float id_hf, float iq_hf) {
    // 解调
    float cos_h = cosf(hfi->phase);
    float sin_h = sinf(hfi->phase);
    
    float demod_d = id_hf * cos_h;
    float demod_q = iq_hf * sin_h;
    
    // 低通滤波后得到位置误差信号
    float err_d = IIR_Process(&hfi->lpf, demod_d);
    
    // PLL提取角度
    // ...（同PLL_Update）
    
    hfi->phase += 2.0f * M_PI * hfi->fh * hfi->Ts;
    if (hfi->phase > 2.0f * M_PI) hfi->phase -= 2.0f * M_PI;
    
    return hfi->theta_est;
}
```

## 6. 弱磁控制

### 6.1 弱磁策略

```
              转速
               ↑
    弱磁区      │      ╱ 最大电压限制
               │    ╱
               │  ╱
    MTPA区     │╱
               └────────────→ 转矩
               
弱磁条件: Vdc已用完，id给负值削弱磁场
```

```c
typedef struct {
    float id_max;       // id最大允许值（负值）
    float Vdc;          // 母线电压
    float Vmax;         // 最大电压矢量幅值 = Vdc/√3
    float Rs, Ld, Lq;
    float flux;
    float we_max;       // 弱磁转折速度
} FieldWeakening;

void FW_Init(FieldWeakening *fw, float Vdc, float Rs, 
             float Ld, float Lq, float flux) {
    fw->Vdc = Vdc;
    fw->Vmax = Vdc / 1.7320508f * 0.9f;  // 留10%余量
    fw->Rs = Rs;
    fw->Ld = Ld;
    fw->Lq = Lq;
    fw->flux = flux;
    fw->id_max = -3.0f * flux / (2.0f * Ld);  // 电流极限圆与MTPA交点
}

// 弱磁时计算id给定
float FW_GetIdRef(FieldWeakening *fw, float iq_ref, float we) {
    // 电压极限圆: (Rs*id - we*Lq*iq)² + (Rs*iq + we*Ld*id + we*flux)² ≤ Vmax²
    // 解出id
    
    float a = fw->Rs * fw->Rs + (fw->we * fw->Ld) * (fw->we * fw->Ld);
    float b = 2.0f * fw->we * fw->Ld * (fw->we * fw->flux + fw->Rs * iq_ref);
    float c = (fw->Rs * iq_ref + fw->we * fw->flux) * (fw->Rs * iq_ref + fw->we * fw->flux) +
              (fw->we * fw->Lq * iq_ref) * (fw->we * fw->Lq * iq_ref) - 
              fw->Vmax * fw->Vmax;
    
    float discriminant = b * b - 4.0f * a * c;
    
    if (discriminant < 0) {
        return fw->id_max;  // 无法满足，输出最大弱磁
    }
    
    // 取较小的根（更负的id）
    float id1 = (-b - sqrtf(discriminant)) / (2.0f * a);
    float id_ref = (id1 < fw->id_max) ? fw->id_max : id1;
    
    return id_ref;
}
```

### 6.2 MTPA（最大转矩电流比）

```c
// 表贴式PMSM: MTPA即id=0
// 内置式PMSM(IPM):
float MTPA_GetIdRef(float iq_ref, float Ld, float Lq, float flux) {
    // MTPA条件: id = flux/(2*(Ld-Lq)) - sqrt(flux²/(4*(Ld-Lq)²) + iq²)
    float dL = Ld - Lq;
    if (fabsf(dL) < 1e-6f) return 0.0f;  // SPMSM
    
    float term = flux / (2.0f * dL);
    return term - sqrtf(term * term + iq_ref * iq_ref);
}
```

## 7. 速度环设计

```c
typedef struct {
    PI_Controller pi;
    float speed_rpm;
    float accel_limit;  // 加速度限幅 (rpm/s)
    float speed_cmd;
    float speed_ref;    // 斜坡后的参考
} SpeedLoop;

void SpeedLoop_Init(SpeedLoop *sl, float Kp, float Ki, float dt,
                     float out_min, float out_max, float accel_limit) {
    sl->pi.Kp = Kp;
    sl->pi.Ki = Ki;
    sl->pi.dt = dt;
    sl->pi.out_min = out_min;
    sl->pi.out_max = out_max;
    sl->pi.integral = 0;
    sl->accel_limit = accel_limit;
    sl->speed_ref = 0;
}

float SpeedLoop_Update(SpeedLoop *sl, float speed_cmd, float speed_fbk) {
    // 斜坡限速
    float delta = speed_cmd - sl->speed_ref;
    float max_delta = sl->accel_limit * sl->pi.dt;
    if (delta > max_delta) delta = max_delta;
    if (delta < -max_delta) delta = -max_delta;
    sl->speed_ref += delta;
    
    // PI控制
    return PI_Update(&sl->pi, sl->speed_ref, speed_fbk);
}
```

## 8. 保护机制

```c
typedef struct {
    float over_current;     // 过流阈值 (A)
    float over_voltage;     // 过压阈值 (V)
    float under_voltage;    // 欠压阈值 (V)
    float over_speed;       // 过速阈值 (rpm)
    uint32_t fault_flags;
} Motor_Protect;

#define FAULT_OC    0x01
#define FAULT_OV    0x02
#define FAULT_UV    0x04
#define FAULT_OS    0x08

uint32_t Motor_CheckFault(Motor_Protect *prot,
                          float ia, float ib, float ic,
                          float vdc, float speed) {
    uint32_t fault = 0;
    
    if (fabsf(ia) > prot->over_current || 
        fabsf(ib) > prot->over_current || 
        fabsf(ic) > prot->over_current)
        fault |= FAULT_OC;
    
    if (vdc > prot->over_voltage) fault |= FAULT_OV;
    if (vdc < prot->under_voltage) fault |= FAULT_UV;
    if (fabsf(speed) > prot->over_speed) fault |= FAULT_OS;
    
    prot->fault_flags = fault;
    return fault;
}
```

## 9. 完整FOC控制流程（1ms中断）

```c
void FOC_ISR_1ms(void) {
    // 1. 采样三相电流
    float ia, ib, ic;
    ADC_ReadCurrent(&ia, &ib, &ic);
    
    // 2. Clarke变换
    float alpha, beta;
    Clarke_2Phase(ia, ib, &alpha, &beta);
    
    // 3. 获取电角度（编码器或观测器）
    float theta = Encoder_GetElecAngle();
    
    // 4. Park变换
    float id, iq;
    Park_Transform(alpha, beta, theta, &id, &iq);
    
    // 5. 电流环PI + 解耦
    float Vd, Vq;
    CurrentLoop_Update(&current_loop, id_ref, iq_ref, id, iq, &Vd, &Vq);
    
    // 6. Park逆变换
    float valpha, vbeta;
    Park_InverseTransform(Vd, Vq, theta, &valpha, &vbeta);
    
    // 7. SVPWM
    float Ta, Tb, Tc;
    SVPWM_Calc(valpha, vbeta, Vdc, &Ta, &Tb, &Tc);
    
    // 8. 更新PWM占空比
    PWM_SetDuty(Ta, Tb, Tc);
}
```
