## 1. 容错控制概述

容错控制（Fault-Tolerant Control, FTC）是指系统在发生故障时，仍能保持稳定性并维持可接受性能的控制策略。

### 1.1 故障类型分类

| 故障类型 | 描述 | 电赛常见场景 |
|---------|------|-------------|
| 传感器故障 | 偏移、漂移、卡死、噪声增大 | 陀螺仪漂移、编码器丢脉冲 |
| 执行器故障 | 卡死、失效、效率下降 | 电机堵转、舵机失灵 |
| 系统参数突变 | 负载变化、结构损坏 | 负载脱落、桨叶断裂 |
| 通信故障 | 丢包、延迟、中断 | 无线链路中断 |

### 1.2 容错控制分类

```
容错控制
├── 被动容错控制（Passive FTC）
│   ├── 鲁棒控制：固定控制器抵御故障
│   └── 可靠控制：设计时考虑所有故障模式
└── 主动容错控制（Active FTC）
    ├── 故障检测与诊断（FDD）
    ├── 控制器重构
    └── 模型自适应
```

## 2. 故障检测与诊断（FDD）

### 2.1 基于模型的故障检测

#### 残差生成法

```
残差 r(k) = y(k) - ŷ(k)
其中 y(k) 为实际输出，ŷ(k) 为模型预测输出

故障判定：
  |r(k)| > 阈值 δ → 故障发生
```

#### C语言实现：残差检测器

```c
typedef struct {
    float threshold;        // 故障检测阈值
    float filter_coeff;     // 残差滤波系数
    float residual_filtered; // 滤波后残差
    int fault_count;        // 连续超限计数
    int fault_trigger;      // 触发计数阈值
    bool fault_detected;    // 故障标志
} FaultDetector_t;

void FaultDetector_Init(FaultDetector_t *fd, float threshold, int trigger_count) {
    fd->threshold = threshold;
    fd->filter_coeff = 0.1f;
    fd->residual_filtered = 0.0f;
    fd->fault_count = 0;
    fd->fault_trigger = trigger_count;
    fd->fault_detected = false;
}

bool FaultDetector_Update(FaultDetector_t *fd, float measured, float predicted) {
    float residual = measured - predicted;
    
    // 一阶低通滤波残差
    fd->residual_filtered += fd->filter_coeff * (residual - fd->residual_filtered);
    
    if (fabsf(fd->residual_filtered) > fd->threshold) {
        fd->fault_count++;
        if (fd->fault_count >= fd->fault_trigger) {
            fd->fault_detected = true;
        }
    } else {
        fd->fault_count = 0;  // 重置计数
        fd->fault_detected = false;
    }
    
    return fd->fault_detected;
}
```

### 2.2 基于信号的故障检测

#### 传感器卡死检测

```c
bool DetectSensorStuck(float *buffer, int len, float epsilon) {
    // 检查最近N个采样值是否几乎不变
    for (int i = 1; i < len; i++) {
        if (fabsf(buffer[i] - buffer[i-1]) > epsilon) {
            return false;  // 有变化，未卡死
        }
    }
    return true;  // 连续N个值几乎不变，判断卡死
}
```

#### 传感器漂移检测

```c
typedef struct {
    float running_mean;
    float running_var;
    float drift_threshold;
    int sample_count;
    float initial_value;
} DriftDetector_t;

bool DriftDetector_Update(DriftDetector_t *dd, float value) {
    dd->sample_count++;
    
    // 在线均值和方差（Welford算法）
    float delta = value - dd->running_mean;
    dd->running_mean += delta / dd->sample_count;
    float delta2 = value - dd->running_mean;
    dd->running_var += (delta * delta2 - dd->running_var) / dd->sample_count;
    
    // 检测漂移：均值偏移过大且方差较小
    float mean_shift = fabsf(dd->running_mean - dd->initial_value);
    float std_dev = sqrtf(dd->running_var);
    
    return (mean_shift > dd->drift_threshold) && (std_dev < dd->drift_threshold * 0.5f);
}
```

### 2.3 基于统计的故障检测

#### 3σ原则检测

```c
bool ThreeSigmaTest(float value, float mean, float std_dev) {
    return fabsf(value - mean) > 3.0f * std_dev;
}
```

#### 卡方检验（多变量）

```c
// 适用于多传感器一致性检验
float ChiSquareTest(float *residuals, float *covariance_inv, int n) {
    float chi2 = 0.0f;
    // χ² = r^T * P^{-1} * r
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            chi2 += residuals[i] * covariance_inv[i * n + j] * residuals[j];
        }
    }
    return chi2;  // 与χ²分布临界值比较
}
```

## 3. 故障隔离

### 3.1 多模型假设检验

```c
// 为每种故障模式建立模型，通过似然函数比较
typedef struct {
    float model_output;
    float likelihood;
    float residual_variance;
} Hypothesis_t;

int FaultIsolation(Hypothesis_t *hypotheses, int n_models, float measurement) {
    int best_model = 0;
    float best_likelihood = -1e10f;
    
    for (int i = 0; i < n_models; i++) {
        float residual = measurement - hypotheses[i].model_output;
        // 高斯似然：ln(L) = -r²/(2σ²) - ln(σ√(2π))
        hypotheses[i].likelihood = -0.5f * residual * residual / hypotheses[i].residual_variance;
        
        if (hypotheses[i].likelihood > best_likelihood) {
            best_likelihood = hypotheses[i].likelihood;
            best_model = i;
        }
    }
    
    return best_model;
}
```

### 3.2 结构化残差法

为每个故障设计专用残差，使其仅对特定故障敏感：

```
残差r1 → 仅对故障f1敏感 → r1超限 → 定位f1
残差r2 → 仅对故障f2敏感 → r2超限 → 定位f2
...
```

### 4.1 增益调度重构

```c
typedef struct {
    float Kp_normal, Ki_normal, Kd_normal;
    float Kp_degraded, Ki_degraded, Kd_degraded;
    bool fault_active;
} AdaptivePID_t;

float AdaptivePID_Compute(AdaptivePID_t *pid, float error) {
    float Kp, Ki, Kd;
    
    if (pid->fault_active) {
        // 降级模式：降低增益，增加稳定性裕度
        Kp = pid->Kp_degraded;
        Ki = pid->Ki_degraded;
        Kd = pid->Kd_degraded;
    } else {
        Kp = pid->Kp_normal;
        Ki = pid->Ki_normal;
        Kd = pid->Kd_degraded;
    }
    
    // PID计算...
    return Kp * error;  // 简化示意
}
```

### 4.2 多控制器切换

```c
typedef struct {
    float (*controller[3])(float error, float dt);  // 正常/降级/应急
    int current_mode;
    float (*mode_selector)(float* sensor_data);
} SwitchableController_t;

float SwitchableController_Run(SwitchableController_t *sc, float error, float dt, float *sensors) {
    // 模式选择器根据传感器状态选择控制器
    int new_mode = (int)sc->mode_selector(sensors);
    
    // 切换时做平滑过渡
    if (new_mode != sc->current_mode) {
        // 可添加无扰切换逻辑
        sc->current_mode = new_mode;
    }
    
    return sc->controller[sc->current_mode](error, dt);
}
```

### 4.3 控制分配重构（冗余执行器）

当某执行器故障时，重新分配控制量到剩余执行器：

```c
// 双电机差速小车，一个电机故障时的处理
void ControlAllocation_2Motor(float v_cmd, float omega_cmd, 
                               bool motor_L_fault, bool motor_R_fault,
                               float *motor_L, float *motor_R) {
    float v_L = v_cmd - omega_cmd;
    float v_R = v_cmd + omega_cmd;
    
    if (motor_L_fault) {
        // 左电机故障：仅用右电机，放弃转向
        *motor_L = 0;
        *motor_R = v_cmd * 2.0f;
    } else if (motor_R_fault) {
        *motor_L = v_cmd * 2.0f;
        *motor_R = 0;
    } else {
        *motor_L = v_L;
        *motor_R = v_R;
    }
}
```

## 5. 冗余设计

### 5.1 传感器冗余

```
方案1：同型号冗余
  传感器A ──┐
  传感器B ──┼── 中值/均值选择 → 输出
  传感器C ──┘

方案2：异构冗余
  IMU姿态 ──┐
  编码器推算 ┼── 加权融合 → 输出
  视觉估计 ──┘
```

#### 中值选择法

```c
float MedianSelect3(float a, float b, float c) {
    if ((a >= b && a <= c) || (a >= c && a <= b)) return a;
    if ((b >= a && b <= c) || (b >= c && b <= a)) return b;
    return c;
}
```

#### 加权表决法

```c
typedef struct {
    float value;
    float variance;  // 估计方差（越小越可信）
} SensorEstimate_t;

float WeightedVoting(SensorEstimate_t *estimates, int n) {
    float weight_sum = 0.0f;
    float value_sum = 0.0f;
    
    for (int i = 0; i < n; i++) {
        float w = 1.0f / estimates[i].variance;
        weight_sum += w;
        value_sum += w * estimates[i].value;
    }
    
    return value_sum / weight_sum;
}
```

### 5.2 执行器冗余

| 冗余方案 | 实现方式 | 适用场景 |
|---------|---------|---------|
| 热备份 | 主执行器+备用执行器同时在线 | 关键系统 |
| 冷备份 | 备用执行器待机，主故障时切换 | 低功耗系统 |
| 功能冗余 | 不同执行器可实现相同功能 | 四旋翼失去一桨 |

### 5.3 计算冗余

```c
// 双机热备：主控和备份同时运行，心跳检测切换
typedef struct {
    bool primary_alive;
    uint32_t last_heartbeat;
    uint32_t timeout_ms;
} RedundancyManager_t;

bool Redundancy_CheckPrimary(RedundancyManager_t *rm, uint32_t current_time) {
    if (current_time - rm->last_heartbeat > rm->timeout_ms) {
        rm->primary_alive = false;
        return false;  // 主控失联，切换到备份
    }
    return true;
}
```

### 6.1 四旋翼容错控制

```
故障场景：单电机失效
容错策略：
1. 检测：通过IMU和电机反馈判断电机故障
2. 隔离：确定故障电机编号
3. 重构：

   - 降低期望：放弃精确悬停，改为安全降落
```

### 6.2 传感器融合容错

```c
// 加速度计+陀螺仪互补融合，任一故障自动降级
typedef struct {
    float angle_complementary;
    float angle_accel;
    float angle_gyro_integral;
    float alpha;  // 融合系数
    bool accel_fault;
    bool gyro_fault;
} ComplementaryFilter_FT_t;

float ComplementaryFilter_FT_Update(ComplementaryFilter_FT_t *cf, 
                                      float accel_angle, float gyro_rate, float dt) {
    if (cf->accel_fault && cf->gyro_fault) {
        return cf->angle_complementary;  // 全故障，保持上次值
    } else if (cf->accel_fault) {
        // 加速度计故障：纯陀螺仪积分（会漂移）
        cf->angle_gyro_integral += gyro_rate * dt;
        cf->angle_complementary = cf->angle_gyro_integral;
    } else if (cf->gyro_fault) {
        // 陀螺仪故障：纯加速度计（有噪声）
        cf->angle_complementary = accel_angle;
    } else {
        // 正常互补融合
        cf->angle_gyro_integral += gyro_rate * dt;
        cf->angle_complementary = cf->alpha * cf->angle_gyro_integral 
                                   + (1.0f - cf->alpha) * accel_angle;
    }
    
    return cf->angle_complementary;
}
```

### 6.3 通信容错

```c
// 丢包容错：预期控制+本地缓存
typedef struct {
    float last_valid_cmd;
    uint32_t last_receive_time;
    uint32_t timeout_ms;
    float safe_default_cmd;  // 超时安全指令
} CommFaultTolerance_t;

float CommFaultTolerance_GetCmd(CommFaultTolerance_t *cft, 
                                  float new_cmd, bool new_valid, uint32_t now) {
    if (new_valid) {
        cft->last_valid_cmd = new_cmd;
        cft->last_receive_time = now;
        return new_cmd;
    }
    
    if (now - cft->last_receive_time < cft->timeout_ms) {
        return cft->last_valid_cmd;  // 短暂丢包，沿用上次值
    }
    
    return cft->safe_default_cmd;  // 超时，进入安全模式
}
```
