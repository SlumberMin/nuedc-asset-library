## 1. 运动控制基础概念

### 1.1 运动控制三要素

```
位置环（外环）
  ├── 速度环（中环）
  │     └── 电流环（内环）
  │           └── PWM驱动
```

### 1.2 运动控制关键参数

| 参数 | 含义 | 典型单位 |
|------|------|----------|
| 位置 | 当前轴所在位置 | mm, °, 脉冲 |
| 速度 | 位置变化率 | mm/s, °/s, rpm |
| 加速度 | 速度变化率 | mm/s², °/s² |
| 加加速度(Jerk) | 加速度变化率 | mm/s³ |
| 跟随误差 | 指令位置与实际位置之差 | mm, 脉冲 |

### 1.3 坐标系

| 坐标系 | 说明 | 应用 |
|--------|------|------|
| 关节空间 | 各轴独立坐标 | 单轴运动 |
| 直角坐标 | XYZ笛卡尔坐标 | 直线/圆弧运动 |
| 工具坐标 | 以工具末端为基准 | 焊接/切割 |
| 用户坐标 | 自定义参考系 | 特定工件 |

## 2. 轨迹规划算法

### 2.1 梯形速度规划

最简单的加减速策略，加速度为常数。

```
速度
 ^
 |    ___________
 |   /           \
 |  /             \
 | /               \
 |/                 \
 +-------------------> 时间
    t_acc  t_run  t_dec
```

```c
// 梯形速度规划核心算法
typedef struct {
    float v_max;      // 最大速度
    float a_max;      // 最大加速度
    float target_pos; // 目标位置
    float t_acc;      // 加速段时间
    float t_run;      // 匀速段时间
    float t_dec;      // 减速段时间
    float t_total;    // 总时间
} TrapezoidProfile;

void trapezoid_init(TrapezoidProfile *p, float v_max, float a_max, float distance) {
    p->v_max = v_max;
    p->a_max = a_max;
    p->target_pos = distance;
    
    float d = fabsf(distance);
    
    // 判断是否能达到最大速度
    float d_acc = v_max * v_max / (2.0f * a_max); // 加速距离
    
    if (2.0f * d_acc > d) {
        // 三角形轮廓（无法达到最大速度）
        p->t_acc = sqrtf(d / a_max);
        p->t_run = 0;
        p->t_dec = p->t_acc;
    } else {
        // 梯形轮廓
        p->t_acc = v_max / a_max;
        p->t_run = (d - 2.0f * d_acc) / v_max;
        p->t_dec = p->t_acc;
    }
    p->t_total = p->t_acc + p->t_run + p->t_dec;
}

float trapezoid_get_pos(TrapezoidProfile *p, float t) {
    float pos;
    float sign = (p->target_pos >= 0) ? 1.0f : -1.0f;
    
    if (t <= 0) {
        pos = 0;
    } else if (t < p->t_acc) {
        // 加速段
        pos = 0.5f * p->a_max * t * t;
    } else if (t < p->t_acc + p->t_run) {
        // 匀速段
        float dt = t - p->t_acc;
        float d_acc = 0.5f * p->a_max * p->t_acc * p->t_acc;
        pos = d_acc + p->v_max * dt;
    } else if (t < p->t_total) {
        // 减速段
        float dt = t - p->t_acc - p->t_run;
        float d_acc = 0.5f * p->a_max * p->t_acc * p->t_acc;
        float d_run = p->v_max * p->t_run;
        pos = d_acc + d_run + p->v_max * dt - 0.5f * p->a_max * dt * dt;
    } else {
        pos = fabsf(p->target_pos);
    }
    
    return sign * pos;
}

float trapezoid_get_vel(TrapezoidProfile *p, float t) {
    float vel;
    float sign = (p->target_pos >= 0) ? 1.0f : -1.0f;
    
    if (t <= 0) {
        vel = 0;
    } else if (t < p->t_acc) {
        vel = p->a_max * t;
    } else if (t < p->t_acc + p->t_run) {
        vel = p->v_max;
    } else if (t < p->t_total) {
        float dt = t - p->t_acc - p->t_run;
        vel = p->v_max - p->a_max * dt;
    } else {
        vel = 0;
    }
    
    return sign * vel;
}
```

### 2.2 S形速度规划（七段式）

控制加加速度(Jerk)，使加速度连续变化，运动更平滑。

```
加速度
 ^
 |    ____
 |   /    \
 |  /      \
 | /        \
 |/          \
 +--\--------\---> 时间
     \      /
      \____/
       速度变化更平滑
```

```c
// S形速度规划（七段式）
typedef struct {
    float v_max, a_max, j_max;  // 最大速度、加速度、加加速度
    float t[8];                  // 7个段的时间节点 t[0]=0, t[7]=total
    float target_pos;
} SProfile;

void s_profile_init(SProfile *p, float v_max, float a_max, float j_max, float distance) {
    p->v_max = v_max;
    p->a_max = a_max;
    p->j_max = j_max;
    p->target_pos = distance;
    
    float d = fabsf(distance);
    
    // 加速段时间：加加速度段 + 恒加速段 + 减加速度段
    float t_j = a_max / j_max;              // Jerk段时间
    float t_a = v_max / a_max - t_j;         // 恒加速段时间
    
    if (t_a < 0) {
        // 速度不够高，无法达到a_max，需要重新计算
        v_max = sqrtf(a_max * a_max / j_max);
        t_j = a_max / j_max;
        t_a = 0;
        p->v_max = v_max;
    }
    
    // 加速阶段
    float d_acc = v_max * (t_j + t_a + t_j); // 加速总距离
    // ... (完整实现需考虑三角形轮廓等特殊情况)
    
    p->t[0] = 0;
    p->t[1] = t_j;                                    // 加加速度结束
    p->t[2] = t_j + t_a;                              // 恒加速结束
    p->t[3] = 2 * t_j + t_a;                          // 减加速度结束(加速完成)
    // t[4]~t[7] 类似处理减速段
    p->t[7] = p->t[3] + (d - 2 * d_acc) / v_max + 3 * t_j + t_a; // 总时间
}
```

### 2.3 五次多项式规划

```c
// 五次多项式：q(t) = a0 + a1*t + a2*t² + a3*t³ + a4*t⁴ + a5*t⁵
// 边界条件：起止位置、速度、加速度共6个约束

typedef struct {
    float a0, a1, a2, a3, a4, a5;
} QuinticPoly;

QuinticPoly quintic_plan(float q0, float qf, float v0, float vf, 
                          float a0_init, float af, float T) {
    QuinticPoly poly;
    poly.a0 = q0;
    poly.a1 = v0;
    poly.a2 = a0_init / 2.0f;
    
    // 求解三元一次方程组得到 a3, a4, a5
    float T2 = T * T;
    float T3 = T2 * T;
    float T4 = T3 * T;
    float T5 = T4 * T;
    
    float h = qf - q0 - v0 * T - 0.5f * a0_init * T2;
    float g = vf - v0 - a0_init * T;
    float k = af - a0_init;
    
    poly.a3 = (20 * h - 8 * g * T + k * T2) / (2 * T3);
    poly.a4 = (-15 * h + 7 * g * T - k * T2) / (T4);
    poly.a5 = (6 * h - 3 * g * T + k * T2) / (2 * T5);
    
    return poly;
}

float quintic_eval(QuinticPoly *p, float t) {
    return p->a0 + t * (p->a1 + t * (p->a2 + t * (p->a3 + t * (p->a4 + t * p->a5))));
}
```

## 3. 插补算法

### 3.1 逐点比较法直线插补

```c
// 逐点比较法 - 直线插补
// 原理：每步计算偏差，决定下一步进给方向

typedef struct {
    int x_end, y_end;   // 终点坐标（脉冲数）
    int x_cur, y_cur;   // 当前位置
    int F;               // 偏差寄存器
    int total_steps;     // 总步数
    int step_count;      // 已走步数
} LineInterpolator;

void line_interp_init(LineInterpolator *li, int xe, int ye) {
    li->x_end = abs(xe);
    li->y_end = abs(ye);
    li->x_cur = 0;
    li->y_cur = 0;
    li->F = 0;
    li->total_steps = li->x_end + li->y_end;
    li->step_count = 0;
}

// 返回：0=X进给, 1=Y进给, -1=完成
int line_interp_step(LineInterpolator *li) {
    if (li->step_count >= li->total_steps) return -1;
    
    if (li->F >= 0) {
        // 偏差≥0，X方向进给
        li->F -= li->y_end;
        li->x_cur++;
        li->step_count++;
        return 0; // X进给
    } else {
        // 偏差<0，Y方向进给
        li->F += li->x_end;
        li->y_cur++;
        li->step_count++;
        return 1; // Y进给
    }
}
```

### 3.2 DDA（数字微分分析器）直线插补

```c
// DDA插补 - 更均匀的脉冲分配
typedef struct {
    int x_end, y_end;
    int Ex, Ey;         // 余数寄存器
    int N;              // 脉冲总数
    int count;
} DDAInterpolator;

void dda_init(DDAInterpolator *dd, int xe, int ye) {
    dd->x_end = abs(xe);
    dd->y_end = abs(ye);
    dd->N = (dd->x_end > dd->y_end) ? dd->x_end : dd->y_end;
    dd->Ex = 0;
    dd->Ey = 0;
    dd->count = 0;
}

// 每次调用返回(X脉冲, Y脉冲)
void dda_step(DDAInterpolator *dd, int *x_pulse, int *y_pulse) {
    if (dd->count >= dd->N) {
        *x_pulse = *y_pulse = 0;
        return;
    }
    
    dd->Ex += dd->x_end;
    dd->Ey += dd->y_end;
    
    *x_pulse = (dd->Ex >= dd->N) ? 1 : 0;
    *y_pulse = (dd->Ey >= dd->N) ? 1 : 0;
    
    if (*x_pulse) dd->Ex -= dd->N;
    if (*y_pulse) dd->Ey -= dd->N;
    
    dd->count++;
}
```

### 3.3 圆弧插补（逐点比较法）

```c
// 逐点比较法 - 圆弧插补
typedef struct {
    int x0, y0;        // 圆心（脉冲数）
    int R;              // 半径（脉冲数）
    int x_cur, y_cur;   // 当前位置
    int F;               // 偏差
    int quadrant;        // 当前象限
    int dir;             // 0=顺时针, 1=逆时针
} ArcInterpolator;

void arc_init(ArcInterpolator *arc, int xc, int yc, int R, 
              int x_start, int y_start, int dir) {
    arc->x0 = xc;
    arc->y0 = yc;
    arc->R = R;
    arc->x_cur = x_start;
    arc->y_cur = y_start;
    arc->dir = dir;
    
    // 初始偏差
    int dx = x_start - xc;
    int dy = y_start - yc;
    arc->F = dx * dx + dy * dy - R * R;
}

// 圆弧插补一步
void arc_step(ArcInterpolator *arc, int *dx, int *dy) {
    if (arc->dir == 1) {
        // 逆时针
        if (arc->F >= 0) {
            // 向圆内进给（X方向）
            int sign = (arc->x_cur > arc->x0) ? -1 : 1;
            *dx = sign; *dy = 0;
            arc->x_cur += *dx;
            arc->F += 2 * (arc->x_cur - arc->x0) + 1;
        } else {
            // 向圆外进给（Y方向）
            int sign = (arc->y_cur > arc->y0) ? -1 : 1;
            *dx = 0; *dy = sign;
            arc->y_cur += *dy;
            arc->F += 2 * (arc->y_cur - arc->y0) + 1;
        }
    }
    // 顺时针类似处理
}
```

### 3.4 Bresenham直线算法（嵌入式常用）

```c
// Bresenham直线算法 - 无浮点运算，适合MCU
void bresenham_line(int x0, int y0, int x1, int y1, 
                    void (*plot)(int, int)) {
    int dx = abs(x1 - x0);
    int dy = abs(y1 - y0);
    int sx = (x0 < x1) ? 1 : -1;
    int sy = (y0 < y1) ? 1 : -1;
    int err = dx - dy;
    
    while (1) {
        plot(x0, y0);
        if (x0 == x1 && y0 == y1) break;
        
        int e2 = 2 * err;
        if (e2 > -dy) {
            err -= dy;
            x0 += sx;
        }
        if (e2 < dx) {
            err += dx;
            y0 += sy;
        }
    }
}

// Bresenham圆弧
void bresenham_circle(int xc, int yc, int r, void (*plot)(int, int)) {
    int x = 0, y = r;
    int d = 3 - 2 * r;
    
    while (x <= y) {
        plot(xc + x, yc + y);
        plot(xc - x, yc + y);
        plot(xc + x, yc - y);
        plot(xc - x, yc - y);
        plot(xc + y, yc + x);
        plot(xc - y, yc + x);
        plot(xc + y, yc - x);
        plot(xc - y, yc - x);
        
        if (d < 0) {
            d += 4 * x + 6;
        } else {
            d += 4 * (x - y) + 10;
            y--;
        }
        x++;
    }
}
```

### 3.5 速度前瞻与拐角处理

```c
// 速度前瞻：预读后续段，提前减速
typedef struct {
    float length;       // 段长度
    float entry_vel;    // 入口速度
    float exit_vel;     // 出口速度
    float cruise_vel;   // 巡航速度
} MotionSegment;

// 速度前瞻算法（backward pass + forward pass）
void velocity_lookahead(MotionSegment *segs, int n, float v_max, float a_max) {
    // 向后规划（从最后一段到第一段）
    segs[n-1].exit_vel = 0; // 最后一段停止
    for (int i = n - 1; i >= 0; i--) {
        if (i < n - 1) {
            // 计算允许的出口速度（考虑下一段的入口约束）
            float max_exit = sqrtf(segs[i+1].entry_vel * segs[i+1].entry_vel 
                                   + 2 * a_max * segs[i].length);
            segs[i].exit_vel = fminf(segs[i].cruise_vel, max_exit);
        }
    }
    
    // 向前规划（从第一段到最后一段）
    segs[0].entry_vel = 0; // 第一段从静止开始
    for (int i = 0; i < n; i++) {
        segs[i].cruise_vel = fminf(v_max, segs[i].cruise_vel);
        if (i > 0) {
            float max_entry = sqrtf(segs[i-1].exit_vel * segs[i-1].exit_vel 
                                    + 2 * a_max * segs[i].length);
            segs[i].entry_vel = fminf(segs[i].entry_vel, max_entry);
        }
    }
}

// 拐角速度限制（基于弦误差或角度）
float corner_velocity_limit(float v1, float v2, float angle, float a_max, float chord_err) {
    // 基于角度的速度限制
    float sin_half = sinf(angle / 2.0f);
    float v_angle = a_max / (1.0f / chord_err); // 简化计算
    
    // 基于弦误差的速度限制
    float v_chord = sqrtf(a_max * chord_err / (1.0f - cosf(angle / 2.0f)));
    
    return fminf(v_angle, v_chord);
}
```

## 4. 同步控制技术

### 4.1 主从同步

```c
// 主从同步：从轴跟踪主轴位置
typedef struct {
    float master_pos;       // 主轴位置
    float slave_pos;        // 从轴位置
    float ratio;            // 传动比
    float offset;           // 偏移量
    float slave_cmd;        // 从轴指令
    
    // 位置环PID
    float Kp, Ki, Kd;
    float err_sum, err_last;
} MasterSlaveSync;

void master_slave_init(MasterSlaveSync *s, float ratio, float offset) {
    s->ratio = ratio;
    s->offset = offset;
    s->Kp = 2.0f;
    s->Ki = 0.01f;
    s->Kd = 0.1f;
    s->err_sum = 0;
    s->err_last = 0;
}

float master_slave_update(MasterSlaveSync *s, float master_pos, float slave_fb) {
    // 计算从轴目标位置
    float target = s->ratio * master_pos + s->offset;
    
    // PID闭环修正
    float err = target - slave_fb;
    s->err_sum += err;
    float d_err = err - s->err_last;
    s->err_last = err;
    
    s->slave_cmd = s->Kp * err + s->Ki * s->err_sum + s->Kd * d_err;
    
    return s->slave_cmd;
}
```

### 4.2 电子齿轮（Electronic Gearing）

```c
// 电子齿轮 - 软件实现的传动比
typedef struct {

    int32_t slave_count;        // 从轴编码器计数

void e_gear_init(ElectronicGear *g, int32_t master_num, int32_t slave_den) {
    g->master_num = master_num;
    g->slave_den = slave_den;
    g->remainder = 0;
    g->gear_error = 0;
    g->master_count = 0;
    g->slave_count = 0;
}

// 每个控制周期调用
int32_t e_gear_compute(ElectronicGear *g, int32_t master_delta) {
    // 累积主轴增量
    g->remainder += master_delta * g->master_num;
    
    // 计算从轴增量
    int32_t slave_delta = g->remainder / g->slave_den;
    g->remainder %= g->slave_den;
    
    g->master_count += master_delta;
    g->slave_count += slave_delta;
    
    return slave_delta;
}

// 实际传动比 = master_num / slave_den
// 例：master_num=3, slave_den=1 → 从轴转3圈，主轴转1圈
```

```c
float cam_lookup(ElectronicCam *cam, float master_angle) {
    // 归一化到[0, master_period)
    float angle = fmodf(master_angle + cam->master_offset, cam->master_period);
    if (angle < 0) angle += cam->master_period;
    
    // 映射到表索引
    float index = angle / cam->master_period * CAM_TABLE_SIZE;
    int idx0 = (int)index;
    int idx1 = (idx0 + 1) % CAM_TABLE_SIZE;
    float frac = index - idx0;
    
    // 线性插值
    return cam->cam_table[idx0] * (1.0f - frac) + cam->cam_table[idx1] * frac;
}

// 凸轮运动控制

    float target = cam_lookup(cam, master_pos);
    float error = target - slave_feedback;
    
    // 简单比例控制（实际应用中用更复杂的控制器）
    return error * 10.0f; // Kp = 10
```

```c
typedef enum {
    CAM_MODIFIED_SINE,      // 修正正弦
    CAM_MODIFIED_TRAPEZOID, // 修正梯形
    CAM_CYCLOIDAL,          // 摆线
    CAM_POLYNOMIAL          // 多项式

// 修正正弦凸轮曲线
void generate_modified_sine_cam(float *table, int size, float lift) {
    for (int i = 0; i < size; i++) {
        float theta = (float)i / size * 2.0f * M_PI;
        float s;
        
        if (theta < M_PI / 4.0f) {
            // 0 ~ π/4: 正弦加速
            s = lift / (4.0f + M_PI) * (M_PI * theta / (M_PI/4.0f) - sinf(4.0f * theta));
        } else if (theta < 3.0f * M_PI / 4.0f) {
            // π/4 ~ 3π/4: 匀速段
            float theta1 = theta - M_PI / 4.0f;
            s = lift / (4.0f + M_PI) * (2.0f + M_PI * theta1 / (M_PI/2.0f));
        } else {
            // 3π/4 ~ π: 正弦减速
            float theta2 = theta - 3.0f * M_PI / 4.0f;
            s = lift / (4.0f + M_PI) * (4.0f + M_PI * theta2 / (M_PI/4.0f) 
                + sinf(4.0f * theta2));
        }
        table[i] = s;
    }
}

// 摆线凸轮曲线
void generate_cycloidal_cam(float *table, int size, float lift) {
    for (int i = 0; i < size; i++) {
        float theta = (float)i / size;
        table[i] = lift * (theta - sinf(2.0f * M_PI * theta) / (2.0f * M_PI));
    }
}
```

```c
    }
```

### 6.2 机器人正逆运动学

```c
// 两连杆机械臂正运动学
typedef struct {
    float L1, L2;  // 连杆长度
    float theta1, theta2; // 关节角度
} TwoLinkArm;

void forward_kinematics(TwoLinkArm *arm, float *x, float *y) {
    *x = arm->L1 * cosf(arm->theta1) + arm->L2 * cosf(arm->theta1 + arm->theta2);
    *y = arm->L1 * sinf(arm->theta1) + arm->L2 * sinf(arm->theta1 + arm->theta2);
}

// 逆运动学
int inverse_kinematics(TwoLinkArm *arm, float x, float y) {
    float r2 = x * x + y * y;
    float r = sqrtf(r2);
    
    // 检查是否在工作空间内
    if (r > arm->L1 + arm->L2 || r < fabsf(arm->L1 - arm->L2)) {
        return -1; // 不可达
    }
    
    // theta2
    float cos_theta2 = (r2 - arm->L1 * arm->L1 - arm->L2 * arm->L2) 
                       / (2.0f * arm->L1 * arm->L2);
    arm->theta2 = acosf(cos_theta2); // 或 -acosf(cos_theta2) 对应另一种构型
    
    // theta1
    float alpha = atan2f(y, x);
    float beta = atan2f(arm->L2 * sinf(arm->theta2), 
                        arm->L1 + arm->L2 * cosf(arm->theta2));
    arm->theta1 = alpha - beta;
    
    return 0; // 成功
}
```

```c
    // 设置方向
    if (target > axis->current_pos) {
        HAL_GPIO_WritePin(axis->dir_port, axis->dir_pin, GPIO_PIN_SET);
    } else {
        HAL_GPIO_WritePin(axis->dir_port, axis->dir_pin, GPIO_PIN_RESET);
    }
    
    // 设置脉冲频率
    uint32_t arr = (uint32_t)(SystemCoreClock / (freq * 2)) - 1;
    __HAL_TIM_SET_AUTORELOAD(axis->htim, arr);
    __HAL_TIM_SET_COMPARE(axis->htim, axis->channel, arr / 2);
    HAL_TIM_PWM_Start(axis->htim, axis->channel);
}

void pulse_dir_update(PulseDirAxis *axis) {
    if (axis->current_pos != axis->target_pos) {
        axis->current_pos += (axis->target_pos > axis->current_pos) ? 1 : -1;
    }
    if (axis->current_pos == axis->target_pos) {
        HAL_TIM_PWM_Stop(axis->htim, axis->channel);
    }
}
```

## 10. 常见问题与解决

### 10.1 运动不平滑

**原因**：加速度不连续、插补周期太长、规划算法选择不当
**解决**：
- 使用S形速度规划替代梯形规划
- 减小插补周期（推荐≤1ms）
- 增加速度前瞻段数

### 10.2 位置精度不够

**原因**：编码器分辨率低、机械间隙、跟随误差大
**解决**：
- 提高编码器分辨率或使用更高精度的编码器
- 消除机械间隙（预紧、补偿）
- 增大位置环增益，减小跟随误差

### 10.3 多轴不同步

**原因**：各轴响应不一致、通信延迟
**解决**：
- 统一各轴的控制参数
- 使用硬件同步触发
- 减小控制周期抖动

### 10.4 高速运动丢步

**原因**：步进电机矩频特性不足、加速度过大
**解决**：
- 降低加速度或最大速度
- 使用伺服电机替代步进电机
- 增加电流或使用细分驱动
