## 一、FFT（快速傅里叶变换）

### 1.1 用途

频谱分析、频率测量、THD分析、滤波器设计验证。

### 1.2 算法原理（基2-FFT/DIT）

```
N点DFT → log₂N级蝶形运算
每级 N/2 个蝶形
总运算量: O(NlogN) vs DFT的 O(N²)
```

### 1.3 C实现（定点优化版）

```c
#include <math.h>
#include <stdint.h>

#define N_FFT 1024
#define LOG2_N 10

typedef struct {
    float re, im;
} complex_t;

static complex_t fft_buf[N_FFT];

// 位反转置换
static void bit_reverse(complex_t *x, int n) {
    int j = 0;
    for (int i = 0; i < n - 1; i++) {
        if (i < j) {
            complex_t tmp = x[i];
            x[i] = x[j];
            x[j] = tmp;
        }
        int k = n >> 1;
        while (k <= j) { j -= k; k >>= 1; }
        j += k;
    }
}

// 原位FFT（基2-DIT）
void fft_compute(complex_t *x, int n) {
    bit_reverse(x, n);
    for (int s = 1; s <= LOG2_N; s++) {
        int m = 1 << s;
        int half = m >> 1;
        complex_t wm = {cosf(-2.0f * M_PI / m),
                        sinf(-2.0f * M_PI / m)};
        for (int k = 0; k < n; k += m) {
            complex_t w = {1.0f, 0.0f};
            for (int j = 0; j < half; j++) {
                complex_t t, u;
                t.re = w.re * x[k+j+half].re - w.im * x[k+j+half].im;
                t.im = w.re * x[k+j+half].im + w.im * x[k+j+half].re;
                u = x[k+j];
                x[k+j].re = u.re + t.re;
                x[k+j].im = u.im + t.im;
                x[k+j+half].re = u.re - t.re;
                x[k+j+half].im = u.im - t.im;
                float wr = w.re * wm.re - w.im * wm.im;
                float wi = w.re * wm.im + w.im * wm.re;
                w.re = wr; w.im = wi;
            }
        }
    }
}

// 计算幅值谱并找峰值频率
float fft_find_peak_freq(complex_t *x, int n, float fs) {
    float max_mag = 0;
    int max_idx = 0;
    for (int i = 1; i < n/2; i++) {
        float mag = sqrtf(x[i].re*x[i].re + x[i].im*x[i].im);
        if (mag > max_mag) { max_mag = mag; max_idx = i; }
    }
    // 抛物线插值提高精度
    float y1 = sqrtf(x[max_idx-1].re*x[max_idx-1].re + x[max_idx-1].im*x[max_idx-1].im);
    float y2 = max_mag;
    float y3 = sqrtf(x[max_idx+1].re*x[max_idx+1].re + x[max_idx+1].im*x[max_idx+1].im);
    float delta = 0.5f * (y1 - y3) / (y1 - 2*y2 + y3);
    return (max_idx + delta) * fs / n;
}
```

## 二、FIR滤波器

### 2.1 基本结构

```
y[n] = Σ h[k]·x[n-k],  k=0..N-1
```

### 2.2 窗函数法设计

```c
// 设计低通FIR滤波器
// fc: 截止频率, fs: 采样率, N: 阶数(奇数)
void fir_design_lowpass(float *h, int N, float fc, float fs) {
    float wc = 2.0f * M_PI * fc / fs;
    int M = N - 1;
    for (int n = 0; n < N; n++) {
        float k = n - M/2.0f;
        if (fabsf(k) < 1e-6f)
            h[n] = wc / M_PI;
        else
            h[n] = sinf(wc * k) / (M_PI * k);
        // Hamming窗
        h[n] *= 0.54f - 0.46f * cosf(2.0f * M_PI * n / M);
    }
    // 归一化
    float sum = 0;
    for (int i = 0; i < N; i++) sum += h[i];
    for (int i = 0; i < N; i++) h[i] /= sum;
}
```

### 2.3 高效实现（滑动窗口）

```c
#define FIR_ORDER 31
static float fir_coeff[FIR_ORDER];
static float fir_delay[FIR_ORDER];
static int fir_idx = 0;

float fir_filter(float x) {
    fir_delay[fir_idx] = x;
    float y = 0;
    int k = fir_idx;
    for (int i = 0; i < FIR_ORDER; i++) {
        y += fir_coeff[i] * fir_delay[k];
        k = (k == 0) ? FIR_ORDER - 1 : k - 1;
    }
    fir_idx = (fir_idx + 1) % FIR_ORDER;
    return y;
}
```

### 2.4 参数选择经验

| 阶数N | 过渡带宽(归一化) | 适用场景 |
|-------|-----------------|---------|
| 15 | ~0.15 | 粗滤波、实时性要求高 |
| 31 | ~0.07 | 一般应用 |
| 63 | ~0.04 | 高精度频率测量 |
| 127 | ~0.02 | 音频处理 |

## 三、IIR滤波器

### 3.1 二阶节（Biquad）级联结构

```
H(z) = Π Hk(z)
Hk(z) = (b0 + b1·z⁻¹ + b2·z⁻²) / (1 + a1·z⁻¹ + a2·z⁻²)
```

### 3.2 Direct Form II 实现

```c
typedef struct {
    float b0, b1, b2;  // 分子系数
    float a1, a2;      // 分母系数（a0=1）
    float w1, w2;      // 状态变量
} Biquad;

float biquad_filter(Biquad *bq, float x) {
    float w = x - bq->a1 * bq->w1 - bq->a2 * bq->w2;
    float y = bq->b0 * w + bq->b1 * bq->w1 + bq->b2 * bq->w2;
    bq->w2 = bq->w1;
    bq->w1 = w;
    return y;
}

// 多级级联
#define IIR_SECTIONS 4
static Biquad iir_sections[IIR_SECTIONS];

float iir_cascade_filter(float x) {
    float y = x;
    for (int i = 0; i < IIR_SECTIONS; i++) {
        y = biquad_filter(&iir_sections[i], y);
    }
    return y;
}
```

### 3.3 常用IIR滤波器系数（MATLAB预计算）


### 3.4 FIR vs IIR 对比

| 特性 | FIR | IIR |
|------|-----|-----|
| 相位 | 线性相位 | 非线性 |
| 稳定性 | 始终稳定 | 可能不稳定 |
| 阶数 | 高（50~200） | 低（4~12） |
| 计算量 | 大 | 小 |
| 电赛推荐 | 频率测量 | 实时控制滤波 |

## 四、抽取与插值

### 4.1 抽取（下采样）

```c
// 先抗混叠滤波，再抽取M倍
// M: 抽取因子
float decimate(float x, int M) {
    static float acc = 0;
    static int cnt = 0;
    acc = biquad_filter(&antialias_biquad, x); // 低通滤波
    cnt++;
    if (cnt >= M) {
        cnt = 0;
        return acc;  // 输出抽取后的样本
    }
    return NAN; // 本周期不输出
}
```

### 4.2 插值（上采样）

```c
// 先零插入L倍，再低通滤波
float interpolate(float x, int L) {
    static int cnt = 0;
    float input = (cnt == 0) ? x * L : 0; // 零插入并补偿增益
    cnt = (cnt + 1) % L;
    return biquad_filter(&interp_biquad, input);
}
```

### 4.3 抽取在电赛中的应用

```
应用场景:
- ADC采样率过高 → 抽取降低数据率，减少MCU负担
- 多速率处理 → 高速ADC + 抽取 → 低速FFT分析
- 例: 1MSPS ADC → 100倍抽取 → 10kSPS → 1024点FFT
```

## 五、正交解调（I/Q解调）

### 5.1 原理

```
r(t) = A·cos(2πft + φ)
与本地载波混频:
I = r(t)·cos(2πft) → ½A·cos(φ) + 高频项
Q = r(t)·sin(2πft) → -½A·sin(φ) + 高频项
低通滤波后:
幅值 = √(I² + Q²)
相位 = atan2(-Q, I)
```

### 5.2 C实现

```c
typedef struct {
    float I, Q;
    float phase_ref;  // 参考相位累加
    float phase_inc;  // 2π·f/fs
} IQ_Demod;

void iq_demod_init(IQ_Demod *d, float freq, float fs) {
    d->phase_ref = 0;
    d->phase_inc = 2.0f * M_PI * freq / fs;
    d->I = d->Q = 0;
}

// 低通滤波系数（简单一阶IIR）
#define LP_ALPHA 0.05f

void iq_demod_process(IQ_Demod *d, float sample) {
    float cos_ref = cosf(d->phase_ref);
    float sin_ref = sinf(d->phase_ref);

    float i_raw = sample * cos_ref;
    float q_raw = -sample * sin_ref;

    // 一阶IIR低通
    d->I += LP_ALPHA * (i_raw - d->I);
    d->Q += LP_ALPHA * (q_raw - d->Q);

    d->phase_ref += d->phase_inc;
    if (d->phase_ref > 2*M_PI) d->phase_ref -= 2*M_PI;
}

float iq_get_magnitude(IQ_Demod *d) {
    return 2.0f * sqrtf(d->I*d->I + d->Q*d->Q);
}

float iq_get_phase(IQ_Demod *d) {
    return atan2f(-d->Q, d->I);
}
```

### 5.3 应用场景

```
- 锁相放大器：微弱信号检测
- 阻抗测量：测量信号幅值和相位
- AM/FM解调：通信题
- 频率响应分析：扫频测量系统传递函数
```

### 6.2 滑动平均（最简单的低通）

```c
#define AVG_N 16
static float avg_buf[AVG_N];
static int avg_idx = 0;
static float avg_sum = 0;

float moving_average(float x) {
    avg_sum -= avg_buf[avg_idx];
    avg_buf[avg_idx] = x;
    avg_sum += x;
    avg_idx = (avg_idx + 1) % AVG_N;
    return avg_sum / AVG_N;
}
```

### 6.3 中值滤波（抗脉冲噪声）

```c
float median3(float a, float b, float c) {
    if (a > b) { float t=a; a=b; b=t; }
    if (b > c) { float t=b; b=c; c=t; }
    if (a > b) { float t=a; a=b; b=t; }
    return b;
}
```
