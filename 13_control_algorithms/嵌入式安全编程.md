## 1. 安全编程概述

嵌入式系统安全编程旨在防止软件缺陷导致系统崩溃、数据损坏或安全隐患。电赛项目虽非安全关键系统，但遵循安全编程规范可显著提高系统可靠性和抗干扰能力。

### 1.1 核心原则

| 原则 | 说明 |
|------|------|
| **防御性编程** | 假设所有输入都不可信，所有函数调用都可能失败 |
| **最小权限** | 模块只访问完成任务所需的最少资源 |
| **故障安全** | 异常情况下系统进入安全状态 |
| **确定性行为** | 避免未定义行为和实现依赖 |

## 2. MISRA-C 编程规范

### 2.1 MISRA-C:2012 核心规则（电赛常用子集）

#### 指针与数组安全
```c
/* MISRA Rule 18.1: 数组下标不能越界 */
#define ARRAY_SIZE  100
uint16_t adc_buffer[ARRAY_SIZE];

/* ✅ 正确：边界检查 */
uint16_t safe_read(uint16_t index) {
    if (index < ARRAY_SIZE) {
        return adc_buffer[index];
    }
    return 0;  /* 默认值 */
}

/* ❌ 错误：未检查越界 */
uint16_t unsafe_read(uint16_t index) {
    return adc_buffer[index];  /* 可能越界！ */
}
```

#### 类型安全
```c
/* MISRA Rule 10.3: 不同位宽类型不要混用 */
/* ✅ 正确 */
uint32_t count = 0U;
int16_t temperature = (int16_t)raw_adc;

/* ❌ 错误 */
int count = 0;           /* 位宽不明确 */
```

#### 控制流安全
```c
/* MISRA Rule 15.5: 函数应只有单一出口 */
/* ✅ 正确 */
int16_t calculate_pid(int16_t error) {
    int16_t result = 0;
    
    if (kp == 0) {
        result = 0;
    } else {
        result = (int16_t)(((int32_t)kp * error) >> 8);
    }
    
    return result;  /* 唯一出口 */
}
```

## 3. CERT-C 安全编码标准

### 3.1 缓冲区溢出防护

```c
/* STR31-C: 确保字符串有足够空间 */
/* ✅ 正确：使用 snprintf 替代 sprintf */
char msg[64];
snprintf(msg, sizeof(msg), "ADC=%d Temp=%d", adc_val, temp_val);

/* ✅ 正确：安全的内存拷贝 */
void safe_copy(char *dst, size_t dst_size, const char *src) {
    if (dst == NULL || src == NULL || dst_size == 0) {
        return;
    }
    strncpy(dst, src, dst_size - 1);
    dst[dst_size - 1] = '\0';  /* 确保终止 */
}

/* ARR30-C: 不要访问超出数组边界的元素 */
#define BUFFER_SIZE 256
static uint8_t rx_buffer[BUFFER_SIZE];

void process_uart_data(uint16_t length) {
    if (length > BUFFER_SIZE) {
        length = BUFFER_SIZE;  /* 截断 */
    }
    for (uint16_t i = 0; i < length; i++) {
        process_byte(rx_buffer[i]);
    }
}
```

### 3.2 整数溢出防护

```c
/* INT31-C: 确保整数转换不丢失数据 */
/* ✅ 正确：先检查再转换 */
bool safe_uint32_to_uint16(uint32_t src, uint16_t *dst) {
    if (src > UINT16_MAX) {
        return false;  /* 溢出检测 */
    }
    *dst = (uint16_t)src;
    return true;
}

/* INT32-C: 有符号整数运算溢出检测 */
int32_t safe_add(int32_t a, int32_t b) {
    if ((b > 0) && (a > INT32_MAX - b)) {
        return INT32_MAX;  /* 上溢饱和 */
    }
    if ((b < 0) && (a < INT32_MIN - b)) {
        return INT32_MIN;  /* 下溢饱和 */
    }
    return a + b;
}

/* INT30-C: 无符号整数乘法溢出 */
uint32_t safe_multiply_u32(uint32_t a, uint32_t b) {
    if (a != 0 && b > UINT32_MAX / a) {
        return UINT32_MAX;  /* 饱和处理 */
    }
    return a * b;
}
```

### 3.3 整数溢出速查表

| 运算 | 风险 | 检测方法 |
|------|------|----------|
| a + b | 结果超出类型范围 | `if (b > MAX - a)` |
| a - b | 下溢 | `if (a < b)` (无符号) |
| a * b | 乘法溢出 | `if (a > MAX / b)` |
| a / b | 除零 | `if (b == 0)` |
| a << n | 移位溢出 | `if (n >= TYPE_BITS)` |
| (T)a | 截断 | `if (a > MAX_OF_T)` |

## 4. 看门狗设计

### 4.1 独立看门狗（IWDG）

```c
/* 独立看门狗配置 - STM32 示例 */
void iwdg_init(uint32_t timeout_ms) {
    /* IWDG 时钟 = LSI ≈ 40kHz
     * 超时 = (4 × 2^PR × RLR) / LSI
     * 例: PR=4, RLR=2500 → 超时 ≈ 1000ms */
    
    uint32_t prescaler, reload;
    
    if (timeout_ms <= 500) {
        prescaler = IWDG_PRESCALER_16;   /* PR=4 */
        reload = (timeout_ms * 40) / 16;  /* 近似 */
    } else if (timeout_ms <= 2000) {
        prescaler = IWDG_PRESCALER_64;   /* PR=6 */
        reload = (timeout_ms * 40) / 64;
    } else {
        prescaler = IWDG_PRESCALER_256;  /* PR=8 */
        reload = (timeout_ms * 40) / 256;
    }
    
    IWDG->KR  = 0x5555;  /* 解锁写保护 */
    IWDG->PR  = prescaler;
    IWDG->RLR = reload & 0xFFF;
    IWDG->KR  = 0xCCCC;  /* 启动看门狗 */
    IWDG->KR  = 0xAAAA;  /* 首次喂狗 */
}

void iwdg_feed(void) {
    IWDG->KR = 0xAAAA;
}
```

### 4.2 窗口看门狗（WWDG）

```c
/* 窗口看门狗 - 防止过快或过慢喂狗 */
void wwdg_init(uint8_t window_min, uint8_t window_max, uint8_t prescaler) {
    RCC->APB1ENR |= RCC_APB1ENR_WWDGEN;
    
    WWDG->CFR = (prescaler << 7) | window_max;
    WWDG->CR  = 0x80 | window_max;  /* 使能并设计数器 */
}

/* 喂狗必须在窗口内 */
void wwdg_feed(uint8_t counter_value) {
    uint8_t window = WWDG->CFR & 0x7F;
    if (counter_value > window) {
        WWDG->CR = 0x80 | counter_value;
    }
    /* 若 counter <= window，不喂狗 → 复位 */
}
```

### 4.3 看门狗分层架构

```
┌─────────────────────────────────────┐
│         应用层看门狗（软件）          │
│   监控各任务是否正常执行              │
│   周期: 100ms ~ 1s                  │
├─────────────────────────────────────┤
│         系统层看门狗                  │
│   监控主循环/RTOS 调度器              │
│   周期: 500ms ~ 2s                  │
├─────────────────────────────────────┤
│       硬件看门狗（IWDG/WWDG）        │
│   最后防线，独立时钟源                │
│   周期: 1s ~ 10s                    │
└─────────────────────────────────────┘
```

```c
/* 分层看门狗实现 */
typedef struct {
    volatile uint32_t last_feed_time;
    uint32_t timeout;
    bool     is_alive;
} soft_watchdog_t;

static soft_watchdog_t task_watchdogs[] = {
    { .timeout = 100  },  /* 电机控制 100ms */
    { .timeout = 500  },  /* 传感器采样 500ms */
    { .timeout = 1000 },  /* 通信处理 1s */
};

void task_feed(uint8_t task_id) {
    if (task_id < sizeof(task_watchdogs)/sizeof(task_watchdogs[0])) {
        task_watchdogs[task_id].last_feed_time = HAL_GetTick();
        task_watchdogs[task_id].is_alive = true;
    }
}

void watchdog_supervisor(void) {
    uint32_t now = HAL_GetTick();
    bool all_alive = true;
    
    for (uint8_t i = 0; i < ARRAY_SIZE(task_watchdogs); i++) {
        if ((now - task_watchdogs[i].last_feed_time) > task_watchdogs[i].timeout) {
            task_watchdogs[i].is_alive = false;
            all_alive = false;
        }
    }
    
    if (all_alive) {
        iwdg_feed();  /* 所有任务正常才喂硬件狗 */
    }
}
```

## 5. 其他安全编程技术

### 5.1 断言机制

```c
/* 轻量级断言 - 适合嵌入式 */
#ifdef DEBUG
    #define ASSERT(cond) do { \
        if (!(cond)) { \
            asm volatile("bkpt #0"); \
        } \
    } while(0)
#else
    #define ASSERT(cond) ((void)0)
#endif

/* 关键参数验证 */
void set_pwm_duty(uint8_t channel, uint16_t duty) {
    ASSERT(channel < 4);
    ASSERT(duty <= 1000);
    
    if (channel >= 4 || duty > 1000) {
        return;  /* Release 模式下安全返回 */
    }
    /* ... 正常执行 */
}
```

### 5.2 安全的状态机

```c
/* 防止非法状态转换 */
typedef enum {
    STATE_IDLE = 0,
    STATE_RUNNING,
    STATE_ERROR,
    STATE_COUNT
} system_state_t;

/* 合法转换表 */
static const bool valid_transitions[STATE_COUNT][STATE_COUNT] = {
    /*              IDLE   RUN   ERR */
    /* IDLE  */  { false, true,  true  },
    /* RUN   */  { true,  false, true  },
    /* ERR   */  { true,  false, false },
};

bool state_transition(system_state_t *current, system_state_t next) {
    if (next >= STATE_COUNT) return false;
    if (!valid_transitions[*current][next]) return false;
    *current = next;
    return true;
}
```

### 5.3 安全的环形缓冲区

```c
typedef struct {
    uint8_t *buffer;
    uint16_t size;
    volatile uint16_t head;
    volatile uint16_t tail;
} ring_buf_t;

bool ring_buf_put(ring_buf_t *rb, uint8_t data) {
    uint16_t next = (rb->head + 1) % rb->size;
    if (next == rb->tail) {
        return false;  /* 满 */
    }
    rb->buffer[rb->head] = data;
    rb->head = next;
    return true;
}

bool ring_buf_get(ring_buf_t *rb, uint8_t *data) {
    if (rb->head == rb->tail) {
        return false;  /* 空 */
    }
    *data = rb->buffer[rb->tail];
    rb->tail = (rb->tail + 1) % rb->size;
    return true;
}
```
