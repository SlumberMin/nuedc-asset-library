> 适用：IMU姿态融合、编码器测速降噪、电池SOC估计等电赛常见场景

---

## 一、直觉理解卡尔曼滤波

### 1.1 一句话解释

**卡尔曼滤波 = 加权平均，权重取决于"谁更准"。**

```
估计值 = 预测值 + K × (测量值 - 预测值)
         ↑              ↑
      由模型推算      由传感器读取

K（卡尔曼增益）→ 自动决定信预测还是信测量
```

### 1.2 两个关键方程

**预测（时间更新）：**
```
x̂⁻ = A × x̂ + B × u        ← 状态预测
P⁻ = A × P × Aᵀ + Q         ← 协方差预测（不确定性增大）
```

**更新（量测更新）：**
```
K = P⁻ × Hᵀ / (H × P⁻ × Hᵀ + R)  ← 卡尔曼增益
x̂ = x̂⁻ + K × (z - H × x̂⁻)         ← 状态修正
P = (I - K × H) × P⁻                ← 协方差修正（不确定性减小）
```

**关键参数：**
- **Q（过程噪声）**：模型有多不准 → Q越大，越信测量
- **R（测量噪声）**：传感器有多不准 → R越大，越信模型

## 二、一维卡尔曼滤波（入门）

### 2.1 完整C实现

```c
/* kalman_1d.h */
#ifndef __KALMAN_1D_H
#define __KALMAN_1D_H

typedef struct {
    float x;    // 状态估计值
    float P;    // 估计协方差
    float Q;    // 过程噪声
    float R;    // 测量噪声
    float K;    // 卡尔曼增益（调试用）
} Kalman1D;

void KF1_Init(Kalman1D *kf, float x0, float P0, float Q, float R);
float KF1_Update(Kalman1D *kf, float z);

#endif
```

```c
/* kalman_1d.c */
#include "kalman_1d.h"

void KF1_Init(Kalman1D *kf, float x0, float P0, float Q, float R) {
    kf->x = x0;
    kf->P = P0;
    kf->Q = Q;
    kf->R = R;
    kf->K = 0;
}

float KF1_Update(Kalman1D *kf, float z) {
    /* 预测 */
    kf->P = kf->P + kf->Q;
    
    /* 更新 */
    kf->K = kf->P / (kf->P + kf->R);
    kf->x = kf->x + kf->K * (z - kf->x);
    kf->P = (1.0f - kf->K) * kf->P;
    
    return kf->x;
}
```

## 三、扩展卡尔曼滤波（EKF）

### 3.1 非线性系统

```
x(k+1) = f(x(k), u(k)) + w     ← 非线性状态方程
z(k) = h(x(k)) + v              ← 非线性观测方程
```

**EKF的核心：对非线性函数做线性化（求雅可比矩阵）**

```
F = ∂f/∂x  （状态转移雅可比）
H = ∂h/∂x  （观测雅可比）
```

### 3.2 EKF通用框架

```c
/* ekf.h - 通用EKF框架 */
#ifndef __EKF_H
#define __EKH_H

#define EKF_N 4  // 状态维度
#define EKF_M 2  // 观测维度

typedef struct {
    float x[EKF_N];              // 状态向量
    float P[EKF_N][EKF_N];      // 协方差矩阵
    float Q[EKF_N][EKF_N];      // 过程噪声
    float R[EKF_M][EKF_M];      // 测量噪声
    float K[EKF_N][EKF_M];      // 卡尔曼增益
    float F[EKF_N][EKF_N];      // 状态转移雅可比
    float H[EKF_M][EKF_N];      // 观测雅可比
} EKF;

void EKF_Init(EKF *ekf);
void EKF_Predict(EKF *ekf, float *u);
void EKF_Update(EKF *ekf, float *z);

// 用户需实现：
void EKF_Func_f(EKF *ekf, float *u);      // 状态转移函数
void EKF_Func_h(EKF *ekf, float *hx);     // 观测函数
void EKF_Jacobi_F(EKF *ekf, float *u);    // F雅可比
void EKF_Jacobi_H(EKF *ekf);              // H雅可比

#endif
```

## 四、实战场景1：IMU姿态融合（互补滤波 → 卡尔曼）

### 4.1 问题描述

- 加速度计：测姿态角，低频准，高频噪声大
- 陀螺仪：测角速度，高频准，低频有漂移

### 4.2 状态空间模型

**状态：** x = [角度, 角速度偏置]ᵀ

**状态方程：**
```
角度(k+1) = 角度(k) + (陀螺仪 - 偏置) × dt
偏置(k+1) = 偏置(k)
```

**观测方程：**
```
z = 角度（由加速度计计算）
```

### 4.3 完整实现

```c
/* imu_kalman.h */
#ifndef __IMU_KALMAN_H
#define __IMU_KALMAN_H

typedef struct {
    /* 状态 */
    float angle;    // 融合角度
    float bias;     // 陀螺仪偏置
    
    /* 协方差矩阵 P */
    float P00, P01;
    float P10, P11;
    
    /* 噪声参数 */
    float Q_angle;      // 角度过程噪声
    float Q_bias;       // 偏置过程噪声
    float R_measure;    // 加速度计测量噪声
    
    /* 输出 */
    float rate;         // 融合角速度（无偏置）
} IMU_Kalman;

void IMU_KF_Init(IMU_Kalman *kf);
float IMU_KF_Update(IMU_Kalman *kf, float newAngle, float newRate, float dt);

#endif
```

```c
/* imu_kalman.c */
#include "imu_kalman.h"

void IMU_KF_Init(IMU_Kalman *kf) {
    kf->angle = 0;
    kf->bias = 0;
    kf->P00 = 0; kf->P01 = 0;
    kf->P10 = 0; kf->P11 = 0;
    
    /* === 这三个参数最关键 === */
    kf->Q_angle = 0.001f;    // 加速度计信任度（小=信任加速度计）
    kf->Q_bias = 0.003f;     // 偏置变化率
    kf->R_measure = 0.03f;   // 加速度计噪声（大=不信任加速度计）
}

float IMU_KF_Update(IMU_Kalman *kf, float newAngle, float newRate, float dt) {
    /* === 预测 === */
    // 状态预测
    kf->rate = newRate - kf->bias;
    kf->angle += dt * kf->rate;
    
    // 协方差预测
    kf->P00 += dt * (dt * kf->P11 - kf->P01 - kf->P10 + kf->Q_angle);
    kf->P01 -= dt * kf->P11;
    kf->P10 -= dt * kf->P11;
    kf->P11 += kf->Q_bias * dt;
    
    /* === 更新 === */
    float S = kf->P00 + kf->R_measure;  // 残差协方差
    
    float K0 = kf->P00 / S;  // 卡尔曼增益
    float K1 = kf->P10 / S;
    
    float y = newAngle - kf->angle;  // 残差
    
    kf->angle += K0 * y;
    kf->bias += K1 * y;
    
    float P00_temp = kf->P00;
    float P01_temp = kf->P01;
    
    kf->P00 -= K0 * P00_temp;
    kf->P01 -= K0 * P01_temp;
    kf->P10 -= K1 * P00_temp;
    kf->P11 -= K1 * P01_temp;
    
    return kf->angle;
}
```

### 4.4 在主循环中调用

```c
IMU_Kalman roll_kf, pitch_kf;

int main(void) {
    IMU_KF_Init(&roll_kf);
    IMU_KF_Init(&pitch_kf);
    
    while (1) {
        // 读取MPU6050
        MPU6050_Read(&ax, &ay, &az, &gx, &gy, &gz);
        
        // 加速度计计算角度
        float acc_roll  = atan2f(ay, az) * 57.2958f;
        float acc_pitch = atan2f(-ax, sqrtf(ay*ay + az*az)) * 57.2958f;
        
        // 陀螺仪角速度（°/s）
        float gyro_roll  = gx / 131.0f;
        float gyro_pitch = gy / 131.0f;
        
        // 卡尔曼融合
        float dt = 0.005f;  // 5ms
        float roll  = IMU_KF_Update(&roll_kf,  acc_roll,  gyro_roll,  dt);
        float pitch = IMU_KF_Update(&pitch_kf, acc_pitch, gyro_pitch, dt);
        
        delay_ms(5);
    }
}
```

### 4.5 调参指南

| 参数 | 初始值 | 调大效果 | 调小效果 |
|------|--------|---------|---------|
| Q_angle | 0.001 | 更信任陀螺仪，响应快但可能漂移 | 更信任加速度计，噪声大 |
| Q_bias | 0.003 | 偏置校正更快 | 偏置校正更慢 |
| R_measure | 0.03 | 更信任模型，平滑但滞后 | 更信任测量，噪声大 |

**调参口诀：** R大→平滑但滞后；Q大→快速但有噪

## 五、实战场景2：编码器测速降噪

### 5.1 问题

编码器测速在低速时存在量化噪声（M法测速误差大）。

### 5.2 一阶卡尔曼测速

```c
/* 速度估计用一维卡尔曼 */
typedef struct {
    float speed;      // 估计速度
    float P;          // 协方差
    float Q;          // 过程噪声（加速度不确定性）
    float R;          // 测量噪声（编码器误差）
    float dt;
} Speed_KF;

void Speed_KF_Init(Speed_KF *kf, float dt) {
    kf->speed = 0;
    kf->P = 100;
    kf->Q = 10.0f;     // 加速度不确定性
    kf->R = 5.0f;      // 编码器噪声
    kf->dt = dt;
}

float Speed_KF_Update(Speed_KF *kf, float measured_speed) {
    /* 预测：假设速度不变（匀速模型） */
    float speed_pred = kf->speed;
    float P_pred = kf->P + kf->Q;
    
    /* 更新 */
    float K = P_pred / (P_pred + kf->R);
    kf->speed = speed_pred + K * (measured_speed - speed_pred);
    kf->P = (1 - K) * P_pred;
    
    return kf->speed;
}
```

### 5.3 二阶卡尔曼（含加速度估计）

```c
/* 状态：[速度, 加速度] */
typedef struct {
    float vel, acc;
    float P[2][2];
    float Q_vel, Q_acc;
    float R;
} SpeedAcc_KF;

float SpeedAcc_Update(SpeedAcc_KF *kf, float z, float dt) {
    /* 预测 */
    float vel_pred = kf->vel + kf->acc * dt;
    // ... 完整协方差传播略（参考EKF框架）
    
    kf->vel = vel_pred + K0 * (z - vel_pred);
    kf->acc += K1 * (z - vel_pred);
    
    return kf->vel;
}
```

## 六、实战场景3：电池SOC估计

### 6.1 模型

**安时积分 + 开路电压修正：**

```
SOC(k+1) = SOC(k) - (I × dt) / Q_max + K × (V_ocv - V_terminal)
```

### 6.2 卡尔曼SOC估计

```c
typedef struct {
    float soc;          // SOC估计值 (0~1)
    float P;            // 协方差
    float Q;            // 模型不确定性
    float R;            // OCV-SOC映射不确定性
    float capacity_ah;  // 电池容量(Ah)
} SOC_KF;

/* OCV-SOC查找表（锂电池典型值） */
const float ocv_soc_table[][2] = {
    {3.00, 0.00}, {3.30, 0.05}, {3.50, 0.10}, {3.60, 0.20},
    {3.70, 0.40}, {3.75, 0.50}, {3.80, 0.60}, {3.90, 0.80},
    {4.00, 0.90}, {4.10, 0.95}, {4.20, 1.00}
};

float OCV_to_SOC(float v) {
    int n = sizeof(ocv_soc_table) / sizeof(ocv_soc_table[0]);
    for (int i = 0; i < n - 1; i++) {
        if (v >= ocv_soc_table[i][0] && v < ocv_soc_table[i+1][0]) {
            float ratio = (v - ocv_soc_table[i][0]) / 
                          (ocv_soc_table[i+1][0] - ocv_soc_table[i][0]);
            return ocv_soc_table[i][1] + ratio * 
                   (ocv_soc_table[i+1][1] - ocv_soc_table[i][1]);
        }
    }
    return (v < 3.0f) ? 0.0f : 1.0f;
}

void SOC_KF_Init(SOC_KF *kf, float capacity) {
    kf->soc = 0.5f;
    kf->P = 0.01f;
    kf->Q = 0.0001f;   // 安时积分模型较准
    kf->R = 0.01f;     // OCV-SOC映射有误差
    kf->capacity_ah = capacity;
}

float SOC_KF_Update(SOC_KF *kf, float current, float voltage, float dt) {
    /* 预测：安时积分 */
    float soc_pred = kf->soc - (current * dt) / (kf->capacity_ah * 3600.0f);
    float P_pred = kf->P + kf->Q;
    
    /* 观测：由OCV查表得SOC */
    float soc_meas = OCV_to_SOC(voltage);
    
    /* 更新 */
    float K = P_pred / (P_pred + kf->R);
    kf->soc = soc_pred + K * (soc_meas - soc_pred);
    kf->P = (1 - K) * P_pred;
    
    // 限幅
    if (kf->soc > 1.0f) kf->soc = 1.0f;
    if (kf->soc < 0.0f) kf->soc = 0.0f;
    
    return kf->soc;
}
```

**使用：**
```c
SOC_KF battery;
SOC_KF_Init(&battery, 2.0f);  // 2000mAh电池

while (1) {
    float I = ADC_Read_Current();   // 电流(A)
    float V = ADC_Read_Voltage();   // 电压(V)
    float soc = SOC_KF_Update(&battery, I, V, 0.1f);  // 100ms
    LCD_ShowSOC(soc * 100);
    delay_ms(100);
}
```

```c
/* matrix.c */
#include "matrix.h"
#include <string.h>

void mat_add(float *A, float *B, float *C, int m, int n) {
    for (int i = 0; i < m * n; i++) C[i] = A[i] + B[i];
}

void mat_sub(float *A, float *B, float *C, int m, int n) {
    for (int i = 0; i < m * n; i++) C[i] = A[i] - B[i];
}

void mat_mul(float *A, float *B, float *C, int m, int n, int p) {
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < p; j++) {
            C[i * p + j] = 0;
            for (int k = 0; k < n; k++) {
                C[i * p + j] += A[i * n + k] * B[k * p + j];
            }
        }
    }
}

void mat_trans(float *A, float *AT, int m, int n) {
    for (int i = 0; i < m; i++)
        for (int j = 0; j < n; j++)
            AT[j * m + i] = A[i * n + j];
}

/* 高斯-约旦消元求逆（适用于小矩阵） */
int mat_inv(float *A, float *Ainv, int n) {
    float aug[MAT_MAX_SIZE][2 * MAT_MAX_SIZE];
    
    // 构造增广矩阵 [A|I]
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            aug[i][j] = A[i * n + j];
            aug[i][j + n] = (i == j) ? 1.0f : 0.0f;
        }
    }
    
    // 消元
    for (int col = 0; col < n; col++) {
        // 选主元
        int max_row = col;
        for (int row = col + 1; row < n; row++) {
            if (fabsf(aug[row][col]) > fabsf(aug[max_row][col]))
                max_row = row;
        }
        // 交换行
        if (max_row != col) {
            for (int j = 0; j < 2 * n; j++) {
                float tmp = aug[col][j];
                aug[col][j] = aug[max_row][j];
                aug[max_row][j] = tmp;
            }
        }
        
        if (fabsf(aug[col][col]) < 1e-10f) return -1;  // 奇异矩阵
        
        float pivot = aug[col][col];
        for (int j = 0; j < 2 * n; j++) aug[col][j] /= pivot;
        
        for (int row = 0; row < n; row++) {
            if (row != col) {
                float factor = aug[row][col];
                for (int j = 0; j < 2 * n; j++)
                    aug[row][j] -= factor * aug[col][j];
            }
        }
    }
    
    // 提取逆矩阵
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            Ainv[i * n + j] = aug[i][j + n];
    
    return 0;
}
```

## 八、调参经验总结

### 8.1 Q/R比值决定滤波效果

```
Q/R 大 → 更信测量 → 响应快、噪声多
Q/R 小 → 更信模型 → 平滑、滞后大
```

### 8.2 参数整定步骤

1. **先设Q=0，调R**：增大R直到滤波输出明显滞后
2. **逐步增大Q**：直到响应速度满足要求
3. **微调**：在响应速度和平滑度之间找平衡

### 8.3 常见陷阱

| 问题 | 原因 | 解决 |
|------|------|------|
| 滤波输出不收敛 | 初始P0太小 | 设P0=100~1000 |
| 输出有偏 | 模型有系统误差 | 检查状态方程 |
| 高频噪声没滤掉 | R太小 | 增大R |
| 低频响应太慢 | Q太小 | 增大Q |
| 协方差矩阵发散 | 数值问题 | 用Joseph形式更新P |

### 8.4 Joseph形式（数值稳定）

```c
// 普通形式（可能数值不稳定）
P = (1 - K) * P_pred;

// Joseph形式（保证P正定）
P = (1 - K) * P_pred * (1 - K) + K * R * K;
```

## 九、卡尔曼滤波 vs 其他滤波器对比

| 滤波器 | 适用场景 | 优点 | 缺点 |
|--------|---------|------|------|
| 移动平均 | 简单降噪 | 实现简单 | 滞后大 |
| 一阶低通 | 实时滤波 | 计算量小 | 参数不自适应 |
| 互补滤波 | IMU融合 | 简单高效 | 无最优性保证 |
| **卡尔曼** | **状态估计** | **最优估计、自适应** | **需建模、计算量大** |
| 粒子滤波 | 高度非线性 | 通用性强 | 计算量巨大 |
