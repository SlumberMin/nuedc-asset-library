### 2.2 Unity 测试框架示例

```c
/* test_pid_controller.c */
#include "unity.h"
#include "pid_controller.h"

static pid_ctrl_t pid;

void setUp(void) {
    pid_init(&pid, 10, 5, 2, 100);  /* Kp=10, Ki=5, Kd=2, dt=100ms */
}

void tearDown(void) {
    /* 清理 */
}

/* 测试基本比例控制 */
void test_proportional_response(void) {
    int16_t output = pid_update(&pid, 100, 80);  /* setpoint=100, measured=80 */
    /* error=20, output = 20*10/100 = 2 (带Q格式) */
    TEST_ASSERT_INT16_WITHIN(5, 20, output);
}

/* 测试积分饱和限制 */
void test_integral_windup(void) {
    /* 持续大误差 */
    for (int i = 0; i < 1000; i++) {
        pid_update(&pid, 1000, 0);
    }
    TEST_ASSERT_TRUE(pid.integral <= pid.i_limit);
}

/* 测试误差为零 */
void test_zero_error(void) {
    int16_t output = pid_update(&pid, 50, 50);
    TEST_ASSERT_EQUAL_INT16(0, output);
}

int main(void) {
    UNITY_BEGIN();
    RUN_TEST(test_proportional_response);
    RUN_TEST(test_integral_windup);
    RUN_TEST(test_zero_error);
    return UNITY_END();
}
```

### 2.3 Mock 技术（隔离硬件依赖）

```c
/* mock_hal.c - 模拟 HAL 层 */
static uint16_t mock_adc_values[4] = {2048, 2048, 2048, 2048};

void mock_set_adc(uint8_t channel, uint16_t value) {
    if (channel < 4) mock_adc_values[channel] = value;

    return mock_adc_values[0];  /* 返回模拟值 */

    static uint32_t mock_tick = 0;
    return mock_tick += 10;  /* 模拟时间流逝 */
}
```

## 3. 集成测试

### 3.1 硬件在环测试（HIL）

```c
/* hil_test_runner.c - 硬件在环集成测试 */
typedef struct {
    const char *test_name;
    bool (*test_func)(void);
} hil_test_t;

/* 测试 ADC 采集链路 */
bool test_adc_pipeline(void) {
    /* 1. 配置 ADC */
    adc_init(ADC_CHANNEL_0, ADC_SAMPLETIME_239CYCLES);
    
    /* 2. 读取采样值 */
    uint16_t raw = adc_read();
    
    /* 3. 验证合理性（假设传感器范围） */
    if (raw < 100 || raw > 4000) {
        printf("ADC 值异常: %d\n", raw);
        return false;
    }
    
    /* 4. 验证滤波输出 */
    uint16_t filtered = moving_average(raw, 16);
    if (filtered < 50 || filtered > 4050) {
        return false;
    }
    
    return true;
}

/* 测试电机驱动链路 */
bool test_motor_drive(void) {
    motor_set_speed(50);  /* 50% 占空比 */
    delay_ms(500);
    
    /* 读取电流反馈 */
    int16_t current = read_current_sensor();
    if (current < 0 || current > 3000) {  /* mA */
        printf("电流异常: %d mA\n", current);
        motor_stop();
        return false;
    }
    
    motor_stop();
    return true;
}

/* 运行所有集成测试 */
void run_all_hil_tests(void) {
    hil_test_t tests[] = {
        {"ADC Pipeline", test_adc_pipeline},
        {"Motor Drive",  test_motor_drive},
        {"UART Comm",    test_uart_comm},
        {"Display",      test_lcd_display},
    };
    
    uint16_t pass = 0, fail = 0;
    uint16_t total = sizeof(tests) / sizeof(tests[0]);
    
    printf("=== HIL Integration Tests ===\n");
    for (uint16_t i = 0; i < total; i++) {
        printf("[%d/%d] %s: ", i+1, total, tests[i].test_name);
        if (tests[i].test_func()) {
            printf("PASS ✓\n");
            pass++;
        } else {
            printf("FAIL ✗\n");
            fail++;
        }
    }
    printf("Result: %d/%d passed\n", pass, total);
}
```

### 3.2 通信协议测试

```c
/* 测试串口通信协议 */
bool test_uart_protocol(void) {
    uint8_t test_frame[] = {0xAA, 0x01, 0x02, 0x00, 0x10, 0xBB};
    uint8_t rx_buf[32];
    
    /* 发送测试帧 */
    uart_send(test_frame, sizeof(test_frame));
    delay_ms(100);
    
    /* 接收并验证 */
    uint16_t rx_len = uart_receive(rx_buf, sizeof(rx_buf));
    
    if (rx_len == 0) {
        printf("无响应\n");
        return false;
    }
    
    /* 校验帧头帧尾 */
    if (rx_buf[0] != 0xAA || rx_buf[rx_len-1] != 0xBB) {
        printf("帧格式错误\n");
        return false;
    }
    
    return true;
}
```

## 4. 代码审查

### 4.1 代码审查检查清单

#### 正确性
- [ ] 算法逻辑是否正确
- [ ] 边界条件是否处理
- [ ] 错误路径是否覆盖
- [ ] 资源是否正确释放

#### 可读性
- [ ] 命名是否清晰（变量、函数、宏）
- [ ] 注释是否必要且准确
- [ ] 函数长度是否合理（<50行）
- [ ] 嵌套深度是否过深（<4层）

#### 性能
- [ ] 是否有不必要的内存拷贝
- [ ] 中断处理是否足够快
- [ ] 是否有阻塞式等待
- [ ] 栈使用是否合理

#### 安全
- [ ] 输入是否验证
- [ ] 整数运算是否溢出安全
- [ ] 指针是否判空
- [ ] 并发访问是否保护

### 4.2 电赛快速审查规则（15分钟审查法）

```markdown
## 快速审查要点（按优先级）

### P0 - 必须修复
1. 会导致系统崩溃的 bug
2. 硬件损坏风险（过流、过压）
3. 死锁或资源泄漏

### P1 - 应该修复
4. 逻辑错误导致功能异常
5. 性能瓶颈
6. 明显的代码重复

### P2 - 建议改进
7. 命名不清晰
8. 缺少错误处理
9. 注释缺失
```

## 5. 静态分析

### 5.1 常用静态分析工具

| 工具 | 费用 | 集成方式 | 检测能力 |
|------|------|----------|----------|
| **Cppcheck** | 免费 | CLI/Makefile | 内存泄漏、未初始化变量 |
| **PC-lint** | 商业 | IDE/CLI | MISRA合规、逻辑错误 |
| **SonarQube** | 社区版免费 | CI/CD | 代码质量门禁 |
| **Coverity** | 商业 | CI/CD | 深度缺陷检测 |
| **GCC -Wall** | 免费 | 编译器 | 基础警告 |

### 5.2 Cppcheck 使用示例

```bash
# 基本扫描
cppcheck --enable=all --std=c11 src/

# 指定 MISRA 规则
cppcheck --enable=all --addon=misra.json src/

# 生成 XML 报告
cppcheck --enable=all --xml --xml-version=2 src/ 2> report.xml

# misra.json 配置示例
{
    "script": "addons/misra.py",
    "args": ["--rule-texts=misra_rules.txt"]
}
```

### 5.3 GCC 编译警告配置

```makefile
# Makefile 中的安全警告配置
CFLAGS += -Wall            # 常见警告
CFLAGS += -Wextra          # 额外警告
CFLAGS += -Wpedantic       # 严格标准检查
CFLAGS += -Wshadow         # 变量遮蔽
CFLAGS += -Wformat=2       # printf 格式检查
CFLAGS += -Wconversion     # 类型转换警告
CFLAGS += -Wsign-conversion # 符号转换警告
CFLAGS += -Wdouble-promotion # 浮点提升
CFLAGS += -Wnull-dereference # 空指针解引用
CFLAGS += -Wstack-usage=1024 # 栈使用警告

# 将警告视为错误（生产构建）
CFLAGS += -Werror
```

## 6. CI/CD 流程

### 6.1 电赛项目简化 CI 流程

```yaml
# .github/workflows/build.yml（如使用 GitHub）
name: Build & Test

on: [push, pull_request]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Install ARM toolchain
        run: |
          sudo apt-get update
          sudo apt-get install -y gcc-arm-none-eabi
      
      - name: Static Analysis
        run: |
          sudo apt-get install -y cppcheck
          cppcheck --enable=warning,style src/
      
      - name: Build
        run: make all
      
      - name: Unit Tests
        run: |
          cd test
          make test
      
      - name: Size Report
        run: arm-none-eabi-size build/*.elf
```

### 6.2 本地预提交钩子

```bash
#!/bin/bash
# .git/hooks/pre-commit

echo "=== 静态检查 ==="
cppcheck --enable=warning src/ 2>&1 | tee /tmp/cppcheck.log
if grep -q "error:" /tmp/cppcheck.log; then
    echo "❌ 静态检查发现错误，阻止提交"
    exit 1
fi

echo "=== 编译检查 ==="
make clean && make all
if [ $? -ne 0 ]; then
    echo "❌ 编译失败，阻止提交"
    exit 1
fi

echo "✅ 所有检查通过"
exit 0
```

### 6.3 电赛实际 CI/CD 方案

由于电赛环境通常无网络，推荐本地 CI 方案：

```bash
#!/bin/bash
# local_ci.sh - 本地持续集成脚本

set -e

echo "============================="
echo "  电赛项目本地 CI 检查"
echo "============================="

# 1. 静态分析
echo "[1/4] 静态分析..."
cppcheck --enable=warning,style,performance src/ 2>&1
echo ""

# 2. 编译
echo "[2/4] 编译..."
make clean && make all 2>&1
echo ""

# 3. 单元测试（如有）
echo "[3/4] 单元测试..."
if [ -d "test" ]; then
    cd test && make test && cd ..
else
    echo "  跳过（无测试目录）"
fi
echo ""

# 4. 资源报告
echo "[4/4] 资源使用..."
arm-none-eabi-size build/*.elf 2>/dev/null || echo "  无 ELF 文件"
echo ""

echo "============================="
echo "  CI 检查完成"
echo "============================="
```

