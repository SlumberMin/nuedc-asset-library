## 五、软件设计

### 5.1 FFT 算法实现

```c
// 基于 CMSIS-DSP 的 FFT 实现
#include "arm_math.h"

#define FFT_LENGTH  2048

arm_rfft_fast_instance_f32 fft_instance;
float fft_input[FFT_LENGTH];
float fft_output[FFT_LENGTH];
float fft_magnitude[FFT_LENGTH/2];

void spectrum_analysis(float *signal, uint16_t length) {
    // 1. 加窗
    for (int i = 0; i < length; i++) {
        // Hanning 窗
        float w = 0.5f * (1.0f - arm_cos_f32(2.0f * PI * i / length));
        fft_input[i] = signal[i] * w;
    }

    // 2. FFT 计算
    arm_rfft_fast_f32(&fft_instance, fft_input, fft_output, 0);

    // 3. 计算幅度谱
    arm_cmplx_mag_f32(fft_output, fft_magnitude, length/2);

    // 4. 转换为 dB
    for (int i = 0; i < length/2; i++) {
        fft_magnitude[i] = 20.0f * log10f(fft_magnitude[i] / length + 1e-10f);
    }
}
```

### 5.2 峰值检测算法

```c
typedef struct {
    float freq;
    float magnitude;
    int bin_index;
} PeakInfo;

PeakInfo peaks[MAX_PEAKS];

int detect_peaks(float *spectrum, int length, float threshold, PeakInfo *peaks) {
    int count = 0;
    for (int i = 2; i < length - 2; i++) {
        if (spectrum[i] > threshold &&
            spectrum[i] > spectrum[i-1] && spectrum[i] > spectrum[i+1] &&
            spectrum[i] > spectrum[i-2] && spectrum[i] > spectrum[i+2]) {
            // 抛物线插值提高频率精度
            float alpha = spectrum[i-1];
            float beta = spectrum[i];
            float gamma = spectrum[i+1];
            float delta = 0.5f * (alpha - gamma) / (alpha - 2*beta + gamma);

            peaks[count].bin_index = i;
            peaks[count].freq = (i + delta) * sample_rate / fft_length;
            peaks[count].magnitude = beta - 0.25f * (alpha - gamma) * delta;
            count++;
            if (count >= MAX_PEAKS) break;
        }
    }
    return count;
}
```
